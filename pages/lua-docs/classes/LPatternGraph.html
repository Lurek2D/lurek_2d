Class

LPatternGraph

Lua-facing graph data structure with directed/undirected edges, BFS, DFS, and connectivity queries.

Docs home/lurek.patterns
Class Example · Exact example · patterns.lua
do
    local g = lurek.patterns.newGraph(true)
    local a = g:addNode("A", {cost = 10})
    local b = g:addNode("B", {cost = 5})
    g:addEdge(a, b, 1.0, "road")
    print("nodes = " .. g:nodeCount())
    print("has a = " .. tostring(g:hasNode(a)))
end
