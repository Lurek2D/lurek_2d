Class

LChunkMap

Lua-side handle wrapping a `ChunkMap` for infinite or very large tile grids stored in dynamically loaded chunks.

Docs home/lurek.tilemap
Class Example · Exact example · tilemap.lua
do
    ---@type LChunkMap
    local cm = lurek.tilemap.newChunkMap(16)
    cm:setTile(10, 20, 5)
    local gid = cm:getTile(10, 20)
    print("tile at 10,20 = " .. gid)
end

    local visible = cm:getChunksInView(0, 0, 800, 600, 32, 32)
