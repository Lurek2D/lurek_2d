LTransform

Lua-side wrapper for a 2D transform matrix.

Docs home/lurek.math
Class Example · Exact example · math.lua
do
    local t = lurek.math.newTransform()
    t:translate(50, 50)
    local x, y = t:transformPoint(0, 0)
    print("point = " .. x .. "," .. y)
end
