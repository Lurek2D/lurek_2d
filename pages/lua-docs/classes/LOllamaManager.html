pullModel

LOllamaManager:pullModel(name: string, callback: function) -> integer

methodsrc/lua_api/agent_api.rs:613
Example · Exact example · agent.lua
do
    local ollama = lurek.agent.newOllama()
    local id     = ollama:pullModel("llama3", function(success, err_msg)
        if success then
            print("Model downloaded successfully.")
        else
            print("Pull failed:", err_msg)
        end
    end)
    print("Pull started, callback id =", id)
