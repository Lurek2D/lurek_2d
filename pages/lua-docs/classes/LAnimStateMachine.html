Class

LAnimStateMachine

Lua-side animation state machine that switches clips from named states and parameters.

Docs home/lurek.animation
Class Example · Exact example · animation.lua
do
    local anim = lurek.animation.new()
    anim:addFrame(0, 0, 32, 32)
    anim:addClip("idle", { 0 }, 5, true)
    local sm = lurek.animation.newStateMachine(anim, "idle")
    sm:addState("idle", "idle", true)
    sm:update(0.016)
    print("sm updated, state = " .. sm:getState())
end

do

    local anim = lurek.animation.new()

    anim:addClip("idle", { 0 }, 8, true)

end
