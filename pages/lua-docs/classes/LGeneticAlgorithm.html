Class

LGeneticAlgorithm

Lua handle for a floating-point genetic algorithm population.

Docs home/lurek.learning
Class Example · Exact example · learning.lua
do
    local ga = lurek.learning.newGeneticAlgorithm(5, 4, 11)

    for index = 0, ga:popSize() - 1 do
        ga:setFitness(index, index * 0.5)
    end

    ga:evolve()
    print("LGeneticAlgorithm:evolve generation", ga:generation())
end
