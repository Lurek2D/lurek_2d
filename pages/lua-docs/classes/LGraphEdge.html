LGraphEdge

Lua-side edge handle referencing one edge id inside a graph.

Docs home/lurek.flownet
Class Example · Exact example · flownet.lua
do
    local g = lurek.graph.newGraph()
    local a = g:addNode()
    local b = g:addNode()
    local e = g:addEdge(a, b, "conveyor")
    print("edge type = " .. e:getType())
end
