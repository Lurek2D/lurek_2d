LFlowField

Lua-side wrapper for a flow field over a navigation grid.

Docs home/lurek.pathfind
Class Example · Exact example · pathfind.lua
do
    local nav = lurek.pathfind.newNavGrid(20, 20)

    nav:fill(1)

    local ff = lurek.pathfind.newFlowField(nav)
    ff:calculate(20, 10)

    print("calculated = " .. tostring(ff:isCalculated()))
    print("targets = " .. #ff:getTargets())
end
