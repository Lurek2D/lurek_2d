LFlowField

Lua-side wrapper for a flow field over a navigation grid.

Docs home/lurek.pathfind
Class Example · Exact example · pathfind.lua
do
    local nav = lurek.pathfind.newNavGrid(20, 20)

    nav:fill(1)

    local ff = lurek.pathfind.newFlowField(nav)
    ff:calculate(20, 10)

    print("calculated = " .. tostring(ff:isCalculated()))
    print("targets = " .. #ff:getTargets())
end

calculateMulti

LFlowField:calculateMulti(targets: table, unit_size: integer?)

Calculates a flow field toward multiple target cells.

methodsrc/lua_api/pathfind_api.rs:622
Example · Exact example · pathfind.lua
do
    local nav = lurek.pathfind.newNavGrid(15, 15)

    nav:fill(1)

    local ff = lurek.pathfind.newFlowField(nav)
    ff:calculateMulti({
        { x = 5, y = 5 },
        { x = 10, y = 10 },
    })

    local targets = ff:getTargets()
    print("targets = " .. #targets)
    print("first = " .. targets[1].x .. "," .. targets[1].y)
    print("last = " .. targets[#targets].x .. "," .. targets[#targets].y)
end
