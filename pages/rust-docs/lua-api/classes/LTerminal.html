Class

LTerminal

Lua-side userdata wrapping a terminal emulator grid with cell access, widgets, input, and rendering.

Docs home/lurek.terminal
Class Example · Exact example · terminal.lua
do
    ---@type LTerminal
    local term = lurek.terminal.newTerminal(80, 25)
    term:set(1, 1, "H", 1, 1, 1, 1, 0, 0, 0, 0)
    local ch, fr, fg, fb, fa, br, bg, bb, ba = term:get(1, 1)
    print("cell(1,1) ch=" .. ch .. " fg=(" .. fr .. "," .. fg .. "," .. fb .. ")")
end
