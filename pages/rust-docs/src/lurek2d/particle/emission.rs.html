1//! Spawn-offset sampling for particle emission shapes and area distributions.
2//!
3//! - Supports uniform, normal, ellipse, border, rectangle, ring, cone, star, and spiral modes.
4//! - Handles area-angle rotation for distribution-based offsets.
5
6use super::config::{AreaDistribution, EmissionShape, ParticleConfig};
7use super::math::{rand_normal, rand_range};
8/// Return a random spawn offset `(dx, dy)` sampled from the `ParticleConfig` area distribution and rotated by `area_angle`.
9pub fn emission_offset(config: &ParticleConfig) -> (f32, f32) {
10    let (dx, dy) = match config.area_distribution {
11        AreaDistribution::None => (0.0, 0.0),
12        AreaDistribution::Uniform => {
13            let x = rand_range(-config.area_width * 0.5, config.area_width * 0.5);
14            let y = rand_range(-config.area_height * 0.5, config.area_height * 0.5);
15            (x, y)
16        }
17        AreaDistribution::Normal => {
18            let x = (rand_normal() * 0.25).clamp(-0.5, 0.5) * config.area_width;
19            let y = (rand_normal() * 0.25).clamp(-0.5, 0.5) * config.area_height;
20            (x, y)
21        }
22        AreaDistribution::Ellipse => {
23            let angle = fastrand::f32() * 2.0 * std::f32::consts::PI;
24            let r = fastrand::f32().sqrt();
25            let x = angle.cos() * r * config.area_width * 0.5;
26            let y = angle.sin() * r * config.area_height * 0.5;
27            (x, y)
28        }
29        AreaDistribution::BorderEllipse => {
30            let angle = fastrand::f32() * 2.0 * std::f32::consts::PI;
31            let x = angle.cos() * config.area_width * 0.5;
32            let y = angle.sin() * config.area_height * 0.5;
33            (x, y)
34        }
35        AreaDistribution::BorderRectangle => {
36            let perimeter = 2.0 * config.area_width + 2.0 * config.area_height;
37            if perimeter < f32::EPSILON {
38                return (0.0, 0.0);
39            }
40            let t = fastrand::f32() * perimeter;
41            let hw = config.area_width * 0.5;
42            let hh = config.area_height * 0.5;
43            if t < config.area_width {
44                (t - hw, -hh)
45            } else if t < config.area_width + config.area_height {
46                (hw, t - config.area_width - hh)
47            } else if t < 2.0 * config.area_width + config.area_height {
48                (hw - (t - config.area_width - config.area_height), hh)
49            } else {
50                (-hw, hh - (t - 2.0 * config.area_width - config.area_height))
51            }
52        }
53    };
54    if config.area_angle.abs() > f32::EPSILON {
55        let cos_a = config.area_angle.cos();
56        let sin_a = config.area_angle.sin();
57        (dx * cos_a - dy * sin_a, dx * sin_a + dy * cos_a)
58    } else {
59        (dx, dy)
60    }
61}
62/// Return a random spawn offset `(dx, dy)` sampled from an `EmissionShape`; `Custom` always returns `(0, 0)`.
63pub fn emission_shape_offset(shape: &EmissionShape) -> (f32, f32) {
64    match shape {
65        EmissionShape::Point => (0.0, 0.0),
66        EmissionShape::Circle { radius, fill } => {
67            let angle = fastrand::f32() * 2.0 * std::f32::consts::PI;
68            let r = if *fill {
69                fastrand::f32().sqrt() * radius
70            } else {
71                *radius
72            };
73            (angle.cos() * r, angle.sin() * r)
74        }
75        EmissionShape::Rectangle { width, height } => {
76            let x = rand_range(-width * 0.5, *width * 0.5);
77            let y = rand_range(-height * 0.5, *height * 0.5);
78            (x, y)
79        }
80        EmissionShape::Ring {
81            inner_radius,
82            outer_radius,
83        } => {
84            let angle = fastrand::f32() * 2.0 * std::f32::consts::PI;
85            let r_sq = rand_range(inner_radius * inner_radius, outer_radius * outer_radius);
86            let r = r_sq.sqrt();
87            (angle.cos() * r, angle.sin() * r)
88        }
89        EmissionShape::Line { length, angle } => {
90            let t = rand_range(-0.5, 0.5);
91            (t * length * angle.cos(), t * length * angle.sin())
92        }
93        EmissionShape::Cone {
94            radius,
95            angle,
96            spread,
97        } => {
98            let a = angle + rand_range(-spread, *spread);
99            let r = fastrand::f32().sqrt() * radius;
100            (a.cos() * r, a.sin() * r)
101        }
102        EmissionShape::Star {
103            points,
104            outer_radius,
105            inner_radius,
106        } => {
107            let n = (*points).max(3) as f32;
108            let step = std::f32::consts::PI / n;
109            let segment = fastrand::u32(0..*points * 2);
110            let t = fastrand::f32();
111            let a0 = segment as f32 * step;
112            let a1 = a0 + step;
113            let r0 = if segment % 2 == 0 {
114                *outer_radius
115            } else {
116                *inner_radius
117            };
118            let r1 = if segment % 2 == 0 {
119                *inner_radius
120            } else {
121                *outer_radius
122            };
123            let x0 = a0.cos() * r0;
124            let y0 = a0.sin() * r0;
125            let x1 = a1.cos() * r1;
126            let y1 = a1.sin() * r1;
127            (x0 + (x1 - x0) * t, y0 + (y1 - y0) * t)
128        }
129        EmissionShape::Spiral {
130            revolutions,
131            radius,
132        } => {
133            let max_angle = *revolutions * 2.0 * std::f32::consts::PI;
134            let t = fastrand::f32();
135            let angle = t * max_angle;
136            let r = t * radius;
137            (angle.cos() * r, angle.sin() * r)
138        }
139        EmissionShape::Custom { .. } => (0.0, 0.0),
140    }
141}
