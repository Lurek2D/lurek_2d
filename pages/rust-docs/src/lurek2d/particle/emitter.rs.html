1//! Live particle emitter that owns the active particle pool, physics stepping, and sub-system list.
2//!
3//! - Per-frame integration: gravity, radial/tangential acceleration, linear damping, drag, orbit, and turbulence.
4//! - Point-attractor influence applied per particle each frame with distance falloff.
5//! - Axis-aligned bounce boundaries that reflect particles with configurable restitution.
6//! - Continuous emission via fractional accumulator, burst spawning, and three insert-order modes.
7//! - Child sub-system spawning on particle death with configurable burst count and config clone.
8//! - State machine: Active, Paused, Stopped with lifetime-based auto-stop.
9//! - Render command building: shape mapping, color/alpha/size interpolation, texture quads, and animated frames.
10//! - Warm-up simulation pre-populates the pool by stepping in fixed 50 ms increments.
11//! - Custom emission shape callback support via pending offset indices drained by the Lua bridge.
12//! - Death event queue exposing world-space position and velocity for gameplay hooks.
13

26/// Live particle emitter containing the active particle pool, physics state, and sub-system list.
27#[derive(Clone, Debug)]
28pub struct ParticleSystem {
29    /// Configuration snapshot used for spawning and update parameters.
30    pub config: ParticleConfig,

33    /// World-space X position of this emitter.

38    pub emit_accumulator: f32,
39    /// Current emitter operating state: active, paused, or stopped.
40    pub state: EmitterState,
41    /// Total age of the emitter in seconds since last `start`.
42    pub emitter_age: f32,
43    /// Emitter X position at the previous frame; used for motion interpolation.
44    pub prev_emitter_x: f32,
45    /// Emitter Y position at the previous frame; used for motion interpolation.
46    pub prev_emitter_y: f32,
47    /// Active point attractors applied to all particles each frame.
48    pub attractors: Vec<Attractor>,
49    /// Optional axis-aligned bounce boundary that reflects particles.
50    pub bounce_bounds: Option<BounceBounds>,
51    /// Child sub-systems spawned on particle death.

53    /// Particle pool indices waiting for a custom spawn-offset callback.
54    pub pending_custom_offsets: Vec<usize>,
55    /// `(world_x, world_y, vx, vy)` entries for particles that died this frame.
56    pub pending_deaths: Vec<(f32, f32, f32, f32)>,
57}

61        log_msg!(debug, PE01, "max {} particles", config.max_particles);
62        Self {
63            particles: Vec::with_capacity(config.max_particles as usize),
64            config,
65            emitter_x: 0.0,
66            emitter_y: 0.0,
67            emit_accumulator: 0.0,
68            state: EmitterState::Active,
69            emitter_age: 0.0,
70            prev_emitter_x: 0.0,
71            prev_emitter_y: 0.0,
72            attractors: Vec::new(),
73            bounce_bounds: None,
74            sub_systems: Vec::new(),
75            pending_custom_offsets: Vec::new(),
76            pending_deaths: Vec::new(),
77        }

79    /// Advance all particles by `dt` seconds: integrate physics, retire dead particles, and spawn new ones.
80    #[allow(clippy::unnecessary_unwrap)]
81    pub fn update(&mut self, dt: f32) {
82        if self.state == EmitterState::Paused {
83            return;
84        }
85        if self.state == EmitterState::Active {
86            self.emitter_age += dt;
87            if self.config.emitter_lifetime >= 0.0
88                && self.emitter_age >= self.config.emitter_lifetime
89            {
90                self.state = EmitterState::Stopped;
91            }
92        }
93        for p in &mut self.particles {
94            let dx = p.x - p.origin_x;
95            let dy = p.y - p.origin_y;
96            let dist = (dx * dx + dy * dy).sqrt();
97            if dist > f32::EPSILON {
98                let radial_x = dx / dist;
99                let radial_y = dy / dist;
100                let tangential_x = -radial_y;
101                let tangential_y = radial_x;
102                p.vx += (radial_x * p.radial_accel + tangential_x * p.tangential_accel) * dt;
103                p.vy += (radial_y * p.radial_accel + tangential_y * p.tangential_accel) * dt;
104            }
105            p.vx += self.config.gravity_x * dt;
106            p.vy += self.config.gravity_y * dt;
107            if p.linear_damping > f32::EPSILON {
108                let damping = 1.0 / (1.0 + p.linear_damping * dt);
109                p.vx *= damping;
110                p.vy *= damping;
111            }
112            if self.config.drag > f32::EPSILON {
113                let speed = (p.vx * p.vx + p.vy * p.vy).sqrt();
114                if speed > f32::EPSILON {
115                    let drag_factor = 1.0 / (1.0 + speed * self.config.drag * dt);
116                    p.vx *= drag_factor;
117                    p.vy *= drag_factor;
118                }
119            }
120            if self.config.orbit_speed.abs() > f32::EPSILON {
121                let orbit_angle = self.config.orbit_speed * dt;
122                let ca = orbit_angle.cos();
123                let sa = orbit_angle.sin();
124                let new_vx = p.vx * ca - p.vy * sa;
125                let new_vy = p.vx * sa + p.vy * ca;
126                p.vx = new_vx;
127                p.vy = new_vy;
128            }
129            if self.config.turbulence > f32::EPSILON {
130                p.vx += rand_normal() * self.config.turbulence * dt;
131                p.vy += rand_normal() * self.config.turbulence * dt;
132            }
133            let wx = p.x + self.emitter_x;
134            let wy = p.y + self.emitter_y;
135            for attr in &self.attractors {
136                let adx = attr.x - wx;
137                let ady = attr.y - wy;
138                let dist2 = adx * adx + ady * ady;
139                let r2 = attr.radius * attr.radius;
140                if dist2 < r2 && dist2 > f32::EPSILON {
141                    let dist = dist2.sqrt();
142                    let factor = attr.strength * (1.0 - dist / attr.radius) * dt;
143                    p.vx += (adx / dist) * factor;
144                    p.vy += (ady / dist) * factor;
145                }
146            }
147            p.x += p.vx * dt;
148            p.y += p.vy * dt;
149            if self.config.relative_rotation {
150                p.rotation = p.vy.atan2(p.vx);
151            } else {
152                p.rotation += p.spin * dt;
153            }
154            if let Some(ref bb) = self.bounce_bounds {
155                let wx = p.x + self.emitter_x;
156                let wy = p.y + self.emitter_y;
157                let r = bb.restitution.clamp(0.0, 1.0);
158                if wx < bb.x_min {
159                    p.x = bb.x_min - self.emitter_x;
160                    p.vx = p.vx.abs() * r;
161                } else if wx > bb.x_max {
162                    p.x = bb.x_max - self.emitter_x;
163                    p.vx = -p.vx.abs() * r;
164                }
165                if wy < bb.y_min {
166                    p.y = bb.y_min - self.emitter_y;
167                    p.vy = p.vy.abs() * r;
168                } else if wy > bb.y_max {
169                    p.y = bb.y_max - self.emitter_y;
170                    p.vy = -p.vy.abs() * r;
171                }
172            }
173            p.life -= dt;
174        }
175        if self.config.death_emitter.is_some() && self.config.death_burst_count > 0 {
176            let death_cfg = self.config.death_emitter.as_ref().unwrap().as_ref().clone();
177            let burst = self.config.death_burst_count;
178            let ex = self.emitter_x;
179            let ey = self.emitter_y;
180            let mut dead_data: Vec<(f32, f32, f32, f32)> = Vec::new();
181            self.particles.retain(|p| {
182                if p.life <= 0.0 {
183                    dead_data.push((ex + p.x, ey + p.y, p.vx, p.vy));
184                    false
185                } else {
186                    true
187                }
188            });
189            for &(dx, dy, _, _) in &dead_data {
190                let mut sub = ParticleSystem::new(death_cfg.clone());
191                sub.emitter_x = dx;
192                sub.emitter_y = dy;
193                sub.emit(burst);
194                sub.stop();
195                self.sub_systems.push(sub);
196            }
197            self.pending_deaths.extend(dead_data);
198        } else {
199            let ex = self.emitter_x;
200            let ey = self.emitter_y;
201            let mut dead_data: Vec<(f32, f32, f32, f32)> = Vec::new();
202            self.particles.retain(|p| {
203                if p.life <= 0.0 {
204                    dead_data.push((ex + p.x, ey + p.y, p.vx, p.vy));
205                    false
206                } else {
207                    true
208                }
209            });
210            self.pending_deaths.extend(dead_data);
211        }
212        self.sub_systems.retain_mut(|sub| {
213            sub.update(dt);
214            !sub.is_empty() || sub.is_active()
215        });
216        if self.state == EmitterState::Active {
217            self.emit_accumulator += self.config.emission_rate * dt;
218            let to_emit = self.emit_accumulator as u32;
219            self.emit_accumulator -= to_emit as f32;
220            for _ in 0..to_emit {
221                if self.particles.len() >= self.config.max_particles as usize {
222                    break;
223                }
224                self.emit_one();
225            }
226        }
227        self.prev_emitter_x = self.emitter_x;
228        self.prev_emitter_y = self.emitter_y;
229    }
230    /// Spawn a single particle using the current config; inserts according to `insert_mode`.
231    fn emit_one(&mut self) {
232        let lifetime = rand_range(self.config.lifetime_min, self.config.lifetime_max);
233        let speed = rand_range(self.config.speed_min, self.config.speed_max);
234        let angle = self.config.direction + rand_range(-self.config.spread, self.config.spread);
235        let base_spin = rand_range(self.config.spin_min, self.config.spin_max);
236        let spin = base_spin * (1.0 - self.config.spin_variation * fastrand::f32());
237        let rotation = rand_range(self.config.rotation_min, self.config.rotation_max);
238        let radial_accel = rand_range(self.config.radial_accel_min, self.config.radial_accel_max);
239        let tangential_accel = rand_range(
240            self.config.tangential_accel_min,
241            self.config.tangential_accel_max,
242        );
243        let linear_damping = rand_range(
244            self.config.linear_damping_min,
245            self.config.linear_damping_max,
246        );
247        let size_variation = self.config.size_variation * fastrand::f32();
248        let (offset_x, offset_y) = if self.config.emission_shape != EmissionShape::Point {
249            emission_shape_offset(&self.config.emission_shape)
250        } else {
251            emission_offset(&self.config)
252        };
253        let particle = Particle {
254            x: offset_x,
255            y: offset_y,
256            vx: angle.cos() * speed,
257            vy: angle.sin() * speed,
258            life: lifetime,
259            max_life: lifetime,
260            rotation,
261            spin,
262            radial_accel,
263            tangential_accel,
264            linear_damping,
265            size_variation,
266            origin_x: offset_x,
267            origin_y: offset_y,
268            shape_seed: fastrand::u32(..),
269        };
270        let new_idx = match self.config.insert_mode {
271            InsertMode::Top => {
272                let idx = self.particles.len();
273                self.particles.push(particle);
274                idx
275            }
276            InsertMode::Bottom => {
277                self.particles.insert(0, particle);
278                0
279            }
280            InsertMode::Random => {
281                let idx = if self.particles.is_empty() {
282                    0
283                } else {
284                    fastrand::usize(..=self.particles.len())
285                };
286                self.particles.insert(idx, particle);
287                idx
288            }
289        };
290        if matches!(self.config.emission_shape, EmissionShape::Custom { .. }) {
291            self.pending_custom_offsets.push(new_idx);
292        }
293    }
294    /// Burst-spawn up to `count` particles immediately, capped by `max_particles`.
295    pub fn emit(&mut self, count: u32) {
296        for _ in 0..count {
297            if self.particles.len() >= self.config.max_particles as usize {
298                break;
299            }
300            self.emit_one();
301        }
302    }
303    /// Return the number of live particles in the pool.

307    /// Clear all particles and reset the accumulator and age.
308    pub fn reset(&mut self) {
309        log_msg!(debug, PE04);
310        self.particles.clear();
311        self.emit_accumulator = 0.0;
312        self.emitter_age = 0.0;
313        self.pending_custom_offsets.clear();

315    }
316    /// Transition to `Active` and reset `emitter_age` to zero.
317    pub fn start(&mut self) {

320        self.emitter_age = 0.0;
321    }
322    /// Transition to `Stopped`; existing particles continue to live but no new ones are emitted.

327    /// Transition to `Paused`; update loop stops advancing but particles freeze in place.
328    pub fn pause(&mut self) {

331    /// Resume from `Paused` or `Stopped`; transitions to `Active`.
332    pub fn resume(&mut self) {
333        if self.state == EmitterState::Paused || self.state == EmitterState::Stopped {
334            self.state = EmitterState::Active;
335        }
336    }

368    /// Build `RenderCommand` values for all live particles at world offset `(ox, oy)`, including sub-systems.
369    pub fn build_render_commands(&self, ox: f32, oy: f32) -> Vec<RenderCommand> {
370        if self.particles.is_empty() {
371            return Vec::new();
372        }
373        let mut instances = Vec::with_capacity(self.particles.len());
374        for p in &self.particles {
375            let t = 1.0 - (p.life / p.max_life);
376            let size = interpolate_sizes(&self.config.sizes, t, p.size_variation);
377            let color_t = if self.config.color_by_speed {
378                let speed = (p.vx * p.vx + p.vy * p.vy).sqrt();
379                let range = self.config.speed_color_max - self.config.speed_color_min;
380                if range > f32::EPSILON {
381                    ((speed - self.config.speed_color_min) / range).clamp(0.0, 1.0)
382                } else {
383                    0.0
384                }
385            } else {
386                t
387            };
388            let [r, g, b, mut a] = interpolate_colors(&self.config.colors, color_t);
389            if !self.config.alpha_keyframes.is_empty() {
390                a = interpolate_alphas(&self.config.alpha_keyframes, t);
391            }
392            let px = ox + self.emitter_x + p.x;
393            let py = oy + self.emitter_y + p.y;
394            let render_shape = match &self.config.shape {
395                ParticleShape::Square => ParticleRenderShape::Square,
396                ParticleShape::Circle => ParticleRenderShape::Circle,
397                ParticleShape::Triangle => ParticleRenderShape::Triangle,
398                ParticleShape::Spark => ParticleRenderShape::Spark,
399                ParticleShape::Diamond => ParticleRenderShape::Diamond,
400                ParticleShape::Shrapnel { edges } => ParticleRenderShape::Shrapnel {
401                    edges: *edges,
402                    seed: p.shape_seed,
403                },
404                ParticleShape::Ray { aspect } => ParticleRenderShape::Ray { aspect: *aspect },
405                ParticleShape::Puff => ParticleRenderShape::Puff,
406                ParticleShape::Ring { thickness } => ParticleRenderShape::Ring {
407                    thickness: *thickness,
408                },
409                ParticleShape::Capsule => ParticleRenderShape::Capsule,
410            };
411            let (texture_key, quad, quad_tex_dims) = if let Some(tex_key) = self.config.texture_id {
412                if !self.config.quads.is_empty() {
413                    let quad_idx = if self.config.animated_frames > 0 {
414                        let age = p.max_life - p.life;
415                        (age * self.config.frame_rate) as usize
416                            % (self.config.animated_frames as usize).min(self.config.quads.len())
417                    } else {
418                        ((t * self.config.quads.len() as f32) as usize)
419                            .min(self.config.quads.len() - 1)
420                    };
421                    let [qx, qy, qw, qh] = self.config.quads[quad_idx];
422                    (Some(tex_key), Some([qx, qy, qw, qh]), Some((qw, qh)))
423                } else {
424                    (Some(tex_key), None, None)
425                }
426            } else {
427                (None, None, None)
428            };
429            instances.push(ParticleInstance {
430                x: px,
431                y: py,
432                r,
433                g,
434                b,
435                a,
436                rotation: p.rotation,
437                size,
438                shape: render_shape,
439                texture_key,
440                quad,
441                quad_tex_dims,
442            });
443        }
444        let mut all_cmds = vec![RenderCommand::DrawParticleSystem {
445            particles: instances,
446        }];
447        for sub in &self.sub_systems {
448            all_cmds.extend(sub.build_render_commands(ox, oy));
449        }
450        all_cmds
451    }
452    /// Run the update loop for up to `seconds` in 50 ms steps to pre-populate the particle pool.
453    pub fn warm_up(&mut self, seconds: f32) {
454        const STEP: f32 = 0.05;
455        let clamped = seconds.clamp(0.0, 30.0);
456        let mut remaining = clamped;
457        while remaining > 0.0 {
458            let dt = remaining.min(STEP);
459            self.update(dt);
460            remaining -= dt;
461        }
462    }

480    /// Set the axis-aligned bounce boundary; particles reflect on crossing any edge.
481    pub fn set_bounds(&mut self, x_min: f32, x_max: f32, y_min: f32, y_max: f32, restitution: f32) {
482        self.bounce_bounds = Some(BounceBounds {
483            x_min,
484            x_max,
485            y_min,
486            y_max,
487            restitution: restitution.clamp(0.0, 1.0),
488        });
489    }
490    /// Remove the bounce boundary. This function is part of the public API.
491    pub fn clear_bounds(&mut self) {
492        self.bounce_bounds = None;
493    }
494    /// Append a child sub-system; returns its index in `sub_systems`.
495    pub fn add_sub_system(&mut self, config: ParticleConfig) -> usize {
496        let sub = ParticleSystem::new(config);
497        self.sub_systems.push(sub);
498        self.sub_systems.len() - 1
499    }
500    /// Return the number of active sub-systems.

504    /// Drain and return all `(world_x, world_y, vx, vy)` death events accumulated since the last call.
505    pub fn drain_pending_deaths(&mut self) -> Vec<(f32, f32, f32, f32)> {
506        std::mem::take(&mut self.pending_deaths)
507    }
508    /// Drain and return particle pool indices that need a custom spawn-offset callback applied.
509    pub fn drain_custom_offsets(&mut self) -> Vec<usize> {
510        std::mem::take(&mut self.pending_custom_offsets)
511    }
