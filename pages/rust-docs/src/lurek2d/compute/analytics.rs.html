10/// Compute cumulative sum and return a 1D array with running totals.
11pub fn cumsum(a: &NdArray) -> Result<NdArray, String> {
12    let n = a.size();
13    let mut out = NdArray::zeros(&[n], a.dtype())?;
14    let mut acc = 0.0_f64;
15    for i in 0..n {
16        acc += a.get_f64(i);
17        out.set_f64(i, acc);
18    }
19    Ok(out)
20}
21/// Compute finite difference of requested order and return derived 1D array.
22pub fn diff(a: &NdArray, order: usize) -> Result<NdArray, String> {
23    if order == 0 {
24        return Ok(a.clone());
25    }
26    let n = a.size();
27    if order >= n {
28        return Err(format!(
29            "diff order {} >= array size {}; result would be empty",
30            order, n
31        ));
32    }
33    let mut vals: Vec<f64> = (0..n).map(|i| a.get_f64(i)).collect();
34    for _ in 0..order {
35        vals = vals.windows(2).map(|w| w[1] - w[0]).collect();
36    }
37    let shape = [vals.len()];
38    let mut out = NdArray::zeros(&shape, a.dtype())?;
39    for (i, v) in vals.into_iter().enumerate() {
40        out.set_f64(i, v);
41    }
42    Ok(out)
43}
44/// Compute histogram bins and return (lo, hi, count) tuples for each bin.
45pub fn histogram(
46    a: &NdArray,
47    bins: usize,
48    range_lo: Option<f64>,
49    range_hi: Option<f64>,
50) -> Result<Vec<(f64, f64, u64)>, String> {
51    if bins == 0 {
52        return Err("histogram: bins must be >= 1".to_string());
53    }
54    let n = a.size();
55    if n == 0 {
56        return Err("histogram: empty array".to_string());
57    }
58    let lo = range_lo.unwrap_or_else(|| {
59        let mut m = a.get_f64(0);
60        for i in 1..n {
61            let v = a.get_f64(i);
62            if v < m {
63                m = v;
64            }
65        }
66        m
67    });
68    let hi = range_hi.unwrap_or_else(|| {
69        let mut m = a.get_f64(0);
70        for i in 1..n {
71            let v = a.get_f64(i);
72            if v > m {
73                m = v;
74            }
75        }
76        m
77    });
78    if hi <= lo {
79        return Err(format!(
80            "histogram: range_hi ({hi}) must be > range_lo ({lo})"
81        ));
82    }
83    let width = (hi - lo) / bins as f64;
84    let mut counts = vec![0u64; bins];
85    for i in 0..n {
86        let v = a.get_f64(i);
87        if v < lo || v > hi {
88            continue;
89        }
90        let mut bin = ((v - lo) / width) as usize;
91        if bin >= bins {
92            bin = bins - 1;
93        }
94        counts[bin] += 1;
95    }
96    let result = (0..bins)
97        .map(|b| {
98            let bin_lo = lo + b as f64 * width;
99            let bin_hi = lo + (b + 1) as f64 * width;
100            (bin_lo, bin_hi, counts[b])
101        })
102        .collect();
103    Ok(result)
104}
105/// Compute percentile value and return interpolated sample at p in [0, 100].
106pub fn percentile(a: &NdArray, p: f64) -> Result<f64, String> {
107    if !(0.0..=100.0).contains(&p) {
108        return Err(format!("percentile p must be in [0, 100], got {p}"));
109    }
110    let n = a.size();
111    if n == 0 {
112        return Err("percentile: empty array".to_string());
113    }
114    let mut vals: Vec<f64> = (0..n).map(|i| a.get_f64(i)).collect();
115    vals.sort_by(|x, y| x.partial_cmp(y).unwrap_or(std::cmp::Ordering::Equal));
116    let idx = (p / 100.0) * (n - 1) as f64;
117    let lo = idx.floor() as usize;
118    let hi = idx.ceil() as usize;
119    if lo == hi {
120        return Ok(vals[lo]);
121    }
122    let frac = idx - lo as f64;
123    Ok(vals[lo] * (1.0 - frac) + vals[hi] * frac)
124}
125/// Compute population covariance and return scalar covariance between arrays.
126pub fn covariance(a: &NdArray, b: &NdArray) -> Result<f64, String> {
127    let n = a.size();
128    if n != b.size() {
129        return Err(format!("covariance: size mismatch {} vs {}", n, b.size()));
130    }
131    if n == 0 {
132        return Err("covariance: empty arrays".to_string());
133    }
134    let mean_a: f64 = (0..n).map(|i| a.get_f64(i)).sum::<f64>() / n as f64;
135    let mean_b: f64 = (0..n).map(|i| b.get_f64(i)).sum::<f64>() / n as f64;
136    let cov: f64 = (0..n)
137        .map(|i| (a.get_f64(i) - mean_a) * (b.get_f64(i) - mean_b))
138        .sum::<f64>()
139        / n as f64;
140    Ok(cov)
141}
142/// Compute Pearson correlation and return normalized linear correlation coefficient.
143pub fn pearson_corr(a: &NdArray, b: &NdArray) -> Result<f64, String> {
144    let n = a.size();
145    if n != b.size() {
146        return Err(format!("pearson_corr: size mismatch {} vs {}", n, b.size()));
147    }
148    if n < 2 {
149        return Err("pearson_corr: need at least 2 elements".to_string());
150    }
151    let mean_a: f64 = (0..n).map(|i| a.get_f64(i)).sum::<f64>() / n as f64;
152    let mean_b: f64 = (0..n).map(|i| b.get_f64(i)).sum::<f64>() / n as f64;
153    let mut num = 0.0_f64;
154    let mut ss_a = 0.0_f64;
155    let mut ss_b = 0.0_f64;
156    for i in 0..n {
157        let da = a.get_f64(i) - mean_a;
158        let db = b.get_f64(i) - mean_b;
159        num += da * db;
160        ss_a += da * da;
161        ss_b += db * db;
162    }
163    let denom = (ss_a * ss_b).sqrt();
164    if denom == 0.0 {
165        return Err("pearson_corr: zero variance in one or both arrays".to_string());
166    }
167    Ok(num / denom)
168}
169/// Normalize values to output range and return scaled 1D array.
170pub fn normalize_range(a: &NdArray, out_min: f64, out_max: f64) -> Result<NdArray, String> {
171    if out_max <= out_min {
172        return Err(format!(
173            "normalize_range: out_max ({out_max}) must be > out_min ({out_min})"
174        ));
175    }
176    let n = a.size();
177    let mut lo = a.get_f64(0);
178    let mut hi = a.get_f64(0);
179    for i in 1..n {
180        let v = a.get_f64(i);
181        if v < lo {
182            lo = v;
183        }
184        if v > hi {
185            hi = v;
186        }
187    }
188    let span = hi - lo;
189    let out_span = out_max - out_min;
190    let mut out = NdArray::zeros(&[n], a.dtype())?;
191    for i in 0..n {
192        let norm = if span == 0.0 {
193            0.5
194        } else {
195            (a.get_f64(i) - lo) / span
196        };
197        out.set_f64(i, out_min + norm * out_span);
198    }
199    Ok(out)
200}
201/// Normalize values to z-scores and return array with mean-zero unit variance.
202pub fn zscore(a: &NdArray) -> Result<NdArray, String> {
203    let n = a.size();
204    if n == 0 {
205        return Err("zscore: empty array".to_string());
206    }
207    let mean: f64 = (0..n).map(|i| a.get_f64(i)).sum::<f64>() / n as f64;
208    let var: f64 = (0..n).map(|i| (a.get_f64(i) - mean).powi(2)).sum::<f64>() / n as f64;
209    let std = var.sqrt();
210    if std == 0.0 {
211        return Err("zscore: zero standard deviation".to_string());
212    }
213    let mut out = NdArray::zeros(&[n], a.dtype())?;
214    for i in 0..n {
215        out.set_f64(i, (a.get_f64(i) - mean) / std);
216    }
217    Ok(out)
218}

220/// Compute full 1D convolution and return output signal array.
221pub fn convolve1d(signal: &NdArray, kernel: &NdArray) -> Result<NdArray, String> {
222    if signal.ndim() != 1 {
223        return Err(format!(
224            "convolve1d: signal must be 1D, got {}D",
225            signal.ndim()
226        ));
227    }
228    if kernel.ndim() != 1 {
229        return Err(format!(
230            "convolve1d: kernel must be 1D, got {}D",
231            kernel.ndim()
232        ));
233    }
234    let sn = signal.size();
235    let kn = kernel.size();
236    let out_len = sn + kn - 1;
237    let mut out = NdArray::zeros(&[out_len], signal.dtype())?;
238    let kflip: Vec<f64> = (0..kn).rev().map(|i| kernel.get_f64(i)).collect();
239    for o in 0..out_len {
240        let mut acc = 0.0_f64;
241        for k in 0..kn {
242            let s_idx = o as isize - k as isize;
243            if s_idx >= 0 && (s_idx as usize) < sn {
244                acc += signal.get_f64(s_idx as usize) * kflip[k];
245            }
246        }
247        out.set_f64(o, acc);
248    }
249    Ok(out)
250}
251/// Compute valid 1D correlation and return sliding dot-product array.
252pub fn correlate1d(signal: &NdArray, template: &NdArray) -> Result<NdArray, String> {
253    if signal.ndim() != 1 {
254        return Err(format!(
255            "correlate1d: signal must be 1D, got {}D",
256            signal.ndim()
257        ));
258    }
259    if template.ndim() != 1 {
260        return Err(format!(
261            "correlate1d: template must be 1D, got {}D",
262            template.ndim()
263        ));
264    }
265    let sn = signal.size();
266    let tn = template.size();
267    if tn > sn {
268        return Err(format!(
269            "correlate1d: template length {tn} > signal length {sn}"
270        ));
271    }
272    let out_len = sn - tn + 1;
273    let mut out = NdArray::zeros(&[out_len], signal.dtype())?;
274    for o in 0..out_len {
275        let mut acc = 0.0_f64;
276        for k in 0..tn {
277            acc += signal.get_f64(o + k) * template.get_f64(k);
278        }
279        out.set_f64(o, acc);
280    }
281    Ok(out)
