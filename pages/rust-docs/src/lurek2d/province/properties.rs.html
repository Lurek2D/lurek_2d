1//! Province property storage: per-province key-value metadata and stat tables.
2//!
3//! - `ProvinceProperties` maps `ProvinceId → HashMap<String, PropertyValue>`.
4//! - `PropertyValue` is an enum covering `Int`, `Float`, `Bool`, and `Text` variants.
5//! - Properties are set from Lua via `lurek.province.set_property(id, key, value)`.
6//! - Serialized into the save file as a flat list for fast round-trip loading.
7
8use std::collections::HashMap;
9
10/// Generic per-province property store.
11/// Game defines what "population", "food" etc. mean through Lua.
12pub struct ProvinceProperties {
13    /// Named numeric values per province.
14    numeric: HashMap<u32, HashMap<String, f64>>,
15    /// Named string attributes per province.
16    string_attrs: HashMap<u32, HashMap<String, String>>,
17    /// Bitfield flags per province (64 custom flags).
18    flags: HashMap<u32, u64>,
19}
20
21impl ProvinceProperties {
22    /// Create a new empty property store with no province data.
23    pub fn new() -> Self {
24        Self {
25            numeric: HashMap::new(),
26            string_attrs: HashMap::new(),
27            flags: HashMap::new(),
28        }
29    }
30
31    /// Set a named numeric value for the given province.
32    pub fn set_numeric(&mut self, province_id: u32, key: &str, value: f64) {
33        self.numeric
34            .entry(province_id)
35            .or_default()
36            .insert(key.to_string(), value);
37    }
38
39    /// Get the named numeric value for a province, or `None` if not set.
40    pub fn get_numeric(&self, province_id: u32, key: &str) -> Option<f64> {
41        self.numeric.get(&province_id)?.get(key).copied()
42    }
43
44    /// Set a named string attribute for the given province.
45    pub fn set_string(&mut self, province_id: u32, key: &str, value: &str) {
46        self.string_attrs
47            .entry(province_id)
48            .or_default()
49            .insert(key.to_string(), value.to_string());
50    }
51
52    /// Get the named string attribute for a province, or `None` if not set.
53    pub fn get_string(&self, province_id: u32, key: &str) -> Option<&str> {
54        self.string_attrs
55            .get(&province_id)?
56            .get(key)
57            .map(|s| s.as_str())
58    }
59
60    /// Set or clear a bitfield flag (0-63) for the given province.
61    pub fn set_flag(&mut self, province_id: u32, bit: u8, value: bool) {
62        let flags = self.flags.entry(province_id).or_insert(0);
63        if value {
64            *flags |= 1u64 << (bit.min(63));
65        } else {
66            *flags &= !(1u64 << (bit.min(63)));
67        }
68    }
69
70    /// Return `true` if the specified bitfield flag is set for the given province.
71    pub fn has_flag(&self, province_id: u32, bit: u8) -> bool {
72        self.flags
73            .get(&province_id)
74            .is_some_and(|f| f & (1u64 << (bit.min(63))) != 0)
75    }
76
77    /// Remove all numeric, string, and flag data stored for the given province.
78    pub fn clear_province(&mut self, province_id: u32) {
79        self.numeric.remove(&province_id);
80        self.string_attrs.remove(&province_id);
81        self.flags.remove(&province_id);
82    }
83
84    /// Return a sorted list of all province IDs that have any stored properties.
85    pub fn province_ids(&self) -> Vec<u32> {
86        let mut ids: std::collections::HashSet<u32> = self.numeric.keys().copied().collect();
87        ids.extend(self.string_attrs.keys());
88        ids.extend(self.flags.keys());
89        let mut v: Vec<u32> = ids.into_iter().collect();
90        v.sort_unstable();
91        v
92    }
93}
94
95impl Default for ProvinceProperties {
96    fn default() -> Self {
97        Self::new()
98    }
99}
