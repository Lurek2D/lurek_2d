1//! Province GPU upload helpers for id, border-index, and distance-field textures.
2//!
3//! - Consistent texture descriptors for `R32Uint`, `R16Uint`, and `R8Unorm` data.
4//! - Byte packing utilities used by upload paths and unit tests.
5
6/// GPU texture bundle used by the province map renderer.
7pub struct ProvinceGpuTextures {
8    /// Province id map texture (`R32Uint`).
9    pub province_id_texture: wgpu::Texture,
10    /// Province id map view.
11    pub province_id_view: wgpu::TextureView,
12    /// Border index texture (`R16Uint`).
13    pub border_index_texture: wgpu::Texture,
14    /// Border index texture view.
15    pub border_index_view: wgpu::TextureView,
16    /// Distance field texture (`R8Unorm`).
17    pub distance_field_texture: wgpu::Texture,
18    /// Distance field texture view.
19    pub distance_field_view: wgpu::TextureView,
20    /// Grid width in pixels.
21    pub width: u32,
22    /// Grid height in pixels.
23    pub height: u32,
24}

26fn make_extent(width: u32, height: u32) -> wgpu::Extent3d {
27    wgpu::Extent3d {
28        width,
29        height,
30        depth_or_array_layers: 1,
31    }
32}

34/// Pack `u32` row-major pixels into little-endian bytes for `R32Uint` uploads.
35pub fn pack_u32_pixels_le(data: &[u32]) -> Vec<u8> {
36    let mut out = Vec::with_capacity(data.len() * 4);
37    for value in data {
38        out.extend_from_slice(&value.to_le_bytes());
39    }
40    out
41}
42
43/// Pack `u16` row-major pixels into little-endian bytes for `R16Uint` uploads.
44pub fn pack_u16_pixels_le(data: &[u16]) -> Vec<u8> {
45    let mut out = Vec::with_capacity(data.len() * 2);
46    for value in data {
47        out.extend_from_slice(&value.to_le_bytes());
48    }
49    out
50}

52/// Create and upload province id map texture (`R32Uint`).
53pub fn create_province_id_texture(
54    device: &wgpu::Device,
55    queue: &wgpu::Queue,
56    width: u32,
57    height: u32,
58    province_ids: &[u32],
59) -> (wgpu::Texture, wgpu::TextureView) {
60    assert_eq!(
61        province_ids.len(),
62        (width as usize).saturating_mul(height as usize),
63        "province_ids length must be width*height"
64    );
65
66    let texture = device.create_texture(&wgpu::TextureDescriptor {
67        label: Some("province_id_map"),
68        size: make_extent(width, height),
69        mip_level_count: 1,
70        sample_count: 1,
71        dimension: wgpu::TextureDimension::D2,
72        format: wgpu::TextureFormat::R32Uint,
73        usage: wgpu::TextureUsages::TEXTURE_BINDING | wgpu::TextureUsages::COPY_DST,
74        view_formats: &[],
75    });
76
77    let bytes = pack_u32_pixels_le(province_ids);
78    queue.write_texture(
79        wgpu::ImageCopyTexture {
80            texture: &texture,
81            mip_level: 0,
82            origin: wgpu::Origin3d::ZERO,
83            aspect: wgpu::TextureAspect::All,
84        },
85        &bytes,
86        wgpu::ImageDataLayout {
87            offset: 0,
88            bytes_per_row: Some(4 * width),
89            rows_per_image: Some(height),
90        },
91        make_extent(width, height),
92    );
93
94    let view = texture.create_view(&wgpu::TextureViewDescriptor::default());
95    (texture, view)
96}

189/// Create all core province textures in one call.
190pub fn create_province_gpu_textures(
191    device: &wgpu::Device,
192    queue: &wgpu::Queue,
193    width: u32,
194    height: u32,
195    province_ids: &[u32],
196    border_index: &[u16],
197    distance_field: &[u8],
198) -> ProvinceGpuTextures {
199    let (province_id_texture, province_id_view) =
200        create_province_id_texture(device, queue, width, height, province_ids);
201    let (border_index_texture, border_index_view) =
202        create_border_index_texture(device, queue, width, height, border_index);
203    let (distance_field_texture, distance_field_view) =
204        create_distance_field_texture(device, queue, width, height, distance_field);
205
206    ProvinceGpuTextures {
207        province_id_texture,
208        province_id_view,
209        border_index_texture,
210        border_index_view,
211        distance_field_texture,
212        distance_field_view,
213        width,
214        height,
215    }
