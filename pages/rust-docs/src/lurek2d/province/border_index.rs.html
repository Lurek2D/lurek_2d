1//! Precompute border-pair index map from province id grid.
2//!
3//! - Assigns a stable u16 pair id for each detected border pixel.
4//! - Optional dilation expands border coverage for thick styled borders.
5
6use crate::province::registry::ProvinceRegistry;
7use crate::province::types::{BorderPairStyle, ProvinceId};
8use std::collections::HashMap;
9
10/// Dense per-pixel border pair index map.
11#[derive(Debug, Clone, PartialEq, Eq)]
12pub struct ProvinceBorderIndex {
13    /// Row-major pair ids. Zero means "not a border pixel".
14    pub data: Vec<u16>,
15    /// Grid width in pixels.
16    pub width: u32,
17    /// Grid height in pixels.
18    pub height: u32,
19    /// Pair id -> (province_a, province_b), index 0 unused.
20    pub id_to_pair: Vec<(ProvinceId, ProvinceId)>,
21    /// (province_a, province_b) -> pair id.
22    pub pair_to_id: HashMap<(ProvinceId, ProvinceId), u16>,
23}
24
25impl ProvinceBorderIndex {
26    /// Return pair id at (x, y), or None when out of bounds.
27    pub fn at(&self, x: u32, y: u32) -> Option<u16> {
28        if x >= self.width || y >= self.height {
29            return None;
30        }
31        self.data.get((y * self.width + x) as usize).copied()
32    }
33
34    /// Return number of unique province pairs referenced by this index.
35    pub fn pair_count(&self) -> usize {
36        self.id_to_pair.len().saturating_sub(1)
37    }
38}
39
40fn norm_pair(a: ProvinceId, b: ProvinceId) -> (ProvinceId, ProvinceId) {
41    if a < b {
42        (a, b)
43    } else {
44        (b, a)
45    }
46}
47
48fn assign_pair_id(
49    pair_to_id: &mut HashMap<(ProvinceId, ProvinceId), u16>,
50    id_to_pair: &mut Vec<(ProvinceId, ProvinceId)>,
51    a: ProvinceId,
52    b: ProvinceId,
53) -> Option<u16> {
54    if a == b {
55        return None;
56    }
57    let pair = norm_pair(a, b);
58    if let Some(id) = pair_to_id.get(&pair) {
59        return Some(*id);
60    }
61    let next = (id_to_pair.len()) as u16;
62    if next == 0 {
63        return None;
64    }
65    pair_to_id.insert(pair, next);
66    id_to_pair.push(pair);
67    Some(next)
68}

70/// Build border index map from raw province id grid.
71pub fn build_border_index(grid: &[u32], width: u32, height: u32) -> ProvinceBorderIndex {
72    let expected_len = (width as usize).saturating_mul(height as usize);
73    assert_eq!(
74        grid.len(), expected_len,
75        "grid length must be width*height"
76    );
77
78    let mut data = vec![0_u16; expected_len];
79    let mut pair_to_id: HashMap<(ProvinceId, ProvinceId), u16> = HashMap::new();
80    let mut id_to_pair: Vec<(ProvinceId, ProvinceId)> = vec![(ProvinceId(0), ProvinceId(0))];
81
82    for y in 0..height {
83        for x in 0..width {
84            let idx = (y * width + x) as usize;
85            let id = ProvinceId(grid[idx]);
86
87            let mut assigned = 0_u16;
88            if x + 1 < width {
89                let right = ProvinceId(grid[(y * width + (x + 1)) as usize]);
90                if right != id {
91                    if let Some(pid) = assign_pair_id(&mut pair_to_id, &mut id_to_pair, id, right) {
92                        assigned = pid;
93                    }
94                }
95            }
96            if assigned == 0 && y + 1 < height {
97                let down = ProvinceId(grid[((y + 1) * width + x) as usize]);
98                if down != id {
99                    if let Some(pid) = assign_pair_id(&mut pair_to_id, &mut id_to_pair, id, down) {
100                        assigned = pid;
101                    }
102                }
103            }
104            data[idx] = assigned;
105        }
106    }
107
108    ProvinceBorderIndex {
109        data,
110        width,
111        height,
112        id_to_pair,
113        pair_to_id,
114    }
115}

117/// Expand border pixels by radius per pair style thickness.
118pub fn dilate_border_index_with_styles(
119    index: &mut ProvinceBorderIndex,
120    styles: &HashMap<(ProvinceId, ProvinceId), BorderPairStyle>,
121) {
122    let width = index.width;
123    let height = index.height;
124    let src = index.data.clone();
125    let mut dst = src.clone();
126
127    for y in 0..height {
128        for x in 0..width {
129            let idx = (y * width + x) as usize;
130            let pair_id = src[idx];
131            if pair_id == 0 {
132                continue;
133            }
134            let pair = index.id_to_pair[pair_id as usize];
135            let style = styles.get(&pair).copied().unwrap_or_default();
136            let radius = ((style.thickness.max(1.0) - 1.0) * 0.5).ceil() as i32;
137            if radius <= 0 {
138                continue;
139            }
140            for oy in -radius..=radius {
141                for ox in -radius..=radius {
142                    let nx = x as i32 + ox;
143                    let ny = y as i32 + oy;
144                    if nx < 0 || ny < 0 {
145                        continue;
146                    }
147                    if nx >= width as i32 || ny >= height as i32 {
148                        continue;
149                    }
150                    let nidx = (ny as u32 * width + nx as u32) as usize;
151                    if dst[nidx] == 0 {
152                        dst[nidx] = pair_id;
153                    }
154                }
155            }
156        }
157    }
158
159    index.data = dst;
160}

162/// Build border index map directly from current registry pixel grid.
163pub fn build_border_index_from_registry(registry: &ProvinceRegistry) -> ProvinceBorderIndex {
164    let width = registry.width();
165    let height = registry.height();
166    let mut ids = Vec::with_capacity((width as usize).saturating_mul(height as usize));
167    for y in 0..height {
168        for x in 0..width {
169            ids.push(registry.get_at(x, y));
170        }
171    }
172    build_border_index(&ids, width, height)
173}
