1//! Province map rendering: convert registry data into a flat RenderCommand list.
2//!
3//! - Viewport culling based on screen bounds and zoom/pan transform.
4//! - Fill rendering via per-province span rectangles coloured by the active map mode.
5//! - Border rendering with config-based color from registered border types.
6//! - Capital dot markers and text labels with shadow offset.
7//! - Hover and selection highlight outlines for interactive feedback.
8
9use crate::province::map_modes::resolve_color_fallback;
10use crate::province::registry::ProvinceRegistry;
11use crate::province::types::{BorderPairFlags, ProvinceId};
12use crate::render::renderer::{DrawMode, RenderCommand};
13use crate::runtime::resource_keys::FontKey;

15/// Strategic/tactical map rendering mode.
16#[derive(Debug, Clone, Copy, PartialEq, Eq)]
17pub enum ProvinceZoomMode {
18    /// Derive mode from zoom threshold.
19    Auto,
20    /// Minimal detail mode intended for world-scale view.
21    Strategic,
22    /// Full detail mode intended for close map view.
23    Tactical,
24}
25
26/// Options controlling what gets rendered and how the province map is projected onto the screen.
27#[derive(Debug, Clone)]
28pub struct ProvinceRenderOptions {
29    /// Horizontal screen translation applied before zoom.
30    pub x: f32,
31    /// Vertical screen translation applied before zoom.
32    pub y: f32,
33    /// Zoom multiplier applied after translation.
34    pub zoom: f32,
35    /// Size in screen pixels of one province map pixel; combined with zoom for final scale.
36    pub pixel_size: f32,
37    /// Screen width in pixels, used for viewport culling.
38    pub screen_w: f32,
39    /// Screen height in pixels, used for viewport culling.
40    pub screen_h: f32,
41    /// Active map mode name that drives fill colour selection.
42    pub map_mode: String,
43    /// When true, emit fill rectangles for province spans.
44    pub draw_fills: bool,
45    /// When true, emit line segments for province borders.
46    pub draw_borders: bool,
47    /// When true, emit text labels at province label or centroid positions.
48    pub draw_labels: bool,
49    /// When true, emit capital dot markers.
50    pub draw_capitals: bool,
51    /// Line width in screen pixels for border segments.
52    pub border_width: f32,
53    /// Optional explicit zoom mode override.
54    pub zoom_mode: Option<ProvinceZoomMode>,
55    /// Auto mode threshold: zoom >= threshold enters tactical mode.
56    pub tactical_zoom_threshold: f32,
57    /// When true and tactical mode is active, draw adjacency roads between visible capitals.
58    pub draw_roads: bool,
59    /// Province to highlight with a white hover outline, or None.
60    pub hovered_id: Option<ProvinceId>,
61    /// Province to highlight with a yellow selection outline, or None.
62    pub selected_id: Option<ProvinceId>,
63}

66impl Default for ProvinceRenderOptions {
67    fn default() -> Self {
68        Self {

70            y: 0.0,
71            zoom: 1.0,
72            pixel_size: 1.0,
73            screen_w: 1280.0,
74            screen_h: 720.0,
75            map_mode: "political".to_string(),
76            draw_fills: true,
77            draw_borders: true,
78            draw_labels: false,
79            draw_capitals: true,
80            border_width: 1.0,
81            zoom_mode: None,
82            tactical_zoom_threshold: 3.0,
83            draw_roads: true,
84            hovered_id: None,
85            selected_id: None,
86        }
87    }

90/// Return the RGBA line colour for a border segment based on its registered type config.
91fn border_color_from_registry(registry: &ProvinceRegistry, a: ProvinceId, b: ProvinceId) -> [f32; 4] {
92    let border_type = registry.get_border_type(a, b).unwrap_or(0);
93    registry
94        .get_border_type_config(border_type)
95        .map(|c| c.color)
96        .unwrap_or([0.5, 0.5, 0.5, 1.0])
97}

99/// Compute the visible province-space bounds (left, top, right, bottom) from the render options.
100fn viewport_bounds(opts: &ProvinceRenderOptions) -> (f32, f32, f32, f32) {
101    let zoom_ps = (opts.zoom * opts.pixel_size).max(0.0001);
102    let left = -opts.x / zoom_ps;
103    let top = -opts.y / zoom_ps;
104    let right = (opts.screen_w - opts.x) / zoom_ps;
105    let bottom = (opts.screen_h - opts.y) / zoom_ps;
106    (left, top, right, bottom)
107}

109const VISIBILITY_HIDDEN: u8 = 0;
110const VISIBILITY_DISCOVERED: u8 = 1;
111const VISIBILITY_VISIBLE_MIN: u8 = 2;
112
113fn discovered_fill_color() -> [f32; 4] {
114    [0.2, 0.2, 0.2, 1.0]
115}
116
117fn is_hidden(visibility_state: u8) -> bool {
118    visibility_state == VISIBILITY_HIDDEN
119}
120
121fn is_discovered(visibility_state: u8) -> bool {
122    visibility_state == VISIBILITY_DISCOVERED
123}
124
125fn is_fully_visible(visibility_state: u8) -> bool {
126    visibility_state >= VISIBILITY_VISIBLE_MIN
127}
128
129fn resolve_zoom_mode(opts: &ProvinceRenderOptions) -> ProvinceZoomMode {
130    if let Some(mode) = opts.zoom_mode {
131        if mode == ProvinceZoomMode::Auto {
132            if opts.zoom >= opts.tactical_zoom_threshold {
133                return ProvinceZoomMode::Tactical;
134            }
135            return ProvinceZoomMode::Strategic;
136        }
137        return mode;
138    }
139    // Preserve historic behavior: if no explicit mode is supplied, render full detail.
140    ProvinceZoomMode::Tactical
141}

143fn should_render_border_in_mode(
144    mode: ProvinceZoomMode,
145    is_country: bool,
146) -> bool {
147    match mode {
148        ProvinceZoomMode::Tactical => true,
149        ProvinceZoomMode::Strategic => is_country,
150        ProvinceZoomMode::Auto => is_country,
151    }
152}
153/// Generate a RenderCommand Vec for the province map: fills, borders, capitals, and labels with viewport culling.
154pub fn generate_render_commands(
155    registry: &ProvinceRegistry,
156    opts: &ProvinceRenderOptions,
157    font_key: Option<FontKey>,
158) -> Vec<RenderCommand> {
159    let mut cmds: Vec<RenderCommand> = Vec::new();
160    let (left, top, right, bottom) = viewport_bounds(opts);
161    let zoom_mode = resolve_zoom_mode(opts);
162    cmds.push(RenderCommand::PushTransform);
163    cmds.push(RenderCommand::Translate {
164        x: opts.x,
165        y: opts.y,
166    });
167    cmds.push(RenderCommand::Scale {
168        sx: opts.zoom,
169        sy: opts.zoom,
170    });
171    if opts.draw_fills {
172        for id in registry.province_ids() {
173            let Some(bb) = registry.bbox_for(id) else {
174                continue;
175            };
176            if (bb.2 as f32) < left {
177                continue;
178            }
179            if (bb.0 as f32) > right {
180                continue;
181            }
182            if (bb.3 as f32) < top {
183                continue;
184            }
185            if (bb.1 as f32) > bottom {
186                continue;
187            }
188            let Some(style) = registry.style_for(id) else {
189                continue;
190            };
191            if is_hidden(style.visibility_state) {
192                continue;
193            }
194            let c = if is_discovered(style.visibility_state) {
195                discovered_fill_color()
196            } else {
197                let mode_config = registry.map_mode_config();
198                resolve_color_fallback(mode_config, style)
199            };
200            cmds.push(RenderCommand::SetColor(c[0], c[1], c[2], c[3]));
201            if let Some(spans) = registry.spans_for(id) {
202                // Merge vertically-adjacent spans with same x0,x1 into taller rectangles
203                let mut i = 0;
204                while i < spans.len() {
205                    let (y, x0, x1) = spans[i];
206                    if (y as f32) > bottom || (x1 as f32) <= left || (x0 as f32) >= right {
207                        i += 1;
208                        continue;
209                    }
210                    if (y as f32) < top {
211                        i += 1;
212                        continue;
213                    }
214                    // Try to merge consecutive spans with same x0, x1
215                    let mut height: u32 = 1;
216                    while i + (height as usize) < spans.len() {
217                        let (ny, nx0, nx1) = spans[i + height as usize];
218                        if nx0 == x0 && nx1 == x1 && ny == y + height {
219                            height += 1;
220                            if (ny as f32) > bottom {
221                                break;
222                            }
223                        } else {
224                            break;
225                        }
226                    }
227                    cmds.push(RenderCommand::Rectangle {
228                        mode: DrawMode::Fill,
229                        x: x0 as f32 * opts.pixel_size,
230                        y: y as f32 * opts.pixel_size,
231                        w: (x1 - x0) as f32 * opts.pixel_size,
232                        h: height as f32 * opts.pixel_size,
233                    });
234                    i += height as usize;
235                }
236            }
237        }
238    }
239    if opts.draw_borders {
240        let effective_scale = opts.zoom * opts.pixel_size;
241        // LOD: skip all borders when they'd be sub-pixel
242        if effective_scale >= 0.5 {
243            let mut active_width: Option<f32> = None;
244            let mut active_color: Option<[f32; 4]> = None;
245            for &(a, b, x0, y0, x1, y1) in registry.border_segments() {
246                let min_x = x0.min(x1) as f32;
247                let max_x = x0.max(x1) as f32;
248                let min_y = y0.min(y1) as f32;
249                let max_y = y0.max(y1) as f32;
250                if max_x < left || min_x > right || max_y < top || min_y > bottom {
251                    continue;
252                }
253                let Some(sa) = registry.style_for(ProvinceId(a)) else {
254                    continue;
255                };
256                let Some(sb) = registry.style_for(ProvinceId(b)) else {
257                    continue;
258                };
259                if !is_fully_visible(sa.visibility_state) || !is_fully_visible(sb.visibility_state) {
260                    continue;
261                }
262                let pair_style_override = registry.get_border_pair_style(ProvinceId(a), ProvinceId(b));
263                let pair_style = pair_style_override.unwrap_or_default();
264                let is_country = pair_style.flags.contains_bits(BorderPairFlags::COUNTRY);
265                if !should_render_border_in_mode(zoom_mode, is_country) {
266                    continue;
267                }
268
269                let width = if pair_style_override.is_some() {
270                    pair_style.thickness.max(1.0)
271                } else {
272                    opts.border_width.max(1.0)
273                };
274                if active_width != Some(width) {
275                    cmds.push(RenderCommand::SetLineWidth(width));
276                    active_width = Some(width);
277                }
278
279                let color = pair_style.color.unwrap_or_else(|| border_color_from_registry(registry, ProvinceId(a), ProvinceId(b)));
280                if active_color != Some(color) {
281                    cmds.push(RenderCommand::SetColor(color[0], color[1], color[2], color[3]));
282                    active_color = Some(color);
283                }
284
285                cmds.push(RenderCommand::Line {
286                    x1: x0 as f32 * opts.pixel_size,
287                    y1: y0 as f32 * opts.pixel_size,
288                    x2: x1 as f32 * opts.pixel_size,
289                    y2: y1 as f32 * opts.pixel_size,
290                });
291            }
292        }
293    }
294    if opts.draw_roads && zoom_mode == ProvinceZoomMode::Tactical {
295        cmds.push(RenderCommand::SetLineWidth((opts.border_width * 1.25).max(1.0)));
296        cmds.push(RenderCommand::SetColor(
297            140.0 / 255.0,
298            100.0 / 255.0,
299            62.0 / 255.0,
300            0.85,
301        ));
302        for (a, b) in registry.adjacency_pairs() {
303            let Some(sa) = registry.style_for(a) else {
304                continue;
305            };
306            let Some(sb) = registry.style_for(b) else {
307                continue;
308            };
309            if !is_fully_visible(sa.visibility_state) || !is_fully_visible(sb.visibility_state) {
310                continue;
311            }
312            let Some((ax, ay)) = registry.capital_for(a) else {
313                continue;
314            };
315            let Some((bx, by)) = registry.capital_for(b) else {
316                continue;
317            };
318            let min_x = ax.min(bx);
319            let max_x = ax.max(bx);
320            let min_y = ay.min(by);
321            let max_y = ay.max(by);
322            if max_x < left || min_x > right || max_y < top || min_y > bottom {
323                continue;
324            }
325            cmds.push(RenderCommand::Line {
326                x1: ax * opts.pixel_size,
327                y1: ay * opts.pixel_size,
328                x2: bx * opts.pixel_size,
329                y2: by * opts.pixel_size,
330            });
331        }
332    }
333    if opts.draw_capitals {
334        for id in registry.province_ids() {
335            let Some(bb) = registry.bbox_for(id) else {
336                continue;
337            };
338            if (bb.2 as f32) < left {
339                continue;
340            }
341            if (bb.0 as f32) > right {
342                continue;
343            }
344            if (bb.3 as f32) < top {
345                continue;
346            }
347            if (bb.1 as f32) > bottom {
348                continue;
349            }
350            let Some(style) = registry.style_for(id) else {
351                continue;
352            };
353            if !is_fully_visible(style.visibility_state) {
354                continue;
355            }
356            let Some((cx, cy)) = registry.capital_for(id) else {
357                continue;
358            };
359            cmds.push(RenderCommand::SetColor(
360                1.0,
361                220.0 / 255.0,
362                70.0 / 255.0,
363                1.0,
364            ));
365            cmds.push(RenderCommand::Circle {
366                mode: DrawMode::Fill,
367                x: cx * opts.pixel_size,
368                y: cy * opts.pixel_size,
369                r: (opts.pixel_size * 0.42).max(2.0),
370            });
371            cmds.push(RenderCommand::SetColor(
372                20.0 / 255.0,
373                20.0 / 255.0,
374                20.0 / 255.0,
375                1.0,
376            ));
377            cmds.push(RenderCommand::Circle {
378                mode: DrawMode::Fill,
379                x: cx * opts.pixel_size,
380                y: cy * opts.pixel_size,
381                r: (opts.pixel_size * 0.24).max(1.0),
382            });
383        }
384    }
385    if opts.draw_labels {
386        if let Some(font) = font_key {
387            for id in registry.province_ids() {
388                let Some(bb) = registry.bbox_for(id) else {
389                    continue;
390                };
391                if (bb.2 as f32) < left {
392                    continue;
393                }
394                if (bb.0 as f32) > right {
395                    continue;
396                }
397                if (bb.3 as f32) < top {
398                    continue;
399                }
400                if (bb.1 as f32) > bottom {
401                    continue;
402                }
403                let Some(style) = registry.style_for(id) else {
404                    continue;
405                };
406                if !is_fully_visible(style.visibility_state) {
407                    continue;
408                }
409                let text = registry
410                    .label_text_for(id)
411                    .map(|s| s.to_string())
412                    .unwrap_or_else(|| id.to_string());
413                let ((ax, ay), (bx, by)) = registry
414                    .label_line_for(id)
415                    .unwrap_or(((bb.0 as f32, bb.1 as f32), (bb.2 as f32, bb.3 as f32)));
416                let mx = (ax + bx) * 0.5;
417                let my = (ay + by) * 0.5;
418                cmds.push(RenderCommand::SetColor(0.0, 0.0, 0.0, 0.65));
419                cmds.push(RenderCommand::Print {
420                    font_key: font,
421                    text: text.clone(),
422                    x: mx * opts.pixel_size + 1.0,
423                    y: my * opts.pixel_size + 1.0,
424                    scale: 0.8,
425                });
426                cmds.push(RenderCommand::SetColor(0.92, 0.92, 0.86, 1.0));
427                cmds.push(RenderCommand::Print {
428                    font_key: font,
429                    text,
430                    x: mx * opts.pixel_size,
431                    y: my * opts.pixel_size,
432                    scale: 0.8,
433                });
434            }
435        }
436    }
437    if let Some(id) = opts.hovered_id {
438        let can_draw_hover = registry
439            .style_for(id)
440            .map(|style| is_fully_visible(style.visibility_state))
441            .unwrap_or(false);
442        if can_draw_hover {
443        if let Some((min_x, min_y, max_x, max_y)) = registry.bbox_for(id) {
444            cmds.push(RenderCommand::SetColor(1.0, 1.0, 1.0, 0.35));
445            cmds.push(RenderCommand::SetLineWidth(2.0));
446            cmds.push(RenderCommand::Rectangle {
447                mode: DrawMode::Line,
448                x: min_x as f32 * opts.pixel_size,
449                y: min_y as f32 * opts.pixel_size,
450                w: (max_x.saturating_sub(min_x) + 1) as f32 * opts.pixel_size,
451                h: (max_y.saturating_sub(min_y) + 1) as f32 * opts.pixel_size,
452            });
453        }
454        }
455    }
456    if let Some(id) = opts.selected_id {
457        let can_draw_selected = registry
458            .style_for(id)
459            .map(|style| is_fully_visible(style.visibility_state))
460            .unwrap_or(false);
461        if can_draw_selected {
462        if let Some((min_x, min_y, max_x, max_y)) = registry.bbox_for(id) {
463            cmds.push(RenderCommand::SetColor(1.0, 0.9, 0.1, 0.9));
464            cmds.push(RenderCommand::SetLineWidth(3.0));
465            cmds.push(RenderCommand::Rectangle {
466                mode: DrawMode::Line,
467                x: min_x as f32 * opts.pixel_size,
468                y: min_y as f32 * opts.pixel_size,
469                w: (max_x.saturating_sub(min_x) + 1) as f32 * opts.pixel_size,
470                h: (max_y.saturating_sub(min_y) + 1) as f32 * opts.pixel_size,
471            });
472        }
473        }
474    }
475    cmds.push(RenderCommand::PopTransform);
476    cmds
