2//!
3//! - ProvinceId newtype, BorderType (u8) for game-defined adjacency classification, and ProvinceStyle for per-province visuals.
4//! - ProvinceSnapshot provides an immutable point-in-time view of province state.

6use std::collections::HashMap;

8/// Unique identifier for a province in the map system; 0 is reserved for "no province" / ocean pixels.
9#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, PartialOrd, Ord)]
10pub struct ProvinceId(pub u32);
11
12impl ProvinceId {
13    /// Creates a new ProvinceId from a raw u32.
14    pub fn new(id: u32) -> Self { Self(id) }
15    /// Returns the raw u32 underlying value.
16    pub fn raw(self) -> u32 { self.0 }
17}
18
19impl std::fmt::Display for ProvinceId {
20    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
21        write!(f, "{}", self.0)
22    }
23}
24
25impl From<u32> for ProvinceId {
26    fn from(v: u32) -> Self { Self(v) }
27}
28
29impl From<ProvinceId> for u32 {
30    fn from(id: ProvinceId) -> Self { id.0 }
31}

49/// Bit-flag set controlling semantic styling of a border pair override.
50#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Default)]
51pub struct BorderPairFlags {
52    bits: u8,
53}
54
55impl BorderPairFlags {
56    /// Country border visual marker.
57    pub const COUNTRY: u8 = 0x01;
58    /// Alliance border visual marker.
59    pub const ALLIANCE: u8 = 0x02;
60    /// War state border visual flag marker.
61    pub const WAR: u8 = 0x04;
62    /// Truce state border visual flag marker.
63    pub const TRUCE: u8 = 0x08;
64
65    /// Create a new empty border flag set.
66    pub fn empty() -> Self {
67        Self { bits: 0 }
68    }
69
70    /// Return the raw flag bit pattern.
71    pub fn bits(self) -> u8 {
72        self.bits
73    }
74
75    /// Build a flag set from raw bits.
76    pub fn from_bits(bits: u8) -> Self {
77        Self { bits }
78    }
79
80    /// Insert a raw flag bitmask into this set.
81    pub fn insert_bits(&mut self, mask: u8) {
82        self.bits |= mask;
83    }
84
85    /// Return true if all bits from mask are present.
86    pub fn contains_bits(self, mask: u8) -> bool {
87        (self.bits & mask) == mask
88    }
89
90    /// Parse a single canonical flag token.
91    pub fn parse_token(token: &str) -> Option<u8> {
92        match token {
93            "country" => Some(Self::COUNTRY),
94            "alliance" => Some(Self::ALLIANCE),
95            "war" => Some(Self::WAR),
96            "truce" => Some(Self::TRUCE),
97            _ => None,
98        }
99    }
100
101    /// Return canonical tokens for all set bits.
102    pub fn to_tokens(self) -> Vec<&'static str> {
103        let mut out = Vec::new();
104        if self.contains_bits(Self::COUNTRY) {
105            out.push("country");
106        }
107        if self.contains_bits(Self::ALLIANCE) {
108            out.push("alliance");
109        }
110        if self.contains_bits(Self::WAR) {
111            out.push("war");
112        }
113        if self.contains_bits(Self::TRUCE) {
114            out.push("truce");
115        }
116        out
117    }
118}

120/// Per-adjacency border style override keyed by ordered province pair.
121#[derive(Debug, Clone, Copy, PartialEq)]
122pub struct BorderPairStyle {
123    /// Optional RGBA color override. When None, renderer uses the class default color.
124    pub color: Option<[f32; 4]>,
125    /// Line width in map pixels for this border pair.
126    pub thickness: f32,
127    /// Semantic flags (country/alliance/war/truce) used for mode filtering.
128    pub flags: BorderPairFlags,
129}
130
131impl Default for BorderPairStyle {
132    fn default() -> Self {
133        Self {
134            color: None,
135            thickness: 1.0,
136            flags: BorderPairFlags::empty(),
137        }
138    }
139}

141/// Visual and gameplay state attached to a single province, stored in ProvinceRegistry.
142#[derive(Debug, Clone, PartialEq)]
143pub struct ProvinceStyle {
144    /// RGBA fill colour used in political map mode; default grey [0.5, 0.5, 0.5, 1.0].
145    pub political_color: [f32; 4],
146    /// Terrain index: 0 = water/sea, non-zero = land class.
147    pub terrain_type: u32,
148    /// Border style index forwarded to the renderer for line variant selection.
149    pub border_style: u32,
150    /// Fog-of-war state byte; 0 = fully fogged.
151    pub fog_state: u8,
152    /// Visibility state byte; 0 = hidden, 1 = discovered, 2+ = fully visible.
153    pub visibility_state: u8,
154}
155
156/// Default ProvinceStyle: grey political color, water terrain, no fog, visible.
157impl Default for ProvinceStyle {
158    fn default() -> Self {
159        Self {
160            political_color: [0.5, 0.5, 0.5, 1.0],
161            terrain_type: 0,
162            border_style: 0,
163            fog_state: 0,
164            visibility_state: 2,
165        }
166    }
167}

169/// Immutable point-in-time view of a province returned by ProvinceRegistry::get_province.
170#[derive(Debug, Clone, PartialEq)]
171pub struct ProvinceSnapshot {
172    /// Identifier of the province this snapshot describes.
173    pub province_id: ProvinceId,
174    /// Visual and gameplay style at snapshot time.
175    pub style: ProvinceStyle,
176    /// Registry revision counter at the time of snapshot creation.
177    pub revision: u64,
178    /// Weighted pixel centroid; None if province has no spans.
179    pub centroid: Option<(f32, f32)>,
180    /// Arbitrary key-value metadata set via set_attr.
181    pub attrs: HashMap<String, String>,
182}
