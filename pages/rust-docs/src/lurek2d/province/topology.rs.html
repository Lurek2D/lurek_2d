1//! Undirected adjacency graph storing sorted neighbour lists per province.
2//!
3//! - Rebuild from raw id pairs with dedup and self-loop filtering.
4//! - Binary-search-based neighbour lookup and adjacency queries.
5//! - Extraction of all province ids and unique adjacency pairs.
6
7use crate::province::types::ProvinceId;
8use std::collections::{HashMap, HashSet};
9
10/// Undirected adjacency graph over provinces; neighbours stored sorted for binary search.
11#[derive(Debug, Clone, Default)]
12pub struct ProvinceGraph {
13    /// Sorted neighbour lists keyed by province id.
14    neighbors: HashMap<ProvinceId, Vec<ProvinceId>>,
15}
16
17impl ProvinceGraph {
18    /// Return a new empty graph. This function is part of the public API.
19    pub fn new() -> Self {
20        Self::default()
21    }
22
23    /// Rebuild the graph from a slice of adjacent id pairs; clears previous data.
24    pub fn rebuild_from_pairs(&mut self, pairs: &[(ProvinceId, ProvinceId)]) {
25        self.neighbors.clear();
26        for &(a, b) in pairs {
27            if a == ProvinceId(0) || b == ProvinceId(0) || a == b {
28                continue;
29            }
30            self.neighbors.entry(a).or_default().push(b);
31            self.neighbors.entry(b).or_default().push(a);
32        }
33        for list in self.neighbors.values_mut() {
34            list.sort_unstable();
35            list.dedup();
36        }
37    }
38
39    /// Return the sorted neighbour slice for id; returns an empty slice if id has no entry.
40    pub fn neighbors_of(&self, id: ProvinceId) -> &[ProvinceId] {
41        static EMPTY: [ProvinceId; 0] = [];
42        self.neighbors.get(&id).map_or(&EMPTY, Vec::as_slice)
43    }
44
45    /// Return true if a and b share a border in the graph.
46    pub fn is_adjacent(&self, a: ProvinceId, b: ProvinceId) -> bool {
47        self.neighbors_of(a).binary_search(&b).is_ok()
48    }
49
50    /// Return all province ids present in the graph, sorted ascending.
51    pub fn province_ids(&self) -> Vec<ProvinceId> {
52        let mut ids: Vec<ProvinceId> = self.neighbors.keys().copied().collect();
53        ids.sort_unstable();
54        ids
55    }
56
57    /// Return all unique adjacency pairs (a < b) sorted ascending.
58    pub fn adjacency_pairs(&self) -> Vec<(ProvinceId, ProvinceId)> {
59        let mut out = Vec::new();
60        let mut seen = HashSet::new();
61        for (&a, list) in &self.neighbors {
62            for &b in list {
63                let (x, y) = if a < b { (a, b) } else { (b, a) };
64                if seen.insert((x, y)) {
65                    out.push((x, y));
66                }
67            }
68        }
69        out.sort_unstable();
70        out
71    }
72}
