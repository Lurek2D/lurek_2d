1//! Nine-slice (9-patch) descriptor that splits a texture into corners, edges, and a center.
2//!
3//! - Generates source/destination patch tuples for scalable UI borders and panels.
4//! - Preserves corner pixel ratios while stretching edges and center to fit target dimensions.
5
6use crate::runtime::resource_keys::TextureKey;
7
8/// Source and destination rect tuple: (src_x, src_y, src_w, src_h, dst_x, dst_y, dst_w, dst_h).
9pub type Patch = (f32, f32, f32, f32, f32, f32, f32, f32);
10/// Nine-slice border descriptor: stores texture key and the four border inset sizes.
11#[derive(Debug, Clone)]
12pub struct NineSlice {
13    /// Texture containing the nine-slice source art.
14    pub texture_key: TextureKey,
15    /// Height of the top border strip in source pixels.
16    pub top: f32,
17    /// Width of the right border strip in source pixels.
18    pub right: f32,
19    /// Height of the bottom border strip in source pixels.
20    pub bottom: f32,
21    /// Width of the left border strip in source pixels.
22    pub left: f32,
23    /// Total source texture width in pixels.
24    pub tex_width: f32,
25    /// Total source texture height in pixels.
26    pub tex_height: f32,
27}
28/// Constructor and patch generation for NineSlice.
29impl NineSlice {
30    /// Create a NineSlice with explicit border insets and full texture dimensions.
31    pub fn new(
32        texture_key: TextureKey,
33        top: f32,
34        right: f32,
35        bottom: f32,
36        left: f32,
37        tex_width: f32,
38        tex_height: f32,
39    ) -> Self {
40        NineSlice {
41            texture_key,
42            top,
43            right,
44            bottom,
45            left,
46            tex_width,
47            tex_height,
48        }
49    }
50    /// Return the 9 Patch tuples for drawing a nine-slice box at (x, y) with target dimensions (w, h).
51    /// Order: top-left, top-centre, top-right, mid-left, mid-centre, mid-right, bot-left, bot-centre, bot-right.
52    pub fn patches(&self, x: f32, y: f32, w: f32, h: f32) -> [Patch; 9] {
53        let l = self.left;
54        let r = self.right;
55        let t = self.top;
56        let b = self.bottom;
57        let tw = self.tex_width;
58        let th = self.tex_height;
59        let src_cx = l;
60        let src_cw = (tw - l - r).max(0.0);
61        let src_rx = tw - r;
62        let src_cy = t;
63        let src_ch = (th - t - b).max(0.0);
64        let src_by = th - b;
65        let dst_cw = (w - l - r).max(0.0);
66        let dst_ch = (h - t - b).max(0.0);
67        [
68            (0.0, 0.0, l, t, x, y, l, t),
69            (src_cx, 0.0, src_cw, t, x + l, y, dst_cw, t),
70            (src_rx, 0.0, r, t, x + l + dst_cw, y, r, t),
71            (0.0, src_cy, l, src_ch, x, y + t, l, dst_ch),
72            (src_cx, src_cy, src_cw, src_ch, x + l, y + t, dst_cw, dst_ch),
73            (src_rx, src_cy, r, src_ch, x + l + dst_cw, y + t, r, dst_ch),
74            (0.0, src_by, l, b, x, y + t + dst_ch, l, b),
75            (src_cx, src_by, src_cw, b, x + l, y + t + dst_ch, dst_cw, b),
76            (src_rx, src_by, r, b, x + l + dst_cw, y + t + dst_ch, r, b),
77        ]
78    }
79}
