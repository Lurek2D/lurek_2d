1//! Xorshift64 pseudo-random number generator for deterministic dataframe sampling
2//!
3//! - Float, integer, and index generation from 64-bit state
4//! - Zero-seed remap to avoid degenerate all-zero output
5
6/// Hold xorshift64 state used by dataframe-local random helpers.
7pub(crate) struct Xorshift64 {
8    /// Store current PRNG state word.
9    state: u64,
10}
11impl Xorshift64 {
12    /// Create generator from seed and remap zero seed to one.
13    pub(crate) fn new(seed: u64) -> Self {
14        Self {
15            state: if seed == 0 { 1 } else { seed },
16        }
17    }
18    /// Advance generator and return next 64-bit pseudo-random value.
19    pub(crate) fn next_u64(&mut self) -> u64 {
20        let mut x = self.state;
21        x ^= x << 13;
22        x ^= x >> 7;
23        x ^= x << 17;
24        self.state = x;
25        x
26    }
27    /// Return pseudo-random float in the half-open range [0, 1).
28    pub(crate) fn next_f64(&mut self) -> f64 {
29        (self.next_u64() & 0x001F_FFFF_FFFF_FFFF) as f64 / (1u64 << 53) as f64
30    }
31    /// Return pseudo-random index in the half-open range [0, max).
32    pub(crate) fn next_usize(&mut self, max: usize) -> usize {
33        (self.next_u64() % max as u64) as usize
34    }
35}
