16/// Hold typed columnar storage with optional validity mask.
17pub enum ColumnStore {
18    /// Store 64-bit floating point values.
19    Float64 {
20        /// Store row values.
21        data: Vec<f64>,
22        /// Store optional per-row validity flags.
23        validity: Option<Vec<bool>>,
24    },
25    /// Store 64-bit integer values.
26    Int64 {
27        /// Store row values.
28        data: Vec<i64>,
29        /// Store optional per-row validity flags.
30        validity: Option<Vec<bool>>,
31    },
32    /// Store boolean values.
33    Bool {
34        /// Store row values.
35        data: Vec<bool>,
36        /// Store optional per-row validity flags.
37        validity: Option<Vec<bool>>,
38    },
39    /// Store UTF-8 text values.
40    Text {
41        /// Store row values.
42        data: Vec<String>,
43        /// Store optional per-row validity flags.
44        validity: Option<Vec<bool>>,
45    },
46}
47impl ColumnStore {
48    /// Return static type name for this column variant.
49    pub fn dtype_name(&self) -> &'static str {
50        match self {
51            ColumnStore::Float64 { .. } => "float64",
52            ColumnStore::Int64 { .. } => "int64",
53            ColumnStore::Bool { .. } => "bool",
54            ColumnStore::Text { .. } => "text",
55        }
56    }
57    /// Return number of rows in this column.
58    pub fn len(&self) -> usize {
59        match self {
60            ColumnStore::Float64 { data, .. } => data.len(),
61            ColumnStore::Int64 { data, .. } => data.len(),
62            ColumnStore::Bool { data, .. } => data.len(),
63            ColumnStore::Text { data, .. } => data.len(),
64        }
65    }
66    /// Return true when this column has no rows.
67    pub fn is_empty(&self) -> bool {
68        self.len() == 0
69    }
70    /// Return true when row at index is valid according to validity mask.
71    pub fn is_valid(&self, i: usize) -> bool {
72        let validity = match self {
73            ColumnStore::Float64 { validity, .. } => validity,
74            ColumnStore::Int64 { validity, .. } => validity,
75            ColumnStore::Bool { validity, .. } => validity,
76            ColumnStore::Text { validity, .. } => validity,
77        };
78        validity
79            .as_ref()
80            .is_none_or(|v| *v.get(i).unwrap_or(&false))
81    }
82    /// Return valid f64 values for Float64 columns, skipping nil rows.
83    pub fn valid_f64s(&self) -> Option<Vec<f64>> {
84        if let ColumnStore::Float64 { data, validity } = self {
85            let vals: Vec<f64> = if let Some(v) = validity {
86                data.iter()
87                    .zip(v.iter())
88                    .filter_map(|(x, ok)| if *ok { Some(*x) } else { None })
89                    .collect()
90            } else {
91                data.clone()
92            };
93            Some(vals)
94        } else {
95            None
96        }
97    }
98    /// Filter rows by boolean mask and return new column.
99    pub fn filter(&self, mask: &[bool]) -> ColumnStore {
100        match self {
101            ColumnStore::Float64 { data, validity } => {
102                let new_data: Vec<f64> = data
103                    .iter()
104                    .zip(mask.iter())
105                    .filter_map(|(v, ok)| if *ok { Some(*v) } else { None })
106                    .collect();
107                let new_validity = validity.as_ref().map(|val| {
108                    val.iter()
109                        .zip(mask.iter())
110                        .filter_map(|(v, ok)| if *ok { Some(*v) } else { None })
111                        .collect()
112                });
113                ColumnStore::Float64 {
114                    data: new_data,
115                    validity: new_validity,
116                }
117            }
118            ColumnStore::Int64 { data, validity } => {
119                let new_data: Vec<i64> = data
120                    .iter()
121                    .zip(mask.iter())
122                    .filter_map(|(v, ok)| if *ok { Some(*v) } else { None })
123                    .collect();
124                let new_validity = validity.as_ref().map(|val| {
125                    val.iter()
126                        .zip(mask.iter())
127                        .filter_map(|(v, ok)| if *ok { Some(*v) } else { None })
128                        .collect()
129                });
130                ColumnStore::Int64 {
131                    data: new_data,
132                    validity: new_validity,
133                }
134            }
135            ColumnStore::Bool { data, validity } => {
136                let new_data: Vec<bool> = data
137                    .iter()
138                    .zip(mask.iter())
139                    .filter_map(|(v, ok)| if *ok { Some(*v) } else { None })
140                    .collect();
141                let new_validity = validity.as_ref().map(|val| {
142                    val.iter()
143                        .zip(mask.iter())
144                        .filter_map(|(v, ok)| if *ok { Some(*v) } else { None })
145                        .collect()
146                });
147                ColumnStore::Bool {
148                    data: new_data,
149                    validity: new_validity,
150                }
151            }
152            ColumnStore::Text { data, validity } => {
153                let new_data: Vec<String> = data
154                    .iter()
155                    .zip(mask.iter())
156                    .filter_map(|(v, ok)| if *ok { Some(v.clone()) } else { None })
157                    .collect();
158                let new_validity = validity.as_ref().map(|val| {
159                    val.iter()
160                        .zip(mask.iter())
161                        .filter_map(|(v, ok)| if *ok { Some(*v) } else { None })
162                        .collect()
163                });
164                ColumnStore::Text {
165                    data: new_data,
166                    validity: new_validity,
167                }
168            }
169        }
170    }
171}

337    /// Convert DataFrame to VecFrame by inferring column types.
338    pub fn from_dataframe(df: &DataFrame) -> VecFrame {
339        let mut vf = VecFrame::new();
340        for (col_idx, name) in df.columns().iter().enumerate() {
341            let raw = df.raw_data();
342            let cells = &raw[col_idx];
343            let store = infer_column(cells);
344            vf.column_names.push(name.clone());
345            vf.columns.push(store);
346        }
347        vf
348    }
349    /// Convert VecFrame back to DataFrame.
350    pub fn to_dataframe(&self) -> DataFrame {
351        let nrows = self.nrows();
352        let mut col_data: Vec<Vec<CellValue>> = Vec::with_capacity(self.columns.len());
353        for (i, store) in self.columns.iter().enumerate() {
354            let _ = i;
355            let cells: Vec<CellValue> = (0..nrows)
356                .map(|row| {
357                    if !store.is_valid(row) {
358                        return CellValue::Nil;
359                    }
360                    match store {
361                        ColumnStore::Float64 { data, .. } => CellValue::Number(data[row]),
362                        ColumnStore::Int64 { data, .. } => CellValue::Number(data[row] as f64),
363                        ColumnStore::Bool { data, .. } => CellValue::Bool(data[row]),
364                        ColumnStore::Text { data, .. } => CellValue::Text(data[row].clone()),
365                    }
366                })
367                .collect();
368            col_data.push(cells);
369        }
370        DataFrame::from_raw(self.column_names.clone(), col_data)
371    }
372    /// Resolve column name to zero-based index.
373    fn resolve(&self, col: &str) -> Result<usize, String> {
374        self.column_names
375            .iter()
376            .position(|n| n == col)
377            .ok_or_else(|| format!("column not found: {col}"))
378    }
379    /// Apply scalar operation to Float64 column in place.
380    pub fn col_scalar_op(&mut self, col: &str, op: ScalarOp, val: f64) -> Result<(), String> {
381        let idx = self.resolve(col)?;
382        if let ColumnStore::Float64 { data, validity } = &mut self.columns[idx] {
383            let vref = validity.as_ref();
384            match op {
385                ScalarOp::Add => {
386                    for (i, x) in data.iter_mut().enumerate() {
387                        if vref.is_none_or(|v| v[i]) {
388                            *x += val;
389                        }
390                    }
391                }
392                ScalarOp::Sub => {
393                    for (i, x) in data.iter_mut().enumerate() {
394                        if vref.is_none_or(|v| v[i]) {
395                            *x -= val;
396                        }
397                    }
398                }
399                ScalarOp::Mul => {
400                    for (i, x) in data.iter_mut().enumerate() {
401                        if vref.is_none_or(|v| v[i]) {
402                            *x *= val;
403                        }
404                    }
405                }
406                ScalarOp::Div => {
407                    if val == 0.0 {
408                        return Err("division by zero in col_scalar_op".into());
409                    }
410                    for (i, x) in data.iter_mut().enumerate() {
411                        if vref.is_none_or(|v| v[i]) {
412                            *x /= val;
413                        }
414                    }
415                }
416                ScalarOp::Abs => {
417                    for (i, x) in data.iter_mut().enumerate() {
418                        if vref.is_none_or(|v| v[i]) {
419                            *x = x.abs();
420                        }
421                    }
422                }
423                ScalarOp::Sqrt => {
424                    for (i, x) in data.iter_mut().enumerate() {
425                        if vref.is_none_or(|v| v[i]) {
426                            *x = x.sqrt();
427                        }
428                    }
429                }
430                ScalarOp::Floor => {
431                    for (i, x) in data.iter_mut().enumerate() {
432                        if vref.is_none_or(|v| v[i]) {
433                            *x = x.floor();
434                        }
435                    }
436                }
437                ScalarOp::Ceil => {
438                    for (i, x) in data.iter_mut().enumerate() {
439                        if vref.is_none_or(|v| v[i]) {
440                            *x = x.ceil();
441                        }
442                    }
443                }
444                ScalarOp::Neg => {
445                    for (i, x) in data.iter_mut().enumerate() {
446                        if vref.is_none_or(|v| v[i]) {
447                            *x = -(*x);
448                        }
449                    }
450                }
451            }
452            Ok(())
453        } else {
454            Err(format!(
455                "col_scalar_op: column '{col}' is not Float64 (type: {})",
456                self.columns[idx].dtype_name()
457            ))
458        }
459    }

478    /// Compute element-wise binary operation between two numeric columns and write result column.
479    pub fn col_binary_op(
480        &mut self,
481        out_col: &str,
482        left_col: &str,
483        op: BinaryOp,
484        right_col: &str,
485    ) -> Result<(), String> {
486        let li = self.resolve(left_col)?;
487        let ri = self.resolve(right_col)?;
488        let nrows = self.nrows();
489        let left_data: Vec<f64> = match &self.columns[li] {
490            ColumnStore::Float64 { data, .. } => data.clone(),
491            ColumnStore::Int64 { data, .. } => data.iter().map(|v| *v as f64).collect(),
492            t => {
493                return Err(format!(
494                    "col_binary_op: left column '{left_col}' type '{}' is not numeric",
495                    t.dtype_name()
496                ))
497            }
498        };
499        let right_data: Vec<f64> = match &self.columns[ri] {
500            ColumnStore::Float64 { data, .. } => data.clone(),
501            ColumnStore::Int64 { data, .. } => data.iter().map(|v| *v as f64).collect(),
502            t => {
503                return Err(format!(
504                    "col_binary_op: right column '{right_col}' type '{}' is not numeric",
505                    t.dtype_name()
506                ))
507            }
508        };
509        let left_valid = match &self.columns[li] {
510            ColumnStore::Float64 { validity, .. } | ColumnStore::Int64 { validity, .. } => {
511                validity.clone()
512            }
513            _ => None,
514        };
515        let right_valid = match &self.columns[ri] {
516            ColumnStore::Float64 { validity, .. } | ColumnStore::Int64 { validity, .. } => {
517                validity.clone()
518            }
519            _ => None,
520        };
521        let mut out_data = Vec::with_capacity(nrows);
522        let mut out_validity: Option<Vec<bool>> = None;
523        for i in 0..nrows {
524            let l_ok = left_valid.as_ref().is_none_or(|v| v[i]);
525            let r_ok = right_valid.as_ref().is_none_or(|v| v[i]);
526            if !l_ok || !r_ok {
527                out_data.push(0.0f64);
528                out_validity.get_or_insert_with(|| vec![true; nrows])[i] = false;
529            } else {
530                let l = left_data[i];
531                let r = right_data[i];
532                let result = match op {
533                    BinaryOp::Add => l + r,
534                    BinaryOp::Sub => l - r,
535                    BinaryOp::Mul => l * r,
536                    BinaryOp::Div => {
537                        if r == 0.0 {
538                            out_data.push(f64::NAN);
539                            if out_validity.is_none() {
540                                out_validity = Some(vec![true; nrows]);
541                            }
542                            out_validity.as_mut().unwrap()[i] = false;
543                            continue;
544                        }
545                        l / r
546                    }
547                    BinaryOp::Min => l.min(r),
548                    BinaryOp::Max => l.max(r),
549                };
550                out_data.push(result);
551            }
552        }
553        let store = ColumnStore::Float64 {
554            data: out_data,
555            validity: out_validity,
556        };
557        if let Some(oi) = self.column_names.iter().position(|n| n == out_col) {
558            self.columns[oi] = store;
559        } else {
560            self.column_names.push(out_col.to_string());
561            self.columns.push(store);
562        }
563        Ok(())
564    }
565    /// Reduce numeric column to single value using selected aggregation.
566    pub fn col_reduce(&self, col: &str, op: ReduceOp) -> Result<Option<f64>, String> {
567        let idx = self.resolve(col)?;
568        let vals = match &self.columns[idx] {
569            ColumnStore::Float64 { .. } => self.columns[idx]
570                .valid_f64s()
571                .ok_or_else(|| "impossible".to_string())?,
572            ColumnStore::Int64 { data, validity } => {
573                let base: Vec<f64> = data.iter().map(|v| *v as f64).collect();
574                if let Some(v) = validity {
575                    base.iter()
576                        .zip(v.iter())
577                        .filter_map(|(x, ok)| if *ok { Some(*x) } else { None })
578                        .collect()
579                } else {
580                    base
581                }
582            }
583            t => {
584                return Err(format!(
585                    "col_reduce: column '{col}' type '{}' is not numeric",
586                    t.dtype_name()
587                ))
588            }
589        };
590        if vals.is_empty() {
591            return Ok(None);
592        }
593        let result = match op {
594            ReduceOp::Count => Some(vals.len() as f64),
595            ReduceOp::Sum => Some(vals.iter().sum()),
596            ReduceOp::Mean => Some(vals.iter().sum::<f64>() / vals.len() as f64),
597            ReduceOp::Min => vals.iter().cloned().reduce(f64::min),
598            ReduceOp::Max => vals.iter().cloned().reduce(f64::max),
599            ReduceOp::Var => {
600                let mean = vals.iter().sum::<f64>() / vals.len() as f64;
601                let var = vals.iter().map(|x| (x - mean).powi(2)).sum::<f64>() / vals.len() as f64;
602                Some(var)
603            }
604            ReduceOp::Std => {
605                let mean = vals.iter().sum::<f64>() / vals.len() as f64;
606                let var = vals.iter().map(|x| (x - mean).powi(2)).sum::<f64>() / vals.len() as f64;
607                Some(var.sqrt())
608            }
609        };
610        Ok(result)
611    }
612    /// Build boolean mask by comparing numeric column against scalar value.
613    pub fn filter_mask(&self, col: &str, op: CmpOp, val: f64) -> Result<Vec<bool>, String> {
614        let idx = self.resolve(col)?;
615        let nrows = self.nrows();
616        let mut mask = vec![false; nrows];
617        match &self.columns[idx] {
618            ColumnStore::Float64 { data, validity } => {
619                for (i, x) in data.iter().enumerate() {
620                    if validity.as_ref().is_none_or(|v| v[i]) {
621                        mask[i] = cmp_f64(*x, op, val);
622                    }
623                }
624            }
625            ColumnStore::Int64 { data, validity } => {
626                for (i, x) in data.iter().enumerate() {
627                    if validity.as_ref().is_none_or(|v| v[i]) {
628                        mask[i] = cmp_f64(*x as f64, op, val);
629                    }
630                }
631            }
632            t => {
633                return Err(format!(
634                    "filter_mask: column '{col}' type '{}' is not numeric",
635                    t.dtype_name()
636                ))
637            }
638        }
639        Ok(mask)
640    }
641    /// Filter all columns by boolean mask and return new VecFrame.
642    pub fn apply_mask(&self, mask: &[bool]) -> Result<VecFrame, String> {
643        if mask.len() != self.nrows() {
644            return Err(format!(
645                "apply_mask: mask length {} != nrows {}",
646                mask.len(),
647                self.nrows()
648            ));
649        }
650        let columns: Vec<ColumnStore> = self.columns.iter().map(|c| c.filter(mask)).collect();
651        Ok(VecFrame {
652            column_names: self.column_names.clone(),
653            columns,
654        })
655    }
656    /// Cast named column to target type in place.
657    pub fn col_cast(&mut self, col: &str, dtype: &str) -> Result<(), String> {
658        let idx = self.resolve(col)?;
659        let nrows = self.nrows();
660        let new_store = match (self.columns[idx].dtype_name(), dtype) {
661            ("float64", "int64") => {
662                if let ColumnStore::Float64 { data, validity } = &self.columns[idx] {
663                    ColumnStore::Int64 {
664                        data: data.iter().map(|v| *v as i64).collect(),
665                        validity: validity.clone(),
666                    }
667                } else {
668                    unreachable!()
669                }
670            }
671            ("int64", "float64") => {
672                if let ColumnStore::Int64 { data, validity } = &self.columns[idx] {
673                    ColumnStore::Float64 {
674                        data: data.iter().map(|v| *v as f64).collect(),
675                        validity: validity.clone(),
676                    }
677                } else {
678                    unreachable!()
679                }
680            }
681            ("float64", "text") => {
682                if let ColumnStore::Float64 { data, validity } = &self.columns[idx] {
683                    ColumnStore::Text {
684                        data: data.iter().map(|v| format!("{v}")).collect(),
685                        validity: validity.clone(),
686                    }
687                } else {
688                    unreachable!()
689                }
690            }
691            ("int64", "text") => {
692                if let ColumnStore::Int64 { data, validity } = &self.columns[idx] {
693                    ColumnStore::Text {
694                        data: data.iter().map(|v| format!("{v}")).collect(),
695                        validity: validity.clone(),
696                    }
697                } else {
698                    unreachable!()
699                }
700            }
701            (from, to) if from == to => return Ok(()),
702            (from, to) => {
703                let _ = nrows;
704                return Err(format!("col_cast: cannot cast '{col}' from {from} to {to}"));
705            }
706        };
707        self.columns[idx] = new_store;
708        Ok(())
709    }
710    /// Reduce multiple columns in parallel and return name-to-result map.
711    pub fn par_reduce(&self, cols: &[&str], op: ReduceOp) -> HashMap<String, Option<f64>> {
712        cols.par_iter()
713            .map(|col| {
714                let result = self.col_reduce(col, op).ok().flatten();
715                (col.to_string(), result)
716            })
717            .collect()
718    }
719    /// Apply scalar operation across multiple Float64 columns in parallel.
720    pub fn par_scalar_op(&mut self, cols: &[&str], op: ScalarOp, val: f64) -> Result<(), String> {
721        let indices: Result<Vec<usize>, String> = cols.iter().map(|c| self.resolve(c)).collect();
722        let indices = indices?;
723        let updates: Vec<(usize, Vec<f64>)> = indices
724            .par_iter()
725            .filter_map(|&idx| {
726                if let ColumnStore::Float64 { data, validity } = &self.columns[idx] {
727                    let vref = validity.as_ref();
728                    let new_data: Vec<f64> = data
729                        .iter()
730                        .enumerate()
731                        .map(|(i, &x)| {
732                            if vref.is_none_or(|v| v[i]) {
733                                apply_scalar_f64(x, op, val)
734                            } else {
735                                x
736                            }
737                        })
738                        .collect();
739                    Some((idx, new_data))
740                } else {
741                    None
742                }
743            })
744            .collect();
745        for (idx, new_data) in updates {
746            if let ColumnStore::Float64 { data, .. } = &mut self.columns[idx] {
747                *data = new_data;
748            }
749        }
750        Ok(())
751    }
752}
753/// Implement `Default` for `VecFrame` by delegating to `VecFrame::new`.
754impl Default for VecFrame {
755    /// Create default empty vectorized frame.
756    fn default() -> Self {
757        Self::new()
758    }

760impl DataFrame {
761    /// Parallel element-wise transform on a column.
762    /// Uses rayon when the column has 10,000+ rows; sequential otherwise.
763    pub fn par_apply_column<F>(&mut self, col: ColRef, f: F) -> Result<(), String>
764    where
765        F: Fn(&CellValue) -> CellValue + Send + Sync,
766    {
767        let ci = self.resolve_col(col)?;
768        let col_data = &self.data[ci];
769        const PAR_THRESHOLD: usize = 10_000;
770
771        let new_col: Vec<CellValue> = if col_data.len() >= PAR_THRESHOLD {
772            col_data.par_iter().map(&f).collect()
773        } else {
774            col_data.iter().map(f).collect()
775        };
776        self.data[ci] = new_col;
777        Ok(())
778    }
779}
780/// Infer best column storage type from a vector of cell values.
781fn infer_column(cells: &[CellValue]) -> ColumnStore {
782    let mut has_num = false;
783    let mut has_bool = false;
784    let mut has_text = false;
785    let mut has_nil = false;
786    for c in cells.iter() {
787        match c {
788            CellValue::Number(_) => has_num = true,
789            CellValue::Bool(_) => has_bool = true,
790            CellValue::Text(_) => has_text = true,
791            CellValue::Nil => has_nil = true,
792        }
793    }
794    let n = cells.len();
795    if has_bool && !has_num && !has_text {
796        let mut data = Vec::with_capacity(n);
797        let mut validity = if has_nil { Some(vec![true; n]) } else { None };
798        for (i, c) in cells.iter().enumerate() {
799            match c {
800                CellValue::Bool(b) => data.push(*b),
801                CellValue::Nil => {
802                    data.push(false);
803                    if let Some(v) = &mut validity {
804                        v[i] = false;
805                    }
806                }
807                _ => data.push(false),
808            }
809        }
810        ColumnStore::Bool { data, validity }
811    } else if has_text && !has_num && !has_bool {
812        let mut data = Vec::with_capacity(n);
813        let mut validity = if has_nil { Some(vec![true; n]) } else { None };
814        for (i, c) in cells.iter().enumerate() {
815            match c {
816                CellValue::Text(s) => data.push(s.clone()),
817                CellValue::Nil => {
818                    data.push(String::new());
819                    if let Some(v) = &mut validity {
820                        v[i] = false;
821                    }
822                }
823                _ => data.push(String::new()),
824            }
825        }
826        ColumnStore::Text { data, validity }
827    } else {
828        let mut data = Vec::with_capacity(n);
829        let mut validity = if has_nil || has_text || has_bool {
830            Some(vec![true; n])
831        } else {
832            None
833        };
834        for (i, c) in cells.iter().enumerate() {
835            match c {
836                CellValue::Number(v) => data.push(*v),
837                CellValue::Nil => {
838                    data.push(0.0);
839                    if let Some(v) = &mut validity {
840                        v[i] = false;
841                    }
842                }
843                CellValue::Bool(b) => data.push(if *b { 1.0 } else { 0.0 }),
844                CellValue::Text(_) => {
845                    data.push(0.0);
846                    if let Some(v) = &mut validity {
847                        v[i] = false;
848                    }
849                }
850            }
851        }
852        ColumnStore::Float64 { data, validity }
853    }
854}
855/// Apply scalar operation to one f64 value and return transformed value.
856fn apply_scalar_f64(x: f64, op: ScalarOp, val: f64) -> f64 {
857    match op {
858        ScalarOp::Add => x + val,
859        ScalarOp::Sub => x - val,
860        ScalarOp::Mul => x * val,
861        ScalarOp::Div => x / val,
862        ScalarOp::Abs => x.abs(),
863        ScalarOp::Sqrt => x.sqrt(),
864        ScalarOp::Floor => x.floor(),
865        ScalarOp::Ceil => x.ceil(),
866        ScalarOp::Neg => -x,
867    }
868}
869/// Compare one value against scalar and return predicate result.
870fn cmp_f64(x: f64, op: CmpOp, val: f64) -> bool {
871    match op {
872        CmpOp::Lt => x < val,
873        CmpOp::Le => x <= val,
874        CmpOp::Gt => x > val,
875        CmpOp::Ge => x >= val,
876        CmpOp::Eq => (x - val).abs() < f64::EPSILON,
877        CmpOp::Ne => (x - val).abs() >= f64::EPSILON,
878    }
879}
