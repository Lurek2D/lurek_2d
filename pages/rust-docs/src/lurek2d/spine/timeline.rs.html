9/// Interpolation curve applied between two adjacent keyframes.
10#[derive(Debug, Clone, PartialEq)]
11pub enum EasingType {
12    /// Constant interpolation — t passes through unchanged.
13    Linear,
14    /// Quadratic ease-in — t*t.
15    EaseIn,
16    /// Quadratic ease-out — 1-(1-t)^2.
17    EaseOut,
18    /// Piecewise quadratic S-curve.
19    EaseInOut,
20    /// Holds the previous keyframe value until the next key is reached.
21    Step,
22}
23/// Easing evaluation for EasingType.
24impl EasingType {
25    /// Evaluate this easing curve for t in [0, 1]; clamps input; Step always returns 0.0 (caller uses prev value).
26    pub fn apply(&self, t: f32) -> f32 {
27        let t = t.clamp(0.0, 1.0);
28        match self {
29            Self::Linear => t,
30            Self::EaseIn => t * t,
31            Self::EaseOut => 1.0 - (1.0 - t) * (1.0 - t),
32            Self::EaseInOut => {
33                if t < 0.5 {
34                    2.0 * t * t
35                } else {
36                    1.0 - 2.0 * (1.0 - t) * (1.0 - t)
37                }
38            }
39            Self::Step => 0.0,
40        }
41    }
42}
43/// Bone transform property targeted by a BoneTimeline.
44#[derive(Debug, Clone, PartialEq)]
45pub enum BoneProperty {
46    /// Animates Bone::local_x.
47    X,
48    /// Animates Bone::local_y.
49    Y,
50    /// Animates Bone::local_rotation in radians.
51    Rotation,
52    /// Animates Bone::local_scale_x.
53    ScaleX,
54    /// Animates Bone::local_scale_y.
55    ScaleY,
56}
57/// Single keyframe holding a time, value, and easing curve to the next keyframe.
58#[derive(Debug, Clone)]
59pub struct Keyframe {
60    /// Time in seconds at which this keyframe occurs.
61    pub time: f32,
62    /// Bone property value at this keyframe.
63    pub value: f32,
64    /// Easing curve used to interpolate from this keyframe to the next.
65    pub easing: EasingType,
66}
67/// Ordered keyframe sequence for a single bone property.
68#[derive(Debug, Clone)]
69pub struct BoneTimeline {
70    /// Index into the skeleton bone array.
71    pub bone_idx: usize,
72    /// Which bone local transform field this timeline drives.
73    pub property: BoneProperty,
74    /// Sorted keyframe list in ascending time order.
75    pub keys: Vec<Keyframe>,
76}
77/// Constructor and evaluation methods for BoneTimeline.
78impl BoneTimeline {
79    /// Create an empty BoneTimeline targeting bone_idx and the given property.
80    pub fn new(bone_idx: usize, property: BoneProperty) -> Self {
81        Self {
82            bone_idx,
83            property,
84            keys: Vec::new(),
85        }
86    }
87    /// Insert a keyframe at the correct sorted position by time.
88    pub fn add_key(&mut self, time: f32, value: f32, easing: EasingType) {
89        let kf = Keyframe {
90            time,
91            value,
92            easing,
93        };
94        let pos = self.keys.partition_point(|k| k.time <= time);
95        self.keys.insert(pos, kf);
96    }
97    /// Return the interpolated bone property value at the given time; extrapolates from first or last key outside the range.
98    pub fn evaluate(&self, time: f32) -> f32 {
99        if self.keys.is_empty() {
100            return 0.0;
101        }
102        if self.keys.len() == 1 || time <= self.keys[0].time {
103            return self.keys[0].value;
104        }
105        let last = self.keys.last().unwrap();
106        if time >= last.time {
107            return last.value;
108        }
109        let next_idx = self.keys.partition_point(|k| k.time <= time);
110        let prev_idx = next_idx - 1;
111        let prev = &self.keys[prev_idx];
112        let next = &self.keys[next_idx];
113        if prev.easing == EasingType::Step {
114            return prev.value;
115        }
116        let span = next.time - prev.time;
117        if span <= 0.0 {
118            return next.value;
119        }
120        let t = (time - prev.time) / span;
121        let eased_t = prev.easing.apply(t);
122        prev.value + (next.value - prev.value) * eased_t
123    }
124}

147/// Named animation clip: a set of bone timelines and event keyframes over a fixed duration.
148#[derive(Debug, Clone)]
149pub struct SkeletonAnimation {
150    /// Clip name used for lookup in Skeleton::play_animation.
151    pub name: String,
152    /// Total clip duration in seconds.
153    pub duration: f32,
154    /// Bone property timelines evaluated during playback.
155    pub timelines: Vec<BoneTimeline>,
156    /// Timed events fired via collect_events during playback.
157    pub events: Vec<EventKeyframe>,
158}
159/// Primary playback methods for SkeletonAnimation.
160impl SkeletonAnimation {
161    /// Create an empty animation clip with the given name and duration.
162    pub fn new(name: impl Into<String>, duration: f32) -> Self {
163        Self {
164            name: name.into(),
165            duration,
166            timelines: Vec::new(),
167            events: Vec::new(),
168        }
169    }
170    /// Append a BoneTimeline to this animation.
171    pub fn add_timeline(&mut self, timeline: BoneTimeline) {
172        self.timelines.push(timeline);
173    }
174    /// Insert an event at the given time; events are kept sorted by time.
175    pub fn add_event_key(&mut self, time: f32, name: impl Into<String>, value: f32) {
176        self.events.push(EventKeyframe::new(time, name, value));
177        self.events.sort_by(|a, b| {
178            a.time
179                .partial_cmp(&b.time)
180                .unwrap_or(std::cmp::Ordering::Equal)
181        });
182    }
183    /// Add a keyframe to a specific bone property timeline. Creates the timeline if it doesn't exist.
184    pub fn add_keyframe(
185        &mut self,
186        bone_idx: usize,
187        property: BoneProperty,
188        time: f32,
189        value: f32,
190        easing: EasingType,
191    ) {
192        if let Some(tl) = self
193            .timelines
194            .iter_mut()
195            .find(|tl| tl.bone_idx == bone_idx && tl.property == property)
196        {
197            tl.add_key(time, value, easing);
198        } else {
199            let mut tl = BoneTimeline::new(bone_idx, property);
200            tl.add_key(time, value, easing);
201            self.timelines.push(tl);
202        }
203    }
204    /// Return all event (name, value) pairs whose time is in the half-open range (from, to].
205    pub fn collect_events(&self, from: f32, to: f32) -> Vec<(String, f32)> {
206        self.events
207            .iter()
208            .filter(|e| e.time > from && e.time <= to)
209            .map(|e| (e.name.clone(), e.value))
210            .collect()
211    }
212    /// Apply all bone timelines at the given time to the target skeleton by setting local properties directly.
213    pub fn apply_to_skeleton(&self, skeleton: &mut super::skeleton::Skeleton, time: f32) {
214        for tl in &self.timelines {
215            let value = tl.evaluate(time);
216            if let Some(bone) = skeleton.bones.get_mut(tl.bone_idx) {
217                apply_bone_property(bone, &tl.property, value);
218            }
219        }
220    }
221    /// Apply all bone timelines blended with blend_weight in [0, 1]; weight=1 is full override, weight=0 is no-op.
222    pub fn apply_to_skeleton_blended(
223        &self,
224        skeleton: &mut super::skeleton::Skeleton,
225        time: f32,
226        blend_weight: f32,
227    ) {
228        let w = blend_weight.clamp(0.0, 1.0);
229        for tl in &self.timelines {
230            let target = tl.evaluate(time);
231            if let Some(bone) = skeleton.bones.get_mut(tl.bone_idx) {
232                let current = current_bone_property(bone, &tl.property);
233                let blended = current + (target - current) * w;
234                apply_bone_property(bone, &tl.property, blended);
235            }
236        }
237    }
238}
239/// Apply a property value to the matching local bone field.
240fn apply_bone_property(bone: &mut Bone, prop: &BoneProperty, value: f32) {
241    match prop {
242        BoneProperty::X => bone.local_x = value,
243        BoneProperty::Y => bone.local_y = value,
244        BoneProperty::Rotation => bone.local_rotation = value,
245        BoneProperty::ScaleX => bone.local_scale_x = value,
246        BoneProperty::ScaleY => bone.local_scale_y = value,
247    }
248}
249/// Read the current local bone field matching the given property.
250fn current_bone_property(bone: &Bone, prop: &BoneProperty) -> f32 {
251    match prop {
252        BoneProperty::X => bone.local_x,
253        BoneProperty::Y => bone.local_y,
254        BoneProperty::Rotation => bone.local_rotation,
255        BoneProperty::ScaleX => bone.local_scale_x,
256        BoneProperty::ScaleY => bone.local_scale_y,
257    }
258}
259/// Utility and data-query methods for SkeletonAnimation.
260impl SkeletonAnimation {
261    /// Return the evaluated (bone_idx, property, value) pose snapshot for all timelines at the given time.
262    /// Return the evaluated (bone_idx, property, value) pose snapshot for all timelines at the given time.
263    pub fn pose_at(&self, time: f32) -> Vec<(usize, BoneProperty, f32)> {
264        self.timelines
265            .iter()
266            .map(|tl| (tl.bone_idx, tl.property.clone(), tl.evaluate(time)))
267            .collect()
268    }
269    /// Return a new animation with all keyframe times mirrored around the clip duration.
270    pub fn reverse(&self) -> Self {
271        let dur = self.duration;
272        let timelines: Vec<BoneTimeline> = self
273            .timelines
274            .iter()
275            .map(|tl| {
276                let mut flipped = BoneTimeline::new(tl.bone_idx, tl.property.clone());
277                for kf in tl.keys.iter().rev() {
278                    flipped.add_key(dur - kf.time, kf.value, kf.easing.clone());
279                }
280                flipped
281            })
282            .collect();
283        let events: Vec<EventKeyframe> = self
284            .events
285            .iter()
286            .map(|e| EventKeyframe::new(dur - e.time, &e.name, e.value))
287            .collect();
288        Self {
289            name: format!("{}_reversed", self.name),
290            duration: dur,
291            timelines,
292            events,
293        }
294    }
295    /// Parse a SkeletonAnimation from a serde_json Value; returns None when required fields are missing or malformed.
296    pub fn from_json(v: &serde_json::Value) -> Option<Self> {
297        let name = v.get("name")?.as_str()?.to_owned();
298        let duration = v.get("duration")?.as_f64()? as f32;
299        let mut anim = Self::new(name, duration);
300        if let Some(timelines) = v.get("timelines").and_then(|t| t.as_array()) {
301            for tl_val in timelines {
302                let bone_idx = tl_val.get("bone_idx")?.as_u64()? as usize;
303                let property = match tl_val.get("property")?.as_str()? {
304                    "x" => BoneProperty::X,
305                    "y" => BoneProperty::Y,
306                    "rotation" => BoneProperty::Rotation,
307                    "scale_x" => BoneProperty::ScaleX,
308                    "scale_y" => BoneProperty::ScaleY,
309                    _ => continue,
310                };
311                let mut tl = BoneTimeline::new(bone_idx, property);
312                if let Some(keys) = tl_val.get("keys").and_then(|k| k.as_array()) {
313                    for kf in keys {
314                        let t = kf.get("time").and_then(|v| v.as_f64()).unwrap_or(0.0) as f32;
315                        let val = kf.get("value").and_then(|v| v.as_f64()).unwrap_or(0.0) as f32;
316                        let easing = match kf
317                            .get("easing")
318                            .and_then(|e| e.as_str())
319                            .unwrap_or("linear")
320                        {
321                            "ease_in" => EasingType::EaseIn,
322                            "ease_out" => EasingType::EaseOut,
323                            "ease_in_out" => EasingType::EaseInOut,
324                            "step" => EasingType::Step,
325                            _ => EasingType::Linear,
326                        };
327                        tl.add_key(t, val, easing);
328                    }
329                }
330                anim.add_timeline(tl);
331            }
332        }
333        if let Some(events) = v.get("events").and_then(|e| e.as_array()) {
334            for ev in events {
335                let t = ev.get("time").and_then(|v| v.as_f64()).unwrap_or(0.0) as f32;
336                let ev_name = ev
337                    .get("name")
338                    .and_then(|v| v.as_str())
339                    .unwrap_or("")
340                    .to_owned();
341                let val = ev.get("value").and_then(|v| v.as_f64()).unwrap_or(0.0) as f32;
342                anim.add_event_key(t, ev_name, val);
343            }
344        }
345        Some(anim)
346    }
