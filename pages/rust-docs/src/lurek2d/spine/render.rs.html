1//! Convert a Skeleton's bone and slot state into a flat list of RenderCommands.
2//!
3//! - Draw bones as filled circles at world positions with slot-derived colors.
4//! - Draw slot attachments as outline rectangles around their parent bone.
5
6use super::skeleton::Skeleton;
7use crate::render::renderer::{DrawMode, RenderCommand};
8
9/// Render methods added to Skeleton by this file.
10impl Skeleton {
11    /// Build a RenderCommand list for all bones (filled circles) and slot attachments (outline rectangles).
12    /// at world offset (x, y); returns empty vec when no bones exist.
13    pub fn generate_render_commands(&self, x: f32, y: f32) -> Vec<RenderCommand> {
14        let mut cmds = Vec::new();
15        if self.bones.is_empty() {
16            return cmds;
17        }
18        let radius = 4.0 * self.scale_x.abs().max(self.scale_y.abs()).max(1.0);
19        let mut bone_colors: Vec<Option<[f32; 4]>> = vec![None; self.bones.len()];
20        for slot in &self.slots {
21            if slot.bone_index < self.bones.len() && bone_colors[slot.bone_index].is_none() {
22                bone_colors[slot.bone_index] =
23                    Some([slot.color_r, slot.color_g, slot.color_b, slot.color_a]);
24            }
25        }
26        for (i, bone) in self.bones.iter().enumerate() {
27            let bx = x + bone.world_x;
28            let by = y + bone.world_y;
29            let [cr, cg, cb, ca] = bone_colors[i].unwrap_or([1.0, 1.0, 1.0, 1.0]);
30            cmds.push(RenderCommand::SetColor(cr, cg, cb, ca));
31            cmds.push(RenderCommand::Circle {
32                mode: DrawMode::Fill,
33                x: bx,
34                y: by,
35                r: radius,
36            });
37        }
38        for slot in &self.slots {
39            if slot.attachment_name.is_none() {
40                continue;
41            }
42            if slot.bone_index >= self.bones.len() {
43                continue;
44            }
45            let bone = &self.bones[slot.bone_index];
46            let bx = x + bone.world_x;
47            let by = y + bone.world_y;
48            let half = radius;
49            cmds.push(RenderCommand::SetColor(
50                slot.color_r,
51                slot.color_g,
52                slot.color_b,
53                slot.color_a * 0.5,
54            ));
55            cmds.push(RenderCommand::Rectangle {
56                mode: DrawMode::Line,
57                x: bx - half,
58                y: by - half,
59                w: half * 2.0,
60                h: half * 2.0,
61            });
62        }
63        cmds
64    }
65}
