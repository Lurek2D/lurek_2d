3//! - Parent-child hierarchy via optional parent index.

6/// Single bone in the skeleton tree with local and accumulated world-space transform.
7#[derive(Debug, Clone)]
8pub struct Bone {
9    /// Unique name identifying this bone within the skeleton.
10    pub name: String,
11    /// Index of the parent bone in the skeleton bone array; None for the root bone.
12    pub parent_index: Option<usize>,
13    /// X translation relative to the parent bone (or world origin for root), in pixels.
14    pub local_x: f32,
15    /// Y translation relative to the parent bone (or world origin for root), in pixels.
16    pub local_y: f32,
17    /// Rotation relative to the parent, in radians counter-clockwise.
18    pub local_rotation: f32,
19    /// Horizontal scale relative to the parent; 1.0 = no scale.
20    pub local_scale_x: f32,
21    /// Vertical scale relative to the parent; 1.0 = no scale.
22    pub local_scale_y: f32,
23    /// Accumulated world-space X position after pose update.
24    pub world_x: f32,
25    /// Accumulated world-space Y position after pose update.
26    pub world_y: f32,
27    /// Accumulated world-space rotation in radians.
28    pub world_rotation: f32,
29    /// Accumulated world-space horizontal scale.
30    pub world_scale_x: f32,
31    /// Accumulated world-space vertical scale.
32    pub world_scale_y: f32,
33}
34
35/// Constructor methods for Bone.
36impl Bone {
37    /// Create a root bone at world origin with identity transform.
38    pub fn new(name: impl Into<String>) -> Self {
39        Self {
40            name: name.into(),
41            parent_index: None,
42            local_x: 0.0,
43            local_y: 0.0,
44            local_rotation: 0.0,
45            local_scale_x: 1.0,
46            local_scale_y: 1.0,
47            world_x: 0.0,
48            world_y: 0.0,
49            world_rotation: 0.0,
50            world_scale_x: 1.0,
51            world_scale_y: 1.0,
52        }
53    }
54    /// Create a child bone with the given parent index and local (x, y) offset; rotation and scale default to identity.
55    pub fn with_parent(name: impl Into<String>, parent: usize, x: f32, y: f32) -> Self {
56        Self {
57            name: name.into(),
58            parent_index: Some(parent),
59            local_x: x,
60            local_y: y,
61            local_rotation: 0.0,
62            local_scale_x: 1.0,
63            local_scale_y: 1.0,
64            world_x: 0.0,
65            world_y: 0.0,
66            world_rotation: 0.0,
67            world_scale_x: 1.0,
68            world_scale_y: 1.0,
69        }
70    }
71}
