1//! Parse TOML strings into engine-internal `SerialValue` trees.
2//!
3//! - Encode `SerialValue` back to TOML text for config persistence.
4//! - Bridge between the `toml` crate's value types and the serial layer.
5
6use super::lua_table::SerialValue;
7use indexmap::IndexMap;
8
9/// Parse a raw TOML string into a `toml::Value` tree.
10pub fn parse_toml(input: &str) -> Result<toml::Value, String> {
11    input
12        .parse::<toml::Value>()
13        .map_err(|e| format!("TOML parse error: {e}"))
14}
15
16/// Parse a TOML string and convert it into a `SerialValue`.
17pub fn from_toml(s: &str) -> Result<SerialValue, String> {
18    let v = parse_toml(s)?;
19    Ok(toml_to_serial(v))
20}
21
22/// Encode a `toml::Value` table into a TOML-formatted string.
23pub fn encode_toml(value: &toml::Value) -> Result<String, String> {
24    match value {
25        toml::Value::Table(t) => toml::to_string(t).map_err(|e| format!("TOML encode error: {e}")),
26        _ => Err("encode_toml expects a table value".into()),
27    }
28}
29
30/// Convert a `SerialValue` into a TOML-encoded string.
31pub fn to_toml(val: &SerialValue) -> Result<String, String> {
32    let tv = serial_to_toml(val)?;
33    encode_toml(&tv)
34}
35
36/// Convert a `toml::Value` tree into the engine's `SerialValue` representation.
37fn toml_to_serial(v: toml::Value) -> SerialValue {
38    match v {
39        toml::Value::String(s) => SerialValue::Str(s),
40        toml::Value::Integer(n) => SerialValue::Int(n),
41        toml::Value::Float(f) => SerialValue::Float(f),
42        toml::Value::Boolean(b) => SerialValue::Bool(b),
43        toml::Value::Datetime(dt) => SerialValue::Str(dt.to_string()),
44        toml::Value::Array(arr) => SerialValue::Seq(arr.into_iter().map(toml_to_serial).collect()),
45        toml::Value::Table(map) => {
46            let mut ordered = IndexMap::new();
47            for (k, v) in map {
48                ordered.insert(k, toml_to_serial(v));
49            }
50            SerialValue::Map(ordered)
51        }
52    }
53}
54
55/// Convert a `SerialValue` into a `toml::Value`, failing on null.
56fn serial_to_toml(val: &SerialValue) -> Result<toml::Value, String> {
57    match val {
58        SerialValue::Null => Err("to_toml: null values are not supported in TOML".to_string()),
59        SerialValue::Bool(b) => Ok(toml::Value::Boolean(*b)),
60        SerialValue::Int(n) => Ok(toml::Value::Integer(*n)),
61        SerialValue::Float(f) => Ok(toml::Value::Float(*f)),
62        SerialValue::Str(s) => Ok(toml::Value::String(s.clone())),
63        SerialValue::Seq(arr) => {
64            let res: Result<Vec<_>, _> = arr.iter().map(serial_to_toml).collect();
65            Ok(toml::Value::Array(res?))
66        }
67        SerialValue::Map(map) => {
68            let mut table = toml::map::Map::new();
69            for (k, v) in map {
70                table.insert(k.clone(), serial_to_toml(v)?);
71            }
72            Ok(toml::Value::Table(table))
73        }
74    }
75}
