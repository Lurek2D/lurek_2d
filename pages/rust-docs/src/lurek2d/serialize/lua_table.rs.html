1//! Bidirectional conversion between Lua tables and a typed serial value tree.
2//!
3//! - Supports null, bool, int, float, string, sequence, and map variants.
4//! - Detects array-like tables automatically and emits `Seq`; otherwise emits `Map`.
5
6use indexmap::IndexMap;
7use mlua::prelude::{Lua, LuaResult, LuaValue};
8use std::fmt;
9/// Type-erased value tree used for Lua-to-Rust serialization.
10#[derive(Debug, Clone)]
11pub enum SerialValue {
12    /// Nil / absent value.
13    Null,
14    /// Boolean value.
15    Bool(bool),
16    /// Integer value (whole numbers stored losslessly).
17    Int(i64),
18    /// Floating-point value.
19    Float(f64),
20    /// UTF-8 string value.
21    Str(String),
22    /// Ordered sequence (array-like Lua table).
23    Seq(Vec<SerialValue>),
24    /// String-keyed map (non-sequential Lua table).
25    Map(IndexMap<String, SerialValue>),
26}
27/// Display scalar values inline; complex values show `[complex]`.
28impl fmt::Display for SerialValue {
29    /// Format this value for user-facing display.
30    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
31        match self {
32            SerialValue::Null => write!(f, ""),
33            SerialValue::Bool(b) => write!(f, "{b}"),
34            SerialValue::Int(n) => write!(f, "{n}"),
35            SerialValue::Float(v) => write!(f, "{v}"),
36            SerialValue::Str(s) => write!(f, "{s}"),
37            SerialValue::Seq(_) | SerialValue::Map(_) => write!(f, "[complex]"),
38        }
39    }
40}
41/// Convert a `SerialValue` tree into a Lua value (tables for Seq/Map).
42pub fn to_lua<'lua>(lua: &'lua Lua, val: &SerialValue) -> LuaResult<LuaValue<'lua>> {
43    match val {
44        SerialValue::Null => Ok(LuaValue::Nil),
45        SerialValue::Bool(b) => Ok(LuaValue::Boolean(*b)),
46        SerialValue::Int(n) => Ok(LuaValue::Integer(*n)),
47        SerialValue::Float(f) => Ok(LuaValue::Number(*f)),
48        SerialValue::Str(s) => Ok(LuaValue::String(lua.create_string(s)?)),
49        SerialValue::Seq(arr) => {
50            let t = lua.create_table()?;
51            for (i, v) in arr.iter().enumerate() {
52                t.set(i as i64 + 1, to_lua(lua, v)?)?;
53            }
54            Ok(LuaValue::Table(t))
55        }
56        SerialValue::Map(map) => {
57            let t = lua.create_table()?;
58            for (k, v) in map {
59                t.set(k.as_str(), to_lua(lua, v)?)?;
60            }
61            Ok(LuaValue::Table(t))
62        }
63    }
64}
65/// Convert a Lua value into a `SerialValue` tree, detecting arrays automatically.
66pub fn from_lua(val: &LuaValue) -> LuaResult<SerialValue> {
67    match val {
68        LuaValue::Nil => Ok(SerialValue::Null),
69        LuaValue::Boolean(b) => Ok(SerialValue::Bool(*b)),
70        LuaValue::Integer(n) => Ok(SerialValue::Int(*n)),
71        LuaValue::Number(f) => {
72            if f.fract() == 0.0 && *f >= i64::MIN as f64 && *f <= i64::MAX as f64 {
73                Ok(SerialValue::Int(*f as i64))
74            } else {
75                Ok(SerialValue::Float(*f))
76            }
77        }
78        LuaValue::String(s) => Ok(SerialValue::Str(
79            s.to_str()
80                .map_err(|e| mlua::Error::RuntimeError(format!("Invalid UTF-8: {e}")))?
81                .to_string(),
82        )),
83        LuaValue::Table(t) => {
84            let raw_len = t.raw_len();
85            if raw_len > 0 {
86                let mut is_seq = true;
87                for i in 1..=raw_len as i64 {
88                    let v: LuaValue = t.get(i)?;
89                    if v == LuaValue::Nil {
90                        is_seq = false;
91                        break;
92                    }
93                }
94                if is_seq {
95                    let mut arr = Vec::with_capacity(raw_len);
96                    for i in 1..=raw_len as i64 {
97                        let v: LuaValue = t.get(i)?;
98                        arr.push(from_lua(&v)?);
99                    }
100                    return Ok(SerialValue::Seq(arr));
101                }
102            }
103            let mut map = IndexMap::new();
104            for pair in t.clone().pairs::<LuaValue, LuaValue>() {
105                let (k, v) = pair?;
106                let key = match &k {
107                    LuaValue::String(s) => s
108                        .to_str()
109                        .map_err(|e| mlua::Error::RuntimeError(format!("Invalid UTF-8 key: {e}")))?
110                        .to_string(),
111                    LuaValue::Integer(n) => n.to_string(),
112                    LuaValue::Number(f) => f.to_string(),
113                    _ => {
114                        return Err(mlua::Error::RuntimeError(
115                            "serial: table keys must be strings or numbers".to_string(),
116                        ))
117                    }
118                };
119                map.insert(key, from_lua(&v)?);
120            }
121            Ok(SerialValue::Map(map))
122        }
123        _ => Err(mlua::Error::RuntimeError(
124            "serial: unsupported Lua value type".to_string(),
125        )),
126    }
