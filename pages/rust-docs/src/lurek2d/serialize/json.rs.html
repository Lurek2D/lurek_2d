1//! Parse JSON strings into the engine's `SerialValue` intermediate representation.
2//!
3//! - Encode `SerialValue` trees back to JSON (compact or pretty-printed).
4//! - Map JSON types (null, bool, number, string, array, object) to `SerialValue` variants bidirectionally.
5
6use super::lua_table::SerialValue;
7use crate::log_msg;
8use crate::runtime::log_messages::{SR01_JSON_OK, SR03_JSON_ENC};
9use indexmap::IndexMap;
10use serde_json::Value as JsonValue;
11
12/// Parse a JSON string into a `SerialValue` tree.
13pub fn from_json(s: &str) -> Result<SerialValue, String> {
14    let v: JsonValue = serde_json::from_str(s).map_err(|e| format!("JSON parse error: {e}"))?;
15    log_msg!(debug, SR01_JSON_OK);
16    Ok(json_to_serial(v))
17}
18/// Encode a `SerialValue` tree to a JSON string.
19pub fn to_json(val: &SerialValue, pretty: bool) -> Result<String, String> {
20    let jv = serial_to_json(val);
21    let result = if pretty {
22        serde_json::to_string_pretty(&jv).map_err(|e| format!("JSON encode error: {e}"))
23    } else {
24        serde_json::to_string(&jv).map_err(|e| format!("JSON encode error: {e}"))
25    };
26    if result.is_ok() {
27        log_msg!(debug, SR03_JSON_ENC);
28    }
29    result
30}
31/// Convert a `serde_json::Value` into the equivalent `SerialValue`.
32fn json_to_serial(v: JsonValue) -> SerialValue {
33    match v {
34        JsonValue::Null => SerialValue::Null,
35        JsonValue::Bool(b) => SerialValue::Bool(b),
36        JsonValue::Number(n) => {
37            if let Some(i) = n.as_i64() {
38                SerialValue::Int(i)
39            } else {
40                SerialValue::Float(n.as_f64().unwrap_or(0.0))
41            }
42        }
43        JsonValue::String(s) => SerialValue::Str(s),
44        JsonValue::Array(arr) => SerialValue::Seq(arr.into_iter().map(json_to_serial).collect()),
45        JsonValue::Object(map) => {
46            let mut ordered = IndexMap::new();
47            for (k, v) in map {
48                ordered.insert(k, json_to_serial(v));
49            }
50            SerialValue::Map(ordered)
51        }
52    }
53}
54/// Convert a `SerialValue` into the equivalent `serde_json::Value`.
55fn serial_to_json(val: &SerialValue) -> JsonValue {
56    match val {
57        SerialValue::Null => JsonValue::Null,
58        SerialValue::Bool(b) => JsonValue::Bool(*b),
59        SerialValue::Int(n) => JsonValue::Number((*n).into()),
60        SerialValue::Float(f) => serde_json::Number::from_f64(*f)
61            .map(JsonValue::Number)
62            .unwrap_or(JsonValue::Null),
63        SerialValue::Str(s) => JsonValue::String(s.clone()),
64        SerialValue::Seq(arr) => JsonValue::Array(arr.iter().map(serial_to_json).collect()),
65        SerialValue::Map(map) => {
66            let obj: serde_json::Map<_, _> = map
67                .iter()
68                .map(|(k, v)| (k.clone(), serial_to_json(v)))
69                .collect();
70            JsonValue::Object(obj)
71        }
72    }
73}
