1//! Parse INI text into a nested `SerialValue` map.
2//!
3//! - Support sections, key=value pairs, and comment lines.
4//! - Preserve insertion order via `IndexMap`.
5
6use super::lua_table::SerialValue;
7use indexmap::IndexMap;
8
9/// Parse an INI-formatted string into a `SerialValue::Map`.
10pub fn from_ini(s: &str) -> Result<SerialValue, String> {
11    let mut root: IndexMap<String, SerialValue> = IndexMap::new();
12    let mut sections: IndexMap<String, IndexMap<String, SerialValue>> = IndexMap::new();
13    let mut current_section: Option<String> = None;
14    for (line_no, raw_line) in s.lines().enumerate() {
15        let line = raw_line.trim();
16        if line.is_empty() || line.starts_with(';') || line.starts_with('#') {
17            continue;
18        }
19        if line.starts_with('[') && line.ends_with(']') {
20            let section = line[1..line.len() - 1].trim();
21            if section.is_empty() {
22                return Err(format!(
23                    "INI parse error at line {}: empty section name",
24                    line_no + 1
25                ));
26            }
27            current_section = Some(section.to_string());
28            sections.entry(section.to_string()).or_default();
29            continue;
30        }
31        let Some((k, v)) = line.split_once('=') else {
32            return Err(format!(
33                "INI parse error at line {}: expected key=value",
34                line_no + 1
35            ));
36        };
37        let key = k.trim();
38        if key.is_empty() {
39            return Err(format!(
40                "INI parse error at line {}: empty key",
41                line_no + 1
42            ));
43        }
44        let value = SerialValue::Str(v.trim().to_string());
45        if let Some(section) = current_section.as_ref() {
46            sections
47                .entry(section.clone())
48                .or_default()
49                .insert(key.to_string(), value);
50        } else {
51            root.insert(key.to_string(), value);
52        }
53    }
54    for (section, map) in sections {
55        root.insert(section, SerialValue::Map(map));
56    }
57    Ok(SerialValue::Map(root))
58}
