1//! Pack and unpack 24-bit slot + 8-bit generation into a single u32 entity id.
2//!
3//! - Stateless utility struct with no allocation or state.
4//! - Supports up to ~16M slots and 256 generations per slot.
5
6/// Stateless namespace for encoding and decoding packed entity identifiers.
7pub struct GenerationalId;
8
9/// Encoding and decoding helpers for packed generational entity ids.
10impl GenerationalId {
11    /// Packs a 24-bit slot and 8-bit generation into a single entity id.
12    pub fn pack(slot: u32, generation: u8) -> u32 {
13        ((generation as u32) << 24) | (slot & 0x00FF_FFFF)
14    }
15    /// Extracts the slot index from a packed entity id.
16    pub fn unpack_slot(id: u32) -> u32 {
17        id & 0x00FF_FFFF
18    }
19    /// Extracts the generation byte from a packed entity id.
20    pub fn unpack_gen(id: u32) -> u8 {
21        (id >> 24) as u8
22    }
23}
