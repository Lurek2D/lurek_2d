1//! Extended Universe operations: advanced queries, bulk spawning, and state serialization.
2//!
3//! - query_not filters entities by required and excluded component sets.
4//! - query_multi invokes a callback with packed ids and multiple component values per entity.
5//! - spawn_bulk creates many entities from a single blueprint with optional per-entity overrides.
6//! - serialize_to_table / deserialize_from_table convert live universe state to and from Lua tables.
7//! - Serialization captures components, tags, layers, bitmap masks, and parent-child hierarchy.
8
9use super::Universe;
10use crate::ecs::lua_table::deep_copy_table;
11use mlua::{Function, Lua, Result as LuaResult, Table, Value as LuaValue};
12impl Universe {
13    /// Returns entities that contain all required components and none of the excluded ones.
14    pub fn query_not(
15        &self,
16        lua: &Lua,
17        with_names: &[String],
18        without_names: &[String],
19    ) -> LuaResult<Vec<u32>> {
20        let mut result = Vec::new();
21        if let Some(ref key) = self.component_store {
22            let store: Table = lua.registry_value(key)?;
23            for slot in self.candidate_slots_for_all(with_names) {
24                if let Ok(entity_table) = store.get::<_, Table>(slot) {
25                    let mut has_excluded = false;
26                    for name in without_names {
27                        let val: LuaValue = entity_table.get(name.as_str())?;
28                        if !val.is_nil() {
29                            has_excluded = true;
30                            break;
31                        }
32                    }
33                    if !has_excluded {
34                        result.push(Self::pack_id(slot, self.current_gen(slot)));
35                    }
36                } else if with_names.is_empty() {
37                    result.push(Self::pack_id(slot, self.current_gen(slot)));
38                }
39            }
40        } else if with_names.is_empty() {
41            result = self.get_entities();
42        }
43        result.sort();
44        Ok(result)
45    }
46    /// Invokes a Lua callback with entity ids followed by multiple requested component values.
47    pub fn query_multi(&self, lua: &Lua, names: &[String], callback: Function) -> LuaResult<()> {
48        if names.is_empty() {
49            return Ok(());
50        }
51        if let Some(ref key) = self.component_store {
52            let store: Table = lua.registry_value(key)?;
53            let mut slots = self.candidate_slots_for_all(names);
54            slots.sort();
55            for slot in slots {
56                if let Ok(entity_table) = store.get::<_, Table>(slot) {
57                    let mut vals: Vec<LuaValue> = Vec::with_capacity(names.len());
58                    let mut all = true;
59                    for name in names {
60                        let v: LuaValue = entity_table.get(name.as_str())?;
61                        if v.is_nil() {
62                            all = false;
63                            break;
64                        }
65                        vals.push(v);
66                    }
67                    if all {
68                        let id = Self::pack_id(slot, self.current_gen(slot));
69                        let mut args = Vec::with_capacity(1 + vals.len());
70                        args.push(LuaValue::Integer(id as i64));
71                        args.extend(vals);
72                        callback.call::<_, ()>(mlua::MultiValue::from_vec(args))?;
73                    }
74                }
75            }
76        }
77        Ok(())
78    }
79    /// Spawns multiple entities from one blueprint, cloning the optional override table per spawn.
80    pub fn spawn_bulk(
81        &mut self,
82        lua: &Lua,
83        name: &str,
84        count: usize,
85        overrides: Option<Table>,
86    ) -> LuaResult<Vec<u32>> {
87        let mut ids = Vec::with_capacity(count);
88        for _ in 0..count {
89            let ov_copy = if let Some(ref ov) = overrides {
90                Some(deep_copy_table(lua, ov)?)
91            } else {
92                None
93            };
94            ids.push(self.spawn_blueprint(lua, name, ov_copy)?);
95        }
96        Ok(ids)
97    }
98    /// Serializes live entities, components, tags, layers, and hierarchy into a Lua snapshot table.
99    pub fn serialize_to_table<'lua>(&self, lua: &'lua Lua) -> LuaResult<Table<'lua>> {
100        let snapshot = lua.create_table()?;
101        let entities_arr = lua.create_table()?;
102        let mut sorted_slots: Vec<u32> = self.alive.iter().copied().collect();
103        sorted_slots.sort();
104        for (i, slot) in sorted_slots.iter().enumerate() {
105            let slot = *slot;
106            let generation = self.current_gen(slot);
107            let id = Self::pack_id(slot, generation);
108            let entry = lua.create_table()?;
109            entry.set("id", id)?;
110            entry.set("slot", slot)?;
111            entry.set("gen", generation)?;
112            let components = lua.create_table()?;
113            if let Some(ref key) = self.component_store {
114                let store: Table = lua.registry_value(key)?;
115                if let Ok(comp_row) = store.get::<_, Table>(slot) {
116                    for pair in comp_row.clone().pairs::<String, LuaValue>() {
117                        let (k, v) = pair?;
118                        let v_copy = match v {
119                            LuaValue::Table(ref t) => LuaValue::Table(deep_copy_table(lua, t)?),
120                            other => other,
121                        };
122                        components.set(k, v_copy)?;
123                    }
124                }
125            }
126            entry.set("components", components)?;
127            let tags = lua.create_table()?;
128            if let Some(tag_list) = self.string_tags.get(&slot) {
129                for (j, t) in tag_list.iter().enumerate() {
130                    tags.set(j + 1, t.as_str())?;
131                }
132            }
133            entry.set("tags", tags)?;
134            entry.set("layer", self.layers.get(&slot).copied().unwrap_or(0))?;
135            entry.set(
136                "bitmap",
137                self.bitmap_masks.get(&slot).copied().unwrap_or(0u64) as i64,
138            )?;
139            if let Some(&parent_slot) = self.parents.get(&slot) {
140                if self.alive.contains(&parent_slot) {
141                    entry.set(
142                        "parent",
143                        Self::pack_id(parent_slot, self.current_gen(parent_slot)),
144                    )?;
145                }
146            }
147            entities_arr.set(i + 1, entry)?;
148        }
149        snapshot.set("entities", entities_arr)?;
150        let btnames = lua.create_table()?;
151        for (j, name) in self.bitmap_tag_names.iter().enumerate() {
152            btnames.set(j + 1, name.as_str())?;
153        }
154        snapshot.set("bitmap_tags", btnames)?;
155        Ok(snapshot)
156    }
157    /// Rebuilds universe state from a previously serialized Lua snapshot table.
158    pub fn deserialize_from_table(&mut self, lua: &Lua, snapshot: Table) -> LuaResult<()> {
159        self.alive.clear();
160        self.free_list.clear();
161        self.next_id = 1;
162        self.string_tags.clear();
163        self.tag_index.clear();
164        self.generations.clear();
165        self.bitmap_masks.clear();
166        self.layers.clear();
167        self.parents.clear();
168        self.children.clear();
169        self.add_events.clear();
170        self.remove_events.clear();
171        self.dirty_set.clear();
172        #[cfg(feature = "ecs-archetype")]
173        self.component_index.clear();
174        if let Some(ref key) = self.component_store {
175            let store: Table = lua.registry_value(key)?;
176            let ks: Vec<u32> = store
177                .clone()
178                .pairs::<u32, LuaValue>()
179                .filter_map(|p| p.ok().map(|(k, _)| k))
180                .collect();
181            for k in ks {
182                store.set(k, LuaValue::Nil)?;
183            }
184        }
185        self.ensure_stores(lua)?;
186        let comp_store = self.get_component_store(lua)?;
187        if let Ok(btnames) = snapshot.get::<_, Table>("bitmap_tags") {
188            for name in btnames.clone().sequence_values::<String>() {
189                let name = name?;
190                if !self.bitmap_tag_names.contains(&name) {
191                    self.bitmap_tag_names.push(name);
192                }
193            }
194        }
195        let entities: Table = snapshot.get("entities")?;
196        let mut parent_data: Vec<(u32, u32)> = Vec::new();
197        for entry_val in entities.clone().sequence_values::<Table>() {
198            let entry = entry_val?;
199            let id: u32 = entry.get("id")?;
200            let slot = Self::unpack_slot(id);
201            let generation: u8 = entry.get("gen").unwrap_or(0);
202            self.alive.insert(slot);
203            *self.generations.entry(slot).or_insert(0) = generation;
204            if slot >= self.next_id {
205                self.next_id = slot + 1;
206            }
207            let comp_row = lua.create_table()?;
208            if let Ok(components) = entry.get::<_, Table>("components") {
209                for pair in components.clone().pairs::<LuaValue, LuaValue>() {
210                    let (k, v) = pair?;
211                    comp_row.set(k, v)?;
212                }
213            }
214            comp_store.set(slot, comp_row.clone())?;
215            #[cfg(feature = "ecs-archetype")]
216            self.reindex_component_row(slot, &comp_row)?;
217            if let Ok(tags) = entry.get::<_, Table>("tags") {
218                let mut tag_list = Vec::new();
219                for t in tags.sequence_values::<String>() {
220                    let t = t?;
221                    self.tag_index.entry(t.clone()).or_default().push(id);
222                    tag_list.push(t);
223                }
224                if !tag_list.is_empty() {
225                    self.string_tags.insert(slot, tag_list);
226                }
227            }
228            let layer: i32 = entry.get("layer").unwrap_or(0);
229            if layer != 0 {
230                self.layers.insert(slot, layer);
231            }
232            let bitmap: i64 = entry.get("bitmap").unwrap_or(0);
233            if bitmap != 0 {
234                self.bitmap_masks.insert(slot, bitmap as u64);
235            }
236            if let Ok(parent_id) = entry.get::<_, u32>("parent") {
237                parent_data.push((slot, parent_id));
238            }
239        }
240        for (child_slot, parent_id) in parent_data {
241            let parent_slot = Self::unpack_slot(parent_id);
242            if self.alive.contains(&parent_slot) {
243                self.parents.insert(child_slot, parent_slot);
244                self.children
245                    .entry(parent_slot)
246                    .or_default()
247                    .push(child_slot);
248            }
249        }
250        Ok(())
251    }
