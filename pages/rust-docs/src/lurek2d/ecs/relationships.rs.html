1//! Relationship type definitions with named level labels and validated defaults.
2//!
3//! - Pairwise relationship records storing numeric affinity and per-type level state.
4//! - Canonical entity-pair ordering for symmetric, order-independent lookups.
5//! - Directed named links between entities for one-way associations.
6//! - Query helpers: filter by entity, check existence, iterate all relations.
7
8use crate::log_msg;
9use crate::runtime::log_messages::{RL01, RL02, RL03};
10use std::collections::HashMap;
11#[derive(Debug, Clone)]
12/// Declares a named relationship kind and the allowed level labels for it.
13pub struct RelationType {
14    /// Stable relationship type name used as the lookup key.
15    pub name: String,
16    /// Ordered list of valid level labels for this relationship type.
17    pub levels: Vec<String>,
18    /// Fallback level returned when a relation has no explicit value for this type.
19    pub default_level: String,
20}
21impl RelationType {
22    /// Creates a relationship type and coerces the default level to a valid entry when possible.
23    pub fn new(name: &str, levels: Vec<String>, default_level: &str) -> Self {
24        let effective_default = if levels.iter().any(|l| l == default_level) {
25            default_level.to_string()
26        } else if let Some(first) = levels.first() {
27            first.clone()
28        } else {
29            default_level.to_string()
30        };
31        Self {
32            name: name.to_string(),
33            levels,
34            default_level: effective_default,
35        }
36    }
37    /// Returns whether the supplied level label is valid for this relationship type.
38    pub fn has_level(&self, level: &str) -> bool {
39        self.levels.iter().any(|l| l == level)
40    }
41}
42#[derive(Debug, Clone)]
43/// Stores pairwise relationship data for an unordered entity pair.
44pub struct Relationship {
45    /// First packed entity id in canonical sorted order.
46    pub from_id: u32,
47    /// Second packed entity id in canonical sorted order.
48    pub to_id: u32,
49    /// Numeric affinity or score assigned to this relationship.
50    pub value: f64,
51    /// Per-type level labels assigned to this relationship.
52    pub type_levels: HashMap<String, String>,
53}
54impl Relationship {
55    /// Creates an empty relationship record for the given entity pair.
56    fn new(a: u32, b: u32) -> Self {
57        let (from_id, to_id) = ordered(a, b);
58        Self {
59            from_id,
60            to_id,
61            value: 0.0,
62            type_levels: HashMap::new(),
63        }
64    }
65}
66#[derive(Debug, Default)]
67/// Owns relationship types, pairwise relation records, and directed named links.
68pub struct RelationshipManager {
69    /// Registered relationship type definitions keyed by name.
70    types: HashMap<String, RelationType>,
71    /// Unordered pairwise relationships keyed by canonical entity ids.
72    relations: HashMap<(u32, u32), Relationship>,
73    /// Directed link targets keyed by source entity id and link name.
74    directed: HashMap<(u32, String), Vec<u32>>,
75}
76#[inline]
77/// Returns an entity pair in deterministic ascending order for map keys.
78fn ordered(a: u32, b: u32) -> (u32, u32) {
79    if a <= b {
80        (a, b)
81    } else {
82        (b, a)
83    }
84}
85impl RelationshipManager {
86    /// Creates an empty relationship manager.
87    pub fn new() -> Self {
88        Self::default()
89    }
90    /// Registers or replaces a named relationship type definition.
91    pub fn define_type(&mut self, name: &str, levels: Vec<String>, default_level: &str) {
92        log_msg!(debug, RL01, "{} ({} levels)", name, levels.len());
93        self.types.insert(
94            name.to_string(),
95            RelationType::new(name, levels, default_level),
96        );
97    }
98    /// Removes a relationship type and clears its assigned levels from all relations.
99    pub fn remove_type(&mut self, name: &str) -> bool {
100        if self.types.remove(name).is_some() {
101            for rel in self.relations.values_mut() {
102                rel.type_levels.remove(name);
103            }
104            log_msg!(debug, RL02, "{}", name);
105            true
106        } else {
107            false
108        }
109    }
110    /// Returns the relationship type definition for the given name.
111    pub fn get_type(&self, name: &str) -> Option<&RelationType> {
112        self.types.get(name)
113    }
114    /// Returns the set of registered relationship type names.
115    pub fn type_names(&self) -> Vec<String> {
116        self.types.keys().cloned().collect()
117    }
118    /// Returns the relation entry for a pair, creating it on first access.
119    fn ensure(&mut self, a: u32, b: u32) -> &mut Relationship {
120        let key = ordered(a, b);
121        self.relations.entry(key).or_insert_with(|| {
122            log_msg!(trace, RL03, "({}, {})", a, b);
123            Relationship::new(a, b)
124        })
125    }
126    /// Returns the numeric relationship value for a pair, defaulting to zero.
127    pub fn get_value(&self, a: u32, b: u32) -> f64 {
128        self.relations
129            .get(&ordered(a, b))
130            .map(|r| r.value)
131            .unwrap_or(0.0)
132    }
133    /// Sets the numeric relationship value for a pair.
134    pub fn set_value(&mut self, a: u32, b: u32, value: f64) {
135        self.ensure(a, b).value = value;
136    }
137    /// Adds a delta to the numeric relationship value for a pair.
138    pub fn adjust_value(&mut self, a: u32, b: u32, delta: f64) {
139        let rel = self.ensure(a, b);
140        rel.value += delta;
141    }
142    /// Assigns a per-type level label to a relation when the type and level are valid.
143    pub fn set_level(&mut self, a: u32, b: u32, type_name: &str, level: &str) -> bool {
144        match self.types.get(type_name) {
145            None => false,
146            Some(t) if !t.has_level(level) => false,
147            Some(_) => {
148                self.ensure(a, b)
149                    .type_levels
150                    .insert(type_name.to_string(), level.to_string());
151                true
152            }
153        }
154    }
155    /// Returns the explicit or default level label for a relation type on a pair.
156    pub fn get_level(&self, a: u32, b: u32, type_name: &str) -> Option<String> {
157        let def = self.types.get(type_name)?.default_level.clone();
158        let level = self
159            .relations
160            .get(&ordered(a, b))
161            .and_then(|r| r.type_levels.get(type_name))
162            .cloned()
163            .unwrap_or(def);
164        Some(level)
165    }
166    /// Returns whether any relation record exists for the pair.
167    pub fn has_relation(&self, a: u32, b: u32) -> bool {
168        self.relations.contains_key(&ordered(a, b))
169    }
170    /// Deletes the stored relation record for the pair.
171    pub fn remove_relation(&mut self, a: u32, b: u32) -> bool {
172        self.relations.remove(&ordered(a, b)).is_some()
173    }
174    /// Returns all relationship records that involve the given entity id.
175    pub fn all_relations_for(&self, entity_id: u32) -> Vec<&Relationship> {
176        self.relations
177            .values()
178            .filter(|r| r.from_id == entity_id || r.to_id == entity_id)
179            .collect()
180    }
181    /// Iterates over every stored pairwise relationship.
182    pub fn all_relations(&self) -> impl Iterator<Item = &Relationship> {
183        self.relations.values()
184    }
185    /// Returns the total number of stored pairwise relationships.
186    pub fn relation_count(&self) -> usize {
187        self.relations.len()
188    }
189    /// Adds a directed named link from one entity to another without duplicates.
190    pub fn add_link(&mut self, from: u32, name: &str, to: u32) {
191        let targets = self.directed.entry((from, name.to_string())).or_default();
192        if !targets.contains(&to) {
193            targets.push(to);
194        }
195    }
196    /// Returns the directed link targets for a source entity and link name.
197    pub fn get_links(&self, from: u32, name: &str) -> &[u32] {
198        self.directed
199            .get(&(from, name.to_string()))
200            .map(|v| v.as_slice())
201            .unwrap_or(&[])
202    }
203    /// Removes one directed link target from a source entity and link name.
204    pub fn remove_link(&mut self, from: u32, name: &str, to: u32) {
205        if let Some(targets) = self.directed.get_mut(&(from, name.to_string())) {
206            targets.retain(|&id| id != to);
207        }
208    }
209    /// Removes all directed link targets for a source entity and link name.
210    pub fn clear_links(&mut self, from: u32, name: &str) {
211        self.directed.remove(&(from, name.to_string()));
212    }
213    /// Returns whether a directed link target exists for the source entity and link name.
214    pub fn has_link(&self, from: u32, name: &str, to: u32) -> bool {
215        self.directed
216            .get(&(from, name.to_string()))
217            .map(|v| v.contains(&to))
218            .unwrap_or(false)
219    }
220}
