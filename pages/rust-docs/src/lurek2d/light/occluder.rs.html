1//! Convex polygon shape that blocks light and casts shadows in the 2D lighting system.
2//!
3//! - Vertex management: construction from Vec2 list or flat coordinate arrays, runtime replacement.
4//! - Per-occluder properties: world position offset, shadow opacity, light-layer bitmask, enabled toggle.
5
6use crate::math::Vec2;
7
8/// Convex polygon shape that blocks light and casts shadows in `LightWorld`.
9pub struct Occluder {
10    /// Polygon vertices in local space; must be 3..=512 elements.
11    pub vertices: Vec<Vec2>,
12    /// World-space offset applied to all vertices during shadow projection.
13    pub position: Vec2,
14    /// Shadow opacity in [0.0, 1.0]; 1.0 = fully opaque, 0.0 = transparent.
15    pub opacity: f32,
16    /// Bitmask selecting which lights this occluder casts shadows for.
17    pub light_mask: u16,
18    /// Whether this occluder participates in shadow computation; when false, it is skipped.
19    pub enabled: bool,
20}
21impl Occluder {
22    /// Create an occluder from vertices; panics if count is outside 3..=512.
23    pub fn new(vertices: Vec<Vec2>) -> Self {
24        assert!(
25            vertices.len() >= 3 && vertices.len() <= 512,
26            "Occluder vertex count must be 3..=512, got {}",
27            vertices.len()
28        );
29        Self {
30            vertices,
31            position: Vec2::ZERO,
32            opacity: 1.0,
33            light_mask: 0xFFFF,
34            enabled: true,
35        }
36    }
37    /// Replace vertices; panics if new count is outside 3..=512.
38    pub fn set_vertices(&mut self, vertices: Vec<Vec2>) {
39        assert!(
40            vertices.len() >= 3 && vertices.len() <= 512,
41            "Occluder vertex count must be 3..=512, got {}",
42            vertices.len()
43        );
44        self.vertices = vertices;
45    }
46    /// Build an occluder from a flat `[x, y, x, y, ...]` coordinate slice; returns error on invalid length.
47    pub fn from_flat_coords(flat: &[f32]) -> Result<Self, String> {
48        if flat.len() < 6 || flat.len() > 1024 || !flat.len().is_multiple_of(2) {
49            return Err(format!(
50                "vertex array must have 6..=1024 coordinates (3..=512 vertices), got {}",
51                flat.len()
52            ));
53        }
54        let verts: Vec<Vec2> = flat.chunks(2).map(|c| Vec2::new(c[0], c[1])).collect();
55        Ok(Self::new(verts))
56    }
57    /// Return the vertex slice. This function is part of the public API.
58    pub fn get_vertices(&self) -> &[Vec2] {
59        &self.vertices
60    }
61    /// Set the world-space position offset.
62    pub fn set_position(&mut self, position: Vec2) {
63        self.position = position;
64    }
65    /// Return the world-space position offset.
66    pub fn get_position(&self) -> Vec2 {
67        self.position
68    }
69    /// Set shadow opacity; expected range [0.0, 1.0].
70    pub fn set_opacity(&mut self, opacity: f32) {
71        self.opacity = opacity;
72    }
73    /// Return shadow opacity. This function is part of the public API.
74    pub fn get_opacity(&self) -> f32 {
75        self.opacity
76    }
77    /// Set the light-layer bitmask.
78    pub fn set_light_mask(&mut self, mask: u16) {
79        self.light_mask = mask;
80    }
81    /// Return the light-layer bitmask.
82    pub fn get_light_mask(&self) -> u16 {
83        self.light_mask
84    }
85    /// Enable or disable this occluder.
86    pub fn set_enabled(&mut self, enabled: bool) {
87        self.enabled = enabled;
88    }
89    /// Return whether this occluder is enabled.
90    pub fn is_enabled(&self) -> bool {
91        self.enabled
92    }
93}
