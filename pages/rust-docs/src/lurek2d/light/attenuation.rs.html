1//! Distance-based light intensity falloff using quadratic attenuation coefficients.
2//!
3//! - Computes attenuation factor from constant, linear, and quadratic terms.
4//! - Debug visualization of attenuation curves rendered to an image buffer.
5
6/// Quadratic attenuation coefficients for distance-based light intensity falloff.
7#[derive(Debug, Clone, Copy, PartialEq)]
8pub struct Attenuation {
9    /// Constant term added to the denominator; prevents infinite brightness at zero distance.
10    pub constant: f32,
11    /// Linear coefficient in the denominator; controls mid-range falloff slope.
12    pub linear: f32,
13    /// Quadratic coefficient in the denominator; controls rapid long-range decay.
14    pub quadratic: f32,
15}
16/// Core methods for constructing and evaluating attenuation.
17impl Attenuation {
18    /// Create a new attenuation with explicit constant, linear, and quadratic coefficients.
19    pub fn new(constant: f32, linear: f32, quadratic: f32) -> Self {
20        Self {
21            constant,
22            linear,
23            quadratic,
24        }
25    }
26    /// Return attenuation factor at `distance`; returns 1.0 when denominator is <= 0.
27    pub fn factor(&self, distance: f32) -> f32 {
28        let denom = self.constant + self.linear * distance + self.quadratic * distance * distance;
29        if denom <= 0.0 {
30            1.0
31        } else {
32            1.0 / denom
33        }
34    }
35    /// Render labeled attenuation curve plots for each config into an `ImageData` for debug output.
36    pub fn draw_attenuation_curves_to_image(
37        configs: &[(Attenuation, &str)],
38        max_distance: f32,
39        width: u32,
40        height: u32,
41    ) -> crate::image::ImageData {
42        let mut img = crate::image::ImageData::new(width, height);
43        img.fill(15, 15, 20, 255);
44        let count = configs.len().max(1);
45        let row_height = (height.saturating_sub(20)) / count as u32;
46        let plot_w = width.saturating_sub(20);
47        let colors: [(u8, u8, u8); 6] = [
48            (230, 80, 80),
49            (80, 230, 80),
50            (80, 80, 230),
51            (230, 230, 80),
52            (230, 80, 230),
53            (80, 230, 230),
54        ];
55        for (i, (atten, label)) in configs.iter().enumerate() {
56            let oy = 10 + i as i32 * row_height as i32;
57            let bar_max = (row_height as f32 * 0.8) as i32;
58            let (r, g, b) = colors[i % colors.len()];
59            for x in 0..plot_w {
60                let dist = x as f32 / plot_w as f32 * max_distance;
61                let factor = atten.factor(dist);
62                let bar_h = (factor * bar_max as f32) as i32;
63                if bar_h > 0 {
64                    img.draw_line(
65                        x as i32 + 10,
66                        oy + bar_max,
67                        x as i32 + 10,
68                        oy + bar_max - bar_h,
69                        r,
70                        g,
71                        b,
72                        200,
73                    );
74                }
75            }
76            img.draw_label(label, 10, oy, 200, 200, 200);
77        }
78        img.draw_label(
79            "ATTENUATION CURVES",
80            (width / 3) as i32,
81            (height - 10) as i32,
82            100,
83            255,
84            100,
85        );
86        img
87    }
88}

90/// Provides no-op attenuation (constant=1, linear=0, quadratic=0) for full-intensity lights.
91impl Default for Attenuation {
92    fn default() -> Self {
93        Self {
94            constant: 1.0,
95            linear: 0.0,
96            quadratic: 0.0,
97        }
98    }
