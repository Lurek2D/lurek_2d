1//! Scene-level container (`LightWorld`) managing all `Light2D` instances and `Occluder` shapes via slotmaps.
2//!
3//! - Add, remove, query, and bulk-update lights and occluders by stable keys.
4//! - Group operations: enable/disable, intensity, and color changes on a named group ID.
5//! - Flicker system: lazy-indexed advance loop that only touches lights with active flicker state.
6//! - Renderer hints: ambient color array, directional light tuples, and normal-map snapshot extraction.
7//! - Debug visualization: rasterize an approximate light-map preview into an `ImageData` bitmap.
8
9use crate::light::light2d::Light2D;
10use crate::light::light_type::LightType;
11use crate::light::occluder::Occluder;
12use crate::log_msg;
13use crate::color::Color;
14use crate::runtime::log_messages::{LW01_LIGHT_WORLD_INIT, LW02_LIGHT_ADD};
15use crate::runtime::resource_keys::{LightKey, OccluderKey};
16use slotmap::SlotMap;
17
18/// Scene-level container for all `Light2D` instances and `Occluder` shapes.
19pub struct LightWorld {
20    /// Slotmap of all registered lights, keyed by `LightKey`.
21    pub lights: SlotMap<LightKey, Light2D>,
22    /// Slotmap of all registered occluder shapes, keyed by `OccluderKey`.
23    pub occluders: SlotMap<OccluderKey, Occluder>,
24    /// Scene ambient base color added to all illuminated pixels.
25    pub ambient: Color,
26    /// Whether any light processing should run; set to `true` on first `add_light`.
27    pub enabled: bool,
28    /// Maximum number of active lights evaluated per frame by the renderer.
29    pub max_lights: u16,
30    /// Cached list of keys for lights that have flicker enabled; rebuilt when `flicker_index_dirty`.
31    flicker_keys: Vec<LightKey>,
32    /// True when the flicker index is stale and must be rebuilt before next advance.
33    flicker_index_dirty: bool,
34}
35
36/// Snapshot of a single light's normal-map binding used by the renderer for surface shading.
37#[derive(Debug, Clone)]
38pub struct NormalMapLightHint {
39    /// World-space X of the contributing light.
40    pub x: f32,
41    /// World-space Y of the contributing light.
42    pub y: f32,
43    /// Effective radius of the contributing light.
44    pub radius: f32,
45    /// Intensity of the contributing light.
46    pub intensity: f32,
47    /// Direction angle in radians for spot lights.
48    pub direction: f32,
49    /// Path to the normal map texture asset.
50    pub path: String,
51    /// Normal map contribution strength in [0.0, 1.0].
52    pub strength: f32,
53}
54impl LightWorld {
55    /// Create an empty world with ambient=0.1, disabled, and max_lights=64.
56    pub fn new() -> Self {
57        log_msg!(trace, LW01_LIGHT_WORLD_INIT);
58        Self {
59            lights: SlotMap::with_key(),
60            occluders: SlotMap::with_key(),
61            ambient: Color::new(0.1, 0.1, 0.1, 1.0),
62            enabled: false,
63            max_lights: 64,
64            flicker_keys: Vec::new(),
65            flicker_index_dirty: true,
66        }
67    }
68    /// Insert a light, enable the world if it was disabled, and return its key.
69    pub fn add_light(&mut self, light: Light2D) -> LightKey {
70        log_msg!(debug, LW02_LIGHT_ADD);
71        if !self.enabled {
72            self.enabled = true;
73        }
74        let key = self.lights.insert(light);
75        self.flicker_index_dirty = true;
76        key
77    }
78    /// Insert an occluder and return its key.
79    pub fn add_occluder(&mut self, occluder: Occluder) -> OccluderKey {
80        self.occluders.insert(occluder)
81    }
82    /// Remove a light by key and evict it from the flicker index; returns the removed light or `None`.
83    pub fn remove_light(&mut self, key: LightKey) -> Option<Light2D> {
84        self.flicker_keys.retain(|k| *k != key);
85        self.lights.remove(key)
86    }
87    /// Remove an occluder by key; returns the removed occluder or `None`.
88    pub fn remove_occluder(&mut self, key: OccluderKey) -> Option<Occluder> {
89        self.occluders.remove(key)
90    }
91    /// Return a shared reference to the light at `key`, or `None` if not present.
92    pub fn get_light(&self, key: LightKey) -> Option<&Light2D> {
93        self.lights.get(key)
94    }
95    /// Return a mutable reference to the light at `key`, or `None` if not present.
96    pub fn get_light_mut(&mut self, key: LightKey) -> Option<&mut Light2D> {
97        self.lights.get_mut(key)
98    }
99    /// Return a shared reference to the occluder at `key`, or `None` if not present.
100    pub fn get_occluder(&self, key: OccluderKey) -> Option<&Occluder> {
101        self.occluders.get(key)
102    }
103    /// Return a mutable reference to the occluder at `key`, or `None` if not present.
104    pub fn get_occluder_mut(&mut self, key: OccluderKey) -> Option<&mut Occluder> {
105        self.occluders.get_mut(key)
106    }
107    /// Return the number of registered lights.
108    pub fn light_count(&self) -> usize {
109        self.lights.len()
110    }
111    /// Return the number of registered occluders.
112    pub fn occluder_count(&self) -> usize {
113        self.occluders.len()
114    }
115    /// Remove all lights and occluders and reset ambient to 0.1 gray.
116    pub fn clear(&mut self) {
117        self.lights.clear();
118        self.occluders.clear();
119        self.ambient = Color::new(0.1, 0.1, 0.1, 1.0);
120        self.flicker_keys.clear();
121        self.flicker_index_dirty = false;
122    }
123    /// Return `true` if any registered light has `enabled = true`.
124    pub fn has_active_lights(&self) -> bool {
125        self.lights.values().any(|l| l.enabled)
126    }
127    /// Set `enabled` on all lights in `group_id`.
128    pub fn set_group_enabled(&mut self, group_id: u16, enabled: bool) {
129        for light in self.lights.values_mut() {
130            if light.group_id == group_id {
131                light.enabled = enabled;
132            }
133        }
134    }
135    /// Set `intensity` on all lights in `group_id`.
136    pub fn set_group_intensity(&mut self, group_id: u16, intensity: f32) {
137        for light in self.lights.values_mut() {
138            if light.group_id == group_id {
139                light.intensity = intensity;
140            }
141        }
142    }
143    /// Set `color` on all lights in `group_id`.
144    pub fn set_group_color(&mut self, group_id: u16, color: Color) {
145        for light in self.lights.values_mut() {
146            if light.group_id == group_id {
147                light.color = color;
148            }
149        }
150    }
151    /// Return the count of lights in `group_id`.
152    pub fn group_count(&self, group_id: u16) -> usize {
153        self.lights
154            .values()
155            .filter(|l| l.group_id == group_id)
156            .count()
157    }
158    /// Advance all flickering lights by `dt` seconds; rebuilds the flicker index if stale.
159    pub fn advance_flickers(&mut self, dt: f32) {
160        if self.flicker_index_dirty {
161            self.reindex_flickers();
162        }
163        let mut stale = false;
164        for key in self.flicker_keys.iter().copied() {
165            if let Some(light) = self.lights.get_mut(key) {
166                if light.flicker.enabled {
167                    light.flicker.advance(dt);
168                }
169            } else {
170                stale = true;
171            }
172        }
173        if stale {
174            self.flicker_keys.retain(|k| self.lights.contains_key(*k));
175        }
176    }
177    /// Rebuild the flicker key index from all lights that have flicker enabled.
178    pub fn reindex_flickers(&mut self) {
179        self.flicker_keys.clear();
180        for (key, light) in self.lights.iter() {
181            if light.flicker.enabled {
182                self.flicker_keys.push(key);
183            }
184        }
185        self.flicker_index_dirty = false;
186    }
187    /// Render an approximate light-map preview of this world into an `ImageData` debug image.
188    pub fn draw_to_image(&self, width: u32, height: u32) -> crate::image::ImageData {
189        let mut img = crate::image::ImageData::new(width, height);
190        img.fill(10, 10, 15, 255);
191        let light_params: Vec<(f32, f32, f32, f32, f32, f32)> = self
192            .lights
193            .values()
194            .map(|l| (l.x, l.y, l.radius, l.color.r, l.color.g, l.color.b))
195            .collect();
196        for y in 0..height {
197            for x in 0..width {
198                let mut fr = 10.0f32;
199                let mut fg = 10.0f32;
200                let mut fb = 15.0f32;
201                for &(lx, ly, radius, lr, lg, lb) in &light_params {
202                    let dx = x as f32 - lx;
203                    let dy = y as f32 - ly;
204                    let dist = (dx * dx + dy * dy).sqrt();
205                    let atten = (1.0 - dist / radius).max(0.0);
206                    let atten = atten * atten;
207                    fr += lr * atten * 200.0;
208                    fg += lg * atten * 200.0;
209                    fb += lb * atten * 200.0;
210                }
211                img.set_pixel(
212                    x,
213                    y,
214                    fr.min(255.0) as u8,
215                    fg.min(255.0) as u8,
216                    fb.min(255.0) as u8,
217                    255,
218                );
219            }
220        }
221        for occ in self.occluders.values() {
222            let verts = &occ.vertices;
223            if verts.len() >= 2 {
224                let mut min_x = f32::MAX;
225                let mut min_y = f32::MAX;
226                let mut max_x = f32::MIN;
227                let mut max_y = f32::MIN;
228                for v in verts {
229                    min_x = min_x.min(v.x);
230                    min_y = min_y.min(v.y);
231                    max_x = max_x.max(v.x);
232                    max_y = max_y.max(v.y);
233                }
234                img.draw_rect(
235                    min_x as i32,
236                    min_y as i32,
237                    (max_x - min_x) as u32,
238                    (max_y - min_y) as u32,
239                    20,
240                    20,
241                    25,
242                    255,
243                );
244            }
245        }
246        for l in self.lights.values() {
247            img.draw_circle(l.x as i32, l.y as i32, 5, 255, 240, 100, 255);
248        }
249        img
250    }
251    /// Return ambient color as an RGBA `[f32; 4]` array for shader upload.
252    pub fn ambient_color_hint(&self) -> [f32; 4] {
253        [
254            self.ambient.r,
255            self.ambient.g,
256            self.ambient.b,
257            self.ambient.a,
258        ]
259    }
260    /// Return `(x, y, direction)` tuples for all enabled directional lights.
261    pub fn directional_light_hints(&self) -> Vec<(f32, f32, f32)> {
262        self.lights
263            .values()
264            .filter(|l| l.enabled && l.light_type == LightType::Directional)
265            .map(|l| (l.x, l.y, l.direction))
266            .collect()
267    }
268    /// Return `NormalMapLightHint` snapshots for all enabled lights that have a normal map path.
269    pub fn normal_map_light_hints(&self) -> Vec<NormalMapLightHint> {
270        self.lights
271            .values()
272            .filter(|l| l.enabled)
273            .filter_map(|l| {
274                l.get_normal_map_path().map(|path| NormalMapLightHint {
275                    x: l.x,
276                    y: l.y,
277                    radius: l.radius,
278                    intensity: l.intensity,
279                    direction: l.direction,
280                    path: path.to_string(),
281                    strength: l.normal_strength,
282                })
283            })
284            .collect()
285    }
286}
