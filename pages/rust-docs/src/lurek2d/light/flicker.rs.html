1//! Sine-based flicker configuration that modulates light intensity over time.
2//!
3//! - Phase accumulates each frame and wraps at TAU for continuous oscillation.
4//! - Strength controls peak deviation from base intensity; speed sets radians per second.
5//! - Disabled by default; enable to animate torches, candles, or neon lights.
6
7/// Sine-based flicker config that modulates a light's intensity by a small oscillating factor.
8#[derive(Debug, Clone, Copy, PartialEq)]
9pub struct FlickerConfig {
10    /// Whether flicker is active; when false, `multiplier()` always returns 1.0.
11    pub enabled: bool,
12    /// Oscillation frequency in radians per second.
13    pub speed: f32,
14    /// Peak deviation from 1.0; e.g. 0.15 gives range [0.85, 1.15].
15    pub strength: f32,
16    /// Current phase in radians, advanced by `advance`.
17    pub phase: f32,
18}
19impl FlickerConfig {
20    /// Create an enabled flicker with given speed and strength; phase starts at 0.
21    pub fn new(speed: f32, strength: f32) -> Self {
22        Self {
23            enabled: true,
24            speed,
25            strength,
26            phase: 0.0,
27        }
28    }
29    /// Return the current intensity multiplier; returns 1.0 when disabled.
30    pub fn multiplier(&self) -> f32 {
31        if self.enabled {
32            1.0 + self.strength * self.phase.sin()
33        } else {
34            1.0
35        }
36    }
37    /// Advance the flicker phase by `dt` seconds, wrapping at TAU.
38    pub fn advance(&mut self, dt: f32) {
39        if self.enabled {
40            self.phase += self.speed * dt;
41            self.phase %= std::f32::consts::TAU;
42        }
43    }
44}
45
46/// Provides a disabled flicker with default speed=8.0 and strength=0.15.
47impl Default for FlickerConfig {
48    fn default() -> Self {
49        Self {
50            enabled: false,
51            speed: 8.0,
52            strength: 0.15,
53            phase: 0.0,
54        }
55    }
56}
