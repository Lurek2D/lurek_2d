10/// Reads an RGBA byte color from a Lua array table, defaulting missing channels to 255.
11fn parse_color_table(tbl: LuaTable) -> LuaResult<[u8; 4]> {
12    let r: u8 = tbl.get(1).unwrap_or(255);
13    let g: u8 = tbl.get(2).unwrap_or(255);
14    let b: u8 = tbl.get(3).unwrap_or(255);
15    let a: u8 = tbl.get(4).unwrap_or(255);
16    Ok([r, g, b, a])
17}
18/// Resolves a `LuaImage` icon handle and display size for minimap object and marker textures.
19fn parse_lua_image_icon(
20    image_ud: LuaAnyUserData,
21    default_width: Option<f32>,
22    default_height: Option<f32>,
23    method_name: &str,
24) -> LuaResult<(
25    crate::runtime::resource_keys::TextureKey,
26    f32,
27    f32,
28    f32,
29    f32,
30)> {
31    let image = image_ud.borrow::<LuaImage>().map_err(|_| {
32        LuaError::RuntimeError(format!(
33            "lurek.minimap: {} expects an LImage from lurek.render.newImage()",
34            method_name
35        ))
36    })?;
37    let (texture_width, texture_height) = {
38        let state = image.state.borrow();
39        let Some(texture) = state.textures.get(image.key) else {
40            return Err(LuaError::RuntimeError(format!(
41                "lurek.minimap: {} received an image whose texture is no longer resident",
42                method_name
43            )));
44        };
45        (texture.width as f32, texture.height as f32)
46    };
47    Ok((
48        image.key,
49        texture_width,
50        texture_height,
51        default_width.unwrap_or(texture_width),
52        default_height.unwrap_or(texture_height),
53    ))
54}

568        /// @param | a | number? | Alpha channel, defaults to 0.8.
569        methods.add_method_mut(
570            "setViewportColor",
571            |_, this, (r, g, b, a): (f32, f32, f32, Option<f32>)| {
572                this.inner.set_viewport_color([r, g, b, a.unwrap_or(0.8)]);
573                Ok(())
574            },
575        );
576        // -- getViewportColor --
577        /// Returns the viewport rectangle color.
578        /// @return | number | Red channel.
579        /// @return | number | Green channel.
580        /// @return | number | Blue channel.
581        /// @return | number | Alpha channel.
582        methods.add_method("getViewportColor", |_, this, ()| {
583            let c = this.inner.viewport_color();
584            Ok((c[0], c[1], c[2], c[3]))
585        });
586        #[allow(clippy::type_complexity)]
587        // -- addPing --
588        /// Adds a timed ping effect at a minimap world position.
589        /// @param | x | number | World x coordinate of the ping.
590        /// @param | y | number | World y coordinate of the ping.
591        /// @param | duration | number | Duration in seconds before the ping fades out.
592        /// @param | r | number? | Red channel, defaults to 1.0.
593        /// @param | g | number? | Green channel, defaults to 1.0.
594        /// @param | b | number? | Blue channel, defaults to 0.0.
595        /// @param | a | number? | Alpha channel, defaults to 1.0.
596        methods.add_method_mut(
597            "addPing",
598            |_,
599             this,
600             (x, y, duration, r, g, b, a): (
601                f32,
602                f32,
603                f32,
604                Option<f32>,
605                Option<f32>,
606                Option<f32>,
607                Option<f32>,
608            )| {
609                let color = [
610                    r.unwrap_or(1.0),
611                    g.unwrap_or(1.0),
612                    b.unwrap_or(0.0),
613                    a.unwrap_or(1.0),
614                ];
615                this.inner.add_ping(x, y, duration, color);
616                Ok(())
617            },
618        );
619        // -- getPingCount --
620        /// Returns the number of active pings.
621        /// @return | integer | Ping count.
622        methods.add_method("getPingCount", |_, this, ()| Ok(this.inner.ping_count()));
623        #[allow(clippy::type_complexity)]
624        // -- addMarker --
625        /// Adds a world-space marker and returns its unique id.
626        /// @param | x | number | Marker x coordinate.
627        /// @param | y | number | Marker y coordinate.
628        /// @param | desc | string? | Marker description.
629        /// @param | r | number? | Red channel override, defaults to 1.0.
630        /// @param | g | number? | Green channel override, defaults to 0.0.
631        /// @param | b | number? | Blue channel override, defaults to 0.0.
632        /// @param | a | number? | Alpha channel override, defaults to 1.0.
633        /// @return | integer | Marker id.
634        methods.add_method_mut(
635            "addMarker",
636            |_,
637             this,
638             (x, y, desc, r, g, b, a): (
639                f32,
640                f32,
641                Option<String>,
642                Option<f32>,
643                Option<f32>,
644                Option<f32>,
645                Option<f32>,
646            )| {
647                let color = [
648                    r.unwrap_or(1.0),
649                    g.unwrap_or(0.0),
650                    b.unwrap_or(0.0),
651                    a.unwrap_or(1.0),
652                ];
653                let id = this.inner.add_marker(x, y, desc.unwrap_or_default(), color);
654                Ok(id)
655            },
656        );
657        // -- removeMarker --
658        /// Removes a minimap marker by its unique id.
659        /// @param | id | integer | Marker id.
660        /// @return | boolean | True when a marker was removed.
661        methods.add_method_mut("removeMarker", |_, this, id: u32| {
662            Ok(this.inner.remove_marker(id))
663        });
664        // -- hasMarker --
665        /// Returns whether a marker id exists.
666        /// @param | id | integer | Marker id.
667        /// @return | boolean | True when the marker exists.
668        methods.add_method("hasMarker", |_, this, id: u32| {
669            Ok(this.inner.has_marker(id))
670        });
671        // -- getMarkerDescription --
672        /// Returns a marker description by id.
673        /// @param | id | integer | Marker id.
674        /// @return | string | Marker description, or nil when missing.
675        methods.add_method("getMarkerDescription", |_, this, id: u32| {
676            Ok(this.inner.get_marker_description(id).map(|s| s.to_string()))
677        });
678        // -- getMarkerCount --
679        /// Returns the total number of minimap markers.
680        /// @return | integer | Marker count.
681        methods.add_method("getMarkerCount", |_, this, ()| {
682            Ok(this.inner.marker_count())
683        });
684        // -- setMarkerTexture --
685        /// Assigns an image texture to a marker.
686        /// @param | id | integer | Marker id.
687        /// @param | image_ud | LImage | Image handle from `lurek.render.newImage`.
688        /// @param | width | number? | Display width override.
689        /// @param | height | number? | Display height override.
690        methods.add_method_mut(
691            "setMarkerTexture",
692            |_, this, (id, image_ud, width, height): (u32, LuaAnyUserData, Option<f32>, Option<f32>)| {
693                let (texture_key, tex_w, tex_h, display_w, display_h) =
694                    parse_lua_image_icon(image_ud, width, height, "setMarkerTexture")?;
695                this.inner.set_marker_texture(id, texture_key, tex_w, tex_h, display_w, display_h);
696                Ok(())
697            },
698        );
699        // -- clearMarkerTexture --
700        /// Clears image texture from a marker.
701        /// @param | id | integer | Marker id.
702        methods.add_method_mut("clearMarkerTexture", |_, this, id: u32| {
703            this.inner.clear_marker_texture(id);
704            Ok(())
705        });
706        // -- setMarkerAnimation --
707        /// Sets marker animation by type name.
708        /// @param | id | integer | Marker id.
709        /// @param | anim_type | string | Animation type: `blink`, `pulse`, or `rotate`.
710        /// @param | speed | number | Animation speed.
711        methods.add_method_mut(
712            "setMarkerAnimation",
713            |_, this, (id, anim_type, speed): (u32, String, f32)| {
714                let anim = match anim_type.as_str() {
715                    "blink" => MarkerAnimation::Blink { speed, phase: 0.0 },
716                    "pulse" => MarkerAnimation::Pulse { speed, phase: 0.0 },
717                    "rotate" => MarkerAnimation::Rotate { speed, angle: 0.0 },
718                    other => {
719                        return Err(LuaError::RuntimeError(format!(
720                            "lurek.minimap: unknown animation type '{}', \
721                             expected 'blink', 'pulse', or 'rotate'",
722                            other
723                        )))
724                    }
725                };
726                this.inner.set_marker_animation(id, anim);
727                Ok(())
728            },
729        );
730        // -- clearMarkerAnimation --
731        /// Clears the animation assigned to a marker by id.
732        /// @param | id | integer | Marker id.
733        methods.add_method_mut("clearMarkerAnimation", |_, this, id: u32| {
734            this.inner.clear_marker_animation(id);
735            Ok(())
736        });
737        // -- drawLine --
738        /// Adds an overlay line between two world-space points.
739        /// @param | x1 | number | Start x coordinate.
740        /// @param | y1 | number | Start y coordinate.
741        /// @param | x2 | number | End x coordinate.
742        /// @param | y2 | number | End y coordinate.
743        /// @param | color_tbl | table | RGBA byte color table.
744        methods.add_method_mut(
745            "drawLine",
746            |_, this, (x1, y1, x2, y2, color_tbl): (f32, f32, f32, f32, LuaTable)| {
747                let color = parse_color_table(color_tbl)?;
748                this.inner.draw_line(x1, y1, x2, y2, color);
749                Ok(())
750            },
751        );
752        // -- drawRect --
753        /// Adds an overlay rectangle at a world-space position.
754        /// @param | x | number | Rectangle x coordinate.
755        /// @param | y | number | Rectangle y coordinate.
756        /// @param | w | number | Rectangle width.
757        /// @param | h | number | Rectangle height.
758        /// @param | color_tbl | table | RGBA byte color table.
759        methods.add_method_mut(
760            "drawRect",
761            |_, this, (x, y, w, h, color_tbl): (f32, f32, f32, f32, LuaTable)| {
762                let color = parse_color_table(color_tbl)?;
763                this.inner.draw_rect(x, y, w, h, color);
764                Ok(())
765            },
766        );
767        // -- clearOverlay --
768        /// Clears all minimap overlay shapes.
769        methods.add_method_mut("clearOverlay", |_, this, ()| {
770            this.inner.clear_overlay();
771            Ok(())
772        });
773        // -- getOverlayShapeCount --
774        /// Returns the number of overlay shapes.
775        /// @return | integer | Overlay shape count.
776        methods.add_method("getOverlayShapeCount", |_, this, ()| {
777            Ok(this.inner.overlay_shapes().len())
778        });
779        // -- showPath --
780        /// Adds a colored path overlay and returns its id.
781        /// @param | points_tbl | table | Array table of point arrays `{x, y}`.
782        /// @param | color_tbl | table | RGBA byte color table.
783        /// @return | integer | Path id.
784        methods.add_method_mut(
785            "showPath",
786            |_, this, (points_tbl, color_tbl): (LuaTable, LuaTable)| {
787                let color = parse_color_table(color_tbl)?;
788                let len = points_tbl.len()? as usize;
789                let mut points = Vec::with_capacity(len);
790                for i in 1..=len {
791                    let pt: LuaTable = points_tbl.get(i)?;
792                    let x: f32 = pt.get(1)?;
793                    let y: f32 = pt.get(2)?;
794                    points.push((x, y));
795                }
796                let id = this.inner.show_path(points, color);
797                Ok(id)
798            },
799        );
800        // -- clearPath --
801        /// Clears one path by id or all paths when no id is provided.
802        /// @param | id | integer? | Path id to clear.
803        methods.add_method_mut("clearPath", |_, this, id: Option<u32>| {
804            this.inner.clear_path(id);
805            Ok(())
806        });
807        // -- getPathCount --
808        /// Returns the number of active path overlays.
809        /// @return | integer | Path count.
810        methods.add_method("getPathCount", |_, this, ()| Ok(this.inner.paths().len()));
811        // -- setLayer --
812        /// Sets the active minimap display layer index.
813        /// @param | layer | integer | Layer index.
814        methods.add_method_mut("setLayer", |_, this, layer: usize| {
815            this.inner.set_layer(layer);
816            Ok(())
817        });
818        // -- getLayer --
819        /// Returns the active minimap display layer index.
820        /// @return | integer | Layer index.
821        methods.add_method("getLayer", |_, this, ()| Ok(this.inner.get_layer()));
822        // -- getLayerCount --
823        /// Returns the number of minimap layers.
824        /// @return | integer | Layer count.
825        methods.add_method("getLayerCount", |_, this, ()| Ok(this.inner.layer_count()));
826        // -- setLayerData --
827        /// Sets raw cell data for a minimap layer.
828        /// @param | layer | integer | Layer index.
829        /// @param | data_tbl | table | Array table of cell bytes.
830        methods.add_method_mut(
831            "setLayerData",
832            |_, this, (layer, data_tbl): (usize, LuaTable)| {
833                let len = data_tbl.len()? as usize;
834                let mut cells = Vec::with_capacity(len);
835                for i in 1..=len {
836                    let v: u8 = data_tbl.get(i)?;
837                    cells.push(v);
838                }
839                let w = this.inner.grid_width();
840                let h = this.inner.grid_height();
841                this.inner.set_layer_data(
842                    layer,
843                    LayerData {
844                        cells,
845                        width: w,
846                        height: h,
847                    },
848                );
849                Ok(())
850            },
851        );
852        // -- getLayerData --
853        /// Returns raw cell data for a minimap layer.
854        /// @param | layer | integer | Layer index.
855        /// @return | integer[] | Array table of cell bytes, or nil when missing.
856        methods.add_method("getLayerData", |lua, this, layer: usize| {
857            let Some(layer_data) = this.inner.layer_data(layer) else {
858                return Ok(None::<LuaTable>);
859            };
860            let tbl = lua.create_table()?;
861            for (index, cell) in layer_data.cells.iter().enumerate() {
862                tbl.set(index + 1, *cell)?;
863            }
864            Ok(Some(tbl))
865        });
866        // -- setAntiAlias --
867        /// Enables or disables minimap anti-aliasing.
868        /// @param | enabled | boolean | Anti-alias flag.
869        methods.add_method_mut("setAntiAlias", |_, this, enabled: bool| {
870            this.inner.set_anti_alias(enabled);
871            Ok(())
872        });
873        // -- isAntiAlias --
874        /// Returns whether anti-aliasing is enabled.
875        /// @return | boolean | True when enabled.
876        methods.add_method("isAntiAlias", |_, this, ()| Ok(this.inner.anti_alias()));
877        // -- setClickable --
878        /// Enables or disables minimap click handling.
879        /// @param | enabled | boolean | Clickable flag.
880        methods.add_method_mut("setClickable", |_, this, enabled: bool| {
881            this.inner.set_clickable(enabled);
882            Ok(())
883        });
884        // -- isClickable --
885        /// Returns whether minimap click handling is enabled.
886        /// @return | boolean | True when clickable.
887        methods.add_method("isClickable", |_, this, ()| Ok(this.inner.is_clickable()));
888        // -- getHoverInfo --
889        /// Returns hover text for a screen position when available.
890        /// @param | sx | number | Screen x coordinate.
891        /// @param | sy | number | Screen y coordinate.
892        /// @param | mx | number | Minimap x position.
893        /// @param | my | number | Minimap y position.
894        /// @return | string | Hover info text, or nil when unavailable.
895        methods.add_method(
896            "getHoverInfo",
897            |_, this, (sx, sy, mx, my): (f32, f32, f32, f32)| {
898                Ok(this
899                    .inner
900                    .get_hover_info(sx, sy, mx, my)
901                    .map(|s| s.to_string()))
902            },
903        );
904        // -- screenToGrid --
905        /// Converts a screen position to grid coordinates.
906        /// @param | sx | number | Screen x coordinate.
907        /// @param | sy | number | Screen y coordinate.
908        /// @param | mx | number | Minimap x position.
909        /// @param | my | number | Minimap y position.
910        /// @return | number | Grid x coordinate.
911        /// @return | number | Grid y coordinate.
912        methods.add_method(
913            "screenToGrid",
914            |_, this, (sx, sy, mx, my): (f32, f32, f32, f32)| {
915                Ok(this.inner.screen_to_grid(sx, sy, mx, my))
916            },
917        );
918        // -- gridToScreen --
919        /// Converts grid coordinates to screen coordinates.
920        /// @param | gx | number | Grid x coordinate.
921        /// @param | gy | number | Grid y coordinate.
922        /// @param | mx | number | Minimap x position.
923        /// @param | my | number | Minimap y position.
924        /// @return | number | Screen x coordinate.
925        /// @return | number | Screen y coordinate.
926        methods.add_method(
927            "gridToScreen",
928            |_, this, (gx, gy, mx, my): (f32, f32, f32, f32)| {
929                Ok(this.inner.grid_to_screen(gx, gy, mx, my))
930            },
931        );
932        // -- update --
933        /// Advances minimap animations and timers.
934        /// @param | dt | number | Delta time in seconds.
935        methods.add_method_mut("update", |_, this, dt: f32| {
936            this.inner.update(dt);
937            Ok(())
938        });
939        // -- type --
940        /// Returns the Lua-visible type name for this minimap handle.
941        /// @return | string | The string `LMinimap`.
942        methods.add_method("type", |_, _, ()| Ok("LMinimap"));
943        // -- typeOf --
944        /// Returns whether this minimap handle matches a supported type name.
945        /// @param | name | string | Type name to compare against `LMinimap`, `Minimap`, and `Object`.
946        /// @return | boolean | True when the supplied type name matches this handle.
947        methods.add_method("typeOf", |_, _, name: String| {
948            Ok(name == "LMinimap" || name == "LObject")
949        });
950        // -- render --
951        /// Enqueues minimap render commands at an optional screen position.
952        /// @param | x | number? | Screen x coordinate, defaults to 0.
953        /// @param | y | number? | Screen y coordinate, defaults to 0.
954        methods.add_method("render", |_, this, (x, y): (Option<f32>, Option<f32>)| {
955            let sx = x.unwrap_or(0.0);
956            let sy = y.unwrap_or(0.0);
957            let cmds = this.inner.build_render_commands(sx, sy);
958            this.state.borrow_mut().render_commands.extend(cmds);
959            Ok(())
960        });
961        // -- drawToImage --
962        /// Draws the minimap into image data at a pixel size.
963        /// @param | pixel_size | integer | Pixel size scale.
964        /// @return | LImageData | Image data containing the rendered minimap.
965        methods.add_method("drawToImage", |_, this, pixel_size: u32| {
966            let img = this.inner.draw_to_image(pixel_size);
967            Ok(img)
968        });
969    }
970}
971/// Registers the `lurek.minimap` module.
972pub fn register(lua: &Lua, lurek: &LuaTable, state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
973    let tbl = lua.create_table()?;
974    let s = state.clone();
975    // -- newMinimap --
976    /// Creates a minimap with grid dimensions and optional display size.
977    /// @param | grid_w | integer | Grid width in cells.
978    /// @param | grid_h | integer | Grid height in cells.
979    /// @param | display_w | integer? | Display width in pixels, defaults to 200.
980    /// @param | display_h | integer? | Display height in pixels, defaults to 200.
981    /// @return | LMinimap | New minimap handle.
982    tbl.set("newMinimap", lua.create_function(
983            move |_, (grid_w, grid_h, display_w, display_h): (u32, u32, Option<u32>, Option<u32>)| {
984                let dw = display_w.unwrap_or(200);
985                let dh = display_h.unwrap_or(200);
986                Ok(LuaMinimap {
987                    inner: Minimap::new(grid_w, grid_h, dw, dh),
988                    state: s.clone(),
989                })
990            },
991        )?,
992    )?;
993    /// Performs the 'minimap' operation.
994    lurek.set("minimap", tbl)?;
995    Ok(())
996}
