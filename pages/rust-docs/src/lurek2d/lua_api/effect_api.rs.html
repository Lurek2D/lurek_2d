13/// Lua-side handle for a single post-processing effect instance.

18impl LuaPostFxEffect {
19    /// Wraps an owned post-processing effect in shared Lua state.
20    fn from_owned(e: PostFxEffect) -> Self {
21        Self {
22            inner: Rc::new(RefCell::new(e)),

24    }

27        Self { inner: rc }
28    }

199/// Lua-side handle for an ordered post-processing stack.
200pub struct LuaPostFxStack {
201    /// Core stack dimensions, enabled flags, and pass ordering.
202    inner: PostFxStack,
203    /// Shared Lua effect handles used by stack pass entries.
204    effects: Vec<Rc<RefCell<PostFxEffect>>>,
205    /// Unique stack id referenced by renderer commands.
206    stack_id: u64,
207    /// Shared runtime state that receives renderer post-effect commands.
208    state: Rc<RefCell<SharedState>>,
209    /// Feedback blend factor clamped to the range 0.0..=1.0.
210    feedback_factor: f32,
211}
212/// Provides Lua methods for editing post-processing stack order, capture, and renderer submission.
213impl LuaUserData for LuaPostFxStack {
214    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
215        // -- add --
216        /// Appends an effect to the end of this stack.
217        /// @param | effect_ud | LPostFxEffect | Effect handle to append.
218        methods.add_method_mut("add", |_, this, effect_ud: LuaAnyUserData| {
219            let effect = effect_ud.borrow::<LuaPostFxEffect>()?;
220            this.effects.push(Rc::clone(&effect.inner));
221            let idx = this.effects.len() - 1;
222            this.inner.add(idx);
223            Ok(())
224        });
225        // -- remove --
226        /// Removes the first matching effect handle from this stack.
227        /// @param | effect_ud | LPostFxEffect | Effect handle to remove.
228        /// @return | boolean | True when the effect was found and removed.
229        methods.add_method_mut("remove", |_, this, effect_ud: LuaAnyUserData| {
230            let effect = effect_ud.borrow::<LuaPostFxEffect>()?;
231            let ptr = Rc::as_ptr(&effect.inner);
232            if let Some(pos) = this.effects.iter().position(|e| Rc::as_ptr(e) == ptr) {
233                this.effects.remove(pos);
234                if pos < this.inner.effects.len() {
235                    this.inner.effects.remove(pos);
236                    this.inner.enabled.remove(pos);
237                }
238                Ok(true)
239            } else {
240                Ok(false)
241            }
242        });
243        // -- insert --
244        /// Inserts an effect at a one-based stack position.
245        /// @param | position | integer | One-based insertion position, clamped to the stack length.
246        /// @param | effect_ud | LPostFxEffect | Effect handle to insert.
247        methods.add_method_mut(
248            "insert",
249            |_, this, (position, effect_ud): (usize, LuaAnyUserData)| {
250                let effect = effect_ud.borrow::<LuaPostFxEffect>()?;
251                let idx = (position.saturating_sub(1)).min(this.effects.len());
252                this.effects.insert(idx, Rc::clone(&effect.inner));
253                this.inner.effects.insert(idx, idx);
254                this.inner.enabled.insert(idx, true);
255                Ok(())
256            },
257        );
258        // -- setEnabled --
259        /// Enables or disables the effect pass at a one-based stack position.
260        /// @param | position | integer | One-based stack position.
261        /// @param | enabled | boolean | New enabled flag for the pass.
262        methods.add_method_mut(
263            "setEnabled",
264            |_, this, (position, enabled): (usize, bool)| {
265                let idx = position.saturating_sub(1);
266                if idx < this.inner.enabled.len() {
267                    this.inner.enabled[idx] = enabled;
268                }
269                Ok(())
270            },
271        );
272        // -- isEnabled --
273        /// Returns whether the effect pass at a one-based position is enabled.
274        /// @param | position | integer | One-based stack position.
275        /// @return | boolean | True when the pass is enabled; false for out-of-range positions.
276        methods.add_method("isEnabled", |_, this, position: usize| {
277            let idx = position.saturating_sub(1);
278            Ok(this.inner.enabled.get(idx).copied().unwrap_or(false))
279        });
280        // -- getEffectCount --
281        /// Returns the number of effect handles in this stack.
282        /// @return | integer | Effect count.
283        methods.add_method("getEffectCount", |_, this, ()| Ok(this.effects.len()));
284        // -- getEffect --
285        /// Returns the effect handle at a one-based position.
286        /// @param | index | integer | One-based stack position.
287        /// @return | LuaValue | `LPostFxEffect` handle, or nil when the index is out of range.
288        methods.add_method("getEffect", |lua, this, index: usize| {
289            let idx = index.saturating_sub(1);
290            match this.effects.get(idx) {
291                Some(rc) => Ok(LuaValue::UserData(
292                    lua.create_userdata(LuaPostFxEffect::from_rc(Rc::clone(rc)))?,
293                )),
294                None => Ok(LuaValue::Nil),
295            }
296        });
297        // -- getEnabledEffects --
298        /// Returns effect handles whose stack passes are enabled.
299        /// @return | LPostFxEffect[] | Enabled `LPostFxEffect` handles.
300        methods.add_method("getEnabledEffects", |lua, this, ()| {
301            let t = lua.create_table()?;
302            let mut count = 1;
303            for (i, rc) in this.effects.iter().enumerate() {
304                if this.inner.enabled.get(i).copied().unwrap_or(true) {
305                    t.set(
306                        count,
307                        lua.create_userdata(LuaPostFxEffect::from_rc(Rc::clone(rc)))?,
308                    )?;
309                    count += 1;
310                }
311            }
312            Ok(t)
313        });
314        // -- getWidth --
315        /// Returns the stack render width. This method is available to Lua scripts.
316        /// @return | integer | Stack width in pixels.
317        methods.add_method("getWidth", |_, this, ()| Ok(this.inner.get_width()));
318        // -- getHeight --
319        /// Returns the stack render height. This method is available to Lua scripts.
320        /// @return | integer | Stack height in pixels.
321        methods.add_method("getHeight", |_, this, ()| Ok(this.inner.get_height()));
322        // -- getDimensions --
323        /// Returns the stack render dimensions.
324        /// @return | integer | Stack width in pixels.
325        /// @return | integer | Stack height in pixels.
326        methods.add_method("getDimensions", |_, this, ()| {
327            Ok(this.inner.get_dimensions())
328        });
329        // -- resize --
330        /// Resizes the post-processing stack render target dimensions.
331        /// @param | w | integer | New width in pixels.
332        /// @param | h | integer | New height in pixels.
333        methods.add_method_mut("resize", |_, this, (w, h): (u32, u32)| {
334            this.inner.resize(w, h);
335            Ok(())
336        });
337        // -- len --
338        /// Returns the number of effect handles in this stack.
339        /// @return | integer | Effect count.
340        methods.add_method("len", |_, this, ()| Ok(this.effects.len()));
341        // -- isEmpty --
342        /// Returns whether this stack has no effects.
343        /// @return | boolean | True when the stack has zero effects.
344        methods.add_method("isEmpty", |_, this, ()| Ok(this.effects.is_empty()));
345        // -- clear --
346        /// Removes all effects and pass state from this stack.
347        methods.add_method_mut("clear", |_, this, ()| {
348            this.effects.clear();
349            this.inner.clear();
350            Ok(())
351        });
352        // -- dedup --
353        /// Removes duplicate effect handles while preserving first occurrences.
354        /// @return | integer | Number of duplicate effects removed.
355        methods.add_method_mut("dedup", |_, this, ()| {
356            let mut seen_ptrs: Vec<*const ()> = Vec::new();
357            let mut new_lua = Vec::with_capacity(this.effects.len());
358            let mut new_inner_effects = Vec::with_capacity(this.effects.len());
359            let mut new_inner_enabled = Vec::with_capacity(this.effects.len());
360            for (i, rc) in this.effects.iter().enumerate() {
361                let ptr = Rc::as_ptr(rc) as *const ();
362                if !seen_ptrs.contains(&ptr) {
363                    seen_ptrs.push(ptr);
364                    new_lua.push(Rc::clone(rc));
365                    new_inner_effects.push(new_lua.len() - 1);
366                    new_inner_enabled.push(this.inner.enabled.get(i).copied().unwrap_or(true));
367                }
368            }
369            let removed = this.effects.len() - new_lua.len();
370            this.effects = new_lua;
371            this.inner.effects = new_inner_effects;
372            this.inner.enabled = new_inner_enabled;
373            Ok(removed as i64)
374        });
375        // -- isCapturing --
376        /// Returns whether this stack is currently capturing draw commands.
377        /// @return | boolean | True when capture mode is active.
378        methods.add_method("isCapturing", |_, this, ()| Ok(this.inner.capturing));
379        // -- beginCapture --
380        /// Starts post-effect capture and queues a renderer begin-capture command.
381        methods.add_method_mut("beginCapture", |_, this, ()| {
382            this.inner.capturing = true;
383            this.state
384                .borrow_mut()
385                .render_commands
386                .push(RenderCommand::BeginPostFx {
387                    stack_id: this.stack_id,
388                });
389            Ok(())
390        });
391        // -- endCapture --
392        /// Ends post-effect capture and queues a renderer end-capture command.
393        methods.add_method_mut("endCapture", |_, this, ()| {
394            this.inner.capturing = false;
395            this.state
396                .borrow_mut()
397                .render_commands
398                .push(RenderCommand::EndPostFx {
399                    stack_id: this.stack_id,
400                });
401            Ok(())
402        });
403        // -- apply --
404        /// Queues this stack's enabled post-effect passes for renderer application.
405        methods.add_method("apply", |_, this, ()| {
406            let passes: Vec<PostFxPass> = this
407                .effects
408                .iter()
409                .zip(this.inner.enabled.iter())
410                .filter(|(_, &enabled)| enabled)
411                .map(|(effect_rc, _)| {
412                    let e = effect_rc.borrow();
413                    PostFxPass {
414                        effect_name: e.get_type_name().to_string(),
415                        params: e.params.clone(),
416                        shader_id: e.shader_id,
417                        auto_uniforms: e.auto_uniforms,
418                    }
419                })
420                .collect();
421            this.state
422                .borrow_mut()
423                .render_commands
424                .push(RenderCommand::ApplyPostFx {
425                    stack_id: this.stack_id,
426                    passes,
427                    width: this.inner.width,
428                    height: this.inner.height,
429                });
430            Ok(())
431        });
432        // -- type --
433        /// Returns the Lua-visible type name for this post-processing stack handle.
434        /// @return | string | The string `LPostFxStack`.
435        methods.add_method("type", |_, _, ()| Ok("LPostFxStack"));
436        // -- typeOf --
437        /// Returns whether this stack handle matches a supported type name.
438        /// @param | name | string | Type name to compare against `PostFxStack` and `Object`.
439        /// @return | boolean | True when the supplied type name matches this handle.
440        methods.add_method("typeOf", |_, _, name: String| {
441            Ok(name == "LPostFxStack" || name == "LObject")
442        });
443        // -- setFeedback --
444        /// Sets the stack feedback blend factor and clamps it to 0.0 through 1.0.
445        /// @param | factor | number | Feedback blend factor.
446        methods.add_method_mut("setFeedback", |_, this, factor: f32| {
447            this.feedback_factor = factor.clamp(0.0, 1.0);
448            Ok(())
449        });
450        // -- getFeedback --
451        /// Returns the current stack feedback blend factor.
452        /// @return | number | Feedback blend factor in the range 0.0 through 1.0.
453        methods.add_method("getFeedback", |_, this, ()| Ok(this.feedback_factor));
454        // -- clearFeedback --
455        /// Resets the stack feedback blend factor to zero.
456        methods.add_method_mut("clearFeedback", |_, this, ()| {
457            this.feedback_factor = 0.0;
458            Ok(())
459        });
460    }
461}
462/// Lua-side handle for an image effect chain detached from live post-effect capture.
463pub struct LuaImageEffect {
464    /// Image effect chain and its owned effect entries.
465    inner: ImageEffect,
466}
467/// Provides Lua methods for editing image effect chains.
468impl LuaUserData for LuaImageEffect {
469    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
470        // -- addEffect --
471        /// Appends a built-in post-effect by type name to this image effect chain.
472        /// @param | name | string | Built-in effect type name.
473        /// @return | LPostFxEffect | Handle for the effect added to the chain.
474        methods.add_method_mut("addEffect", |lua, this, name: String| {
475            let et = PostFxEffectType::from_name(&name)
476                .ok_or_else(|| LuaError::RuntimeError(format!("unknown effect type: {name}")))?;
477            let rc = Rc::new(RefCell::new(PostFxEffect::new(et)));
478            this.inner.add_effect_rc(Rc::clone(&rc));
479            lua.create_userdata(LuaPostFxEffect::from_rc(rc))
480        });
481        // -- getEffect --
482        /// Looks up an image effect by one-based index or effect type name.
483        /// @param | key | string | Effect name string or one-based integer index.
484        /// @return | LuaValue | `LPostFxEffect` handle, or nil when no matching effect exists.
485        methods.add_method("getEffect", |lua, this, key: LuaValue| {
486            let rc_opt = match &key {
487                LuaValue::Integer(i) => this
488                    .inner
489                    .get_effect_by_index((*i as usize).saturating_sub(1)),
490                LuaValue::Number(n) => this
491                    .inner
492                    .get_effect_by_index((*n as usize).saturating_sub(1)),
493                LuaValue::String(s) => this.inner.get_effect_by_name(s.to_str()?),
494                _ => None,
495            };
496            match rc_opt {
497                Some(rc) => Ok(LuaValue::UserData(
498                    lua.create_userdata(LuaPostFxEffect::from_rc(rc))?,
499                )),
500                None => Ok(LuaValue::Nil),
501            }
502        });
503        // -- removeEffect --
504        /// Removes an image effect by one-based index or effect type name.
505        /// @param | key | string | Effect name string or one-based integer index.
506        /// @return | boolean | True when an effect was removed.
507        methods.add_method_mut("removeEffect", |_, this, key: LuaValue| match &key {
508            LuaValue::Integer(i) => Ok(this.inner.remove_by_index((*i as usize).saturating_sub(1))),
509            LuaValue::Number(n) => Ok(this.inner.remove_by_index((*n as usize).saturating_sub(1))),
510            LuaValue::String(s) => Ok(this.inner.remove_by_name(s.to_str()?)),
511            _ => Ok(false),
512        });
513        // -- clearEffects --
514        /// Removes every effect from this image effect chain.
515        methods.add_method_mut("clearEffects", |_, this, ()| {
516            this.inner.clear();
517            Ok(())
518        });
519        // -- clear --
520        /// Removes every effect from this image effect chain.
521        methods.add_method_mut("clear", |_, this, ()| {
522            this.inner.clear();
523            Ok(())
524        });
525        // -- effectCount --
526        /// Returns the number of effects in this image effect chain.
527        /// @return | integer | Effect count.
528        methods.add_method("effectCount", |_, this, ()| Ok(this.inner.effect_count()));
529        // -- getEffectCount --
530        /// Returns the number of effects in this image effect chain.
531        /// @return | integer | Effect count.
532        methods.add_method("getEffectCount", |_, this, ()| {
533            Ok(this.inner.effect_count())
534        });
535        // -- clone --
536        /// Creates a new image effect chain with cloned effect entries.
537        /// @return | LImageEffect | New image effect handle with the same effect chain.
538        methods.add_method("clone", |lua, this, ()| {
539            let mut new_ie = ImageEffect::new("");
540            for i in 0..this.inner.effect_count() {
541                if let Some(rc) = this.inner.get_effect_by_index(i) {
542                    new_ie.add_effect(rc.borrow().clone());
543                }
544            }
545            lua.create_userdata(LuaImageEffect { inner: new_ie })
546        });
547        // -- save --
548        /// Reports success for the current image effect save placeholder.
549        /// @return | boolean | Always true.
550        methods.add_method("save", |_, _, ()| Ok(true));
551        // -- type --
552        /// Returns the Lua-visible type name for this image effect handle.
553        /// @return | string | The string `LImageEffect`.
554        methods.add_method("type", |_, _, ()| Ok("LImageEffect"));
555        // -- typeOf --
556        /// Returns whether this image effect handle matches a supported type name.
557        /// @param | name | string | Type name to compare against `ImageEffect` and `Object`.
558        /// @return | boolean | True when the supplied type name matches this handle.
559        methods.add_method("typeOf", |_, _, name: String| {
560            Ok(name == "LImageEffect" || name == "LObject")
561        });
562        // -- removeByIndex --
563        /// Removes an image effect by zero-based internal index.
564        /// @param | idx | integer | Zero-based effect index.
565        /// @return | boolean | True when an effect was removed.
566        methods.add_method_mut("removeByIndex", |_, this, idx: usize| {
567            Ok(this.inner.remove_by_index(idx))
568        });
569        // -- removeByName --
570        /// Removes the first image effect with a matching effect type name.
571        /// @param | name | string | Effect type name to remove.
572        /// @return | boolean | True when an effect was removed.
573        methods.add_method_mut("removeByName", |_, this, name: String| {
574            Ok(this.inner.remove_by_name(&name))
575        });
576    }
577}
578/// Registers `lurek.effect` constructors and effect state controls.
579pub fn register(lua: &Lua, lurek: &LuaTable, state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
580    let tbl = lua.create_table()?;
581    // -- newEffect --
582    /// Creates a built-in post-processing effect by type name.
583    /// @param | type_name | string | Built-in effect type name such as `blur`, `bloom`, or `crt`.
584    /// @return | LPostFxEffect | New post-processing effect handle.
585    tbl.set(
586        "newEffect",
587        lua.create_function(|lua, type_name: String| {
588            let effect_type = PostFxEffectType::from_name(&type_name).ok_or_else(|| {
589                LuaError::RuntimeError(format!("unknown effect type: {type_name}"))
590            })?;
591            lua.create_userdata(LuaPostFxEffect::from_owned(PostFxEffect::new(effect_type)))
592        })?,
593    )?;
594    // -- newCustomEffect --
595    /// Creates a custom post-processing effect that references an existing shader id.
596    /// @param | shader_id | integer | Renderer shader identifier used for the custom effect.
597    /// @return | LPostFxEffect | New custom post-processing effect handle.
598    tbl.set(
599        "newCustomEffect",
600        lua.create_function(|lua, shader_id: usize| {
601            lua.create_userdata(LuaPostFxEffect::from_owned(PostFxEffect::new_custom(
602                shader_id,
603            )))
604        })?,
605    )?;
606    let s = state.clone();
607    // -- newStack --
608    /// Creates a post-processing stack using optional dimensions or the current window size.
609    /// @param | w | integer? | Stack width in pixels, defaulting to window width.
610    /// @param | h | integer? | Stack height in pixels, defaulting to window height.
611    /// @return | LPostFxStack | New post-processing stack handle.
612    tbl.set(
613        "newStack",
614        lua.create_function(move |lua, (w, h): (Option<u32>, Option<u32>)| {
615            let (default_w, default_h) = {
616                let s = s.borrow();
617                (s.window_width, s.window_height)
618            };
619            let w = w.unwrap_or(default_w);
620            let h = h.unwrap_or(default_h);
621            lua.create_userdata(LuaPostFxStack {
622                inner: PostFxStack::new(w, h),
623                effects: Vec::new(),
624                stack_id: NEXT_STACK_ID.fetch_add(1, Ordering::Relaxed),
625                state: Rc::clone(&s),
626                feedback_factor: 0.0,
627            })
628        })?,
629    )?;
630    let s = state.clone();
631    // -- newPresetStack --
632    /// Creates a named preset post-processing stack with optional dimensions.
633    /// @param | name | string | Preset stack name.
634    /// @param | w | integer? | Stack width in pixels, defaulting to window width.
635    /// @param | h | integer? | Stack height in pixels, defaulting to window height.
636    /// @return | LPostFxStack | New preset post-processing stack handle.
637    tbl.set(
638        "newPresetStack",
639        lua.create_function(
640            move |lua, (name, w, h): (String, Option<u32>, Option<u32>)| {
641                let (default_w, default_h) = {
642                    let borrow = s.borrow();
643                    (borrow.window_width, borrow.window_height)
644                };
645                let w = w.unwrap_or(default_w);
646                let h = h.unwrap_or(default_h);
647                let preset = build_preset(&name, w, h)
648                    .ok_or_else(|| LuaError::RuntimeError(format!("unknown preset '{}'", name)))?;
649                let effects: Vec<Rc<RefCell<PostFxEffect>>> = preset
650                    .effects
651                    .into_iter()
652                    .map(|e| Rc::new(RefCell::new(e)))
653                    .collect();
654                lua.create_userdata(LuaPostFxStack {
655                    inner: preset.stack,
656                    effects,
657                    stack_id: NEXT_STACK_ID.fetch_add(1, Ordering::Relaxed),
658                    state: Rc::clone(&s),
659                    feedback_factor: 0.0,
660                })
661            },
662        )?,
663    )?;
664    // -- newPass --
665    /// Creates a custom post-processing pass from an existing shader id.
666    /// @param | shader_id | integer | Renderer shader identifier used for the pass.
667    /// @return | LPostFxEffect | New custom post-processing effect handle.
668    tbl.set(
669        "newPass",
670        lua.create_function(|lua, shader_id: usize| {
671            lua.create_userdata(LuaPostFxEffect::from_owned(PostFxEffect::new_custom(
672                shader_id,
673            )))
674        })?,
675    )?;
676    // -- getEffectTypes --
677    /// Returns all built-in post-processing effect type names.
678    /// @return | string[] | Built-in effect type strings.
679    tbl.set(
680        "getEffectTypes",
681        lua.create_function(|_, ()| Ok(PostFxEffectType::built_in_names()))?,
682    )?;
683    // -- newImageEffect --
684    /// Creates an image effect chain from no arguments, a type name and optional parameters, or a chain table.
685    /// @param | spec | LuaValue? | Optional effect type string, or an array table of effect entries, or nil for an empty chain.
686    /// @param | params | table? | Optional parameter table used when `spec` is an effect type string.
687    /// @return | LImageEffect | New image effect chain handle.
688    tbl.set(
689        "newImageEffect",
690        lua.create_function(|lua, args: LuaMultiValue| {
691            let mut ie = ImageEffect::new("");
692            match args.iter().next() {
693                None => {}
694                Some(LuaValue::String(s)) => {
695                    let name = s.to_str().map_err(LuaError::external)?.to_string();
696                    let et = PostFxEffectType::from_name(&name).ok_or_else(|| {
697                        LuaError::RuntimeError(format!("unknown effect type: {name}"))
698                    })?;
699                    let mut eff = PostFxEffect::new(et);
700                    if let Some(LuaValue::Table(params)) = args.iter().nth(1) {
701                        for (k, v) in params.clone().pairs::<String, f32>().flatten() {
702                            eff.set_parameter(&k, v);
703                        }
704                    }
705                    ie.add_effect(eff);
706                }
707                Some(LuaValue::Table(chain)) => {
708                    for entry in chain.clone().sequence_values::<LuaTable>() {
709                        let entry = entry?;
710                        let name: String = entry
711                            .get("type")
712                            .or_else(|_| entry.get(1))
713                            .unwrap_or_default();
714                        let et = PostFxEffectType::from_name(&name).ok_or_else(|| {
715                            LuaError::RuntimeError(format!("unknown effect type: {name}"))
716                        })?;
717                        let mut eff = PostFxEffect::new(et);
718                        for (k, v) in entry.pairs::<String, LuaValue>().flatten() {
719                            if k != "type" {
720                                if let LuaValue::Number(n) = v {
721                                    eff.set_parameter(&k, n as f32);
722                                } else if let LuaValue::Integer(i) = v {
723                                    eff.set_parameter(&k, i as f32);
724                                }
725                            }
726                        }
727                        ie.add_effect(eff);
728                    }
729                }
730                _ => {
731                    return Err(LuaError::RuntimeError(
732                        "newImageEffect: invalid arguments".to_string(),
733                    ))
734                }
735            }
736            lua.create_userdata(LuaImageEffect { inner: ie })
737        })?,
738    )?;
739    let s = state.clone();
740    // -- setShaderErrorDisplay --
741    /// Enables or disables renderer shader error display overlays.
742    /// @param | enabled | boolean | New shader error display flag.
743    tbl.set(
744        "setShaderErrorDisplay",
745        lua.create_function(move |_, enabled: bool| {
746            s.borrow_mut().shader_error_display_enabled = enabled;
747            Ok(())
748        })?,
749    )?;
750    let s = state.clone();
751    // -- getShaderErrorDisplay --
752    /// Returns whether renderer shader error display overlays are enabled.
753    /// @return | boolean | True when shader error display is enabled.
754    tbl.set(
755        "getShaderErrorDisplay",
756        lua.create_function(move |_, ()| Ok(s.borrow().shader_error_display_enabled))?,
757    )?;
758    /// Performs the 'effect' operation.
759    lurek.set("effect", tbl)?;
760    Ok(())
761}
