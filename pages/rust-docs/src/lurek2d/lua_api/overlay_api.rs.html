1//! `lurek.overlay` -- Lua bindings for the overlay system: weather, atmosphere, screen flash/shake/fade, transitions, and ambient controls.

3use super::SharedState;
4use crate::overlay::{Overlay, ScreenTransition, TransitionKind, WeatherType};
5use mlua::prelude::*;
6use std::cell::RefCell;
7use std::rc::Rc;

9/// Lua-side handle for screen overlay, ambient, weather, and transition visual state.
10pub struct LuaOverlay {
11    /// Overlay state that builds renderer commands for full-screen effects.
12    inner: Overlay,
13    /// Shared runtime state used for renderer commands and light ambient synchronization.
14    state: Rc<RefCell<SharedState>>,
15}
16/// Provides Lua methods for overlay animation, ambient, weather, fog, water, and render submission.

554            "setWaterTint",
555            |_, this, (r, g, b, strength): (f32, f32, f32, f32)| {
556                this.inner.water.tint_r = r;
557                this.inner.water.tint_g = g;
558                this.inner.water.tint_b = b;
559                this.inner.water.tint_strength = strength;
560                Ok(())
561            },
562        );
563        // -- setCustomShader --
564        /// Sets or clears the custom overlay shader name.
565        /// @param | name | string? | Optional shader name; nil clears the custom shader.
566        methods.add_method_mut("setCustomShader", |_, this, name: Option<String>| {
567            this.inner.custom_shader = name;
568            Ok(())
569        });
570        // -- getWater --
571        /// Returns a table describing the current water effect settings.
572        /// @return | table | Water state table with enabled, wave, tint, depth, and time fields.
573        /// @field | enabled | boolean | Whether water effect is enabled.
574        /// @field | amplitude | number | Wave amplitude.
575        /// @field | frequency | number | Wave frequency.
576        /// @field | speed | number | Wave speed.
577        /// @field | tint_r | number | Tint red component.
578        /// @field | tint_g | number | Tint green component.
579        /// @field | tint_b | number | Tint blue component.
580        /// @field | tint_strength | number | Tint strength.
581        /// @field | depth_r | number | Depth red component.
582        /// @field | depth_g | number | Depth green component.
583        /// @field | depth_b | number | Depth blue component.
584        /// @field | depth_strength | number | Depth strength.
585        /// @field | time | number | Elapsed time.
586        methods.add_method("getWater", |lua, this, ()| {
587            let w = &this.inner.water;
588            let t = lua.create_table()?;
589            t.set("enabled", w.enabled)?;
590            t.set("amplitude", w.amplitude)?;
591            t.set("frequency", w.frequency)?;
592            t.set("speed", w.speed)?;
593            t.set("tint_r", w.tint_r)?;
594            t.set("tint_g", w.tint_g)?;
595            t.set("tint_b", w.tint_b)?;
596            t.set("tint_strength", w.tint_strength)?;
597            t.set("depth_r", w.depth_r)?;
598            t.set("depth_g", w.depth_g)?;
599            t.set("depth_b", w.depth_b)?;
600            t.set("depth_strength", w.depth_strength)?;
601            t.set("time", w.time)?;
602            Ok(t)
603        });
604        // -- type --
605        /// Returns the Lua-visible type name for this overlay handle.
606        /// @return | string | The string `LOverlay`.
607        methods.add_method("type", |_, _this, ()| Ok("LOverlay"));
608        // -- typeOf --
609        /// Returns whether this overlay handle matches a supported type name.
610        /// @param | name | string | Type name to compare against `Overlay` and `Object`.
611        /// @return | boolean | True when the supplied type name matches this handle.
612        methods.add_method("typeOf", |_, _this, name: String| {
613            Ok(name == "LObject" || name == "LOverlay")
614        });
615    }
616}
617/// Lua-side handle for a timed screen transition effect.
618pub struct LuaScreenTransition {
619    /// Transition kind, timer, direction, and RGBA color.
620    inner: ScreenTransition,
621}
622/// Provides Lua methods for playing and inspecting screen transitions.
623impl mlua::UserData for LuaScreenTransition {
624    fn add_methods<'lua, M: mlua::UserDataMethods<'lua, Self>>(methods: &mut M) {
625        // -- play --
626        /// Starts this screen transition forward from its current state.
627        methods.add_method_mut("play", |_, this, ()| {
628            this.inner.play();
629            Ok(())
630        });
631        // -- reverse --
632        /// Starts this screen transition in reverse from its current state.
633        methods.add_method_mut("reverse", |_, this, ()| {
634            this.inner.reverse();
635            Ok(())
636        });
637        // -- update --
638        /// Advances this transition timer and returns whether it remains active.
639        /// @param | dt | number | Delta time in seconds.
640        /// @return | boolean | True when the transition is still active after the update.
641        methods.add_method_mut("update", |_, this, dt: f32| Ok(this.inner.update(dt)));
642        // -- progress --
643        /// Returns normalized transition progress.
644        /// @return | number | Progress value between the transition start and end.
645        methods.add_method("progress", |_, this, ()| Ok(this.inner.progress()));
646        // -- isActive --
647        /// Returns whether the transition is currently active.
648        /// @return | boolean | True when the transition is active.
649        methods.add_method("isActive", |_, this, ()| Ok(this.inner.is_active()));
650        // -- isDone --
651        /// Returns whether the transition has finished.
652        /// @return | boolean | True when the transition is complete.
653        methods.add_method("isDone", |_, this, ()| Ok(this.inner.is_done()));
654        // -- kind --
655        /// Returns the transition kind name. This method is available to Lua scripts.
656        /// @return | string | Transition kind name.
657        methods.add_method("kind", |_, this, ()| Ok(this.inner.kind.name()));
658        // -- color --
659        /// Returns the transition RGBA color.
660        /// @return | number | Red channel.
661        /// @return | number | Green channel.
662        /// @return | number | Blue channel.
663        /// @return | number | Alpha channel.
664        methods.add_method("color", |_, this, ()| {
665            let c = this.inner.color;
666            Ok((c[0], c[1], c[2], c[3]))
667        });
668        // -- setColor --
669        /// Sets the transition RGBA color from a numeric array table.
670        /// @param | color | table | Numeric color table using indices 1 through 4.
671        methods.add_method_mut("setColor", |_, this, ct: mlua::Table| {
672            this.inner.color = [
673                ct.get::<_, f32>(1).unwrap_or(0.0),
674                ct.get::<_, f32>(2).unwrap_or(0.0),
675                ct.get::<_, f32>(3).unwrap_or(0.0),
676                ct.get::<_, f32>(4).unwrap_or(1.0),
677            ];
678            Ok(())
679        });
680        // -- type --
681        /// Returns the Lua-visible type name for this transition handle.
682        /// @return | string | The string `LScreenTransition`.
683        methods.add_method("type", |_, _, ()| Ok("LScreenTransition"));
684        // -- typeOf --
685        /// Returns whether this transition handle matches a supported type name.
686        /// @param | name | string | Type name to compare against `ScreenTransition` and `Object`.
687        /// @return | boolean | True when the supplied type name matches this handle.
688        methods.add_method("typeOf", |_, _, name: String| {
689            Ok(name == "LScreenTransition" || name == "LObject")
690        });
691    }
692}
693/// Registers `lurek.overlay` constructors for overlay and transition objects.
694pub fn register(lua: &Lua, lurek: &LuaTable, state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
695    let tbl = lua.create_table()?;
696    let s = state.clone();
697    // -- new --
698    /// Creates an overlay controller for screen effects using optional dimensions.
699    /// @param | w | integer? | Overlay width in pixels, defaulting to 800.
700    /// @param | h | integer? | Overlay height in pixels, defaulting to 600.
701    /// @return | LOverlay | New overlay handle.
702    tbl.set(
703        "new",
704        lua.create_function(move |lua, (w, h): (Option<u32>, Option<u32>)| {
705            let width = w.unwrap_or(800);
706            let height = h.unwrap_or(600);
707            lua.create_userdata(LuaOverlay {
708                inner: Overlay::new(width, height),
709                state: s.clone(),
710            })
711        })?,
712    )?;
713    // -- newTransition --
714    /// Creates a timed screen transition with optional kind, duration, and color.
715    /// @param | kind | string? | Transition kind name, defaulting to `fade`.
716    /// @param | duration | number? | Duration in seconds, defaulting to 1.0.
717    /// @param | color_tbl | table? | Numeric RGBA table using indices 1 through 4.
718    /// @return | LScreenTransition | New screen transition handle.
719    tbl.set(
720        "newTransition",
721        lua.create_function(
722            move |lua, (kind, duration, color_tbl): (Option<String>, Option<f32>, Option<LuaTable>)| {
723                let k = TransitionKind::from_str(kind.as_deref().unwrap_or("fade"));
724                let dur = duration.unwrap_or(1.0);
725                let color = if let Some(ct) = color_tbl {
726                    [
727                        ct.get::<_, f32>(1).unwrap_or(0.0),
728                        ct.get::<_, f32>(2).unwrap_or(0.0),
729                        ct.get::<_, f32>(3).unwrap_or(0.0),
730                        ct.get::<_, f32>(4).unwrap_or(1.0),
731                    ]
732                } else {
733                    [0.0, 0.0, 0.0, 1.0]
734                };
735                lua.create_userdata(LuaScreenTransition {
736                    inner: ScreenTransition::new(k, dur, color),
737                })
738            },
739        )?,
740    )?;
741    lurek.set("overlay", tbl)?;
742
743    // Compatibility aliases: lurek.effect.newOverlay / lurek.effect.newTransition
744    // overlay_api is registered after effect_api, so lurek.effect exists here.
745    if let Ok(effect_tbl) = lurek.get::<_, mlua::Table>("effect") {
746        let overlay_tbl: mlua::Table = lurek.get("overlay")?;
747        effect_tbl.set("newOverlay", overlay_tbl.get::<_, mlua::Function>("new")?)?;
748        effect_tbl.set(
749            "newTransition",
750            overlay_tbl.get::<_, mlua::Function>("newTransition")?,
751        )?;
752    }
753
754    Ok(())
755}
