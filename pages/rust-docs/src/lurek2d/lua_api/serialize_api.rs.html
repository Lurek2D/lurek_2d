1//! `lurek.serialize` — Data serialization and deserialization with JSON, TOML, CSV, XML, INI, and MessagePack encoding/decoding for game configuration, save data, and inter-system data exchange.

12/// Resolves the first byte of an optional delimiter string, defaulting to a comma.
13fn parse_delimiter(delim: Option<String>) -> u8 {
14    delim
15        .as_deref()
16        .and_then(|d| d.as_bytes().first().copied())
17        .unwrap_or(b',')
18}
19/// Parses CSV decoding and encoding options from an optional Lua options table.
20fn csv_options_from_table(opts: Option<LuaTable>) -> LuaResult<CsvOptions> {
21    let mut out = CsvOptions::default();
22    if let Some(t) = opts {
23        let delim: Option<String> = t.get("delimiter")?;
24        let headers: Option<bool> = t.get("has_headers")?;
25        out.delimiter = parse_delimiter(delim);
26        out.has_headers = headers.unwrap_or(true);
27    }
28    Ok(out)
29}
30/// Parses shared serialization encode options from an optional Lua options table.
31fn encode_options_from_table(opts: Option<LuaTable>) -> LuaResult<EncodeOptions> {
32    let mut out = EncodeOptions::default();
33    if let Some(t) = opts {
34        let pretty: Option<bool> = t.get("pretty")?;
35        out.json_pretty = pretty.unwrap_or(false);
36        out.csv = csv_options_from_table(Some(t))?;
37    }
38    Ok(out)
39}
40/// Registers the `lurek.serialize` module into the Lua runtime.
41pub fn register(lua: &Lua, lurek: &LuaTable, _state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
42    let tbl = lua.create_table()?;
43
44    // -- fromJson --
45    /// Parses a JSON string into a Lua table. Use this to load configuration files, network responses, or any structured data stored as JSON.
46    /// @param | text | string | A valid JSON string to parse.
47    /// @return | table | The decoded Lua table representing the JSON structure.
48    tbl.set(
49        "fromJson",
50        lua.create_function(|lua, s: String| {
51            let val = crate::serialize::from_json(&s).map_err(LuaError::RuntimeError)?;
52            to_lua(lua, &val)
53        })?,
54    )?;
55
56    // -- toJson --
57    /// Serializes a Lua value (table, string, number, boolean, or nil) into a JSON string. Useful for saving game state, writing config files, or preparing network payloads.
58    /// @param | value | any | The Lua value to serialize into JSON.
59    /// @param | pretty | boolean? | When true, outputs indented human-readable JSON. Defaults to false (compact).
60    /// @return | string | The JSON-encoded string representation of the value.
61    tbl.set(
62        "toJson",
63        lua.create_function(|_, (value, pretty): (LuaValue, Option<bool>)| {
64            let val = from_lua(&value)?;
65            crate::serialize::to_json(&val, pretty.unwrap_or(false)).map_err(LuaError::RuntimeError)
66        })?,
67    )?;
68
69    // -- fromToml --
70    /// Parses a TOML string into a Lua table. Ideal for loading game configuration files, level definitions, and engine settings stored in TOML format.
71    /// @param | text | string | A valid TOML string to parse.
72    /// @return | table | The decoded Lua table representing the TOML structure.
73    tbl.set(
74        "fromToml",
75        lua.create_function(|lua, s: String| {
76            let val = crate::serialize::from_toml(&s).map_err(LuaError::RuntimeError)?;
77            to_lua(lua, &val)
78        })?,
79    )?;
80
81    // -- fromIni --
82    /// Parses an INI-format string into a Lua table. Sections become nested tables, and key-value pairs become string fields. Useful for legacy config files or simple settings.
83    /// @param | text | string | A valid INI string to parse.
84    /// @return | table | The decoded Lua table with section names as keys and their key-value pairs as nested tables.
85    tbl.set(
86        "fromIni",
87        lua.create_function(|lua, s: String| {
88            let val = crate::serialize::from_ini(&s).map_err(LuaError::RuntimeError)?;
89            to_lua(lua, &val)
90        })?,
91    )?;
92
93    // -- toToml --
94    /// Serializes a Lua table into a TOML-formatted string. Use this to write configuration files, save structured settings, or export data in a human-readable format.
95    /// @param | value | table | The Lua table to serialize into TOML.
96    /// @return | string | The TOML-encoded string representation of the table.
97    tbl.set(
98        "toToml",
99        lua.create_function(|_, value: LuaValue| {
100            let val = from_lua(&value)?;
101            crate::serialize::to_toml(&val).map_err(LuaError::RuntimeError)
102        })?,
103    )?;
104
105    // -- fromCsv --
106    /// Parses a CSV string into a Lua table (array of rows). Each row is either a keyed table (when headers are present) or an indexed array of field values. Useful for loading spreadsheet exports, leaderboard data, or tabular game data.
107    /// @param | text | string | The CSV content to parse.
108    /// @param | delimiter | string? | Single-character field delimiter. Defaults to comma (",").
109    /// @param | hasHeaders | boolean? | When true, the first row is treated as column names and each data row becomes a keyed table. Defaults to true.
110    /// @return | table | An array of row tables containing the parsed CSV data.
111    tbl.set(
112        "fromCsv",
113        lua.create_function(
114            |lua, (s, delim, headers): (String, Option<String>, Option<bool>)| {
115                let opts = CsvOptions {
116                    delimiter: parse_delimiter(delim),
117                    has_headers: headers.unwrap_or(true),
118                };
119                let val = crate::serialize::from_csv(&s, opts).map_err(LuaError::RuntimeError)?;
120                to_lua(lua, &val)
121            },
122        )?,
123    )?;
124
125    // -- toCsv --
126    /// Serializes a Lua table (array of row tables) into a CSV-formatted string. Each row table should have consistent keys or be an indexed array. Use this to export leaderboards, save tabular data, or generate spreadsheet-compatible output.
127    /// @param | value | table | An array of row tables to serialize.
128    /// @param | delimiter | string? | Single-character field delimiter. Defaults to comma (",").
129    /// @param | hasHeaders | boolean? | When true, writes column names as the first row. Defaults to true.
130    /// @return | string | The CSV-encoded string of the table data.
131    tbl.set(
132        "toCsv",
133        lua.create_function(
134            |_, (value, delim, headers): (LuaValue, Option<String>, Option<bool>)| {
135                let opts = CsvOptions {
136                    delimiter: parse_delimiter(delim),
137                    has_headers: headers.unwrap_or(true),
138                };
139                let val = from_lua(&value)?;
140                crate::serialize::to_csv(&val, opts).map_err(LuaError::RuntimeError)
141            },
142        )?,
143    )?;
144
145    // -- encodeMsgPack --
146    /// Encodes a Lua table into a compact binary MessagePack string. MessagePack is faster and smaller than JSON, making it ideal for save files, network packets, or any scenario where performance matters more than human readability. The argument must be a table.
147    /// @param | value | table | The Lua table to encode. Must be a table (not a primitive).
148    /// @return | string | A binary string containing the MessagePack-encoded data.
149    tbl.set(
150        "encodeMsgPack",
151        lua.create_function(|lua, value: LuaValue| {
152            if !matches!(value, LuaValue::Table(_)) {
153                return Err(LuaError::RuntimeError(
154                    "encodeMsgPack: argument must be a table".to_string(),
155                ));
156            }
157            let val = from_lua(&value)?;
158            let bytes = crate::serialize::to_msgpack(&val).map_err(LuaError::RuntimeError)?;
159            lua.create_string(&bytes)
160        })?,
161    )?;
162
163    // -- decodeMsgPack --
164    /// Decodes a binary MessagePack string back into a Lua table. Use this to read save files, network packets, or any data previously encoded with encodeMsgPack.
165    /// @param | bytes | string | A binary string containing valid MessagePack data.
166    /// @return | table | The decoded Lua table from the MessagePack payload.
167    tbl.set(
168        "decodeMsgPack",
169        lua.create_function(|lua, bytes: mlua::String| {
170            let val =
171                crate::serialize::from_msgpack(bytes.as_bytes()).map_err(LuaError::RuntimeError)?;
172            crate::serialize::lua_table::to_lua(lua, &val)
173        })?,
174    )?;
175
176    // -- decodeXml --
177    /// Parses an XML string into a Lua table structure. Elements become nested tables with tag names as keys. Useful for loading Tiled map exports, SVG data, UI layout definitions, or other XML-based game assets.
178    /// @param | text | string | A valid XML string to parse.
179    /// @return | table | A nested Lua table representing the XML document structure.
180    tbl.set(
181        "decodeXml",
182        lua.create_function(|lua, s: String| {
183            let val = crate::serialize::from_xml(&s).map_err(LuaError::RuntimeError)?;
184            crate::serialize::lua_table::to_lua(lua, &val)
185        })?,
186    )?;
187
188    // -- validate --
189    /// Validates a Lua value against a schema table. The schema defines expected types, required fields, and constraints. Returns a success boolean and an optional error message string describing the first validation failure. Use this to verify save data integrity or user-provided configuration before processing.
190    /// @param | value | any | The data to validate.
191    /// @param | schema | table | A schema table defining the expected structure and constraints.
192    /// @return | boolean | True if validation passes, false otherwise.
193    /// @return | string | An error message describing the validation failure, or nil on success.
194    tbl.set(
195        "validate",
196        lua.create_function(|_, (value, schema): (LuaValue, LuaValue)| {
197            let val = from_lua(&value)?;
198            let sch = from_lua(&schema)?;
199            match crate::serialize::validate_schema(&val, &sch) {
200                Ok(()) => Ok((true, None::<String>)),
201                Err(msg) => Ok((false, Some(msg))),
202            }
203        })?,
204    )?;
205
206    // -- detectFormat --
207    /// Attempts to auto-detect the serialization format of a string by inspecting its content (e.g., leading `{` for JSON, `[section]` for INI, XML declaration for XML). Returns the format name or nil if detection fails. Useful for loading user-provided files where the format is unknown.
208    /// @param | text | string | The raw text content to analyze.
209    /// @return | string | The detected format name ("json", "toml", "csv", "xml", "ini"), or nil if unrecognized.
210    tbl.set(
211        "detectFormat",
212        lua.create_function(|_, s: String| {
213            Ok(crate::serialize::detect_format(&s).map(|f| f.as_str().to_string()))
214        })?,
215    )?;
216
217    // -- decode --
218    /// Universal decoder that parses a string payload into a Lua table using the specified format. If no format is given, auto-detects from the content. Supports JSON, TOML, CSV, XML, INI, and MessagePack. Use this as a single entry point when handling files of varying or unknown formats.
219    /// @param | payload | string | The raw string (or binary for msgpack) to decode.
220    /// @param | format | string? | Format hint: "json", "toml", "csv", "xml", "ini", or "msgpack". Nil triggers auto-detection.
221    /// @param | opts | table? | Optional settings table. For CSV: `delimiter` (string) and `has_headers` (boolean).
222    /// @return | table | The decoded Lua table.
223    tbl.set(
224        "decode",
225        lua.create_function(
226            |lua, (payload, format, opts): (LuaValue, Option<String>, Option<LuaTable>)| {
227                let fmt = if let Some(name) = format.as_deref() {
228                    Some(SerialFormat::parse(name).ok_or_else(|| {
229                        LuaError::RuntimeError(
230                            "decode: unknown format (expected json/toml/csv/xml/ini/msgpack)"
231                                .to_string(),
232                        )
233                    })?)
234                } else {
235                    None
236                };
237                let val = match (payload, fmt) {
238                    (LuaValue::String(bytes), Some(SerialFormat::MsgPack)) => {
239                        crate::serialize::decode_bytes(bytes.as_bytes(), SerialFormat::MsgPack)
240                            .map_err(LuaError::RuntimeError)?
241                    }
242                    (LuaValue::String(text), Some(f)) => {
243                        let s = text
244                            .to_str()
245                            .map_err(|e| {
246                                LuaError::RuntimeError(format!("decode: expected UTF-8 text: {e}"))
247                            })?
248                            .to_string();
249                        crate::serialize::decode_text(
250                            &s,
251                            Some(f),
252                            DecodeOptions {
253                                csv: csv_options_from_table(opts)?,
254                            },
255                        )
256                        .map_err(LuaError::RuntimeError)?
257                    }
258                    (LuaValue::String(text), None) => {
259                        let s = text
260                            .to_str()
261                            .map_err(|e| {
262                                LuaError::RuntimeError(format!(
263                                    "decode: expected UTF-8 text for auto-detect: {e}"
264                                ))
265                            })?
266                            .to_string();
267                        crate::serialize::decode_text(
268                            &s,
269                            None,
270                            DecodeOptions {
271                                csv: csv_options_from_table(opts)?,
272                            },
273                        )
274                        .map_err(LuaError::RuntimeError)?
275                    }
276                    _ => {
277                        return Err(LuaError::RuntimeError(
278                            "decode: payload must be a string".to_string(),
279                        ));
280                    }
281                };
282                to_lua(lua, &val)
283            },
284        )?,
285    )?;
286
287    // -- encode --
288    /// Universal encoder that serializes a Lua value into the specified format. Supports JSON, TOML, CSV, and MessagePack. Returns a string (text for JSON/TOML/CSV, binary for MessagePack). Use this as a single entry point for all serialization needs.
289    /// @param | value | any | The Lua value to encode.
290    /// @param | format | string | Target format: "json", "toml", "csv", or "msgpack".
291    /// @param | opts | table? | Optional settings table. For JSON: `pretty` (boolean). For CSV: `delimiter` (string) and `has_headers` (boolean).
292    /// @return | string | The encoded string (text or binary depending on format).
293    tbl.set(
294        "encode",
295        lua.create_function(
296            |lua, (value, format, opts): (LuaValue, String, Option<LuaTable>)| {
297                let val = from_lua(&value)?;
298                let fmt = SerialFormat::parse(&format).ok_or_else(|| {
299                    LuaError::RuntimeError(
300                        "encode: unknown format (expected json/toml/csv/msgpack)".to_string(),
301                    )
302                })?;
303                let encoded = crate::serialize::encode(&val, fmt, encode_options_from_table(opts)?)
304                    .map_err(LuaError::RuntimeError)?;
305                match encoded {
306                    EncodedValue::Text(s) => lua.create_string(&s),
307                    EncodedValue::Binary(bytes) => lua.create_string(&bytes),
308                }
309            },
310        )?,
311    )?;
312
313    // -- applyDefaults --
314    /// Merges a schema's default values into a data table, filling in any missing fields without overwriting existing ones. Use this to ensure game config or save data always has complete fields even when the user provides only partial overrides.
315    /// @param | value | any | The data value that may have missing fields.
316    /// @param | schema | table | A schema table containing `default` entries for fields.
317    /// @return | table | A new table with defaults applied for any absent fields.
318    tbl.set(
319        "applyDefaults",
320        lua.create_function(|lua, (value, schema): (LuaValue, LuaValue)| {
321            let val = from_lua(&value)?;
322            let sch = from_lua(&schema)?;
323            let patched =
324                crate::serialize::apply_schema_defaults(&val, &sch).map_err(LuaError::RuntimeError)?;
325            to_lua(lua, &patched)
326        })?,
327    )?;
328    /// Performs the 'serialize' operation.
329    lurek.set("serialize", tbl.clone())?;
330    // Backward-compatibility alias
331    lurek.set("serial", tbl)?;
332    Ok(())
