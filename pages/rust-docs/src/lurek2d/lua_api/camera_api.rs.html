12/// Builds a Lua camera handle with fresh path, zoom tween, and parallax state.
13fn make_lua_camera(inner: Rc<RefCell<Camera2D>>, state: Rc<RefCell<SharedState>>) -> LuaCamera2D {
14    LuaCamera2D {
15        inner,
16        path: RefCell::new(None),
17        zoom_tween: RefCell::new(None),
18        parallax: RefCell::new(HashMap::new()),
19        state,
20    }
21}
22/// Parses a Lua follow easing name into a camera easing mode.
23fn parse_follow_easing(name: &str) -> CameraEasing {
24    match name.to_ascii_lowercase().as_str() {
25        "smoothstep" | "smooth" => CameraEasing::SmoothStep,
26        "easeout" | "ease_out" | "ease-out" => CameraEasing::EaseOutCubic,
27        _ => CameraEasing::Linear,
28    }
29}
30/// Parses an optional Lua zoom easing name into a zoom tween easing mode.
31fn parse_zoom_easing(name: Option<String>) -> CameraEasing {
32    match name
33        .unwrap_or_else(|| "linear".to_string())
34        .to_ascii_lowercase()
35        .as_str()
36    {
37        "smoothstep" | "smooth" => CameraEasing::SmoothStep,
38        "easeout" | "ease_out" | "ease-out" => CameraEasing::EaseOutCubic,
39        _ => CameraEasing::Linear,
40    }
41}
42/// Lua-side 2D camera handle with transforms, effects, bounds, and render command access.
43pub struct LuaCamera2D {
44    /// Shared camera state used by all camera methods.
45    inner: Rc<RefCell<Camera2D>>,
46    /// Optional active path followed by `updatePath`.
47    path: RefCell<Option<CameraPath>>,
48    /// Optional active zoom tween followed by `updateZoom`.
49    zoom_tween: RefCell<Option<ZoomTween>>,
50    /// Per-layer parallax factors keyed by layer name.
51    parallax: RefCell<HashMap<String, f32>>,
52    /// Shared runtime state receiving camera render transform commands.
53    state: Rc<RefCell<SharedState>>,
54}
55impl LuaCamera2D {

59    }
60    /// Returns the camera position tuple for engine-side helpers.
61    pub(crate) fn position(&self) -> (f32, f32) {

686/// Lua-side camera rig that manages named cameras and viewport layouts.
687pub struct LuaCameraRig {
688    /// Shared camera rig state exposed through this userdata handle.
689    inner: Rc<RefCell<CameraRig2D>>,
690    /// Shared runtime state receiving render transform commands.
691    state: Rc<RefCell<SharedState>>,
692}
693impl LuaUserData for LuaCameraRig {
694    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
695        // -- splitScreen --
696        /// Applies a split-screen layout using the current window size.
697        /// @param | window_w | number | Window width in pixels.
698        /// @param | window_h | number | Window height in pixels.
699        methods.add_method(
700            "splitScreen",
701            |_, this, (window_w, window_h): (f32, f32)| {
702                this.inner
703                    .borrow_mut()
704                    .apply_split_screen_layout(window_w, window_h);
705                Ok(())
706            },
707        );
708        // -- minimap --
709        /// Applies a minimap layout using the current window size and optional ratio.
710        /// @param | window_w | number | Window width in pixels.
711        /// @param | window_h | number | Window height in pixels.
712        /// @param | ratio | number? | Minimap size ratio; defaults to 0.25.
713        methods.add_method(
714            "minimap",
715            |_, this, (window_w, window_h, ratio): (f32, f32, Option<f32>)| {
716                this.inner.borrow_mut().apply_minimap_layout(
717                    window_w,
718                    window_h,
719                    ratio.unwrap_or(0.25),
720                );
721                Ok(())
722            },
723        );
724        // -- pictureInPicture --
725        /// Applies a picture-in-picture layout using optional inset size.
726        /// @param | window_w | number | Window width in pixels.
727        /// @param | window_h | number | Window height in pixels.
728        /// @param | pip_w | number? | Picture-in-picture width; defaults to 320.
729        /// @param | pip_h | number? | Picture-in-picture height; defaults to 180.
730        methods.add_method(
731            "pictureInPicture",
732            |_, this, (window_w, window_h, pip_w, pip_h): (f32, f32, Option<f32>, Option<f32>)| {
733                this.inner.borrow_mut().apply_picture_in_picture_layout(
734                    window_w,
735                    window_h,
736                    pip_w.unwrap_or(320.0),
737                    pip_h.unwrap_or(180.0),
738                );
739                Ok(())
740            },
741        );
742        // -- setPosition --
743        /// Sets the position of a named rig camera, creating it if needed.
744        /// @param | name | string | Camera name.
745        /// @param | x | number | Camera X position.
746        /// @param | y | number | Camera Y position.
747        methods.add_method(
748            "setPosition",
749            |_, this, (name, x, y): (String, f32, f32)| {
750                this.inner
751                    .borrow_mut()
752                    .ensure_camera(&name, 800.0, 600.0)
753                    .set_position(x, y);
754                Ok(())
755            },
756        );
757        // -- setZoom --
758        /// Sets the zoom of a named rig camera, creating it if needed.
759        /// @param | name | string | Camera name.
760        /// @param | zoom | number | Camera zoom factor.
761        methods.add_method("setZoom", |_, this, (name, zoom): (String, f32)| {
762            this.inner
763                .borrow_mut()
764                .ensure_camera(&name, 800.0, 600.0)
765                .set_zoom(zoom);
766            Ok(())
767        });
768        // -- setTarget --
769        /// Sets the follow target of a named rig camera, creating it if needed.
770        /// @param | name | string | Camera name.
771        /// @param | x | number | Target X position.
772        /// @param | y | number | Target Y position.
773        methods.add_method("setTarget", |_, this, (name, x, y): (String, f32, f32)| {
774            this.inner
775                .borrow_mut()
776                .ensure_camera(&name, 800.0, 600.0)
777                .set_target(x, y);
778            Ok(())
779        });
780        // -- updateAll --
781        /// Advances every camera in this rig. This method is available to Lua scripts.
782        /// @param | dt | number | Elapsed time in seconds.
783        methods.add_method("updateAll", |_, this, dt: f32| {
784            this.inner.borrow_mut().update_all(dt);
785            Ok(())
786        });
787        // -- apply --
788        /// Appends render commands for a named camera in this rig.
789        /// @param | name | string | Camera name to apply.
790        /// @return | boolean | True when the named camera exists.
791        methods.add_method("apply", |_, this, name: String| {
792            let rig = this.inner.borrow();
793            if let Some(cam) = rig.camera(&name) {
794                let mut state = this.state.borrow_mut();
795                cam.append_begin_render_commands(&mut state.render_commands);
796                return Ok(true);
797            }
798            Ok(false)
799        });
800        // -- getViewport --
801        /// Returns a named rig camera viewport with a leading availability flag.
802        /// @param | name | string | Camera name to query.
803        /// @return | boolean, number, number, number, number | Has-camera flag followed by viewport X, Y, width, and height.
804        methods.add_method("getViewport", |_, this, name: String| {
805            let out = if let Some((x, y, w, h)) = this.inner.borrow().viewport_of(&name) {
806                (true, x, y, w, h)
807            } else {
808                (false, 0.0, 0.0, 0.0, 0.0)
809            };
810            Ok(out)
811        });
812        // -- names --
813        /// Returns all camera names in this rig.
814        /// @return | string[] | Camera names.
815        methods.add_method("names", |lua, this, ()| {
816            let names = this.inner.borrow().camera_names();
817            let table = lua.create_table()?;
818            for (idx, name) in names.iter().enumerate() {
819                table.set(idx + 1, name.as_str())?;
820            }
821            Ok(table)
822        });
823        // -- remove --
824        /// Removes a named camera from this rig.
825        /// @param | name | string | Camera name to remove.
826        /// @return | boolean | True when the camera existed and was removed.
827        methods.add_method("remove", |_, this, name: String| {
828            Ok(this.inner.borrow_mut().remove_camera(&name))
829        });
830        // -- has --
831        /// Returns whether this rig contains a named camera.
832        /// @param | name | string | Camera name to check.
833        /// @return | boolean | True when the camera exists.
834        methods.add_method("has", |_, this, name: String| {
835            Ok(this.inner.borrow().has_camera(&name))
836        });
837        // -- type --
838        /// Returns the Lua-visible type name for this camera rig handle.
839        /// @return | string | The string `LCameraRig`.
840        methods.add_method("type", |_, _, ()| Ok("LCameraRig"));
841        // -- typeOf --
842        /// Returns whether this camera rig handle matches a supported type name.
843        /// @param | name | string | Type name to compare against `LCameraRig` and `Object`.
844        /// @return | boolean | True when the supplied type name matches this handle.
845        methods.add_method("typeOf", |_, _, name: String| {
846            Ok(name == "LCameraRig" || name == "LObject")
847        });
848    }
849}
850/// Registers the `lurek.camera` API table with the Lua VM.
851pub fn register(lua: &Lua, lurek: &LuaTable, state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
852    let tbl = lua.create_table()?;
853    // -- new --
854    /// Creates a 2D camera with optional virtual viewport size.
855    /// @param | vw | number? | Virtual viewport width; defaults to 800.
856    /// @param | vh | number? | Virtual viewport height; defaults to 600.
857    /// @return | LCamera | New camera handle.
858    let s = state.clone();
859    tbl.set(
860        "new",
861        lua.create_function(move |lua, (vw, vh): (Option<f32>, Option<f32>)| {
862            let vw = vw.unwrap_or(800.0);
863            let vh = vh.unwrap_or(600.0);
864            lua.create_userdata(make_lua_camera(
865                Rc::new(RefCell::new(Camera2D::new(vw, vh))),
866                s.clone(),
867            ))
868        })?,
869    )?;
870    // -- newCamera --
871    /// Creates a 2D camera with optional virtual viewport size.
872    /// @param | vw | number? | Virtual viewport width; defaults to 800.
873    /// @param | vh | number? | Virtual viewport height; defaults to 600.
874    /// @return | LCamera | New camera handle.
875    let s = state.clone();
876    tbl.set(
877        "newCamera",
878        lua.create_function(move |lua, (vw, vh): (Option<f32>, Option<f32>)| {
879            let vw = vw.unwrap_or(800.0);
880            let vh = vh.unwrap_or(600.0);
881            lua.create_userdata(make_lua_camera(
882                Rc::new(RefCell::new(Camera2D::new(vw, vh))),
883                s.clone(),
884            ))
885        })?,
886    )?;
887    // -- newRig --
888    /// Creates an empty named camera rig. This function is exposed to Lua scripts.
889    /// @return | LCameraRig | New camera rig handle.
890    let s = state.clone();
891    tbl.set(
892        "newRig",
893        lua.create_function(move |lua, ()| {
894            lua.create_userdata(LuaCameraRig {
895                inner: Rc::new(RefCell::new(CameraRig2D::new())),
896                state: s.clone(),
897            })
898        })?,
899    )?;
900    /// Performs the 'camera' operation.
901    lurek.set("camera", tbl)?;
902    Ok(())
903}
