14/// Lua-side animation object containing frame rectangles, named clips, playback state, and blend state.
15pub struct LuaAnimation {
16    /// Owned animation data exposed through this userdata handle.
17    inner: Animation,
18    /// Spritesheet stored by `setImage`; used by `draw` when no image argument is supplied.
19    stored_image: Option<LuaImage>,

21impl LuaUserData for LuaAnimation {
22    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
23        // -- addFrame --
24        /// Adds one frame rectangle to this animation.
25        /// @param | x | number | Frame X coordinate in texture pixels.
26        /// @param | y | number | Frame Y coordinate in texture pixels.
27        /// @param | w | number | Frame width in texture pixels.
28        /// @param | h | number | Frame height in texture pixels.
29        /// @return | integer | Index of the inserted frame.
30        methods.add_method_mut("addFrame", |_, this, (x, y, w, h): (f32, f32, f32, f32)| {
31            Ok(this.inner.add_frame(Rect::new(x, y, w, h)))
32        });
33        // -- addFramesFromGrid --
34        /// Adds frames by slicing a texture grid.
35        /// @param | tw | integer | Texture width in pixels.
36        /// @param | th | integer | Texture height in pixels.
37        /// @param | fw | integer | Frame width in pixels.
38        /// @param | fh | integer | Frame height in pixels.
39        /// @param | start | integer | Zero-based grid cell index where import begins.
40        /// @param | count | integer | Number of frames to add.
41        /// @return | integer | Number of frames inserted.
42        methods.add_method_mut(
43            "addFramesFromGrid",
44            |_, this, (tw, th, fw, fh, start, count): (u32, u32, u32, u32, usize, usize)| {
45                Ok(this
46                    .inner
47                    .add_frames_from_grid(tw, th, fw, fh, start, count))
48            },
49        );
50        // -- addFramesFromRects --
51        /// Adds frames from an array of rectangle tables.
52        /// @param | rects | table | Array of tables with numeric `x`, `y`, `w`, and `h` fields.
53        /// @return | integer | Number of frames inserted.
54        methods.add_method_mut("addFramesFromRects", |_, this, rects: LuaTable| {
55            let mut quads: Vec<crate::math::Rect> = Vec::new();
56            for entry in rects.sequence_values::<LuaTable>() {
57                let tbl = entry?;
58                let x: f32 = tbl.get("x")?;
59                let y: f32 = tbl.get("y")?;
60                let w: f32 = tbl.get("w")?;
61                let h: f32 = tbl.get("h")?;
62                quads.push(crate::math::Rect::new(x, y, w, h));
63            }
64            Ok(this.inner.add_frames_from_rects(&quads))
65        });
66        // -- addClip --
67        /// Adds a named clip using existing frame indices.
68        /// @param | name | string | Clip name used by playback and state machines.
69        /// @param | indices_tbl | table | Array of frame indices that make up the clip.
70        /// @param | fps | number | Playback speed in frames per second.
71        /// @param | looping | boolean | True when playback should wrap at the end.
72        /// @param | mode | string? | Playback mode `forward`, `reverse`, or `pingpong`; defaults to `forward`.
73        methods.add_method_mut(
74            "addClip",
75            |_,
76             this,
77             (name, indices_tbl, fps, looping, mode): (
78                String,
79                LuaTable,
80                f32,
81                bool,
82                Option<String>,
83            )| {
84                let mut indices: Vec<usize> = Vec::new();
85                for v in indices_tbl.sequence_values::<usize>() {
86                    indices.push(v?);
87                }
88                let mode = parse_clip_mode(mode.as_deref())?;
89                this.inner
90                    .add_clip_with_mode(&name, indices, fps, looping, mode);
91                Ok(())
92            },
93        );
94        // -- setClipMode --
95        /// Changes the playback mode for an existing clip.
96        /// @param | name | string | Clip name to update.
97        /// @param | mode | string | Playback mode `forward`, `reverse`, or `pingpong`.
98        /// @return | boolean | True when the clip exists and the mode was changed.
99        methods.add_method_mut("setClipMode", |_, this, (name, mode): (String, String)| {
100            let mode = parse_clip_mode(Some(mode.as_str()))?;
101            if let Some(clip) = this.inner.get_clip_mut(&name) {
102                clip.mode = mode;
103                Ok(true)
104            } else {
105                Ok(false)
106            }
107        });
108        // -- getClipMode --
109        /// Returns the playback mode name for a clip when it exists.
110        /// @param | name | string | Clip name to query.
111        /// @return | LuaValue | Playback mode string, or nil when the clip does not exist.
112        methods.add_method("getClipMode", |_, this, name: String| {
113            Ok(this
114                .inner
115                .get_clip(&name)
116                .map(|clip| clip_mode_name(clip.mode).to_string()))
117        });
118        // -- addClipFromGrid --
119        /// Adds frames from a texture grid and creates a clip that references the new frames.
120        /// @param | name | string | Clip name to create.
121        /// @param | tw | integer | Texture width in pixels.
122        /// @param | th | integer | Texture height in pixels.
123        /// @param | fw | integer | Frame width in pixels.
124        /// @param | fh | integer | Frame height in pixels.
125        /// @param | start | integer | Zero-based grid cell index where import begins.
126        /// @param | count | integer | Number of frames to add.
127        /// @param | fps | number | Playback speed in frames per second.
128        /// @param | looping | boolean | True when playback should wrap at the end.
129        methods.add_method_mut(
130            "addClipFromGrid",
131            |_,
132             this,
133             (name, tw, th, fw, fh, start, count, fps, looping): (
134                String,
135                u32,
136                u32,
137                u32,
138                u32,
139                usize,
140                usize,
141                f32,
142                bool,
143            )| {
144                this.inner
145                    .add_clip_from_grid(&name, tw, th, fw, fh, start, count, fps, looping);
146                Ok(())
147            },
148        );
149        // -- play --
150        /// Starts playback of a named clip. This method is available to Lua scripts.
151        /// @param | name | string | Clip name to play.
152        /// @return | boolean | True when the clip exists and playback started.
153        methods.add_method_mut("play", |_, this, name: String| Ok(this.inner.play(&name)));
154        // -- stop --
155        /// Stops playback and resets animation playback state.
156        methods.add_method_mut("stop", |_, this, ()| {
157            this.inner.stop();
158            Ok(())
159        });
160        // -- pause --
161        /// Pauses animation playback without changing the current clip.
162        methods.add_method_mut("pause", |_, this, ()| {
163            this.inner.pause();
164            Ok(())
165        });
166        // -- resume --
167        /// Resumes playback of a paused animation.
168        methods.add_method_mut("resume", |_, this, ()| {
169            this.inner.resume();
170            Ok(())
171        });
172        // -- update --
173        /// Advances animation playback and records any frame or clip events.
174        /// @param | dt | number | Elapsed time in seconds.
175        methods.add_method_mut("update", |_, this, dt: f32| {
176            this.inner.update(dt);
177            Ok(())
178        });
179        // -- getQuad --
180        /// Returns the current frame rectangle as a table.
181        /// @return | LuaValue | Table with `x`, `y`, `w`, and `h`, or nil when no frame is active.
182        methods.add_method("getQuad", |lua, this, ()| {
183            if let Some(q) = this.inner.current_quad() {
184                let t = lua.create_table()?;
185                /// The 'x' field value exposed to Lua scripts.
186                t.set("x", q.x)?;
187                /// The 'y' field value exposed to Lua scripts.
188                t.set("y", q.y)?;
189                /// The 'w' field value exposed to Lua scripts.
190                t.set("w", q.width)?;
191                /// The 'h' field value exposed to Lua scripts.
192                t.set("h", q.height)?;
193                Ok(LuaValue::Table(t))
194            } else {
195                Ok(LuaValue::Nil)
196            }
197        });
198        // -- setImage --
199        /// Stores a spritesheet image on this animation so draw can be called without an explicit image argument.
200        /// @param | image | LImage | Texture atlas or spritesheet containing the animation frames.
201        methods.add_method_mut("setImage", |_, this, image_ud: LuaAnyUserData| {
202            let image = image_ud.borrow::<LuaImage>().map_err(|_| {
203                LuaError::RuntimeError("LAnimation:setImage: argument must be LImage".into())
204            })?;
205            this.stored_image = Some(image.clone());
206            Ok(())
207        });
208        // -- draw --
209        /// Draws the current animation frame without advancing playback.
210        /// Accepts two call forms: `draw(image, x, y, opts)` uses the supplied atlas directly;
211        /// `draw(x, y, opts)` uses the image previously stored with `setImage`.
212        /// @param | image | LImage? | Texture atlas or spritesheet; omit when an image was stored with setImage.
213        /// @param | x | number? | Destination X position (default 0).
214        /// @param | y | number? | Destination Y position (default 0).
215        /// @param | opts | table? | Optional transform table with numeric `rotation`, `scale`, `scaleX`, `scaleY`, `originX`, and `originY` fields.
216        /// @return | boolean | True when a frame draw command was queued; false when no current frame is active.
217        methods.add_method("draw", |_, this, args: LuaMultiValue| {
218            draw_animation_frame_from_args("LAnimation:draw", &this.inner, &this.stored_image, args)
219        });
220        // -- pollEvents --
221        /// Drains animation events produced since the previous poll.
222        /// @return | table | Array of event tables with `type` and optional `frame` fields.
223        /// @field | type | string | Event type name.
224        /// @field | frame | integer? | Frame index when available.
225        methods.add_method_mut("pollEvents", |lua, this, ()| {
226            let events = this.inner.drain_events();
227            let tbl = lua.create_table()?;
228            for (i, ev) in events.iter().enumerate() {
229                let ev_tbl = lua.create_table()?;
230                /// Performs the 'type' operation.
231                ev_tbl.set("type", ev.type_name())?;
232                if let Some(idx) = ev.frame_index() {
233                    /// Performs the 'frame' operation.
234                    ev_tbl.set("frame", idx)?;
235                }
236                tbl.set(i + 1, ev_tbl)?;
237            }
238            Ok(tbl)
239        });
240        // -- isPlaying --
241        /// Returns whether this animation is currently playing.
242        /// @return | boolean | True when playback is active.
243        methods.add_method("isPlaying", |_, this, ()| Ok(this.inner.is_playing()));
244        // -- isLooping --
245        /// Returns whether the current clip loops.
246        /// @return | boolean | True when the active clip is looping.
247        methods.add_method("isLooping", |_, this, ()| Ok(this.inner.is_looping()));
248        // -- getClip --
249        /// Returns the current clip name when a clip is active.
250        /// @return | LuaValue | Current clip name, or nil when no clip is active.
251        methods.add_method("getClip", |_, this, ()| {
252            Ok(this.inner.get_current_clip().map(|s| s.to_owned()))
253        });
254        // -- getSpeed --
255        /// Returns the animation playback speed multiplier.
256        /// @return | number | Current playback speed multiplier.
257        methods.add_method("getSpeed", |_, this, ()| Ok(this.inner.get_speed()));
258        // -- setSpeed --
259        /// Sets the animation playback speed multiplier.
260        /// @param | speed | number | Playback speed multiplier used by future updates.
261        methods.add_method_mut("setSpeed", |_, this, speed: f32| {
262            this.inner.set_speed(speed);
263            Ok(())
264        });
265        // -- getFrameCount --
266        /// Returns the number of frame rectangles stored in this animation.
267        /// @return | integer | Frame count.
268        methods.add_method("getFrameCount", |_, this, ()| {
269            Ok(this.inner.get_frame_count())
270        });
271        // -- getClipCount --
272        /// Returns the number of named clips stored in this animation.
273        /// @return | integer | Clip count.
274        methods.add_method("getClipCount", |_, this, ()| {
275            Ok(this.inner.get_clip_count())
276        });
277        // -- getCurrentFrame --
278        /// Returns the current frame index. This method is available to Lua scripts.
279        /// @return | integer | Current frame index.
280        methods.add_method("getCurrentFrame", |_, this, ()| {
281            Ok(this.inner.current_frame())
282        });
283        // -- setFrame --
284        /// Sets the current frame index directly.
285        /// @param | index | integer | Frame index to make current.
286        methods.add_method_mut("setFrame", |_, this, index: usize| {
287            this.inner.set_frame(index);
288            Ok(())
289        });
290        // -- crossfade --
291        /// Starts a crossfade from the current clip to another clip.
292        /// @param | clip_name | string | Destination clip name.
293        /// @param | duration | number | Crossfade duration in seconds.
294        /// @return | boolean | True when the destination clip exists and crossfade started.
295        methods.add_method_mut(
296            "crossfade",
297            |_, this, (clip_name, duration): (String, f32)| {
298                Ok(this.inner.crossfade(&clip_name, duration))
299            },
300        );
301        // -- getBlendState --
302        /// Returns current crossfade rectangles and blend factor when a crossfade is active.
303        /// @return | LuaValue | Table with `from`, `to`, and `blend`, or nil when no blend is active.
304        methods.add_method("getBlendState", |lua, this, ()| {
305            match this.inner.get_blend_state() {
306                None => Ok(LuaValue::Nil),
307                Some((from_q, to_q, blend)) => {
308                    let from = lua.create_table()?;
309                    /// The 'x' field value exposed to Lua scripts.
310                    from.set("x", from_q.x)?;
311                    /// The 'y' field value exposed to Lua scripts.
312                    from.set("y", from_q.y)?;
313                    /// The 'w' field value exposed to Lua scripts.
314                    from.set("w", from_q.width)?;
315                    /// The 'h' field value exposed to Lua scripts.
316                    from.set("h", from_q.height)?;
317                    let to = lua.create_table()?;
318                    /// The 'x' field value exposed to Lua scripts.
319                    to.set("x", to_q.x)?;
320                    /// The 'y' field value exposed to Lua scripts.
321                    to.set("y", to_q.y)?;
322                    /// The 'w' field value exposed to Lua scripts.
323                    to.set("w", to_q.width)?;
324                    /// The 'h' field value exposed to Lua scripts.
325                    to.set("h", to_q.height)?;
326                    let t = lua.create_table()?;
327                    /// Performs the 'from' operation.
328                    t.set("from", from)?;
329                    /// The 'to' field value exposed to Lua scripts.
330                    t.set("to", to)?;
331                    /// Performs the 'blend' operation.
332                    t.set("blend", blend)?;
333                    Ok(LuaValue::Table(t))
334                }
335            }
336        });
337        // -- drawToImage --
338        /// Rasterizes the current animation frame into an image userdata.
339        /// @param | w | integer | Output image width in pixels.
340        /// @param | h | integer | Output image height in pixels.
341        /// @return | LImageData | Image data containing the rendered frame.
342        methods.add_method("drawToImage", |lua, this, (w, h): (u32, u32)| {
343            let img = this.inner.draw_to_image(w, h);
344            lua.create_userdata(img)
345        });
346        // -- drawPreviewGrid --
347        /// Rasterizes all animation frames into a preview grid image.
348        /// @param | columns | integer | Number of columns in the preview grid.
349        /// @param | cell_size | integer | Size of each preview cell in pixels.
350        /// @return | LImageData | Image data containing the preview grid.
351        methods.add_method(
352            "drawPreviewGrid",
353            |lua, this, (columns, cell_size): (u32, u32)| {
354                let img = this.inner.draw_preview_grid(columns, cell_size);
355                lua.create_userdata(img)
356            },
357        );
358        // -- type --
359        /// Returns the Lua-visible type name for this animation handle.
360        /// @return | string | The string `LAnimation`.
361        methods.add_method("type", |_, _, ()| Ok("LAnimation"));
362        // -- typeOf --
363        /// Returns whether this animation handle matches a supported type name.
364        /// @param | name | string | Type name to compare against `LAnimation` and `Object`.
365        /// @return | boolean | True when the supplied type name matches this handle.
366        methods.add_method("typeOf", |_, _, name: String| {
367            Ok(name == "LAnimation" || name == "LObject")
368        });
369    }

371/// Lua-side animation state machine that switches clips from named states and parameters.
372pub struct LuaAnimStateMachine {
373    /// Owned animation state machine exposed through this userdata handle.
374    inner: AnimStateMachine,
375    /// Spritesheet stored by `setImage`; used by `draw` when no image argument is supplied.
376    stored_image: Option<LuaImage>,
377}
378impl LuaUserData for LuaAnimStateMachine {
379    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
380        // -- update --
381        /// Advances the animation state machine and its owned animation playback.
382        /// @param | dt | number | Elapsed time in seconds.
383        methods.add_method_mut("update", |_, this, dt: f32| {
384            this.inner.update(dt);
385            Ok(())
386        });
387        // -- getState --
388        /// Returns the current animation state name.
389        /// @return | string | Current state name.
390        methods.add_method("getState", |_, this, ()| {
391            Ok(this.inner.get_state().to_owned())
392        });
393        // -- forceState --
394        /// Forces the state machine into a named state.
395        /// @param | name | string | State name to activate immediately.
396        /// @return | boolean | True when the state exists and was activated.
397        methods.add_method_mut("forceState", |_, this, name: String| {
398            Ok(this.inner.force_state(&name))
399        });
400        // -- addState --
401        /// Adds a state that plays a named animation clip.
402        /// @param | name | string | State name.
403        /// @param | clip | string | Clip name to play while this state is active.
404        /// @param | looping | boolean | True when the clip should loop in this state.
405        methods.add_method_mut(
406            "addState",
407            |_, this, (name, clip, looping): (String, String, bool)| {
408                this.inner.add_state(&name, &clip, looping);
409                Ok(())
410            },
411        );
412        // -- addTransition --
413        /// Adds a named-condition transition between two animation states.
414        /// @param | from_state | string | Source state name.
415        /// @param | to_state | string | Destination state name.
416        /// @param | condition | string | Parameter condition expression understood by the state machine.
417        methods.add_method_mut(
418            "addTransition",
419            |_, this, (from_state, to_state, condition): (String, String, String)| {
420                this.inner
421                    .add_transition(&from_state, &to_state, &condition);
422                Ok(())
423            },
424        );
425        // -- setParam --
426        /// Sets a boolean, integer, or numeric state machine parameter.
427        /// @param | name | string | Parameter name used by transition conditions.
428        /// @param | value | LuaValue | Boolean, integer, or number value to store.
429        methods.add_method_mut("setParam", |_, this, (name, value): (String, LuaValue)| {
430            let param = match value {
431                LuaValue::Boolean(b) => AnimParamValue::Bool(b),
432                LuaValue::Integer(i) => AnimParamValue::Int(i as i32),
433                LuaValue::Number(f) => AnimParamValue::Float(f as f32),
434                _ => {
435                    return Err(LuaError::RuntimeError(
436                        "setParam: value must be boolean, integer, or number".into(),
437                    ))
438                }
439            };
440            match param {
441                AnimParamValue::Bool(b) => this.inner.set_param_bool(&name, b),
442                AnimParamValue::Int(i) => this.inner.set_param_int(&name, i),
443                AnimParamValue::Float(f) => this.inner.set_param_float(&name, f),
444            }
445            Ok(())
446        });
447        // -- getQuad --
448        /// Returns the current frame rectangle from the state machine's owned animation.
449        /// @return | LuaValue | Table with `x`, `y`, `w`, and `h`, or nil when no frame is active.
450        methods.add_method("getQuad", |lua, this, ()| {
451            let anim = this.inner.get_animation();
452            if let Some(q) = anim.current_quad() {
453                let t = lua.create_table()?;
454                /// The 'x' field value exposed to Lua scripts.
455                t.set("x", q.x)?;
456                /// The 'y' field value exposed to Lua scripts.
457                t.set("y", q.y)?;
458                /// The 'w' field value exposed to Lua scripts.
459                t.set("w", q.width)?;
460                /// The 'h' field value exposed to Lua scripts.
461                t.set("h", q.height)?;
462                Ok(LuaValue::Table(t))
463            } else {
464                Ok(LuaValue::Nil)
465            }
466        });
467        // -- setImage --
468        /// Stores a spritesheet image on this state machine so draw can be called without an explicit image argument.
469        /// @param | image | LImage | Texture atlas or spritesheet containing the animation frames.
470        methods.add_method_mut("setImage", |_, this, image_ud: LuaAnyUserData| {
471            let image = image_ud.borrow::<LuaImage>().map_err(|_| {
472                LuaError::RuntimeError(
473                    "LAnimStateMachine:setImage: argument must be LImage".into(),
474                )
475            })?;
476            this.stored_image = Some(image.clone());
477            Ok(())
478        });
479        // -- draw --
480        /// Draws the current state-machine animation frame without advancing playback.
481        /// Accepts two call forms: `draw(image, x, y, opts)` uses the supplied atlas directly;
482        /// `draw(x, y, opts)` uses the image previously stored with `setImage`.
483        /// @param | image | LImage? | Texture atlas or spritesheet; omit when an image was stored with setImage.
484        /// @param | x | number? | Destination X position (default 0).
485        /// @param | y | number? | Destination Y position (default 0).
486        /// @param | opts | table? | Optional transform table with numeric `rotation`, `scale`, `scaleX`, `scaleY`, `originX`, and `originY` fields.
487        /// @return | boolean | True when a frame draw command was queued; false when no current frame is active.
488        methods.add_method("draw", |_, this, args: LuaMultiValue| {
489            draw_animation_frame_from_args(
490                "LAnimStateMachine:draw",
491                this.inner.get_animation(),
492                &this.stored_image,
493                args,
494            )
495        });
496        // -- type --
497        /// Returns the Lua-visible type name for this animation state machine handle.
498        /// @return | string | The string `LAnimStateMachine`.
499        methods.add_method("type", |_, _, ()| Ok("LAnimStateMachine"));
500        // -- typeOf --
501        /// Returns whether this animation state machine handle matches a supported type name.
502        /// @param | name | string | Type name to compare against `LAnimStateMachine` and `Object`.
503        /// @return | boolean | True when the supplied type name matches this handle.
504        methods.add_method("typeOf", |_, _, name: String| {
505            Ok(name == "LAnimStateMachine" || name == "LObject")
506        });
507    }
508}
509/// Lua-side blend layer set used to combine animation clips with weights and bone masks.
510pub struct LuaBlendLayerSet {
511    /// Owned blend layer set exposed through this userdata handle.
512    inner: BlendLayerSet,
513}
514impl LuaUserData for LuaBlendLayerSet {
515    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
516        // -- addLayer --
517        /// Adds a weighted animation blend layer with an optional bone mask.
518        /// @param | name | string | Unique layer name.
519        /// @param | clip_name | string | Animation clip name used by the layer.
520        /// @param | weight | number | Blend weight for this layer.
521        /// @param | bones | table? | Optional array or map table of bone names included in the mask.
522        /// @return | boolean | True when the layer was added.
523        methods.add_method_mut(
524            "addLayer",
525            |_, this, (name, clip_name, weight, bones): (String, String, f32, Option<LuaTable>)| {
526                let mask = if let Some(t) = bones {
527                    let mut names: Vec<String> = Vec::new();
528                    for pair in t.pairs::<LuaValue, String>() {
529                        let (_, v) = pair?;
530                        names.push(v);
531                    }
532                    BlendMask::from_bones(names)
533                } else {
534                    BlendMask::all()
535                };
536                let layer = BlendLayer::new(&name, &clip_name, weight, mask);
537                this.inner
538                    .add_layer(layer)
539                    .map_err(LuaError::RuntimeError)?;
540                Ok(true)
541            },
542        );
543        // -- removeLayer --
544        /// Removes a blend layer by name. This method is available to Lua scripts.
545        /// @param | name | string | Layer name to remove.
546        /// @return | boolean | True when the layer was removed.
547        methods.add_method_mut("removeLayer", |_, this, name: String| {
548            this.inner
549                .remove_layer(&name)
550                .map_err(LuaError::RuntimeError)?;
551            Ok(true)
552        });
553        // -- setWeight --
554        /// Sets the blend weight for an existing layer.
555        /// @param | name | string | Layer name to update.
556        /// @param | weight | number | New layer weight.
557        /// @return | boolean | True when the layer exists and the weight was changed.
558        methods.add_method_mut("setWeight", |_, this, (name, weight): (String, f32)| {
559            this.inner
560                .set_weight(&name, weight)
561                .map_err(LuaError::RuntimeError)?;
562            Ok(true)
563        });
564        // -- getWeight --
565        /// Returns the weight for a blend layer when it exists.
566        /// @param | name | string | Layer name to query.
567        /// @return | LuaValue | Layer weight, or nil when the layer does not exist.
568        methods.add_method("getWeight", |_, this, name: String| {
569            Ok(this.inner.get_weight(&name))
570        });
571        // -- setMask --
572        /// Replaces a layer bone mask from a table of bone names.
573        /// @param | name | string | Layer name to update.
574        /// @param | bones | table | Array or map table of bone names included in the mask.
575        /// @return | boolean | True when the layer exists and the mask was changed.
576        methods.add_method_mut("setMask", |_, this, (name, bones): (String, LuaTable)| {
577            let mut bone_names: Vec<String> = Vec::new();
578            for pair in bones.pairs::<LuaValue, String>() {
579                let (_, v) = pair?;
580                bone_names.push(v);
581            }
582            this.inner
583                .set_mask(&name, BlendMask::from_bones(bone_names))
584                .map_err(LuaError::RuntimeError)?;
585            Ok(true)
586        });
587        // -- listLayers --
588        /// Returns all blend layers with names, clip names, weights, and bone masks.
589        /// @return | table | Array of layer tables with `name`, `clip_name`, `weight`, and `bones` fields.
590        /// @field | name | string | Layer name.
591        /// @field | clip_name | string | Clip name.
592        /// @field | weight | number | Blend weight.
593        /// @field | bones | string[] | Bone mask names.
594        methods.add_method("listLayers", |lua, this, ()| {
595            let out = lua.create_table()?;
596            for (i, layer) in this.inner.layers().iter().enumerate() {
597                let t = lua.create_table()?;
598                /// Performs the 'name' operation.
599                t.set("name", layer.name.clone())?;
600                /// Performs the 'clip_name' operation.
601                t.set("clip_name", layer.clip_name.clone())?;
602                /// Performs the 'weight' operation.
603                t.set("weight", layer.weight)?;
604                let bones = lua.create_table()?;
605                for (j, b) in layer.mask.bone_names.iter().enumerate() {
606                    bones.set(j + 1, b.clone())?;
607                }
608                /// Performs the 'bones' operation.
609                t.set("bones", bones)?;
610                out.set(i + 1, t)?;
611            }
612            Ok(out)
613        });
614        // -- len --
615        /// Returns the number of blend layers.
616        /// @return | integer | Blend layer count.
617        methods.add_method("len", |_, this, ()| Ok(this.inner.len()));
618        // -- type --
619        /// Returns the Lua-visible type name for this blend layer set handle.
620        /// @return | string | The string `LBlendLayerSet`.
621        methods.add_method("type", |_, _, ()| Ok("LBlendLayerSet"));
622        // -- typeOf --
623        /// Returns whether this blend layer set handle matches a supported type name.
624        /// @param | name | string | Type name to compare against `LBlendLayerSet` and `Object`.
625        /// @return | boolean | True when the supplied type name matches this handle.
626        methods.add_method("typeOf", |_, _, name: String| {
627            Ok(name == "LBlendLayerSet" || name == "LObject")
628        });
629    }
630}
631/// Registers the `lurek.animation` API table with the Lua VM.
632pub fn register(lua: &Lua, luna: &LuaTable, _state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
633    let tbl = lua.create_table()?;
634    // -- new --
635    /// Creates an empty animation with no frames or clips.
636    /// @return | LAnimation | New animation handle.
637    tbl.set(
638        "new",
639        lua.create_function(|lua, ()| {
640            lua.create_userdata(LuaAnimation {
641                inner: Animation::new(),
642                stored_image: None,
643            })
644        })?,
645    )?;
646    // -- fromAseprite --
647    /// Loads an animation from an Aseprite JSON export string.
648    /// @param | json_str | string | Raw Aseprite JSON document contents.
649    /// @return | LuaValue | Animation handle when parsing succeeds; raises an error when the JSON cannot be parsed.
650    tbl.set(
651        "fromAseprite",
652        lua.create_function(
653            |lua, json_str: String| match load_aseprite_json(&json_str) {
654                Ok(parsed) => {
655                    let anim = Animation::load_from_aseprite(&parsed);
656                    let ud = lua.create_userdata(LuaAnimation { inner: anim, stored_image: None })?;
657                    Ok(LuaValue::UserData(ud))
658                }
659                Err(e) => Err(LuaError::RuntimeError(format!("fromAseprite: {}", e))),
660            },
661        )?,
662    )?;
663    // -- newStateMachine --
664    /// Creates an animation state machine by consuming an animation handle.
665    /// @param | anim_ud | LAnimation | Animation handle moved into the state machine.
666    /// @param | initial | string | Initial state name stored in the state machine.
667    /// @return | LAnimStateMachine | New animation state machine handle.
668    tbl.set(
669        "newStateMachine",
670        lua.create_function(|lua, (anim_ud, initial): (LuaAnyUserData, String)| {
671            let anim = anim_ud.take::<LuaAnimation>()?.inner;
672            lua.create_userdata(LuaAnimStateMachine {
673                inner: AnimStateMachine::new(anim, initial),
674                stored_image: None,
675            })
676        })?,
677    )?;
678    // -- newCurve --
679    /// Creates an empty animation curve. This function is exposed to Lua scripts.
680    /// @return | LAnimCurve | New animation curve handle.
681    tbl.set(
682        "newCurve",
683        lua.create_function(|lua, ()| {
684            lua.create_userdata(LuaAnimCurve {
685                inner: crate::animation::curve::AnimCurve::new(),
686                custom_easing: None,
687            })
688        })?,
689    )?;
690    // -- newSyncGroup --
691    /// Creates an empty animation synchronization group.
692    /// @return | LAnimSyncGroup | New animation sync group handle.
693    tbl.set(
694        "newSyncGroup",
695        lua.create_function(|lua, ()| {
696            lua.create_userdata(LuaAnimSyncGroup {
697                inner: crate::animation::sync_group::AnimSyncGroup::new(),
698            })
699        })?,
700    )?;
701    // -- newBlendLayerSet --
702    /// Creates an empty blend layer set for layered animation playback.
703    /// @return | LBlendLayerSet | New blend layer set handle.
704    tbl.set(
705        "newBlendLayerSet",
706        lua.create_function(|lua, ()| {
707            lua.create_userdata(LuaBlendLayerSet {
708                inner: BlendLayerSet::new(),
709            })
710        })?,
711    )?;
712    // -- buildCharacter --
713    /// Builds a character animation bundle from grid frame and clip configuration.
714    /// @param | cfg | table | Configuration table with texture size, frame size, clips, optional states, and optional transitions.
715    /// @return | table | Table containing `animation` and, when states are supplied, `stateMachine` handles.
716    /// @field | animation | LAnimation | Animation handle.
717    /// @field | stateMachine | LStateMachine | State machine handle.
718    tbl.set(
719        "buildCharacter",
720        lua.create_function(|lua, cfg: LuaTable| {
721            let tex_w: u32 = cfg.get("texW")?;
722            let tex_h: u32 = cfg.get("texH")?;
723            let frame_w: u32 = cfg.get("frameW")?;
724            let frame_h: u32 = cfg.get("frameH")?;
725            let mut anim = Animation::new();
726            let clips: LuaTable = cfg.get("clips")?;
727            for clip_tbl in clips.sequence_values::<LuaTable>() {
728                let clip_tbl = clip_tbl?;
729                let name: String = clip_tbl.get("name")?;
730                let start: usize = clip_tbl.get("start")?;
731                let count: usize = clip_tbl.get("count")?;
732                let fps: f32 = clip_tbl.get::<_, Option<f32>>("fps")?.unwrap_or(8.0);
733                let looping: bool = clip_tbl.get::<_, Option<bool>>("looping")?.unwrap_or(true);
734                let mode = parse_clip_mode(clip_tbl.get::<_, Option<String>>("mode")?.as_deref())?;
735                let base = anim.get_frame_count();
736                let added = anim.add_frames_from_grid(tex_w, tex_h, frame_w, frame_h, start, count);
737                let indices: Vec<usize> = (base..base + added).collect();
738                anim.add_clip_with_mode(&name, indices, fps, looping, mode);
739            }
740            let initial_clip = cfg.get::<_, Option<String>>("initialClip")?;
741            if let Some(name) = initial_clip {
742                let _ = anim.play(&name);
743            }
744            let out = lua.create_table()?;
745            let anim_clone_for_sm = anim.clone();
746            let anim_ud = lua.create_userdata(LuaAnimation { inner: anim, stored_image: None })?;
747            /// Performs the 'animation' operation.
748            out.set("animation", anim_ud)?;
749            if let Some(states) = cfg.get::<_, Option<LuaTable>>("states")? {
750                let initial_state = cfg
751                    .get::<_, Option<String>>("initialState")?
752                    .unwrap_or_else(|| "idle".to_string());
753                let mut sm = AnimStateMachine::new(anim_clone_for_sm, initial_state);
754                for state_tbl in states.sequence_values::<LuaTable>() {
755                    let state_tbl = state_tbl?;
756                    let name: String = state_tbl.get("name")?;
757                    let clip: String = state_tbl.get("clip")?;
758                    let looping: bool =
759                        state_tbl.get::<_, Option<bool>>("looping")?.unwrap_or(true);
760                    sm.add_state(&name, &clip, looping);
761                }
762                if let Some(transitions) = cfg.get::<_, Option<LuaTable>>("transitions")? {
763                    for t in transitions.sequence_values::<LuaTable>() {
764                        let t = t?;
765                        let from: String = t.get("from")?;
766                        let to: String = t.get("to")?;
767                        let condition: String = t.get("condition")?;
768                        sm.add_transition(&from, &to, &condition);
769                    }
770                }
771                let sm_ud = lua.create_userdata(LuaAnimStateMachine { inner: sm, stored_image: None })?;
772                /// Performs the 'stateMachine' operation.
773                out.set("stateMachine", sm_ud)?;
774            }
775            Ok(out)
776        })?,
777    )?;
778    /// Performs the 'animation' operation.
779    luna.set("animation", tbl)?;
780    Ok(())
781}

783/// Parsed transform options for animation draw helpers.
784struct AnimationDrawOptions {
785    x: f32,
786    y: f32,
787    rotation: f32,
788    sx: f32,
789    sy: f32,
790    ox: f32,
791    oy: f32,

794/// Extract optional draw transform values from Lua arguments.
795fn parse_animation_draw_options(
796    method_name: &str,
797    x: Option<f32>,
798    y: Option<f32>,
799    opts: Option<LuaTable>,
800) -> LuaResult<AnimationDrawOptions> {
801    let mut out = AnimationDrawOptions {
802        x: x.unwrap_or(0.0),
803        y: y.unwrap_or(0.0),
804        rotation: 0.0,
805        sx: 1.0,
806        sy: 1.0,
807        ox: 0.0,
808        oy: 0.0,
809    };
810    if let Some(opts) = opts {
811        out.rotation = optional_number_field(method_name, &opts, "rotation")?.unwrap_or(0.0);
812        let scale = optional_number_field(method_name, &opts, "scale")?.unwrap_or(1.0);
813        out.sx = optional_number_field(method_name, &opts, "scaleX")?.unwrap_or(scale);
814        out.sy = optional_number_field(method_name, &opts, "scaleY")?.unwrap_or(scale);
815        out.ox = optional_number_field(method_name, &opts, "originX")?.unwrap_or(0.0);
816        out.oy = optional_number_field(method_name, &opts, "originY")?.unwrap_or(0.0);
817    }
818    Ok(out)
819}

821/// Read an optional numeric field from a Lua options table.
822fn optional_number_field(
823    method_name: &str,
824    opts: &LuaTable,
825    field: &str,

828        LuaValue::Nil => Ok(None),

830        LuaValue::Number(n) => Ok(Some(n as f32)),
831        other => Err(LuaError::RuntimeError(format!(
832            "{method_name}: option '{field}' must be a number, got {}",
833            other.type_name()
834        ))),
835    }
836}

838/// Queue a draw command for an animation's current frame using a resolved `LuaImage`, returning false when no frame is active.
839fn draw_animation_frame(
840    method_name: &str,
841    animation: &Animation,
842    image: &LuaImage,
843    x: Option<f32>,
844    y: Option<f32>,
845    opts: Option<LuaTable>,
846) -> LuaResult<bool> {
847    let texture_key = image.key;
848    let state = image.state.clone();
849    let options = parse_animation_draw_options(method_name, x, y, opts)?;
850    let (tex_w, tex_h) = {
851        let st = state.borrow();
852        let texture = st.textures.get(texture_key).ok_or_else(|| {
853            LuaError::RuntimeError(format!(
854                "{method_name}: image handle is not valid or was released"
855            ))
856        })?;
857        (texture.width as f32, texture.height as f32)
858    };
859    let params = AnimRenderParams {
860        texture_key,
861        tex_w,
862        tex_h,
863        x: options.x,
864        y: options.y,
865        rotation: options.rotation,
866        sx: options.sx,
867        sy: options.sy,
868        ox: options.ox,
869        oy: options.oy,
870    };
871    let Some(command) = animation.generate_render_command(&params) else {
872        return Ok(false);
873    };
874    state.borrow_mut().render_commands.push(command);
875    Ok(true)
876}

878/// Dispatch a draw call by inspecting the first variadic argument.
879/// If arg[0] is `LImage` userdata the explicit-image form is used; otherwise the stored image is used.
880fn draw_animation_frame_from_args(
881    method_name: &str,
882    animation: &Animation,
883    stored_image: &Option<LuaImage>,
884    args: LuaMultiValue,
885) -> LuaResult<bool> {
886    let mut iter = args.into_iter();
887    let first = iter.next();
888    let is_image_ud = matches!(&first, Some(LuaValue::UserData(_)));
889    if is_image_ud {
890        let ud = match first {
891            Some(LuaValue::UserData(ud)) => ud,
892            _ => unreachable!(),
893        };
894        let image = ud.borrow::<LuaImage>().map_err(|_| {
895            LuaError::RuntimeError(format!(
896                "{method_name}: first argument must be LImage, a number, or nil"
897            ))
898        })?.clone();
899        let x = parse_opt_coord(method_name, "x", iter.next().as_ref())?;
900        let y = parse_opt_coord(method_name, "y", iter.next().as_ref())?;
901        let opts = parse_opt_table_arg(method_name, "opts", iter.next())?;
902        draw_animation_frame(method_name, animation, &image, x, y, opts)
903    } else {
904        let image = stored_image.as_ref().ok_or_else(|| {
905            LuaError::RuntimeError(format!(
906                "{method_name}: no image stored; call setImage first or pass an LImage as the first argument"
907            ))
908        })?.clone();
909        let x = parse_opt_coord(method_name, "x", first.as_ref())?;
910        let y = parse_opt_coord(method_name, "y", iter.next().as_ref())?;
911        let opts = parse_opt_table_arg(method_name, "opts", iter.next())?;
912        draw_animation_frame(method_name, animation, &image, x, y, opts)
913    }
914}

923            "{method_name}: '{field}' must be a number, got {}",

945/// Parses a Lua clip mode string and returns the matching playback mode.

964/// Lua-side animation curve with keyframes and optional custom easing callback.
965pub struct LuaAnimCurve {
966    /// Owned curve data exposed through this userdata handle.
967    inner: crate::animation::curve::AnimCurve,
968    /// Optional Lua registry key used for custom easing evaluation.
969    custom_easing: Option<LuaRegistryKey>,
970}
971impl LuaUserData for LuaAnimCurve {
972    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
973        // -- addKeyframe --
974        /// Adds a keyframe to the curve. This method is available to Lua scripts.
975        /// @param | t | number | Keyframe time or normalized position.
976        /// @param | v | number | Keyframe value.
977        methods.add_method_mut("addKeyframe", |_, this, (t, v): (f32, f32)| {
978            this.inner.add_keyframe(t, v);
979            Ok(())
980        });
981        // -- eval --
982        /// Evaluates the curve at a time or normalized position.
983        /// @param | t | number | Time or normalized position to evaluate.
984        /// @return | number | Interpolated curve value.
985        methods.add_method("eval", |lua, this, t: f32| {
986            if let Some(key) = &this.custom_easing {
987                let func: mlua::Function = lua.registry_value(key)?;
988                let v: f64 = func.call(t as f64)?;
989                return Ok(v as f32);
990            }
991            Ok(this.inner.eval(t))
992        });
993        // -- setEasing --
994        /// Sets the built-in easing mode used between keyframes.
995        /// @param | mode | string | Easing mode `step`, `linear`, `ease_in`, `ease_out`, or `ease_in_out`.
996        methods.add_method_mut("setEasing", |_, this, mode: String| {
997            use crate::animation::curve::EasingKind;
998            this.inner.easing = match mode.as_str() {
999                "step" => EasingKind::Step,
1000                "linear" => EasingKind::Linear,
1001                "ease_in" => EasingKind::EaseIn,
1002                "ease_out" => EasingKind::EaseOut,
1003                "ease_in_out" => EasingKind::EaseInOut,
1004                other => {
1005                    return Err(LuaError::RuntimeError(format!(
1006                        "unknown easing mode '{other}' — expected step|linear|ease_in|ease_out|ease_in_out"
1007                    )));
1008                }
1009            };
1010            Ok(())
1011        });
1012        // -- keyframeCount --
1013        /// Returns the number of keyframes stored in this curve.
1014        /// @return | integer | Keyframe count.
1015        methods.add_method("keyframeCount", |_, this, ()| {
1016            Ok(this.inner.keyframe_count())
1017        });
1018        // -- setCustomEasing --
1019        /// Sets or clears a Lua callback used to evaluate custom easing.
1020        /// @param | func | function | Function used as custom easing callback, or nil to clear custom easing.
1021        methods.add_method_mut("setCustomEasing", |lua, this, func: LuaValue| {
1022            use crate::animation::curve::EasingKind;
1023            if let Some(old_key) = this.custom_easing.take() {
1024                lua.remove_registry_value(old_key)?;
1025            }
1026            match func {
1027                LuaValue::Function(f) => {
1028                    let key = lua.create_registry_value(f)?;
1029                    this.custom_easing = Some(key);
1030                    this.inner.easing = EasingKind::Custom { callback_id: 0 };
1031                }
1032                LuaValue::Nil => {
1033                    this.inner.easing = EasingKind::Linear;
1034                }
1035                _ => {
1036                    return Err(LuaError::RuntimeError(
1037                        "setCustomEasing: expected function or nil".into(),
1038                    ))
1039                }
1040            }
1041            Ok(())
1042        });
1043        // -- clear --
1044        /// Removes all keyframes from this curve.
1045        methods.add_method_mut("clear", |_, this, ()| {
1046            this.inner.clear();
1047            Ok(())
1048        });
1049        // -- type --
1050        /// Returns the Lua-visible type name for this animation curve handle.
1051        /// @return | string | The string `LAnimCurve`.
1052        methods.add_method("type", |_, _, ()| Ok("LAnimCurve"));
1053        // -- typeOf --
1054        /// Returns whether this animation curve handle matches a supported type name.
1055        /// @param | name | string | Type name to compare against `LAnimCurve` and `Object`.
1056        /// @return | boolean | True when the supplied type name matches this handle.
1057        methods.add_method("typeOf", |_, _, name: String| {
1058            Ok(name == "LAnimCurve" || name == "LObject")
1059        });
1060    }
1061}
1062/// Lua-side animation synchronization group for coordinating multiple animation handles.
1063pub struct LuaAnimSyncGroup {
1064    /// Owned sync group data exposed through this userdata handle.
1065    inner: crate::animation::sync_group::AnimSyncGroup,
1066}
1067impl LuaUserData for LuaAnimSyncGroup {
1068    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
1069        // -- add --
1070        /// Adds an animation-like handle to the sync group.
1071        /// @param | handle | integer | Animation handle accepted by future sync group implementations.
1072        methods.add_method_mut("add", |_, _this, _handle: LuaValue| Ok(()));
1073        // -- remove --
1074        /// Removes an animation-like handle from the sync group.
1075        /// @param | handle | integer | Animation handle accepted by future sync group implementations.
1076        methods.add_method_mut("remove", |_, _this, _handle: LuaValue| Ok(()));
