1//! `lurek.visibility` -- Lua bindings for universal fog-of-war, discovery, and line-of-sight.

3use super::SharedState;
4use crate::visibility::{FogConfig, VisibilityEvent, VisibilityFlags, VisibilityGrid};
5use mlua::prelude::*;
6use std::cell::RefCell;
7use std::rc::Rc;

9/// Lua-side wrapper for a visibility grid instance.
10struct LuaVisibilityGrid {
11    inner: RefCell<VisibilityGrid>,
12}
13
14impl LuaUserData for LuaVisibilityGrid {
15    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
16        // -- reveal --
17        /// Reveals a region for a player (and their allies). Optional flags argument.
18        /// @param | player_id | integer | Player index (0-based).
19        /// @param | region_id | integer | Region index (0-based).
20        /// @param | flags | integer? | Optional bitfield flags to set on the region.
21        methods.add_method("reveal", |_, this, (player_id, region_id, flags): (u32, u32, Option<u64>)| {
22            let reveal_flags = VisibilityFlags(flags.unwrap_or(0));
23            this.inner.borrow_mut().reveal(player_id, region_id, reveal_flags);
24            Ok(())
25        });
26
27        // -- hide --
28        /// Hides a region for a player (moves from Visible to Discovered).
29        /// @param | player_id | integer | Player index (0-based).
30        /// @param | region_id | integer | Region index (0-based).
31        methods.add_method("hide", |_, this, (player_id, region_id): (u32, u32)| {
32            this.inner.borrow_mut().hide(player_id, region_id);
33            Ok(())
34        });
35
36        // -- getState --
37        /// Gets the visibility state for a player at a region.
38        /// @param | player_id | integer | Player index (0-based).
39        /// @param | region_id | integer | Region index (0-based).
40        /// @return | string | "hidden", "discovered", "visible", or a number for custom levels.
41        methods.add_method("getState", |_, this, (player_id, region_id): (u32, u32)| {
42            let state = this.inner.borrow().get_state(player_id, region_id);
43            let result = match state {
44                crate::visibility::VisibilityState::Hidden => "hidden".to_string(),
45                crate::visibility::VisibilityState::Discovered => "discovered".to_string(),
46                crate::visibility::VisibilityState::Visible => "visible".to_string(),
47                crate::visibility::VisibilityState::Custom(v) => v.to_string(),
48            };
49            Ok(result)
50        });
51
52        // -- getFogIntensity --
53        /// Gets the fog intensity for a region from a player's perspective.
54        /// @param | player_id | integer | Player index (0-based).
55        /// @param | region_id | integer | Region index (0-based).
56        /// @return | number | Fog intensity from 0.0 (clear) to 1.0 (fully fogged).
57        methods.add_method("getFogIntensity", |_, this, (player_id, region_id): (u32, u32)| {
58            Ok(this.inner.borrow().get_fog_intensity(player_id, region_id))
59        });
60
61        // -- setCost --
62        /// Sets the discovery cost for a region.
63        /// @param | region_id | integer | Region index (0-based).
64        /// @param | cost | number | Discovery cost value.
65        methods.add_method("setCost", |_, this, (region_id, cost): (u32, f32)| {
66            this.inner.borrow_mut().set_cost(region_id, cost);
67            Ok(())
68        });
69
70        // -- getCost --
71        /// Gets the discovery cost for a region.
72        /// @param | region_id | integer | Region index (0-based).
73        /// @return | number | Discovery cost value.
74        methods.add_method("getCost", |_, this, region_id: u32| {
75            Ok(this.inner.borrow().get_cost(region_id))
76        });
77
78        // -- setFlag --
79        /// Sets a visibility flag bit on a region.
80        /// @param | region_id | integer | Region index (0-based).
81        /// @param | bit | integer | Flag bit index (0-63).
82        /// @param | value | boolean | Whether to set or clear the bit.
83        methods.add_method("setFlag", |_, this, (region_id, bit, value): (u32, u8, bool)| {
84            let mut grid = this.inner.borrow_mut();
85            let mut flags = grid.get_flags(region_id);
86            flags.set(bit, value);
87            grid.set_flags(region_id, flags);
88            Ok(())
89        });
90
91        // -- hasFlag --
92        /// Checks if a visibility flag bit is set on a region.
93        /// @param | region_id | integer | Region index (0-based).
94        /// @param | bit | integer | Flag bit index (0-63).
95        /// @return | boolean | Whether the bit is set.
96        methods.add_method("hasFlag", |_, this, (region_id, bit): (u32, u8)| {
97            Ok(this.inner.borrow().get_flags(region_id).has(bit))
98        });
99
100        // -- setGroup --
101        /// Sets an alliance group for a list of players (shared visibility).
102        /// @param | players | table | Array of player IDs (0-based) to group together.
103        /// @return | integer | The assigned group ID.
104        methods.add_method("setGroup", |_, this, players: LuaTable| {
105            let player_ids: Vec<u32> = players
106                .sequence_values::<u32>()
107                .filter_map(|r| r.ok())
108                .collect();
109            let gid = this.inner.borrow_mut().set_group(&player_ids);
110            Ok(gid)
111        });
112
113        // -- sharesVisibility --
114        /// Checks if two players share visibility (same alliance group or same player).
115        /// @param | player_a | integer | First player index (0-based).
116        /// @param | player_b | integer | Second player index (0-based).
117        /// @return | boolean | Whether they share visibility.
118        methods.add_method("sharesVisibility", |_, this, (a, b): (u32, u32)| {
119            Ok(this.inner.borrow().shares_visibility(a, b))
120        });
121
122        // -- revealAll --
123        /// Reveals all regions for a player (debug/cheat).
124        /// @param | player_id | integer | Player index (0-based).
125        methods.add_method("revealAll", |_, this, player_id: u32| {
126            this.inner.borrow_mut().reveal_all(player_id);
127            Ok(())
128        });
129
130        // -- reset --
131        /// Resets all visibility to Hidden for a player.
132        /// @param | player_id | integer | Player index (0-based).
133        methods.add_method("reset", |_, this, player_id: u32| {
134            this.inner.borrow_mut().reset(player_id);
135            Ok(())
136        });
137
138        // -- drainEvents --
139        /// Drains and returns all pending visibility events.
140        /// @return | table | Array of event tables with `type`, `player_id`, and `region_id` fields.
141        methods.add_method("drainEvents", |lua, this, ()| {
142            let events = this.inner.borrow_mut().drain_events();
143            let result = lua.create_table()?;
144            for (i, event) in events.iter().enumerate() {
145                let tbl = lua.create_table()?;
146                match event {
147                    VisibilityEvent::Revealed { player_id, region_id } => {
148                        tbl.set("type", "revealed")?;
149                        tbl.set("player_id", *player_id)?;
150                        tbl.set("region_id", *region_id)?;
151                    }
152                    VisibilityEvent::Hidden { player_id, region_id } => {
153                        tbl.set("type", "hidden")?;
154                        tbl.set("player_id", *player_id)?;
155                        tbl.set("region_id", *region_id)?;
156                    }
157                    VisibilityEvent::Forgotten { player_id, region_id } => {
158                        /// Event type string for this forgotten visibility event.
159                        tbl.set("type", "forgotten")?;
160                        /// Player index affected by this forgotten visibility event.
161                        tbl.set("player_id", *player_id)?;
162                        /// Region index hidden from this player.
163                        tbl.set("region_id", *region_id)?;
164                    }
165                    VisibilityEvent::GroupChanged { player_id, group_id } => {
166                        /// Event type string for this group-changed visibility event.
167                        tbl.set("type", "group_changed")?;
168                        /// Player index whose group membership changed.
169                        tbl.set("player_id", *player_id)?;
170                        /// New group id assigned to the player.
171                        tbl.set("group_id", *group_id)?;
172                    }
173                }
174                result.set(i + 1, tbl)?;
175            }
176            Ok(result)
177        });
178
179        // -- regionCount --
180        /// Returns the total number of regions in the grid.
181        /// @return | integer | Region count.
182        methods.add_method("regionCount", |_, this, ()| {
183            Ok(this.inner.borrow().region_count())
184        });
185
186        // -- playerCount --
187        /// Returns the total number of players in the grid.
188        /// @return | integer | Player count.
189        methods.add_method("playerCount", |_, this, ()| {
190            Ok(this.inner.borrow().player_count())
191        });
192    }
193}

195/// Registers the `lurek.visibility` module into the Lua `lurek` table.
196pub fn register(lua: &Lua, lurek: &LuaTable, _state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
197    let tbl = lua.create_table()?;
198
199    // -- new --
200    /// Create a new visibility grid for shadow-cast computation.
201    /// @param | config | table | Configuration table with `regions` (integer) and `players` (integer) fields. Optional `fog` sub-table with `discovered` (number), `hidden` (number), `smooth` (boolean), `speed` (number).
202    /// @return | LVisibilityGrid | New visibility grid handle.
203    tbl.set(
204        "new",
205        lua.create_function(|_, config: LuaTable| {
206            let regions: u32 = config.get("regions")?;
207            let players: u32 = config.get("players")?;
208
209            let mut grid = VisibilityGrid::new(regions, players);
210
211            if let Ok(fog_tbl) = config.get::<_, LuaTable>("fog") {
212                let discovered = fog_tbl.get::<_, f32>("discovered").unwrap_or(0.5);
213                let hidden = fog_tbl.get::<_, f32>("hidden").unwrap_or(1.0);
214                let smooth = fog_tbl.get::<_, bool>("smooth").unwrap_or(true);
215                let speed = fog_tbl.get::<_, f32>("speed").unwrap_or(0.5);
216                grid.set_fog_config(FogConfig {
217                    discovered_intensity: discovered,
218                    hidden_intensity: hidden,
219                    smooth_transitions: smooth,
220                    transition_speed: speed,
221                });
222            }
223
224            Ok(LuaVisibilityGrid {
225                inner: RefCell::new(grid),
226            })
227        })?,
228    )?;
229
230    lurek.set("visibility", tbl)?;
231    Ok(())
232}
