11#[derive(Clone)]
12/// Lua-side HTML document handle with DOM state, callbacks, and render command access.
13struct LuaHtmlDocument {
14    /// Parsed document, layout tree, dirty state, and input state.
15    inner: Rc<RefCell<HtmlDocument>>,
16    /// Document and element listener registry state.
17    callbacks: Rc<RefCell<HtmlCallbacks>>,
18    /// Shared runtime state receiving HTML render commands.
19    state: Rc<RefCell<SharedState>>,
20}
21impl LuaHtmlDocument {
22    /// Wraps a document with callback state and shared runtime access.
23    fn new(document: HtmlDocument, state: Rc<RefCell<SharedState>>) -> Self {
24        Self {
25            inner: Rc::new(RefCell::new(document)),
26            callbacks: Rc::new(RefCell::new(HtmlCallbacks::default())),
27            state,
28        }
29    }
30    /// Creates an element handle tied to the document generation at lookup time.
31    fn element_handle(&self, element_id: HtmlElementId) -> LuaHtmlElement {
32        LuaHtmlElement {
33            document: self.inner.clone(),
34            callbacks: self.callbacks.clone(),
35            state: self.state.clone(),
36            element_id,
37            generation: self.inner.borrow().generation(),
38        }
39    }
40}
41#[derive(Clone)]
42/// Lua-side DOM element handle with stale-generation detection.
43struct LuaHtmlElement {
44    /// Shared document containing this element.
45    document: Rc<RefCell<HtmlDocument>>,
46    /// Shared callback registry for document and element listeners.
47    callbacks: Rc<RefCell<HtmlCallbacks>>,
48    /// Shared runtime state used when returning document handles.
49    state: Rc<RefCell<SharedState>>,
50    /// Element id inside the document.
51    element_id: HtmlElementId,
52    /// Document generation captured when this handle was created.
53    generation: u64,
54}
55#[derive(Default)]
56/// Stores HTML event callbacks and the next listener handle.
57struct HtmlCallbacks {
58    /// Next numeric listener handle exposed by the lurek engine.
59    next_handle: u64,
60    /// Document-level listeners keyed by listener handle.
61    document: HashMap<u64, HtmlListener>,
62    /// Element-level listeners keyed by element id and listener handle.
63    elements: HashMap<HtmlElementId, HashMap<u64, HtmlListener>>,
64}
65/// Stores one Lua event listener registry key and event name.
66struct HtmlListener {
67    /// Event name matched during dispatch.
68    event: String,
69    /// Lua registry key for the callback function.
70    key: LuaRegistryKey,
71}
72#[derive(Default)]
73/// Mutable event dispatch flags shared with event table helper functions.
74struct HtmlEventFlags {
75    /// Whether the callback requested default prevention.
76    default_prevented: bool,
77    /// Whether the callback requested propagation stop.
78    propagation_stopped: bool,
79}
80#[derive(Default)]
81/// Optional event payload fields inserted into Lua event tables.
82struct HtmlEventPayload {
83    /// Mouse x coordinate when available.
84    x: Option<f32>,
85    /// Mouse y coordinate when available.
86    y: Option<f32>,
87    /// Mouse button index when available.
88    button: Option<u32>,
89    /// Keyboard key string when available.
90    key: Option<String>,
91    /// Text input string when available.
92    text: Option<String>,
93    /// Element value string after input events when available.
94    value: Option<String>,
95}

98    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
99        // -- setHtml --
100        /// Replaces the document markup and invalidates existing element handles.
101        /// @param | html | string | New HTML markup.
102        methods.add_method("setHtml", |_, this, html: String| {
103            this.inner.borrow_mut().set_html(html);
104            Ok(())
105        });
106        // -- getHtml --
107        /// Returns the current document markup string.
108        /// @return | string | Current HTML markup.
109        methods.add_method("getHtml", |_, this, ()| {
110            Ok(this.inner.borrow().get_html().to_string())
111        });
112        // -- setCss --
113        /// Replaces the document stylesheet text.
114        /// @param | css | string | CSS source text.
115        methods.add_method("setCss", |_, this, css: String| {
116            this.inner.borrow_mut().set_css(css);
117            Ok(())
118        });
119        // -- addCss --
120        /// Appends CSS source text to the document stylesheet.
121        /// @param | css | string | CSS source text to append.
122        methods.add_method("addCss", |_, this, css: String| {
123            this.inner.borrow_mut().add_css(css);
124            Ok(())
125        });
126        // -- clearCss --
127        /// Clears all CSS source text from the document.
128        methods.add_method("clearCss", |_, this, ()| {
129            this.inner.borrow_mut().clear_css();
130            Ok(())
131        });
132        // -- setViewport --
133        /// Sets the document layout viewport size.
134        /// @param | w | number | Viewport width in pixels.
135        /// @param | h | number | Viewport height in pixels.
136        methods.add_method("setViewport", |_, this, (w, h): (f32, f32)| {
137            this.inner.borrow_mut().set_viewport(w, h);
138            Ok(())
139        });
140        // -- getViewport --
141        /// Returns the document layout viewport size.
142        /// @return | number | Viewport width in pixels.
143        /// @return | number | Viewport height in pixels.
144        methods.add_method("getViewport", |_, this, ()| {
145            Ok(this.inner.borrow().viewport())
146        });
147        // -- update --
148        /// Advances document timers and animated state.
149        /// @param | dt | number | Delta time in seconds.
150        methods.add_method("update", |_, this, dt: f32| {
151            this.inner.borrow_mut().update(dt);
152            Ok(())
153        });
154        // -- draw --
155        /// Queues render commands for this document at an optional offset.
156        /// @param | x | number? | X offset, defaulting to 0.
157        /// @param | y | number? | Y offset, defaulting to 0.
158        methods.add_method("draw", |_, this, (x, y): (Option<f32>, Option<f32>)| {
159            let dx = x.unwrap_or(0.0);
160            let dy = y.unwrap_or(0.0);
161            let mut document = this.inner.borrow_mut();
162            let mut state = this.state.borrow_mut();
163            enqueue_html_draw_commands(&mut document, &mut state, dx, dy);
164            Ok(())
165        });
166        // -- render --
167        /// Queues render commands for this document at an optional offset.
168        /// @param | x | number? | X offset, defaulting to 0.
169        /// @param | y | number? | Y offset, defaulting to 0.
170        methods.add_method("render", |_, this, (x, y): (Option<f32>, Option<f32>)| {
171            let dx = x.unwrap_or(0.0);
172            let dy = y.unwrap_or(0.0);
173            let mut document = this.inner.borrow_mut();
174            let mut state = this.state.borrow_mut();
175            enqueue_html_draw_commands(&mut document, &mut state, dx, dy);
176            Ok(())
177        });
178        // -- relayout --
179        /// Rebuilds document layout immediately.
180        methods.add_method("relayout", |_, this, ()| {
181            this.inner.borrow_mut().relayout();
182            Ok(())
183        });
184        // -- isDirty --
185        /// Returns whether the document layout is dirty.
186        /// @return | boolean | True when a relayout is needed.
187        methods.add_method("isDirty", |_, this, ()| Ok(this.inner.borrow().is_dirty()));
188        // -- getRoot --
189        /// Returns the root DOM element handle.
190        /// @return | LHtmlElement | Root element handle.
191        methods.add_method("getRoot", |lua, this, ()| {
192            lua.create_userdata(this.element_handle(this.inner.borrow().root()))
193        });
194        // -- getElementById --
195        /// Looks up the first element with a matching id attribute.
196        /// @param | id | string | Element id attribute.
197        /// @return | LuaValue | `LHtmlElement` handle, or nil when no element matches.
198        methods.add_method("getElementById", |lua, this, id: String| {
199            html_element_value(lua, this, this.inner.borrow().get_element_by_id(&id))
200        });
201        // -- query --
202        /// Looks up the first element matching a selector.
203        /// @param | selector | string | Selector supported by the HTML engine.
204        /// @return | LuaValue | `LHtmlElement` handle, or nil when no element matches.
205        methods.add_method("query", |lua, this, selector: String| {
206            html_element_value(lua, this, this.inner.borrow().query(&selector))
207        });
208        // -- queryAll --
209        /// Returns all elements matching a selector.
210        /// @param | selector | string | Selector supported by the HTML engine.
211        /// @return | LHtmlElement[] | `LHtmlElement` handles.
212        methods.add_method("queryAll", |lua, this, selector: String| {
213            html_element_table(lua, this, this.inner.borrow().query_all(&selector))
214        });
215        // -- on --
216        /// Registers a document-level event listener.
217        /// @param | event | string | Event name to listen for.
218        /// @param | func | function | Lua callback receiving an event table.
219        /// @return | integer | Listener handle used by `off`.
220        methods.add_method("on", |lua, this, (event, func): (String, LuaFunction)| {
221            let key = lua.create_registry_value(func)?;
222            let mut callbacks = this.callbacks.borrow_mut();
223            let handle = callbacks.next_handle.saturating_add(1);
224            callbacks.next_handle = handle;
225            callbacks
226                .document
227                .insert(handle, HtmlListener { event, key });
228            Ok(handle)
229        });
230        // -- off --
231        /// Removes a document-level event listener by handle.
232        /// @param | handle | integer | Listener handle returned by `on`.
233        methods.add_method("off", |lua, this, handle: u64| {
234            if let Some(listener) = this.callbacks.borrow_mut().document.remove(&handle) {
235                lua.remove_registry_value(listener.key)?;
236            }
237            Ok(())
238        });
239        // -- mousepressed --
240        /// Forwards a mouse press to the document and dispatches a click event when an element is hit.
241        /// @param | x | number | Mouse x coordinate.
242        /// @param | y | number | Mouse y coordinate.
243        /// @param | button | integer? | Mouse button, defaulting to 1.
244        /// @return | boolean | True when the event was consumed or default was prevented.
245        methods.add_method(
246            "mousepressed",
247            |lua, this, (x, y, button): (f32, f32, Option<u32>)| {
248                let button = button.unwrap_or(1);
249                let target = this.inner.borrow_mut().mouse_pressed(x, y, button);
250                let consumed = target.is_some();
251                let prevented = emit_html_event(
252                    lua,
253                    this,
254                    "click",
255                    target,
256                    HtmlEventPayload {
257                        x: Some(x),
258                        y: Some(y),
259                        button: Some(button),
260                        ..HtmlEventPayload::default()
261                    },
262                )?;
263                Ok(consumed || prevented)
264            },
265        );
266        // -- mousereleased --
267        /// Forwards a mouse release to the document.
268        /// @param | x | number | Mouse x coordinate.
269        /// @param | y | number | Mouse y coordinate.
270        /// @param | button | integer? | Mouse button, defaulting to 1.
271        /// @return | boolean | True when an element handled the release.
272        methods.add_method(
273            "mousereleased",
274            |_, this, (x, y, button): (f32, f32, Option<u32>)| {
275                Ok(this
276                    .inner
277                    .borrow_mut()
278                    .mouse_released(x, y, button.unwrap_or(1))
279                    .is_some())
280            },
281        );
282        // -- mousemoved --
283        /// Forwards mouse movement to the document.
284        /// @param | x | number | Mouse x coordinate.
285        /// @param | y | number | Mouse y coordinate.
286        /// @return | boolean | True when an element handled the move.
287        methods.add_method("mousemoved", |_, this, (x, y): (f32, f32)| {
288            Ok(this.inner.borrow_mut().mouse_moved(x, y).is_some())
289        });
290        // -- wheelmoved --
291        /// Forwards mouse wheel movement to the document.
292        /// @param | dx | number | Horizontal wheel delta.
293        /// @param | dy | number | Vertical wheel delta.
294        /// @return | boolean | True when an element handled the wheel event.
295        methods.add_method("wheelmoved", |_, this, (dx, dy): (f32, f32)| {
296            Ok(this.inner.borrow().wheel_moved(dx, dy).is_some())
297        });
298        // -- keypressed --
299        /// Forwards a key press to the focused document element and dispatches `keydown`.
300        /// @param | key | string | Key name.
301        /// @return | boolean | True when the event was consumed or default was prevented.
302        methods.add_method("keypressed", |lua, this, key: String| {
303            let target = this.inner.borrow().key_pressed(&key);
304            let consumed = target.is_some();
305            let prevented = emit_html_event(
306                lua,
307                this,
308                "keydown",
309                target,
310                HtmlEventPayload {
311                    key: Some(key),
312                    ..HtmlEventPayload::default()
313                },
314            )?;
315            Ok(consumed || prevented)
316        });
317        // -- textinput --
318        /// Forwards text input to the focused document element and dispatches `input`.
319        /// @param | text | string | Input text.
320        /// @return | boolean | True when the event was consumed or default was prevented.
321        methods.add_method("textinput", |lua, this, text: String| {
322            let target = this.inner.borrow_mut().text_input(&text);
323            let value = target.and_then(|id| this.inner.borrow().text(id));
324            let consumed = target.is_some();
325            let prevented = emit_html_event(
326                lua,
327                this,
328                "input",
329                target,
330                HtmlEventPayload {
331                    text: Some(text),
332                    value,
333                    ..HtmlEventPayload::default()
334                },
335            )?;
336            Ok(consumed || prevented)
337        });
338        // -- type --
339        /// Returns the Lua-visible type name for this HTML document handle.
340        /// @return | string | The string `LHtmlDocument`.
341        methods.add_method("type", |_, _, ()| Ok("LHtmlDocument"));
342        // -- typeOf --
343        /// Returns whether this document handle matches a supported type name.
344        /// @param | name | string | Type name to compare against `LHtmlDocument` and `Object`.
345        /// @return | boolean | True when the supplied type name matches this handle.
346        methods.add_method("typeOf", |_, _, name: String| {
347            Ok(name == "LHtmlDocument" || name == "LObject")
348        });
349    }

609        // -- queryAll --
610        /// Returns all descendant elements matching a selector.
611        /// @param | selector | string | Selector supported by the HTML engine.
612        /// @return | LHtmlElement[] | `LHtmlElement` handles.
613        methods.add_method("queryAll", |lua, this, selector: String| {
614            check_html_element(this)?;
615            let document = LuaHtmlDocument {
616                inner: this.document.clone(),
617                callbacks: this.callbacks.clone(),
618                state: this.state.clone(),
619            };
620            html_element_table(
621                lua,
622                &document,
623                this.document
624                    .borrow()
625                    .query_all_from(this.element_id, &selector),
626            )
627        });
628        // -- on --
629        /// Registers an element-level event listener.
630        /// @param | event | string | Event name to listen for.
631        /// @param | func | function | Lua callback receiving an event table.
632        /// @return | integer | Listener handle used by `off`.
633        methods.add_method("on", |lua, this, (event, func): (String, LuaFunction)| {
634            check_html_element(this)?;
635            let key = lua.create_registry_value(func)?;
636            let mut callbacks = this.callbacks.borrow_mut();
637            let handle = callbacks.next_handle.saturating_add(1);
638            callbacks.next_handle = handle;
639            callbacks
640                .elements
641                .entry(this.element_id)
642                .or_default()
643                .insert(handle, HtmlListener { event, key });
644            Ok(handle)
645        });
646        // -- off --
647        /// Removes an element-level event listener by handle.
648        /// @param | handle | integer | Listener handle returned by `on`.
649        methods.add_method("off", |lua, this, handle: u64| {
650            if let Some(listeners) = this
651                .callbacks
652                .borrow_mut()
653                .elements
654                .get_mut(&this.element_id)
655            {
656                if let Some(listener) = listeners.remove(&handle) {
657                    lua.remove_registry_value(listener.key)?;
658                }
659            }
660            Ok(())
661        });
662        // -- type --
663        /// Returns the Lua-visible type name for this HTML element handle.
664        /// @return | string | The string `LHtmlElement`.
665        methods.add_method("type", |_, _, ()| Ok("LHtmlElement"));
666        // -- typeOf --
667        /// Returns whether this element handle matches a supported type name.
668        /// @param | name | string | Type name to compare against `LHtmlElement` and `Object`.
669        /// @return | boolean | True when the supplied type name matches this handle.
670        methods.add_method("typeOf", |_, _, name: String| {
671            Ok(name == "LHtmlElement" || name == "LObject")
672        });
673    }
674}
675/// Converts an optional element id into an element userdata value or nil.
676fn html_element_value<'lua>(
677    lua: &'lua Lua,
678    document: &LuaHtmlDocument,
679    element_id: Option<HtmlElementId>,
680) -> LuaResult<LuaValue<'lua>> {
681    match element_id {
682        Some(element_id) => Ok(LuaValue::UserData(
683            lua.create_userdata(document.element_handle(element_id))?,
684        )),
685        None => Ok(LuaValue::Nil),
686    }
687}
688/// Converts element ids into a Lua array table of element handles.
689fn html_element_table<'lua>(
690    lua: &'lua Lua,
691    document: &LuaHtmlDocument,
692    element_ids: Vec<HtmlElementId>,
693) -> LuaResult<LuaTable<'lua>> {
694    let table = lua.create_table()?;
695    for (index, element_id) in element_ids.into_iter().enumerate() {
696        table.set(index + 1, document.element_handle(element_id))?;
697    }
698    Ok(table)
699}
700/// Verifies an element handle still points to a live element in the same document generation.
701fn check_html_element(element: &LuaHtmlElement) -> LuaResult<()> {
702    let document = element.document.borrow();
703    if document.generation() != element.generation {
704        return Err(LuaError::RuntimeError(
705            "HtmlElement handle is stale after document markup replacement".to_string(),
706        ));
707    }
708    if document.element(element.element_id).is_none() {
709        return Err(LuaError::RuntimeError(
710            "HtmlElement handle no longer points to a live element".to_string(),
711        ));
712    }
713    Ok(())
714}
715/// Dispatches an HTML event through element bubbling and document listeners.
716fn emit_html_event(
717    lua: &Lua,
718    document: &LuaHtmlDocument,
719    event_name: &str,
720    target: Option<HtmlElementId>,
721    payload: HtmlEventPayload,
722) -> LuaResult<bool> {
723    let flags = Rc::new(RefCell::new(HtmlEventFlags::default()));
724    let mut called = false;
725    if let Some(target_id) = target {
726        let chain = document.inner.borrow().ancestors_inclusive(target_id);
727        for current_target in chain {
728            let listeners = collect_element_listeners(lua, document, current_target, event_name)?;
729            for listener in listeners {
730                called = true;
731                let event = create_html_event_table(
732                    lua,
733                    document,
734                    event_name,
735                    target,
736                    Some(current_target),
737                    &payload,
738                    flags.clone(),
739                )?;
740                listener.call::<_, ()>(event)?;
741                if flags.borrow().propagation_stopped {
742                    return Ok(true);
743                }
744            }
745        }
746    }
747    let listeners = collect_document_listeners(lua, document, event_name)?;
748    for listener in listeners {
749        called = true;
750        let event = create_html_event_table(
751            lua,
752            document,
753            event_name,
754            target,
755            None,
756            &payload,
757            flags.clone(),
758        )?;
759        listener.call::<_, ()>(event)?;
760        if flags.borrow().propagation_stopped {
761            break;
762        }
763    }
764    Ok(called || flags.borrow().default_prevented)
765}
766/// Collects Lua callbacks for one element and event name.
767fn collect_element_listeners<'lua>(
768    lua: &'lua Lua,
769    document: &LuaHtmlDocument,
770    element_id: HtmlElementId,
771    event_name: &str,
772) -> LuaResult<Vec<LuaFunction<'lua>>> {
773    let callbacks = document.callbacks.borrow();
774    let Some(listeners) = callbacks.elements.get(&element_id) else {
775        return Ok(Vec::new());
776    };
777    listeners
778        .values()
779        .filter(|listener| listener.event == event_name)
780        .map(|listener| lua.registry_value(&listener.key))
781        .collect()
782}
783/// Collects Lua document callbacks for one event name.
784fn collect_document_listeners<'lua>(
785    lua: &'lua Lua,
786    document: &LuaHtmlDocument,
787    event_name: &str,
788) -> LuaResult<Vec<LuaFunction<'lua>>> {
789    let callbacks = document.callbacks.borrow();
790    callbacks
791        .document
792        .values()
793        .filter(|listener| listener.event == event_name)
794        .map(|listener| lua.registry_value(&listener.key))
795        .collect()
796}
797/// Builds the Lua event table passed to HTML event listeners.
798fn create_html_event_table<'lua>(
799    lua: &'lua Lua,
800    document: &LuaHtmlDocument,
801    event_name: &str,
802    target: Option<HtmlElementId>,
803    current_target: Option<HtmlElementId>,
804    payload: &HtmlEventPayload,
805    flags: Rc<RefCell<HtmlEventFlags>>,
806) -> LuaResult<LuaTable<'lua>> {
807    let table = lua.create_table()?;
808    /// Performs the 'type' operation.
809    table.set("type", event_name)?;
810    /// Performs the 'document' operation.
811    table.set("document", document.clone())?;
812    if let Some(target_id) = target {
813        /// Performs the 'target' operation.
814        table.set("target", document.element_handle(target_id))?;
815    }
816    match current_target {
817        Some(current_target) => {
818            /// Performs the 'currentTarget' operation.
819            table.set("currentTarget", document.element_handle(current_target))?
820        }
821        /// Performs the 'currentTarget' operation.
822        None => table.set("currentTarget", document.clone())?,
823    }
824    if let Some(x) = payload.x {
825        /// The 'x' field value exposed to Lua scripts.
826        table.set("x", x)?;
827    }
828    if let Some(y) = payload.y {
829        /// The 'y' field value exposed to Lua scripts.
830        table.set("y", y)?;
831    }
832    if let Some(button) = payload.button {
833        /// Performs the 'button' operation.
834        table.set("button", button)?;
835    }
836    if let Some(key) = &payload.key {
837        /// The 'key' field value exposed to Lua scripts.
838        table.set("key", key.clone())?;
839    }
840    if let Some(text) = &payload.text {
841        /// Performs the 'text' operation.
842        table.set("text", text.clone())?;
843    }
844    if let Some(value) = &payload.value {
845        /// Performs the 'value' operation.
846        table.set("value", value.clone())?;
847    }
848    let prevent_flags = flags.clone();
849    /// Marks the event as having its default action prevented.
850    table.set(
851        "preventDefault",
852        lua.create_function(move |_, ()| {
853            prevent_flags.borrow_mut().default_prevented = true;
854            Ok(())
855        })?,
856    )?;
857    let stop_flags = flags.clone();
858    /// Stops event propagation to remaining listeners.
859    table.set(
860        "stopPropagation",
861        lua.create_function(move |_, ()| {
862            stop_flags.borrow_mut().propagation_stopped = true;
863            Ok(())
864        })?,
865    )?;
866    /// Returns whether the default action was prevented.
867    /// @return | boolean | True when the default was prevented.
868    table.set(
869        "isDefaultPrevented",
870        lua.create_function(move |_, ()| Ok(flags.borrow().default_prevented))?,
871    )?;
872    Ok(table)
873}
874/// Converts document draw commands into engine render commands.
875fn enqueue_html_draw_commands(
876    document: &mut HtmlDocument,
877    state: &mut SharedState,
878    x: f32,
879    y: f32,
880) {
881    let commands = document.draw_commands(x, y);
882    for command in commands {
883        if command.rect.w <= 0.0 || command.rect.h <= 0.0 {
884            continue;
885        }
886        match command.kind.as_str() {
887            "box" => {
888                if let Some(bg) = command
889                    .background_color
890                    .as_deref()
891                    .and_then(parse_css_color)
892                {
893                    state
894                        .render_commands
895                        .push(RenderCommand::SetColor(bg[0], bg[1], bg[2], bg[3]));
896                    state.render_commands.push(RenderCommand::Rectangle {
897                        mode: DrawMode::Fill,
898                        x: command.rect.x,
899                        y: command.rect.y,
900                        w: command.rect.w,
901                        h: command.rect.h,
902                    });
903                }
904                if let Some(border) = command.color.as_deref().and_then(parse_css_color) {
905                    state.render_commands.push(RenderCommand::SetColor(
906                        border[0], border[1], border[2], border[3],
907                    ));
908                    state.render_commands.push(RenderCommand::Rectangle {
909                        mode: DrawMode::Line,
910                        x: command.rect.x,
911                        y: command.rect.y,
912                        w: command.rect.w,
913                        h: command.rect.h,
914                    });
915                }
916            }
917            "text" => {
918                let text = command.text.trim();
919                if text.is_empty() {
920                    continue;
921                }
922                let Some(font_key) = state.active_font.or(state.default_font) else {
923                    continue;
924                };
925                let fg = command
926                    .color
927                    .as_deref()
928                    .and_then(parse_css_color)
929                    .unwrap_or([1.0, 1.0, 1.0, 1.0]);
930                state
931                    .render_commands
932                    .push(RenderCommand::SetColor(fg[0], fg[1], fg[2], fg[3]));
933                state.render_commands.push(RenderCommand::Print {
934                    font_key,
935                    text: text.to_string(),
936                    x: command.rect.x + 4.0,
937                    y: command.rect.y + 4.0,
938                    scale: 1.0,
939                });
940            }
941            _ => {}
942        }
943    }
944}
945/// Parses a CSS color string into normalized RGBA channels.
946fn parse_css_color(raw: &str) -> Option<[f32; 4]> {
947    parse_css_color_rgba(raw)
948}
949/// Parses optional Lua document options and separate CSS path override.
950fn parse_document_options(
951    opts: Option<LuaTable>,
952) -> LuaResult<(HtmlDocumentOptions, Option<String>)> {
953    let mut options = HtmlDocumentOptions::default();
954    let mut css_path = None;
955    if let Some(opts) = opts {
956        options.css = opts.get::<_, Option<String>>("css")?;
957        css_path = opts.get::<_, Option<String>>("cssPath")?;
958        if let Some(width) = opts.get::<_, Option<f32>>("width")? {
959            options.width = width;
960        }
961        if let Some(height) = opts.get::<_, Option<f32>>("height")? {
962            options.height = height;
963        }
964    }
965    Ok((options, css_path))
966}
967/// Returns the CSS path beside an HTML path by replacing the extension with `.css`.
968fn companion_css_path(path: &str) -> Option<String> {
969    let trimmed = path.trim();
970    if trimmed.is_empty() {
971        return None;
972    }
973    Some(
974        Path::new(trimmed)
975            .with_extension("css")
976            .to_string_lossy()
977            .into_owned(),
978    )
979}
980/// Registers `lurek.html` document constructors and feature support checks.
981pub fn register(lua: &Lua, luna: &LuaTable, state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
982    let html = lua.create_table()?;
983    let state_for_new = state.clone();
984    // -- newDocument --
985    /// Creates an HTML document from optional source and layout/style options.
986    /// @param | source | string? | HTML source, defaulting to an empty document.
987    /// @param | opts | table? | Table with `css`, `cssPath`, `width`, and `height` fields.
988    /// @return | LHtmlDocument | New HTML document handle.
989    html.set(
990        "newDocument",
991        lua.create_function(
992            move |lua, (source, opts): (Option<String>, Option<LuaTable>)| {
993                let (options, _) = parse_document_options(opts)?;
994                lua.create_userdata(LuaHtmlDocument::new(
995                    HtmlDocument::with_options(source.unwrap_or_default(), options),
996                    state_for_new.clone(),
997                ))
998            },
999        )?,
1000    )?;
1001    let state_for_load = state.clone();
1002    // -- loadDocument --
1003    /// Loads an HTML document from GameFS and optionally loads CSS from options or companion file.
1004    /// @param | path | string | GameFS path to the HTML file.
1005    /// @param | opts | table? | Table with `css`, `cssPath`, `width`, and `height` fields.
1006    /// @return | LHtmlDocument | Loaded HTML document handle.
1007    html.set(
1008        "loadDocument",
1009        lua.create_function(move |lua, (path, opts): (String, Option<LuaTable>)| {
1010            let (mut options, css_path) = parse_document_options(opts)?;
1011            let source = {
1012                let state = state_for_load.borrow();
1013                state.fs.read_string(&path).map_err(|e| {
1014                    LuaError::RuntimeError(format!(
1015                        "lurek.html.loadDocument: cannot read '{path}': {e}"
1016                    ))
1017                })?
1018            };
1019            if options.css.is_none() {
1020                if let Some(css_path) = css_path {
1021                    let css = {
1022                        let state = state_for_load.borrow();
1023                        state.fs.read_string(&css_path).map_err(|e| {
1024                            LuaError::RuntimeError(format!(
1025                                "lurek.html.loadDocument: cannot read cssPath '{css_path}': {e}"
1026                            ))
1027                        })?
1028                    };
1029                    options.css = Some(css);
1030                } else if let Some(css_companion) = companion_css_path(&path) {
1031                    let css = {
1032                        let state = state_for_load.borrow();
1033                        state.fs.read_string(&css_companion)
1034                    };
1035                    if let Ok(css) = css {
1036                        options.css = Some(css);
1037                    }
1038                }
1039            }
1040            lua.create_userdata(LuaHtmlDocument::new(
1041                HtmlDocument::with_options(source, options),
1042                state_for_load.clone(),
1043            ))
1044        })?,
1045    )?;
1046    // -- supports --
1047    /// Returns whether the HTML engine supports a named feature.
1048    /// @param | feature | string | Feature name to query.
1049    /// @return | boolean | True when the feature is supported.
1050    html.set(
1051        "supports",
1052        lua.create_function(|_, feature: String| Ok(HtmlDocument::supports(&feature)))?,
1053    )?;
1054    /// Performs the 'html' operation.
1055    luna.set("html", html)?;
1056    Ok(())
1057}
