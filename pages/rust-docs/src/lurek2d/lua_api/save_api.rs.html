11/// Extracts a logical slot name from a `slot_<name>.sav` filename.
12fn slot_name_from_filename(filename: &str) -> Option<&str> {
13    filename
14        .strip_prefix("slot_")
15        .and_then(|s| s.strip_suffix(".sav"))
16}
17/// Parses validated save content and evaluates it back into a Lua table.
18fn eval_save_content<'a>(vm: &'a Lua, content: &str) -> LuaResult<LuaTable<'a>> {
19    let validated = SaveManager::parse_save_string(content).map_err(LuaError::RuntimeError)?;
20    vm.load(validated.as_str()).eval()
21}
22/// Manages persistent game state: registering data collectors/restorers, serializing to named.
23/// slots, handling schema migrations, auto-save timers, compression, and lifecycle hooks.
24pub struct LuaSaveManager {
25    manager: SaveManager,
26    state: Rc<RefCell<SharedState>>,
27    collectors: HashMap<String, LuaRegistryKey>,
28    restorers: HashMap<String, LuaRegistryKey>,
29    migrations: HashMap<i32, LuaRegistryKey>,
30    compress: bool,
31    before_save: Option<LuaRegistryKey>,
32    after_load: Option<LuaRegistryKey>,
33}
34impl LuaSaveManager {
35    /// Creates a new Lua-visible save manager bound to shared engine state.
36    pub fn new(state: Rc<RefCell<SharedState>>) -> Self {
37        Self {
38            manager: SaveManager::new(),
39            state,
40            collectors: HashMap::new(),
41            restorers: HashMap::new(),
42            migrations: HashMap::new(),
43            compress: false,
44            before_save: None,
45            after_load: None,
46        }
47    }
48    fn remove_key<K: std::hash::Hash + Eq>(
49        lua: &Lua,
50        map: &mut HashMap<K, LuaRegistryKey>,
51        key: &K,
52    ) -> LuaResult<()> {
53        if let Some(rk) = map.remove(key) {
54            lua.remove_registry_value(rk)?;
55        }
56        Ok(())
57    }
58    fn collect_data<'a>(&self, lua: &'a Lua) -> LuaResult<LuaTable<'a>> {
59        let result = lua.create_table()?;
60        for name in self.manager.registered_names() {
61            if let Some(key) = self.collectors.get(name) {
62                let func = lua.registry_value::<LuaFunction>(key)?;
63                let val: LuaValue = func.call(())?;
64                result.set(name.as_str(), val)?;
65            }
66        }
67        /// Performs the '__schema_version' operation.
68        result.set("__schema_version", self.manager.schema_version())?;
69        let timestamp = std::time::SystemTime::now()
70            .duration_since(std::time::UNIX_EPOCH)
71            .map(|d| d.as_secs_f64())
72            .unwrap_or(0.0);
73        /// Performs the '__timestamp' operation.
74        result.set("__timestamp", timestamp)?;
75        /// Performs the '__summary' operation.
76        result.set("__summary", self.manager.summary())?;
77        Ok(result)
78    }
79    fn apply_migrations<'a>(
80        &self,
81        lua: &'a Lua,
82        mut data: LuaTable<'a>,
83        saved_ver: i32,
84    ) -> LuaResult<LuaTable<'a>> {
85        for ver in self.manager.applicable_migrations(saved_ver) {
86            if let Some(key) = self.migrations.get(&ver) {
87                let func = lua.registry_value::<LuaFunction>(key)?;
88                let result: LuaValue = func.call(data.clone())?;
89                if let LuaValue::Table(t) = result {
90                    data = t;
91                }
92            }
93        }
94        Ok(data)
95    }
96    fn call_restorers(&self, lua: &Lua, data: &LuaTable) -> LuaResult<()> {
97        for name in self.manager.registered_names() {
98            if let Some(key) = self.restorers.get(name) {
99                let func = lua.registry_value::<LuaFunction>(key)?;
100                let val: LuaValue = data.get(name.as_str())?;
101                func.call::<_, ()>(val)?;
102            }
103        }
104        Ok(())
105    }
106    fn restore_from_table<'a>(&mut self, lua: &'a Lua, data: LuaTable<'a>) -> LuaResult<()> {
107        let saved_ver: i32 = data.get("__schema_version").unwrap_or(0);
108        let data = self.apply_migrations(lua, data, saved_ver)?;
109        self.call_restorers(lua, &data)?;
110        drop(data);
111        self.manager.clear_dirty();
112        Ok(())
113    }
114    fn serialize_collected(&self, lua: &Lua) -> LuaResult<String> {
115        let data_table = self.collect_data(lua)?;
116        let mut data_map = HashMap::new();
117        for pair in data_table.pairs::<String, LuaValue>() {
118            let (k, v) = pair?;
119            data_map.insert(k, SaveValue::from_lua(&v)?);
120        }
121        let body = serialize_table(&data_map, 0).map_err(LuaError::RuntimeError)?;
122        Ok(format!("return {}\n", body))
123    }
124    fn save_to_slot(&mut self, lua: &Lua, slot: &str) -> LuaResult<()> {
125        if let Some(ref key) = self.before_save {
126            let func = lua.registry_value::<LuaFunction>(key)?;
127            func.call::<_, ()>(slot)?;
128        }
129        let plain = self.serialize_collected(lua)?;
130        let content = if self.compress {
131            compress_save_content(&plain).map_err(LuaError::RuntimeError)?
132        } else {
133            plain
134        };
135        let path = SaveManager::slot_path(slot);
136        self.state
137            .borrow()
138            .fs
139            .write_string(&path, &content)
140            .map_err(|e| LuaError::RuntimeError(format!("lurek.save:save: {}", e)))?;
141        self.manager.clear_dirty();
142        Ok(())
143    }
144    fn load_from_slot(&mut self, lua: &Lua, slot: &str) -> LuaResult<(bool, Option<String>)> {
145        let path = SaveManager::slot_path(slot);
146        let raw = match self.state.borrow().fs.read_string(&path) {
147            Ok(c) => c,
148            Err(e) => return Ok((false, Some(format!("lurek.save:load: {}", e)))),
149        };
150        let content: String = match decompress_save_content(&raw) {
151            Ok(s) => s,
152            Err(e) => return Ok((false, Some(format!("lurek.save:load: {}", e)))),
153        };
154        let data: LuaTable = match eval_save_content(lua, &content) {
155            Ok(t) => t,
156            Err(e) => return Ok((false, Some(format!("lurek.save:load: corrupt save: {}", e)))),
157        };
158        self.restore_from_table(lua, data)?;
159        if let Some(ref key) = self.after_load {
160            let func = lua.registry_value::<LuaFunction>(key)?;
161            func.call::<_, ()>(slot)?;
162        }
163        Ok((true, None))
164    }
165    fn delete_slot(&self, slot: &str) -> LuaResult<()> {
166        let path = SaveManager::slot_path(slot);
167        self.state
168            .borrow()
169            .fs
170            .remove(&path)
171            .map_err(|e| LuaError::RuntimeError(format!("lurek.save:delete: {}", e)))?;
172        Ok(())
173    }
174    fn slot_exists(&self, slot: &str) -> bool {
175        let path = SaveManager::slot_path(slot);
176        self.state.borrow().fs.exists(&path)
177    }
178    fn read_slot_meta<'a>(&self, lua: &'a Lua, slot: &str) -> LuaResult<Option<LuaTable<'a>>> {
179        let path = SaveManager::slot_path(slot);
180        let content = {
181            let state_ref = self.state.borrow();
182            let game_fs = &state_ref.fs;
183            if !game_fs.exists(&path) {
184                return Ok(None);
185            }
186            match game_fs.read_string(&path) {
187                Ok(c) => c,
188                Err(_) => return Ok(None),
189            }
190        };
191        match eval_save_content(lua, &content) {
192            Ok(data) => {
193                let info = lua.create_table()?;
194                /// Performs the 'slot' operation.
195                info.set("slot", slot)?;
196                let ver: i32 = data.get("__schema_version").unwrap_or(0);
197                let ts: f64 = data.get("__timestamp").unwrap_or(0.0);
198                let summary: String = data.get("__summary").unwrap_or_default();
199                /// Performs the 'version' operation.
200                info.set("version", ver)?;
201                /// Performs the 'timestamp' operation.
202                info.set("timestamp", ts)?;
203                /// Performs the 'summary' operation.
204                info.set("summary", summary)?;
205                Ok(Some(info))
206            }
207            Err(_) => Ok(None),
208        }
209    }
210    fn list_slots<'a>(&self, lua: &'a Lua) -> LuaResult<LuaTable<'a>> {
211        let result = lua.create_table()?;
212        let entries = match self.state.borrow().fs.list("save") {
213            Ok(e) => e,
214            Err(_) => return Ok(result),
215        };
216        let mut idx = 1;
217        for entry in &entries {
218            if let Some(slot_name) = slot_name_from_filename(entry) {
219                if let Some(info) = self.read_slot_meta(lua, slot_name)? {
220                    result.set(idx, info)?;
221                    idx += 1;
222                }
223            }
224        }
225        Ok(result)
226    }
227}
228impl LuaUserData for LuaSaveManager {
229    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
230        // -- register --
231        /// Register a named data section with a collector and restorer function pair.
232        /// The collector is called during save to gather the current state; the restorer is called during load to apply saved state.
233        /// @param | name | string | Unique section name identifying this chunk of save data (e.g. "player", "inventory").
234        /// @param | collectFn | function | Called with no arguments during save; must return the data to persist for this section.
235        /// @param | restoreFn | function | Called with the saved value during load; responsible for applying it back to game state.
236        methods.add_method_mut(
237            "register",
238            |lua, this, (name, collect_fn, restore_fn): (String, LuaFunction, LuaFunction)| {
239                Self::remove_key(lua, &mut this.collectors, &name)?;
240                Self::remove_key(lua, &mut this.restorers, &name)?;
241                this.collectors
242                    .insert(name.clone(), lua.create_registry_value(collect_fn)?);
243                this.restorers
244                    .insert(name.clone(), lua.create_registry_value(restore_fn)?);
245                this.manager.register(&name);
246                Ok(())
247            },
248        );
249        // -- unregister --
250        /// Remove a previously registered data section by name, cleaning up its collector and restorer callbacks.
251        /// @param | name | string | The section name to unregister.
252        methods.add_method_mut("unregister", |lua, this, name: String| {
253            this.manager.unregister(&name);
254            Self::remove_key(lua, &mut this.collectors, &name)?;
255            Self::remove_key(lua, &mut this.restorers, &name)?;
256            Ok(())
257        });
258        // -- setSchemaVersion --
259        /// Set the current schema version number for saves produced by this game build.
260        /// When loading an older save, migrations registered via addMigration will run in order.
261        /// @param | version | integer | Integer schema version (must increase with each breaking data format change).
262        methods.add_method_mut("setSchemaVersion", |_, this, version: i32| {
263            this.manager.set_schema_version(version);
264            Ok(())
265        });
266        // -- getSchemaVersion --
267        /// Return the current schema version number set for this save manager.
268        /// @return | integer | The active schema version.
269        methods.add_method("getSchemaVersion", |_, this, ()| {
270            Ok(this.manager.schema_version())
271        });
272        // -- addMigration --
273        /// Register a migration function that transforms save data from one schema version to the next.
274        /// Migrations run in version order when loading saves older than the current schema version.
275        /// @param | fromVersion | integer | The schema version this migration upgrades FROM (it produces fromVersion+1).
276        /// @param | func | function | Receives the full save data table and must return the transformed table.
277        methods.add_method_mut(
278            "addMigration",
279            |lua, this, (from_ver, func): (i32, LuaFunction)| {
280                Self::remove_key(lua, &mut this.migrations, &from_ver)?;
281                this.migrations
282                    .insert(from_ver, lua.create_registry_value(func)?);
283                this.manager.add_migration(from_ver);
284                Ok(())
285            },
286        );
287        // -- collect --
288        /// Invoke all registered collectors and return the assembled save-data table without writing to disk.
289        /// Useful for preview, debug display, or manual serialization.
290        /// @return | table | The full save-data table including __schema_version, __timestamp, and __summary metadata.
291        methods.add_method("collect", |lua, this, ()| this.collect_data(lua));
292        // -- restore --
293        /// Apply a previously collected save-data table back into game state by invoking all registered restorers.
294        /// Migrations are applied if the table's __schema_version is older than the current version.
295        /// @param | data | table | A save-data table (as produced by collect or loaded from disk).
296        methods.add_method_mut("restore", |lua, this, data: LuaTable| {
297            this.restore_from_table(lua, data)
298        });
299        // -- markDirty --
300        /// Mark the save state as dirty, indicating unsaved changes exist.
301        /// When auto-save is enabled, this flag triggers a write on the next auto-save interval.
302        methods.add_method_mut("markDirty", |_, this, ()| {
303            this.manager.mark_dirty();
304            Ok(())
305        });
306        // -- isDirty --
307        /// Check whether unsaved changes exist since the last save or load.
308        /// @return | boolean | True if game state has been modified and not yet persisted.
309        methods.add_method("isDirty", |_, this, ()| Ok(this.manager.is_dirty()));
310        // -- enableAutoSave --
311        /// Enable periodic auto-saving: when the dirty flag is set, the system writes to the target slot every interval seconds.
312        /// @param | interval | number | Time in seconds between auto-save checks (e.g. 30.0 for every 30 seconds).
313        /// @param | slot | string | The slot name to auto-save into (e.g. "autosave").
314        methods.add_method_mut(
315            "enableAutoSave",
316            |_, this, (interval, slot): (f64, String)| {
317                this.manager.enable_auto_save(interval, slot);
318                Ok(())
319            },
320        );
321        // -- disableAutoSave --
322        /// Disable the periodic auto-save timer. Manual saves via save() still work.
323        methods.add_method_mut("disableAutoSave", |_, this, ()| {
324            this.manager.disable_auto_save();
325            Ok(())
326        });
327        // -- update --
328        /// Advance the auto-save timer by dt seconds. Call this once per frame from your game loop.
329        /// Returns true if an auto-save was triggered this tick (dirty flag was set and interval elapsed).
330        /// @param | dt | number | Delta time in seconds since the last frame.
331        /// @return | boolean | True if an auto-save was triggered during this update.
332        methods.add_method_mut("update", |_, this, dt: f64| Ok(this.manager.update(dt)));
333        // -- setSummary --
334        /// Set a human-readable summary string stored alongside save metadata (e.g. "Level 5 – Forest").
335        /// This appears in slot listings so players can identify saves without loading them.
336        /// @param | summary | string | Short description of the current game progress.
337        methods.add_method_mut("setSummary", |_, this, summary: String| {
338            this.manager.set_summary(summary);
339            Ok(())
340        });
341        // -- getSummary --
342        /// Get the current summary string that will be embedded in the next save.
343        /// @return | string | The summary text, or an empty string if none was set.
344        methods.add_method("getSummary", |_, this, ()| {
345            Ok(this.manager.summary().to_string())
346        });
347        // -- reset --
348        /// Completely reset the save manager: unregister all sections, clear migrations, hooks, compression, and dirty state.
349        /// Use this when returning to a main menu or starting a new game session.
350        methods.add_method_mut("reset", |lua, this, ()| {
351            for (_, key) in this.collectors.drain() {
352                lua.remove_registry_value(key)?;
353            }
354            for (_, key) in this.restorers.drain() {
355                lua.remove_registry_value(key)?;
356            }
357            for (_, key) in this.migrations.drain() {
358                lua.remove_registry_value(key)?;
359            }
360            if let Some(key) = this.before_save.take() {
361                lua.remove_registry_value(key)?;
362            }
363            if let Some(key) = this.after_load.take() {
364                lua.remove_registry_value(key)?;
365            }
366            this.compress = false;
367            this.manager.reset();
368            Ok(())
369        });
370        // -- setCompress --
371        /// Enable or disable LZ4 compression for save files. Compressed saves are smaller on disk.
372        /// but slightly slower to write and read. Decompression is handled transparently on load.
373        /// @param | enabled | boolean | True to compress future saves, false to write plain text.
374        methods.add_method_mut("setCompress", |_, this, enabled: bool| {
375            this.compress = enabled;
376            Ok(())
377        });
378        // -- isCompressed --
379        /// Check whether save compression is currently enabled.
380        /// @return | boolean | True if future saves will be LZ4-compressed.
381        methods.add_method("isCompressed", |_, this, ()| Ok(this.compress));
382        // -- onBeforeSave --
383        /// Set a hook function called immediately before each save operation begins.
384        /// Useful for last-moment state snapshots or UI feedback ("Saving...").
385        /// Pass nil to clear a previously registered hook.
386        /// @param | func | function? | Callback receiving the slot name as its argument, or nil to clear.
387        methods.add_method_mut("onBeforeSave", |lua, this, func: LuaValue| {
388            if let Some(key) = this.before_save.take() {
389                lua.remove_registry_value(key)?;
390            }
391            if let LuaValue::Function(f) = func {
392                this.before_save = Some(lua.create_registry_value(f)?);
393            }
394            Ok(())
395        });
396        // -- onAfterLoad --
397        /// Set a hook function called immediately after a save file is successfully loaded and all restorers have run.
398        /// Useful for refreshing derived state, triggering UI updates, or logging.
399        /// Pass nil to clear a previously registered hook.
400        /// @param | func | function? | Callback receiving the slot name as its argument, or nil to clear.
401        methods.add_method_mut("onAfterLoad", |lua, this, func: LuaValue| {
402            if let Some(key) = this.after_load.take() {
403                lua.remove_registry_value(key)?;
404            }
405            if let LuaValue::Function(f) = func {
406                this.after_load = Some(lua.create_registry_value(f)?);
407            }
408            Ok(())
409        });
410        // -- save --
411        /// Persist all registered data sections to the named slot file on disk.
412        /// Calls the onBeforeSave hook, collects data, optionally compresses, and writes to save/<slot>.sav.
413        /// @param | slot | string | Slot name (e.g. "slot1", "quicksave"). The file is stored as save/slot_<name>.sav.
414        methods.add_method_mut("save", |lua, this, slot: String| {
415            this.save_to_slot(lua, &slot)
416        });
417        // -- load --
418        /// Load game state from a named slot file. Decompresses if needed, applies migrations, calls restorers, then fires onAfterLoad.
419        /// @param | slot | string | Slot name to load (e.g. "slot1").
420        /// @return | boolean | True if the load succeeded, false on error.
421        /// @return | string | Error message if the load failed, nil on success.
422        methods.add_method_mut("load", |lua, this, slot: String| {
423            this.load_from_slot(lua, &slot)
424        });
425        // -- delete --
426        /// Permanently delete a save slot file from disk. This action cannot be undone.
427        /// @param | slot | string | Slot name to delete (e.g. "slot1").
428        /// @return | nil | No return value.
429        methods.add_method("delete", |_, this, slot: String| this.delete_slot(&slot));
430        // -- exists --
431        /// Check whether a save slot file exists on disk without reading its contents.
432        /// @param | slot | string | Slot name to check.
433        /// @return | boolean | True if the slot file is present.
434        methods.add_method(
435            "exists",
436            |_, this, slot: String| Ok(this.slot_exists(&slot)),
437        );
438        // -- getSlots --
439        /// List all save slots found on disk with their metadata (version, timestamp, summary).
440        /// @return | table | Array of info tables, each with fields: slot, version, timestamp, summary.
441        /// @field | slot | string | Slot name.
442        /// @field | version | integer | Schema version.
443        /// @field | timestamp | integer | Save timestamp.
444        /// @field | summary | string | Save summary.
445        methods.add_method("getSlots", |lua, this, ()| this.list_slots(lua));
446        // -- getSlotInfo --
447        /// Read metadata for a single save slot without loading its full game state.
448        /// Returns nil if the slot does not exist or is corrupted.
449        /// @param | slot | string | Slot name to inspect.
450        /// @return | table | Info table with fields: slot, version, timestamp, summary, or nil if not found.
451        /// @field | slot | string | Slot name.
452        /// @field | version | integer | Schema version.
453        /// @field | timestamp | integer | Save timestamp.
454        /// @field | summary | string | Save summary.
455        methods.add_method("getSlotInfo", |lua, this, slot: String| {
456            match this.read_slot_meta(lua, &slot)? {
457                Some(info) => Ok(LuaValue::Table(info)),
458                None => Ok(LuaValue::Nil),
459            }
460        });
461        // -- type --
462        /// Return the type name string for this userdata object.
463        /// @return | string | Always "LSaveManager".
464        methods.add_method("type", |_, _, ()| Ok("LSaveManager"));
465        // -- typeOf --
466        /// Check whether this object matches a given type name. Supports "LSaveManager" and "Object".
467        /// @param | name | string | Type name to test against.
468        /// @return | boolean | True if the object matches the given type name.
469        methods.add_method("typeOf", |_, _, name: String| {
470            Ok(name == "LSaveManager" || name == "LObject")
471        });
472    }
473}
474/// Register the `lurek.save` module into the Lua runtime.
475pub fn register(lua: &Lua, lurek: &LuaTable, state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
476    let tbl = lua.create_table()?;
477    let s = state.clone();
478    // -- newSaveManager --
479    /// Create a new SaveManager instance for managing persistent game saves.
480    /// @return | LSaveManager | A fresh save manager with no registered sections.
481    tbl.set(
482        "newSaveManager",
483        lua.create_function(move |lua, ()| lua.create_userdata(LuaSaveManager::new(s.clone())))?,
484    )?;
485    /// Performs the 'save' operation.
486    lurek.set("save", tbl)?;
487    Ok(())
488}
