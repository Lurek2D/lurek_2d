12/// Parses a Lua shape table into positive array dimensions.
13fn parse_shape(value: LuaValue) -> LuaResult<Vec<usize>> {
14    let table = value
15        .as_table()
16        .ok_or_else(|| LuaError::RuntimeError("shape must be a table".into()))?;
17    let mut shape = Vec::new();
18    for i in 1..=table.len()? {
19        let dim: i64 = table.get(i)?;
20        if dim <= 0 {
21            return Err(LuaError::RuntimeError(
22                "shape dimensions must be positive".into(),
23            ));
24        }
25        shape.push(dim as usize);
26    }
27    if shape.is_empty() {
28        return Err(LuaError::RuntimeError("shape must not be empty".into()));
29    }
30    Ok(shape)
31}
32/// Parses an optional Lua dtype name into an array data type.
33fn parse_dtype(s: Option<String>) -> LuaResult<DataType> {
34    let name = s.as_deref().unwrap_or("float32");
35    DataType::parse(name).map_err(LuaError::RuntimeError)
36}
37/// Converts one-based Lua indices into zero-based array indices.
38fn parse_lua_indices(args: &[LuaValue]) -> LuaResult<Vec<usize>> {
39    args.iter()
40        .map(|v| match v {
41            LuaValue::Integer(n) => {
42                if *n < 1 {
43                    Err(LuaError::RuntimeError("index must be >= 1".into()))
44                } else {
45                    Ok((*n - 1) as usize)
46                }
47            }
48            LuaValue::Number(n) => {
49                let i = *n as i64;
50                if i < 1 {
51                    Err(LuaError::RuntimeError("index must be >= 1".into()))
52                } else {
53                    Ok((i - 1) as usize)
54                }
55            }
56            _ => Err(LuaError::RuntimeError("indices must be integers".into())),
57        })
58        .collect()
59}
60macro_rules! dispatch_arith {
61    ($methods:ident, $name:expr, $doc:expr, $arr_fn:path, $scalar_fn:path) => {
62        $methods.add_method($name, |lua, this, value: LuaValue| {
63            let result = match value {
64                LuaValue::Number(n) => {
65                    $scalar_fn(&this.inner, n).map_err(LuaError::RuntimeError)?
66                }
67                LuaValue::Integer(n) => {
68                    $scalar_fn(&this.inner, n as f64).map_err(LuaError::RuntimeError)?
69                }
70                LuaValue::UserData(ud) => {
71                    let other = ud.borrow::<LuaArray>()?;
72                    $arr_fn(&this.inner, &other.inner).map_err(LuaError::RuntimeError)?
73                }
74                _ => return Err(LuaError::RuntimeError("expected Array or number".into())),
75            };
76            lua.create_userdata(LuaArray { inner: result })
77        });
78    };
79}

141                    ))
142                }
143            };
144            let indices = parse_lua_indices(&args_vec[..args_vec.len() - 1])?;

578                    .map_err(LuaError::RuntimeError)?;
579                lua.create_userdata(LuaArray { inner: result })
580            },

582        // -- setRegion --
583        /// Writes a source array into this array at a one-based row and column.
584        /// @param | row | integer | One-based destination row.

600        /// Returns cumulative sum over the flattened array.
601        /// @return | LArray | New array containing cumulative sums.
602        methods.add_method("cumsum", |lua, this, ()| {
603            let r = analytics::cumsum(&this.inner).map_err(LuaError::RuntimeError)?;
604            lua.create_userdata(LuaArray { inner: r })
605        });
606        // -- diff --
607        /// Returns finite differences over the flattened array.
608        /// @param | order | integer? | Difference order; defaults to 1.
609        /// @return | LArray | New array containing differences.
610        methods.add_method("diff", |lua, this, order: Option<usize>| {
611            let r =
612                analytics::diff(&this.inner, order.unwrap_or(1)).map_err(LuaError::RuntimeError)?;
613            lua.create_userdata(LuaArray { inner: r })
614        });
615        // -- histogram --
616        /// Returns histogram bins for the array values.
617        /// @param | bins | integer | Number of histogram bins.
618        /// @param | lo | number? | Optional lower bound.
619        /// @param | hi | number? | Optional upper bound.
620        /// @return | table | Array of bin tables with `lo`, `hi`, and `count` fields.
621        /// @field | lo | number | Bin lower bound.
622        /// @field | hi | number | Bin upper bound.
623        /// @field | count | integer | Number of values in bin.
624        methods.add_method(
625            "histogram",
626            |lua, this, (bins, lo, hi): (usize, Option<f64>, Option<f64>)| {
627                let bins_data = analytics::histogram(&this.inner, bins, lo, hi)
628                    .map_err(LuaError::RuntimeError)?;
629                let out = lua.create_table()?;
630                for (i, (bin_lo, bin_hi, count)) in bins_data.iter().enumerate() {
631                    let entry = lua.create_table()?;
632                    /// The 'lo' field value exposed to Lua scripts.
633                    entry.set("lo", *bin_lo)?;
634                    /// The 'hi' field value exposed to Lua scripts.
635                    entry.set("hi", *bin_hi)?;
636                    /// Performs the 'count' operation.
637                    entry.set("count", *count)?;
638                    out.set(i + 1, entry)?;
639                }
640                Ok(out)
641            },
642        );
643        // -- percentile --
644        /// Returns a percentile value from the array.
645        /// @param | p | number | Percentile between 0 and 100.
646        /// @return | number | Percentile result.
647        methods.add_method("percentile", |_, this, p: f64| {
648            analytics::percentile(&this.inner, p).map_err(LuaError::RuntimeError)
649        });

651        /// Returns covariance with another array.

653        /// @return | number | Covariance value.

655            let other = other.borrow::<LuaArray>()?;
656            analytics::covariance(&this.inner, &other.inner).map_err(LuaError::RuntimeError)
657        });
658        // -- pearsonCorr --
659        /// Returns Pearson correlation with another array.
660        /// @param | other | LArray | Array used as the second variable.
661        /// @return | number | Pearson correlation coefficient.
662        methods.add_method("pearsonCorr", |_, this, other: LuaAnyUserData| {
663            let other = other.borrow::<LuaArray>()?;
664            analytics::pearson_corr(&this.inner, &other.inner).map_err(LuaError::RuntimeError)
665        });

667        /// Returns array values normalized into a target range.
668        /// @param | lo | number | Target lower bound.
669        /// @param | hi | number | Target upper bound.
670        /// @return | LArray | New normalized array.
671        methods.add_method("normalizeRange", |lua, this, (lo, hi): (f64, f64)| {
672            let r =
673                analytics::normalize_range(&this.inner, lo, hi).map_err(LuaError::RuntimeError)?;
674            lua.create_userdata(LuaArray { inner: r })
675        });
676        // -- zscore --
677        /// Returns z-score normalized array values.
678        /// @return | LArray | New z-score normalized array.
679        methods.add_method("zscore", |lua, this, ()| {
680            let r = analytics::zscore(&this.inner).map_err(LuaError::RuntimeError)?;
681            lua.create_userdata(LuaArray { inner: r })
682        });
683        // -- convolve1d --
684        /// Returns one-dimensional convolution with a kernel array.
685        /// @param | kernel | LArray | Kernel array used for convolution.
686        /// @return | LArray | New array containing convolution result.
687        methods.add_method("convolve1d", |lua, this, kernel: LuaAnyUserData| {

690                .map_err(LuaError::RuntimeError)?;
691            lua.create_userdata(LuaArray { inner: r })
692        });
693        // -- correlate1d --
694        /// Returns one-dimensional correlation with a template array.
695        /// @param | template | LArray | Template array used for correlation.
696        /// @return | LArray | New array containing correlation result.
697        methods.add_method("correlate1d", |lua, this, template: LuaAnyUserData| {
698            let template = template.borrow::<LuaArray>()?;
699            let r = analytics::correlate1d(&this.inner, &template.inner)
700                .map_err(LuaError::RuntimeError)?;
701            lua.create_userdata(LuaArray { inner: r })
702        });
703        // -- normalizeVec --
704        /// Returns this vector normalized to unit length.
705        /// @return | LArray | New normalized vector array.
706        methods.add_method("normalizeVec", |lua, this, ()| {
707            let r = linalg::normalize_vec(&this.inner).map_err(LuaError::RuntimeError)?;
708            lua.create_userdata(LuaArray { inner: r })
709        });
710        // -- outer --
711        /// Returns outer product with another vector array.
712        /// @param | other | LArray | Vector array used as the second operand.
713        /// @return | LArray | New array containing outer product result.
714        methods.add_method("outer", |lua, this, other: LuaAnyUserData| {
715            let other = other.borrow::<LuaArray>()?;
716            let r = linalg::outer(&this.inner, &other.inner).map_err(LuaError::RuntimeError)?;
717            lua.create_userdata(LuaArray { inner: r })
718        });
719        // -- cross2d --
720        /// Returns two-dimensional cross product with another vector.
721        /// @param | other | LArray | Vector array used as the second operand.
722        /// @return | number | Scalar 2D cross product result.
723        methods.add_method("cross2d", |_, this, other: LuaAnyUserData| {
724            let other = other.borrow::<LuaArray>()?;
725            linalg::cross2d(&this.inner, &other.inner).map_err(LuaError::RuntimeError)
726        });
727        // -- transformPoints --
728        /// Transforms a point array by this transform matrix.
729        /// @param | pts | LArray | Point array to transform.
730        /// @return | LArray | New array containing transformed points.
731        methods.add_method("transformPoints", |lua, this, pts: LuaAnyUserData| {
732            let pts = pts.borrow::<LuaArray>()?;
733            let r = linalg::transform_points(&this.inner, &pts.inner)
734                .map_err(LuaError::RuntimeError)?;
735            lua.create_userdata(LuaArray { inner: r })
736        });
737        // -- sobel --
738        /// Computes Sobel gradients for this array.
739        /// @return | table | Table with `gx` and `gy` gradient arrays.
740        /// @field | gx | LArray | Gradient X array.
741        /// @field | gy | LArray | Gradient Y array.
742        methods.add_method("sobel", |lua, this, ()| {
743            let (gx, gy) = linalg::sobel(&this.inner).map_err(LuaError::RuntimeError)?;
744            let t = lua.create_table()?;
745            /// The 'gx' field value exposed to Lua scripts.
746            t.set("gx", lua.create_userdata(LuaArray { inner: gx })?)?;
747            /// The 'gy' field value exposed to Lua scripts.
748            t.set("gy", lua.create_userdata(LuaArray { inner: gy })?)?;
749            Ok(t)
750        });
751        // -- linsolve --
752        /// Solves a linear system using this matrix and a right-hand side array.
753        /// @param | b | LArray | Right-hand side array.
754        /// @return | LArray | Solution array.
755        methods.add_method("linsolve", |lua, this, b: LuaAnyUserData| {
756            let b = b.borrow::<LuaArray>()?;
757            let r = linalg::linsolve(&this.inner, &b.inner).map_err(LuaError::RuntimeError)?;
758            lua.create_userdata(LuaArray { inner: r })
759        });
760        // -- luDecompose --
761        /// Decomposes this matrix into LU data and permutation metadata.
762        /// @return | table | Table containing `n`, `det_sign`, `perm`, and `lu_data` fields.
763        /// @field | n | integer | N.
764        /// @field | det_sign | integer | Det sign.
765        /// @field | perm | integer[] | Permutation array.
766        /// @field | lu_data | number[] | LU decomposition data.
767        methods.add_method("luDecompose", |lua, this, ()| {
768            let decomp = crate::compute::linalg::lu_decompose(&this.inner)
769                .map_err(LuaError::RuntimeError)?;
770            let result = lua.create_table()?;
771            /// The 'n' field value exposed to Lua scripts.
772            result.set("n", decomp.n as i64)?;
773            /// Performs the 'det_sign' operation.
774            result.set("det_sign", decomp.det_sign as i64)?;
775            let perm_tbl = lua.create_table()?;
776            for (i, &p) in decomp.perm.iter().enumerate() {
777                perm_tbl.set(i + 1, p as i64 + 1)?;
778            }
779            /// Performs the 'perm' operation.
780            result.set("perm", perm_tbl)?;
781            let lu_tbl = lua.create_table()?;
782            for (i, &v) in decomp.lu_data.iter().enumerate() {
783                lu_tbl.set(i + 1, v)?;
784            }
785            /// Performs the 'lu_data' operation.
786            result.set("lu_data", lu_tbl)?;
787            Ok(result)
788        });
789        // -- eigenPower --
790        /// Estimates dominant eigenvalue and eigenvector using power iteration.
791        /// @param | max_iter | integer? | Maximum iteration count; zero uses the engine default.
792        /// @param | tol | number? | Convergence tolerance; zero uses the engine default.
793        /// @return | table | Table containing `value` and `vector` fields.
794        /// @field | value | number | Dominant eigenvalue.
795        /// @field | vector | number[] | Eigenvector.
796        methods.add_method(
797            "eigenPower",
798            |lua, this, (max_iter, tol): (Option<u32>, Option<f64>)| {
799                let (eigenvalue, vec) = crate::compute::linalg::eigenvalue_power(
800                    &this.inner,
801                    max_iter.unwrap_or(0),
802                    tol.unwrap_or(0.0),
803                )
804                .map_err(LuaError::RuntimeError)?;
805                let result = lua.create_table()?;
806                /// Performs the 'value' operation.
807                result.set("value", eigenvalue)?;
808                let v_tbl = lua.create_table()?;
809                for (i, &x) in vec.iter().enumerate() {
810                    v_tbl.set(i + 1, x)?;
811                }
812                /// Performs the 'vector' operation.
813                result.set("vector", v_tbl)?;
814                Ok(result)
815            },
816        );
817        // -- map --
818        /// Maps each element through a Lua function and returns a new array.
819        /// @param | func | function | Function called with each element value and returning a number.
820        /// @return | LArray | New array containing mapped values.
821        methods.add_method("map", |_lua, this, func: LuaFunction| {
822            let src = &this.inner;
823            let n = src.size();
824            let mut out =
825                NdArray::zeros(src.shape(), src.dtype()).map_err(LuaError::RuntimeError)?;
826            for i in 0..n {
827                let v = src.get_f64(i);
828                let result: f64 = func.call(v)?;
829                out.set_f64(i, result);
830            }
831            Ok(LuaArray { inner: out })
832        });
833        // -- eval --
834        /// Maps each element through a Lua expression compiled as `function(x) return expression end`.
835        /// @param | expr | string | Lua expression that can read the current element as `x`.
836        /// @return | LArray | New array containing expression results.
837        methods.add_method("eval", |lua, this, expr: String| {
838            let src_code = format!("return function(x) return {} end", expr);
839            // LUA-EVAL-JUSTIFIED: compute.Array:eval compiles a user expression into a per-element mapper.
840            let func: LuaFunction = lua.load(&src_code).eval()?;
841            let src = &this.inner;
842            let n = src.size();
843            let mut out =
844                NdArray::zeros(src.shape(), src.dtype()).map_err(LuaError::RuntimeError)?;
845            for i in 0..n {
846                let v = src.get_f64(i);
847                let result: f64 = func.call(v)?;
848                out.set_f64(i, result);
849            }
850            Ok(LuaArray { inner: out })
851        });
852        // -- reduce --
853        /// Reduces array values with a Lua accumulator function.
854        /// @param | func | function | Function called as `(accumulator, value)` and returning the next accumulator.
855        /// @param | init | number | Initial accumulator value.
856        /// @return | number | Final accumulator value.
857        methods.add_method("reduce", |_, this, (func, init): (LuaFunction, f64)| {
858            let src = &this.inner;
859            let n = src.size();
860            let mut acc = init;
861            for i in 0..n {
862                let v = src.get_f64(i);
863                acc = func.call((acc, v))?;
864            }
865            Ok(acc)
866        });
867        // -- scan --
868        /// Produces prefix accumulator values with a Lua function.
869        /// @param | func | function | Function called as `(accumulator, value)` and returning the next accumulator.
870        /// @param | init | number | Initial accumulator value.
871        /// @return | LArray | New array containing accumulator values.
872        methods.add_method("scan", |_lua, this, (func, init): (LuaFunction, f64)| {
873            let src = &this.inner;
874            let n = src.size();
875            let mut out =
876                NdArray::zeros(src.shape(), src.dtype()).map_err(LuaError::RuntimeError)?;
877            let mut acc = init;
878            for i in 0..n {
879                let v = src.get_f64(i);
880                acc = func.call((acc, v))?;
881                out.set_f64(i, acc);
882            }
883            Ok(LuaArray { inner: out })
884        });
885        // -- type --
886        /// Returns the Lua-visible type name for this array handle.
887        /// @return | string | The string `LArray`.
888        methods.add_method("type", |_, _, ()| Ok("LArray"));
889        // -- typeOf --
890        /// Returns whether this array handle matches a supported type name.
891        /// @param | name | string | Type name to compare against `LArray`, `Array`, and `Object`.
892        /// @return | boolean | True when the supplied type name matches this handle.
893        methods.add_method("typeOf", |_, _, name: String| {
894            Ok(name == "LArray" || name == "LObject")
895        });
896    }
897}

1012    /// @param | ty | number | Translation Y component.

1020            |lua, (tx, ty, angle_rad, sx, sy): (f64, f64, f64, f64, f64)| {

1022                    linalg::affine2d(tx, ty, angle_rad, sx, sy).map_err(LuaError::RuntimeError)?;

1033    tbl.set(
1034        "fft",
1035        lua.create_function(|lua, samples: LuaTable| {
1036            let data: Vec<f64> = samples.sequence_values::<f64>().flatten().collect();
1037            let output = crate::compute::fft::fft(&data);
1038            let t = lua.create_table()?;
1039            for (i, (re, im)) in output.iter().enumerate() {
1040                let pair = lua.create_table()?;
1041                /// The 're' field value exposed to Lua scripts.
1042                pair.set("re", *re)?;
1043                /// The 'im' field value exposed to Lua scripts.
1044                pair.set("im", *im)?;
1045                t.set(i + 1, pair)?;
1046            }
1047            Ok(t)
1048        })?,
1049    )?;
1050    // -- ifft --
1051    /// Computes the inverse FFT of complex frequency pairs.
1052    /// @param | freqs | table | Array table of complex pairs with `re` and `im` fields.
1053    /// @return | number[] | Array table of reconstructed real-valued samples.
1054    tbl.set(
1055        "ifft",
1056        lua.create_function(|lua, freqs: LuaTable| {
1057            let pairs: Vec<(f64, f64)> = freqs
1058                .sequence_values::<LuaTable>()
1059                .flatten()
1060                .map(|entry| {
1061                    let re: f64 = entry.get("re").unwrap_or(0.0);
1062                    let im: f64 = entry.get("im").unwrap_or(0.0);
1063                    (re, im)
1064                })
1065                .collect();
1066            let output = crate::compute::fft::ifft(&pairs);
1067            let t = lua.create_table()?;
1068            for (i, v) in output.iter().enumerate() {
1069                t.set(i + 1, *v)?;
1070            }
1071            Ok(t)
1072        })?,
1073    )?;
1074    // -- fftMagnitude --
1075    /// Computes FFT magnitudes for real-valued samples.
1076    /// @param | samples | table | Array table of real-valued samples.
1077    /// @return | number[] | Array table of magnitude values.
1078    tbl.set(
1079        "fftMagnitude",
1080        lua.create_function(|lua, samples: LuaTable| {
1081            let data: Vec<f64> = samples.sequence_values::<f64>().flatten().collect();
1082            let mag = crate::compute::fft::fft_magnitude(&data);
1083            let t = lua.create_table()?;
1084            for (i, v) in mag.iter().enumerate() {
1085                t.set(i + 1, *v)?;
1086            }
1087            Ok(t)
1088        })?,
1089    )?;
1090    // -- getParThreshold --
1091    /// Returns the global compute parallelism threshold.
1092    /// @return | integer | Current parallel threshold.
1093    tbl.set(
1094        "getParThreshold",
1095        lua.create_function(|_, ()| Ok(crate::compute::get_par_threshold() as i64))?,
1096    )?;
1097    // -- setParThreshold --
1098    /// Sets the global compute parallelism threshold and returns the previous value.
1099    /// @param | threshold | integer | New threshold; values below one are clamped to one.
1100    /// @return | integer | Previous parallel threshold.
1101    tbl.set(
1102        "setParThreshold",
1103        lua.create_function(|_, threshold: i64| {
1104            let new_threshold = (threshold as usize).max(1);
1105            let prev = crate::compute::set_par_threshold(new_threshold);
1106            Ok(prev as i64)
1107        })?,
1108    )?;
1109    /// Performs the 'compute' operation.
1110    lurek.set("compute", tbl)?;
1111    Ok(())
