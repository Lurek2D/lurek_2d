32/// Flattens nested Lua translation tables into dot-separated string entries.
33fn flatten_lua_table(tbl: &LuaTable, prefix: &str, out: &mut HashMap<String, String>) {
34    for pair in tbl.clone().pairs::<LuaValue, LuaValue>() {
35        let Ok((k, v)) = pair else { continue };
36        let key_seg = match &k {
37            LuaValue::String(s) => s.to_str().unwrap_or("?").to_string(),
38            LuaValue::Integer(n) => n.to_string(),
39            _ => continue,
40        };
41        let full_key = if prefix.is_empty() {
42            key_seg.clone()
43        } else {
44            format!("{prefix}.{key_seg}")
45        };
46        match v {
47            LuaValue::Table(sub) => flatten_lua_table(&sub, &full_key, out),
48            LuaValue::String(s) => {
49                out.insert(full_key, s.to_str().unwrap_or("").to_string());
50            }
51            LuaValue::Integer(n) => {
52                out.insert(full_key, n.to_string());
53            }
54            LuaValue::Number(f) => {
55                out.insert(full_key, f.to_string());
56            }
57            LuaValue::Boolean(b) => {
58                out.insert(full_key, b.to_string());
59            }
60            _ => {}
61        }
62    }
63}

622            move |_, (locale, content, format): (String, String, String)| {
623                let flat = match format.to_lowercase().as_str() {
624                    "toml" => flat_table_from_toml(&content).map_err(mlua::Error::RuntimeError)?,
625                    "json" => flat_table_from_json(&content).map_err(mlua::Error::RuntimeError)?,
626                    other => {
627                        return Err(mlua::Error::RuntimeError(format!(
628                            "loadString: unknown format '{}'; expected 'toml' or 'json'",
629                            other
630                        )))
631                    }
632                };
633                s.borrow_mut().catalog.load(&locale, flat);
634                Ok(())
635            },
636        )?,
637    )?;
638    let s = shared.clone();
639    // -- localeCoverage --
640    /// Returns missing translation keys for all locales compared to a reference locale.
641    /// @param | reference | string | Locale code used as the coverage reference.
642    /// @return | table | Array of gap rows with `key` and `missing_in` fields.
643    /// @field | key | string | Translation key.
644    /// @field | missing_in | string[] | Locales missing this key.
645    loc.set(
646        "localeCoverage",
647        lua.create_function(move |lua, reference: String| {
648            let gaps = s.borrow().catalog.coverage_gaps(&reference);
649            let tbl = lua.create_table()?;
650            for (i, gap) in gaps.iter().enumerate() {
651                let row = lua.create_table()?;
652                /// The 'key' field value exposed to Lua scripts.
653                row.set("key", gap.key.as_str())?;
654                let missing = lua.create_table()?;
655                for (j, loc) in gap.missing_in.iter().enumerate() {
656                    missing.set(j + 1, loc.as_str())?;
657                }
658                /// Performs the 'missing_in' operation.
659                row.set("missing_in", missing)?;
660                tbl.set(i + 1, row)?;
661            }
662            Ok(tbl)
663        })?,
664    )?;
665    /// Performs the 'i18n' operation.
666    lurek.set("i18n", loc)?;
667    Ok(())
668}
