11/// Lua-visible wrapper around a SpriteSheet, providing grid-based frame access,.
12/// named animation groups, and row/column slicing for sprite sheet textures.
13pub struct LuaSpriteSheet {
14    inner: SpriteSheet,
15}
16impl LuaUserData for LuaSpriteSheet {
17    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
18        // -- getFrame --
19        /// Returns the UV quad for a single frame by its 1-based index.
20        /// @param | index | integer | 1-based frame index in the sprite sheet.
21        /// @return | table | Quad table `{x, y, w, h}` with normalized UV coordinates, or nil if the index is out of range.
22        /// @field | x | number | X.
23        /// @field | y | number | Y.
24        /// @field | w | number | W.
25        /// @field | h | number | H.
26        methods.add_method("getFrame", |lua, this, index: usize| {
27            match this.inner.get_frame(index) {
28                Some(r) => {
29                    let t = quad_table(lua, r)?;
30                    Ok(LuaValue::Table(t))
31                }
32                None => Ok(LuaValue::Nil),
33            }
34        });
35        // -- getFrameCount --
36        /// Returns the total number of frames in this sprite sheet.
37        /// @return | integer | Total frame count (columns × rows).
38        methods.add_method("getFrameCount", |_, this, ()| {
39            Ok(this.inner.get_frame_count())
40        });
41        // -- getRow --
42        /// Returns all frame quads in the given row of the sprite sheet grid.
43        /// @param | row | integer | 0-based row index.
44        /// @return | table | Array of quad tables `{x, y, w, h}`.
45        /// @field | x | number | X.
46        /// @field | y | number | Y.
47        /// @field | w | number | W.
48        /// @field | h | number | H.
49        methods.add_method("getRow", |lua, this, row: u32| {
50            let frames = this.inner.get_row(row);
51            frames_to_table(lua, &frames)
52        });
53        // -- getColumn --
54        /// Returns all frame quads in the given column of the sprite sheet grid.
55        /// @param | col | integer | 0-based column index.
56        /// @return | table | Array of quad tables `{x, y, w, h}`.
57        /// @field | x | number | X.
58        /// @field | y | number | Y.
59        /// @field | w | number | W.
60        /// @field | h | number | H.
61        methods.add_method("getColumn", |lua, this, col: u32| {
62            let frames = this.inner.get_column(col);
63            frames_to_table(lua, &frames)
64        });
65        // -- getGroupFrames --
66        /// Returns the frame quads for a named animation group.
67        /// @param | name | string | Name of the animation group (e.g. "walk", "idle").
68        /// @return | table | Array of quad tables for the group, or nil if the group does not exist.
69        /// @field | x | number | X position in atlas.
70        /// @field | y | number | Y position in atlas.
71        /// @field | w | number | Width.
72        /// @field | h | number | Height.
73        methods.add_method("getGroupFrames", |lua, this, name: String| {
74            match this.inner.get_group(&name) {
75                Some(frames) => {
76                    let t = frames_to_table(lua, &frames)?;
77                    Ok(LuaValue::Table(t))
78                }
79                None => Ok(LuaValue::Nil),
80            }
81        });
82        // -- getGroupNames --
83        /// Returns an array of all named animation group names defined on this sheet.
84        /// @return | string[] | Group name strings.
85        methods.add_method("getGroupNames", |lua, this, ()| {
86            let names = this.inner.get_group_names();
87            let t = lua.create_table()?;
88            for (i, n) in names.iter().enumerate() {
89                t.set(i + 1, n.as_str())?;
90            }
91            Ok(t)
92        });
93        // -- nameGroup --
94        /// Defines a named animation group as a contiguous range of frames.
95        /// @param | name | string | Name for the group (e.g. "attack").
96        /// @param | start | integer | 1-based start frame index.
97        /// @param | count | integer | Number of frames in the group.
98        methods.add_method_mut(
99            "nameGroup",
100            |_, this, (name, start, count): (String, usize, usize)| {
101                this.inner.name_group(name, start, count);
102                Ok(())
103            },
104        );
105        // -- getFrameSize --
106        /// Returns the pixel dimensions of a single frame cell.
107        /// @return | integer | Frame width in pixels.
108        /// @return | integer | Frame height in pixels.
109        methods.add_method("getFrameSize", |_, this, ()| {
110            let (w, h) = this.inner.get_frame_size();
111            Ok((w, h))
112        });
113        // -- getGridSize --
114        /// Returns the number of columns and rows in the sprite sheet grid.
115        /// @return | integer | Number of columns.
116        /// @return | integer | Number of rows.
117        methods.add_method("getGridSize", |_, this, ()| {
118            let (cols, rows) = this.inner.get_grid_size();
119            Ok((cols, rows))
120        });
121        // -- drawToImage --
122        /// Renders the sprite sheet grid into an LImage of the given size for debugging or previews.
123        /// @param | w | integer | Output image width in pixels.
124        /// @param | h | integer | Output image height in pixels.
125        /// @return | LImage | A new image containing the rendered sprite sheet.
126        methods.add_method("drawToImage", |lua, this, (w, h): (u32, u32)| {
127            let img = this.inner.draw_to_image(w, h);
128            lua.create_userdata(img)
129        });
130        // -- type --
131        /// Returns the type name of this object.
132        /// @return | string | Always `"LSpriteSheet"`.
133        methods.add_method("type", |_, _, ()| Ok("LSpriteSheet"));
134        // -- typeOf --
135        /// Checks whether this object matches the given type name.
136        /// @param | name | string | Type name to check (e.g. `"LSpriteSheet"` or `"Object"`).
137        /// @return | boolean | True if the object is the given type.
138        methods.add_method("typeOf", |_, _, name: String| {
139            Ok(name == "LSpriteSheet" || name == "LObject")
140        });
141    }
142}

144/// Lua-visible wrapper around a SpriteAtlas, providing named region lookups.
145/// for packed texture atlases exported from tools like TexturePacker or Aseprite.
146pub struct LuaSpriteAtlas {
147    inner: SpriteAtlas,
148}
149impl LuaUserData for LuaSpriteAtlas {
150    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
151        // -- getEntry --
152        /// Looks up a named sprite region in the atlas by its original filename or tag.
153        /// @param | name | string | Entry name (e.g. `"player_idle_0"`).
154        /// @return | table | Entry table `{name, x, y, w, h, rotated}`, or nil if the entry is not found.
155        /// @field | name | string | Entry name.
156        /// @field | x | number | X.
157        /// @field | y | number | Y.
158        /// @field | w | number | W.
159        /// @field | h | number | H.
160        /// @field | rotated | boolean | Whether the entry is rotated.
161        methods.add_method("getEntry", |lua, this, name: String| {
162            match this.inner.get_entry(&name) {
163                Some(e) => {
164                    let t = lua.create_table()?;
165                    /// Performs the 'name' operation.
166                    t.set("name", e.name.as_str())?;
167                    /// The 'x' field value exposed to Lua scripts.
168                    t.set("x", e.x)?;
169                    /// The 'y' field value exposed to Lua scripts.
170                    t.set("y", e.y)?;
171                    /// The 'w' field value exposed to Lua scripts.
172                    t.set("w", e.w)?;
173                    /// The 'h' field value exposed to Lua scripts.
174                    t.set("h", e.h)?;
175                    /// Performs the 'rotated' operation.
176                    t.set("rotated", e.rotated)?;
177                    Ok(LuaValue::Table(t))
178                }
179                None => Ok(LuaValue::Nil),
180            }
181        });
182        // -- getByIndex --
183        /// Returns a sprite region by its 1-based index in the atlas.
184        /// @param | index | integer | 1-based entry index.
185        /// @return | table | Entry table `{name, x, y, w, h, rotated}`, or nil if the index is out of range.
186        /// @field | name | string | Entry name.
187        /// @field | x | number | X.
188        /// @field | y | number | Y.
189        /// @field | w | number | W.
190        /// @field | h | number | H.
191        /// @field | rotated | boolean | Whether the entry is rotated.
192        /// @field | flip_x | boolean | Flip horizontally.
193        /// @field | flip_y | boolean | Flip vertically.
194        methods.add_method("getByIndex", |lua, this, index: usize| {
195            match this.inner.get_by_index(index.saturating_sub(1)) {
196                Some(e) => {
197                    let t = lua.create_table()?;
198                    /// Performs the 'name' operation.
199                    t.set("name", e.name.as_str())?;
200                    /// The 'x' field value exposed to Lua scripts.
201                    t.set("x", e.x)?;
202                    /// The 'y' field value exposed to Lua scripts.
203                    t.set("y", e.y)?;
204                    /// The 'w' field value exposed to Lua scripts.
205                    t.set("w", e.w)?;
206                    /// The 'h' field value exposed to Lua scripts.
207                    t.set("h", e.h)?;
208                    /// Performs the 'rotated' operation.
209                    t.set("rotated", e.rotated)?;
210                    Ok(LuaValue::Table(t))
211                }
212                None => Ok(LuaValue::Nil),
213            }
214        });
215        // -- entryCount --
216        /// Returns the total number of entries (sprite regions) in the atlas.
217        /// @return | integer | Entry count.
218        methods.add_method("entryCount", |_, this, ()| Ok(this.inner.entry_count()));
219        // -- entryNames --
220        /// Returns an array of all entry names in the atlas.
221        /// @return | string[] | Name strings.
222        methods.add_method("entryNames", |lua, this, ()| {
223            let names = this.inner.entry_names();
224            let t = lua.create_table()?;
225            for (i, n) in names.iter().enumerate() {
226                t.set(i + 1, *n)?;
227            }
228            Ok(t)
229        });
230        // -- getFlipped --
231        /// Returns a copy of a named atlas entry with the specified flip flags applied.
232        /// @param | name | string | Entry name to look up.
233        /// @param | flip_x | boolean | Mirror horizontally.
234        /// @param | flip_y | boolean | Mirror vertically.
235        /// @return | table | Entry table with added `flip_x` and `flip_y` fields, or nil if the entry is not found.
236        /// @field | name | string | Entry name.
237        /// @field | x | number | X.
238        /// @field | y | number | Y.
239        /// @field | w | number | W.
240        /// @field | h | number | H.
241        /// @field | rotated | boolean | Whether the entry is rotated.
242        /// @field | flip_x | boolean | Flip horizontally.
243        /// @field | flip_y | boolean | Flip vertically.
244        methods.add_method(
245            "getFlipped",
246            |lua, this, (name, flip_x, flip_y): (String, bool, bool)| match this
247                .inner
248                .get_entry(&name)
249            {
250                Some(e) => {
251                    let flipped = e.get_flipped(flip_x, flip_y);
252                    let t = lua.create_table()?;
253                    /// Performs the 'name' operation.
254                    t.set("name", flipped.name.as_str())?;
255                    /// The 'x' field value exposed to Lua scripts.
256                    t.set("x", flipped.x)?;
257                    /// The 'y' field value exposed to Lua scripts.
258                    t.set("y", flipped.y)?;
259                    /// The 'w' field value exposed to Lua scripts.
260                    t.set("w", flipped.w)?;
261                    /// The 'h' field value exposed to Lua scripts.
262                    t.set("h", flipped.h)?;
263                    /// Performs the 'rotated' operation.
264                    t.set("rotated", flipped.rotated)?;
265                    /// Performs the 'flip_x' operation.
266                    t.set("flip_x", flipped.flip_x)?;
267                    /// Performs the 'flip_y' operation.
268                    t.set("flip_y", flipped.flip_y)?;
269                    Ok(LuaValue::Table(t))
270                }
271                None => Ok(LuaValue::Nil),
272            },
273        );
274        // -- type --
275        /// Returns the type name of this object.
276        /// @return | string | Always `"LSpriteAtlas"`.
277        methods.add_method("type", |_, _, ()| Ok("LSpriteAtlas"));
278        // -- typeOf --
279        /// Checks whether this object matches the given type name.
280        /// @param | name | string | Type name to check (e.g. `"LSpriteAtlas"` or `"Object"`).
281        /// @return | boolean | True if the object is the given type.
282        methods.add_method("typeOf", |_, _, name: String| {
283            Ok(name == "LSpriteAtlas" || name == "LObject")
284        });
285    }
286}

288/// Registers the `lurek.sprite` module, exposing sprite sheet and texture atlas constructors.
289pub fn register(lua: &Lua, lurek: &LuaTable, _state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
290    let tbl = lua.create_table()?;
291
292    // -- newSheet --
293    /// Creates a new sprite sheet by dividing a texture of the given pixel size into a grid of equal-sized frames.
294    /// @param | tw | integer | Full texture width in pixels.
295    /// @param | th | integer | Full texture height in pixels.
296    /// @param | fw | integer | Single frame width in pixels.
297    /// @param | fh | integer | Single frame height in pixels.
298    /// @return | LSpriteSheet | A new sprite sheet object.
299    tbl.set(
300        "newSheet",
301        lua.create_function(|lua, (tw, th, fw, fh): (u32, u32, u32, u32)| {
302            lua.create_userdata(LuaSpriteSheet {
303                inner: SpriteSheet::new(tw, th, fw, fh),
304            })
305        })?,
306    )?;
307    // -- newRPGMakerSheet --
308    /// Creates a sprite sheet using RPG Maker's standard character layout (4 columns × 4 rows per character block).
309    /// @param | tw | integer | Full texture width in pixels.
310    /// @param | th | integer | Full texture height in pixels.
311    /// @return | LSpriteSheet | A new sprite sheet configured for RPG Maker character sprites.
312    tbl.set(
313        "newRPGMakerSheet",
314        lua.create_function(|lua, (tw, th): (u32, u32)| {
315            lua.create_userdata(LuaSpriteSheet {
316                inner: SpriteSheet::from_rpgmaker(tw, th),
317            })
318        })?,
319    )?;
320    // -- parseAtlas --
321    /// Parses a TexturePacker JSON atlas string and returns a sprite atlas object.
322    /// @param | json_str | string | Raw JSON content of the TexturePacker atlas file.
323    /// @return | LSpriteAtlas | A new atlas with named sprite regions.
324    tbl.set(
325        "parseAtlas",
326        lua.create_function(
327            |lua, json_str: String| match parse_texturepacker_json(&json_str) {
328                Ok(atlas) => {
329                    let ud = lua.create_userdata(LuaSpriteAtlas { inner: atlas })?;
330                    Ok(LuaValue::UserData(ud))
331                }
332                Err(e) => Err(LuaError::RuntimeError(format!("parseAtlas: {}", e))),
333            },
334        )?,
335    )?;
336    // -- newAtlasSheet --
337    /// Creates a sprite sheet from an existing atlas, treating each atlas entry as a frame within the given sheet dimensions.
338    /// @param | atlas | LSpriteAtlas | A previously parsed sprite atlas.
339    /// @param | sw | integer | Sheet texture width in pixels.
340    /// @param | sh | integer | Sheet texture height in pixels.
341    /// @return | LSpriteSheet | A new sprite sheet derived from the atlas entries.
342    tbl.set(
343        "newAtlasSheet",
344        lua.create_function(|lua, (atlas_ud, sw, sh): (LuaAnyUserData, u32, u32)| {
345            let atlas = atlas_ud.borrow::<LuaSpriteAtlas>()?;
346            lua.create_userdata(LuaSpriteSheet {
347                inner: SpriteSheet::from_atlas(&atlas.inner, sw, sh),
348            })
349        })?,
350    )?;
351    // -- parseAsepriteAtlas --
352    /// Parses an Aseprite JSON atlas string and returns a sprite atlas object.
353    /// @param | json_str | string | Raw JSON content of the Aseprite export atlas file.
354    /// @return | LSpriteAtlas | A new atlas with named sprite regions from Aseprite frames.
355    tbl.set(
356        "parseAsepriteAtlas",
357        lua.create_function(|lua, json_str: String| {
358            let atlas = parse_aseprite_json(&json_str).map_err(LuaError::RuntimeError)?;
359            lua.create_userdata(LuaSpriteAtlas { inner: atlas })
360        })?,
361    )?;
362    /// Performs the 'sprite' operation.
363    lurek.set("sprite", tbl)?;
364    Ok(())
365}
366/// Converts a sprite rectangle into the Lua quad table returned by atlas helpers.
367fn quad_table(lua: &Lua, r: Rect) -> LuaResult<LuaTable<'_>> {
368    let t = lua.create_table()?;
369    /// The 'x' field value exposed to Lua scripts.
370    t.set("x", r.x)?;
371    /// The 'y' field value exposed to Lua scripts.
372    t.set("y", r.y)?;
373    /// The 'w' field value exposed to Lua scripts.
374    t.set("w", r.width)?;
375    /// The 'h' field value exposed to Lua scripts.
376    t.set("h", r.height)?;
377    Ok(t)
378}
379/// Converts a slice of sprite rectangles into an array-style Lua table.
380fn frames_to_table<'lua>(lua: &'lua Lua, frames: &[Rect]) -> LuaResult<LuaTable<'lua>> {
381    let t = lua.create_table()?;
382    for (i, r) in frames.iter().enumerate() {
383        t.set(i + 1, quad_table(lua, *r)?)?;
384    }
385    Ok(t)
