10/// Lua-facing publish/subscribe event bus allowing decoupled communication between game systems.
11#[derive(Clone)]
12struct LuaEventBus {
13    bus: Rc<RefCell<crate::patterns::EventBus>>,
14    callbacks: Rc<RefCell<HashMap<u64, LuaRegistryKey>>>,
15}
16impl LurekType for LuaEventBus {
17    const TYPE_NAME: &'static str = "LEventBus";
18    const TYPE_HIERARCHY: &'static [&'static str] = &["LEventBus", "LObject"];
19}
20impl LuaUserData for LuaEventBus {
21    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
22        add_type_methods(methods);
23        // -- on --
24        /// Subscribe a callback to a named event. Higher priority listeners fire first.
25        /// @param | event | string | The event name to listen for.
26        /// @param | callback | function | The function to invoke when the event fires.
27        /// @param | priority | integer? | Listener priority (default 0). Higher values execute first.
28        /// @return | number | A subscription ID used to unsubscribe later.
29        methods.add_method(
30            "on",
31            |lua, this, (event, callback, priority): (String, LuaFunction, Option<i64>)| {
32                let priority = priority.unwrap_or(0);
33                let id = this.bus.borrow_mut().subscribe(&event, priority, false);
34                let key = lua.create_registry_value(callback)?;
35                this.callbacks.borrow_mut().insert(id, key);
36                Ok(id)
37            },
38        );
39        // -- off --
40        /// Unsubscribe a listener by its subscription ID. Removes the callback from the event bus.
41        /// @param | id | integer | The subscription ID returned by `on()`.
42        methods.add_method("off", |lua, this, id: u64| {
43            this.bus.borrow_mut().unsubscribe(id);
44            if let Some(key) = this.callbacks.borrow_mut().remove(&id) {
45                lua.remove_registry_value(key)?;
46            }
47            Ok(())
48        });
49        // -- emit --
50        /// Emit an event, invoking all subscribed listeners in priority order with optional payload arguments.
51        /// @param | event | string | The event name to emit.
52        /// @param | ... | any | Additional arguments passed to each listener callback.
53        methods.add_method("emit", |lua, this, args: LuaMultiValue| {
54            let mut args_iter = args.into_iter();
55            let event: String = match args_iter.next() {
56                Some(v) => lua.unpack(v)?,
57                None => return Err(LuaError::external("emit requires an event name")),
58            };
59            let extra: Vec<LuaValue> = args_iter.collect();
60            let ids = this.bus.borrow().get_listeners(&event);
61            for id in &ids {
62                let cbs = this.callbacks.borrow();
63                if let Some(key) = cbs.get(id) {
64                    let func: LuaFunction = lua.registry_value(key)?;
65                    drop(cbs);
66                    func.call::<_, ()>(LuaMultiValue::from_iter(extra.clone()))?;
67                }
68            }
69            let removed = this.bus.borrow_mut().drain_once(&ids);
70            let mut cbs = this.callbacks.borrow_mut();
71            for id in removed {
72                if let Some(key) = cbs.remove(&id) {
73                    lua.remove_registry_value(key)?;
74                }
75            }
76            Ok(())
77        });
78        // -- clear --
79        /// Remove all listeners subscribed to a specific event name.
80        /// @param | event | string | The event name whose listeners will be removed.
81        methods.add_method("clear", |lua, this, event: String| {
82            let removed_ids = this.bus.borrow_mut().clear_event(&event);
83            let mut cbs = this.callbacks.borrow_mut();
84            for id in removed_ids {
85                if let Some(key) = cbs.remove(&id) {
86                    lua.remove_registry_value(key)?;
87                }
88            }
89            Ok(())
90        });
91        // -- clearAll --
92        /// Remove all listeners from every event on this bus. Resets the bus to empty.
93        methods.add_method("clearAll", |lua, this, ()| {
94            let _ = this.bus.borrow_mut().clear_all();
95            let drained: Vec<(u64, LuaRegistryKey)> = this.callbacks.borrow_mut().drain().collect();
96            for (_, key) in drained {
97                lua.remove_registry_value(key)?;
98            }
99            Ok(())
100        });
101        // -- getListenerCount --
102        /// Return the number of active listeners for a given event name.
103        /// @param | event | string | The event name to query.
104        /// @return | number | Count of currently registered listeners.
105        methods.add_method("getListenerCount", |_lua, this, event: String| {
106            Ok(this.bus.borrow().listener_count(&event))
107        });
108        // -- getEvents --
109        /// Return an array of all event names that have at least one listener.
110        /// @return | string[] | Event name strings.
111        methods.add_method("getEvents", |lua, this, ()| {
112            let names = this.bus.borrow().event_names();
113            let table = lua.create_table()?;
114            for (i, name) in names.iter().enumerate() {
115                table.set(i + 1, name.as_str())?;
116            }
117            Ok(table)
118        });
119    }

121/// Lua-facing object pool for reusing pre-allocated game objects (bullets, particles, enemies) to avoid per-frame allocations.
122#[derive(Clone)]
123struct LuaObjectPool {
124    pool: Rc<RefCell<crate::patterns::ObjectPool>>,
125    idle_objects: Rc<RefCell<HashMap<u64, LuaRegistryKey>>>,
126    active_queue: Rc<RefCell<VecDeque<u64>>>,
127}
128impl LurekType for LuaObjectPool {
129    const TYPE_NAME: &'static str = "LObjectPool";
130    const TYPE_HIERARCHY: &'static [&'static str] = &["LObjectPool", "LObject"];
131}
132impl LuaUserData for LuaObjectPool {
133    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
134        add_type_methods(methods);
135        // -- add --
136        /// Add an object to the pool's idle set, making it available for future acquisition.
137        /// @param | value | any | The object value to store in the pool.
138        methods.add_method("add", |lua, this, value: LuaValue| {
139            let total = this.pool.borrow().total_count();
140            let new_ids = this.pool.borrow_mut().prewarm(total + 1);
141            if let Some(&id) = new_ids.first() {
142                let key = lua.create_registry_value(value)?;
143                this.idle_objects.borrow_mut().insert(id, key);
144            }
145            Ok(())
146        });
147        // -- acquire --
148        /// Take an idle object from the pool and mark it active. Returns nil if the pool is empty.
149        /// @return | table | The acquired object table.
150        /// @return | nil | If none available.
151        methods.add_method("acquire", |lua, this, ()| {
152            if let Some(id) = this.pool.borrow_mut().acquire() {
153                if let Some(key) = this.idle_objects.borrow_mut().remove(&id) {
154                    let val: LuaValue = lua.registry_value(&key)?;
155                    lua.remove_registry_value(key)?;
156                    this.active_queue.borrow_mut().push_back(id);
157                    return Ok(val);
158                }
159                this.pool.borrow_mut().release(id);
160            }
161            Ok(LuaValue::Nil)
162        });
163        // -- release --
164        /// Return an active object back to the pool's idle set so it can be reused.
165        /// @param | value | any | The object value to release back into the pool.
166        methods.add_method("release", |lua, this, value: LuaValue| {
167            if let Some(id) = this.active_queue.borrow_mut().pop_front() {
168                this.pool.borrow_mut().release(id);
169                let key = lua.create_registry_value(value)?;
170                this.idle_objects.borrow_mut().insert(id, key);
171            }
172            Ok(())
173        });
174        // -- getActiveCount --
175        /// Return the number of objects currently checked out from the pool.
176        /// @return | number | Count of active (in-use) objects.
177        methods.add_method("getActiveCount", |_lua, this, ()| {
178            Ok(this.pool.borrow().active_count())
179        });
180        // -- getAvailableCount --
181        /// Return the number of idle objects ready for acquisition.
182        /// @return | number | Count of available (idle) objects.
183        methods.add_method("getAvailableCount", |_lua, this, ()| {
184            Ok(this.pool.borrow().idle_count())
185        });
186        // -- getTotalCount --
187        /// Return the total number of objects managed by this pool (active + idle).
188        /// @return | number | Total object count.
189        methods.add_method("getTotalCount", |_lua, this, ()| {
190            Ok(this.pool.borrow().total_count())
191        });
192        // -- clearAll --
193        /// Destroy all objects (active and idle) and reset the pool to empty.
194        methods.add_method("clearAll", |lua, this, ()| {
195            let cap = this.pool.borrow().capacity;
196            *this.pool.borrow_mut() = crate::patterns::ObjectPool::new("", cap);
197            this.active_queue.borrow_mut().clear();
198            let drained: Vec<(u64, LuaRegistryKey)> =
199                this.idle_objects.borrow_mut().drain().collect();
200            for (_, key) in drained {
201                lua.remove_registry_value(key)?;
202            }
203            Ok(())
204        });
205    }

207/// Lua-facing undo/redo command stack. Records executed actions with optional undo functions for full history navigation.
208#[derive(Clone)]
209struct LuaCommandStack {
210    stack: Rc<RefCell<crate::patterns::CommandStack>>,
211    exec_fns: Rc<RefCell<HashMap<u64, LuaRegistryKey>>>,
212    undo_fns: Rc<RefCell<HashMap<u64, LuaRegistryKey>>>,
213    history_ids: Rc<RefCell<Vec<u64>>>,
214}
215impl LurekType for LuaCommandStack {
216    const TYPE_NAME: &'static str = "LCommandStack";
217    const TYPE_HIERARCHY: &'static [&'static str] = &["LCommandStack", "LObject"];
218}
219impl LuaUserData for LuaCommandStack {
220    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
221        add_type_methods(methods);
222        // -- execute --
223        /// Execute a named command immediately, recording it in history. Discards any redo history ahead of the current position.
224        /// @param | name | string | A descriptive name for the command (shown in history).
225        /// @param | execFn | function | The function that performs the action.
226        /// @param | undoFn | function? | An optional function that reverses the action. If omitted, command cannot be undone.
227        methods.add_method(
228            "execute",
229            |lua, this, (name, exec_fn, undo_fn): (String, LuaFunction, Option<LuaFunction>)| {
230                let undo_count = this.stack.borrow().undo_count();
231                let discarded: Vec<u64> = {
232                    let mut ids = this.history_ids.borrow_mut();
233                    ids.drain(undo_count..).collect()
234                };
235                {
236                    let mut exec_fns = this.exec_fns.borrow_mut();
237                    let mut undo_fns = this.undo_fns.borrow_mut();
238                    for id in discarded {
239                        if let Some(k) = exec_fns.remove(&id) {
240                            lua.remove_registry_value(k)?;
241                        }
242                        if let Some(k) = undo_fns.remove(&id) {
243                            lua.remove_registry_value(k)?;
244                        }
245                    }
246                }
247                let has_undo = undo_fn.is_some();
248                let entry_id = this.stack.borrow_mut().push(&name, has_undo);
249                this.history_ids.borrow_mut().push(entry_id);
250                let expected_total = {
251                    let s = this.stack.borrow();
252                    s.undo_count() + s.redo_count()
253                };
254                while this.history_ids.borrow().len() > expected_total {
255                    let oldest_id = this.history_ids.borrow_mut().remove(0);
256                    if let Some(k) = this.exec_fns.borrow_mut().remove(&oldest_id) {
257                        lua.remove_registry_value(k)?;
258                    }
259                    if let Some(k) = this.undo_fns.borrow_mut().remove(&oldest_id) {
260                        lua.remove_registry_value(k)?;
261                    }
262                }
263                exec_fn.call::<_, ()>(())?;
264                this.exec_fns
265                    .borrow_mut()
266                    .insert(entry_id, lua.create_registry_value(exec_fn)?);
267                if let Some(f) = undo_fn {
268                    this.undo_fns
269                        .borrow_mut()
270                        .insert(entry_id, lua.create_registry_value(f)?);
271                }
272                Ok(())
273            },
274        );
275        // -- undo --
276        /// Undo the most recent command by calling its undo function. Moves the pointer back in history.
277        /// @return | boolean | True if undo succeeded, false if nothing to undo or no undo function registered.
278        methods.add_method("undo", |lua, this, ()| {
279            let peek_id = this.stack.borrow().peek_undo();
280            if let Some(id) = peek_id {
281                let has_undo = this
282                    .stack
283                    .borrow()
284                    .get_entry(id)
285                    .map(|e| e.has_undo)
286                    .unwrap_or(false);
287                if !has_undo {
288                    return Ok(false);
289                }
290                let func_opt = this
291                    .undo_fns
292                    .borrow()
293                    .get(&id)
294                    .map(|k| lua.registry_value::<LuaFunction>(k));
295                if let Some(Ok(func)) = func_opt {
296                    func.call::<_, ()>(())?;
297                    this.stack.borrow_mut().step_undo();
298                    Ok(true)
299                } else {
300                    Ok(false)
301                }
302            } else {
303                Ok(false)
304            }
305        });
306        // -- redo --
307        /// Redo a previously undone command by re-calling its execute function. Moves the pointer forward.
308        /// @return | boolean | True if redo succeeded, false if nothing to redo.
309        methods.add_method("redo", |lua, this, ()| {
310            let peek_id = this.stack.borrow().peek_redo();
311            if let Some(id) = peek_id {
312                let func_opt = this
313                    .exec_fns
314                    .borrow()
315                    .get(&id)
316                    .map(|k| lua.registry_value::<LuaFunction>(k));
317                if let Some(Ok(func)) = func_opt {
318                    func.call::<_, ()>(())?;
319                    this.stack.borrow_mut().step_redo();
320                    Ok(true)
321                } else {
322                    Ok(false)
323                }
324            } else {
325                Ok(false)
326            }
327        });
328        // -- canUndo --
329        /// Check whether an undo operation is possible (there is a command with an undo function behind the pointer).
330        /// @return | boolean | True if undo is available.
331        methods.add_method("canUndo", |_lua, this, ()| {
332            let s = this.stack.borrow();
333            Ok(s.peek_undo()
334                .and_then(|id| s.get_entry(id))
335                .map(|e| e.has_undo)
336                .unwrap_or(false))
337        });
338        // -- canRedo --
339        /// Check whether a redo operation is possible (there are commands ahead of the pointer).
340        /// @return | boolean | True if redo is available.
341        methods.add_method("canRedo", |_lua, this, ()| {
342            Ok(this.stack.borrow().redo_count() > 0)
343        });
344        // -- getHistorySize --
345        /// Return the total number of commands in the history (both undone and available for redo).
346        /// @return | number | Total history depth.
347        methods.add_method("getHistorySize", |_lua, this, ()| {
348            let s = this.stack.borrow();
349            Ok(s.undo_count() + s.redo_count())
350        });
351        // -- getCurrentName --
352        /// Return the name of the most recently executed (or undone-to) command, or nil if history is empty.
353        /// @return | string | The command name, or nil if history is empty.
354        methods.add_method("getCurrentName", |_lua, this, ()| {
355            let s = this.stack.borrow();
356            Ok(s.peek_undo()
357                .and_then(|id| s.get_entry(id))
358                .map(|e| e.name.clone()))
359        });
360        // -- clearAll --
361        /// Discard all command history and free associated callbacks.
362        methods.add_method("clearAll", |lua, this, ()| {
363            this.stack.borrow_mut().clear();
364            this.history_ids.borrow_mut().clear();
365            for (_, key) in this.exec_fns.borrow_mut().drain() {
366                lua.remove_registry_value(key)?;
367            }
368            for (_, key) in this.undo_fns.borrow_mut().drain() {
369                lua.remove_registry_value(key)?;
370            }
371            Ok(())
372        });
373    }
374}
375/// Lua-facing service locator for registering and retrieving shared services by name at runtime.
376#[derive(Clone)]
377struct LuaServiceLocator {
378    locator: Rc<RefCell<crate::patterns::ServiceLocator>>,
379    services: Rc<RefCell<HashMap<String, LuaRegistryKey>>>,
380}
381impl LurekType for LuaServiceLocator {
382    const TYPE_NAME: &'static str = "LServiceLocator";
383    const TYPE_HIERARCHY: &'static [&'static str] = &["LServiceLocator", "LObject"];

385impl LuaUserData for LuaServiceLocator {
386    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
387        add_type_methods(methods);
388        // -- provide --
389        /// Register a service instance under a given name. Replaces any previously registered service with the same name.
390        /// @param | name | string | Unique identifier for the service.
391        /// @param | value | any | The service value to register.
392        methods.add_method("provide", |lua, this, (name, value): (String, LuaValue)| {
393            this.locator.borrow_mut().register(&name);
394            let key = lua.create_registry_value(value)?;
395            if let Some(old) = this.services.borrow_mut().insert(name, key) {
396                lua.remove_registry_value(old)?;
397            }
398            Ok(())
399        });
400        // -- locate --
401        /// Retrieve a registered service by name. Returns nil if not found.
402        /// @param | name | string | The service name to look up.
403        /// @return | table | The service table.
404        /// @return | nil | If not registered.
405        methods.add_method("locate", |lua, this, name: String| {
406            let svc = this.services.borrow();
407            match svc.get(&name) {
408                Some(key) => Ok(lua.registry_value::<LuaValue>(key)?),
409                None => Ok(LuaValue::Nil),
410            }
411        });
412        // -- has --
413        /// Check whether a service with the given name is currently registered.
414        /// @param | name | string | The service name to check.
415        /// @return | boolean | True if the service exists.
416        methods.add_method("has", |_lua, this, name: String| {
417            Ok(this.locator.borrow().has(&name))
418        });
419        // -- remove --
420        /// Unregister and discard a service by name.
421        /// @param | name | string | The service name to remove.
422        methods.add_method("remove", |lua, this, name: String| {
423            this.locator.borrow_mut().unregister(&name);
424            if let Some(key) = this.services.borrow_mut().remove(&name) {
425                lua.remove_registry_value(key)?;
426            }
427            Ok(())
428        });
429        // -- getServices --
430        /// Return an array of all registered service names.
431        /// @return | string[] | Service name strings.
432        methods.add_method("getServices", |lua, this, ()| {
433            let names: Vec<String> = this
434                .locator
435                .borrow()
436                .names()
437                .iter()
438                .map(|s| s.to_string())
439                .collect();
440            let table = lua.create_table()?;
441            for (i, name) in names.iter().enumerate() {
442                table.set(i + 1, name.as_str())?;
443            }
444            Ok(table)
445        });
446        // -- clearAll --
447        /// Remove all registered services and reset the locator.
448        methods.add_method("clearAll", |lua, this, ()| {
449            this.locator.borrow_mut().clear();
450            let drained: Vec<(String, LuaRegistryKey)> =
451                this.services.borrow_mut().drain().collect();
452            for (_, key) in drained {
453                lua.remove_registry_value(key)?;
454            }
455            Ok(())
456        });
457    }

459/// Lua-facing factory pattern for creating typed game objects from registered constructor functions.
460#[derive(Clone)]
461struct LuaFactory {
462    factory: Rc<RefCell<crate::patterns::Factory>>,
463    constructors: Rc<RefCell<HashMap<String, LuaRegistryKey>>>,
464}
465impl LurekType for LuaFactory {
466    const TYPE_NAME: &'static str = "LFactory";
467    const TYPE_HIERARCHY: &'static [&'static str] = &["LFactory", "LObject"];
468}
469impl LuaUserData for LuaFactory {
470    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
471        add_type_methods(methods);
472        // -- register --
473        /// Register a constructor function for a given type name. Future `create()` calls with this type will invoke it.
474        /// @param | typeName | string | The type identifier (e.g. "enemy", "bullet").
475        /// @param | ctor | function | A constructor function that returns a new instance.
476        methods.add_method(
477            "register",
478            |lua, this, (type_name, ctor): (String, LuaFunction)| {
479                this.factory.borrow_mut().register(&type_name);
480                let key = lua.create_registry_value(ctor)?;
481                if let Some(old) = this.constructors.borrow_mut().insert(type_name, key) {
482                    lua.remove_registry_value(old)?;
483                }
484                Ok(())
485            },
486        );
487        // -- create --
488        /// Create a new object by type name, passing additional arguments to the constructor.
489        /// @param | typeName | string | The registered type to instantiate.
490        /// @param | ... | any | Extra arguments forwarded to the constructor.
491        /// @return | table | The created object table.
492        /// @return | nil | When not available.
493        methods.add_method("create", |lua, this, args: LuaMultiValue| {
494            let mut args_iter = args.into_iter();
495            let type_name: String = match args_iter.next() {
496                Some(v) => lua.unpack(v)?,
497                None => return Err(LuaError::external("create requires a type name")),
498            };
499            let extra: Vec<LuaValue> = args_iter.collect();
500            let canonical = this.factory.borrow().resolve(&type_name).to_string();
501            let ctors = this.constructors.borrow();
502            let key = ctors.get(&canonical).ok_or_else(|| {
503                LuaError::external(format!("no constructor registered for type '{canonical}'"))
504            })?;
505            let func: LuaFunction = lua.registry_value(key)?;
506            drop(ctors);
507            func.call::<_, LuaValue>(LuaMultiValue::from_iter(extra))
508        });
509        // -- has --
510        /// Check whether a constructor is registered for the given type name.
511        /// @param | typeName | string | The type name to query.
512        /// @return | boolean | True if a constructor exists for this type.
513        methods.add_method("has", |_lua, this, type_name: String| {
514            Ok(this.factory.borrow().has(&type_name))
515        });
516        // -- alias --
517        /// Create an alias that maps to an existing type name. `create(alias)` will use the canonical constructor.
518        /// @param | alias | string | The alternative name.
519        /// @param | canonical | string | The existing registered type name.
520        methods.add_method(
521            "alias",
522            |_lua, this, (alias, canonical): (String, String)| {
523                this.factory.borrow_mut().add_alias(&alias, &canonical);
524                Ok(())
525            },
526        );
527        // -- getTypes --
528        /// Return an array of all registered type names.
529        /// @return | string[] | Type name strings.
530        methods.add_method("getTypes", |lua, this, ()| {
531            let names: Vec<String> = this
532                .factory
533                .borrow()
534                .type_names()
535                .iter()
536                .map(|s| s.to_string())
537                .collect();
538            let table = lua.create_table()?;
539            for (i, name) in names.iter().enumerate() {
540                table.set(i + 1, name.as_str())?;
541            }
542            Ok(table)
543        });
544        // -- remove --
545        /// Unregister a type and discard its constructor function.
546        /// @param | typeName | string | The type name to remove.
547        methods.add_method("remove", |lua, this, type_name: String| {
548            this.factory.borrow_mut().unregister(&type_name);
549            if let Some(key) = this.constructors.borrow_mut().remove(&type_name) {
550                lua.remove_registry_value(key)?;
551            }
552            Ok(())
553        });

578}
579impl LuaUserData for LuaSimpleState {
580    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
581        add_type_methods(methods);
582        // -- addState --
583        /// Register a named state with optional enter, exit, and update callbacks.
584        /// @param | name | string | Unique state identifier.
585        /// @param | callbacks | table? | Table with optional fields: `enter` (function), `exit` (function), `update` (function receiving dt).
586        methods.add_method(
587            "addState",
588            |lua, this, (name, callbacks): (String, Option<LuaTable>)| {
589                {
590                    let mut enter = this.enter_keys.borrow_mut();
591                    let mut exit = this.exit_keys.borrow_mut();
592                    let mut update = this.update_keys.borrow_mut();
593                    if let Some(k) = enter.remove(&name) {
594                        lua.remove_registry_value(k)?;
595                    }
596                    if let Some(k) = exit.remove(&name) {
597                        lua.remove_registry_value(k)?;
598                    }
599                    if let Some(k) = update.remove(&name) {
600                        lua.remove_registry_value(k)?;
601                    }
602                }
603                this.state.borrow_mut().add(&name);
604                if let Some(tbl) = callbacks {
605                    if let Ok(f) = tbl.get::<_, LuaFunction>("enter") {
606                        this.enter_keys
607                            .borrow_mut()
608                            .insert(name.clone(), lua.create_registry_value(f)?);
609                    }
610                    if let Ok(f) = tbl.get::<_, LuaFunction>("exit") {
611                        this.exit_keys
612                            .borrow_mut()
613                            .insert(name.clone(), lua.create_registry_value(f)?);
614                    }
615                    if let Ok(f) = tbl.get::<_, LuaFunction>("update") {
616                        this.update_keys
617                            .borrow_mut()
618                            .insert(name.clone(), lua.create_registry_value(f)?);
619                    }
620                }
621                Ok(())
622            },
623        );
624        // -- transitionTo --
625        /// Transition to a new state. Calls the current state's `exit` and the target state's `enter` callbacks.
626        /// @param | name | string | The state to transition to. Must be previously added.
627        /// @return | boolean | True if the transition happened, false if the target state does not exist.
628        methods.add_method("transitionTo", |lua, this, name: String| {
629            if !this.state.borrow().has(&name) {
630                return Ok(false);
631            }
632            let current_opt = this.state.borrow().current().map(|s| s.to_string());
633            if let Some(ref current) = current_opt {
634                let func_opt = this
635                    .exit_keys
636                    .borrow()
637                    .get(current.as_str())
638                    .map(|k| lua.registry_value::<LuaFunction>(k));
639                if let Some(Ok(func)) = func_opt {
640                    func.call::<_, ()>(())?;
641                }
642            }
643            this.state.borrow_mut().set_current(&name);
644            let func_opt = this
645                .enter_keys
646                .borrow()
647                .get(&name)
648                .map(|k| lua.registry_value::<LuaFunction>(k));
649            if let Some(Ok(func)) = func_opt {
650                func.call::<_, ()>(())?;
651            }
652            Ok(true)
653        });
654        // -- update --
655        /// Call the current state's update callback with the frame delta time.
656        /// @param | dt | number | Delta time in seconds since last frame.
657        methods.add_method("update", |lua, this, dt: f64| {
658            let current_opt = this.state.borrow().current().map(|s| s.to_string());
659            if let Some(ref current) = current_opt {
660                let func_opt = this
661                    .update_keys
662                    .borrow()
663                    .get(current.as_str())
664                    .map(|k| lua.registry_value::<LuaFunction>(k));
665                if let Some(Ok(func)) = func_opt {
666                    func.call::<_, ()>(dt)?;
667                }
668            }
669            Ok(())
670        });
671        // -- getCurrent --
672        /// Return the name of the currently active state, or nil if no state is set.
673        /// @return | string | Current state name, or nil if no state is set.
674        methods.add_method("getCurrent", |_lua, this, ()| {
675            Ok(this.state.borrow().current().map(|s| s.to_string()))
676        });
677        // -- hasState --
678        /// Check whether a state with the given name is registered.
679        /// @param | name | string | State name to check.
680        /// @return | boolean | True if the state exists.
681        methods.add_method("hasState", |_lua, this, name: String| {
682            Ok(this.state.borrow().has(&name))
683        });
684        // -- getStates --
685        /// Return an array of all registered state names.
686        /// @return | string[] | State name strings.
687        methods.add_method("getStates", |lua, this, ()| {
688            let names: Vec<String> = this
689                .state
690                .borrow()
691                .states()
692                .iter()
693                .map(|s| s.to_string())
694                .collect();
695            let table = lua.create_table()?;
696            for (i, name) in names.iter().enumerate() {
697                table.set(i + 1, name.as_str())?;
698            }
699            Ok(table)
700        });
701        // -- clearAll --
702        /// Remove all states and their callbacks, resetting the state machine.
703        methods.add_method("clearAll", |lua, this, ()| {
704            *this.state.borrow_mut() = crate::patterns::SimpleState::new();
705            for (_, key) in this.enter_keys.borrow_mut().drain() {
706                lua.remove_registry_value(key)?;
707            }
708            for (_, key) in this.exit_keys.borrow_mut().drain() {
709                lua.remove_registry_value(key)?;
710            }
711            for (_, key) in this.update_keys.borrow_mut().drain() {
712                lua.remove_registry_value(key)?;
713            }
714            Ok(())
715        });
716    }

718/// Lua-facing shared key-value blackboard supporting bool/number/string values with watchers for reactive game logic.
719#[derive(Clone)]
720struct LuaBlackboard {
721    board: Rc<RefCell<crate::patterns::Blackboard>>,
722    watchers: Rc<RefCell<HashMap<u64, LuaRegistryKey>>>,
723    watcher_keys: Rc<RefCell<HashMap<u64, String>>>,

725}
726impl LurekType for LuaBlackboard {
727    const TYPE_NAME: &'static str = "LBlackboard";
728    const TYPE_HIERARCHY: &'static [&'static str] = &["LBlackboard", "LObject"];
729}
730impl LuaUserData for LuaBlackboard {
731    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
732        add_type_methods(methods);
733        // -- set --
734        /// Set a key to a value (boolean, number, string, or nil to clear). Notifies registered watchers if value changed.
735        /// @param | key | string | The key name.
736        /// @param | value | boolean|number|string|nil | The value to store. Pass nil to clear the key.
737        methods.add_method("set", |lua, this, (key, value): (String, LuaValue)| {
738            let prev_rev = this.board.borrow().revision;
739            match &value {
740                LuaValue::Boolean(b) => this.board.borrow_mut().set_bool(&key, *b),
741                LuaValue::Integer(n) => this.board.borrow_mut().set_number(&key, *n as f64),
742                LuaValue::Number(n) => this.board.borrow_mut().set_number(&key, *n),
743                LuaValue::String(s) => this
744                    .board
745                    .borrow_mut()
746                    .set_text(&key, s.to_str()?.to_string()),
747                LuaValue::Nil => this.board.borrow_mut().clear(),
748                _ => {
749                    return Err(LuaError::external(
750                        "Blackboard only supports bool/number/string/nil values",
751                    ))
752                }
753            }
754            let new_rev = this.board.borrow().revision;
755            if new_rev != prev_rev {
756                let sub_ids: Vec<u64> = {
757                    let wk = this.watcher_keys.borrow();
758                    wk.iter()
759                        .filter(|(_, k)| k == &&key || k.as_str() == "*")
760                        .map(|(id, _)| *id)
761                        .collect()
762                };
763                for id in sub_ids {
764                    let cb = this.watchers.borrow();
765                    if let Some(rk) = cb.get(&id) {
766                        let func: LuaFunction = lua.registry_value(rk)?;
767                        drop(cb);
768                        func.call::<_, ()>((key.clone(), value.clone()))?;
769                    }
770                }
771            }
772            Ok(())
773        });
774        // -- get --
775        /// Retrieve the value stored under a key. Returns nil if the key does not exist.
776        /// @param | key | string | The key name to look up.
777        /// @return | boolean|number|string|nil | The stored value.
778        methods.add_method("get", |lua, this, key: String| {
779            match this.board.borrow().get(&key) {
780                Some(crate::patterns::BlackboardValue::Bool(b)) => Ok(LuaValue::Boolean(*b)),
781                Some(crate::patterns::BlackboardValue::Number(n)) => Ok(LuaValue::Number(*n)),
782                Some(crate::patterns::BlackboardValue::Text(s)) => {
783                    Ok(LuaValue::String(lua.create_string(s)?))
784                }
785                Some(crate::patterns::BlackboardValue::Nil) | None => Ok(LuaValue::Nil),
786            }
787        });
788        // -- has --
789        /// Check whether a key exists on the blackboard.
790        /// @param | key | string | The key to check.
791        /// @return | boolean | True if the key has a stored value.
792        methods.add_method("has", |_, this, key: String| {
793            Ok(this.board.borrow().has(&key))
794        });
795        // -- clear --
796        /// Remove a single key from the blackboard.
797        /// @param | key | string | The key to remove.
798        methods.add_method("clear", |_, this, key: String| {
799            this.board.borrow_mut().remove(&key);
800            Ok(())
801        });
802        // -- keys --
803        /// Return an array of all keys currently stored on the blackboard.
804        /// @return | string[] | Key name strings.
805        methods.add_method("keys", |lua, this, ()| {
806            let keys = this.board.borrow().keys();
807            let tbl = lua.create_table()?;
808            for (i, k) in keys.iter().enumerate() {
809                tbl.set(i + 1, k.as_str())?;
810            }
811            Ok(tbl)
812        });
813        // -- watch --
814        /// Register a watcher callback that fires whenever the specified key changes. Use `"*"` to watch all keys.
815        /// @param | key | string | The key to watch, or `"*"` for all changes.
816        /// @param | callback | function | Called with (key, newValue) when a change occurs.
817        /// @return | number | A watcher ID for later removal with `unwatch()`.
818        methods.add_method(
819            "watch",
820            |lua, this, (key, callback): (String, LuaFunction)| {
821                let id = {
822                    let mut nid = this.next_watcher_id.borrow_mut();
823                    let id = *nid;
824                    *nid += 1;
825                    id
826                };
827                let rk = lua.create_registry_value(callback)?;
828                this.watchers.borrow_mut().insert(id, rk);
829                this.watcher_keys.borrow_mut().insert(id, key);
830                Ok(id)
831            },
832        );
833        // -- unwatch --
834        /// Remove a previously registered watcher by its ID.
835        /// @param | id | integer | The watcher ID returned by `watch()`.
836        methods.add_method("unwatch", |lua, this, id: u64| {
837            if let Some(rk) = this.watchers.borrow_mut().remove(&id) {
838                lua.remove_registry_value(rk)?;
839            }
840            this.watcher_keys.borrow_mut().remove(&id);
841            Ok(())
842        });
843        // -- getRevision --
844        /// Return the current revision counter. Increments on every value change.
845        /// @return | number | The revision number.
846        methods.add_method("getRevision", |_, this, ()| {
847            Ok(this.board.borrow().revision)
848        });
849        // -- snapshot --
850        /// Return a table containing all current key-value pairs as a snapshot. Useful for serialization or debug display.
851        /// @return | table | A table mapping key strings to their stored values.
852        methods.add_method("snapshot", |lua, this, ()| {
853            let tbl = lua.create_table()?;
854            for (k, v) in this.board.borrow().snapshot() {
855                match v {
856                    crate::patterns::BlackboardValue::Bool(b) => tbl.set(k, *b)?,
857                    crate::patterns::BlackboardValue::Number(n) => tbl.set(k, *n)?,
858                    crate::patterns::BlackboardValue::Text(s) => tbl.set(k, s.as_str())?,
859                    crate::patterns::BlackboardValue::Nil => {}
860                }
861            }
862            Ok(tbl)
863        });
864        // -- clearAll --
865        /// Remove all keys and values from the blackboard.
866        methods.add_method("clearAll", |_, this, ()| {
867            this.board.borrow_mut().clear_all();
868            Ok(())
869        });
870    }

1095        /// @param | priority | integer | The priority value (higher = dequeued sooner).
1096        /// @param | value | any | The payload value to store.
1097        /// @param | label | string? | Optional human-readable label for debugging.
1098        /// @return | number | The internal ID of the enqueued item.
1099        methods.add_method(
1100            "push",
1101            |lua, this, (priority, value, label): (i64, LuaValue, Option<String>)| {
1102                let id = this
1103                    .queue
1104                    .borrow_mut()
1105                    .push(priority, label.as_deref().unwrap_or(""));
1106                let rk = lua.create_registry_value(value)?;
1107                this.payloads.borrow_mut().insert(id, rk);
1108                Ok(id)
1109            },
1110        );
1111        // -- pop --
1112        /// Remove and return the highest-priority item. Returns nil if the queue is empty.
1113        /// @return | table | The item table.
1114        /// @return | nil | When not available.
1115        methods.add_method("pop", |lua, this, ()| match this.queue.borrow_mut().pop() {
1116            Some((id, _priority)) => {
1117                if let Some(rk) = this.payloads.borrow_mut().remove(&id) {
1118                    let val: LuaValue = lua.registry_value(&rk)?;
1119                    lua.remove_registry_value(rk)?;
1120                    Ok(val)
1121                } else {
1122                    Ok(LuaValue::Nil)
1123                }
1124            }
1125            None => Ok(LuaValue::Nil),
1126        });
1127        // -- peek --
1128        /// Return the highest-priority item without removing it. Returns nil if empty.
1129        /// @return | table | The item table.
1130        /// @return | nil | When not available.
1131        methods.add_method("peek", |lua, this, ()| match this.queue.borrow().peek() {
1132            Some(item) => {
1133                let id = item.id;
1134                let payloads = this.payloads.borrow();
1135                if let Some(rk) = payloads.get(&id) {
1136                    Ok(lua.registry_value::<LuaValue>(rk)?)
1137                } else {
1138                    Ok(LuaValue::Nil)
1139                }
1140            }
1141            None => Ok(LuaValue::Nil),
1142        });
1143        // -- len --
1144        /// Return the number of items currently in the queue.
1145        /// @return | number | Item count.
1146        methods.add_method("len", |_, this, ()| Ok(this.queue.borrow().len()));
1147        // -- isEmpty --
1148        /// Check whether the queue contains no items.
1149        /// @return | boolean | True if the queue is empty.
1150        methods.add_method("isEmpty", |_, this, ()| Ok(this.queue.borrow().is_empty()));
1151        // -- clearAll --
1152        /// Remove all items from the queue. This method is available to Lua scripts.
1153        methods.add_method("clearAll", |lua, this, ()| {
1154            this.queue.borrow_mut().clear();
1155            let drained: Vec<(u64, LuaRegistryKey)> = this.payloads.borrow_mut().drain().collect();
1156            for (_, rk) in drained {
1157                lua.remove_registry_value(rk)?;
1158            }
1159            Ok(())
1160        });
1161    }
1162}

1279/// Lua-facing batching funnel that collects events over a time window and flushes them together.
1280#[derive(Clone)]
1281struct LuaFunnel {
1282    funnel: Rc<RefCell<crate::patterns::Funnel>>,
1283    on_flush: Rc<RefCell<Option<LuaRegistryKey>>>,
1284}
1285impl LurekType for LuaFunnel {
1286    const TYPE_NAME: &'static str = "LFunnel";
1287    const TYPE_HIERARCHY: &'static [&'static str] = &["LFunnel", "LObject"];
1288}
1289impl LuaUserData for LuaFunnel {
1290    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
1291        add_type_methods(methods);
1292        // -- onFlush --
1293        /// Set the callback invoked when the funnel flushes. Receives an array of {tag, value} entries.
1294        /// @param | f | function | Callback receiving a table array of batched entries.
1295        methods.add_method("onFlush", |lua, this, f: LuaFunction| {
1296            let rk = lua.create_registry_value(f)?;
1297            if let Some(old) = this.on_flush.borrow_mut().replace(rk) {
1298                lua.remove_registry_value(old)?;
1299            }
1300            Ok(())
1301        });
1302        // -- push --
1303        /// Push a tagged event into the funnel. May trigger an immediate flush if the max entry count is reached.
1304        /// @param | tag | string | A category label for the event.
1305        /// @param | value | number? | Optional numeric value (default 0).
1306        methods.add_method("push", |lua, this, (tag, value): (String, Option<f64>)| {
1307            let (_, should_flush) = this.funnel.borrow_mut().push(&tag, value.unwrap_or(0.0));
1308            if should_flush {
1309                Self::do_flush(lua, this)?;
1310            }
1311            Ok(())
1312        });
1313        // -- update --
1314        /// Advance the funnel's time window. Flushes and invokes the callback if the window elapsed.
1315        /// @param | dt | number | Delta time in seconds.
1316        /// @return | boolean | True if a flush occurred this frame.
1317        methods.add_method("update", |lua, this, dt: f64| {
1318            let should_flush = this.funnel.borrow_mut().update(dt);
1319            if should_flush {
1320                Self::do_flush(lua, this)?;
1321                return Ok(true);
1322            }
1323            Ok(false)
1324        });
1325        // -- flush --
1326        /// Force an immediate flush of all pending entries, invoking the callback.
1327        methods.add_method("flush", |lua, this, ()| Self::do_flush(lua, this));
1328        // -- discard --
1329        /// Discard all pending entries without flushing or calling the callback.
1330        methods.add_method("discard", |_, this, ()| {
1331            this.funnel.borrow_mut().discard();
1332            Ok(())
1333        });
1334        // -- pendingCount --
1335        /// Return the number of entries waiting to be flushed.
1336        /// @return | number | Pending entry count.
1337        methods.add_method("pendingCount", |_, this, ()| {
1338            Ok(this.funnel.borrow().pending_count())
1339        });
1340        // -- getFlushCount --
1341        /// Return the total number of times this funnel has flushed since creation.
1342        /// @return | number | Total flush count.
1343        methods.add_method("getFlushCount", |_, this, ()| {
1344            Ok(this.funnel.borrow().flush_count)
1345        });
1346    }
1347}
1348impl LuaFunnel {
1349    fn do_flush(lua: &Lua, this: &LuaFunnel) -> LuaResult<()> {
1350        let entries = this.funnel.borrow_mut().flush();
1351        if entries.is_empty() {
1352            return Ok(());
1353        }
1354        if let Some(rk) = &*this.on_flush.borrow() {
1355            let func: LuaFunction = lua.registry_value(rk)?;
1356            let tbl = lua.create_table()?;
1357            for (i, e) in entries.iter().enumerate() {
1358                let et = lua.create_table()?;
1359                /// The 'tag' field value exposed to Lua scripts.
1360                et.set("tag", e.tag.as_str())?;
1361                /// Performs the 'value' operation.
1362                et.set("value", e.value)?;
1363                tbl.set(i + 1, et)?;
1364            }
1365            func.call::<_, ()>(tbl)?;
1366        }
1367        Ok(())
1368    }

1592            }
1593            Ok(())
1594        });

1629        });

1642                .get(&id)
1643                .ok_or_else(|| LuaError::runtime("Strategy function missing"))?;

1664        /// @return | boolean | True if the strategy was found and removed.
1665        methods.add_method("remove", |lua, this, name: String| {
1666            let id = {

1668                if st.has(&name) {
1669                    st.get_current_id()
1670                } else {
1671                    return Ok(false);
1672                }
1673            };
1674            let removed = this.strategy.borrow_mut().remove(&name);
1675            if removed {
1676                if let Some(id) = id {
1677                    if let Some(key) = this.callbacks.borrow_mut().remove(&id) {
1678                        lua.remove_registry_value(key)?;
1679                    }
1680                }
1681            }
1682            Ok(removed)
1683        });
1684        // -- names --
1685        /// Return an array of all registered strategy names.
1686        /// @return | string[] | Strategy name strings.
1687        methods.add_method("names", |_, this, ()| Ok(this.strategy.borrow().names()));
1688        // -- clear --
1689        /// Remove all strategies and reset the selection.
1690        methods.add_method("clear", |lua, this, ()| {
1691            this.strategy.borrow_mut().clear();
1692            let mut cbs = this.callbacks.borrow_mut();
1693            for (_, key) in cbs.drain() {
1694                lua.remove_registry_value(key)?;
1695            }
1696            Ok(())
1697        });
1698    }

2754impl LuaBehaviorTree {
2755    fn tick_node(
2756        lua: &Lua,
2757        this: &LuaBehaviorTree,
2758        id: crate::patterns::NodeId,
2759    ) -> LuaResult<crate::patterns::BtStatus> {
2760        use crate::patterns::{BtStatus, NodeKind};
2761        let (kind, children) = {
2762            let t = this.tree.borrow();
2763            let node = match t.get_node(id) {
2764                Some(n) => n,
2765                None => return Ok(BtStatus::Failure),
2766            };
2767            (node.kind.clone(), node.children.clone())
2768        };
2769        match kind {
2770            NodeKind::Sequence => {
2771                for child in children {
2772                    match Self::tick_node(lua, this, child)? {
2773                        BtStatus::Success => continue,
2774                        other => return Ok(other),
2775                    }
2776                }
2777                Ok(BtStatus::Success)
2778            }
2779            NodeKind::Selector => {
2780                for child in children {
2781                    match Self::tick_node(lua, this, child)? {
2782                        BtStatus::Failure => continue,
2783                        other => return Ok(other),
2784                    }
2785                }
2786                Ok(BtStatus::Failure)
2787            }
2788            NodeKind::Parallel { min_success } => {
2789                let mut successes = 0usize;
2790                let mut any_running = false;
2791                for child in &children {
2792                    match Self::tick_node(lua, this, *child)? {
2793                        BtStatus::Success => successes += 1,
2794                        BtStatus::Running => any_running = true,
2795                        BtStatus::Failure => {}
2796                    }
2797                }
2798                if successes >= min_success {
2799                    Ok(BtStatus::Success)
2800                } else if any_running {
2801                    Ok(BtStatus::Running)
2802                } else {
2803                    Ok(BtStatus::Failure)
2804                }
2805            }
2806            NodeKind::Inverter => {
2807                let child = match children.first() {
2808                    Some(&c) => c,
2809                    None => return Ok(BtStatus::Failure),
2810                };
2811                Ok(match Self::tick_node(lua, this, child)? {
2812                    BtStatus::Success => BtStatus::Failure,
2813                    BtStatus::Failure => BtStatus::Success,
2814                    BtStatus::Running => BtStatus::Running,
2815                })
2816            }
2817            NodeKind::Repeat { count } => {
2818                let child = match children.first() {
2819                    Some(&c) => c,
2820                    None => return Ok(BtStatus::Failure),
2821                };
2822                let iterations = count.max(1);
2823                for _ in 0..iterations {
2824                    match Self::tick_node(lua, this, child)? {
2825                        BtStatus::Failure => return Ok(BtStatus::Failure),
2826                        BtStatus::Running => return Ok(BtStatus::Running),
2827                        BtStatus::Success => {}
2828                    }
2829                }
2830                Ok(BtStatus::Success)
2831            }
2832            NodeKind::Leaf { name } => {
2833                let fns = this.leaf_fns.borrow();
2834                match fns.get(&name) {
2835                    None => Ok(BtStatus::Failure),
2836                    Some(rk) => {
2837                        let func: LuaFunction = lua.registry_value(rk)?;
2838                        drop(fns);
2839                        let result: String = func.call(())?;
2840                        Ok(match result.as_str() {
2841                            "success" => BtStatus::Success,
2842                            "running" => BtStatus::Running,
2843                            _ => BtStatus::Failure,
2844                        })
2845                    }
2846                }
2847            }
2848        }
2849    }
