21impl LuaStep {
22    /// Wraps an existing pipeline step in a Lua-visible userdata handle.
23    pub fn new(step: PipelineStep) -> Self {
24        Self {
25            inner: Rc::new(RefCell::new(step)),
26            callback_key: Rc::new(RefCell::new(None)),
27            condition_key: Rc::new(RefCell::new(None)),
28            on_error_key: Rc::new(RefCell::new(None)),
29        }

31    /// Execute sync. This function is part of the public API.
32    pub(crate) fn execute_sync<'lua>(
33        &self,
34        lua: &'lua Lua,
35        ctx: &LuaTable<'lua>,
36        abort_on_fail: bool,
37    ) -> LuaResult<bool> {
38        let name = self.inner.borrow().name.clone();
39        let retry_count = self.inner.borrow().retry_count;
40        if let Some(key) = self.condition_key.borrow().as_ref() {
41            let cond_fn: LuaFunction = lua.registry_value(key)?;
42            let should_run: bool = cond_fn.call(ctx.clone())?;
43            if !should_run {
44                self.inner.borrow_mut().status = StepStatus::Skipped;
45                return Ok(true);
46            }
47        }
48        let cb_key_opt = self.callback_key.borrow();
49        let cb_key = cb_key_opt.as_ref().ok_or_else(|| {
50            LuaError::runtime(format!("step '{}' has no callback registered", name))
51        })?;
52        let cb: LuaFunction = lua.registry_value(cb_key)?;
53        drop(cb_key_opt);
54        let max_attempts = retry_count + 1;
55        let mut last_error: Option<LuaError> = None;
56        let started = Instant::now();
57        for attempt in 0..max_attempts {
58            self.inner.borrow_mut().attempt = attempt + 1;
59            self.inner.borrow_mut().status = StepStatus::Running;
60            match cb.call::<_, LuaValue<'_>>(ctx.clone()) {
61                Ok(result) => {
62                    let results: LuaTable = match ctx.get("results") {
63                        Ok(t) => t,
64                        Err(_) => lua.create_table()?,
65                    };
66                    results.set(name.clone(), result)?;
67                    /// Performs the 'results' operation.
68                    ctx.set("results", results)?;
69                    self.inner.borrow_mut().status = StepStatus::Completed;
70                    self.inner.borrow_mut().duration = started.elapsed().as_secs_f32();
71                    self.inner.borrow_mut().error_msg = None;
72                    return Ok(true);
73                }
74                Err(e) => {
75                    last_error = Some(e);
76                    if attempt + 1 < max_attempts {
77                        continue;
78                    }
79                }
80            }
81        }
82        let err_msg = last_error
83            .as_ref()
84            .map(|e| e.to_string())
85            .unwrap_or_default();
86        {
87            let mut inner = self.inner.borrow_mut();
88            inner.status = StepStatus::Failed;
89            inner.duration = started.elapsed().as_secs_f32();
90            inner.error_msg = Some(err_msg.clone());
91        }
92        if let Some(key) = self.on_error_key.borrow().as_ref() {
93            let err_fn: LuaFunction = lua.registry_value(key)?;
94            if let Err(e) = err_fn.call::<_, LuaValue<'_>>((name.clone(), err_msg)) {
95                log_msg!(warn, LA02_PIPELINE_CALLBACK_FAIL, "on_error: {e}");
96            }
97        }
98        if abort_on_fail {
99            Ok(false)
100        } else {
101            Ok(true)
102        }
103    }
104    fn is_async_enabled(&self) -> bool {
105        self.inner
106            .borrow()
107            .metadata
108            .get("__pipeline_async")
109            .map(|v| v == "true")
110            .unwrap_or(false)
111    }
112    fn set_async_enabled(&self, enabled: bool) {
113        self.inner
114            .borrow_mut()
115            .metadata
116            .insert("__pipeline_async".to_string(), enabled.to_string());
117    }

406/// Pipeline result to lua. This function is part of the public API.
407pub(crate) fn pipeline_result_to_lua<'lua>(
408    lua: &'lua Lua,
409    result: &crate::pipeline::PipelineResult,
410) -> LuaResult<LuaTable<'lua>> {
411    let t = lua.create_table()?;
412    let completed = lua.create_table()?;
413    for (i, name) in result.completed.iter().enumerate() {
414        completed.set(i + 1, name.clone())?;
415    }
416    let failed = lua.create_table()?;
417    for (i, name) in result.failed.iter().enumerate() {
418        failed.set(i + 1, name.clone())?;
419    }
420    let skipped = lua.create_table()?;
421    for (i, name) in result.skipped.iter().enumerate() {
422        skipped.set(i + 1, name.clone())?;
423    }
424    let cancelled = lua.create_table()?;
425    for (i, name) in result.cancelled.iter().enumerate() {
426        cancelled.set(i + 1, name.clone())?;
427    }
428    let errors = lua.create_table()?;
429    for (i, (name, msg)) in result.errors.iter().enumerate() {
430        let entry = lua.create_table()?;
431        entry.set(1, name.clone())?;
432        entry.set(2, msg.clone())?;
433        errors.set(i + 1, entry)?;
434    }
435    /// Performs the 'success' operation.
436    t.set("success", result.is_success())?;
437    /// Performs the 'completed' operation.
438    t.set("completed", completed)?;
439    /// Performs the 'failed' operation.
440    t.set("failed", failed)?;
441    /// Performs the 'skipped' operation.
442    t.set("skipped", skipped)?;
443    /// Performs the 'cancelled' operation.
444    t.set("cancelled", cancelled)?;
445    /// Performs the 'totalDuration' operation.
446    t.set("totalDuration", result.total_duration)?;
447    /// Performs the 'errors' operation.
448    t.set("errors", errors)?;
449    Ok(t)
450}
451/// Cancel remaining steps. This function is part of the public API.
452pub(crate) fn cancel_remaining_steps(wrappers: &HashMap<String, LuaStep>, order: &[String]) {
453    for name in order {
454        if let Some(w) = wrappers.get(name) {
455            if w.inner.borrow().status == StepStatus::Pending {
456                w.inner.borrow_mut().status = StepStatus::Cancelled;
457            }
458        }
459    }
460}
461/// Fire step callbacks. This function is part of the public API.
462pub(crate) fn fire_step_callbacks<'lua>(
463    lua: &'lua Lua,
464    this: &LuaPipeline,
465    step_name: &str,
466    ctx: &LuaTable<'lua>,
467    wrapper: &LuaStep,
468) -> LuaResult<()> {
469    let step_status = wrapper.inner.borrow().status.clone();
470    if step_status == StepStatus::Completed {
471        if let Some(key) = this.on_step_complete_key.borrow().as_ref() {
472            let f: LuaFunction = lua.registry_value(key)?;
473            if let Err(e) = f.call::<_, LuaValue<'_>>((step_name.to_string(), ctx.clone())) {
474                log_msg!(warn, LA02_PIPELINE_CALLBACK_FAIL, "on_step_complete: {e}");
475            }
476        }
477    } else if step_status == StepStatus::Failed {
478        if let Some(key) = this.on_step_error_key.borrow().as_ref() {
479            let f: LuaFunction = lua.registry_value(key)?;
480            let err = wrapper.inner.borrow().error_msg.clone().unwrap_or_default();
481            if let Err(e) = f.call::<_, LuaValue<'_>>((step_name.to_string(), err)) {
482                log_msg!(warn, LA02_PIPELINE_CALLBACK_FAIL, "on_step_error: {e}");
483            }
484        }
485    }
486    if let Some(key) = this.on_progress_key.borrow().as_ref() {
487        let status_str = format!("{:?}", step_status).to_lowercase();
488        let f: LuaFunction = lua.registry_value(key)?;
489        if let Err(e) = f.call::<_, LuaValue<'_>>((step_name.to_string(), status_str)) {
490            log_msg!(warn, LA02_PIPELINE_CALLBACK_FAIL, "on_progress: {e}");
491        }
492    }
493    let detail: LuaValue = if step_status == StepStatus::Failed {
494        LuaValue::String(
495            lua.create_string(wrapper.inner.borrow().error_msg.clone().unwrap_or_default())?,
496        )
497    } else {
498        LuaValue::Nil
499    };
500    fire_pipeline_event(
501        lua,
502        this,
503        "step_finished",
504        step_name,
505        step_status.as_str(),
506        detail,
507    )?;
508    Ok(())
509}
510/// Dispatches one pipeline lifecycle event to the optional Lua `on_event` callback.
511fn fire_pipeline_event<'lua>(
512    lua: &'lua Lua,
513    this: &LuaPipeline,
514    event_name: &str,
515    step_name: &str,
516    status: &str,
517    detail: LuaValue<'lua>,
518) -> LuaResult<()> {
519    if let Some(key) = this.on_event_key.borrow().as_ref() {
520        let f: LuaFunction = lua.registry_value(key)?;
521        if let Err(e) = f.call::<_, LuaValue<'_>>((
522            event_name.to_string(),
523            step_name.to_string(),
524            status.to_string(),
525            detail,
526        )) {
527            log_msg!(warn, LA02_PIPELINE_CALLBACK_FAIL, "on_event: {e}");
528        }
529    }
530    Ok(())
531}
532/// Starts or resumes an async coroutine-backed pipeline step and reports whether it completed.
533fn execute_async_coroutine_step<'lua>(
534    lua: &'lua Lua,
535    this: &LuaPipeline,
536    step_name: &str,
537    wrapper: &LuaStep,
538    ctx: &LuaTable<'lua>,
539) -> LuaResult<bool> {
540    let mut first_resume = false;
541    let mut thread_key = this.running_threads.borrow_mut().remove(step_name);
542    if thread_key.is_none() {
543        if let Some(key) = wrapper.condition_key.borrow().as_ref() {
544            let cond_fn: LuaFunction = lua.registry_value(key)?;
545            let should_run: bool = cond_fn.call(ctx.clone())?;
546            if !should_run {
547                wrapper.inner.borrow_mut().status = StepStatus::Skipped;
548                return Ok(true);
549            }
550        }
551        let name = wrapper.inner.borrow().name.clone();
552        let cb_key_opt = wrapper.callback_key.borrow();
553        let cb_key = cb_key_opt.as_ref().ok_or_else(|| {
554            LuaError::runtime(format!("step '{}' has no callback registered", name))
555        })?;
556        let cb: LuaFunction = lua.registry_value(cb_key)?;
557        drop(cb_key_opt);
558        let thread = lua.create_thread(cb)?;
559        let key = lua.create_registry_value(thread)?;
560        thread_key = Some(key);
561        first_resume = true;
562        let mut inner = wrapper.inner.borrow_mut();
563        inner.attempt = inner.attempt.saturating_add(1);
564        inner.status = StepStatus::Running;
565        inner.error_msg = None;
566        drop(inner);
567        this.started_at
568            .borrow_mut()
569            .insert(step_name.to_string(), Instant::now());
570    }
571    let Some(reg_key) = thread_key else {
572        return Ok(false);
573    };
574    let thread_val: LuaValue = lua.registry_value(&reg_key)?;
575    let thread = match thread_val {
576        LuaValue::Thread(t) => t,
577        _ => {
578            this.running_threads.borrow_mut().remove(step_name);
579            lua.remove_registry_value(reg_key)?;
580            return Err(LuaError::runtime(format!(
581                "step '{}' async state is invalid",
582                step_name
583            )));
584        }
585    };
586    let resumed = if first_resume {
587        thread.resume::<_, LuaValue<'_>>(ctx.clone())
588    } else {
589        thread.resume::<_, LuaValue<'_>>(())
590    };
591    match resumed {
592        Ok(result) => {
593            let coroutine_tbl: LuaTable = lua.globals().get("coroutine")?;
594            let status_fn: LuaFunction = coroutine_tbl.get("status")?;
595            let coroutine_status: String = status_fn.call(thread.clone())?;
596            if coroutine_status != "dead" {
597                wrapper.inner.borrow_mut().status = StepStatus::Running;
598                this.running_threads
599                    .borrow_mut()
600                    .insert(step_name.to_string(), reg_key);
601                return Ok(false);
602            }
603            let results: LuaTable = match ctx.get("results") {
604                Ok(t) => t,
605                Err(_) => lua.create_table()?,
606            };
607            results.set(step_name.to_string(), result)?;
608            /// Performs the 'results' operation.
609            ctx.set("results", results)?;
610            let mut inner = wrapper.inner.borrow_mut();
611            inner.status = StepStatus::Completed;
612            if let Some(started) = this.started_at.borrow_mut().remove(step_name) {
613                inner.duration = started.elapsed().as_secs_f32();
614            }
615            drop(inner);
616            this.running_threads.borrow_mut().remove(step_name);
617            lua.remove_registry_value(reg_key)?;
618            Ok(true)
619        }
620        Err(e) => {
621            this.running_threads.borrow_mut().remove(step_name);
622            lua.remove_registry_value(reg_key)?;
623            let mut inner = wrapper.inner.borrow_mut();
624            let max_attempts = inner.retry_count.saturating_add(1);
625            if inner.attempt < max_attempts {
626                inner.status = StepStatus::Waiting;
627                return Ok(false);
628            }
629            inner.status = StepStatus::Failed;
630            inner.error_msg = Some(e.to_string());
631            if let Some(started) = this.started_at.borrow_mut().remove(step_name) {
632                inner.duration = started.elapsed().as_secs_f32();
633            }
634            drop(inner);
635            if let Some(key) = wrapper.on_error_key.borrow().as_ref() {
636                let err_fn: LuaFunction = lua.registry_value(key)?;
637                let err = wrapper.inner.borrow().error_msg.clone().unwrap_or_default();
638                if let Err(cb_err) = err_fn.call::<_, LuaValue<'_>>((step_name.to_string(), err)) {
639                    log_msg!(warn, LA02_PIPELINE_CALLBACK_FAIL, "on_error: {cb_err}");
640                }
641            }
642            Ok(true)
643        }
644    }
645}
646/// Finalize pipeline result. This function is part of the public API.
647pub(crate) fn finalize_pipeline_result<'lua>(
648    lua: &'lua Lua,
649    this: &LuaPipeline,
650    elapsed: f32,
651) -> LuaResult<LuaTable<'lua>> {
652    let step_statuses: HashMap<String, (StepStatus, Option<String>)> = this
653        .step_wrappers
654        .borrow()
655        .iter()
656        .map(|(k, v)| {
657            let inner = v.inner.borrow();
658            (k.clone(), (inner.status.clone(), inner.error_msg.clone()))
659        })
660        .collect();
661    let result = this.inner.borrow().collect_result(&step_statuses, elapsed);
662    let result_tbl = pipeline_result_to_lua(lua, &result)?;
663    if let Some(key) = this.on_complete_key.borrow().as_ref() {
664        let f: LuaFunction = lua.registry_value(key)?;
665        if let Err(e) = f.call::<_, LuaValue<'_>>(result_tbl.clone()) {
666            log_msg!(warn, LA02_PIPELINE_CALLBACK_FAIL, "on_complete: {e}");
667        }
668    }
669    Ok(result_tbl)
670}
671impl LuaUserData for LuaPipeline {
672    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
673        // -- addStep --
674        /// Adds an existing step object to this pipeline. The step will be scheduled according to its declared dependencies.
675        /// @param | step | LPipelineStep | The step to add.
676        /// @return | LPipeline | Returns self for method chaining.
677        methods.add_method("addStep", |_, this, step_ud: LuaAnyUserData| {
678            let step = step_ud.borrow::<LuaStep>()?.clone();
679            let step_def = step.inner.borrow().clone();
680            this.inner
681                .borrow_mut()
682                .add_step(step_def)
683                .map_err(LuaError::runtime)?;
684            let name = step.inner.borrow().name.clone();
685            this.step_wrappers.borrow_mut().insert(name, step);
686            Ok(this.clone())
687        });
688
689        // -- removeStep --
690        /// Removes a step from the pipeline by name. Any other steps that depend on it may fail or be skipped.
691        /// @param | name | string | Name of the step to remove.
692        methods.add_method("removeStep", |_, this, name: String| {
693            this.inner.borrow_mut().remove_step(&name);
694            this.step_wrappers.borrow_mut().remove(&name);
695            Ok(())
696        });
697
698        // -- getStep --
699        /// Retrieves a step object by name, or nil if no step with that name exists in this pipeline.
700        /// @param | name | string | Name of the step to find.
701        /// @return | LPipelineStep | The step object, or nil if no step with that name exists.
702        methods.add_method("getStep", |_, this, name: String| {
703            Ok(this.step_wrappers.borrow().get(&name).cloned())
704        });
705
706        // -- getSteps --
707        /// Returns a table containing all step objects currently in this pipeline.
708        /// @return | LPipelineStep[] | LPipelineStep objects.
709        methods.add_method("getSteps", |lua, this, ()| {
710            let t = lua.create_table()?;
711            for (i, wrapper) in this.step_wrappers.borrow().values().enumerate() {
712                t.set(i + 1, wrapper.clone())?;
713            }
714            Ok(t)
715        });
716
717        // -- getStepCount --
718        /// Returns the total number of steps in this pipeline.
719        /// @return | integer | Step count.
720        methods.add_method("getStepCount", |_, this, ()| {
721            Ok(this.inner.borrow().get_step_count())
722        });
723
724        // -- getStepsByTag --
725        /// Returns all steps that have the specified tag assigned.
726        /// @param | tag | string | The tag to filter by.
727        /// @return | LPipelineStep[] | Matching LPipelineStep objects.
728        methods.add_method("getStepsByTag", |lua, this, tag: String| {
729            let t = lua.create_table()?;
730            let mut i = 1usize;
731            for wrapper in this.step_wrappers.borrow().values() {
732                if wrapper.inner.borrow().tag.as_deref() == Some(&tag) {
733                    t.set(i, wrapper.clone())?;
734                    i += 1;
735                }
736            }
737            Ok(t)
738        });
739
740        // -- clear --
741        /// Removes all steps from the pipeline, resetting it to an empty state.
742        methods.add_method("clear", |_, this, ()| {
743            this.inner.borrow_mut().clear();
744            this.step_wrappers.borrow_mut().clear();
745            Ok(())
746        });
747
748        // -- validate --
749        /// Validates the pipeline structure, checking for missing dependencies and circular references.
750        /// @return | boolean | True if the pipeline is valid.
751        /// @return | string[] | Error message strings (empty if valid).
752        methods.add_method("validate", |lua, this, ()| {
753            let (ok, errs) = this.inner.borrow().validate();
754            let t = lua.create_table()?;
755            for (i, e) in errs.iter().enumerate() {
756                t.set(i + 1, e.clone())?;
757            }
758            Ok((ok, t))
759        });
760
761        // -- getExecutionOrder --
762        /// Computes the topologically sorted execution order of all steps, respecting dependencies.
763        /// @return | string[] | Step names in execution order, or nil on error.
764        /// @return | string | Error message if ordering failed (e.g., circular dependency), or nil on success.
765        methods.add_method("getExecutionOrder", |lua, this, ()| {
766            match this.inner.borrow().get_execution_order() {
767                Ok(order) => {
768                    let t = lua.create_table()?;
769                    for (i, n) in order.iter().enumerate() {
770                        t.set(i + 1, n.clone())?;
771                    }
772                    Ok((Some(t), None::<String>))
773                }
774                Err(e) => Ok((None, Some(e))),
775            }
776        });
777
778        // -- getParallelGroups --
779        /// Groups steps into parallel execution tiers. Steps within the same group have no mutual dependencies and can run concurrently.
780        /// @return | string[] | Array of arrays, each inner array is a group of step names. Nil on error.
781        /// @return | string | Error message if grouping failed, or nil on success.
782        methods.add_method("getParallelGroups", |lua, this, ()| {
783            match this.inner.borrow().get_parallel_groups() {
784                Ok(groups) => {
785                    let outer = lua.create_table()?;
786                    for (i, group) in groups.iter().enumerate() {
787                        let inner_t = lua.create_table()?;
788                        for (j, n) in group.iter().enumerate() {
789                            inner_t.set(j + 1, n.clone())?;
790                        }
791                        outer.set(i + 1, inner_t)?;
792                    }
793                    Ok((Some(outer), None::<String>))
794                }
795                Err(e) => Ok((None, Some(e))),
796            }
797        });
798
799        // -- run --
800        /// Executes all pipeline steps synchronously in dependency order. Blocks until all steps complete, fail, or are cancelled.
801        /// @param | context | table? | An optional shared context table passed to every step callback. A fresh table is created if omitted.
802        /// @return | table | A result table with fields: success (boolean), completed, failed, skipped, cancelled (arrays of names), totalDuration (number), errors (array of {name, msg}).
803        /// @field | success | boolean | Success flag.
804        /// @field | completed | string[] | Completed step names.
805        /// @field | failed | string[] | Failed step names.
806        /// @field | skipped | string[] | Skipped step names.
807        /// @field | cancelled | string[] | Cancelled step names.
808        /// @field | totalDuration | number | Total duration in seconds.
809        /// @field | errors | table | Array of error entries.
810        methods.add_method("run", |lua, this, context: Option<LuaTable>| {
811            let order = this
812                .inner
813                .borrow()
814                .get_execution_order()
815                .map_err(LuaError::runtime)?;
816            this.inner.borrow_mut().reset();
817            for w in this.step_wrappers.borrow().values() {
818                w.inner.borrow_mut().reset();
819            }
820            let ctx = context.unwrap_or(lua.create_table()?);
821            /// Performs the 'results' operation.
822            ctx.set("results", lua.create_table()?)?;
823            let abort_on_fail = this.inner.borrow().error_mode == ErrorMode::Abort;
824            let start = std::time::Instant::now();
825            'steps: for step_name in &order {
826                let wrapper = match this.step_wrappers.borrow().get(step_name).cloned() {
827                    Some(w) => w,
828                    None => continue,
829                };
830                let statuses: HashMap<String, StepStatus> = this
831                    .step_wrappers
832                    .borrow()
833                    .iter()
834                    .map(|(k, v)| (k.clone(), v.inner.borrow().status.clone()))
835                    .collect();
836                match this.inner.borrow().are_deps_satisfied(step_name, &statuses) {
837                    Ok(true) => {}
838                    Ok(false) => {
839                        if abort_on_fail {
840                            cancel_remaining_steps(&this.step_wrappers.borrow(), &order);
841                            break 'steps;
842                        } else {
843                            wrapper.inner.borrow_mut().status = StepStatus::Skipped;
844                            fire_step_callbacks(lua, this, step_name, &ctx, &wrapper)?;
845                            continue;
846                        }
847                    }
848                    Err(_) => {
849                        wrapper.inner.borrow_mut().status = StepStatus::Skipped;
850                        fire_step_callbacks(lua, this, step_name, &ctx, &wrapper)?;
851                        continue;
852                    }
853                }
854                fire_pipeline_event(
855                    lua,
856                    this,
857                    "step_started",
858                    step_name,
859                    StepStatus::Running.as_str(),
860                    LuaValue::Nil,
861                )?;
862                let succeeded = wrapper.execute_sync(lua, &ctx, abort_on_fail)?;
863                fire_step_callbacks(lua, this, step_name, &ctx, &wrapper)?;
864                if !succeeded {
865                    cancel_remaining_steps(&this.step_wrappers.borrow(), &order);
866                    break;
867                }
868            }
869            finalize_pipeline_result(lua, this, start.elapsed().as_secs_f32())
870        });
871
872        // -- runAsync --
873        /// Starts asynchronous (coroutine-based) execution of the pipeline. Call update(dt) each frame to advance steps.
874        /// @param | context | table? | An optional shared context table. A fresh table is created if omitted.
875        methods.add_method("runAsync", |lua, this, context: Option<LuaTable>| {
876            this.inner.borrow_mut().reset();
877            for wrapper in this.step_wrappers.borrow().values() {
878                wrapper.inner.borrow_mut().reset();
879            }
880            let ctx = match context {
881                Some(c) => c,
882                None => lua.create_table()?,
883            };
884            let results_table = lua.create_table()?;
885            /// Performs the 'results' operation.
886            ctx.set("results", results_table)?;
887            *this.context_key.borrow_mut() = Some(lua.create_registry_value(ctx)?);
888            *this.is_async.borrow_mut() = true;
889            this.running_threads.borrow_mut().clear();
890            this.started_at.borrow_mut().clear();
891            let pipeline = this.inner.borrow();
892            this.scheduler.borrow_mut().start(&pipeline);
893            let wrappers = this.step_wrappers.borrow();
894            for (name, wrapper) in wrappers.iter() {
895                let has_deps = !pipeline
896                    .get_step(name)
897                    .map(|s| s.deps.is_empty())
898                    .unwrap_or(true);
899                if !has_deps {
900                    wrapper.inner.borrow_mut().status = StepStatus::Waiting;
901                    this.scheduler
902                        .borrow_mut()
903                        .mark_step_waiting(name, &pipeline);
904                }
905            }
906            Ok(())
907        });
908
909        // -- update --
910        /// Advances an async pipeline by one frame tick. Resumes coroutines, checks dependencies, and fires callbacks. Call every frame after runAsync().
911        /// @param | dt | number | Delta time in seconds since last frame.
912        /// @return | boolean | True when the entire pipeline has finished (all steps done); false if still running.
913        methods.add_method("update", |lua, this, dt: f32| {
914            if !*this.is_async.borrow() {
915                return Ok(false);
916            }
917            let ctx: LuaTable = {
918                let key_opt = this.context_key.borrow();
919                match key_opt.as_ref() {
920                    Some(k) => lua.registry_value(k)?,
921                    None => return Ok(false),
922                }
923            };
924            let abort_on_fail = this.inner.borrow().error_mode == ErrorMode::Abort;
925            let running_names: Vec<String> =
926                this.running_threads.borrow().keys().cloned().collect();
927            for step_name in running_names {
928                let wrapper = match this.step_wrappers.borrow().get(&step_name).cloned() {
929                    Some(w) => w,
930                    None => continue,
931                };
932                let done = execute_async_coroutine_step(lua, this, &step_name, &wrapper, &ctx)?;
933                if done {
934                    fire_step_callbacks(lua, this, &step_name, &ctx, &wrapper)?;
935                    if wrapper.inner.borrow().status == StepStatus::Failed && abort_on_fail {
936                        this.step_wrappers.borrow().values().for_each(|w| {
937                            let s = w.inner.borrow().status.clone();
938                            if s == StepStatus::Pending || s == StepStatus::Waiting {
939                                w.inner.borrow_mut().status = StepStatus::Cancelled;
940                            }
941                        });
942                        *this.is_async.borrow_mut() = false;
943                        this.scheduler.borrow_mut().reset();
944                        return finalize_pipeline_result(
945                            lua,
946                            this,
947                            this.scheduler.borrow().elapsed,
948                        )
949                        .map(|_| true);
950                    }
951                }
952            }
953            let ready_names: Vec<String> = {
954                let pipeline = this.inner.borrow();
955                this.scheduler.borrow_mut().update(dt, &pipeline)
956            };
957            for step_name in ready_names {
958                let wrapper = match this.step_wrappers.borrow().get(&step_name).cloned() {
959                    Some(w) => w,
960                    None => continue,
961                };
962                fire_pipeline_event(
963                    lua,
964                    this,
965                    "step_started",
966                    &step_name,
967                    StepStatus::Running.as_str(),
968                    LuaValue::Nil,
969                )?;
970                let succeeded = if wrapper.is_async_enabled() {
971                    let done = execute_async_coroutine_step(lua, this, &step_name, &wrapper, &ctx)?;
972                    if done {
973                        fire_step_callbacks(lua, this, &step_name, &ctx, &wrapper)?;
974                    }
975                    done || wrapper.inner.borrow().status != StepStatus::Failed
976                } else {
977                    let ok = wrapper.execute_sync(lua, &ctx, abort_on_fail)?;
978                    fire_step_callbacks(lua, this, &step_name, &ctx, &wrapper)?;
979                    ok
980                };
981                if !succeeded {
982                    this.step_wrappers.borrow().values().for_each(|w| {
983                        let s = w.inner.borrow().status.clone();
984                        if s == StepStatus::Pending || s == StepStatus::Waiting {
985                            w.inner.borrow_mut().status = StepStatus::Cancelled;
986                        }
987                    });
988                    *this.is_async.borrow_mut() = false;
989                    this.scheduler.borrow_mut().reset();
990                    return finalize_pipeline_result(lua, this, this.scheduler.borrow().elapsed)
991                        .map(|_| true);
992                }
993                let statuses: HashMap<String, StepStatus> = this
994                    .step_wrappers
995                    .borrow()
996                    .iter()
997                    .map(|(k, v)| (k.clone(), v.inner.borrow().status.clone()))
998                    .collect();
999                for (name, w) in this.step_wrappers.borrow().iter() {
1000                    if w.inner.borrow().status != StepStatus::Pending {
1001                        continue;
1002                    }
1003                    let pipeline = this.inner.borrow();
1004                    if let Ok(true) = pipeline.are_deps_satisfied(name, &statuses) {
1005                        w.inner.borrow_mut().status = StepStatus::Waiting;
1006                        this.scheduler
1007                            .borrow_mut()
1008                            .mark_step_waiting(name, &pipeline);
1009                    } else if let Ok(false) = pipeline.are_deps_satisfied(name, &statuses) {
1010                        w.inner.borrow_mut().status = StepStatus::Skipped;
1011                        fire_step_callbacks(lua, this, name, &ctx, w)?;
1012                    }
1013                }
1014            }
1015            let all_done = this.step_wrappers.borrow().values().all(|w| {
1016                matches!(
1017                    w.inner.borrow().status,
1018                    StepStatus::Completed
1019                        | StepStatus::Failed
1020                        | StepStatus::Skipped
1021                        | StepStatus::Cancelled
1022                )
1023            });
1024            if all_done {
1025                *this.is_async.borrow_mut() = false;
1026                this.scheduler.borrow_mut().reset();
1027                finalize_pipeline_result(lua, this, this.scheduler.borrow().elapsed).map(|_| true)
1028            } else {
1029                Ok(false)
1030            }
1031        });
1032
1033        // -- cancel --
1034        /// Cancels all pending and waiting steps. Steps already running or completed are unaffected.
1035        methods.add_method("cancel", |_, this, ()| {
1036            let wrappers = this.step_wrappers.borrow();
1037            for w in wrappers.values() {
1038                let s = w.inner.borrow().status.clone();
1039                if s == StepStatus::Pending || s == StepStatus::Waiting {
1040                    w.inner.borrow_mut().status = StepStatus::Cancelled;
1041                }
1042            }
1043            this.running_threads.borrow_mut().clear();
1044            this.started_at.borrow_mut().clear();
1045            Ok(())
1046        });
1047
1048        // -- reset --
1049        /// Resets the pipeline and all steps back to their initial pending state, clearing context and async state.
1050        methods.add_method("reset", |_, this, ()| {
1051            this.inner.borrow_mut().reset();
1052            for w in this.step_wrappers.borrow().values() {
1053                w.inner.borrow_mut().reset();
1054            }
1055            *this.context_key.borrow_mut() = None;
1056            *this.is_async.borrow_mut() = false;
1057            this.scheduler.borrow_mut().reset();
1058            this.running_threads.borrow_mut().clear();
1059            this.started_at.borrow_mut().clear();
1060            Ok(())
1061        });
1062
1063        // -- isRunning --
1064        /// Returns whether the pipeline is currently in async execution mode (started via runAsync and not yet finished).
1065        /// @return | boolean | True if the pipeline is actively running.
1066        methods.add_method("isRunning", |_, this, ()| Ok(*this.is_async.borrow()));
1067
1068        // -- isComplete --
1069        /// Returns whether all steps have reached a terminal state (completed, failed, skipped, or cancelled).
1070        /// @return | boolean | True if no steps are still pending or running.
1071        methods.add_method("isComplete", |_, this, ()| {
1072            let wrappers = this.step_wrappers.borrow();
1073            let all_done = wrappers.values().all(|w| {
1074                matches!(
1075                    w.inner.borrow().status,
1076                    StepStatus::Completed
1077                        | StepStatus::Failed
1078                        | StepStatus::Skipped
1079                        | StepStatus::Cancelled
1080                )
1081            });
1082            Ok(all_done)
1083        });
1084
1085        // -- setErrorMode --
1086        /// Sets how the pipeline handles step failures. "abort" stops on first failure; "continue" runs remaining steps.
1087        /// @param | mode | string | Either "abort" or "continue".
1088        methods.add_method("setErrorMode", |_, this, mode: String| {
1089            let em = ErrorMode::from_str_lua(&mode).map_err(LuaError::runtime)?;
1090            this.inner.borrow_mut().error_mode = em;
1091            Ok(())
1092        });
1093
1094        // -- getErrorMode --
1095        /// Returns the current error mode of the pipeline as a string.
1096        /// @return | string | "abort" or "continue".
1097        methods.add_method("getErrorMode", |_, this, ()| {
1098            Ok(this.inner.borrow().error_mode.as_str().to_string())
1099        });
1100
1101        // -- getResult --
1102        /// Returns the current pipeline result summary table, or nil if no steps exist. Useful for inspecting state after run or during async execution.
1103        /// @return | table | Result table with success, completed, failed, skipped, cancelled, totalDuration, errors fields, or nil if no steps exist.
1104        /// @field | success | boolean | Success flag.
1105        /// @field | completed | string[] | Completed step names.
1106        /// @field | failed | string[] | Failed step names.
1107        /// @field | skipped | string[] | Skipped step names.
1108        /// @field | cancelled | string[] | Cancelled step names.
1109        /// @field | totalDuration | number | Total duration in seconds.
1110        /// @field | errors | table | Array of error entries.
1111        methods.add_method("getResult", |lua, this, ()| {
1112            if this.step_wrappers.borrow().is_empty() {
1113                return Ok(None);
1114            }
1115            let elapsed = this.scheduler.borrow().elapsed;
1116            let step_statuses: HashMap<String, (StepStatus, Option<String>)> = this
1117                .step_wrappers
1118                .borrow()
1119                .iter()
1120                .map(|(k, v)| {
1121                    let i = v.inner.borrow();
1122                    (k.clone(), (i.status.clone(), i.error_msg.clone()))
1123                })
1124                .collect();
1125            let result = this.inner.borrow().collect_result(&step_statuses, elapsed);
1126            Ok(Some(pipeline_result_to_lua(lua, &result)?))
1127        });
1128
1129        // -- getContext --
1130        /// Returns the shared context table used by the current or most recent pipeline execution, or nil if none exists.
1131        /// @return | table | The pipeline context table.
1132        /// @return | nil | If no context has been set.
1133        methods.add_method("getContext", |lua, this, ()| {
1134            let key_opt = this.context_key.borrow();
1135            match key_opt.as_ref() {
1136                Some(k) => {
1137                    let t: LuaTable = lua.registry_value(k)?;
1138                    Ok(Some(t))
1139                }
1140                None => Ok(None),
1141            }
1142        });
1143
1144        // -- setOnComplete --
1145        /// Registers a callback invoked when the entire pipeline finishes execution. Receives the result table.
1146        /// @param | callback | function? | A function receiving the result table. Pass nil to remove.
1147        methods.add_method("setOnComplete", |lua, this, cb: Option<LuaFunction>| {
1148            *this.on_complete_key.borrow_mut() = match cb {
1149                Some(f) => Some(lua.create_registry_value(f)?),
1150                None => None,
1151            };
1152            Ok(())
1153        });
1154
1155        // -- setOnStepComplete --
1156        /// Registers a callback invoked each time any step completes successfully. Receives (stepName, context).
1157        /// @param | callback | function? | A function receiving (stepName, context). Pass nil to remove.
1158        methods.add_method("setOnStepComplete", |lua, this, cb: Option<LuaFunction>| {
1159            *this.on_step_complete_key.borrow_mut() = match cb {
1160                Some(f) => Some(lua.create_registry_value(f)?),
1161                None => None,
1162            };
1163            Ok(())
1164        });
1165
1166        // -- setOnStepError --
1167        /// Registers a callback invoked each time any step fails. Receives (stepName, errorMessage).
1168        /// @param | callback | function? | A function receiving (stepName, errorMessage). Pass nil to remove.
1169        methods.add_method("setOnStepError", |lua, this, cb: Option<LuaFunction>| {
1170            *this.on_step_error_key.borrow_mut() = match cb {
1171                Some(f) => Some(lua.create_registry_value(f)?),
1172                None => None,
1173            };
1174            Ok(())
1175        });
1176
1177        // -- getName --
1178        /// Returns the name of this pipeline. This method is available to Lua scripts.
1179        /// @return | string | Pipeline name.
1180        methods.add_method("getName", |_, this, ()| {
1181            Ok(this.inner.borrow().name.clone())
1182        });
1183
1184        // -- setName --
1185        /// Changes the name of this pipeline. This method is available to Lua scripts.
1186        /// @param | name | string | New pipeline name.
1187        methods.add_method("setName", |_, this, name: String| {
1188            this.inner.borrow_mut().name = name;
1189            Ok(())
1190        });
1191
1192        // -- toTable --
1193        /// Serializes the pipeline configuration into a plain Lua table for inspection or persistence.
1194        /// @return | table | A table with name, errorMode, and steps array fields.
1195        /// @field | name | string | Pipeline name.
1196        /// @field | errorMode | string | Error handling mode.
1197        /// @field | steps | table | Array of step tables.
1198        methods.add_method("toTable", |lua, this, ()| {
1199            let t = lua.create_table()?;
1200            let pipeline = this.inner.borrow();
1201            /// Performs the 'name' operation.
1202            t.set("name", pipeline.name.clone())?;
1203            /// Performs the 'errorMode' operation.
1204            t.set("errorMode", pipeline.error_mode.as_str())?;
1205            let steps_t = lua.create_table()?;
1206            let wrappers = this.step_wrappers.borrow();
1207            let mut i = 1usize;
1208            for (name, wrapper) in wrappers.iter() {
1209                let st = lua.create_table()?;
1210                let inner = wrapper.inner.borrow();
1211                /// Performs the 'name' operation.
1212                st.set("name", name.clone())?;
1213                let deps_t = lua.create_table()?;
1214                for (j, d) in inner.deps.iter().enumerate() {
1215                    deps_t.set(j + 1, d.clone())?;
1216                }
1217                /// Performs the 'deps' operation.
1218                st.set("deps", deps_t)?;
1219                /// Performs the 'delay' operation.
1220                st.set("delay", inner.delay)?;
1221                /// Performs the 'optional' operation.
1222                st.set("optional", inner.optional)?;
1223                /// Performs the 'retryCount' operation.
1224                st.set("retryCount", inner.retry_count)?;
1225                /// Performs the 'retryDelay' operation.
1226                st.set("retryDelay", inner.retry_delay)?;
1227                /// Performs the 'async' operation.
1228                st.set("async", wrapper.is_async_enabled())?;
1229                if let Some(ref tag) = inner.tag {
1230                    /// The 'tag' field value exposed to Lua scripts.
1231                    st.set("tag", tag.clone())?;
1232                }
1233                steps_t.set(i, st)?;
1234                i += 1;
1235            }
1236            /// Performs the 'steps' operation.
1237            t.set("steps", steps_t)?;
1238            Ok(t)
1239        });
1240
1241        methods.add_meta_method(LuaMetaMethod::ToString, |_, this, ()| {
1242            let inner = this.inner.borrow();
1243            Ok(format!(
1244                "Pipeline(\"{}\", {} steps)",
1245                inner.name,
1246                inner.get_step_count()
1247            ))
1248        });
1249
1250        // -- type --
1251        /// Returns the type name of this object ("LPipeline").
1252        /// @return | string | Type identifier.
1253        methods.add_method("type", |_, _, ()| Ok("LPipeline"));
1254
1255        // -- addConditional --
1256        /// Convenience method to create and add a step with dependencies and a condition in one call.
1257        /// @param | name | string | Unique step name.
1258        /// @param | deps | table | Array of dependency step names.
1259        /// @param | callback | function | The step callback function.
1260        /// @param | condition | function | Predicate function; step runs only if it returns true.
1261        /// @return | LPipeline | Returns self for method chaining.
1262        methods.add_method(
1263            "addConditional",
1264            |lua,
1265             this,
1266             (name, deps_tbl, cb, cond): (String, LuaTable, LuaFunction, LuaFunction)| {
1267                let dep_names: Vec<String> = deps_tbl
1268                    .sequence_values::<String>()
1269                    .collect::<LuaResult<Vec<_>>>()?;
1270                let mut step = PipelineStep::new(name.clone());
1271                step.deps = dep_names;
1272                let wrapper = LuaStep::new(step.clone());
1273                *wrapper.callback_key.borrow_mut() = Some(lua.create_registry_value(cb)?);
1274                *wrapper.condition_key.borrow_mut() = Some(lua.create_registry_value(cond)?);
1275                this.inner
1276                    .borrow_mut()
1277                    .add_step(step)
1278                    .map_err(LuaError::runtime)?;
1279                this.step_wrappers.borrow_mut().insert(name, wrapper);
1280                Ok(this.clone())
1281            },
1282        );
1283
1284        // -- addBranch --
1285        /// Adds a branching construct: evaluates a predicate, then runs either the "then" or "else" callback based on the result.
1286        /// @param | name | string | Base name for the branch (generates internal guard/then/else sub-steps).
1287        /// @param | deps | table | Array of dependency step names that must complete before the branch evaluates.
1288        /// @param | when | function | Predicate function receiving context; returns true for the "then" path.
1289        /// @param | thenFn | function | Callback executed if the predicate returns true.
1290        /// @param | elseFn | function? | Callback executed if the predicate returns false. Defaults to a no-op.
1291        /// @return | LPipeline | Returns self for method chaining.
1292        methods.add_method(
1293            "addBranch",
1294            |lua,
1295             this,
1296             (name, deps_tbl, when_fn, then_fn, else_fn): (
1297                String,
1298                LuaTable,
1299                LuaFunction,
1300                LuaFunction,
1301                Option<LuaFunction>,
1302            )| {
1303                let dep_names: Vec<String> = deps_tbl
1304                    .sequence_values::<String>()
1305                    .collect::<LuaResult<Vec<_>>>()?;
1306                let guard_name = format!("{}__branch_guard", name);
1307                let then_name = format!("{}__then", name);
1308                let else_name = format!("{}__else", name);
1309                let branch_key = format!("pipeline.branch.{}", name);
1310                let when_key = lua.create_registry_value(when_fn)?;
1311                let guard_wrapper = LuaStep::new(PipelineStep::new(guard_name.clone()));
1312                {
1313                    let mut inner = guard_wrapper.inner.borrow_mut();
1314                    inner.deps = dep_names;
1315                }
1316                let branch_key_guard = branch_key.clone();
1317                let when_key_guard = when_key;
1318                let guard_cb = lua.create_function(move |lua, ctx: LuaTable| {
1319                    let branch_tbl: LuaTable = match ctx.get("branch") {
1320                        Ok(t) => t,
1321                        Err(_) => {
1322                            let t = lua.create_table()?;
1323                            /// Performs the 'branch' operation.
1324                            ctx.set("branch", t.clone())?;
1325                            t
1326                        }
1327                    };
1328                    let predicate: LuaFunction = lua.registry_value(&when_key_guard)?;
1329                    let pass: bool = predicate.call(ctx.clone())?;
1330                    branch_tbl.set(branch_key_guard.clone(), pass)?;
1331                    Ok(pass)
1332                })?;
1333                *guard_wrapper.callback_key.borrow_mut() =
1334                    Some(lua.create_registry_value(guard_cb)?);
1335                let then_wrapper = LuaStep::new(PipelineStep::new(then_name));
1336                then_wrapper
1337                    .inner
1338                    .borrow_mut()
1339                    .deps
1340                    .push(guard_name.clone());
1341                *then_wrapper.callback_key.borrow_mut() = Some(lua.create_registry_value(then_fn)?);
1342                let branch_key_then = branch_key.clone();
1343                let then_cond = lua.create_function(move |_, ctx: LuaTable| {
1344                    let branch_tbl: LuaTable = match ctx.get("branch") {
1345                        Ok(t) => t,
1346                        Err(_) => return Ok(false),
1347                    };
1348                    Ok(branch_tbl
1349                        .get::<_, bool>(branch_key_then.clone())
1350                        .unwrap_or(false))
1351                })?;
1352                *then_wrapper.condition_key.borrow_mut() =
1353                    Some(lua.create_registry_value(then_cond)?);
1354                let else_wrapper = LuaStep::new(PipelineStep::new(else_name));
1355                else_wrapper
1356                    .inner
1357                    .borrow_mut()
1358                    .deps
1359                    .push(guard_name.clone());
1360                let else_cb: LuaFunction = match else_fn {
1361                    Some(f) => f,
1362                    None => lua.create_function(|_, _: LuaTable| Ok(LuaValue::Nil))?,
1363                };
1364                *else_wrapper.callback_key.borrow_mut() = Some(lua.create_registry_value(else_cb)?);
1365                let branch_key_else = branch_key;
1366                let else_cond = lua.create_function(move |_, ctx: LuaTable| {
1367                    let branch_tbl: LuaTable = match ctx.get("branch") {
1368                        Ok(t) => t,
1369                        Err(_) => return Ok(false),
1370                    };
1371                    Ok(!branch_tbl
1372                        .get::<_, bool>(branch_key_else.clone())
1373                        .unwrap_or(false))
1374                })?;
1375                *else_wrapper.condition_key.borrow_mut() =
1376                    Some(lua.create_registry_value(else_cond)?);
1377                let mut pipeline = this.inner.borrow_mut();
1378                pipeline
1379                    .add_step(guard_wrapper.inner.borrow().clone())
1380                    .map_err(LuaError::runtime)?;
1381                pipeline
1382                    .add_step(then_wrapper.inner.borrow().clone())
1383                    .map_err(LuaError::runtime)?;
1384                pipeline
1385                    .add_step(else_wrapper.inner.borrow().clone())
1386                    .map_err(LuaError::runtime)?;
1387                drop(pipeline);
1388                this.step_wrappers
1389                    .borrow_mut()
1390                    .insert(guard_name, guard_wrapper);
1391                this.step_wrappers
1392                    .borrow_mut()
1393                    .insert(format!("{}__then", name), then_wrapper);
1394                this.step_wrappers
1395                    .borrow_mut()
1396                    .insert(format!("{}__else", name), else_wrapper);
1397                Ok(this.clone())
1398            },
1399        );
1400
1401        // -- onProgress --
1402        /// Registers a progress callback invoked after each step finishes (regardless of outcome). Receives (stepName, statusString).
1403        /// @param | callback | function | A function receiving (stepName, status).
1404        methods.add_method("onProgress", |lua, this, cb: LuaFunction| {
1405            *this.on_progress_key.borrow_mut() = Some(lua.create_registry_value(cb)?);
1406            Ok(())
1407        });
1408
1409        // -- onEvent --
1410        /// Registers a low-level event callback for all pipeline lifecycle events. Receives (eventName, stepName, status, detail).
1411        /// @param | callback | function | A function receiving (eventName, stepName, status, detail).
1412        methods.add_method("onEvent", |lua, this, cb: LuaFunction| {
1413            *this.on_event_key.borrow_mut() = Some(lua.create_registry_value(cb)?);
1414            Ok(())
1415        });
1416
1417        // -- toAscii --
1418        /// Returns an ASCII art diagram of the pipeline's dependency graph for debugging and visualization.
1419        /// @return | string | Multi-line ASCII diagram.
1420        methods.add_method("toAscii", |_, this, ()| {
1421            Ok(this.inner.borrow().to_ascii_diagram())
1422        });
1423
1424        // -- addSubPipeline --
1425        /// Embeds another pipeline's steps into this pipeline under an alias prefix, with optional outer dependencies.
1426        /// @param | subPipeline | LPipeline | The pipeline whose steps will be merged in.
1427        /// @param | alias | string | A prefix applied to all merged step names to avoid collisions.
1428        /// @param | deps | table? | Optional array of step names that all merged steps depend on.
1429        methods.add_method("addSubPipeline", |_, this, (sub_ud, alias, deps_tbl): (mlua::AnyUserData, String, Option<mlua::Table>)| {
1430                let sub_ref = sub_ud.borrow::<LuaPipeline>().map_err(mlua::Error::external)?;
1431                let sub_clone = sub_ref.inner.borrow().clone();
1432                let outer_deps: Vec<String> = if let Some(tbl) = deps_tbl {
1433                    tbl.pairs::<mlua::Value, String>()
1434                        .filter_map(|r| r.ok().map(|(_, v)| v))
1435                        .collect()
1436                } else {
1437                    Vec::new()
1438                };
1439                this.inner
1440                    .borrow_mut()
1441                    .add_sub_pipeline(sub_clone, &alias, outer_deps)
1442                    .map_err(mlua::Error::external)?;
1443                Ok(())
1444            },
1445        );
1446
1447        // -- typeOf --
1448        /// Checks whether this object is of a given type name. Accepts "LPipeline", "Pipeline", or "Object".
1449        /// @param | name | string | Type name to check against.
1450        /// @return | boolean | True if the type matches.
1451        methods.add_method("typeOf", |_, _, name: String| {
1452            Ok(name == "LPipeline" || name == "LObject")
1453        });
1454    }

1456/// Registers the `lurek.pipeline` module into the Lua runtime.
1457pub fn register(lua: &Lua, lurek: &LuaTable, _state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
1458    let tbl = lua.create_table()?;
1459
1460    // -- newStep --
1461    /// Creates a new pipeline step with the given name and an optional callback function.
1462    /// @param | name | string | Unique step name.
1463    /// @param | callback | function? | Optional callback executed when this step runs.
1464    /// @return | LPipelineStep | The new step object.
1465    tbl.set(
1466        "newStep",
1467        lua.create_function(|lua, (name, callback): (String, Option<LuaFunction>)| {
1468            let step = PipelineStep::new(name);
1469            let wrapper = LuaStep::new(step);
1470            if let Some(cb) = callback {
1471                *wrapper.callback_key.borrow_mut() = Some(lua.create_registry_value(cb)?);
1472            }
1473            Ok(wrapper)
1474        })?,
1475    )?;
1476
1477    // -- newPipeline --
1478    /// Creates a new empty pipeline with an optional name. Add steps via addStep() or addConditional().
1479    /// @param | name | string? | Pipeline name (defaults to "pipeline").
1480    /// @return | LPipeline | The new pipeline object.
1481    tbl.set(
1482        "newPipeline",
1483        lua.create_function(|_, name: Option<String>| {
1484            let pipeline = Pipeline::new(name.unwrap_or_else(|| "pipeline".to_string()));
1485            Ok(LuaPipeline::new(pipeline))
1486        })?,
1487    )?;
1488
1489    // -- fromTable --
1490    /// Creates a pipeline pre-populated with steps from a declarative table definition. Each step entry can specify name, deps, delay, optional, retryCount, retryDelay, async, tag, and fn.
1491    /// @param | definition | table | A table with optional name, errorMode, and a steps array.
1492    /// @return | LPipeline | The constructed pipeline.
1493    tbl.set(
1494        "fromTable",
1495        lua.create_function(|lua, def: LuaTable| {
1496            let name: Option<String> = def.get("name")?;
1497            let error_mode_str: Option<String> = def.get("errorMode")?;
1498            let mut pipeline = Pipeline::new(name.unwrap_or_else(|| "pipeline".to_string()));
1499            if let Some(ref mode) = error_mode_str {
1500                pipeline.error_mode = match mode.as_str() {
1501                    "continue" => ErrorMode::Continue,
1502                    _ => ErrorMode::Abort,
1503                };
1504            }
1505            let pipeline_rc = Rc::new(RefCell::new(pipeline));
1506            let wrappers_rc: Rc<RefCell<HashMap<String, LuaStep>>> =
1507                Rc::new(RefCell::new(HashMap::new()));
1508            if let Ok(steps_t) = def.get::<_, LuaTable<'_>>("steps") {
1509                for pair in steps_t.sequence_values::<LuaTable<'_>>() {
1510                    let st: LuaTable = pair?;
1511                    let sname: String = st.get("name")?;
1512                    let mut step = PipelineStep::new(sname.clone());
1513                    if let Ok(deps_t) = st.get::<_, LuaTable<'_>>("deps") {
1514                        for dep_v in deps_t.sequence_values::<String>() {
1515                            step.deps.push(dep_v?);
1516                        }
1517                    }
1518                    if let Ok(delay) = st.get::<_, f32>("delay") {
1519                        step.delay = delay;
1520                    }
1521                    if let Ok(optional) = st.get::<_, bool>("optional") {
1522                        step.optional = optional;
1523                    }
1524                    if let Ok(rc) = st.get::<_, u32>("retryCount") {
1525                        step.retry_count = rc;
1526                    }
1527                    if let Ok(rd) = st.get::<_, f32>("retryDelay") {
1528                        step.retry_delay = rd;
1529                    }
1530                    if let Ok(tag) = st.get::<_, String>("tag") {
1531                        step.tag = Some(tag);
1532                    }
1533                    if let Ok(async_enabled) = st.get::<_, bool>("async") {
1534                        step.metadata
1535                            .insert("__pipeline_async".to_string(), async_enabled.to_string());
1536                    }
1537                    let wrapper = LuaStep::new(step.clone());
1538                    if let Ok(cb) = st.get::<_, LuaFunction<'_>>("fn") {
1539                        *wrapper.callback_key.borrow_mut() = Some(lua.create_registry_value(cb)?);
1540                    }
1541                    pipeline_rc
1542                        .borrow_mut()
1543                        .add_step(step)
1544                        .map_err(LuaError::runtime)?;
1545                    wrappers_rc.borrow_mut().insert(sname, wrapper);
1546                }
1547            }
1548            Ok(LuaPipeline::from_parts(pipeline_rc, wrappers_rc))
1549        })?,
1550    )?;
1551    /// Performs the 'pipeline' operation.
1552    lurek.set("pipeline", tbl)?;
1553    Ok(())
