13/// Parses a Lua table of nodes into a Vec of LayoutNode.
14fn parse_nodes(tbl: &LuaTable) -> LuaResult<Vec<LayoutNode>> {
15    let mut nodes = Vec::new();
16    for pair in tbl.clone().sequence_values::<LuaTable>() {
17        let t = pair?;
18        let id: NodeId = t.get("id")?;
19        let mut node = LayoutNode::new(id);
20        if let Ok(w) = t.get::<_, f64>("width") {
21            node.width = w;
22        }
23        if let Ok(h) = t.get::<_, f64>("height") {
24            node.height = h;
25        }
26        if let Ok(label) = t.get::<_, String>("label") {
27            node.label = Some(label);
28        }
29        nodes.push(node);
30    }
31    Ok(nodes)
32}
33
34/// Parses a Lua table of edges into a Vec of LayoutEdge.
35fn parse_edges(tbl: &LuaTable) -> LuaResult<Vec<LayoutEdge>> {
36    let mut edges = Vec::new();
37    for pair in tbl.clone().sequence_values::<LuaTable>() {
38        let t = pair?;
39        let from: NodeId = t.get("from")?;
40        let to: NodeId = t.get("to")?;
41        let weight: f64 = t.get("weight").unwrap_or(1.0);
42        edges.push(LayoutEdge { from, to, weight });
43    }
44    Ok(edges)
45}

47/// Converts a LayoutResult into a Lua table.
48fn result_to_lua<'lua>(lua: &'lua Lua, result: &LayoutResult) -> LuaResult<LuaTable<'lua>> {
49    let tbl = lua.create_table()?;
50    let nodes_tbl = lua.create_table()?;
51    for (i, node) in result.nodes.iter().enumerate() {
52        let n = lua.create_table()?;
53        n.set("id", node.id)?;
54        n.set("x", node.x)?;
55        n.set("y", node.y)?;
56        n.set("width", node.width)?;
57        n.set("height", node.height)?;
58        if let Some(ref label) = node.label {
59            n.set("label", label.clone())?;
60        }
61        nodes_tbl.set(i + 1, n)?;
62    }
63    /// Laid-out node array with id, x, y, width, height, and optional label fields.
64    tbl.set("nodes", nodes_tbl)?;
65    /// Total width of the laid-out graph.
66    tbl.set("width", result.width)?;
67    /// Total height of the laid-out graph.
68    tbl.set("height", result.height)?;
69    Ok(tbl)
70}

72/// Parses a LayoutConfig from a Lua table or returns defaults.
73fn parse_config(tbl: Option<LuaTable>) -> LayoutConfig {
74    match tbl {
75        Some(t) => LayoutConfig {
76            h_spacing: t.get("hSpacing").unwrap_or(50.0),
77            v_spacing: t.get("vSpacing").unwrap_or(80.0),
78            margin: t.get("margin").unwrap_or(20.0),
79        },
80        None => LayoutConfig::default(),
81    }
82}

84/// Registers the `lurek.layout` namespace with tree, dag, force, and post-processing functions.
85/// @module layout
86pub fn register(lua: &Lua, lurek: &LuaTable, _state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
87    let layout_table = lua.create_table()?;
88
89    // lurek.layout.tree(nodes, children, root, config?) → result
90    /// Lays out a tree using the Reingold-Tilford algorithm.
91    /// @param | nodes | table | Array of node tables with id, width, height, label fields.
92    /// @param | children | table | Map of parent node ID to array of child node IDs.
93    /// @param | root | integer | ID of the root node.
94    /// @param | config | table|nil | Optional config with hSpacing, vSpacing, margin.
95    /// @return | table | Layout result with nodes array, width, height.
96    layout_table.set(
97        "tree",
98        lua.create_function(
99            |lua,
100             (nodes_tbl, children_tbl, root, config_tbl): (
101                LuaTable,
102                LuaTable,
103                NodeId,
104                Option<LuaTable>,
105            )| {
106                let nodes = parse_nodes(&nodes_tbl)?;
107                let config = parse_config(config_tbl);
108
109                let mut children: HashMap<NodeId, Vec<NodeId>> = HashMap::new();
110                for pair in children_tbl.pairs::<LuaValue, LuaTable>() {
111                    let (key, val) = pair?;
112                    let parent_id: NodeId = match key {
113                        LuaValue::Integer(i) => i as NodeId,
114                        LuaValue::Number(n) => n as NodeId,
115                        _ => continue,
116                    };
117                    let kids: Vec<NodeId> =
118                        val.sequence_values::<usize>().filter_map(|v| v.ok()).collect();
119                    children.insert(parent_id, kids);
120                }
121
122                let result = layout_tree(&nodes, &children, root, &config);
123                result_to_lua(lua, &result)
124            },
125        )?,
126    )?;
127
128    // lurek.layout.dag(nodes, edges, config?) → result
129    /// Lays out a DAG using the Sugiyama layered algorithm.
130    /// @param | nodes | table | Array of node tables with id, width, height, label fields.
131    /// @param | edges | table | Array of edge tables with from, to, weight fields.
132    /// @param | config | table|nil | Optional config with hSpacing, vSpacing, margin.
133    /// @return | table | Layout result with nodes array, width, height.
134    layout_table.set(
135        "dag",
136        lua.create_function(
137            |lua, (nodes_tbl, edges_tbl, config_tbl): (LuaTable, LuaTable, Option<LuaTable>)| {
138                let nodes = parse_nodes(&nodes_tbl)?;
139                let edges = parse_edges(&edges_tbl)?;
140                let config = parse_config(config_tbl);
141                let result = layout_dag(&nodes, &edges, &config);
142                result_to_lua(lua, &result)
143            },
144        )?,
145    )?;
146
147    // lurek.layout.force(nodes, edges, config?) → result
148    /// Lays out a graph using force-directed Fruchterman-Reingold simulation.
149    /// @param | nodes | table | Array of node tables with id, width, height, label fields.
150    /// @param | edges | table | Array of edge tables with from, to, weight fields.
151    /// @param | config | table|nil | Optional config with iterations, repulsion, attraction, cooling, areaWidth, areaHeight.
152    /// @return | table | Layout result with nodes array, width, height.
153    layout_table.set(
154        "force",
155        lua.create_function(
156            |lua, (nodes_tbl, edges_tbl, config_tbl): (LuaTable, LuaTable, Option<LuaTable>)| {
157                let nodes = parse_nodes(&nodes_tbl)?;
158                let edges = parse_edges(&edges_tbl)?;
159
160                let force_config = match config_tbl {
161                    Some(t) => ForceConfig {
162                        iterations: t.get("iterations").unwrap_or(100),
163                        repulsion: t.get("repulsion").unwrap_or(10000.0),
164                        attraction: t.get("attraction").unwrap_or(0.01),
165                        cooling: t.get("cooling").unwrap_or(0.95),
166                        area_width: t.get("areaWidth").unwrap_or(800.0),
167                        area_height: t.get("areaHeight").unwrap_or(600.0),
168                    },
169                    None => ForceConfig::default(),
170                };
171
172                let result = layout_force(&nodes, &edges, &force_config);
173                result_to_lua(lua, &result)
174            },
175        )?,
176    )?;
177
178    // lurek.layout.snapToGrid(result, gridSize) → result
179    /// Snaps all node positions to the nearest grid point.
180    /// @param | result | table | A layout result table with nodes array.
181    /// @param | gridSize | number | Grid cell size in pixels.
182    /// @return | table | New layout result with snapped positions.
183    layout_table.set(
184        "snapToGrid",
185        lua.create_function(|lua, (result_tbl, grid_size): (LuaTable, f64)| {
186            let nodes_tbl: LuaTable = result_tbl.get("nodes")?;
187            let nodes = parse_nodes(&nodes_tbl)?;
188            let mut result = LayoutResult::new(nodes);
189            snap_to_grid(&mut result, grid_size);
190            result_to_lua(lua, &result)
191        })?,
192    )?;
193
194    // lurek.layout.centerInArea(result, width, height) → result
195    /// Centers the layout within a given area.
196    /// @param | result | table | A layout result table with nodes array.
197    /// @param | width | number | Target area width.
198    /// @param | height | number | Target area height.
199    /// @return | table | New layout result centered in the area.
200    layout_table.set(
201        "centerInArea",
202        lua.create_function(|lua, (result_tbl, width, height): (LuaTable, f64, f64)| {
203            let nodes_tbl: LuaTable = result_tbl.get("nodes")?;
204            let nodes = parse_nodes(&nodes_tbl)?;
205            let mut result = LayoutResult::new(nodes);
206            center_in_area(&mut result, width, height);
207            result_to_lua(lua, &result)
208        })?,
209    )?;
210
211    lurek.set("layout", layout_table)?;
212    Ok(())
213}
