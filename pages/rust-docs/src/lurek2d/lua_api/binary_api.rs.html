16/// Converts Lua varargs into binary pack values.
17fn lua_values_to_pack(vals: LuaMultiValue) -> Vec<PackValue> {
18    vals.into_iter()
19        .map(|v| match v {
20            LuaValue::Integer(n) => PackValue::Int(n),
21            LuaValue::Number(n) => PackValue::Double(n),
22            LuaValue::String(s) => PackValue::Str(s.to_str().unwrap_or("").to_string()),
23            LuaValue::Boolean(b) => PackValue::Int(b as i64),
24            _ => PackValue::Int(0),
25        })
26        .collect()
27}
28/// Converts binary pack values into Lua return values.
29fn pack_values_to_lua(lua: &Lua, vals: Vec<PackValue>) -> LuaResult<Vec<LuaValue<'_>>> {
30    vals.into_iter()
31        .map(|v| match v {
32            PackValue::Int(n) => Ok(LuaValue::Integer(n)),
33            PackValue::UInt(n) => Ok(LuaValue::Integer(n as i64)),
34            PackValue::Float(f) => Ok(LuaValue::Number(f as f64)),
35            PackValue::Double(f) => Ok(LuaValue::Number(f)),
36            PackValue::Str(s) => lua.create_string(s.as_bytes()).map(LuaValue::String),
37            PackValue::Bytes(b) => lua.create_string(&b[..]).map(LuaValue::String),
38        })
39        .collect()
40}
41/// Converts Lua varargs into binary reader and writer values.
42fn lua_values_to_bin(vals: LuaMultiValue) -> Vec<BinValue> {
43    vals.into_iter()
44        .map(|v| match v {
45            LuaValue::Integer(n) => BinValue::I64(n),
46            LuaValue::Number(n) => BinValue::F64(n),
47            LuaValue::String(s) => BinValue::Str(s.to_str().unwrap_or("").to_string()),
48            LuaValue::Boolean(b) => BinValue::Bool(b),
49            _ => BinValue::I64(0),
50        })
51        .collect()
52}
53/// Converts binary reader values into Lua return values.
54fn bin_values_to_lua(lua: &Lua, vals: Vec<BinValue>) -> LuaResult<Vec<LuaValue<'_>>> {
55    vals.into_iter()
56        .map(|v| match v {
57            BinValue::U8(n) => Ok(LuaValue::Integer(n as i64)),
58            BinValue::U16(n) => Ok(LuaValue::Integer(n as i64)),
59            BinValue::U32(n) => Ok(LuaValue::Integer(n as i64)),
60            BinValue::U64(n) => Ok(LuaValue::Integer(n as i64)),
61            BinValue::I8(n) => Ok(LuaValue::Integer(n as i64)),
62            BinValue::I16(n) => Ok(LuaValue::Integer(n as i64)),
63            BinValue::I32(n) => Ok(LuaValue::Integer(n as i64)),
64            BinValue::I64(n) => Ok(LuaValue::Integer(n)),
65            BinValue::F32(f) => Ok(LuaValue::Number(f as f64)),
66            BinValue::F64(f) => Ok(LuaValue::Number(f)),
67            BinValue::Bool(b) => Ok(LuaValue::Boolean(b)),
68            BinValue::Str(s) => lua.create_string(&s).map(LuaValue::String),
69            BinValue::Bytes(b) => lua.create_string(&b[..]).map(LuaValue::String),
70        })
71        .collect()
72}
73/// Converts a Lua string or table of strings into byte chunks.
74fn lua_value_to_byte_chunks(value: LuaValue) -> LuaResult<Vec<Vec<u8>>> {
75    match value {
76        LuaValue::String(s) => Ok(vec![s.as_bytes().to_vec()]),
77        LuaValue::Table(t) => {
78            let mut chunks = Vec::new();
79            for (idx, value) in t.sequence_values::<LuaValue>().enumerate() {
80                let value = value?;
81                match value {
82                    LuaValue::String(s) => chunks.push(s.as_bytes().to_vec()),
83                    _ => {
84                        return Err(LuaError::RuntimeError(format!(
85                            "chunk table entry #{} must be a string",
86                            idx + 1
87                        )))
88                    }
89                }
90            }
91            if chunks.is_empty() {
92                return Err(LuaError::RuntimeError(
93                    "chunk table must contain at least one string".to_string(),
94                ));
95            }
96            Ok(chunks)
97        }
98        _ => Err(LuaError::RuntimeError(
99            "expected a string or an array-like table of strings".to_string(),
100        )),
101    }
102}
103/// Lua-side fixed-capacity FIFO buffer that stores registry-protected Lua values.
104pub struct LuaRingBuffer {
105    /// Stored Lua registry keys in oldest-to-newest order.
106    inner: VecDeque<LuaRegistryKey>,
107    /// Maximum number of values retained by the buffer.
108    capacity: usize,
109}
110impl LurekType for LuaRingBuffer {
111    const TYPE_NAME: &'static str = "LRingBuffer";
112    const TYPE_HIERARCHY: &'static [&'static str] = &["LRingBuffer", "LObject"];
113}
114impl LuaUserData for LuaRingBuffer {
115    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
116        // -- push --
117        /// Pushes a value into the ring buffer and evicts the oldest value when full.
118        /// @param | value | any | Lua value to store in the buffer.
119        /// @return | boolean | True when the push evicted an older value.
120        methods.add_method_mut("push", |lua, this, value: LuaValue| {
121            let key = lua.create_registry_value(value)?;
122            let was_full = this.inner.len() >= this.capacity;
123            if was_full {
124                if let Some(old_key) = this.inner.pop_front() {
125                    lua.remove_registry_value(old_key)?;
126                }
127            }
128            this.inner.push_back(key);
129            Ok(was_full)
130        });
131        // -- pop --
132        /// Removes and returns the oldest stored value from the ring buffer.
133        /// @return | LuaValue | Oldest stored value, or nil when the buffer is empty.
134        methods.add_method_mut("pop", |lua, this, ()| match this.inner.pop_front() {
135            Some(key) => {
136                let v: LuaValue = lua.registry_value(&key)?;
137                lua.remove_registry_value(key)?;
138                Ok(v)
139            }
140            None => Ok(LuaValue::Nil),
141        });
142        // -- peek --
143        /// Returns the oldest stored value without removing it from the ring buffer.
144        /// @return | LuaValue | Oldest stored value, or nil when the buffer is empty.
145        methods.add_method("peek", |lua, this, ()| match this.inner.front() {
146            Some(key) => lua.registry_value::<LuaValue>(key),
147            None => Ok(LuaValue::Nil),
148        });
149        // -- peekNewest --
150        /// Returns the newest stored value without removing it from the ring buffer.
151        /// @return | LuaValue | Newest stored value, or nil when the buffer is empty.
152        methods.add_method("peekNewest", |lua, this, ()| match this.inner.back() {
153            Some(key) => lua.registry_value::<LuaValue>(key),
154            None => Ok(LuaValue::Nil),
155        });
156        // -- len --
157        /// Returns the number of values currently stored.
158        /// @return | integer | Current buffer length.
159        methods.add_method("len", |_, this, ()| Ok(this.inner.len() as i64));
160        // -- capacity --
161        /// Returns the maximum capacity of the ring buffer.
162        /// @return | integer | Maximum number of stored values.
163        methods.add_method("capacity", |_, this, ()| Ok(this.capacity as i64));
164        // -- isEmpty --
165        /// Returns whether the ring buffer has no values.
166        /// @return | boolean | True when the buffer is empty.
167        methods.add_method("isEmpty", |_, this, ()| Ok(this.inner.is_empty()));
168        // -- isFull --
169        /// Returns whether the ring buffer is at capacity.
170        /// @return | boolean | True when the buffer length is at least its capacity.
171        methods.add_method(
172            "isFull",
173            |_, this, ()| Ok(this.inner.len() >= this.capacity),
174        );
175        // -- clear --
176        /// Removes every stored value and releases their registry keys.
177        methods.add_method_mut("clear", |lua, this, ()| {
178            while let Some(key) = this.inner.pop_front() {
179                lua.remove_registry_value(key)?;
180            }
181            Ok(())
182        });
183        // -- toTable --
184        /// Returns stored values in oldest-to-newest order.
185        /// @return | number[] | Array table of stored values.
186        methods.add_method("toTable", |lua, this, ()| {
187            let t = lua.create_table()?;
188            for (i, key) in this.inner.iter().enumerate() {
189                let v: LuaValue = lua.registry_value(key)?;
190                t.set(i + 1, v)?;
191            }
192            Ok(t)
193        });
194        // -- type --
195        /// Returns the Lua-visible type name for this ring buffer handle.
196        /// @return | string | The string `LRingBuffer`.
197        methods.add_method("type", |_, _, ()| Ok("LRingBuffer"));
198        // -- typeOf --
199        /// Returns whether this ring buffer handle matches a supported type name.
200        /// @param | name | string | Type name to compare against `LRingBuffer` and `Object`.
201        /// @return | boolean | True when the supplied type name matches this handle.
202        methods.add_method("typeOf", |_, _, name: String| {
203            Ok(name == "LRingBuffer" || name == "LObject")
204        });
205    }
206}
207/// Converts a TOML value into a Lua value.
208fn toml_value_to_lua<'lua>(lua: &'lua Lua, value: &toml::Value) -> LuaResult<LuaValue<'lua>> {
209    match value {
210        toml::Value::String(s) => lua.create_string(s.as_bytes()).map(LuaValue::String),
211        toml::Value::Integer(n) => Ok(LuaValue::Integer(*n)),
212        toml::Value::Float(f) => Ok(LuaValue::Number(*f)),
213        toml::Value::Boolean(b) => Ok(LuaValue::Boolean(*b)),
214        toml::Value::Array(arr) => {
215            let tbl = lua.create_table()?;
216            for (i, v) in arr.iter().enumerate() {
217                tbl.set(i + 1, toml_value_to_lua(lua, v)?)?;
218            }
219            Ok(LuaValue::Table(tbl))
220        }
221        toml::Value::Table(map) => {
222            let tbl = lua.create_table()?;
223            for (k, v) in map {
224                tbl.set(k.as_str(), toml_value_to_lua(lua, v)?)?;
225            }
226            Ok(LuaValue::Table(tbl))
227        }
228        toml::Value::Datetime(dt) => lua
229            .create_string(dt.to_string().as_bytes())
230            .map(LuaValue::String),
231    }
232}
233/// Converts a Lua value into a TOML value.
234fn lua_table_to_toml_value(value: &LuaValue) -> LuaResult<toml::Value> {
235    match value {
236        LuaValue::Boolean(b) => Ok(toml::Value::Boolean(*b)),
237        LuaValue::Integer(n) => Ok(toml::Value::Integer(*n)),
238        LuaValue::Number(f) => Ok(toml::Value::Float(*f)),
239        LuaValue::String(s) => {
240            let st = s
241                .to_str()
242                .map_err(|e| LuaError::RuntimeError(e.to_string()))?;
243            Ok(toml::Value::String(st.to_string()))
244        }
245        LuaValue::Table(tbl) => {
246            let len = tbl.raw_len();
247            if len > 0 {
248                let mut arr = Vec::new();
249                for i in 1..=len {
250                    let v: LuaValue = tbl.raw_get(i)?;
251                    arr.push(lua_table_to_toml_value(&v)?);
252                }
253                Ok(toml::Value::Array(arr))
254            } else {
255                let mut map = toml::map::Map::new();
256                for pair in tbl.clone().pairs::<LuaString, LuaValue>() {
257                    let (k, v) = pair?;
258                    let key = k
259                        .to_str()
260                        .map_err(|e| LuaError::RuntimeError(e.to_string()))?;
261                    map.insert(key.to_string(), lua_table_to_toml_value(&v)?);
262                }
263                Ok(toml::Value::Table(map))
264            }
265        }
266        _ => Err(LuaError::RuntimeError(
267            "Cannot convert this Lua type to TOML".to_string(),
268        )),
269    }
270}
271/// Registers the `lurek.data` API table with the Lua VM.
272pub fn register(lua: &Lua, luna: &LuaTable, _state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
273    let tbl = lua.create_table()?;
274    // -- pack --
275    /// Packs Lua values into a binary string using a format string.
276    /// @param | fmt | string | Binary pack format string.
277    /// @param | ... | any | Values to pack according to the format.
278    /// @return | string | Packed binary byte string.
279    tbl.set(
280        "pack",
281        lua.create_function(|lua, (fmt, vals): (String, LuaMultiValue)| {
282            let pvs = lua_values_to_pack(vals);
283            let bd = binary::pack(&fmt, &pvs).map_err(LuaError::RuntimeError)?;
284            lua.create_string(bd.as_bytes())
285        })?,
286    )?;
287    // -- unpack --
288    /// Unpacks values from a binary string using a format string.
289    /// @param | fmt | string | Binary unpack format string.
290    /// @param | raw | string | Binary byte string to unpack.
291    /// @param | offset | integer? | Optional zero-based byte offset; defaults to zero.
292    /// @return | LuaValue | Unpacked values followed by the next byte offset.
293    tbl.set(
294        "unpack",
295        lua.create_function(
296            |lua, (fmt, raw, offset): (String, LuaString, Option<usize>)| {
297                let (values, next_pos) = binary::unpack(&fmt, raw.as_bytes(), offset.unwrap_or(0))
298                    .map_err(LuaError::RuntimeError)?;
299                let mut result = pack_values_to_lua(lua, values)?;
300                result.push(LuaValue::Integer(next_pos as i64));
301                Ok(LuaMultiValue::from_vec(result))
302            },
303        )?,
304    )?;
305    // -- getPackedSize --
306    /// Computes the packed byte size for values and a format string.
307    /// @param | fmt | string | Binary pack format string.
308    /// @param | ... | any | Values measured according to the format.
309    /// @return | integer | Packed byte size.
310    tbl.set(
311        "getPackedSize",
312        lua.create_function(|_, (fmt, vals): (String, LuaMultiValue)| {
313            let pvs = lua_values_to_pack(vals);
314            binary::get_packed_size(&fmt, &pvs)
315                .map_err(LuaError::RuntimeError)
316                .map(|n| n as i64)
317        })?,
318    )?;
319    // -- compress --
320    /// Compresses a binary string using a named compression format.
321    /// @param | format_str | string | Compression format name.
322    /// @param | raw_data | string | Raw binary data to compress.
323    /// @param | level | integer? | Optional compression level; defaults to 6.
324    /// @return | string | Compressed binary byte string.
325    tbl.set(
326        "compress",
327        lua.create_function(
328            |lua, (format_str, raw_data, level): (String, LuaString, Option<u32>)| {
329                let format =
330                    CompressFormat::parse_str(&format_str).map_err(LuaError::RuntimeError)?;
331                let result = binary::compress(raw_data.as_bytes(), format, level.unwrap_or(6))
332                    .map_err(LuaError::RuntimeError)?;
333                lua.create_string(&result)
334            },
335        )?,
336    )?;
337    // -- decompress --
338    /// Decompresses a binary string using a named compression format.
339    /// @param | format_str | string | Compression format name.
340    /// @param | compressed | string | Compressed binary data.
341    /// @return | string | Decompressed binary byte string.
342    tbl.set(
343        "decompress",
344        lua.create_function(|lua, (format_str, compressed): (String, LuaString)| {
345            let format = CompressFormat::parse_str(&format_str).map_err(LuaError::RuntimeError)?;
346            let result =
347                binary::decompress(compressed.as_bytes(), format).map_err(LuaError::RuntimeError)?;
348            lua.create_string(&result)
349        })?,
350    )?;
351    // -- compressChunks --
352    /// Compresses a string or table of strings as a chunked byte stream.
353    /// @param | format_str | string | Compression format name.
354    /// @param | chunks | any | Binary string or array table of binary strings.
355    /// @param | level | integer? | Optional compression level; defaults to 6.
356    /// @return | string | Compressed binary byte string.
357    tbl.set(
358        "compressChunks",
359        lua.create_function(
360            |lua, (format_str, chunks, level): (String, LuaValue, Option<u32>)| {
361                let format =
362                    CompressFormat::parse_str(&format_str).map_err(LuaError::RuntimeError)?;
363                let chunks = lua_value_to_byte_chunks(chunks)?;
364                let chunk_refs: Vec<&[u8]> = chunks.iter().map(|c| c.as_slice()).collect();
365                let result = binary::compress_chunks(&chunk_refs, format, level.unwrap_or(6))
366                    .map_err(LuaError::RuntimeError)?;
367                lua.create_string(&result)
368            },
369        )?,
370    )?;
371    // -- decompressChunks --
372    /// Decompresses a string or table of strings as a chunked byte stream.
373    /// @param | format_str | string | Compression format name.
374    /// @param | chunks | any | Binary string or array table of binary strings.
375    /// @return | string | Decompressed binary byte string.
376    tbl.set(
377        "decompressChunks",
378        lua.create_function(|lua, (format_str, chunks): (String, LuaValue)| {
379            let format = CompressFormat::parse_str(&format_str).map_err(LuaError::RuntimeError)?;
380            let chunks = lua_value_to_byte_chunks(chunks)?;
381            let chunk_refs: Vec<&[u8]> = chunks.iter().map(|c| c.as_slice()).collect();
382            let result =
383                binary::decompress_chunks(&chunk_refs, format).map_err(LuaError::RuntimeError)?;
384            lua.create_string(&result)
385        })?,
386    )?;
387    // -- encode --
388    /// Encodes a binary string using a named text encoding format.
389    /// @param | format_str | string | Encoding format name.
390    /// @param | raw_data | string | Raw binary data to encode.
391    /// @return | string | Encoded string.
392    tbl.set(
393        "encode",
394        lua.create_function(|_, (format_str, raw_data): (String, LuaString)| {
395            let format = EncodeFormat::parse_str(&format_str).map_err(LuaError::RuntimeError)?;
396            Ok(binary::encode(format, raw_data.as_bytes()))
397        })?,
398    )?;
399    // -- decode --
400    /// Decodes a string using a named text encoding format.
401    /// @param | format_str | string | Encoding format name.
402    /// @param | encoded | string | Encoded string to decode.
403    /// @return | string | Decoded binary byte string.
404    tbl.set(
405        "decode",
406        lua.create_function(|lua, (format_str, encoded): (String, String)| {
407            let format = EncodeFormat::parse_str(&format_str).map_err(LuaError::RuntimeError)?;
408            let result = binary::decode(format, &encoded).map_err(LuaError::RuntimeError)?;
409            lua.create_string(&result)
410        })?,
411    )?;
412    // -- hash --
413    /// Hashes a binary string with a named algorithm.
414    /// @param | algo_str | string | Hash algorithm name.
415    /// @param | raw_data | string | Raw binary data to hash.
416    /// @return | string | Hash digest string.
417    tbl.set(
418        "hash",
419        lua.create_function(|_, (algo_str, raw_data): (String, LuaString)| {
420            let algo = HashAlgorithm::parse_str(&algo_str).map_err(LuaError::RuntimeError)?;
421            Ok(binary::hash(algo, raw_data.as_bytes()))
422        })?,
423    )?;
424    // -- crc32 --
425    /// Computes CRC32 for a binary string.
426    /// @param | raw_data | string | Raw binary data to checksum.
427    /// @return | integer | CRC32 checksum value.
428    tbl.set(
429        "crc32",
430        lua.create_function(|_, raw_data: LuaString| Ok(binary::crc32(raw_data.as_bytes())))?,
431    )?;
432    // -- newByteData --
433    /// Creates ByteData from a size or string.
434    /// @param | value | any | Integer size for zeroed bytes, or string used as initial bytes.
435    /// @return | LByteData | New LByteData userdata.
436    tbl.set(
437        "newByteData",
438        lua.create_function(|lua, value: LuaValue| {
439            let bd = match value {
440                LuaValue::Integer(n) => ByteData::new(n.max(0) as usize),
441                LuaValue::Number(n) => ByteData::new(n.max(0.0) as usize),
442                LuaValue::String(s) => ByteData::from_string(
443                    s.to_str()
444                        .map_err(|e| LuaError::RuntimeError(e.to_string()))?,
445                ),
446                _ => {
447                    return Err(LuaError::RuntimeError(
448                        "newByteData expects a number (size) or string".to_string(),
449                    ))
450                }
451            };
452            lua.create_userdata(bd)
453        })?,
454    )?;
455    // -- newDataView --
456    /// Creates a DataView over a binary string slice.
457    /// @param | raw | string | Binary byte string backing the view.
458    /// @param | offset | integer? | Optional zero-based start offset; defaults to zero.
459    /// @param | size | integer? | Optional view size in bytes; defaults to the remaining bytes.
460    /// @return | LDataView | New data view handle.
461    tbl.set(
462        "newDataView",
463        lua.create_function(
464            |lua, (raw, offset, size): (LuaString, Option<usize>, Option<usize>)| {
465                let bytes: Arc<Vec<u8>> = Arc::new(raw.as_bytes().to_vec());
466                let total = bytes.len();
467                let off = offset.unwrap_or(0);
468                let sz = size.unwrap_or_else(|| total.saturating_sub(off));
469                let dv = DataView::new_slice(bytes, off, sz).map_err(LuaError::RuntimeError)?;
470                lua.create_userdata(LuaDataView::new(dv))
471            },
472        )?,
473    )?;
474    // -- write --
475    /// Writes binary values into a byte string using a format string.
476    /// @param | fmt | string | Binary writer format string.
477    /// @param | ... | any | Values to write according to the format.
478    /// @return | string | Binary byte string containing written values.
479    tbl.set(
480        "write",
481        lua.create_function(|lua, (fmt, vals): (String, LuaMultiValue)| {
482            let bvs = lua_values_to_bin(vals);
483            let bd = binary::bin_write(&fmt, &bvs).map_err(LuaError::RuntimeError)?;
484            lua.create_string(bd.as_bytes())
485        })?,
486    )?;
487    // -- read --
488    /// Reads binary values from a byte string using a format string.
489    /// @param | fmt | string | Binary reader format string.
490    /// @param | raw | string | Binary byte string to read.
491    /// @param | offset | integer? | Optional zero-based byte offset; defaults to zero.
492    /// @return | LuaValue | Values read from the byte string.
493    tbl.set(
494        "read",
495        lua.create_function(
496            |lua, (fmt, raw, offset): (String, LuaString, Option<usize>)| {
497                let (bvs, _) = binary::bin_read(&fmt, raw.as_bytes(), offset.unwrap_or(0))
498                    .map_err(LuaError::RuntimeError)?;
499                let lv = bin_values_to_lua(lua, bvs)?;
500                Ok(LuaMultiValue::from_vec(lv))
501            },
502        )?,
503    )?;
504    // -- size --
505    /// Measures fixed byte size for a binary format string.
506    /// @param | fmt | string | Binary format string to measure.
507    /// @return | integer | Fixed byte size for the format.
508    tbl.set(
509        "size",
510        lua.create_function(|_, fmt: String| {
511            binary::bin_measure_size(&fmt).map_err(LuaError::RuntimeError)
512        })?,
513    )?;
514    // -- parseToml --
515    /// Parses TOML text into Lua tables and scalar values.
516    /// @param | text | string | TOML document text.
517    /// @return | table | Lua representation of the TOML document.
518    tbl.set(
519        "parseToml",
520        lua.create_function(|lua, text: String| {
521            let value = serial::parse_toml(&text).map_err(LuaError::RuntimeError)?;
522            toml_value_to_lua(lua, &value)
523        })?,
524    )?;
525    // -- encodeToml --
526    /// Encodes a Lua table into a TOML document string.
527    /// @param | tbl | table | Lua table to encode as TOML.
528    /// @return | string | TOML document text.
529    tbl.set(
530        "encodeToml",
531        lua.create_function(|_, tbl: LuaTable| {
532            let value = lua_table_to_toml_value(&LuaValue::Table(tbl))?;
533            serial::encode_toml(&value).map_err(LuaError::RuntimeError)
534        })?,
535    )?;
536    // -- newRingBuffer --
537    /// Creates a fixed-capacity ring buffer for Lua values.
538    /// @param | capacity | integer | Maximum value count; must be greater than zero.
539    /// @return | LRingBuffer | New ring buffer handle.
540    tbl.set(
541        "newRingBuffer",
542        lua.create_function(|_, capacity: usize| {
543            if capacity == 0 {
544                return Err(LuaError::RuntimeError(
545                    "newRingBuffer: capacity must be greater than 0".to_string(),
546                ));
547            }
548            Ok(LuaRingBuffer {
549                inner: VecDeque::with_capacity(capacity),
550                capacity,
551            })
552        })?,
553    )?;
554    // -- toMsgPack --
555    /// Encodes a Lua value into the current structured binary interchange payload.
556    /// @param | value | any | Lua value to encode through the serial table converter.
557    /// @return | string | Encoded binary payload.
558    tbl.set(
559        "toMsgPack",
560        lua.create_function(|lua, value: LuaValue| {
561            let serial_val =
562                crate::serialize::lua_table::from_lua(&value).map_err(LuaError::external)?;
563            let json_val = serial_value_to_json(&serial_val);
564            let bytes = crate::serialize::encode_json(&json_val).map_err(LuaError::RuntimeError)?;
565            lua.create_string(&bytes)
566        })?,
567    )?;
568    // -- fromMsgPack --
569    /// Decodes a structured binary interchange payload back into Lua values.
570    /// @param | bytes | string | Encoded binary payload.
571    /// @return | LuaValue | Decoded Lua value.
572    tbl.set(
573        "fromMsgPack",
574        lua.create_function(|lua, bytes: LuaString| {
575            let raw: &[u8] = bytes.as_bytes();
576            let json_val = crate::serialize::decode_json(raw).map_err(LuaError::RuntimeError)?;
577            let serial = json_value_to_serial(&json_val);
578            crate::serialize::lua_table::to_lua(lua, &serial)
579        })?,
580    )?;
581    // -- newWriter --
582    /// Creates an empty binary data writer.
583    /// @return | LDataWriter | New data writer handle.
584    tbl.set(
585        "newWriter",
586        lua.create_function(|lua, ()| {
587            lua.create_userdata(LuaDataWriter {
588                inner: DataWriter::new(),
589            })
590        })?,
591    )?;
592    /// Performs the 'binary' operation.
593    luna.set("binary", tbl.clone())?;
594    /// Backward-compatible alias so existing `lurek.data` scripts still work.
595    luna.set("data", tbl)?;
596    Ok(())
597}
598/// Converts a serial Lua-table value into JSON for binary payload encoding.
599fn serial_value_to_json(sv: &crate::serialize::lua_table::SerialValue) -> serde_json::Value {
600    use crate::serialize::lua_table::SerialValue;
601    match sv {
602        SerialValue::Null => serde_json::Value::Null,
603        SerialValue::Bool(b) => serde_json::Value::Bool(*b),
604        SerialValue::Int(n) => serde_json::Value::Number((*n).into()),
605        SerialValue::Float(f) => serde_json::Number::from_f64(*f)
606            .map(serde_json::Value::Number)
607            .unwrap_or(serde_json::Value::Null),
608        SerialValue::Str(s) => serde_json::Value::String(s.clone()),
609        SerialValue::Seq(arr) => {
610            serde_json::Value::Array(arr.iter().map(serial_value_to_json).collect())
611        }
612        SerialValue::Map(map) => {
613            let obj: serde_json::Map<String, serde_json::Value> = map
614                .iter()
615                .map(|(k, v)| (k.clone(), serial_value_to_json(v)))
616                .collect();
617            serde_json::Value::Object(obj)
618        }
619    }
620}
621/// Converts JSON into the serial Lua-table value representation.
622fn json_value_to_serial(val: &serde_json::Value) -> crate::serialize::lua_table::SerialValue {
623    use crate::serialize::lua_table::SerialValue;
624    match val {
625        serde_json::Value::Null => SerialValue::Null,
626        serde_json::Value::Bool(b) => SerialValue::Bool(*b),
627        serde_json::Value::Number(n) => {
628            if let Some(i) = n.as_i64() {
629                SerialValue::Int(i)
630            } else if let Some(f) = n.as_f64() {
631                SerialValue::Float(f)
632            } else {
633                SerialValue::Int(0)
634            }
635        }
636        serde_json::Value::String(s) => SerialValue::Str(s.clone()),
637        serde_json::Value::Array(arr) => {
638            SerialValue::Seq(arr.iter().map(json_value_to_serial).collect())
639        }
640        serde_json::Value::Object(obj) => {
641            let mut map = LuaIndexMap::new();
642            for (k, v) in obj {
643                map.insert(k.clone(), json_value_to_serial(v));
644            }
645            SerialValue::Map(map)
646        }
647    }
648}

863/// Exposes byte-buffer inspection and bit editing methods to Lua.
864impl mlua::UserData for ByteData {
865    fn add_methods<'lua, M: mlua::UserDataMethods<'lua, Self>>(methods: &mut M) {
866        // -- getSize --
867        /// Returns the byte buffer length in bytes.
868        /// @return | integer | Buffer length in bytes.
869        methods.add_method("getSize", |_, this, ()| Ok(this.len()));
870        // -- getString --
871        /// Returns the byte buffer as a string.
872        /// @return | string | Byte buffer contents as a Lua string.
873        methods.add_method("getString", |_, this, ()| Ok(this.get_string()));
874        // -- getByte --
875        /// Reads one byte at a zero-based offset.
876        /// @param | offset | integer | Zero-based byte offset.
877        /// @return | integer | Byte value from 0 to 255.
878        methods.add_method("getByte", |_, this, offset: usize| {
879            this.get_byte(offset).ok_or_else(|| {
880                LuaError::RuntimeError(format!(
881                    "Offset {} out of bounds (size {})",
882                    offset,
883                    this.len()
884                ))
885            })
886        });
887        // -- setByte --
888        /// Writes one byte at a zero-based offset inside the buffer.
889        /// @param | offset | integer | Zero-based byte offset.
890        /// @param | value | integer | Byte value from 0 to 255.
891        methods.add_method_mut("setByte", |_, this, (offset, value): (usize, u8)| {
892            if this.set_byte(offset, value) {
893                Ok(())
894            } else {
895                Err(LuaError::RuntimeError(format!(
896                    "Offset {} out of bounds (size {})",
897                    offset,
898                    this.len()
899                )))
900            }
901        });
902        // -- clone --
903        /// Returns a deep copy of the entire byte buffer.
904        /// @return | LByteData | New LByteData userdata containing copied bytes.
905        methods.add_method("clone", |lua, this, ()| {
906            lua.create_userdata(this.clone_data())
907        });
908        // -- setBit --
909        /// Sets or clears one bit inside a byte at the given offset.
910        /// @param | byte_offset | integer | Zero-based byte offset.
911        /// @param | bit_offset | integer | Bit offset from 0 to 7 inside the byte.
912        /// @param | value | boolean | True to set the bit, false to clear it.
913        methods.add_method_mut(
914            "setBit",
915            |_, this, (byte_offset, bit_offset, value): (usize, u8, bool)| {
916                if byte_offset >= this.len() {
917                    return Err(LuaError::RuntimeError(format!(
918                        "lurek.data: setBit byte_offset {} out of range (size={})",
919                        byte_offset,
920                        this.len()
921                    )));
922                }
923                if bit_offset > 7 {
924                    return Err(LuaError::RuntimeError(format!(
925                        "lurek.data: setBit bit_offset {} out of range [0..7]",
926                        bit_offset
927                    )));
928                }
929                let current = this.get_byte(byte_offset).unwrap();
930                let new_val = if value {
931                    current | (1u8 << bit_offset)
932                } else {
933                    current & !(1u8 << bit_offset)
934                };
935                this.set_byte(byte_offset, new_val);
936                Ok(())
937            },
938        );
939        // -- getBit --
940        /// Reads one bit inside a byte at the given offsets.
941        /// @param | byte_offset | integer | Zero-based byte offset.
942        /// @param | bit_offset | integer | Bit offset from 0 to 7 inside the byte.
943        /// @return | boolean | True when the bit is set.
944        methods.add_method(
945            "getBit",
946            |_, this, (byte_offset, bit_offset): (usize, u8)| {
947                if byte_offset >= this.len() {
948                    return Err(LuaError::RuntimeError(format!(
949                        "lurek.data: getBit byte_offset {} out of range (size={})",
950                        byte_offset,
951                        this.len()
952                    )));
953                }
954                if bit_offset > 7 {
955                    return Err(LuaError::RuntimeError(format!(
956                        "lurek.data: getBit bit_offset {} out of range [0..7]",
957                        bit_offset
958                    )));
959                }
960                let byte = this.get_byte(byte_offset).unwrap();
961                Ok((byte >> bit_offset) & 1 == 1)
962            },
963        );
964        // -- readBits --
965        /// Reads up to 32 bits starting at a byte and bit offset.
966        /// @param | byte_offset | integer | Zero-based byte offset.
967        /// @param | bit_offset | integer | Bit offset from 0 to 7 inside the starting byte.
968        /// @param | count | integer | Number of bits to read, from 1 to 32.
969        /// @return | integer | Unsigned integer containing the requested bits.
970        methods.add_method(
971            "readBits",
972            |_, this, (byte_offset, bit_offset, count): (usize, u8, u8)| {
973                if count == 0 || count > 32 {
974                    return Err(LuaError::RuntimeError(format!(
975                        "lurek.data: readBits count {} out of range [1..32]",
976                        count
977                    )));
978                }
979                let mut result: u32 = 0;
980                for i in 0..count {
981                    let total_bit = bit_offset as usize + i as usize;
982                    let b = byte_offset + total_bit / 8;
983                    let bit = (total_bit % 8) as u8;
984                    if b >= this.len() {
985                        return Err(LuaError::RuntimeError(format!(
986                            "lurek.data: readBits reads beyond buffer end (size={})",
987                            this.len()
988                        )));
989                    }
990                    let byte = this.get_byte(b).unwrap();
991                    if (byte >> bit) & 1 == 1 {
992                        result |= 1u32 << i;
993                    }
994                }
995                Ok(result)
996            },
997        );
998        // -- type --
999        /// Returns the type name of this object for runtime type-checking.
1000        /// @return | string | Always returns "LByteData".
1001        methods.add_method("type", |_, _, ()| Ok("LByteData"));
1002        // -- typeOf --
1003        /// Checks whether this object matches the given type name.
1004        /// @param | name | string | Type name to check (e.g. "LByteData" or "LObject").
1005        /// @return | boolean | True if this object matches the given type.
1006        methods.add_method("typeOf", |_, _, name: String| {
1007            Ok(name == "LByteData" || name == "LObject")
1008        });
1009    }
