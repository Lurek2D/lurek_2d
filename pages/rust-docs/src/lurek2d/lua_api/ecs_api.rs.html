1//! `lurek.ecs` -- Entity-component-system bindings for creating universes, managing entities and components, running Lua systems, querying tags and blueprints, serializing state, and tracking hierarchy and relation data.
2
3use super::SharedState;
4use crate::ecs::Universe;
5use mlua::prelude::*;
6use std::cell::RefCell;
7use std::collections::HashMap;
8use std::rc::Rc;
9#[derive(Clone)]
10/// Lua-side handle for one ECS universe.
11pub struct LuaUniverse {
12    /// Shared ECS universe storage used by all cloned Lua handles.
13    inner: Rc<RefCell<Universe>>,
14    /// Registered callbacks keyed by component name for add events.
15    add_observers: Rc<RefCell<HashMap<String, Vec<LuaRegistryKey>>>>,
16    /// Registered callbacks keyed by component name for remove events.
17    remove_observers: Rc<RefCell<HashMap<String, Vec<LuaRegistryKey>>>>,
18}
19/// Provides Lua methods for ECS entity, component, system, tag, blueprint, hierarchy, observer, and relation operations.
20impl LuaUserData for LuaUniverse {
21    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
22        // -- spawn --
23        /// Creates a new entity in this universe.
24        /// @return | integer | Numeric entity id for the spawned entity.
25        methods.add_method("spawn", |_, this, ()| {
26            Ok(this.inner.borrow_mut().spawn().raw())
27        });
28        // -- kill --
29        /// Deletes an entity and removes its components from this universe.
30        /// @param | id | integer | Entity id to delete.
31        methods.add_method("kill", |lua, this, id: u32| {
32            this.inner
33                .borrow_mut()
34                .kill(crate::ecs::EntityId(id), lua)
35        });
36        // -- isAlive --
37        /// Returns whether an entity id currently exists in this universe.
38        /// @param | id | integer | Entity id to test.
39        /// @return | boolean | True when the entity is alive.
40        methods.add_method("isAlive", |_, this, id: u32| {
41            Ok(this.inner.borrow().is_alive(id))
42        });
43        // -- set --
44        /// Stores or replaces a component value on an entity.
45        /// @param | id | integer | Entity id that receives the component.
46        /// @param | name | string | Component name.
47        /// @param | value | any | Lua value stored as the component payload.
48        methods.add_method(
49            "set",
50            |lua, this, (id, name, value): (u32, String, LuaValue)| {
51                this.inner.borrow_mut().set_component(lua, id, &name, value)
52            },
53        );
54        // -- get --
55        /// Returns a component value from an entity.
56        /// @param | id | integer | Entity id to read.
57        /// @param | name | string | Component name to read.
58        /// @return | table|number|string|boolean|nil | Stored component value, or nil when the entity does not have that component.
59        methods.add_method("get", |lua, this, (id, name): (u32, String)| {
60            this.inner.borrow().get_component(lua, id, &name)
61        });
62        // -- has --
63        /// Returns whether an entity has a named component.
64        /// @param | id | integer | Entity id to inspect.
65        /// @param | name | string | Component name to check.
66        /// @return | boolean | True when the component exists on the entity.
67        methods.add_method("has", |lua, this, (id, name): (u32, String)| {
68            this.inner.borrow().has_component(lua, id, &name)
69        });
70        // -- remove --
71        /// Removes a named component from an entity.
72        /// @param | id | integer | Entity id to mutate.
73        /// @param | name | string | Component name to remove.
74        methods.add_method("remove", |lua, this, (id, name): (u32, String)| {
75            this.inner.borrow_mut().remove_component(lua, id, &name)
76        });
77        // -- getComponents --
78        /// Returns component names currently stored on an entity.
79        /// @param | id | integer | Entity id to inspect.
80        /// @return | string[] | Component name strings.
81        methods.add_method("getComponents", |lua, this, id: u32| {
82            this.inner.borrow().get_component_names(lua, id)
83        });
84        // -- query --
85        /// Returns entities that have all component names passed as varargs.
86        /// @param | ... | string | Component names that every returned entity must have.
87        /// @return | integer[] | Array table of matching entity ids.
88        methods.add_method("query", |lua, this, args: LuaMultiValue| {
89            let names: Vec<String> = args
90                .into_iter()
91                .filter_map(|v| match v {
92                    LuaValue::String(s) => s.to_str().ok().map(|s| s.to_string()),
93                    _ => None,
94                })
95                .collect();
96            this.inner.borrow().query(lua, &names)
97        });
98        // -- each --
99        /// Iterates entities with one component and calls a Lua callback for each match.
100        /// @param | name | string | Component name used to select entities.
101        /// @param | callback | function | Callback invoked by the ECS backend for each matching entity.
102        methods.add_method(
103            "each",
104            |lua, this, (name, callback): (String, LuaFunction)| {
105                this.inner.borrow().each(lua, &name, callback)
106            },
107        );
108        // -- getEntities --
109        /// Returns all live entity ids in this universe.
110        /// @return | integer[] | Array table of live entity ids.
111        methods.add_method("getEntities", |_, this, ()| {
112            Ok(this.inner.borrow().get_entities())
113        });
114        // -- getEntityCount --
115        /// Returns the number of live entities in this universe.
116        /// @return | integer | Live entity count.
117        methods.add_method("getEntityCount", |_, this, ()| {
118            Ok(this.inner.borrow().get_entity_count())
119        });
120        // -- addSystem --
121        /// Registers a Lua system table with optional phase, priority, name, and dependency metadata.
122        /// @param | system | table | System table containing update, render, draw, or event methods.
123        /// @param | opts | table? | Optional table with priority, phase, name, and after fields.
124        methods.add_method(
125            "addSystem",
126            |lua, this, (system, opts): (LuaTable, Option<LuaTable>)| {
127                let priority = opts
128                    .as_ref()
129                    .and_then(|o| o.get::<_, i32>("priority").ok())
130                    .unwrap_or(0);
131                let phase = opts
132                    .as_ref()
133                    .and_then(|o| o.get::<_, String>("phase").ok())
134                    .unwrap_or_default();
135                let name = opts
136                    .as_ref()
137                    .and_then(|o| o.get::<_, String>("name").ok())
138                    .unwrap_or_default();
139                let deps: Vec<String> = opts
140                    .as_ref()
141                    .and_then(|o| o.get::<_, LuaTable>("after").ok())
142                    .map(|t| {
143                        t.sequence_values::<String>()
144                            .filter_map(|r| r.ok())
145                            .collect()
146                    })
147                    .unwrap_or_default();
148                this.inner
149                    .borrow_mut()
150                    .add_system(lua, system, priority, phase, name, deps)
151            },
152        );
153        // -- removeSystem --
154        /// Removes a previously registered Lua system table.
155        /// @param | system | table | System table to remove from this universe.
156        methods.add_method("removeSystem", |lua, this, system: LuaTable| {
157            this.inner.borrow_mut().remove_system(lua, system)
158        });
159        // -- update --
160        /// Runs registered update-phase systems with a frame delta.
161        /// @param | dt | number | Frame delta time in seconds.
162        methods.add_method("update", |lua, this, dt: f64| {
163            let count = this.inner.borrow().get_system_count(lua)?;
164            if count == 0 {
165                return Ok(());
166            }
167            let order = this
168                .inner
169                .borrow()
170                .get_sorted_system_indices_for_phase("update");
171            let store = this.inner.borrow().get_system_store(lua)?;
172            let world = this.clone();
173            for i in order {
174                let system: LuaTable = store.get(i)?;
175                if let Ok(func) = system.get::<_, LuaFunction>("update") {
176                    func.call::<_, ()>((system.clone(), world.clone(), dt))?;
177                }
178            }
179            Ok(())
180        });
181        // -- render --
182        /// Runs registered render-phase systems using their render or draw callbacks.
183        methods.add_method("render", |lua, this, ()| {
184            let count = this.inner.borrow().get_system_count(lua)?;
185            if count == 0 {
186                return Ok(());
187            }
188            let order = this
189                .inner
190                .borrow()
191                .get_sorted_system_indices_for_phase("render");
192            let store = this.inner.borrow().get_system_store(lua)?;
193            let world = this.clone();
194            for i in order {
195                let system: LuaTable = store.get(i)?;
196                let func = system
197                    .get::<_, LuaFunction>("render")
198                    .or_else(|_| system.get::<_, LuaFunction>("draw"));
199                if let Ok(f) = func {
200                    f.call::<_, ()>((system.clone(), world.clone()))?;
201                }
202            }
203            Ok(())
204        });
205        // -- emit --
206        /// Calls matching event-named functions on registered systems.
207        /// @param | event | string | Function name looked up on each system table.
208        /// @param | ... | any | Extra values forwarded after the system and universe arguments.
209        methods.add_method("emit", |lua, this, args: LuaMultiValue| {
210            let mut args_iter = args.into_iter();
211            let event: String = match args_iter.next() {
212                Some(LuaValue::String(s)) => s.to_str()?.to_string(),
213                _ => {
214                    return Err(LuaError::runtime(
215                        "emit() requires event name as first argument",
216                    ))
217                }
218            };
219            let extra_args: Vec<LuaValue> = args_iter.collect();
220            let count = this.inner.borrow().get_system_count(lua)?;
221            if count == 0 {
222                return Ok(());
223            }
224            let order = this.inner.borrow().get_sorted_system_indices_all();
225            let store = this.inner.borrow().get_system_store(lua)?;
226            let world = this.clone();
227            for i in order {
228                let system: LuaTable = store.get(i)?;
229                if let Ok(func) = system.get::<_, LuaFunction>(event.as_str()) {
230                    let mut call_args = Vec::with_capacity(2 + extra_args.len());
231                    call_args.push(LuaValue::Table(system.clone()));
232                    call_args.push(LuaValue::UserData(lua.create_userdata(world.clone())?));
233                    call_args.extend(extra_args.iter().cloned());
234                    func.call::<_, ()>(LuaMultiValue::from_vec(call_args))?;
235                }
236            }
237            Ok(())
238        });
239        // -- getSystemCount --
240        /// Returns the number of registered systems.
241        /// @return | integer | Registered system count.
242        methods.add_method("getSystemCount", |lua, this, ()| {
243            this.inner.borrow().get_system_count(lua)
244        });
245        // -- updatePhase --
246        /// Runs registered systems assigned to a named phase.
247        /// @param | phase | string | System phase name to run.
248        /// @param | dt | number | Frame delta time in seconds.
249        methods.add_method("updatePhase", |lua, this, (phase, dt): (String, f64)| {
250            let count = this.inner.borrow().get_system_count(lua)?;
251            if count == 0 {
252                return Ok(());
253            }
254            let order = this
255                .inner
256                .borrow()
257                .get_sorted_system_indices_for_phase(&phase);
258            let store = this.inner.borrow().get_system_store(lua)?;
259            let world = this.clone();
260            for i in order {
261                let system: LuaTable = store.get(i)?;
262                if let Ok(func) = system.get::<_, LuaFunction>("update") {
263                    func.call::<_, ()>((system.clone(), world.clone(), dt))?;
264                }
265            }
266            Ok(())
267        });
268        // -- getDirtyEntities --
269        /// Returns entities marked dirty by recent ECS mutations.
270        /// @return | integer[] | Array table of dirty entity ids.
271        methods.add_method("getDirtyEntities", |lua, this, ()| {
272            let ids = this.inner.borrow().get_dirty_entities();
273            let t = lua.create_table()?;
274            for (i, id) in ids.iter().enumerate() {
275                t.set(i + 1, *id)?;
276            }
277            Ok(LuaValue::Table(t))
278        });
279        // -- queryMulti --
280        /// Iterates entities that have all component names from a table.
281        /// @param | names_table | table | Array table of component names.
282        /// @param | callback | function | Callback invoked by the ECS backend for each matching entity.
283        methods.add_method(
284            "queryMulti",
285            |lua, this, (names_table, callback): (LuaTable, LuaFunction)| {
286                let mut names: Vec<String> = Vec::new();
287                for pair in names_table.sequence_values::<String>() {
288                    names.push(pair?);
289                }
290                this.inner.borrow().query_multi(lua, &names, callback)
291            },
292        );
293        // -- snapshot --
294        /// Serializes this universe into a Lua table snapshot.
295        /// @return | table | Snapshot table containing entities and component state.
296        /// @field | added_components | table | Added components.
297        /// @field | removed_components | table | Removed components.
298        /// @field | deleted_entities | integer[] | Deleted entity ids.
299        /// @field | dirty_entities | integer[] | Dirty entity ids.
300        methods.add_method("snapshot", |lua, this, ()| {
301            this.inner.borrow().serialize_to_table(lua)
302        });
303        // -- applySnapshot --
304        /// Replaces this universe state from a Lua table snapshot.
305        /// @param | snapshot | table | Snapshot table previously produced by `snapshot` or `serialize`.
306        methods.add_method("applySnapshot", |lua, this, snapshot: LuaTable| {
307            this.inner
308                .borrow_mut()
309                .deserialize_from_table(lua, snapshot)
310        });
311        // -- takeSnapshotDiff --
312        /// Returns and clears accumulated ECS snapshot diff data.
313        /// @return | table | Diff table with added_components, removed_components, deleted_entities, and dirty_entities arrays.
314        /// @field | added_components | table | Array of {entity_id, name} tables.
315        /// @field | removed_components | table | Array of {entity_id, name} tables.
316        /// @field | deleted_entities | integer[] | Deleted entity ids.
317        /// @field | dirty_entities | integer[] | Modified entity ids.
318        methods.add_method("takeSnapshotDiff", |lua, this, ()| {
319            let diff = this.inner.borrow_mut().take_snapshot_diff();
320            let out = lua.create_table()?;
321            let added = lua.create_table()?;
322            for (i, (id, name)) in diff.added_components.iter().enumerate() {
323                let entry = lua.create_table()?;
324                /// Performs the 'entity_id' operation.
325                entry.set("entity_id", *id)?;
326                /// Performs the 'name' operation.
327                entry.set("name", name.clone())?;
328                added.set(i + 1, entry)?;
329            }
330            /// Performs the 'added_components' operation.
331            out.set("added_components", added)?;
332            let removed = lua.create_table()?;
333            for (i, (id, name)) in diff.removed_components.iter().enumerate() {
334                let entry = lua.create_table()?;
335                /// Performs the 'entity_id' operation.
336                entry.set("entity_id", *id)?;
337                /// Performs the 'name' operation.
338                entry.set("name", name.clone())?;
339                removed.set(i + 1, entry)?;
340            }
341            /// Performs the 'removed_components' operation.
342            out.set("removed_components", removed)?;
343            let deleted = lua.create_table()?;
344            for (i, id) in diff.deleted_entities.iter().enumerate() {
345                deleted.set(i + 1, *id)?;
346            }
347            /// Performs the 'deleted_entities' operation.
348            out.set("deleted_entities", deleted)?;
349            let dirty = lua.create_table()?;
350            for (i, id) in diff.dirty_entities.iter().enumerate() {
351                dirty.set(i + 1, *id)?;
352            }
353            /// Performs the 'dirty_entities' operation.
354            out.set("dirty_entities", dirty)?;
355            Ok(out)
356        });

541        // -- removeBlueprint --
542        /// Removes a named blueprint from this universe.
543        /// @param | name | string | Blueprint name to remove.
544        /// @return | boolean | True when a blueprint was removed.
545        methods.add_method("removeBlueprint", |lua, this, name: String| {
546            this.inner.borrow().remove_blueprint(lua, &name)
547        });
548        // -- listBlueprints --
549        /// Returns names of all registered blueprints.
550        /// @return | string[] | Blueprint names.
551        methods.add_method("listBlueprints", |lua, this, ()| {
552            this.inner.borrow().list_blueprints(lua)
553        });
554        // -- getBlueprintComponents --
555        /// Returns the component table stored for a blueprint.
556        /// @param | name | string | Blueprint name to inspect.
557        /// @return | table | Blueprint component table.
558        methods.add_method("getBlueprintComponents", |lua, this, name: String| {
559            this.inner.borrow().get_blueprint_components(lua, &name)
560        });
561        // -- setParent --
562        /// Sets or clears the parent entity for a child entity.
563        /// @param | child_id | integer | Entity id whose parent changes.
564        /// @param | parent_id | integer? | Parent entity id, or nil to clear the parent.
565        methods.add_method(
566            "setParent",
567            |_, this, (child_id, parent_id): (u32, Option<u32>)| {
568                this.inner.borrow_mut().set_parent(child_id, parent_id);
569                Ok(())
570            },
571        );
572        // -- getParent --
573        /// Returns the parent entity id for a child entity.
574        /// @param | child_id | integer | Entity id whose parent is read.
575        /// @return | integer | Parent entity id, or nil when the entity has no parent.
576        methods.add_method("getParent", |_, this, child_id: u32| {
577            Ok(this.inner.borrow().get_parent(child_id))
578        });
579        // -- getChildren --
580        /// Returns child entity ids for a parent entity.
581        /// @param | parent_id | integer | Parent entity id to inspect.
582        /// @return | integer[] | Array table of child entity ids.
583        methods.add_method("getChildren", |_, this, parent_id: u32| {
584            Ok(this.inner.borrow().get_children(parent_id))
585        });
586        // -- killRecursive --
587        /// Deletes an entity and all descendant entities in its hierarchy.
588        /// @param | id | integer | Root entity id to delete.
589        methods.add_method("killRecursive", |lua, this, id: u32| {
590            this.inner.borrow_mut().kill_recursive(id, lua)
591        });
592        // -- queryNot --
593        /// Returns entities that include one component set and exclude another component set.
594        /// @param | with_tbl | table | Array table of required component names.
595        /// @param | without_tbl | table | Array table of forbidden component names.
596        /// @return | integer[] | Array table of matching entity ids.
597        methods.add_method(
598            "queryNot",
599            |lua, this, (with_tbl, without_tbl): (LuaTable, LuaTable)| {
600                let with_names: Vec<String> = with_tbl
601                    .sequence_values::<String>()
602                    .collect::<LuaResult<_>>()?;
603                let without_names: Vec<String> = without_tbl
604                    .sequence_values::<String>()
605                    .collect::<LuaResult<_>>()?;
606                this.inner
607                    .borrow()
608                    .query_not(lua, &with_names, &without_names)
609            },
610        );
611        // -- serialize --
612        /// Serializes this universe into a Lua table snapshot.
613        /// @return | table | Snapshot table containing entities and component state.
614        /// @field | entities | integer[] | Array of entity ids.
615        /// @field | components | table | Map of entity id to component data tables.
616        methods.add_method("serialize", |lua, this, ()| {
617            this.inner.borrow().serialize_to_table(lua)
618        });
619        // -- deserialize --
620        /// Replaces this universe state from a serialized Lua snapshot.
621        /// @param | snapshot | table | Snapshot table previously produced by `serialize` or `snapshot`.
622        methods.add_method("deserialize", |lua, this, snapshot: LuaTable| {
623            this.inner
624                .borrow_mut()
625                .deserialize_from_table(lua, snapshot)
626        });
627        // -- onComponentAdded --
628        /// Registers a callback for queued component-add events with a given component name.
629        /// @param | name | string | Component name whose add events are observed.
630        /// @param | cb | function | Callback receiving entity id and component name.
631        methods.add_method(
632            "onComponentAdded",
633            |lua, this, (name, cb): (String, LuaFunction)| {
634                let key = lua.create_registry_value(cb)?;
635                this.add_observers
636                    .borrow_mut()
637                    .entry(name)
638                    .or_default()
639                    .push(key);
640                Ok(())
641            },
642        );
643        // -- onComponentRemoved --
644        /// Registers a callback for queued component-remove events with a given component name.
645        /// @param | name | string | Component name whose remove events are observed.
646        /// @param | cb | function | Callback receiving entity id and component name.
647        methods.add_method(
648            "onComponentRemoved",
649            |lua, this, (name, cb): (String, LuaFunction)| {
650                let key = lua.create_registry_value(cb)?;
651                this.remove_observers
652                    .borrow_mut()
653                    .entry(name)
654                    .or_default()
655                    .push(key);
656                Ok(())
657            },
658        );
659        // -- flushObservers --
660        /// Delivers queued component add and remove events to registered observer callbacks.
661        methods.add_method("flushObservers", |lua, this, ()| {
662            let (add_evs, remove_evs) = this.inner.borrow_mut().take_component_events();
663            for (id, name) in &add_evs {
664                if let Some(keys) = this
665                    .add_observers
666                    .borrow()
667                    .get(name.as_str())
668                    .map(|v| v.len())
669                {
670                    let _ = keys;
671                }
672                let keys_opt: Option<Vec<LuaFunction>> = {
673                    let obs = this.add_observers.borrow();
674                    obs.get(name.as_str()).map(|keys| {
675                        keys.iter()
676                            .filter_map(|k| lua.registry_value::<LuaFunction>(k).ok())
677                            .collect()
678                    })
679                };
680                if let Some(fns) = keys_opt {
681                    for f in fns {
682                        f.call::<_, ()>((*id, name.as_str()))?;
683                    }
684                }
685            }
686            for (id, name) in &remove_evs {
687                let keys_opt: Option<Vec<LuaFunction>> = {
688                    let obs = this.remove_observers.borrow();
689                    obs.get(name.as_str()).map(|keys| {
690                        keys.iter()
691                            .filter_map(|k| lua.registry_value::<LuaFunction>(k).ok())
692                            .collect()
693                    })
694                };
695                if let Some(fns) = keys_opt {
696                    for f in fns {
697                        f.call::<_, ()>((*id, name.as_str()))?;
698                    }
699                }
700            }
701            Ok(())
702        });
703        // -- spawnBulk --
704        /// Spawns multiple entities from a blueprint using shared optional overrides.
705        /// @param | name | string | Blueprint name to instantiate.
706        /// @param | count | integer | Number of entities to spawn.
707        /// @param | overrides | table? | Optional component overrides applied to each spawned entity.
708        /// @return | integer[] | Array table of spawned entity ids.
709        methods.add_method(
710            "spawnBulk",
711            |lua, this, (name, count, overrides): (String, usize, Option<LuaTable>)| {
712                this.inner
713                    .borrow_mut()
714                    .spawn_bulk(lua, &name, count, overrides)
715            },
716        );
717        // -- addRelation --
718        /// Adds a named directed relation from one entity to another.
719        /// @param | from | integer | Source entity id.
720        /// @param | name | string | Relation name.
721        /// @param | to | integer | Target entity id.
722        methods.add_method(
723            "addRelation",
724            |_, this, (from, name, to): (u32, String, u32)| {
725                this.inner
726                    .borrow_mut()
727                    .relationships
728                    .add_link(from, &name, to);
729                Ok(())
730            },
731        );
732        // -- getRelated --
733        /// Returns targets linked from an entity by a named relation.
734        /// @param | from | integer | Source entity id.
735        /// @param | name | string | Relation name.
736        /// @return | integer[] | Array table of related target entity ids.
737        methods.add_method("getRelated", |lua, this, (from, name): (u32, String)| {
738            let inner = this.inner.borrow();
739            let ids = inner.relationships.get_links(from, &name);
740            let tbl = lua.create_table()?;
741            for (i, id) in ids.iter().enumerate() {
742                tbl.set(i + 1, *id)?;
743            }
744            Ok(tbl)
745        });
746        // -- removeRelation --
747        /// Removes a named directed relation between two entities.
748        /// @param | from | integer | Source entity id.
749        /// @param | name | string | Relation name.
750        /// @param | to | integer | Target entity id.
751        methods.add_method(
752            "removeRelation",
753            |_, this, (from, name, to): (u32, String, u32)| {
754                this.inner
755                    .borrow_mut()
756                    .relationships
757                    .remove_link(from, &name, to);
758                Ok(())
759            },
760        );
761        // -- clearRelations --
762        /// Removes every target for one named relation from an entity.
763        /// @param | from | integer | Source entity id.
764        /// @param | name | string | Relation name to clear.
765        methods.add_method("clearRelations", |_, this, (from, name): (u32, String)| {
766            this.inner
767                .borrow_mut()
768                .relationships
769                .clear_links(from, &name);
770            Ok(())
771        });
772        // -- hasRelation --
773        /// Returns whether a named directed relation exists between two entities.
774        /// @param | from | integer | Source entity id.
775        /// @param | name | string | Relation name.
776        /// @param | to | integer | Target entity id.
777        /// @return | boolean | True when the relation exists.
778        methods.add_method(
779            "hasRelation",
780            |_, this, (from, name, to): (u32, String, u32)| {
781                Ok(this.inner.borrow().relationships.has_link(from, &name, to))
782            },
783        );
784        // -- type --
785        /// Returns the Lua-visible type name for this universe handle.
786        /// @return | string | The string `LUniverse`.
787        methods.add_method("type", |_, _, ()| Ok("LUniverse"));
788        // -- typeOf --
789        /// Returns whether this universe handle matches a supported type name.
790        /// @param | name | string | Type name to compare against `LUniverse` and `Object`.
791        /// @return | boolean | True when the supplied type name matches this handle.
792        methods.add_method("typeOf", |_, _, name: String| {
793            Ok(name == "LUniverse" || name == "LObject")
794        });
795    }
796}
797/// Registers the `lurek.ecs` API table with the Lua VM.
798pub fn register(lua: &Lua, lurek: &LuaTable, _state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
799    let tbl = lua.create_table()?;
800    // -- newUniverse --
801    /// Creates an empty ECS universe for entity, component, system, and relationship management.
802    /// @return | LUniverse | New universe handle.
803    tbl.set(
804        "newUniverse",
805        lua.create_function(|_, ()| {
806            Ok(LuaUniverse {
807                inner: Rc::new(RefCell::new(Universe::new())),
808                add_observers: Rc::new(RefCell::new(HashMap::new())),
809                remove_observers: Rc::new(RefCell::new(HashMap::new())),
810            })
811        })?,
812    )?;
813    /// The 'ecs' field value exposed to Lua scripts.
814    lurek.set("ecs", tbl)?;
815    Ok(())
816}
