15/// Lua userdata that controls cursor appearance and system cursor selection.
16struct LuaCursorManager {
17    inner: Rc<RefCell<CursorManager>>,
18}
19
20impl LuaUserData for LuaCursorManager {
21    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
22        /// Set the active cursor to a system cursor by name.
23        /// @param | name | string | System cursor name (arrow, hand, crosshair, ibeam, wait, no, etc.).
24        methods.add_method("setSystem", |_, this, name: String| {
25            let cursor = SystemCursor::from_name(&name)
26                .ok_or_else(|| LuaError::runtime(format!("unknown system cursor: {name}")))?;
27            this.inner.borrow_mut().set_system(cursor);
28            Ok(())
29        });
30
31        /// Set the active cursor to a custom image cursor.
32        /// @param | cursor | LuaCustomCursor | Custom cursor object.
33        methods.add_method("setCustom", |_, this, cursor: LuaAnyUserData| {
34            let cursor = cursor.borrow::<LuaCustomCursor>()?.clone();
35            this.inner.borrow_mut().set_custom(cursor.inner.borrow().clone());
36            Ok(())
37        });
38
39        /// Set the active cursor to an animated cursor.
40        /// @param | cursor | LuaAnimatedCursor | Animated cursor object.
41        methods.add_method("setAnimated", |_, this, cursor: LuaAnyUserData| {
42            let cursor = cursor.borrow::<LuaAnimatedCursor>()?.clone();
43            this.inner.borrow_mut().set_animated(cursor.inner.borrow().clone());
44            Ok(())
45        });
46
47        /// Set the current context for context-sensitive switching.
48        /// @param | ctx | string | Context name (default, raycaster, globe, tilemap, ui_button, etc.).
49        methods.add_method("setContext", |_, this, ctx: String| {
50            this.inner.borrow_mut().set_context(CursorContext::from_name(&ctx));
51            Ok(())
52        });
53
54        /// Add a context rule that maps a context to a system cursor.
55        /// @param | ctx | string | Context name.
56        /// @param | cursor_name | string | System cursor name.
57        methods.add_method("addRule", |_, this, (ctx, cursor_name): (String, String)| {
58            let cursor = SystemCursor::from_name(&cursor_name)
59                .ok_or_else(|| LuaError::runtime(format!("unknown system cursor: {cursor_name}")))?;
60            use crate::cursor::context::{ContextRule, CursorState};
61            this.inner.borrow_mut().add_rule(ContextRule {
62                context: CursorContext::from_name(&ctx),
63                cursor: CursorState::System(cursor),
64            });
65            Ok(())
66        });
67
68        /// Remove a context rule for this object.
69        /// @param | ctx | string | Context name to remove.
70        methods.add_method("removeRule", |_, this, ctx: String| {
71            this.inner.borrow_mut().remove_rule(&CursorContext::from_name(&ctx));
72            Ok(())
73        });
74
75        /// Update cursor state (call each frame).
76        /// @param | x | number | Cursor X position.
77        /// @param | y | number | Cursor Y position.
78        /// @param | dt | number | Delta time in seconds.
79        methods.add_method("update", |_, this, (x, y, dt): (f32, f32, f32)| {
80            this.inner.borrow_mut().update(x, y, dt);
81            Ok(())
82        });
83
84        /// Set cursor visibility for this object.
85        /// @param | visible | boolean | Whether the cursor is visible.
86        methods.add_method("setVisible", |_, this, visible: bool| {
87            this.inner.borrow_mut().set_visible(visible);
88            Ok(())
89        });
90
91        /// Get cursor visibility for this object.
92        /// @return | boolean | Whether the cursor is visible.
93        methods.add_method("isVisible", |_, this, ()| {
94            Ok(this.inner.borrow().is_visible())
95        });
96
97        /// Lock the cursor position using the system grab mode.
98        /// @param | locked | boolean | Whether the cursor is locked.
99        methods.add_method("setLocked", |_, this, locked: bool| {
100            this.inner.borrow_mut().set_locked(locked);
101            Ok(())
102        });
103
104        /// Get cursor lock state for this object.
105        /// @return | boolean | Whether the cursor is locked.
106        methods.add_method("isLocked", |_, this, ()| {
107            Ok(this.inner.borrow().is_locked())
108        });
109
110        /// Get cursor position for this object.
111        /// @return | number | X position.
112        /// @return | number | Y position.
113        methods.add_method("getPosition", |_, this, ()| {
114            let (x, y) = this.inner.borrow().position();
115            Ok((x, y))
116        });
117
118        /// Get current context name for this object.
119        /// @return | string | Active context name.
120        methods.add_method("getContext", |_, this, ()| {
121            Ok(this.inner.borrow().context().as_str().to_string())
122        });
123
124        /// Enable cursor trail with fade points mode.
125        /// @param | r | number | Red (0-1).
126        /// @param | g | number | Green (0-1).
127        /// @param | b | number | Blue (0-1).
128        /// @param | lifetime | number | Seconds before trail fades.
129        methods.add_method("enableTrail", |_, this, (r, g, b, lifetime): (f32, f32, f32, f32)| {
130            let trail = CursorTrail::new(TrailMode::FadePoints {
131                color: [r, g, b, 1.0],
132                lifetime,
133            });
134            this.inner.borrow_mut().set_trail(Some(trail));
135            Ok(())
136        });
137
138        /// Enable cursor trail with line mode.
139        /// @param | r | number | Red (0-1).
140        /// @param | g | number | Green (0-1).
141        /// @param | b | number | Blue (0-1).
142        /// @param | width | number | Line width in pixels.
143        methods.add_method("enableLineTrail", |_, this, (r, g, b, width): (f32, f32, f32, f32)| {
144            let trail = CursorTrail::new(TrailMode::Line {
145                color: [r, g, b, 1.0],
146                width,
147                fade: true,
148            });
149            this.inner.borrow_mut().set_trail(Some(trail));
150            Ok(())
151        });
152
153        /// Disable cursor trail for this object.
154        methods.add_method("disableTrail", |_, this, ()| {
155            this.inner.borrow_mut().set_trail(None);
156            Ok(())
157        });
158
159        /// Enable zoom/magnifier at cursor position.
160        /// @param | magnification | number | Zoom factor (1-10).
161        /// @param | radius | number | Lens radius in pixels.
162        methods.add_method("enableZoom", |_, this, (mag, radius): (f32, f32)| {
163            this.inner.borrow_mut().set_zoom(Some(CursorZoom::new(mag, radius)));
164            Ok(())
165        });
166
167        /// Disable cursor zoom for this object.
168        methods.add_method("disableZoom", |_, this, ()| {
169            this.inner.borrow_mut().set_zoom(None);
170            Ok(())
171        });
172    }
173}

179/// Lua userdata representing a custom-drawn cursor image with a configurable hot-spot.
180#[derive(Clone)]
181struct LuaCustomCursor {
182    inner: Rc<RefCell<CustomCursor>>,
183}
184
185impl LuaUserData for LuaCustomCursor {
186    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
187        /// Set a pixel color — Lua userdata object exposed by the engine.
188        /// @param | x | integer | X coordinate.
189        /// @param | y | integer | Y coordinate.
190        /// @param | r | integer | Red (0-255).
191        /// @param | g | integer | Green (0-255).
192        /// @param | b | integer | Blue (0-255).
193        /// @param | a | integer | Alpha (0-255).
194        methods.add_method("setPixel", |_, this, (x, y, r, g, b, a): (u32, u32, u8, u8, u8, u8)| {
195            this.inner.borrow_mut().set_pixel(x, y, r, g, b, a);
196            Ok(())
197        });
198
199        /// Get the pixel color at the specified cursor image position.
200        /// @param | x | integer | X coordinate.
201        /// @param | y | integer | Y coordinate.
202        /// @return | integer | Red.
203        /// @return | integer | Green.
204        /// @return | integer | Blue.
205        /// @return | integer | Alpha.
206        methods.add_method("getPixel", |_, this, (x, y): (u32, u32)| {
207            match this.inner.borrow().get_pixel(x, y) {
208                Some((r, g, b, a)) => Ok((r, g, b, a)),
209                None => Err(LuaError::runtime("pixel out of bounds")),
210            }
211        });
212
213        /// Get the pixel width and height of the cursor image.
214        /// @return | integer | Width.
215        /// @return | integer | Height.
216        methods.add_method("getSize", |_, this, ()| {
217            let (w, h) = this.inner.borrow().size();
218            Ok((w, h))
219        });
220
221        /// Get hotspot position for this object.
222        /// @return | integer | Hotspot X.
223        /// @return | integer | Hotspot Y.
224        methods.add_method("getHotspot", |_, this, ()| {
225            let c = this.inner.borrow();
226            Ok((c.hotspot_x, c.hotspot_y))
227        });
228    }
229}

235/// Lua userdata representing an animated cursor that cycles through image frames.
236#[derive(Clone)]
237struct LuaAnimatedCursor {
238    inner: Rc<RefCell<AnimatedCursor>>,
239}
240
241impl LuaUserData for LuaAnimatedCursor {
242    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
243        /// Add a frame from a custom cursor image.
244        /// @param | cursor | LuaCustomCursor | Frame image.
245        /// @param | duration_ms | integer | Frame duration in milliseconds.
246        methods.add_method("addFrame", |_, this, (cursor, dur): (LuaAnyUserData, u32)| {
247            let cursor = cursor.borrow::<LuaCustomCursor>()?.clone();
248            let img = cursor.inner.borrow().clone();
249            this.inner.borrow_mut().add_frame(img, dur);
250            Ok(())
251        });
252
253        /// Update animation (call each frame).
254        /// @param | dt | number | Delta time in seconds.
255        methods.add_method("update", |_, this, dt: f32| {
256            this.inner.borrow_mut().update(dt);
257            Ok(())
258        });
259
260        /// Get current frame index for this object.
261        /// @return | integer | Zero-based frame index.
262        methods.add_method("currentIndex", |_, this, ()| {
263            Ok(this.inner.borrow().current_index())
264        });
265
266        /// Get total frame count for this object.
267        /// @return | integer | Number of frames.
268        methods.add_method("frameCount", |_, this, ()| {
269            Ok(this.inner.borrow().frame_count())
270        });
271
272        /// Get current scale from pulse animation.
273        /// @return | number | Current scale factor.
274        methods.add_method("currentScale", |_, this, ()| {
275            Ok(this.inner.borrow().current_scale())
276        });
277
278        /// Set the pulse animation speed and scale factor parameters.
279        /// @param | min_scale | number | Minimum scale.
280        /// @param | max_scale | number | Maximum scale.
281        /// @param | speed | number | Pulse speed.
282        methods.add_method("setPulse", |_, this, (min_s, max_s, speed): (f32, f32, f32)| {
283            this.inner.borrow_mut().set_pulse(Some(PulseConfig {
284                min_scale: min_s,
285                max_scale: max_s,
286                speed,
287            }));
288            Ok(())
289        });
290
291        /// Disable pulse animation for this object.
292        methods.add_method("clearPulse", |_, this, ()| {
293            this.inner.borrow_mut().set_pulse(None);
294            Ok(())
295        });
296
297        /// Reset the cursor animation playback to the first frame.
298        methods.add_method("reset", |_, this, ()| {
299            this.inner.borrow_mut().reset();
300            Ok(())
301        });
302    }
303}

309/// Register the `lurek.cursor` module.
310///
311/// ## Functions (see lurek Lua API reference for details).
312///
313/// ### newManager (see lurek Lua API reference for details).
314/// Create a new cursor manager exposed by the lurek engine.
315/// @return | LuaCursorManager | Cursor manager instance.
316///
317/// ### newCustom (see lurek Lua API reference for details).
318/// Create a blank custom cursor image.
319/// @param | width | integer | Image width in pixels.
320/// @param | height | integer | Image height in pixels.
321/// @param | hotspot_x | integer | Hotspot X offset.
322/// @param | hotspot_y | integer | Hotspot Y offset.
323/// @return | LuaCustomCursor | Custom cursor instance.
324///
325/// ### newAnimated (see lurek Lua API reference for details).
326/// Create a new animated cursor exposed by the lurek engine.
327/// @param | looping | boolean | Whether the animation loops.
328/// @return | LuaAnimatedCursor | Animated cursor instance.
329///
330/// ### systemCursors (see lurek Lua API reference for details).
331/// Get list of available system cursor names.
332/// @return | table | Array of cursor name strings.
333pub fn register(lua: &Lua, lurek: &LuaTable, _state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
334    let module = lua.create_table()?;
335
336    /// Creates a new cursor manager for handling cursor state and visibility.
337    ///
338    /// @return | LCursorManager | A new cursor manager instance.
339    module.set(
340        "newManager",
341        lua.create_function(|_, ()| {
342            Ok(LuaCursorManager {
343                inner: Rc::new(RefCell::new(CursorManager::new())),
344            })
345        })?,
346    )?;
347
348    /// Creates a new custom cursor with specified dimensions and hotspot position.
349    ///
350    /// @param | w | integer | Width of the cursor image in pixels.
351    /// @param | h | integer | Height of the cursor image in pixels.
352    /// @param | hx | integer | Hotspot X offset from cursor origin.
353    /// @param | hy | integer | Hotspot Y offset from cursor origin.
354    /// @return | LCustomCursor | A new custom cursor instance.
355    module.set(
356        "newCustom",
357        lua.create_function(|_, (w, h, hx, hy): (u32, u32, u32, u32)| {
358            Ok(LuaCustomCursor {
359                inner: Rc::new(RefCell::new(CustomCursor::new(w, h, hx, hy))),
360            })
361        })?,
362    )?;
363
364    /// Creates a new animated cursor that can cycle through frames.
365    ///
366    /// @param | looping | boolean | Whether the animation loops continuously.
367    /// @return | LAnimatedCursor | A new animated cursor instance.
368    module.set(
369        "newAnimated",
370        lua.create_function(|_, looping: bool| {
371            Ok(LuaAnimatedCursor {
372                inner: Rc::new(RefCell::new(AnimatedCursor::new(looping))),
373            })
374        })?,
375    )?;
376
377    /// Returns a list of all available system cursor names as a string array.
378    ///
379    /// @return | table | Array of system cursor name strings.
380    module.set(
381        "systemCursors",
382        lua.create_function(|lua, ()| {
383            let names = vec![
384                "arrow", "ibeam", "wait", "crosshair", "wait_arrow",
385                "size_nwse", "size_nesw", "size_we", "size_ns", "size_all",
386                "no", "hand",
387            ];
388            let tbl = lua.create_table()?;
389            for (i, name) in names.iter().enumerate() {
390                tbl.set(i + 1, *name)?;
391            }
392            Ok(tbl)
393        })?,
394    )?;
395
396    lurek.set("cursor", module)?;
397    Ok(())
