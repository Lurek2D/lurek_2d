1//! `lurek.automation` -- Automation bindings for loading simulator scripts, controlling playback, inspecting state, saving macros, and waiting on Lua predicates.
2
3use super::SharedState;
4use crate::automation::{Action, Script, Simulator, Step};
5use mlua::prelude::*;
6use std::cell::RefCell;
7use std::rc::Rc;
8/// Registers the `lurek.automation` API table with the Lua VM.
9pub fn register(lua: &Lua, lurek: &LuaTable, state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
10    let tbl = lua.create_table()?;
11    let simulator = Rc::new(RefCell::new(Simulator::new()));
12    let wait_state: Rc<RefCell<Option<(LuaRegistryKey, f32, f32)>>> = Rc::new(RefCell::new(None));
13    // -- load --
14    /// Loads an automation script from a Lua table of steps and optional metadata.
15    /// @param | name | string | Script name used by `start`, macros, and lookup calls.
16    /// @param | data | table | Script data table with a `steps` array and optional `meta.description` string.
17    let sim = simulator.clone();
18    tbl.set(
19        "load",
20        lua.create_function(move |_, (name, data): (String, LuaTable)| {
21            let steps_table: LuaTable = data.get::<_, LuaTable>("steps").map_err(|_| {
22                LuaError::external("simulator.load: data table must have a 'steps' array")
23            })?;
24            let steps = Step::vec_from_lua_table(&steps_table)?;
25            let description: Option<String> = data
26                .get::<_, Option<LuaTable>>("meta")?
27                .and_then(|meta| meta.get::<_, Option<String>>("description").ok().flatten());
28            let script = match description {
29                Some(desc) => Script::with_description(name, desc, steps),
30                None => Script::new(name, steps),
31            };
32            sim.borrow_mut().load(script);
33            Ok(())
34        })?,
35    )?;
36    // -- unload --
37    /// Unloads a named automation script.
38    /// @param | name | string | Script name to remove.
39    /// @return | boolean | True when the script existed and was removed.
40    let sim = simulator.clone();
41    tbl.set(
42        "unload",
43        lua.create_function(move |_, name: String| Ok(sim.borrow_mut().unload(&name)))?,
44    )?;
45    // -- hasScript --
46    /// Returns whether a script is loaded.
47    /// @param | name | string | Script name to check.
48    /// @return | boolean | True when the script is loaded.
49    let sim = simulator.clone();
50    tbl.set(
51        "hasScript",
52        lua.create_function(move |_, name: String| Ok(sim.borrow().has_script(&name)))?,
53    )?;
54    // -- getScripts --
55    /// Returns the names of loaded automation scripts.
56    /// @return | string[] | Script names.
57    let sim = simulator.clone();
58    tbl.set(
59        "getScripts",
60        lua.create_function(move |_, ()| Ok(sim.borrow().get_scripts()))?,
61    )?;
62    // -- start --
63    /// Starts playback of a loaded automation script.
64    /// @param | name | string | Loaded script name to start.
65    let sim = simulator.clone();
66    tbl.set(
67        "start",
68        lua.create_function(move |_, name: String| {
69            sim.borrow_mut().start(&name).map_err(LuaError::external)
70        })?,
71    )?;
72    // -- stop --
73    /// Stops the current automation script.
74    let sim = simulator.clone();
75    tbl.set(
76        "stop",
77        lua.create_function(move |_, ()| {
78            sim.borrow_mut().stop();
79            Ok(())
80        })?,
81    )?;
82    // -- pause --
83    /// Pauses automation playback. This function is exposed to Lua scripts.
84    let sim = simulator.clone();
85    tbl.set(
86        "pause",
87        lua.create_function(move |_, ()| {
88            sim.borrow_mut().pause();
89            Ok(())
90        })?,
91    )?;
92    // -- resume --
93    /// Resumes automation playback. This function is exposed to Lua scripts.
94    let sim = simulator.clone();
95    tbl.set(
96        "resume",
97        lua.create_function(move |_, ()| {
98            sim.borrow_mut().resume();
99            Ok(())
100        })?,
101    )?;
102    // -- update --
103    /// Advances automation playback and dispatches generated input events.
104    /// @param | dt | number | Elapsed time in seconds.
105    let sim = simulator.clone();
106    let s = state.clone();
107    let ws = wait_state.clone();
108    tbl.set(
109        "update",
110        lua.create_function(move |lua, dt: f32| {
111            if ws.borrow().is_some() {
112                let (resolved, timed_out) = {
113                    let state_opt = ws.borrow();
114                    if let Some((ref key, timeout, elapsed)) = *state_opt {
115                        let new_elapsed = elapsed + dt;
116                        let predicate: LuaFunction = lua.registry_value(key)?;
117                        let done: bool = predicate.call(())?;
118                        (done, new_elapsed >= timeout)
119                    } else {
120                        (false, false)
121                    }
122                };
123                if resolved || timed_out {
124                    *ws.borrow_mut() = None;
125                } else {
126                    if let Some(ref mut triple) = ws.borrow_mut().as_mut() {
127                        triple.2 += dt;
128                    }
129                    return Ok(());
130                }
131            }
132            sim.borrow_mut().update(dt, &mut s.borrow_mut().event_queue);
133            Ok(())
134        })?,
135    )?;
136    // -- isRunning --
137    /// Returns whether automation playback is running.
138    /// @return | boolean | True when a script is running.
139    let sim = simulator.clone();
140    tbl.set(
141        "isRunning",
142        lua.create_function(move |_, ()| Ok(sim.borrow().is_running()))?,
143    )?;
144    // -- isPaused --
145    /// Returns whether automation playback is paused.
146    /// @return | boolean | True when playback is paused.
147    let sim = simulator.clone();
148    tbl.set(
149        "isPaused",
150        lua.create_function(move |_, ()| Ok(sim.borrow().is_paused()))?,
151    )?;
152    // -- isComplete --
153    /// Returns whether the current automation script completed.
154    /// @return | boolean | True when the current script has completed.
155    let sim = simulator.clone();
156    tbl.set(
157        "isComplete",
158        lua.create_function(move |_, ()| Ok(sim.borrow().is_complete()))?,
159    )?;
160    // -- isFailed --
161    /// Returns whether the current automation script failed.
162    /// @return | boolean | True when the current script has failed.
163    let sim = simulator.clone();
164    tbl.set(
165        "isFailed",
166        lua.create_function(move |_, ()| Ok(sim.borrow().is_failed()))?,
167    )?;
168    // -- getLastError --
169    /// Returns the last automation error message when one exists.
170    /// @return | string | Last error string, or nil when no error is stored.
171    let sim = simulator.clone();
172    tbl.set(
173        "getLastError",
174        lua.create_function(move |_, ()| Ok(sim.borrow().last_error().map(|s| s.to_string())))?,
175    )?;
176    // -- setCondition --
177    /// Sets a named boolean condition used by automation steps.
178    /// @param | name | string | Condition name.
179    /// @param | value | boolean | Condition value.
180    let sim = simulator.clone();
181    tbl.set(
182        "setCondition",
183        lua.create_function(move |_, (name, value): (String, bool)| {
184            sim.borrow_mut().set_condition(name, value);
185            Ok(())
186        })?,
187    )?;
188    // -- getCondition --
189    /// Returns a named automation condition value.
190    /// @param | name | string | Condition name.
191    /// @return | boolean | Current condition value.
192    let sim = simulator.clone();
193    tbl.set(
194        "getCondition",
195        lua.create_function(move |_, name: String| Ok(sim.borrow().get_condition(&name)))?,
196    )?;
197    // -- getCurrentStep --
198    /// Returns the current step index of the active script.
199    /// @return | integer | Current step index.
200    let sim = simulator.clone();
201    tbl.set(
202        "getCurrentStep",
203        lua.create_function(move |_, ()| Ok(sim.borrow().current_step()))?,
204    )?;
205    // -- getStepCount --
206    /// Returns the number of steps in the active script.
207    /// @return | integer | Active script step count.
208    let sim = simulator.clone();
209    tbl.set(
210        "getStepCount",
211        lua.create_function(move |_, ()| Ok(sim.borrow().step_count()))?,
212    )?;
213    // -- getCurrentScript --
214    /// Returns the current script name when a script is active.
215    /// @return | string | Current script name, or nil when no script is active.
216    let sim = simulator.clone();
217    tbl.set(
218        "getCurrentScript",
219        lua.create_function(move |_, ()| Ok(sim.borrow().current_script().map(|s| s.to_string())))?,
220    )?;
221    // -- getElapsedTime --
222    /// Returns elapsed playback time for the current script.
223    /// @return | number | Elapsed time in seconds.
224    let sim = simulator.clone();
225    tbl.set(
226        "getElapsedTime",
227        lua.create_function(move |_, ()| Ok(sim.borrow().elapsed_time()))?,
228    )?;
229    // -- loadFromToml --
230    /// Loads an automation script from TOML text.
231    /// @param | name | string | Script name used by `start`, macros, and lookup calls.
232    /// @param | toml_str | string | TOML automation script contents.
233    let sim = simulator.clone();
234    tbl.set(
235        "loadFromToml",
236        lua.create_function(move |_, (name, toml_str): (String, String)| {
237            let script = Script::from_toml(&name, &toml_str)
238                .map_err(|e| LuaError::external(format!("loadFromToml: {e}")))?;
239            sim.borrow_mut().load(script);
240            Ok(())
241        })?,
242    )?;
243    // -- getStepLimit --
244    /// Returns the configured step limit for a loaded script.
245    /// @param | name | string | Script name to query.
246    /// @return | integer | Step limit, or nil when no limit is set.
247    let sim = simulator.clone();
248    tbl.set(
249        "getStepLimit",
250        lua.create_function(move |_, name: String| {
251            Ok(sim.borrow().get_script_step_limit(&name).map(|v| v as u64))
252        })?,
253    )?;
254    // -- setStepLimit --
255    /// Sets the maximum step count for a loaded script.
256    /// @param | name | string | Script name to update.
257    /// @param | n | integer | Maximum step count.
258    /// @return | boolean | True when the script exists and the limit was set.
259    let sim = simulator.clone();
260    tbl.set(
261        "setStepLimit",
262        lua.create_function(move |_, (name, n): (String, u64)| {
263            Ok(sim.borrow_mut().set_script_step_limit(&name, n as usize))
264        })?,
265    )?;
266    // -- saveMacro --
267    /// Saves a loaded script as a named macro.
268    /// @param | macro_name | string | Macro name to save.
269    /// @param | script_name | string | Loaded script name to copy into the macro store.
270    let sim = simulator.clone();
271    tbl.set(
272        "saveMacro",
273        lua.create_function(move |_, (macro_name, script_name): (String, String)| {
274            let script = sim.borrow().get_script(&script_name).ok_or_else(|| {
275                LuaError::external(format!("saveMacro: script '{}' not found", script_name))
276            })?;
277            sim.borrow_mut().save_macro(macro_name, script);
278            Ok(())
279        })?,
280    )?;
281    // -- playMacro --
282    /// Starts playback of a saved macro. This function is exposed to Lua scripts.
283    /// @param | name | string | Macro name to play.
284    let sim = simulator.clone();
285    tbl.set(
286        "playMacro",
287        lua.create_function(move |_, name: String| {
288            sim.borrow_mut()
289                .play_macro(&name)
290                .map_err(LuaError::external)
291        })?,
292    )?;
293    // -- hasMacro --
294    /// Returns whether a macro is saved. This function is exposed to Lua scripts.
295    /// @param | name | string | Macro name to check.
296    /// @return | boolean | True when the macro exists.
297    let sim = simulator.clone();
298    tbl.set(
299        "hasMacro",
300        lua.create_function(move |_, name: String| Ok(sim.borrow().has_macro(&name)))?,
301    )?;
302    // -- listMacros --
303    /// Returns the names of saved macros. This function is exposed to Lua scripts.
304    /// @return | string[] | Macro names.
305    let sim = simulator.clone();
306    tbl.set(
307        "listMacros",
308        lua.create_function(move |_, ()| Ok(sim.borrow().list_macros()))?,
309    )?;
310    // -- setPlaybackSpeed --
311    /// Sets automation playback speed multiplier.
312    /// @param | factor | number | Playback speed multiplier.
313    let sim = simulator.clone();
314    tbl.set(
315        "setPlaybackSpeed",
316        lua.create_function(move |_, factor: f32| {
317            sim.borrow_mut().set_playback_speed(factor);
318            Ok(())
319        })?,
320    )?;
321    // -- getPlaybackSpeed --
322    /// Returns automation playback speed multiplier.
323    /// @return | number | Current playback speed multiplier.
324    let sim = simulator.clone();
325    tbl.set(
326        "getPlaybackSpeed",
327        lua.create_function(move |_, ()| Ok(sim.borrow().get_playback_speed()))?,
328    )?;
329    // -- setHighlightMode --
330    /// Enables or disables automation highlight mode.
331    /// @param | enable | boolean | True to enable highlight mode.
332    let sim = simulator.clone();
333    tbl.set(
334        "setHighlightMode",
335        lua.create_function(move |_, enable: bool| {
336            sim.borrow_mut().set_highlight_mode(enable);
337            Ok(())
338        })?,
339    )?;
340    // -- isHighlightMode --
341    /// Returns whether automation highlight mode is enabled.
342    /// @return | boolean | True when highlight mode is enabled.
343    let sim = simulator.clone();
344    tbl.set(
345        "isHighlightMode",
346        lua.create_function(move |_, ()| Ok(sim.borrow().is_highlight_mode()))?,
347    )?;
348    // -- waitUntil --
349    /// Suspends automation updates until a predicate returns true or a timeout elapses.
350    /// @param | predicate | function | Function called each update; true resolves the wait.
351    /// @param | timeout | number | Maximum wait duration in seconds.
352    let ws = wait_state.clone();
353    tbl.set(
354        "waitUntil",
355        lua.create_function(move |lua, (predicate, timeout): (LuaFunction, f32)| {
356            let key = lua.create_registry_value(predicate)?;
357            *ws.borrow_mut() = Some((key, timeout.max(0.0), 0.0));
358            Ok(())
359        })?,
360    )?;
361    /// Performs the 'automation' operation.
362    lurek.set("automation", tbl)?;
363    Ok(())
364}
365impl Step {
366    /// Converts a Lua array of step tables into automation steps.
367    pub fn vec_from_lua_table(t: &LuaTable) -> LuaResult<Vec<Self>> {
368        let len = t.len()? as usize;
369        let mut steps = Vec::with_capacity(len);
370        for i in 1..=len {
371            let entry: LuaTable = t.get(i)?;
372            let action_str: String = entry.get::<_, String>("action").map_err(|_| {
373                LuaError::external("simulator.load: each step must have an 'action' field")
374            })?;
375            let action = Action::parse_action(&action_str).ok_or_else(|| {
376                LuaError::external(format!(
377                    "simulator.load: unknown action '{}' \u{2014} expected one of: keypress, keyrelease, mousemove, mousepress, mouserelease, mousewheel, textinput, wait, repeat, callmacro, assert, visualassert",
378                    action_str
379                ))
380            })?;
381            let time: f32 = entry.get::<_, Option<f32>>("time")?.unwrap_or(0.0);
382            let mut step = Self::new(time, action);
383            step.key = entry.get::<_, Option<String>>("key")?;
384            step.scancode = entry.get::<_, Option<String>>("scancode")?;
385            step.x = entry.get::<_, Option<f64>>("x")?;
386            step.y = entry.get::<_, Option<f64>>("y")?;
387            step.dx = entry.get::<_, Option<f64>>("dx")?;
388            step.dy = entry.get::<_, Option<f64>>("dy")?;
389            step.button = entry.get::<_, Option<u32>>("button")?;
390            step.text = entry.get::<_, Option<String>>("text")?;
391            step.is_repeat = entry.get::<_, Option<bool>>("isRepeat")?.unwrap_or(false);
392            step.clicks = entry.get::<_, Option<u32>>("clicks")?;
393            step.repeat = entry.get::<_, Option<u32>>("repeat")?;
394            step.repeat_interval = entry.get::<_, Option<f32>>("repeatInterval")?;
395            step.macro_name = entry.get::<_, Option<String>>("macro")?;
396            step.when = entry.get::<_, Option<String>>("when")?;
397            step.assert = entry.get::<_, Option<String>>("assert")?;
398            step.baseline = entry.get::<_, Option<String>>("baseline")?;
399            step.actual = entry.get::<_, Option<String>>("actual")?;
400            step.max_diff = entry.get::<_, Option<u32>>("maxDiff")?;
401            steps.push(step);
402        }
403        Ok(steps)
404    }
