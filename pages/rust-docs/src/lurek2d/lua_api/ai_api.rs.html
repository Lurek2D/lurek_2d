18/// Lua handle for an AI world that owns named agents, global blackboard data, and custom callback registrations.
19#[derive(Clone)]
20struct LuaAIWorld {
21    /// Shared AI world state used by every world, agent, and blackboard wrapper cloned from this handle.
22    inner: Rc<RefCell<AIWorld>>,
23    /// Registry of Lua callbacks used by custom agent decision models created through this world.
24    custom_callbacks: Rc<RefCell<CallbackRegistry>>,
25}
26impl LuaUserData for LuaAIWorld {
27    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
28        // -- addAgent --
29        /// Creates a named agent in this world and returns a handle that can edit its movement and decision state.
30        /// @param | name | string | Unique agent name used by later lookup, tags, custom callbacks, and squad membership references.
31        /// @return | LAgent | Lua handle for the newly inserted agent.
32        methods.add_method("addAgent", |_, this, name: String| {
33            let mut w = this.inner.borrow_mut();
34            w.add_agent(&name).map_err(LuaError::RuntimeError)?;
35            Ok(LuaAgent {
36                world: this.inner.clone(),
37                name,
38                callbacks: this.custom_callbacks.clone(),
39            })
40        });
41        // -- getAgent --
42        /// Returns the named agent handle when it exists in this world.
43        /// @param | name | string | Agent name previously passed to `addAgent`.
44        /// @return | LuaValue | Agent handle when found, or nil when the world has no agent with that name.
45        methods.add_method("getAgent", |_, this, name: String| {
46            let w = this.inner.borrow();
47            if w.get_agent_index(&name).is_some() {
48                Ok(Some(LuaAgent {
49                    world: this.inner.clone(),
50                    name,
51                    callbacks: this.custom_callbacks.clone(),
52                }))
53            } else {
54                Ok(None)
55            }
56        });
57        // -- removeAgent --
58        /// Removes an agent from this world by using an existing agent handle.
59        /// @param | agent | LAgent | Agent handle whose stored name identifies the world entry to remove.
60        methods.add_method("removeAgent", |_, this, agent: LuaAnyUserData| {
61            let a = agent.borrow::<LuaAgent>()?;
62            this.inner.borrow_mut().remove_agent(&a.name);
63            Ok(())
64        });
65        // -- getAgentCount --
66        /// Returns the number of agents currently stored in this world.
67        /// @return | integer | Current agent count.
68        methods.add_method("getAgentCount", |_, this, ()| {
69            Ok(this.inner.borrow().agent_count())
70        });
71        // -- getGlobalBlackboard --
72        /// Returns a blackboard snapshot containing the world's shared AI facts.
73        /// @return | LAIBlackboard | Blackboard handle initialized from the world's global blackboard values at call time.
74        methods.add_method("getGlobalBlackboard", |_, this, ()| {
75            let w = this.inner.borrow();
76            Ok(LuaAIBlackboard {
77                inner: Rc::new(RefCell::new(w.global_blackboard().clone())),
78            })
79        });
80        // -- update --
81        /// Advances the world simulation and invokes custom decision callbacks for agents that use a custom model.
82        /// @param | dt | number | Elapsed simulation time in seconds for this update step.
83        methods.add_method("update", |lua, this, dt: f32| {
84            this.inner.borrow_mut().update(dt);
85            let custom_agents: Vec<(String, u32)> = {
86                let w = this.inner.borrow();
87                w.agents
88                    .iter()
89                    .filter_map(|a| {
90                        if let crate::ai::DecisionModel::Custom { callback_id } = a.decision_model {
91                            Some((a.name.clone(), callback_id))
92                        } else {
93                            None
94                        }
95                    })
96                    .collect()
97            };
98            for (name, callback_id) in custom_agents {
99                let lua_agent = LuaAgent {
100                    world: this.inner.clone(),
101                    name: name.clone(),
102                    callbacks: this.custom_callbacks.clone(),
103                };
104                let lua_bb = {
105                    let w = this.inner.borrow();
106                    match w.agent(&name) {
107                        Some(agent) => LuaAIBlackboard {
108                            inner: Rc::new(RefCell::new(agent.blackboard.clone())),
109                        },
110                        None => continue,
111                    }
112                };
113                let func_opt: Option<LuaFunction> = {
114                    let cb = this.custom_callbacks.borrow();
115                    cb.get(callback_id)
116                        .and_then(|key| lua.registry_value(key).ok())
117                };
118                if let Some(func) = func_opt {
119                    if let Err(e) = func.call::<_, ()>((lua_agent, lua_bb, dt)) {
120                        eprintln!("[lurek.ai] custom model callback error for '{name}': {e}");
121                    }
122                }
123            }
124            Ok(())
125        });
126        // -- type --
127        /// Returns the Lua-visible type name for this AI world handle.
128        /// @return | string | The string `LAIWorld`.
129        methods.add_method("type", |_, _, ()| Ok("LAIWorld"));
130        // -- typeOf --
131        /// Returns whether this AI world handle matches a supported type name.
132        /// @param | name | string | Type name to compare against `AIWorld` and `Object`.
133        /// @return | boolean | True when the supplied type name matches this handle.
134        methods.add_method("typeOf", |_, _, name: String| {
135            Ok(name == "LAIWorld" || name == "LObject")
136        });
137    }

174        });
175        // -- setVelocity --
176        /// Sets this agent's velocity vector when the agent still exists in its world.
177        /// @param | x | number | New X velocity in world units per second.
178        /// @param | y | number | New Y velocity in world units per second.
179        methods.add_method("setVelocity", |_, this, (x, y): (f32, f32)| {
180            if let Some(agent) = this.world.borrow_mut().agent_mut(&this.name) {
181                agent.velocity = (x, y);
182            }
183            Ok(())
184        });
185        // -- getVelocity --
186        /// Returns this agent's velocity vector or zero velocity when the agent has been removed.
187        /// @return | number, number | X and Y velocity in world units per second.
188        methods.add_method("getVelocity", |_, this, ()| {
189            if let Some(agent) = this.world.borrow().agent(&this.name) {
190                Ok(agent.velocity)
191            } else {
192                Ok((0.0, 0.0))
193            }
194        });
195        // -- setMaxSpeed --
196        /// Sets this agent's maximum movement speed when the agent still exists in its world.
197        /// @param | v | number | Maximum speed in world units per second.
198        methods.add_method("setMaxSpeed", |_, this, v: f32| {
199            if let Some(agent) = this.world.borrow_mut().agent_mut(&this.name) {
200                agent.max_speed = v;
201            }
202            Ok(())
203        });
204        // -- getMaxSpeed --
205        /// Returns this agent's maximum movement speed or the default speed for a missing agent.
206        /// @return | number | Maximum speed in world units per second.
207        methods.add_method("getMaxSpeed", |_, this, ()| {
208            if let Some(agent) = this.world.borrow().agent(&this.name) {

211                Ok(100.0)

215        /// Sets this agent's maximum steering force when the agent still exists in its world.
216        /// @param | v | number | Maximum steering force applied during steering calculations.

218            if let Some(agent) = this.world.borrow_mut().agent_mut(&this.name) {
219                agent.max_force = v;
220            }
221            Ok(())
222        });
223        // -- getMaxForce --

225        /// @return | number | Maximum steering force value.

227            if let Some(agent) = this.world.borrow().agent(&this.name) {
228                Ok(agent.max_force)

230                Ok(200.0)

233        // -- setPriority --
234        /// Sets this agent's integer priority when the agent still exists in its world.
235        /// @param | p | integer | Priority value used by game-side AI scheduling or ordering logic.
236        methods.add_method("setPriority", |_, this, p: i32| {
237            if let Some(agent) = this.world.borrow_mut().agent_mut(&this.name) {
238                agent.priority = p;
239            }
240            Ok(())
241        });
242        // -- getPriority --
243        /// Returns this agent's integer priority or zero when the agent has been removed.
244        /// @return | integer | Current priority value.
245        methods.add_method("getPriority", |_, this, ()| {
246            if let Some(agent) = this.world.borrow().agent(&this.name) {
247                Ok(agent.priority)
248            } else {
249                Ok(0)
250            }
251        });
252        // -- setDecisionModel --
253        /// Sets this agent's built-in decision model from a string name when the name is recognized.
254        /// @param | model | string | Decision model name such as `fsm`, `bt`, `utility`, or another engine-supported model string.

256            let mut w = this.world.borrow_mut();
257            if let Some(idx) = w.get_agent_index(&this.name) {
258                if let Some(dm) = DecisionModel::parse_str(&model) {

263        });
264        // -- getDecisionModel --

271                Ok("fsm".to_string())

273        });
274        // -- setCustomModel --
275        /// Installs a Lua callback as this agent's decision model and stores it in the callback registry.
276        /// @param | callback | function | Function called during world updates with `(agent, blackboard, dt)` for this agent.

284        });
285        // -- addTag --
286        /// Adds a tag string to this agent when the agent still exists in its world.
287        /// @param | tag | string | Tag name to insert into the agent tag set.

289            if let Some(agent) = this.world.borrow_mut().agent_mut(&this.name) {
290                agent.tags.insert(tag);
291            }
292            Ok(())
293        });
294        // -- removeTag --
295        /// Removes a tag string from this agent when the agent still exists in its world.
296        /// @param | tag | string | Tag name to remove from the agent tag set.
297        methods.add_method("removeTag", |_, this, tag: String| {
298            if let Some(agent) = this.world.borrow_mut().agent_mut(&this.name) {
299                agent.tags.remove(&tag);
300            }
301            Ok(())
302        });
303        // -- hasTag --
304        /// Returns whether this agent currently has the given tag.
305        /// @param | tag | string | Tag name to check in the agent tag set.
306        /// @return | boolean | True when the tag exists on the agent.
307        methods.add_method("hasTag", |_, this, tag: String| {
308            if let Some(agent) = this.world.borrow().agent(&this.name) {
309                Ok(agent.tags.contains(&tag))
310            } else {
311                Ok(false)
312            }
313        });
314        // -- getBlackboard --
315        /// Returns a blackboard snapshot for this agent or an empty blackboard when the agent has been removed.
316        /// @return | LAIBlackboard | Blackboard handle initialized from the agent's local blackboard values at call time.
317        methods.add_method("getBlackboard", |_, this, ()| {
318            let w = this.world.borrow();
319            if let Some(idx) = w.get_agent_index(&this.name) {
320                Ok(LuaAIBlackboard {
321                    inner: Rc::new(RefCell::new(w.agents[idx].blackboard.clone())),
322                })
323            } else {
324                Ok(LuaAIBlackboard {
325                    inner: Rc::new(RefCell::new(Blackboard::default())),
326                })
327            }
328        });
329        // -- type --
330        /// Returns the Lua-visible type name for this agent handle.
331        /// @return | string | The string `LAgent`.
332        methods.add_method("type", |_, _, ()| Ok("LAgent"));
333        // -- typeOf --
334        /// Returns whether this agent handle matches a supported type name.
335        /// @param | name | string | Type name to compare against `Agent` and `Object`.
336        /// @return | boolean | True when the supplied type name matches this handle.
337        methods.add_method("typeOf", |_, _, name: String| {
338            Ok(name == "LAgent" || name == "LObject")
339        });
340    }

342/// Lua handle for a typed AI blackboard storing local key-value facts.
343#[derive(Clone)]
344struct LuaAIBlackboard {
345    /// Shared blackboard values exposed through typed setter and getter methods.

347}
348impl LuaUserData for LuaAIBlackboard {
349    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
350        // -- setNumber --
351        /// Stores a numeric fact under the given blackboard key.
352        /// @param | key | string | Blackboard key to write.
353        /// @param | value | number | Numeric value stored for later numeric reads.
354        methods.add_method("setNumber", |_, this, (key, value): (String, f64)| {
355            this.inner.borrow_mut().set_number(&key, value);
356            Ok(())
357        });
358        // -- getNumber --
359        /// Returns a numeric blackboard fact or the provided fallback when the key is missing or not numeric.
360        /// @param | key | string | Blackboard key to read.
361        /// @param | default | number? | Fallback value used when the key has no numeric entry; defaults to zero.
362        /// @return | number | Stored numeric value or fallback value.
363        methods.add_method(
364            "getNumber",
365            |_, this, (key, default): (String, Option<f64>)| {
366                Ok(this.inner.borrow().get_number(&key, default.unwrap_or(0.0)))
367            },
368        );
369        // -- setBool --
370        /// Stores a boolean fact under the given blackboard key.
371        /// @param | key | string | Blackboard key to write.
372        /// @param | value | boolean | Boolean value stored for later boolean reads.
373        methods.add_method("setBool", |_, this, (key, value): (String, bool)| {
374            this.inner.borrow_mut().set_bool(&key, value);
375            Ok(())
376        });
377        // -- getBool --
378        /// Returns a boolean blackboard fact or the provided fallback when the key is missing or not boolean.
379        /// @param | key | string | Blackboard key to read.
380        /// @param | default | boolean? | Fallback value used when the key has no boolean entry; defaults to false.
381        /// @return | boolean | Stored boolean value or fallback value.
382        methods.add_method(
383            "getBool",
384            |_, this, (key, default): (String, Option<bool>)| {
385                Ok(this.inner.borrow().get_bool(&key, default.unwrap_or(false)))
386            },
387        );
388        // -- setString --
389        /// Stores a string fact under the given blackboard key.
390        /// @param | key | string | Blackboard key to write.
391        /// @param | value | string | String value stored for later string reads.
392        methods.add_method("setString", |_, this, (key, value): (String, String)| {
393            this.inner.borrow_mut().set_string(&key, &value);
394            Ok(())
395        });
396        // -- getString --
397        /// Returns a string blackboard fact or the provided fallback when the key is missing or not a string.
398        /// @param | key | string | Blackboard key to read.
399        /// @param | default | string? | Fallback value used when the key has no string entry; defaults to an empty string.
400        /// @return | string | Stored string value or fallback value.
401        methods.add_method(
402            "getString",
403            |_, this, (key, default): (String, Option<String>)| {
404                let def = default.unwrap_or_default();
405                Ok(this.inner.borrow().get_string(&key, &def))
406            },
407        );
408        // -- has --
409        /// Returns whether the blackboard contains any entry for the given key.
410        /// @param | key | string | Blackboard key to check.
411        /// @return | boolean | True when any typed value is stored at the key.
412        methods.add_method("has", |_, this, key: String| {
413            Ok(this.inner.borrow().has(&key))
414        });
415        // -- remove --
416        /// Removes the given key from the blackboard if it exists.
417        /// @param | key | string | Blackboard key to remove.
418        methods.add_method("remove", |_, this, key: String| {
419            this.inner.borrow_mut().remove(&key);
420            Ok(())
421        });
422        // -- clear --
423        /// Removes every local entry from this blackboard.
424        methods.add_method("clear", |_, this, ()| {
425            this.inner.borrow_mut().clear();
426            Ok(())
427        });
428        // -- getKeys --
429        /// Returns every local blackboard key in an array-style Lua table.
430        /// @return | string[] | Array table containing all stored key names as strings.
431        methods.add_method("getKeys", |lua, this, ()| {
432            let keys = this.inner.borrow().keys();
433            let tbl = lua.create_table()?;
434            for (i, k) in keys.iter().enumerate() {
435                tbl.set(i as i64 + 1, k.as_str())?;
436            }
437            Ok(tbl)
438        });
439        // -- getSize --
440        /// Returns the number of entries currently stored in this blackboard.
441        /// @return | integer | Current blackboard entry count.
442        methods.add_method("getSize", |_, this, ()| Ok(this.inner.borrow().size()));
443        // -- type --
444        /// Returns the Lua-visible type name for this blackboard handle.
445        /// @return | string | The string `LAIBlackboard`.
446        methods.add_method("type", |_, _, ()| Ok("LAIBlackboard"));
447        // -- typeOf --
448        /// Returns whether this blackboard handle matches a supported type name.
449        /// @param | name | string | Type name to compare against `AIBlackboard`, `Blackboard`, and `Object`.
450        /// @return | boolean | True when the supplied type name matches this handle.
451        methods.add_method("typeOf", |_, _, name: String| {
452            Ok(name == "LAIBlackboard" || name == "LBlackboard" || name == "LObject")
453        });
454    }

456/// Lua handle for a finite state machine with Lua-backed state callbacks and transition guards.
457#[derive(Clone)]
458struct LuaStateMachine {
459    /// Shared state machine containing states, transitions, current state, and timing data.
460    inner: Rc<RefCell<crate::ai::StateMachine>>,
461}
462impl LuaUserData for LuaStateMachine {
463    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
464        // -- addState --
465        /// Adds a state with optional Lua lifecycle callbacks.
466        /// @param | name | string | State name used by transitions and direct state changes.
467        /// @param | opts | table | Optional table with `onEnter`, `onUpdate`, and `onExit` callback functions.
468        methods.add_method("addState", |lua, this, (name, opts): (String, LuaTable)| {
469            let on_enter: Option<LuaFunction> = opts.get("onEnter").ok();
470            let on_update: Option<LuaFunction> = opts.get("onUpdate").ok();
471            let on_exit: Option<LuaFunction> = opts.get("onExit").ok();
472            let enter_key = on_enter.map(|f| lua.create_registry_value(f)).transpose()?;
473            let update_key = on_update
474                .map(|f| lua.create_registry_value(f))
475                .transpose()?;
476            let exit_key = on_exit.map(|f| lua.create_registry_value(f)).transpose()?;
477            this.inner
478                .borrow_mut()
479                .add_state_raw(name, enter_key, update_key, exit_key);
480            Ok(())
481        });
482        // -- addTransition --
483        /// Adds a transition between two states with an optional guard callback and priority.
484        /// @param | from | string | Source state name.
485        /// @param | to | string | Destination state name.
486        /// @param | guard | function? | Optional function that must return true for the transition to run.
487        /// @param | priority | integer? | Transition priority used when multiple transitions are available; defaults to zero.
488        methods.add_method("addTransition", |lua, this, (from, to, guard, priority): (String, String, Option<LuaFunction>, Option<i32>)| {
489                let guard_key = guard.map(|f| lua.create_registry_value(f)).transpose()?;
490                this.inner.borrow_mut().add_transition_raw(from, to, priority.unwrap_or(0), guard_key);
491                Ok(())
492            },
493        );
494        // -- setInitialState --
495        /// Sets the initial state and also enters it when the machine has no current state yet.
496        /// @param | name | string | State name to use as the initial state.
497        methods.add_method("setInitialState", |_, this, name: String| {
498            let mut fsm = this.inner.borrow_mut();
499            fsm.initial_state = Some(name.clone());
500            if fsm.current_state.is_none() {
501                fsm.current_state = Some(name);
502            }
503            Ok(())
504        });
505        // -- getCurrentState --
506        /// Returns the current state name when the state machine has entered a state.
507        /// @return | LuaValue | Current state name, or nil before any state is active.
508        methods.add_method("getCurrentState", |_, this, ()| {
509            Ok(this.inner.borrow().current_state().map(|s| s.to_string()))
510        });
511        // -- forceState --
512        /// Immediately switches the current state and resets the time spent in state.
513        /// @param | name | string | State name to set as current without transition checks.
514        methods.add_method("forceState", |_, this, name: String| {
515            let mut fsm = this.inner.borrow_mut();
516            fsm.current_state = Some(name);
517            fsm.time_in_state = 0.0;
518            Ok(())
519        });
520        // -- getTimeInState --
521        /// Returns how long the machine has spent in the current state.
522        /// @return | number | Elapsed time in seconds since the current state was entered.
523        methods.add_method("getTimeInState", |_, this, ()| {
524            Ok(this.inner.borrow().time_in_state())
525        });
526        // -- type --
527        /// Returns the Lua-visible type name for this state machine handle.
528        /// @return | string | The string `LStateMachine`.
529        methods.add_method("type", |_, _, ()| Ok("LStateMachine"));
530        // -- typeOf --

1071                            }
1072                        }

1074                }

1076            },
