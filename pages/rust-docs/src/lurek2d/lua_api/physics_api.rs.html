12/// Parses a strict Lua body type string into the corresponding engine body type.
13fn parse_body_type(s: &str) -> LuaResult<BodyType> {
14    match s {
15        "static" => Ok(BodyType::Static),
16        "dynamic" => Ok(BodyType::Dynamic),
17        "kinematic" => Ok(BodyType::Kinematic),
18        "sensor" => Ok(BodyType::Sensor),
19        _ => Err(LuaError::external(format!(
20            "invalid body type '{}': expected static, dynamic, kinematic, or sensor",
21            s
22        ))),
23    }
24}
25/// Parses a Lua body type string but falls back to `Dynamic` for unknown values.
26fn parse_body_type_lenient(s: &str) -> BodyType {
27    match s {
28        "static" => BodyType::Static,
29        "kinematic" => BodyType::Kinematic,
30        "sensor" => BodyType::Sensor,
31        _ => BodyType::Dynamic,
32    }
33}
34/// Converts Lua shape constructor arguments into an engine physics shape definition.
35fn shape_from_lua(lua: &Lua, shape_type: &str, args: LuaMultiValue) -> LuaResult<Shape> {
36    let mut float_args: Vec<f32> = Vec::new();
37    let mut closed = false;
38    let mut iter = args.into_iter();
39    let first = iter.next().unwrap_or(LuaValue::Nil);
40    if matches!(shape_type, "polygon" | "chain") {
41        let tbl: LuaTable = lua.unpack(first)?;
42        let len = tbl.raw_len();
43        let mut i = 1i64;
44        while i < len as i64 {
45            float_args.push(tbl.raw_get(i)?);
46            float_args.push(tbl.raw_get(i + 1)?);
47            i += 2;
48        }
49        if shape_type == "chain" {
50            closed = lua
51                .unpack(iter.next().unwrap_or(LuaValue::Boolean(false)))
52                .unwrap_or(false);
53        }
54    } else {
55        float_args.push(lua.unpack(first)?);
56        for v in iter {
57            if let Ok(f) = lua.unpack::<f32>(v) {
58                float_args.push(f);
59            }
60        }
61    }
62    Shape::from_parts(shape_type, &float_args, closed).map_err(LuaError::runtime)
63}
64/// Serializes a physics raycast hit into the Lua table shape exposed by the bindings.
65fn raycast_hit_to_table<'lua>(lua: &'lua Lua, hit: &RaycastHit) -> LuaResult<LuaTable<'lua>> {
66    let tbl = lua.create_table()?;
67    /// Performs the 'bodyId' operation.
68    tbl.set("bodyId", hit.body_id)?;
69    /// The 'x' field value exposed to Lua scripts.
70    tbl.set("x", hit.point.0)?;
71    /// The 'y' field value exposed to Lua scripts.
72    tbl.set("y", hit.point.1)?;
73    /// Performs the 'normalX' operation.
74    tbl.set("normalX", hit.normal.0)?;
75    /// Performs the 'normalY' operation.
76    tbl.set("normalY", hit.normal.1)?;
77    /// The 'toi' field value exposed to Lua scripts.
78    tbl.set("toi", hit.toi)?;
79    Ok(tbl)
80}
