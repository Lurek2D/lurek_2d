3use mlua::prelude::*;
4/// Defines the lurek type behavior required by this module.
5pub trait LurekType {
6    const TYPE_NAME: &'static str;
7    const TYPE_HIERARCHY: &'static [&'static str];
8}
9/// Adds type methods. This function is part of the public API.
10pub fn add_type_methods<'lua, T, M>(methods: &mut M)
11where
12    T: LurekType + LuaUserData,
13    M: LuaUserDataMethods<'lua, T>,
14{
15    methods.add_method("type", |_, _, ()| Ok(T::TYPE_NAME));
16    methods.add_method("typeOf", |_, _, name: String| {
17        Ok(T::TYPE_HIERARCHY.contains(&name.as_str()))
18    });
19    methods.add_meta_method(LuaMetaMethod::ToString, |_, _, ()| {
20        Ok(format!("{}()", T::TYPE_NAME))
21    });
22}
