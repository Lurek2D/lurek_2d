25/// Lua-visible wrapper around the biome classification engine, used to assign biome types based on height, moisture, and temperature.
26pub struct LuaBiomeClassifier(BiomeClassifier);
27impl LuaUserData for LuaBiomeClassifier {
28    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
29        // -- classify --
30        /// Classify a single point into a biome type based on its environmental parameters.
31        /// @param | height | number | Elevation value (0.0–1.0) of the terrain point.
32        /// @param | moisture | number | Moisture level (0.0–1.0) at the point.
33        /// @param | temperature | number | Temperature value (0.0–1.0) at the point.
34        /// @return | string | Biome name such as "ocean", "desert", "grassland", "taiga", etc.
35        methods.add_method("classify", |_, this, (h, m, t): (f32, f32, f32)| {
36            Ok(this.0.classify(h, m, t).as_str())
37        });
38        // -- classifyMap --
39        /// Classify an entire grid of points into biome types in bulk.
40        /// @param | width | integer | Grid width in cells.
41        /// @param | height | integer | Grid height in cells.
42        /// @param | heights | table | Flat array of height values (length = width*height).
43        /// @param | moisture | table | Flat array of moisture values (length = width*height).
44        /// @param | temperature | table? | Optional flat array of temperature values. If omitted, temperature is ignored.
45        /// @return | string[] | Biome name strings (length = width*height).
46        methods.add_method(
47            "classifyMap",
48            |lua, this, (width, height, ht, mt, tt): (u32, u32, LuaTable, LuaTable, Option<LuaTable>)| {
49                let n = (width * height) as usize;
50                let heights: Vec<f32> = (1..=n).filter_map(|i| ht.get::<_, f32>(i).ok()).collect();
51                let moisture: Vec<f32> = (1..=n).filter_map(|i| mt.get::<_, f32>(i).ok()).collect();
52                let temperature: Vec<f32> = if let Some(t) = tt {
53                    (1..=n).filter_map(|i| t.get::<_, f32>(i).ok()).collect()
54                } else {
55                    Vec::new()
56                };
57                let biomes = this.0.classify_map(width, height, &heights, &moisture, &temperature);
58                let out = lua.create_table()?;
59                for (i, b) in biomes.iter().enumerate() {
60                    out.set(i + 1, b.as_str())?;
61                }
62                Ok(out)
63            },
64        );
65        // -- type --
66        /// Returns the type name of this object.
67        /// @return | string | Always returns "LBiomeClassifier".
68        methods.add_method("type", |_, _, ()| Ok("LBiomeClassifier"));
69        // -- typeOf --
70        /// Check whether this object matches a given type name.
71        /// @param | name | string | Type name to test (e.g. "LBiomeClassifier" or "Object").
72        /// @return | boolean | True if the object is of the specified type.
73        methods.add_method("typeOf", |_, _, name: String| {
74            Ok(name == "LBiomeClassifier" || name == "LObject")
75        });
76    }
77}

136/// Resolves a Lua noise kind string into a noise kind enum.
137fn resolve_noise_kind(name: &str) -> NoiseKind {
138    match name.to_lowercase().as_str() {
139        "simplex" => NoiseKind::Simplex,
140        _ => NoiseKind::Perlin,
141    }
142}
143/// Resolves a Lua distance type string into a distance type enum.
144fn resolve_dist_type(name: &str) -> DistType {
145    match name.to_lowercase().as_str() {
146        "manhattan" => DistType::Manhattan,
147        "chebyshev" => DistType::Chebyshev,
148        _ => DistType::Euclidean,
149    }
150}
151/// Resolves a Lua fractal type string into a fractal type enum.
152fn resolve_fractal_type(name: &str) -> FractalType {
153    match name.to_lowercase().as_str() {
154        "ridged" => FractalType::Ridged,
155        "turbulence" => FractalType::Turbulence,
156        _ => FractalType::Fbm,
157    }
158}

474/// Registers the `lurek.procgen` module and all its functions on the given Lua table.
475pub fn register(lua: &Lua, luna: &LuaTable, _state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
476    let tbl = lua.create_table()?;
477    // -- cellularAutomata --
478    /// Generate a cave or organic map using cellular automata rules.
479    /// @param | width | integer | Grid width in cells.
480    /// @param | height | integer | Grid height in cells.
481    /// @param | opts | table? | Options: fill (0.0–1.0 initial fill ratio), iterations, birth threshold, survive threshold, seed.
482    /// @return | integer[] | Flat array of cell values (0=empty, 1=wall) with length width×height.
483    tbl.set(
484        "cellularAutomata",
485        lua.create_function(|lua, (w, h, opts): (u32, u32, Option<LuaTable>)| {
486            let cfg = opts
487                .map(|t| CellularOpts::from_lua_table(&t))
488                .transpose()?
489                .unwrap_or_default();
490            let data = cellular_automata(w, h, &cfg);
491            let out = lua.create_table()?;
492            for (i, v) in data.iter().enumerate() {
493                out.set(i + 1, *v)?;
494            }
495            Ok(out)
496        })?,
497    )?;
498    // -- floodFill --
499    /// Flood-fill a grid from a starting cell, marking all connected cells that pass a threshold test.
500    /// @param | data | table | Flat array of u8 cell values (length = width*height).
501    /// @param | width | number | Grid width.
502    /// @param | height | number | Grid height.
503    /// @param | startX | number | Start column (0-based).
504    /// @param | startY | number | Start row (0-based).
505    /// @param | threshold | number? | Value threshold (default 128).
506    /// @param | above | boolean? | If true, fill cells >= threshold; if false (default), fill cells < threshold.
507    /// @return | integer[] | Flat array of fill values (1=filled, 0=not filled) with length width×height.
508    tbl.set(
509        "floodFill",
510        lua.create_function(
511            |lua,
512             (data_tbl, w, h, sx, sy, threshold, above): (
513                LuaTable,
514                u32,
515                u32,
516                u32,
517                u32,
518                Option<u8>,
519                Option<bool>,
520            )| {
521                let mut data: Vec<u8> = Vec::with_capacity((w * h) as usize);
522                for v in data_tbl.sequence_values::<u8>() {
523                    data.push(v?);
524                }
525                let result = flood_fill(
526                    &data,
527                    w,
528                    h,
529                    sx,
530                    sy,
531                    threshold.unwrap_or(128),

1566    // -- simplexNoise --
1567    /// Sample a 2D or 3D simplex noise value at a given point.
1568    /// @param | x | number | X coordinate.
1569    /// @param | y | number | Y coordinate.
1570    /// @param | z | number? | Z coordinate for 3D noise.

1582    // -- newBiomeClassifier --
1583    /// Create a BiomeClassifier object with custom threshold rules for mapping height/moisture/temperature to biome types.
1584    /// @param | opts | table? | Optional rules: ocean_threshold, coast_threshold, mountain_threshold, ice_cap_threshold, cold_temperature, warm_temperature, dry_moisture, wet_moisture.

1586    tbl.set(

1589            let rules = opts
1590                .map(|t| BiomeRules::from_lua_table(&t))
1591                .transpose()?
1592                .unwrap_or_default();
1593            lua.create_userdata(LuaBiomeClassifier(BiomeClassifier::new(rules)))
1594        })?,
1595    )?;
1596    // -- biomeColor --
1597    /// Get the default RGBA display color for a biome type name. Useful for minimap or debug visualization.
1598    /// @param | name | string | Biome name (e.g. "ocean", "desert", "taiga").
1599    /// @return | number | Red component (0–255).
1600    /// @return | number | Green component (0–255).

1608            Ok((r, g, b, a))
1609        })?,
1610    )?;
1611    /// Performs the 'procgen' operation.
1612    luna.set("procgen", tbl)?;
