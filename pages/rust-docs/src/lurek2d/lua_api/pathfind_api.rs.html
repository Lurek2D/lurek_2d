19fn one_based_to_zero_based(value: u32, label: &str) -> LuaResult<u32> {
20    value.checked_sub(1).ok_or_else(|| {
21        LuaError::RuntimeError(format!(
22            "{} must be >= 1 for one-based coordinates",
23            label
24        ))
25    })
26}

28/// Converts zero-based Rust waypoints into one-based Lua point tables.
29fn waypoints_to_lua<'a>(lua: &'a Lua, path: &[Waypoint]) -> LuaResult<LuaTable<'a>> {
30    let tbl = lua.create_table()?;
31    for (i, wp) in path.iter().enumerate() {
32        let entry = lua.create_table()?;
33        /// The 'x' field value exposed to Lua scripts.
34        entry.set("x", wp.x + 1)?;
35        /// The 'y' field value exposed to Lua scripts.
36        entry.set("y", wp.y + 1)?;
37        tbl.set(i + 1, entry)?;
38    }
39    Ok(tbl)
40}
41/// Converts one-based Lua point tables into zero-based Rust waypoints.
42fn lua_to_waypoints(tbl: &LuaTable) -> LuaResult<Vec<Waypoint>> {
43    let mut waypoints = Vec::new();
44    for pair in tbl.clone().sequence_values::<LuaTable>() {
45        let entry = pair?;
46        let x: u32 = entry.get("x")?;
47        let y: u32 = entry.get("y")?;
48        waypoints.push(Waypoint {
49            x: one_based_to_zero_based(x, "x")?,
50            y: one_based_to_zero_based(y, "y")?,
51        });
52    }
53    Ok(waypoints)
54}

1130        // -- addPolygon --

1141            }

1169        /// @param | gx | number | Goal X in world coordinates.

1172        /// @field | x | number | X.
1173        /// @field | y | number | Y.
1174        methods.add_method(
1175            "findPath",

1188                        }
1189                        Ok(LuaValue::Table(out))
1190                    }

1192                }
1193            },
1194        );
