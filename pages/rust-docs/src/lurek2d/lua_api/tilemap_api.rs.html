20/// Converts a one-based Lua index into a zero-based `usize` tile index.
21fn one_based_usize(name: &str, val: usize) -> LuaResult<usize> {
22    val.checked_sub(1)
23        .ok_or_else(|| mlua::Error::RuntimeError(format!("{name} must be >= 1 (got {val})")))
24}
25/// Converts a one-based Lua index into a zero-based `u32` tile index.

27    val.checked_sub(1)
28        .ok_or_else(|| mlua::Error::RuntimeError(format!("{name} must be >= 1 (got {val})")))
29}

790        /// @param | layer | integer | Layer index (1-based).

843                    for (cb_gid, key) in callbacks.iter() {
844                        if *cb_gid == gid {
845                            let func: LuaFunction = lua.registry_value(key)?;
846                            func.call::<_, ()>((wx, wy, tx, ty))?;
847                        }
848                    }
849                }
850                Ok(())
851            },
852        );
853        // -- onTileStep --
854        /// Registers a callback invoked each frame an entity remains on a tile with the given GID.
855        /// @param | gid | integer | Global tile ID to watch for.

873                this.tile_exit_callbacks.borrow_mut().insert(gid, key);
874                Ok(())
875            },
876        );
877        // -- fireTileStep --
878        /// Manually fires the tile-step callback for a specific GID and entity at a tile position.
879        /// @param | gid | integer | Global tile ID.
880        /// @param | entity | table | Entity table to pass to the callback.
881        /// @param | tx | integer | Tile column.
882        /// @param | ty | integer | Tile row.
883        methods.add_method(
884            "fireTileStep",
885            |lua, this, (gid, entity, tx, ty): (u32, LuaTable, i32, i32)| {
886                if let Some(key) = this.tile_step_callbacks.borrow().get(&gid) {
887                    let func: mlua::Function = lua.registry_value(key)?;
888                    let _: () = func.call((entity, tx, ty))?;
889                }
890                Ok(())
891            },
892        );
893        // -- fireTileExit --
894        /// Manually fires the tile-exit callback for a specific GID and entity at a tile position.
895        /// @param | gid | integer | Global tile ID.
896        /// @param | entity | table | Entity table to pass to the callback.
897        /// @param | tx | integer | Tile column.
898        /// @param | ty | integer | Tile row.
899        methods.add_method(
900            "fireTileExit",
901            |lua, this, (gid, entity, tx, ty): (u32, LuaTable, i32, i32)| {
902                if let Some(key) = this.tile_exit_callbacks.borrow().get(&gid) {
903                    let func: mlua::Function = lua.registry_value(key)?;
904                    let _: () = func.call((entity, tx, ty))?;
905                }
906                Ok(())
907            },
908        );
909        // -- type --
910        /// Returns the type name of this userdata.
911        /// @return | string | Always `"LTileMap"`.
912        methods.add_method("type", |_, _, ()| Ok("LTileMap"));
913        // -- typeOf --
914        /// Checks whether this object matches the given type name.
915        /// @param | name | string | Type name to check against.
916        /// @return | boolean | True if `name` is `"LTileMap"` or `"Object"`.
917        methods.add_method("typeOf", |_, _, name: String| {
918            Ok(name == "LTileMap" || name == "LObject")
919        });
920    }

1704            Ok(this.inner.borrow().get_step_count())
1705        });
1706        // -- addStep --
1707        /// Appends a generation step. The step table must have a `type` field and optional parameters.
1708        /// @param | stepDef | table | Step definition with `type` and parameters like `x`, `y`, `w`, `h`, `gid`, `chance`, etc.
1709        methods.add_method("addStep", |_, this, step_def: LuaTable| {
1710            let step_type_str: String = step_def.get("type")?;
1711            let st = match step_type_str.as_str() {
1712                "fillRandom"  => StepType::FillRandom,
1713                "placeBlock"  => StepType::PlaceBlock,
1714                "placeRandom" => StepType::PlaceRandom,
1715                "placeLine"   => StepType::PlaceLine,
1716                "floodFill"   => StepType::FloodFill,
1717                "fillArea"    => StepType::FillArea,
1718                "drawPath"    => StepType::DrawPath,
1719                "fillRect"    => StepType::FillRect,

1723                        other
1724                    )))
1725                }
1726            };
1727            let get_u32_field = |tbl: &LuaTable, key: &str| -> u32 {
1728                match tbl.get::<_, LuaValue>(key) {

1733            };

1738                    _ => 1.0,
1739                }
1740            };
1741            let get_i32_field = |tbl: &LuaTable, key: &str, default: i32| -> i32 {
1742                match tbl.get::<_, LuaValue>(key) {
1743                    Ok(LuaValue::Integer(n)) => n as i32,
1744                    Ok(LuaValue::Number(n)) => n as i32,
1745                    _ => default,
1746                }
1747            };
1748            let step = ScriptStep {
1749                step_type: st,
1750                x: get_u32_field(&step_def, "x"),
1751                y: get_u32_field(&step_def, "y"),
1752                width: get_u32_field(&step_def, "w"),
1753                height: get_u32_field(&step_def, "h"),
1754                tile_id: get_u32_field(&step_def, "gid"),
1755                chance: get_f32_field(&step_def, "chance"),
1756                direction: get_u32_field(&step_def, "direction"),
1757                path_width: {
1758                    let v = get_u32_field(&step_def, "pathWidth");

1760                },
1761                repeat_count: {
1762                    let v = get_u32_field(&step_def, "repeatCount");
1763                    if v == 0 { 1 } else { v }
1764                },
1765                count: {
1766                    let v = get_u32_field(&step_def, "count");
1767                    if v == 0 { 1 } else { v }
1768                },
1769                group_index: get_i32_field(&step_def, "groupIndex", -1),
1770                block_index: get_i32_field(&step_def, "blockIndex", -1),
1771                tile_layer: get_u32_field(&step_def, "tileLayer"),
1772                ..Default::default()
1773            };
1774            this.inner.borrow_mut().add_step(step);
1775            Ok(())
1776        });
1777        // -- type --
1778        /// Returns the type name of this userdata.
1779        /// @return | string | Always `"LMapScript"`.
1780        methods.add_method("type", |_, _, ()| Ok("LMapScript"));
1781        // -- typeOf --

1784        /// @return | boolean | True if `name` is `"LMapScript"` or `"Object"`.

1786            Ok(name == "LMapScript" || name == "LObject")
1787        });
