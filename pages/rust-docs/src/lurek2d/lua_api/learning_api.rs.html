11/// Lua handle for a Q-learning table with configurable exploration and learning parameters.
12#[derive(Clone)]
13pub(crate) struct LuaQLearner {
14    /// Shared Q-learning model containing Q-values, episode counters, and tuning parameters.
15    pub(crate) inner: Rc<RefCell<QLearner>>,
16}
17impl LuaUserData for LuaQLearner {
18    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
19        // -- chooseAction --
20        /// Chooses an action for a one-based state index using the learner's exploration policy.
21        /// @param | state | integer | One-based state index.
22        /// @return | integer | One-based chosen action index.
23        methods.add_method("chooseAction", |_, this, state: usize| {
24            Ok(this.inner.borrow().choose_action(state.saturating_sub(1)) + 1)
25        });
26        // -- bestAction --
27        /// Returns the highest-valued action for a one-based state index without exploration.
28        /// @param | state | integer | One-based state index.
29        /// @return | integer | One-based best action index.
30        methods.add_method("bestAction", |_, this, state: usize| {
31            Ok(this.inner.borrow().best_action(state.saturating_sub(1)) + 1)
32        });
33        // -- learn --
34        /// Applies one Q-learning update from a transition and reward.
35        /// @param | state | integer | One-based previous state index.
36        /// @param | action | integer | One-based action index taken in the previous state.
37        /// @param | reward | number | Reward received for the transition.
38        /// @param | next_state | integer | One-based next state index.
39        methods.add_method(
40            "learn",
41            |_, this, (state, action, reward, next_state): (usize, usize, f64, usize)| {
42                this.inner.borrow_mut().learn(
43                    state.saturating_sub(1),
44                    action.saturating_sub(1),
45                    reward,
46                    next_state.saturating_sub(1),
47                );
48                Ok(())
49            },
50        );
51        // -- getQValue --
52        /// Returns the stored Q-value for a one-based state and action pair.
53        /// @param | state | integer | One-based state index.
54        /// @param | action | integer | One-based action index.
55        /// @return | number | Current Q-value.
56        methods.add_method("getQValue", |_, this, (state, action): (usize, usize)| {
57            Ok(this
58                .inner
59                .borrow()
60                .get_q(state.saturating_sub(1), action.saturating_sub(1)))
61        });
62        // -- setQValue --
63        /// Sets the stored Q-value for a one-based state and action pair.
64        /// @param | state | integer | One-based state index.
65        /// @param | action | integer | One-based action index.
66        /// @param | value | number | Q-value to store.
67        methods.add_method(
68            "setQValue",
69            |_, this, (state, action, value): (usize, usize, f64)| {
70                this.inner.borrow_mut().set_q(
71                    state.saturating_sub(1),
72                    action.saturating_sub(1),
73                    value,
74                );
75                Ok(())
76            },
77        );
78        // -- endEpisode --
79        /// Decays epsilon and increments the episode count.
80        methods.add_method("endEpisode", |_, this, ()| {
81            this.inner.borrow_mut().end_episode();
82            Ok(())
83        });
84        // -- getEpisodeCount --
85        /// Returns the total number of episodes completed so far.
86        /// @return | integer | Episode count.
87        methods.add_method("getEpisodeCount", |_, this, ()| {
88            Ok(this.inner.borrow().episode_count)
89        });
90        // -- getStateCount --
91        /// Returns the number of states represented by this learner.
92        /// @return | integer | State count.
93        methods.add_method("getStateCount", |_, this, ()| {
94            Ok(this.inner.borrow().state_count)
95        });
96        // -- getActionCount --
97        /// Returns the number of actions represented by this learner.
98        /// @return | integer | Action count.
99        methods.add_method("getActionCount", |_, this, ()| {
100            Ok(this.inner.borrow().action_count)
101        });
102        // -- setLearningRate --
103        /// Sets the Q-learning alpha learning rate.
104        /// @param | v | number | Learning rate used by future updates.
105        methods.add_method("setLearningRate", |_, this, v: f64| {
106            this.inner.borrow_mut().alpha = v;
107            Ok(())
108        });
109        // -- getLearningRate --
110        /// Returns the Q-learning alpha learning rate.
111        /// @return | number | Current learning rate.
112        methods.add_method("getLearningRate", |_, this, ()| {
113            Ok(this.inner.borrow().alpha)
114        });
115        // -- setDiscountFactor --
116        /// Sets the Q-learning gamma discount factor.
117        /// @param | v | number | Discount factor used by future updates.
118        methods.add_method("setDiscountFactor", |_, this, v: f64| {
119            this.inner.borrow_mut().gamma = v;
120            Ok(())
121        });
122        // -- getDiscountFactor --
123        /// Returns the Q-learning gamma discount factor.
124        /// @return | number | Current discount factor.
125        methods.add_method("getDiscountFactor", |_, this, ()| {
126            Ok(this.inner.borrow().gamma)
127        });
128        // -- setExplorationRate --
129        /// Sets the exploration rate used by action selection.
130        /// @param | v | number | Exploration probability for future `chooseAction` calls.
131        methods.add_method("setExplorationRate", |_, this, v: f64| {
132            this.inner.borrow_mut().epsilon = v;
133            Ok(())
134        });
135        // -- getExplorationRate --
136        /// Returns the exploration rate used by action selection.
137        /// @return | number | Current exploration rate.
138        methods.add_method("getExplorationRate", |_, this, ()| {
139            Ok(this.inner.borrow().epsilon)
140        });
141        // -- setExplorationDecay --
142        /// Sets the exploration decay multiplier applied across episodes.
143        /// @param | v | number | Exploration decay multiplier.
144        methods.add_method("setExplorationDecay", |_, this, v: f64| {
145            this.inner.borrow_mut().epsilon_decay = v;
146            Ok(())
147        });
148        // -- getExplorationDecay --
149        /// Returns the exploration decay multiplier.
150        /// @return | number | Current exploration decay multiplier.
151        methods.add_method("getExplorationDecay", |_, this, ()| {
152            Ok(this.inner.borrow().epsilon_decay)
153        });
154        // -- serialize --
155        /// Serializes the Q-learner state to a JSON string.
156        /// @return | string | JSON representation of this learner.
157        methods.add_method("serialize", |_, this, ()| {
158            Ok(this.inner.borrow().serialize())
159        });
160        // -- deserialize --
161        /// Replaces the Q-learner state from a JSON string.
162        /// @param | json | string | JSON data previously produced by `serialize`.
163        methods.add_method("deserialize", |_, this, json: String| {
164            this.inner
165                .borrow_mut()
166                .deserialize(&json)
167                .map_err(LuaError::RuntimeError)?;
168            Ok(())
169        });
170        // -- type --
171        /// Returns the Lua-visible type name for this Q-learner handle.
172        /// @return | string | The string `LQLearner`.
173        methods.add_method("type", |_, _, ()| Ok("LQLearner"));
174        // -- typeOf --
175        /// Returns whether this Q-learner handle matches a supported type name.
176        /// @param | name | string | Type name to compare against `LQLearner` and `Object`.
177        /// @return | boolean | True when the supplied type name matches this handle.
178        methods.add_method("typeOf", |_, _, name: String| {
179            Ok(name == "LQLearner" || name == "LObject")
180        });
181    }
182}

184/// Lua handle for a feed-forward neural network.
185#[derive(Clone)]
186pub(crate) struct LuaNeuralNet {
187    /// Shared neural network containing layers and flattened weights.
188    pub(crate) inner: Rc<RefCell<NeuralNet>>,
189}
190impl LuaUserData for LuaNeuralNet {
191    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
192        // -- addLayer --
193        /// Adds a neural network layer with an activation function.
194        /// @param | inputs | integer | Input count for the layer.
195        /// @param | outputs | integer | Output count for the layer.
196        /// @param | activation | string | Activation name such as `relu`, `sigmoid`, `tanh`, `linear`, or `softmax`.
197        methods.add_method_mut(
198            "addLayer",
199            |_, this, (inputs, outputs, activation): (usize, usize, String)| {
200                let act = Activation::from_str(&activation);
201                this.inner.borrow_mut().add_layer(inputs, outputs, act);
202                Ok(())
203            },
204        );
205        // -- forward --
206        /// Runs a forward pass and returns output values.
207        /// @param | input | table | Array of numeric input values.
208        /// @return | number[] | Numeric output values.
209        methods.add_method("forward", |lua, this, input: Vec<f32>| {
210            let out = this.inner.borrow().forward(&input);
211            let t = lua.create_table()?;
212            for (i, v) in out.into_iter().enumerate() {
213                t.raw_set(i + 1, v)?;
214            }
215            Ok(t)
216        });
217        // -- setWeights --
218        /// Replaces the network weights from a flat numeric array.
219        /// @param | weights | table | Flat array of numeric weights in engine layer order.
220        /// @return | boolean | True when the supplied weight slice matches the network shape.
221        methods.add_method_mut("setWeights", |_, this, weights: Vec<f32>| {
222            Ok(this.inner.borrow_mut().set_weights(&weights))
223        });
224        // -- getWeights --
225        /// Returns the network weights as a flat numeric array.
226        /// @return | number[] | Numeric weights in engine layer order.
227        methods.add_method("getWeights", |lua, this, ()| {
228            let w = this.inner.borrow().get_weights();
229            let t = lua.create_table()?;
230            for (i, v) in w.into_iter().enumerate() {
231                t.raw_set(i + 1, v)?;
232            }
233            Ok(t)
234        });
235        // -- paramCount --
236        /// Returns the total number of trainable parameters.
237        /// @return | integer | Parameter count.
238        methods.add_method("paramCount", |_, this, ()| {
239            Ok(this.inner.borrow().param_count() as i64)
240        });
241        // -- layerCount --
242        /// Returns the number of layers in the network.
243        /// @return | integer | Layer count.
244        methods.add_method("layerCount", |_, this, ()| {
245            Ok(this.inner.borrow().layer_count() as i64)
246        });
247        // -- type --
248        /// Returns the Lua-visible type name for this neural network handle.
249        /// @return | string | The string `LNeuralNet`.
250        methods.add_method("type", |_, _, ()| Ok("LNeuralNet"));
251        // -- typeOf --
252        /// Returns whether this neural network handle matches a supported type name.
253        /// @param | name | string | Type name to compare against `LNeuralNet` and `Object`.
254        /// @return | boolean | True when the supplied type name matches this handle.
255        methods.add_method("typeOf", |_, _, name: String| {
256            Ok(name == "LNeuralNet" || name == "LObject")
257        });
258    }

261/// Lua handle for a floating-point genetic algorithm population.
262#[derive(Clone)]
263pub(crate) struct LuaGeneticAlgorithm {
264    /// Shared genetic algorithm population and generation state.
265    pub(crate) inner: Rc<RefCell<GeneticAlgorithm>>,
266}
267impl LuaUserData for LuaGeneticAlgorithm {
268    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
269        // -- evolve --
270        /// Advances the genetic algorithm by one generation.
271        methods.add_method_mut("evolve", |_, this, ()| {
272            this.inner.borrow_mut().evolve();
273            Ok(())
274        });
275        // -- generation --
276        /// Returns the current generation index.
277        /// @return | integer | Current generation count.
278        methods.add_method("generation", |_, this, ()| {
279            Ok(this.inner.borrow().generation as i64)
280        });
281        // -- popSize --
282        /// Returns the population size. This method is available to Lua scripts.
283        /// @return | integer | Current population size.
284        methods.add_method("popSize", |_, this, ()| {
285            Ok(this.inner.borrow().pop_size() as i64)
286        });
287        // -- setFitness --
288        /// Sets the fitness value for a chromosome by zero-based index.
289        /// @param | idx | integer | Zero-based chromosome index.
290        /// @param | fitness | number | Fitness value used by the next evolution step.
291        methods.add_method_mut("setFitness", |_, this, (idx, fitness): (usize, f32)| {
292            if let Some(c) = this.inner.borrow_mut().population.get_mut(idx) {
293                c.fitness = fitness;
294            }
295            Ok(())
296        });
297        // -- getGenes --
298        /// Returns the genes for a chromosome by zero-based index.
299        /// @param | idx | integer | Zero-based chromosome index.
300        /// @return | number[] | Gene values, or an empty table for an invalid index.
301        methods.add_method("getGenes", |lua, this, idx: usize| {
302            let ga = this.inner.borrow();
303            let t = lua.create_table()?;
304            if let Some(c) = ga.population.get(idx) {
305                for (i, &g) in c.genes.iter().enumerate() {
306                    t.raw_set(i + 1, g)?;
307                }
308            }
309            Ok(t)
310        });
311        // -- bestGenes --
312        /// Returns the genes for the best chromosome in the population.
313        /// @return | number[] | Array of best gene values, or an empty array when the population has no best chromosome.
314        methods.add_method("bestGenes", |lua, this, ()| {
315            let ga = this.inner.borrow();
316            let t = lua.create_table()?;
317            if let Some(best) = ga.best() {
318                for (i, &g) in best.genes.iter().enumerate() {
319                    t.raw_set(i + 1, g)?;
320                }
321            }
322            Ok(t)
323        });
324        // -- type --
325        /// Returns the Lua-visible type name for this genetic algorithm handle.
326        /// @return | string | The string `LGeneticAlgorithm`.
327        methods.add_method("type", |_, _, ()| Ok("LGeneticAlgorithm"));
328        // -- typeOf --
329        /// Returns whether this genetic algorithm handle matches a supported type name.
330        /// @param | name | string | Type name to compare against `LGeneticAlgorithm` and `Object`.
331        /// @return | boolean | True when the supplied type name matches this handle.
332        methods.add_method("typeOf", |_, _, name: String| {
333            Ok(name == "LGeneticAlgorithm" || name == "LObject")
334        });
335    }
336}

471/// Registers the `lurek.learning` API table with the Lua VM.
472pub fn register(lua: &Lua, lurek: &LuaTable, _state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
473    let tbl = lua.create_table()?;
474    // -- newQLearner --
475    /// Creates a Q-learner with fixed state and action counts.
476    /// @param | sc | integer | Number of discrete states.
477    /// @param | ac | integer | Number of discrete actions.
478    /// @return | LQLearner | New Q-learner handle.
479    tbl.set(
480        "newQLearner",
481        lua.create_function(|_, (sc, ac): (usize, usize)| {
482            Ok(LuaQLearner {
483                inner: Rc::new(RefCell::new(QLearner::new(sc, ac))),
484            })
485        })?,
486    )?;
487    // -- newNeuralNet --
488    /// Creates an empty feed-forward neural network.
489    /// @return | LNeuralNet | New neural network handle.
490    tbl.set(
491        "newNeuralNet",
492        lua.create_function(|_, ()| {
493            Ok(LuaNeuralNet {
494                inner: Rc::new(RefCell::new(NeuralNet::new())),
495            })
496        })?,
497    )?;
498    // -- newGeneticAlgorithm --
499    /// Creates a genetic algorithm population with fixed chromosome length.
500    /// @param | pop_size | integer | Number of chromosomes in the population.
501    /// @param | gene_count | integer | Number of floating-point genes per chromosome.
502    /// @param | seed | integer | Random seed used for population initialization and evolution.
503    /// @return | LGeneticAlgorithm | New genetic algorithm handle.
504    tbl.set(
505        "newGeneticAlgorithm",
506        lua.create_function(|_, (pop_size, gene_count, seed): (usize, usize, u64)| {
507            Ok(LuaGeneticAlgorithm {
508                inner: Rc::new(RefCell::new(GeneticAlgorithm::new(
509                    pop_size, gene_count, seed,
510                ))),
511            })
512        })?,
513    )?;
514    // -- newBandit --
515    /// Creates a multi-armed bandit with a named selection strategy.
516    /// @param | arm_count | integer | Number of selectable arms.
517    /// @param | strategy | string | Strategy name such as `ucb1`, `thompson`, or an epsilon-greedy fallback.
518    /// @param | epsilon | number | Exploration probability used by epsilon-greedy strategy and clamped to `[0, 1]`.
519    /// @param | seed | integer | Random seed used by the bandit.
520    /// @return | LBandit | New bandit handle.
521    tbl.set(
522        "newBandit",
523        lua.create_function(
524            |_, (arm_count, strategy, epsilon, seed): (usize, String, f32, u64)| {
525                let strat = match strategy.as_str() {
526                    "ucb1" => BanditStrategy::UCB1,
527                    "thompson" | "thompson_sampling" => BanditStrategy::ThompsonSampling,
528                    _ => BanditStrategy::EpsilonGreedy {
529                        epsilon: epsilon.clamp(0.0, 1.0),
530                    },
531                };
532                Ok(LuaBandit {
533                    inner: Rc::new(RefCell::new(Bandit::new(arm_count, strat, seed))),
534                })
535            },
536        )?,
537    )?;
538    // -- newNeuroevolution --
539    /// Creates a neuroevolution population from a layer specification table.
540    /// @param | layer_spec | table | Array of layer tables with `inputs`, `outputs`, and optional `activation` fields.
541    /// @param | pop_size | integer | Number of chromosomes in the population.
542    /// @param | seed | integer | Random seed used for population initialization and evolution.
543    /// @return | LNeuroevolution | New neuroevolution handle.
544    tbl.set(
545        "newNeuroevolution",
546        lua.create_function(|_, (layer_spec, pop_size, seed): (LuaTable, usize, u64)| {
547            let mut spec: Vec<(usize, usize, &'static str)> = Vec::new();
548            for i in 1..=layer_spec.raw_len() {
549                let entry: LuaTable = layer_spec.raw_get(i)?;
550                let in_size: usize = entry.raw_get("inputs").unwrap_or(1);
551                let out_size: usize = entry.raw_get("outputs").unwrap_or(1);
552                let act_str: String = entry
553                    .raw_get("activation")
554                    .unwrap_or_else(|_| "relu".into());
555                let act: &'static str = match act_str.as_str() {
556                    "sigmoid" => "sigmoid",
557                    "tanh" => "tanh",
