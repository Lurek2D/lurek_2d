18/// Lua-side handle for a named globe stored inside a shared registry.
19pub struct LuaGlobe {
20    /// Shared globe registry containing all named globes.
21    reg: Arc<Mutex<GlobeRegistry>>,
22    /// Registry key for the globe represented by this Lua handle.
23    name: String,
24    #[allow(dead_code)]
25    /// Shared runtime state retained for future renderer integration.
26    state: Rc<RefCell<SharedState>>,
27}
28impl LuaGlobe {
29    /// Reads the named globe from the registry and maps it to a Lua result.
30    fn with<R>(&self, f: impl FnOnce(&Globe) -> R) -> LuaResult<R> {
31        let guard = self
32            .reg
33            .lock()
34            .map_err(|e| mlua::Error::RuntimeError(format!("globe registry lock poisoned: {e}")))?;
35        guard
36            .get(&self.name)
37            .map(f)
38            .ok_or_else(|| mlua::Error::RuntimeError(format!("globe '{}' not found", self.name)))
39    }
40    /// Mutates the named globe in the registry and maps it to a Lua result.
41    fn with_mut<R>(&self, f: impl FnOnce(&mut Globe) -> R) -> LuaResult<R> {
42        let mut guard = self
43            .reg
44            .lock()
45            .map_err(|e| mlua::Error::RuntimeError(format!("globe registry lock poisoned: {e}")))?;
46        guard
47            .get_mut(&self.name)
48            .map(f)
49            .ok_or_else(|| mlua::Error::RuntimeError(format!("globe '{}' not found", self.name)))
50    }
51}

56        /// Adds a province described by id, centroid, vertices, neighbors, and optional base color.

58        /// @return | boolean | True when the province was accepted by the globe.
59        methods.add_method_mut("addProvince", |_, this, p: LuaTable| {
60            let id: u32 = p.get("id")?;
61            let ct: LuaTable = p.get("centroid")?;
62            let centroid = (ct.get::<_, f32>(1)?, ct.get::<_, f32>(2)?);
63            let verts_tbl: LuaTable = p.get::<_, LuaTable>("vertices").map_err(|_| {
64                mlua::Error::RuntimeError(format!("province {id}: 'vertices' field is required"))
65            })?;
66            let mut vertices = Vec::new();
67            for vt in verts_tbl.sequence_values::<LuaTable>() {
68                let vt = vt?;
69                vertices.push((vt.get::<_, f32>(1)?, vt.get::<_, f32>(2)?));
70            }
71            let neighbors: Vec<RegionId> = p
72                .get::<_, LuaTable>("neighbors")
73                .map(|t| t.sequence_values::<u32>().map(|r| r.map(RegionId)).collect::<LuaResult<Vec<_>>>())
74                .unwrap_or_else(|_| Ok(vec![]))?;

80                    ct.get::<_, f32>(3).unwrap_or(0.5),
81                    ct.get::<_, f32>(4).unwrap_or(1.0),
82                ]
83            } else {
84                [0.5, 0.5, 0.5, 1.0]
85            };
86            let region = Region::with_data(RegionId(id), centroid, vertices, neighbors, base_color);

792                        visible: true,

796                })
797            },
798        );
799        // -- removeArc --
800        /// Removes an arc by id. This method is available to Lua scripts.
801        /// @param | id | integer | Arc id.
802        /// @return | boolean | True when an arc was removed.
803        methods.add_method_mut("removeArc", |_, this, id: u32| {
804            this.with_mut(|g| g.remove_arc(id))
805        });
806        // -- getName --
807        /// Returns the registry name of this globe.
808        /// @return | string | Globe registry name.
809        methods.add_method("getName", |_, this, ()| Ok(this.name.clone()));
810        methods.add_meta_method(LuaMetaMethod::ToString, |_, this, ()| {

834/// Provides Lua methods for managing globe registry entries.
835impl LuaUserData for LuaGlobeRegistry {
836    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
837        // -- new --
838        /// Creates a named globe with optional specification fields.
839        /// @param | name | string | Globe registry name.
840        /// @param | spec_tbl | table? | Globe specification table.
841        /// @return | LGlobe | New globe handle.
842        methods.add_method_mut(
843            "new",
844            |_, this, (name, spec_tbl): (String, Option<LuaTable>)| {
845                let spec = parse_globe_spec(spec_tbl);
846                {
847                    let mut guard = this.reg.lock().map_err(|e| {
848                        mlua::Error::RuntimeError(format!("registry lock poisoned: {e}"))
849                    })?;
850                    guard.create(name.clone(), spec);
851                }
852                Ok(LuaGlobe {
853                    reg: this.reg.clone(),
854                    name,
855                    state: this.state.clone(),
856                })
857            },
858        );
859        // -- get --
860        /// Returns a globe handle by registry name.
861        /// @param | name | string | Globe registry name.
862        /// @return | LGlobe | Globe handle, or nil when no globe exists with that name.
863        methods.add_method("get", |_, this, name: String| {
864            let exists = {
865                let guard = this.reg.lock().map_err(|e| {
866                    mlua::Error::RuntimeError(format!("registry lock poisoned: {e}"))
867                })?;
868                guard.get(&name).is_some()
869            };
870            if exists {
871                Ok(Some(LuaGlobe {
872                    reg: this.reg.clone(),
873                    name,
874                    state: this.state.clone(),
875                }))
876            } else {
877                Ok(None)
878            }
879        });
880        // -- remove --
881        /// Removes a globe from the registry by name.
882        /// @param | name | string | Globe registry name.
883        /// @return | boolean | True when a globe was removed.
884        methods.add_method_mut("remove", |_, this, name: String| {
885            let mut guard = this
886                .reg
887                .lock()
888                .map_err(|e| mlua::Error::RuntimeError(format!("registry lock poisoned: {e}")))?;
889            Ok(guard.remove(&name).is_some())
890        });
891        // -- names --
892        /// Returns all globe names currently stored in this registry.
893        /// @return | string[] | Globe names.
894        methods.add_method("names", |lua, this, ()| {
895            let guard = this
896                .reg
897                .lock()
898                .map_err(|e| mlua::Error::RuntimeError(format!("registry lock poisoned: {e}")))?;
899            let names = guard.names();
900            let t = lua.create_table()?;
901            for (i, n) in names.iter().enumerate() {
902                t.set(i + 1, n.clone())?;
903            }
904            Ok(t)
905        });
906        // -- type --
907        /// Returns the Lua-visible type name for this globe registry handle.
908        /// @return | string | The string `LGlobeRegistry`.
909        methods.add_method("type", |_, _, ()| Ok("LGlobeRegistry"));
910        // -- typeOf --
911        /// Returns whether this registry handle matches a supported type name.
912        /// @param | name | string | Type name to compare against `LGlobeRegistry` and `Object`.
913        /// @return | boolean | True when the supplied type name matches this handle.
914        methods.add_method("typeOf", |_, _, name: String| {
915            Ok(name == "LGlobeRegistry" || name == "LObject")
916        });
917    }
918}
919/// Parses optional Lua table fields into a globe specification.
920fn parse_globe_spec(tbl: Option<LuaTable>) -> GlobeSpec {
921    let mut spec = GlobeSpec::default();
922    if let Some(t) = tbl {
923        if let Ok(v) = t.get::<_, f32>("radius") {
924            spec.radius = v;
925        }
926        if let Ok(v) = t.get::<_, f32>("axial_tilt_deg") {
927            spec.axial_tilt_deg = v;
928        }
929        if let Ok(v) = t.get::<_, f32>("rotation_deg") {
930            spec.rotation_deg = v;
931        }
932        if let Ok(v) = t.get::<_, f32>("time_of_day") {
933            spec.time_of_day = v;
934        }
935        if let Ok(v) = t.get::<_, bool>("render_borders") {
936            spec.render_borders = v;
937        }
938        if let Ok(v) = t.get::<_, f32>("border_width") {
939            spec.border_width = v;
940        }
941        if let Ok(v) = t.get::<_, f32>("ambient") {
942            spec.ambient = v;
943        }
944        if let Ok(v) = t.get::<_, f32>("auto_rotation_deg_per_sec") {
945            spec.auto_rotation_deg_per_sec = v;
946        }
947        if let Ok(v) = t.get::<_, f32>("atmosphere_width") {
948            spec.atmosphere_width = v.max(0.0);
949        }
950        if let Ok(v) = t.get::<_, u8>("border_smoothing_passes") {
951            spec.border_smoothing_passes = v;
952        }
953    }
954    spec
955}

1033        // -- loadFromTOML --

1067        // -- loadFromPNG --

1071        /// @param | spec_tbl | table? | Globe specification table.

1077                    let spec = parse_globe_spec(spec_tbl);
1078                    let provinces =
1079                        loader::load_from_png_file(&png_path).map_err(mlua::Error::RuntimeError)?;
1080                    {
1081                        let mut guard = reg.lock().map_err(|e| {
1082                            mlua::Error::RuntimeError(format!("registry lock poisoned: {e}"))
1083                        })?;
1084                        let globe = guard.create(name.clone(), spec);
1085                        for p in provinces {
1086                            let _ = globe.add_province(p);
1087                        }
1088                    }
1089                    Ok(LuaGlobe {
1090                        reg: reg.clone(),
1091                        name,
1092                        state: s.clone(),
1093                    })
1094                },
1095            )?,
1096        )?;
1097    }
1098    {
1099        let reg = registry.clone();
1100        let s = state.clone();
1101        // -- generateVoronoi --
1102        /// Creates a globe and populates provinces from latitude-longitude seed points.
1103        /// @param | name | string | Globe registry name.
1104        /// @param | seeds_tbl | table | Array table of `{lat, lon}` seed pairs.
1105        /// @param | spec_tbl | table? | Globe specification table.
1106        /// @return | LGlobe | New generated globe handle.
1107        tbl.set(
1108            "generateVoronoi",
1109            lua.create_function(
1110                move |_, (name, seeds_tbl, spec_tbl): (String, LuaTable, Option<LuaTable>)| {
1111                    let spec = parse_globe_spec(spec_tbl);
1112                    let mut seeds = Vec::new();
1113                    for item in seeds_tbl.sequence_values::<LuaTable>() {
1114                        let p = item?;
1115                        let lat: f32 = p.get(1)?;
1116                        let lon: f32 = p.get(2)?;
1117                        seeds.push((lat, lon));
1118                    }
1119                    let provinces = loader::generate_voronoi_provinces(&seeds);
1120                    {
1121                        let mut guard = reg.lock().map_err(|e| {
1122                            mlua::Error::RuntimeError(format!("registry lock poisoned: {e}"))
1123                        })?;
1124                        let globe = guard.create(name.clone(), spec);
1125                        for p in provinces {
1126                            let _ = globe.add_province(p);
1127                        }
1128                    }
1129                    Ok(LuaGlobe {
1130                        reg: reg.clone(),
1131                        name,
1132                        state: s.clone(),
1133                    })
1134                },
1135            )?,
1136        )?;
1137    }
1138    // -- greatCircleDistance --
1139    /// Computes great-circle distance between two latitude-longitude points.
1140    /// @param | la | number | Start latitude in degrees.
1141    /// @param | lo | number | Start longitude in degrees.
1142    /// @param | lb | number | End latitude in degrees.

1149        })?,
1150    )?;
1151    // -- greatCirclePath --
1152    /// Computes sampled latitude-longitude points along a great-circle path.
1153    /// @param | la | number | Start latitude in degrees.
1154    /// @param | lo | number | Start longitude in degrees.
1155    /// @param | lb | number | End latitude in degrees.
1156    /// @param | lo2 | number | End longitude in degrees.
1157    /// @param | n | integer | Number of samples.
1158    /// @return | table | Array table of `{lat, lon}` point tables.
1159    /// @field | lat | number | Lat.
1160    /// @field | lon | number | Lon.
1161    tbl.set(
1162        "greatCirclePath",
1163        lua.create_function(|lua, (la, lo, lb, lo2, n): (f32, f32, f32, f32, u32)| {
1164            let pts = great_circle_path(la, lo, lb, lo2, n);
1165            let t = lua.create_table()?;
1166            for (i, (lat, lon)) in pts.iter().enumerate() {
1167                let p = lua.create_table()?;
1168                p.set(1, *lat)?;
1169                p.set(2, *lon)?;
1170                t.set(i + 1, p)?;
1171            }
1172            Ok(t)
1173        })?,
1174    )?;
1175    // -- latLonToUnit --

1177    /// @param | lat | number | Latitude in degrees.

1187            let t = lua.create_table()?;

1189            t.set(2, v.y)?;
1190            t.set(3, v.z)?;

1192        })?,
1193    )?;
