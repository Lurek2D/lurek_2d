12/// Lua-visible font handle storing the slot key and cached metadata.
13#[derive(Clone)]
14pub struct LuaFont {
15    /// SlotMap key into SharedState.fonts.
16    handle_id: FontKey,
17    /// Human-readable font name exposed by the lurek engine.
18    name: String,
19    /// Point size of the font exposed by the lurek engine.
20    size: u32,
21    /// Style string ("regular" or "bold").
22    style: String,
23    /// Shared runtime state for font access.
24    state: Rc<RefCell<SharedState>>,
25}
26
27impl LuaUserData for LuaFont {
28    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
29        // -- getName --
30        /// Returns the human-readable name of this font. This method is available to Lua scripts.
31        /// @return | string | Font name.
32        methods.add_method("getName", |_, this, ()| Ok(this.name.clone()));
33
34        // -- getSize --
35        /// Returns the point size of this font. This method is available to Lua scripts.
36        /// @return | number | Point size.
37        methods.add_method("getSize", |_, this, ()| Ok(this.size));
38
39        // -- getStyle --
40        /// Returns the style string of this font. This method is available to Lua scripts.
41        /// @return | string | Style name ("regular", "bold").
42        methods.add_method("getStyle", |_, this, ()| Ok(this.style.clone()));
43
44        // -- isBold --
45        /// Returns whether this font is the bold variant. This method is available to Lua scripts.
46        /// @return | boolean | True if font style is bold.
47        methods.add_method("isBold", |_, this, ()| Ok(this.style == "bold"));
48
49        // -- lineHeight --
50        /// Returns the line height of this font in pixels. This method is available to Lua scripts.
51        /// @return | number | Line height in pixels.
52        methods.add_method("lineHeight", |_, this, ()| {
53            let st = this.state.borrow();
54            let font = st.fonts.get(this.handle_id).ok_or_else(|| {
55                LuaError::RuntimeError("Font handle is not valid or was released".into())
56            })?;
57            Ok(font.line_height())
58        });
59
60        // -- measure --
61        /// Measures the pixel dimensions of a text string at the given scale. This method is available to Lua scripts.
62        /// @param | text | string | Text to measure.
63        /// @param | scale | number | Scale factor applied to dimensions.
64        /// @return | number, number | Width and height in pixels.
65        methods.add_method("measure", |_, this, (text, scale): (String, Option<f32>)| {
66            let scale = scale.unwrap_or(1.0);
67            let st = this.state.borrow();
68            let font = st.fonts.get(this.handle_id).ok_or_else(|| {
69                LuaError::RuntimeError("Font handle is not valid or was released".into())
70            })?;
71            let width = font.text_width(&text) * scale;
72            let height = font.line_height() * scale;
73            Ok((width, height))
74        });
75
76        // -- wrapText --
77        /// Wraps text into lines fitting within the given max width. This method is available to Lua scripts.
78        /// @param | text | string | Text to wrap.
79        /// @param | maxWidth | number | Maximum line width in pixels.
80        /// @param | scale | number | Scale factor.
81        /// @return | table | Array of wrapped line strings.
82        methods.add_method(
83            "wrapText",
84            |lua, this, (text, max_width, scale): (String, f32, Option<f32>)| {
85                let scale = scale.unwrap_or(1.0);
86                let st = this.state.borrow();
87                let font = st.fonts.get(this.handle_id).ok_or_else(|| {
88                    LuaError::RuntimeError("Font handle is not valid or was released".into())
89                })?;
90                let limit = if scale > 0.0 {
91                    max_width / scale
92                } else {
93                    max_width
94                };
95                let lines = font.wrap_text(&text, limit);
96                let tbl = lua.create_table()?;
97                for (i, line) in lines.iter().enumerate() {
98                    tbl.set(i + 1, line.as_str())?;
99                }
100                Ok(tbl)
101            },
102        );
103
104        // -- containsGlyph --
105        /// Returns whether the font contains a glyph for the given character. This method is available to Lua scripts.
106        /// @param | char | string | A single-character string to check.
107        /// @return | boolean | True if the font has a glyph for this character.
108        methods.add_method("containsGlyph", |_, this, ch: String| {
109            let st = this.state.borrow();
110            let font = st.fonts.get(this.handle_id).ok_or_else(|| {
111                LuaError::RuntimeError("Font handle is not valid or was released".into())
112            })?;
113            let c = ch.chars().next().unwrap_or('\0');
114            Ok(font.glyph(c).is_some())
115        });
116    }
117}

119/// Resolves the active font key, falling back to the default font.
120fn active_font_key(st: &SharedState) -> Option<FontKey> {
121    st.active_font.or(st.default_font)
122}
123
124/// Determines the style string for a font key in shared state.
125fn font_style_str(st: &SharedState, key: FontKey) -> &'static str {
126    for bold_key in &st.default_bold_fonts {
127        if *bold_key == Some(key) {
128            return "bold";
129        }
130    }
131    "regular"
132}
133
134/// Determines a human-readable name for a font key in shared state.
135fn font_name_for_key(st: &SharedState, key: FontKey) -> String {
136    for (i, regular_key) in st.default_fonts.iter().enumerate() {
137        if *regular_key == Some(key) {
138            return format!("font_{}", AVAILABLE_POINT_SIZES[i]);
139        }
140    }
141    for (i, bold_key) in st.default_bold_fonts.iter().enumerate() {
142        if *bold_key == Some(key) {
143            return format!("fontb_{}", AVAILABLE_POINT_SIZES[i]);
144        }
145    }
146    format!("font_{}", key.data().as_ffi())
147}
148
149/// Determines the point size for a font key in shared state.
150fn font_size_for_key(st: &SharedState, key: FontKey) -> u32 {
151    if let Some(font) = st.fonts.get(key) {
152        font.size() as u32
153    } else {
154        0
155    }
156}

158/// Builds a LuaFont userdata from a font key.
159fn make_lua_font(state: Rc<RefCell<SharedState>>, key: FontKey) -> LuaFont {
160    let st = state.borrow();
161    let name = font_name_for_key(&st, key);
162    let size = font_size_for_key(&st, key);
163    let style = font_style_str(&st, key).to_string();
164    drop(st);
165    LuaFont {
166        handle_id: key,
167        name,
168        size,
169        style,
170        state,
171    }
172}

190    /// Returns the default engine font as a LuaFont userdata. This function is available to Lua scripts.
191    /// @return | LuaFont | The default font handle.
192    {
193        let s = state.clone();
194        /// Returns the default engine font as an LFont userdata handle.
195        ///
196        /// @return | LFont | The default engine font handle.
197        font.set(
198            "getDefault",
199            lua.create_function(move |_, ()| {
200                let key = {
201                    let st = s.borrow();
202                    active_font_key(&st).ok_or_else(|| {
203                        LuaError::RuntimeError(

205                        )

207                };
208                Ok(make_lua_font(s.clone(), key))
209            })?,

213    // -- load --
214    /// Loads a TTF/OTF font file at the given point size and returns a LuaFont handle. This function is available to Lua scripts.
215    /// @param | path | string | Relative path to the font file.
216    /// @param | size | number | Point size for rasterisation.
217    /// @return | LuaFont | The loaded font handle.
218    {
219        let s = state.clone();
220        /// Loads a TTF/OTF/PNG font file at the given point size and returns an LFont handle.
221        ///
222        /// @param | path | string | Relative path to the font file.
223        /// @param | size | number | Point size for rasterisation.
224        /// @return | LFont | The loaded font handle.
225        font.set(
226            "load",
227            lua.create_function(move |_, (path, size): (String, f32)| {
228                let key = {
229                    let mut st = s.borrow_mut();
230                    let full_path = st.game_dir.join(&path);
231                    let data = std::fs::read(&full_path).map_err(|e| {
232                        LuaError::RuntimeError(format!(
233                            "lurek.font.load: failed to read '{}': {}",
234                            path, e
235                        ))
236                    })?;
237                    let ext = Path::new(&path)
238                        .extension()
239                        .and_then(|ext| ext.to_str())
240                        .map(|ext| ext.to_ascii_lowercase());
241                    let bitmap_cell_h = size.max(1.0).round() as u32;
242                    let bitmap_cell_w = (size.max(1.0) * 0.6).round().max(1.0) as u32;
243                    let loaded = match ext.as_deref() {
244                        Some("ttf") | Some("otf") | Some("ttc") => {
245                            Font::from_font_bytes(&data, size)
246                        }
247                        Some("png") => {
248                            Font::from_png_bytes(&data, bitmap_cell_w, bitmap_cell_h, false)
249                        }
250                        _ => Font::from_font_bytes(&data, size).or_else(|_| {
251                            Font::from_png_bytes(&data, bitmap_cell_w, bitmap_cell_h, false)
252                        }),
253                    }
254                    .map_err(|e| {
255                        LuaError::RuntimeError(format!("lurek.font.load: {}", e))
256                    })?;
257                    st.fonts.insert(loaded)
258                };
259                let lua_font = {
260                    let st = s.borrow();
261                    let name = Path::new(&path)
262                        .file_stem()
263                        .and_then(|n| n.to_str())
264                        .unwrap_or("custom")
265                        .to_string();
266                    let font_size = st
267                        .fonts
268                        .get(key)
269                        .map(|f| f.size() as u32)
270                        .unwrap_or(size as u32);
271                    drop(st);
272                    LuaFont {
273                        handle_id: key,
274                        name,
275                        size: font_size,
276                        style: "regular".to_string(),
277                        state: s.clone(),
278                    }
279                };
280                Ok(lua_font)
281            })?,
282        )?;
283    }

534    // -- shapeText --
535    /// Shapes text into aligned, wrapped lines with offset data. This function is available to Lua scripts.
536    /// @param | font | LuaFont | Font handle.
537    /// @param | text | string | Text to shape.
538    /// @param | maxWidth | number | Maximum line width in pixels.
539    /// @param | scale | number | Scale factor.
540    /// @param | align | string | Alignment: "left", "center", "right", "justify".
541    /// @param | wrap | string | Wrap mode: "none", "word", "char".
542    /// @return | table | Array of tables with fields: text (string), width (number), xOffset (number).
543    {
544        let s = state.clone();
545        /// Shapes and aligns text into wrapped lines with x-offset data for rendering.
546        ///
547        /// @param | font | LFont | Font handle used for shaping.
548        /// @param | text | string | Text to shape.
549        /// @param | maxWidth | number | Maximum line width in pixels.
550        /// @param | scale | number | Scale factor (default 1.0).
551        /// @param | align | string | Alignment: "left", "center", "right", or "justify".
552        /// @param | wrap | string | Wrap mode: "none", "word", or "char".
553        /// @return | table | Array of tables with fields: text (string), width (number), xOffset (number).
554        font.set(
555            "shapeText",
556            lua.create_function(
557                move |lua,
558                      (font_ud, text, max_width, scale, align, wrap): (
559                    LuaAnyUserData,
560                    String,
561                    f32,
562                    Option<f32>,
563                    Option<String>,
564                    Option<String>,
565                )| {
566                    let scale = scale.unwrap_or(1.0);
567                    let align_str = align.unwrap_or_else(|| "left".to_string());
568                    let wrap_str = wrap.unwrap_or_else(|| "word".to_string());
569
570                    let lua_font = font_ud.borrow::<LuaFont>()?;
571                    let key = lua_font.handle_id;
572                    let state_ref = lua_font.state.clone();
573                    drop(lua_font);
574
575                    let st = state_ref.borrow();
576                    let f = st.fonts.get(key).ok_or_else(|| {
577                        LuaError::RuntimeError("Font handle is not valid or was released".into())
578                    })?;
579
580                    let limit = if scale > 0.0 {
581                        max_width / scale
582                    } else {
583                        max_width
584                    };
585
586                    // Wrap lines based on mode
587                    let raw_lines: Vec<String> = match wrap_str.as_str() {
588                        "none" => text.split('\n').map(String::from).collect(),
589                        "char" => {
590                            // Character-level wrapping
591                            let mut lines = Vec::new();
592                            for paragraph in text.split('\n') {
593                                let mut current = String::new();
594                                let mut current_width = 0.0f32;
595                                for ch in paragraph.chars() {
596                                    let adv = f
597                                        .glyph(ch)
598                                        .map(|g| g.advance_width)
599                                        .unwrap_or(0.0);
600                                    if current_width + adv > limit && !current.is_empty() {
601                                        lines.push(current);
602                                        current = String::new();
603                                        current_width = 0.0;
604                                    }
605                                    current.push(ch);
606                                    current_width += adv;
607                                }
608                                lines.push(current);
609                            }
610                            lines
611                        }
612                        _ => f.wrap_text(&text, limit), // "word" default
613                    };
614
615                    // Compute alignment offsets
616                    let tbl = lua.create_table()?;
617                    for (i, line_text) in raw_lines.iter().enumerate() {
618                        let width = f.text_width(line_text) * scale;
619                        let x_offset = match align_str.as_str() {
620                            "center" => ((max_width - width) * 0.5).max(0.0),
621                            "right" => (max_width - width).max(0.0),
622                            _ => 0.0, // "left" and "justify"
623                        };
624                        let entry = lua.create_table()?;
625                        entry.set("text", line_text.as_str())?;
626                        entry.set("width", width)?;
627                        entry.set("xOffset", x_offset)?;
628                        tbl.set(i + 1, entry)?;
629                    }
630                    Ok(tbl)
631                },
632            )?,
633        )?;
634        let _ = &s;
635    }
636
637    // -- charAdvance --
638    /// Returns the horizontal advance width of a single character. This function is available to Lua scripts.
639    /// @param | font | LuaFont | Font handle.
640    /// @param | char | string | A single-character string.
641    /// @param | scale | number | Scale factor.
642    /// @return | number | Advance width in pixels.
643    {
644        let s = state.clone();
645        /// Returns the horizontal advance width in pixels of a single character using the given font.
646        ///
647        /// @param | font | LFont | Font handle.
648        /// @param | char | string | A single-character string.
649        /// @param | scale | number | Scale factor (default 1.0).
650        /// @return | number | Horizontal advance width in pixels.
651        font.set(
652            "charAdvance",
653            lua.create_function(
654                move |_, (font_ud, ch, scale): (LuaAnyUserData, String, Option<f32>)| {
655                    let scale = scale.unwrap_or(1.0);
656                    let lua_font = font_ud.borrow::<LuaFont>()?;
657                    let key = lua_font.handle_id;
658                    let state_ref = lua_font.state.clone();
659                    drop(lua_font);
660                    let st = state_ref.borrow();
661                    let f = st.fonts.get(key).ok_or_else(|| {
662                        LuaError::RuntimeError("Font handle is not valid or was released".into())
663                    })?;
664                    let c = ch.chars().next().unwrap_or(' ');
665                    let adv = f.glyph(c).map(|g| g.advance_width).unwrap_or(0.0);
666                    Ok(adv * scale)
667                },
668            )?,
669        )?;
670        let _ = &s;
671    }
672
673    // -- lineHeight --
674    /// Returns the line height of a font in pixels. This function is available to Lua scripts.
675    /// @param | font | LuaFont | Font handle.
676    /// @return | number | Line height in pixels.
677    {
678        let s = state.clone();
679        /// Returns the line height of the given font in pixels.
680        ///
681        /// @param | font | LFont | Font handle to query.
682        /// @return | number | Line height in pixels.
683        font.set(
684            "lineHeight",
685            lua.create_function(move |_, font_ud: LuaAnyUserData| {
686                let lua_font = font_ud.borrow::<LuaFont>()?;
687                let key = lua_font.handle_id;
688                let state_ref = lua_font.state.clone();
689                drop(lua_font);
690                let st = state_ref.borrow();
691                let f = st.fonts.get(key).ok_or_else(|| {
692                    LuaError::RuntimeError("Font handle is not valid or was released".into())
693                })?;
694                Ok(f.line_height())
695            })?,
696        )?;
697        let _ = &s;
698    }
699
700    Ok(font)
701}

703/// Standard module registration entry point — registers `lurek.font` and returns `Ok(())`.
704pub fn register(lua: &Lua, lurek: &LuaTable, state: Rc<RefCell<SharedState>>) -> LuaResult<()> {
705    let font = register_font_api(lua, state)?;
706    lurek.set("font", font)?;
707    Ok(())
