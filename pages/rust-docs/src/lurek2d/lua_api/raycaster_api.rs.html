24/// Rebuilds a texture key and raw handle pair from the persisted numeric texture id.
25fn texture_key_from_raw_id(raw_id: u64) -> (TextureKey, u64) {
26    (TextureKey::from(slotmap::KeyData::from_ffi(raw_id)), raw_id)
27}
28/// Parses nil, numeric ids, or `LImage` userdata into a raycaster texture reference.
29fn parse_texture_key_value(
30    value: &LuaValue,
31    api_name: &str,
32) -> LuaResult<Option<(TextureKey, u64)>> {
33    match value {
34        LuaValue::Nil => Ok(None),
35        LuaValue::Integer(v) => {
36            if *v < 0 {
37                return Err(LuaError::RuntimeError(format!(
38                    "{}: texture id must be >= 0",
39                    api_name
40                )));
41            }
42            Ok(Some(texture_key_from_raw_id(*v as u64)))
43        }
44        LuaValue::Number(v) => {
45            if !v.is_finite() || *v < 0.0 || v.fract() != 0.0 {
46                return Err(LuaError::RuntimeError(format!(
47                    "{}: texture must be an integer id, LImage userdata, or nil",
48                    api_name
49                )));
50            }
51            Ok(Some(texture_key_from_raw_id(*v as u64)))
52        }
53        LuaValue::UserData(ud) => {
54            let img = ud.borrow::<LuaImage>().map_err(|_| {
55                LuaError::RuntimeError(format!(
56                    "{}: texture userdata must be LImage from lurek.render.newImage()",
57                    api_name
58                ))
59            })?;
60            let key = img.key;
61            let raw = key.data().as_ffi();
62            Ok(Some((key, raw)))
63        }
64        _ => Err(LuaError::RuntimeError(format!(
65            "{}: texture must be an integer id, LImage userdata, or nil",
66            api_name
67        ))),
68    }
69}
70/// Serializes one raycaster hit result into the Lua table layout returned by cast helpers.
71fn ray_hit_to_table<'lua>(lua: &'lua Lua, hit: &RayHit) -> LuaResult<LuaTable<'lua>> {
72    let t = lua.create_table()?;
73    /// Performs the 'distance' operation.
74    t.set("distance", hit.distance)?;
75    /// Performs the 'raw_distance' operation.
76    t.set("raw_distance", hit.raw_distance)?;
77    /// Performs the 'cell_value' operation.
78    t.set("cell_value", hit.cell_value)?;
79    /// Performs the 'alpha' operation.
80    t.set("alpha", hit.alpha)?;
81    /// Performs the 'side' operation.
82    t.set("side", hit.side)?;
83    /// Performs the 'tex_u' operation.
84    t.set("tex_u", hit.tex_u)?;
85    /// Performs the 'hit_x' operation.
86    t.set("hit_x", hit.hit_x)?;
87    /// Performs the 'hit_y' operation.
88    t.set("hit_y", hit.hit_y)?;
89    /// The 'hit' field value exposed to Lua scripts.
90    t.set("hit", hit.hit)?;
91    Ok(t)
92}
93/// Parses a Lua array of point light tables into raycaster point light definitions.
94fn parse_point_lights(value: LuaValue, api_name: &str) -> LuaResult<Vec<PointLight>> {
95    match value {
96        LuaValue::Nil => Ok(Vec::new()),
97        LuaValue::Table(tbl) => {
98            let mut out = Vec::new();
99            for pair in tbl.sequence_values::<LuaTable>() {
100                let lt = pair?;
101                out.push(PointLight {
102                    x: lt.get::<_, f32>("x").map_err(|_| {
103                        LuaError::RuntimeError(format!("{}: lights[].x is required", api_name))
104                    })?,
105                    y: lt.get::<_, f32>("y").map_err(|_| {
106                        LuaError::RuntimeError(format!("{}: lights[].y is required", api_name))
107                    })?,
108                    radius: lt.get::<_, Option<f32>>("radius")?.unwrap_or(0.0).max(0.0),
109                    color: [
110                        lt.get::<_, Option<f32>>("r")?.unwrap_or(1.0),
111                        lt.get::<_, Option<f32>>("g")?.unwrap_or(1.0),
112                        lt.get::<_, Option<f32>>("b")?.unwrap_or(1.0),
113                    ],
114                    intensity: lt.get::<_, Option<f32>>("intensity")?.unwrap_or(1.0),
115                });
116            }
117            Ok(out)
118        }
119        _ => Err(LuaError::RuntimeError(format!(
120            "{}: lights must be an array table or nil",
121            api_name
122        ))),
123    }
124}
125/// Lua-visible door manager that controls sliding doors within a raycaster map.
126/// Doors can be opened, closed, and animated over time at configurable speeds.

130impl LuaUserData for LuaDoorManager {
131    fn add_methods<'lua, M: LuaUserDataMethods<'lua, Self>>(methods: &mut M) {
132        // -- addDoor --
133        /// Registers a new sliding door at the given grid cell.
134        /// @param | x | integer | Grid column of the door cell.
135        /// @param | y | integer | Grid row of the door cell.
136        /// @param | direction | string | Slide axis: "horizontal" or "vertical".
137        /// @param | speed | number | How fast the door opens/closes (units per second).
138        /// @return | integer | Zero-based index of the newly added door.
139        methods.add_method_mut(
140            "addDoor",
141            |_, this, (x, y, dir_str, speed): (u32, u32, String, f32)| {
142                let dir = match dir_str.as_str() {
143                    "vertical" => DoorDirection::Vertical,
144                    _ => DoorDirection::Horizontal,
145                };
146                Ok(this.inner.borrow_mut().add_door(x, y, dir, speed))
147            },
148        );
149        // -- openDoor --
150        /// Begins opening the door at the given index. The door animates over time via `update()`.
151        /// @param | index | integer | Zero-based index of the door to open.
152        methods.add_method_mut("openDoor", |_, this, index: usize| {
153            this.inner.borrow_mut().open_door(index);
154            Ok(())
155        });
156        // -- closeDoor --
157        /// Begins closing the door at the given index. The door animates over time via `update()`.
158        /// @param | index | integer | Zero-based index of the door to close.
159        methods.add_method_mut("closeDoor", |_, this, index: usize| {
160            this.inner.borrow_mut().close_door(index);
161            Ok(())
162        });
163        // -- update --
164        /// Advances all door animations by the given delta time. Call once per frame.
165        /// @param | dt | number | Delta time in seconds since last frame.
166        methods.add_method_mut("update", |_, this, dt: f32| {
167            this.inner.borrow_mut().update(dt);
168            Ok(())
169        });
170        // -- getDoor --
171        /// Returns a table describing the door at the given index, or nil if index is out of range.
172        /// The table contains: x, y, openAmount (0.0..1.0), state ("closed"|"opening"|"open"|"closing").
173        /// @param | index | integer | Zero-based index of the door to query.
174        /// @return | table | Door info table, or nil if not found.
175        /// @field | x | number | X.
176        /// @field | y | number | Y.
177        /// @field | openAmount | number | Open amount 0.0 to 1.0.
178        /// @field | state | string | Door state.
179        methods.add_method("getDoor", |lua, this, index: usize| {
180            let mgr = this.inner.borrow();
181            if let Some(door) = mgr.doors().get(index) {
182                let tbl = lua.create_table()?;
183                /// The 'x' field value exposed to Lua scripts.
184                tbl.set("x", door.x)?;
185                /// The 'y' field value exposed to Lua scripts.
186                tbl.set("y", door.y)?;
187                /// Performs the 'openAmount' operation.
188                tbl.set("openAmount", door.open_amount)?;
189                let state_str = match door.state {
190                    DoorState::Closed => "closed",
191                    DoorState::Opening => "opening",
192                    DoorState::Open => "open",
193                    DoorState::Closing => "closing",
194                };
195                /// Performs the 'state' operation.
196                tbl.set("state", state_str)?;
197                Ok(LuaValue::Table(tbl))
198            } else {
199                Ok(LuaValue::Nil)
200            }
201        });
202        // -- count --
203        /// Returns the total number of registered doors.
204        /// @return | integer | Door count.
205        methods.add_method("count", |_, this, ()| Ok(this.inner.borrow().doors().len()));
206        // -- type --
207        /// Returns the type name of this object.
208        /// @return | string | Always "LDoorManager".
209        methods.add_method("type", |_, _, ()| Ok("LDoorManager"));
210        // -- typeOf --
211        /// Checks whether this object matches the given type name.
212        /// @param | name | string | Type name to check.
213        /// @return | boolean | True if the name matches this userdata type.
214        methods.add_method("typeOf", |_, _, name: String| {
215            Ok(name == "LDoorManager" || name == "LObject")
216        });
217    }
218}

451        /// @param | x | integer | Grid column.
452        /// @param | y | integer | Grid row.
453        /// @return | integer | Raw texture id or nil.
454        methods.add_method("getCeilingTextureCell", |_, this, (x, y): (u32, u32)| {
455            Ok(this.ceiling_cell_textures.get(&(x, y)).map(|entry| entry.1))
456        });
457        // -- setLoweredFloorCell --
458        /// Marks a cell as a lowered floor (pit) with its own texture, depth, tint, and blocking flag.
459        /// Pass nil to remove the lowered floor designation.
460        /// @param | x | integer | Grid column.
461        /// @param | y | integer | Grid row.
462        /// @param | opts | table? | Options table {texture, depth?, r?, g?, b?, blocked?} or nil to clear.
463        methods.add_method_mut(
464            "setLoweredFloorCell",
465            |_, this, (x, y, opts): (u32, u32, LuaValue)| {
466                match opts {
467                    LuaValue::Nil => {
468                        this.lowered_floor_cells.remove(&(x, y));
469                    }
470                    LuaValue::Table(tbl) => {
471                        let tex_val = tbl.get::<_, LuaValue>("texture")?;
472                        let (texture_key, raw_id) = parse_texture_key_value(
473                            &tex_val,
474                            "lurek.raycaster.setLoweredFloorCell(texture)",
475                        )?
476                        .ok_or_else(|| {
477                            LuaError::RuntimeError(
478                                "lurek.raycaster.setLoweredFloorCell: opts.texture cannot be nil"
479                                    .to_string(),
480                            )
481                        })?;
482                        let depth_offset = tbl
483                            .get::<_, Option<f32>>("depth")?
484                            .unwrap_or(0.25)
485                            .clamp(0.0, 0.75);
486                        let tint = [
487                            tbl.get::<_, Option<f32>>("r")?
488                                .unwrap_or(1.0)
489                                .clamp(0.0, 1.0),
490                            tbl.get::<_, Option<f32>>("g")?
491                                .unwrap_or(1.0)
492                                .clamp(0.0, 1.0),
493                            tbl.get::<_, Option<f32>>("b")?
494                                .unwrap_or(1.0)
495                                .clamp(0.0, 1.0),
496                        ];
497                        let blocked = tbl.get::<_, Option<bool>>("blocked")?.unwrap_or(true);
498                        this.lowered_floor_cells.insert(
499                            (x, y),
500                            LuaLoweredFloorCell {
501                                texture_key,
502                                raw_id,
503                                depth_offset,
504                                tint,
505                                blocked,
506                            },
507                        );
508                    }
509                    _ => {
510                        return Err(LuaError::RuntimeError(
511                            "lurek.raycaster.setLoweredFloorCell: opts must be a table or nil"
512                                .to_string(),
513                        ));
514                    }
515                }
516                Ok(())
517            },
518        );
519        // -- getLoweredFloorCell --
520        /// Returns the lowered floor configuration at a cell, or nil if the cell is normal.
521        /// @param | x | integer | Grid column.
522        /// @param | y | integer | Grid row.
523        /// @return | table | Table {texture, depth, r, g, b, blocked} or nil.
524        /// @field | texture | integer | Texture id.
525        /// @field | depth | number | Floor depth.
526        /// @field | r | number | Red component.
527        /// @field | g | number | Green component.
528        /// @field | b | number | Blue component.
529        /// @field | blocked | boolean | Blocked.
530        methods.add_method("getLoweredFloorCell", |lua, this, (x, y): (u32, u32)| {
531            if let Some(cell) = this.lowered_floor_cells.get(&(x, y)) {
532                let tbl = lua.create_table()?;
533                /// Performs the 'texture' operation.

1486                    /// Performs the 'distance' operation.
1487                    entry.set("distance", dist)?;
1488                    tbl.set(i + 1, entry)?;
1489                }
1490                Ok(tbl)
1491            },
1492        );
1493        // -- type --
