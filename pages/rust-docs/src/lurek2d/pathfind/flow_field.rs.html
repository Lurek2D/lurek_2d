1//! Dijkstra-based flow field that seeds from one or more goal cells and computes shortest paths across a NavGrid.
2//!
3//! - Each reachable cell stores a normalised direction vector toward the nearest goal and its accumulated travel cost.
4//! - Supports variable unit sizes for clearance-aware pathfinding using the backing grid's walkability checks.
5//! - Provides world-space steering that converts pixel coordinates to tile lookups and returns a scaled velocity.
6//! - Includes a debug visualisation helper that renders the field directions and obstacles to an ImageData bitmap.
7
8use crate::runtime::log_messages::{FF01, FF02, FF03};
9
10use crate::log_msg;
11use crate::pathfind::nav_grid::NavGrid;
12use std::cell::RefCell;
13use std::collections::VecDeque;
14use std::rc::Rc;
15/// Grid-resident flow field pointing each reachable cell toward one or more goal cells.
16pub struct FlowField {
17    /// Width of the backing grid in cells.
18    width: u32,
19    /// Height of the backing grid in cells.
20    height: u32,
21    /// Normalised move direction per cell, pointing toward the nearest goal.
22    directions: Vec<(f32, f32)>,
23    /// Dijkstra distance from each cell to the nearest goal.
24    costs: Vec<f32>,
25    /// True after `calculate` or `calculate_multi` has run at least once.
26    calculated: bool,
27    /// Seed cells used for the last computation.
28    targets: Vec<(u32, u32)>,
29    /// Shared grid reference used for walkability and move-cost queries.
30    grid: Rc<RefCell<NavGrid>>,
31}
32/// Construction and query methods for `FlowField`.
33impl FlowField {
34    /// Create an uninitialised flow field linked to `grid`; call `calculate` before querying.
35    pub fn new(grid: Rc<RefCell<NavGrid>>) -> Self {
36        let g = grid.borrow();
37        let w = g.get_width();
38        let h = g.get_height();
39        let size = (w * h) as usize;
40        drop(g);
41        log_msg!(debug, FF01);
42        Self {
43            width: w,
44            height: h,
45            directions: vec![(0.0, 0.0); size],
46            costs: vec![f32::INFINITY; size],
47            calculated: false,
48            targets: Vec::new(),
49            grid,
50        }
51    }
52    /// Seed the field with a single target cell and recompute.
53    pub fn calculate(&mut self, target_x: u32, target_y: u32, unit_size: u32) {
54        self.calculate_multi(&[(target_x, target_y)], unit_size);
55    }
56    /// Recompute the field seeded from all cells in `targets`.
57    pub fn calculate_multi(&mut self, targets: &[(u32, u32)], unit_size: u32) {
58        let grid = self.grid.borrow();
59        let w = grid.get_width();
60        let h = grid.get_height();
61        let size = (w * h) as usize;
62        let us = unit_size.max(1);
63        log_msg!(debug, FF03);
64        self.width = w;
65        self.height = h;
66        self.directions.clear();
67        self.directions.resize(size, (0.0, 0.0));
68        self.costs.clear();
69        self.costs.resize(size, f32::INFINITY);
70        self.targets = targets.to_vec();
71        let mut queue = VecDeque::new();
72        for &(tx, ty) in targets {
73            if tx < w && ty < h && grid.is_walkable(tx, ty, us) {
74                let idx = (ty * w + tx) as usize;
75                self.costs[idx] = 0.0;
76                queue.push_back((tx, ty));
77            }
78        }
79        while let Some((cx, cy)) = queue.pop_front() {
80            let cur_cost = self.costs[(cy * w + cx) as usize];
81            for dy in -1i32..=1 {
82                for dx in -1i32..=1 {
83                    if dx == 0 && dy == 0 {
84                        continue;
85                    }
86                    let nx = cx as i32 + dx;
87                    let ny = cy as i32 + dy;
88                    if nx < 0 || ny < 0 || nx >= w as i32 || ny >= h as i32 {
89                        continue;
90                    }
91                    let (nxu, nyu) = (nx as u32, ny as u32);
92                    if !grid.is_walkable(nxu, nyu, us) {
93                        continue;
94                    }
95                    let is_diag = dx != 0 && dy != 0;
96                    let step = if is_diag {
97                        std::f32::consts::SQRT_2
98                    } else {
99                        1.0
100                    } * grid.get_cost(nxu, nyu) as f32;
101                    let new_cost = cur_cost + step;
102                    let n_idx = (nyu * w + nxu) as usize;
103                    if new_cost < self.costs[n_idx] {
104                        self.costs[n_idx] = new_cost;
105                        queue.push_back((nxu, nyu));
106                    }
107                }
108            }
109        }
110        for cy in 0..h {
111            for cx in 0..w {
112                let idx = (cy * w + cx) as usize;
113                if self.costs[idx] == f32::INFINITY {
114                    continue;
115                }
116                let mut best_cost = self.costs[idx];
117                let mut best_dir = (0.0f32, 0.0f32);
118                for dy in -1i32..=1 {
119                    for dx in -1i32..=1 {
120                        if dx == 0 && dy == 0 {
121                            continue;
122                        }
123                        let nx = cx as i32 + dx;
124                        let ny = cy as i32 + dy;
125                        if nx < 0 || ny < 0 || nx >= w as i32 || ny >= h as i32 {
126                            continue;
127                        }
128                        let n_idx = (ny as u32 * w + nx as u32) as usize;
129                        if self.costs[n_idx] < best_cost {
130                            best_cost = self.costs[n_idx];
131                            best_dir = (dx as f32, dy as f32);
132                        }
133                    }
134                }
135                let len = (best_dir.0 * best_dir.0 + best_dir.1 * best_dir.1).sqrt();
136                if len > 0.0 {
137                    self.directions[idx] = (best_dir.0 / len, best_dir.1 / len);
138                }
139            }
140        }
141        log_msg!(debug, FF02);
142        self.calculated = true;
143    }
144    /// Return the normalised flow direction at `(x, y)`; returns `(0,0)` when out of bounds.
145    pub fn get_direction(&self, x: u32, y: u32) -> (f32, f32) {
146        if x >= self.width || y >= self.height {
147            return (0.0, 0.0);
148        }
149        self.directions[(y * self.width + x) as usize]
150    }
151    /// Return the flow direction at `(x, y)` as an angle in radians relative to the +x axis.
152    pub fn get_direction_angle(&self, x: u32, y: u32) -> f32 {
153        let (dx, dy) = self.get_direction(x, y);
154        dy.atan2(dx)
155    }
156    /// Return the Dijkstra cost from `(x, y)` to the nearest target; `INFINITY` when unreachable.
157    pub fn get_cost_to_target(&self, x: u32, y: u32) -> f32 {
158        if x >= self.width || y >= self.height {
159            return f32::INFINITY;
160        }
161        self.costs[(y * self.width + x) as usize]
162    }
163    /// Return true if `calculate` or `calculate_multi` has been called at least once.
164    pub fn is_calculated(&self) -> bool {
165        self.calculated
166    }
167    /// Return a clone of the target cells used for the last computation.
168    pub fn get_targets(&self) -> Vec<(u32, u32)> {
169        self.targets.clone()
170    }
171    /// Return the grid width. This function is part of the public API.
172    pub fn get_width(&self) -> u32 {
173        self.width
174    }
175    /// Return the grid height. This function is part of the public API.
176    pub fn get_height(&self) -> u32 {
177        self.height
178    }
179    /// Convert world position to tile, sample direction, and return a velocity scaled by `speed`.
180    pub fn steer(
181        &self,
182        world_x: f32,
183        world_y: f32,
184        speed: f32,
185        tile_w: f32,
186        tile_h: f32,
187    ) -> (f32, f32) {
188        if tile_w <= 0.0 || tile_h <= 0.0 {
189            return (0.0, 0.0);
190        }
191        let tx = (world_x / tile_w).floor() as i32;
192        let ty = (world_y / tile_h).floor() as i32;
193        if tx < 0 || ty < 0 || tx >= self.width as i32 || ty >= self.height as i32 {
194            return (0.0, 0.0);
195        }
196        let (dx, dy) = self.get_direction(tx as u32, ty as u32);
197        (dx * speed, dy * speed)
198    }
199    /// Render the flow field to an `ImageData` with `cell_size` pixels per tile for debugging.
200    pub fn draw_to_image(&self, cell_size: u32) -> crate::image::ImageData {
201        let mut img = crate::image::ImageData::new(self.width * cell_size, self.height * cell_size);
202        img.fill(40, 45, 55, 255);
203        {
204            let g = self.grid.borrow();
205            for y in 0..self.height {
206                for x in 0..self.width {
207                    if g.is_blocked(x, y) {
208                        for py in 0..cell_size {
209                            for px in 0..cell_size {
210                                img.set_pixel(
211                                    x * cell_size + px,
212                                    y * cell_size + py,
213                                    90,
214                                    40,
215                                    40,
216                                    255,
217                                );
218                            }
219                        }
220                    }
221                }
222            }
223        }
224        for y in 0..self.height {
225            for x in 0..self.width {
226                if !self.grid.borrow().is_blocked(x, y) {
227                    let (dx, dy) = self.get_direction(x, y);
228                    if dx.abs() > 0.01 || dy.abs() > 0.01 {
229                        let cx = (x * cell_size + cell_size / 2) as i32;
230                        let cy = (y * cell_size + cell_size / 2) as i32;
231                        let ex = cx + (dx * 6.0) as i32;
232                        let ey = cy + (dy * 6.0) as i32;
233                        img.draw_line(cx, cy, ex, ey, 100, 200, 255, 200);
234                    }
235                }
236            }
237        }
238        if let Some(&(tx, ty)) = self.targets.first() {
239            img.draw_circle(
240                (tx * cell_size + cell_size / 2) as i32,
241                (ty * cell_size + cell_size / 2) as i32,
242                5,
243                255,
244                80,
245                80,
246                255,
247            );
248        }
249        img
250    }
