1//! Stateful per-unit pathfinder wrapping a shared `NavGrid` reference.
2//!
3//! - Full A* path search with optional string-pull smoothing for shorter results.
4//! - Partial-path expansion with configurable node budget for real-time budgets.
5//! - BFS reachability test and nearest-walkable-cell search within a radius.
6//! - LRU path cache with configurable max size and manual invalidation.
7//! - Octile heuristic and Bresenham line-of-sight utility helpers.
8
9use crate::runtime::log_messages::{UP01, UP02, UP03};
10
11use crate::log_msg;
12use crate::pathfind::{astar, nav_grid::NavGrid};
13use std::cell::RefCell;
14use std::collections::{HashMap, VecDeque};
15use std::rc::Rc;
16/// Grid cell coordinate returned as a path waypoint.
17#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
18pub struct Waypoint {
19    /// Column index.
20    pub x: u32,
21    /// Row index.
22    pub y: u32,
23}
24/// LRU cache lookup key for a specific unit-size path request.
25#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
26struct CacheKey {
27    /// Start column.
28    x1: u32,
29    /// Start row.
30    y1: u32,
31    /// Goal column.
32    x2: u32,
33    /// Goal row.
34    y2: u32,
35    /// Footprint side length used to compute walkability.
36    unit_size: u32,
37}
38/// Stateful pathfinder for one unit type sharing a grid reference.
39pub struct UnitPathfinder {
40    /// Shared walkability grid.
41    grid: Rc<RefCell<NavGrid>>,
42    /// Cached path results keyed by `CacheKey`.
43    cache: HashMap<CacheKey, Option<Vec<Waypoint>>>,
44    /// Insertion-order key list used for LRU eviction.
45    cache_order: Vec<CacheKey>,
46    /// Whether path caching is active.
47    cache_enabled: bool,
48    /// Maximum number of cached entries before eviction.
49    cache_max_size: usize,
50}
51/// All public and private methods for `UnitPathfinder`.
52impl UnitPathfinder {
53    /// Create a new pathfinder wrapping `grid` with caching enabled and a default max size of 1024.
54    pub fn new(grid: Rc<RefCell<NavGrid>>) -> Self {
55        log_msg!(debug, UP01);
56        Self {
57            grid,
58            cache: HashMap::new(),
59            cache_order: Vec::new(),
60            cache_enabled: true,
61            cache_max_size: 1024,
62        }
63    }
64    /// Find a path from `(x1, y1)` to `(x2, y2)` for a unit of `unit_size`; return waypoints or `None`.
65    pub fn find_path(
66        &mut self,
67        x1: u32,
68        y1: u32,
69        x2: u32,
70        y2: u32,
71        unit_size: u32,
72    ) -> Option<Vec<Waypoint>> {
73        let key = CacheKey {
74            x1,
75            y1,
76            x2,
77            y2,
78            unit_size,
79        };
80        if self.cache_enabled {
81            if let Some(cached) = self.cache.get(&key) {
82                return cached.clone();
83            }
84        }
85        let result = {
86            let grid = self.grid.borrow();
87            let (path, _complete) = astar::astar(&grid, (x1, y1), (x2, y2), unit_size, 0);
88            path.map(|p| p.into_iter().map(|(x, y)| Waypoint { x, y }).collect())
89        };
90        if result.is_some() {
91            log_msg!(debug, UP02, "({}, {}) -> ({}, {})", x1, y1, x2, y2);
92        } else {
93            log_msg!(warn, UP03, "({}, {}) -> ({}, {})", x1, y1, x2, y2);
94        }
95        if self.cache_enabled {
96            self.cache_insert(key, result.clone());
97        }
98        result
99    }
100    /// Find a path then apply A\* string-pull smoothing; return waypoints or `None`.
101    pub fn find_path_smooth(
102        &mut self,
103        x1: u32,
104        y1: u32,
105        x2: u32,
106        y2: u32,
107        unit_size: u32,
108    ) -> Option<Vec<Waypoint>> {
109        let grid = self.grid.borrow();
110        let (path, _complete) = astar::astar(&grid, (x1, y1), (x2, y2), unit_size, 0);
111        path.map(|p| {
112            let smoothed = astar::smooth_path(&grid, &p, unit_size);
113            smoothed
114                .into_iter()
115                .map(|(x, y)| Waypoint { x, y })
116                .collect()
117        })
118    }
119    /// Return the Euclidean length of `path` in cells.
120    pub fn get_path_length(path: &[Waypoint]) -> f32 {
121        let mut total = 0.0f32;
122        for i in 1..path.len() {
123            let dx = path[i].x as f32 - path[i - 1].x as f32;
124            let dy = path[i].y as f32 - path[i - 1].y as f32;
125            total += (dx * dx + dy * dy).sqrt();
126        }
127        total
128    }
129    /// Return the sum of `NavGrid` costs for all waypoints in `path`.
130    pub fn get_path_cost(&self, path: &[Waypoint]) -> f32 {
131        let grid = self.grid.borrow();
132        let mut total = 0.0f32;
133        for wp in path {
134            total += grid.get_cost(wp.x, wp.y) as f32;
135        }
136        total
137    }
138    /// Run A\* limited to `max_nodes` expansions; return `(partial_path, reached_goal)`.
139    pub fn find_partial_path(
140        &self,
141        x1: u32,
142        y1: u32,
143        x2: u32,
144        y2: u32,
145        max_nodes: u32,
146        unit_size: u32,
147    ) -> (Vec<Waypoint>, bool) {
148        let grid = self.grid.borrow();
149        let (path, complete) = astar::astar(&grid, (x1, y1), (x2, y2), unit_size, max_nodes);
150        let waypoints = path
151            .map(|p| p.into_iter().map(|(x, y)| Waypoint { x, y }).collect())
152            .unwrap_or_default();
153        (waypoints, complete)
154    }
155    /// BFS-search for the nearest `unit_size`-walkable cell within `max_radius` steps from `(x, y)`.
156    pub fn find_nearest_walkable(
157        &self,
158        x: u32,
159        y: u32,
160        max_radius: u32,
161        unit_size: u32,
162    ) -> Option<(u32, u32)> {
163        let grid = self.grid.borrow();
164        if grid.is_walkable(x, y, unit_size) {
165            return Some((x, y));
166        }
167        let (w, h) = grid.get_dimensions();
168        let mut visited = vec![false; (w * h) as usize];
169        let mut queue = VecDeque::new();
170        visited[(y * w + x) as usize] = true;
171        queue.push_back((x, y, 0u32));
172        while let Some((cx, cy, dist)) = queue.pop_front() {
173            if dist > max_radius {
174                break;
175            }
176            if grid.is_walkable(cx, cy, unit_size) {
177                return Some((cx, cy));
178            }
179            for (dx, dy) in [(-1i32, 0i32), (1, 0), (0, -1), (0, 1)] {
180                let nx = cx as i32 + dx;
181                let ny = cy as i32 + dy;
182                if nx < 0 || ny < 0 || nx >= w as i32 || ny >= h as i32 {
183                    continue;
184                }
185                let (nxu, nyu) = (nx as u32, ny as u32);
186                let idx = (nyu * w + nxu) as usize;
187                if !visited[idx] {
188                    visited[idx] = true;
189                    queue.push_back((nxu, nyu, dist + 1));
190                }
191            }
192        }
193        Option::None
194    }
195    /// Return true when `(x2, y2)` is reachable from `(x1, y1)` via BFS for a unit of `unit_size`.
196    pub fn is_reachable(&self, x1: u32, y1: u32, x2: u32, y2: u32, unit_size: u32) -> bool {
197        let grid = self.grid.borrow();
198        let us = unit_size.max(1);
199        if !grid.is_walkable(x1, y1, us) || !grid.is_walkable(x2, y2, us) {
200            return false;
201        }
202        let (w, h) = grid.get_dimensions();
203        let mut visited = vec![false; (w * h) as usize];
204        let mut queue = VecDeque::new();
205        visited[(y1 * w + x1) as usize] = true;
206        queue.push_back((x1, y1));
207        while let Some((cx, cy)) = queue.pop_front() {
208            if cx == x2 && cy == y2 {
209                return true;
210            }
211            for (nx, ny) in grid.neighbors(cx, cy) {
212                if grid.is_walkable(nx, ny, us) {
213                    let idx = (ny * w + nx) as usize;
214                    if !visited[idx] {
215                        visited[idx] = true;
216                        queue.push_back((nx, ny));
217                    }
218                }
219            }
220        }
221        false
222    }
223    /// Return the octile distance heuristic between two cell coordinates.
224    pub fn heuristic_distance(x1: u32, y1: u32, x2: u32, y2: u32) -> f32 {
225        let dx = (x1 as f32 - x2 as f32).abs();
226        let dy = (y1 as f32 - y2 as f32).abs();
227        let min = dx.min(dy);
228        let max = dx.max(dy);
229        min * std::f32::consts::SQRT_2 + (max - min)
230    }
231    /// Return true when the Bresenham line from `(x1, y1)` to `(x2, y2)` passes only walkable cells for `unit_size`.
232    pub fn line_of_sight(&self, x1: u32, y1: u32, x2: u32, y2: u32, unit_size: u32) -> bool {
233        let grid = self.grid.borrow();
234        astar::line_of_sight(&grid, x1, y1, x2, y2, unit_size)
235    }
236    /// Enable or disable path caching; clears existing cache when disabled.
237    pub fn set_cache_enabled(&mut self, enabled: bool) {
238        self.cache_enabled = enabled;
239        if !enabled {
240            self.cache.clear();
241            self.cache_order.clear();
242        }
243    }
244    /// Return true when path caching is currently enabled.
245    pub fn is_cache_enabled(&self) -> bool {
246        self.cache_enabled
247    }
248    /// Remove all cached paths. This function is part of the public API.
249    pub fn clear_cache(&mut self) {
250        self.cache.clear();
251        self.cache_order.clear();
252    }
253    /// Return the current number of cached entries.
254    pub fn get_cache_size(&self) -> usize {
255        self.cache.len()
256    }
257    /// Set the maximum cache size and evict old entries if needed.
258    pub fn set_cache_max_size(&mut self, max_size: usize) {
259        self.cache_max_size = max_size;
260        self.evict();
261    }
262    /// Return a reference to the shared `NavGrid`.
263    pub fn nav_grid(&self) -> &std::rc::Rc<std::cell::RefCell<NavGrid>> {
264        &self.grid
265    }
266    /// Insert a path into the cache and trigger LRU eviction if over `cache_max_size`.
267    fn cache_insert(&mut self, key: CacheKey, value: Option<Vec<Waypoint>>) {
268        self.cache.insert(key, value);
269        self.cache_order.push(key);
270        self.evict();
271    }
272    /// Remove the oldest cache entry until the cache is at or below `cache_max_size`.
273    fn evict(&mut self) {
274        while self.cache.len() > self.cache_max_size && !self.cache_order.is_empty() {
275            let oldest = self.cache_order.remove(0);
276            self.cache.remove(&oldest);
277        }
278    }
