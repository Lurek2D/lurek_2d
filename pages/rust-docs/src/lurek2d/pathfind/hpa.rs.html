1//! Hierarchical Pathfinding A* (HPA*) over a chunked NavGrid abstraction.
2//!
3//! - Partition the grid into fixed-size chunks and detect entrance nodes at chunk boundaries.
4//! - Build an abstract graph of entrance-to-entrance edges with A*-computed costs.
5//! - Run abstract-level A* search using octile distance heuristic.
6//! - Refine abstract waypoints back into full grid-level paths via per-segment A*.
7//! - BFS-based reachability test over chunk connectivity without computing a full path.
8//! - Temporary start/goal insertion into the abstract graph for single queries.
9//! - Boundary scanning logic handles both horizontal and vertical chunk edges.
10//! - Supports variable unit sizes passed through to underlying A* refinement.
11
12use crate::runtime::log_messages::{HP01, HP02, HP03};

16use std::cmp::Ordering;
17use std::collections::{BinaryHeap, HashMap, HashSet, VecDeque};
18/// Directed edge in the abstract graph between two entrance nodes.
19#[derive(Debug, Clone)]
20pub struct AbstractEdge {
21    /// Index of the destination node in `AbstractGraph::nodes`.
22    pub to: usize,
23    /// Traversal cost of this edge in grid-distance units.
24    pub cost: f32,
25}
26/// Abstract graph node positioned at a chunk entrance cell.
27#[derive(Debug, Clone)]
28pub struct AbstractNode {
29    /// Grid column of this entrance cell.
30    pub x: u32,
31    /// Grid row of this entrance cell.
32    pub y: u32,
33    /// Chunk coordinates `(chunk_col, chunk_row)` that own this node.
34    pub chunk: (u32, u32),
35}
36/// A rectangular region of the grid used as the unit of the abstract hierarchy.
37#[derive(Debug, Clone)]
38pub struct Chunk {
39    /// Chunk column index in the chunk grid.
40    pub col: u32,
41    /// Chunk row index in the chunk grid.
42    pub row: u32,
43    /// Left grid column of this chunk.
44    pub x: u32,
45    /// Top grid row of this chunk.
46    pub y: u32,
47    /// Width in grid cells.
48    pub w: u32,
49    /// Height in grid cells.
50    pub h: u32,
51    /// Indices into `AbstractGraph::nodes` for entrances belonging to this chunk.
52    pub entrance_indices: Vec<usize>,
53}
54/// Full hierarchical graph: entrance nodes, inter-chunk edges, and chunk metadata.
55#[derive(Debug, Clone)]
56pub struct AbstractGraph {
57    /// All entrance nodes in the abstract graph.
58    pub nodes: Vec<AbstractNode>,
59    /// Adjacency list: `edges[i]` is the list of edges from node `i`.
60    pub edges: Vec<Vec<AbstractEdge>>,
61    /// All chunks indexed by `(chunk_col, chunk_row)`.
62    pub chunks: HashMap<(u32, u32), Chunk>,
63    /// Underlying grid width in tiles.
64    pub grid_width: u32,
65    /// Underlying grid height in tiles.
66    pub grid_height: u32,
67    /// Tile size of each chunk (clamped to ≥ 2).
68    pub chunk_size: u32,
69}
70/// Internal heap node for the abstract-level A\* search.
71#[derive(Debug, Clone)]
72struct HpaNode {
73    /// f-score = g + h.
74    f: f32,
75    /// Actual cost from start to this node.
76    g: f32,
77    /// Index into the temporary node list.
78    idx: usize,
79}
80/// Equality by f-score.
81impl PartialEq for HpaNode {
82    fn eq(&self, other: &Self) -> bool {
83        self.f == other.f
84    }
85}
86
87/// Marker impl required by `Ord`.
88impl Eq for HpaNode {}
89
90/// Reverse ordering so `BinaryHeap` acts as a min-heap.
91impl Ord for HpaNode {
92    fn cmp(&self, other: &Self) -> Ordering {
93        other.f.partial_cmp(&self.f).unwrap_or(Ordering::Equal)
94    }
95}
96/// Delegates to `Ord`.
97impl PartialOrd for HpaNode {
98    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
99        Some(self.cmp(other))
100    }
101}
102/// Build an `AbstractGraph` from `grid` by partitioning into chunks of `chunk_size` and computing entrance nodes.
103pub fn build_abstract(grid: &NavGrid, chunk_size: u32) -> AbstractGraph {
104    let cs = chunk_size.max(2);
105    let (gw, gh) = grid.get_dimensions();
106    let cols = gw.div_ceil(cs);
107    let rows = gh.div_ceil(cs);
108    log_msg!(debug, HP01, "{}x{}", gw, gh);
109    let mut graph = AbstractGraph {
110        nodes: Vec::new(),
111        edges: Vec::new(),
112        chunks: HashMap::new(),
113        grid_width: gw,
114        grid_height: gh,
115        chunk_size: cs,
116    };
117    for cr in 0..rows {
118        for cc in 0..cols {
119            let cx = cc * cs;
120            let cy = cr * cs;
121            let cw = cs.min(gw - cx);
122            let ch = cs.min(gh - cy);
123            graph.chunks.insert(
124                (cc, cr),
125                Chunk {
126                    col: cc,
127                    row: cr,
128                    x: cx,
129                    y: cy,
130                    w: cw,
131                    h: ch,
132                    entrance_indices: Vec::new(),
133                },
134            );
135        }
136    }
137    for cr in 0..rows.saturating_sub(1) {
138        for cc in 0..cols {
139            let chunk = graph.chunks[&(cc, cr)].clone();
140            let boundary_y = chunk.y + chunk.h - 1;
141            let below_y = boundary_y + 1;
142            if below_y >= gh {
143                continue;
144            }
145            find_boundary_entrances(
146                grid,
147                &mut graph,
148                &BoundaryScan {
149                    chunk_a: (cc, cr),
150                    chunk_b: (cc, cr + 1),
151                    range_start: chunk.x,
152                    range_end: chunk.x + chunk.w,
153                    coord_a: boundary_y,
154                    coord_b: below_y,
155                    horizontal: true,
156                },
157            );
158        }
159    }
160    for cr in 0..rows {
161        for cc in 0..cols.saturating_sub(1) {
162            let chunk = graph.chunks[&(cc, cr)].clone();
163            let boundary_x = chunk.x + chunk.w - 1;
164            let right_x = boundary_x + 1;
165            if right_x >= gw {
166                continue;
167            }
168            find_boundary_entrances(
169                grid,
170                &mut graph,
171                &BoundaryScan {
172                    chunk_a: (cc, cr),
173                    chunk_b: (cc + 1, cr),
174                    range_start: chunk.y,
175                    range_end: chunk.y + chunk.h,
176                    coord_a: boundary_x,
177                    coord_b: right_x,
178                    horizontal: false,
179                },
180            );
181        }
182    }
183    log_msg!(debug, HP03);
184    let chunk_keys: Vec<(u32, u32)> = graph.chunks.keys().cloned().collect();
185    for key in chunk_keys {
186        let indices = graph.chunks[&key].entrance_indices.clone();
187        let len = indices.len();
188        for i in 0..len {
189            for j in (i + 1)..len {
190                let a = indices[i];
191                let b = indices[j];
192                let ax = graph.nodes[a].x;
193                let ay = graph.nodes[a].y;
194                let bx = graph.nodes[b].x;
195                let by = graph.nodes[b].y;
196                let (path, complete) = astar::astar(grid, (ax, ay), (bx, by), 1, 0);
197                if complete {
198                    if let Some(ref p) = path {
199                        let cost = path_cost_f32(p);
200                        graph.edges[a].push(AbstractEdge { to: b, cost });
201                        graph.edges[b].push(AbstractEdge { to: a, cost });
202                    }
203                }
204            }
205        }
206    }
207    log_msg!(debug, HP02);
208    graph
209}
210/// Parameters for scanning a boundary between two adjacent chunks to detect entrance pairs.
211struct BoundaryScan {
212    /// First chunk coordinate pair.
213    chunk_a: (u32, u32),
214    /// Second chunk coordinate pair.
215    chunk_b: (u32, u32),
216    /// Start of the shared boundary range (inclusive).
217    range_start: u32,
218    /// End of the shared boundary range (exclusive).
219    range_end: u32,
220    /// Boundary row for chunk_a (horizontal) or column for chunk_a (vertical).
221    coord_a: u32,
222    /// Boundary row for chunk_b (horizontal) or column for chunk_b (vertical).
223    coord_b: u32,
224    /// True for a horizontal boundary (top/bottom), false for vertical (left/right).
225    horizontal: bool,
226}
227/// Scan the boundary described by `scan` and add entrance node pairs to `graph`.
228fn find_boundary_entrances(grid: &NavGrid, graph: &mut AbstractGraph, scan: &BoundaryScan) {
229    let BoundaryScan {
230        chunk_a,
231        chunk_b,
232        range_start,
233        range_end,
234        coord_a,
235        coord_b,
236        horizontal,
237    } = *scan;
238    let mut i = range_start;
239    while i < range_end {
240        let (ax, ay, bx, by) = if horizontal {
241            (i, coord_a, i, coord_b)
242        } else {
243            (coord_a, i, coord_b, i)
244        };
245        if grid.is_blocked(ax, ay) || grid.is_blocked(bx, by) {
246            i += 1;
247            continue;
248        }
249        let run_start = i;
250        while i < range_end {
251            let (nax, nay, nbx, nby) = if horizontal {
252                (i, coord_a, i, coord_b)
253            } else {
254                (coord_a, i, coord_b, i)
255            };
256            if grid.is_blocked(nax, nay) || grid.is_blocked(nbx, nby) {
257                break;
258            }
259            i += 1;
260        }
261        let run_end = i;
262        let positions: Vec<u32> = if run_end - run_start <= 2 {
263            vec![run_start]
264        } else {
265            vec![run_start, run_end - 1]
266        };
267        for pos in positions {
268            let (eax, eay, ebx, eby) = if horizontal {
269                (pos, coord_a, pos, coord_b)
270            } else {
271                (coord_a, pos, coord_b, pos)
272            };
273            let idx_a = get_or_create_node(graph, eax, eay, chunk_a);
274            let idx_b = get_or_create_node(graph, ebx, eby, chunk_b);
275            graph.edges[idx_a].push(AbstractEdge {
276                to: idx_b,
277                cost: 1.0,
278            });
279            graph.edges[idx_b].push(AbstractEdge {
280                to: idx_a,
281                cost: 1.0,
282            });
283        }
284    }
285}
286/// Return the index of the node at `(x, y)`, inserting it into `graph` under `chunk_key` if absent.
287fn get_or_create_node(graph: &mut AbstractGraph, x: u32, y: u32, chunk_key: (u32, u32)) -> usize {
288    for (i, n) in graph.nodes.iter().enumerate() {
289        if n.x == x && n.y == y {
290            return i;
291        }
292    }
293    let idx = graph.nodes.len();
294    graph.nodes.push(AbstractNode {
295        x,
296        y,
297        chunk: chunk_key,
298    });
299    graph.edges.push(Vec::new());
300    if let Some(chunk) = graph.chunks.get_mut(&chunk_key) {
301        chunk.entrance_indices.push(idx);
302    }
303    idx
304}
305/// Sum Euclidean distances of all consecutive pairs in `path`.
306fn path_cost_f32(path: &[(u32, u32)]) -> f32 {
307    let mut cost = 0.0f32;
308    for i in 1..path.len() {
309        let dx = (path[i].0 as f32 - path[i - 1].0 as f32).abs();
310        let dy = (path[i].1 as f32 - path[i - 1].1 as f32).abs();
311        cost += (dx * dx + dy * dy).sqrt();
312    }
313    cost
314}
315/// Run HPA\* on `abstract_graph`, returning a refined grid-level path from `start` to `goal`.
316pub fn hpa_star(
317    grid: &NavGrid,
318    abstract_graph: &AbstractGraph,
319    start: (u32, u32),
320    goal: (u32, u32),
321    unit_size: u32,
322) -> Option<Vec<(u32, u32)>> {
323    let cs = abstract_graph.chunk_size;
324    let start_chunk = (start.0 / cs, start.1 / cs);
325    let goal_chunk = (goal.0 / cs, goal.1 / cs);
326    let mut nodes = abstract_graph.nodes.clone();
327    let mut edges = abstract_graph.edges.clone();
328    let start_idx = nodes.len();
329    nodes.push(AbstractNode {
330        x: start.0,
331        y: start.1,
332        chunk: start_chunk,
333    });
334    edges.push(Vec::new());
335    let goal_idx = nodes.len();
336    nodes.push(AbstractNode {
337        x: goal.0,
338        y: goal.1,
339        chunk: goal_chunk,
340    });
341    edges.push(Vec::new());
342    for &temp_idx in &[start_idx, goal_idx] {
343        let chunk_key = nodes[temp_idx].chunk;
344        if let Some(chunk) = abstract_graph.chunks.get(&chunk_key) {
345            for &ent_idx in &chunk.entrance_indices {
346                let (path, complete) = astar::astar(
347                    grid,
348                    (nodes[temp_idx].x, nodes[temp_idx].y),
349                    (nodes[ent_idx].x, nodes[ent_idx].y),
350                    unit_size,
351                    0,
352                );
353                if complete {
354                    if let Some(ref p) = path {
355                        let cost = path_cost_f32(p);
356                        edges[temp_idx].push(AbstractEdge { to: ent_idx, cost });
357                        edges[ent_idx].push(AbstractEdge { to: temp_idx, cost });
358                    }
359                }
360            }
361        }
362    }
363    if start_chunk == goal_chunk {
364        let (path, complete) = astar::astar(grid, start, goal, unit_size, 0);
365        if complete {
366            if let Some(ref p) = path {
367                let cost = path_cost_f32(p);
368                edges[start_idx].push(AbstractEdge { to: goal_idx, cost });
369            }
370        }
371    }
372    let node_count = nodes.len();
373    let mut g_costs = vec![f32::INFINITY; node_count];
374    let mut came_from: Vec<usize> = vec![usize::MAX; node_count];
375    let mut closed = vec![false; node_count];
376    g_costs[start_idx] = 0.0;
377    let mut open = BinaryHeap::new();
378    open.push(HpaNode {
379        f: octile_dist(
380            nodes[start_idx].x,
381            nodes[start_idx].y,
382            nodes[goal_idx].x,
383            nodes[goal_idx].y,
384        ),
385        g: 0.0,
386        idx: start_idx,
387    });
388    while let Some(current) = open.pop() {
389        if closed[current.idx] {
390            continue;
391        }
392        closed[current.idx] = true;
393        if current.idx == goal_idx {
394            let abstract_path = reconstruct_abstract(&came_from, start_idx, goal_idx, &nodes);
395            return refine_path(grid, &abstract_path, unit_size);
396        }
397        for edge in &edges[current.idx] {
398            if closed[edge.to] {
399                continue;
400            }
401            let tentative_g = current.g + edge.cost;
402            if tentative_g < g_costs[edge.to] {
403                g_costs[edge.to] = tentative_g;
404                came_from[edge.to] = current.idx;
405                let h = octile_dist(
406                    nodes[edge.to].x,
407                    nodes[edge.to].y,
408                    nodes[goal_idx].x,
409                    nodes[goal_idx].y,
410                );
411                open.push(HpaNode {
412                    f: tentative_g + h,
413                    g: tentative_g,
414                    idx: edge.to,
415                });
416            }
417        }
418    }
419    Option::None
420}
421/// Return true when `goal` is reachable from `start` via the abstract chunk graph (BFS over chunks).
422pub fn is_reachable(abstract_graph: &AbstractGraph, start: (u32, u32), goal: (u32, u32)) -> bool {
423    let cs = abstract_graph.chunk_size;
424    let start_chunk = (start.0 / cs, start.1 / cs);
425    let goal_chunk = (goal.0 / cs, goal.1 / cs);
426    if start_chunk == goal_chunk {
427        return true;
428    }
429    let mut visited: HashSet<(u32, u32)> = HashSet::new();
430    let mut queue = VecDeque::new();
431    visited.insert(start_chunk);
432    queue.push_back(start_chunk);
433    while let Some(current_chunk) = queue.pop_front() {
434        if current_chunk == goal_chunk {
435            return true;
436        }
437        if let Some(chunk) = abstract_graph.chunks.get(&current_chunk) {
438            for &ent_idx in &chunk.entrance_indices {
439                for edge in &abstract_graph.edges[ent_idx] {
440                    let target_chunk = abstract_graph.nodes[edge.to].chunk;
441                    if visited.insert(target_chunk) {
442                        queue.push_back(target_chunk);
443                    }
444                }
445            }
446        }
447    }
448    false
449}
450/// Compute octile (Chebyshev with diagonal weight √2) distance between two grid cells.
451fn octile_dist(x1: u32, y1: u32, x2: u32, y2: u32) -> f32 {
452    let dx = (x1 as f32 - x2 as f32).abs();
453    let dy = (y1 as f32 - y2 as f32).abs();
454    let min = dx.min(dy);
455    let max = dx.max(dy);
456    min * std::f32::consts::SQRT_2 + (max - min)
457}
458/// Walk `came_from` from `goal` back to `start` and return the coordinate path in forward order.
459fn reconstruct_abstract(
460    came_from: &[usize],
461    start: usize,
462    goal: usize,
463    nodes: &[AbstractNode],
464) -> Vec<(u32, u32)> {
465    let mut path = Vec::new();
466    let mut cur = goal;
467    loop {
468        path.push((nodes[cur].x, nodes[cur].y));
469        if cur == start {
470            break;
471        }
472        let prev = came_from[cur];
473        if prev == usize::MAX || prev == cur {
474            break;
475        }
476        cur = prev;
477    }
478    path.reverse();
479    path
480}
481/// Replace each consecutive pair in `abstract_path` with a full A\* segment; return `None` on failure.
482fn refine_path(
483    grid: &NavGrid,
484    abstract_path: &[(u32, u32)],
485    unit_size: u32,
486) -> Option<Vec<(u32, u32)>> {
487    if abstract_path.is_empty() {
488        return Option::None;
489    }
490    if abstract_path.len() == 1 {
491        return Some(abstract_path.to_vec());
492    }
493    let mut full_path: Vec<(u32, u32)> = Vec::new();
494    for i in 0..abstract_path.len() - 1 {
495        let (path, complete) =
496            astar::astar(grid, abstract_path[i], abstract_path[i + 1], unit_size, 0);
497        if !complete {
498            return Option::None;
499        }
500        if let Some(segment) = path {
501            if full_path.is_empty() {
502                full_path.extend_from_slice(&segment);
503            } else {
504                full_path.extend_from_slice(&segment[1..]);
505            }
506        }
507    }
508    if full_path.is_empty() {
509        Option::None
510    } else {
511        Some(full_path)
512    }
