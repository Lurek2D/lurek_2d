1//! Province-level A* pathfinding across adjacency graphs with configurable move costs.
2//!
3//! - Dijkstra-based reachability flood to find all provinces within a cost budget.
4//! - Per-province and per-edge-tag cost modelling with blocked-province exclusion.
5//! - Min-heap priority queue node with reverse ordering for standard `BinaryHeap`.
6//! - Euclidean centroid heuristic for A* admissibility.
7
8use std::cmp::Ordering;
9use std::collections::{BinaryHeap, HashMap, HashSet};
10/// Result of a province-level pathfinding query.
11#[derive(Debug, Clone)]
12pub struct ProvincePath {
13    /// Ordered sequence of province ids from start to destination, inclusive.
14    pub provinces: Vec<u32>,
15    /// Summed move cost along the chosen path.
16    pub total_cost: f64,
17}
18/// Per-province and per-edge-tag move-cost configuration passed to path queries.
19#[derive(Debug, Clone, Default)]
20pub struct ProvinceCostFn {
21    /// Base cost applied to every traversable province.
22    pub default_cost: f64,
23    /// Per-province cost additions keyed by province id.
24    pub province_costs: HashMap<u32, f64>,
25    /// Edge surcharges applied when a crossing edge carries the given tag.
26    pub tag_costs: HashMap<String, f64>,
27    /// Province ids that cannot be entered; paths will not pass through them.
28    pub blocked: HashSet<u32>,
29}
30/// Construction and cost helpers for `ProvinceCostFn`.
31impl ProvinceCostFn {
32    /// Create a default cost function with `default_cost = 1.0` and no blocked provinces.
33    pub fn new() -> Self {
34        Self {
35            default_cost: 1.0,
36            ..Default::default()
37        }
38    }
39    /// Return the total cost to enter `province_id`, or `None` when blocked or infinite.
40    fn cost_for(&self, province_id: u32) -> Option<f64> {
41        if self.blocked.contains(&province_id) {
42            return None;
43        }
44        let base = self.default_cost;
45        let extra = self
46            .province_costs
47            .get(&province_id)
48            .copied()
49            .unwrap_or(0.0);
50        let total = base + extra;
51        if total.is_infinite() {
52            None
53        } else {
54            Some(total)
55        }
56    }
57    /// Sum tag surcharges for an edge whose tag set intersects `edge_tags`.
58    fn edge_cost(&self, edge_tags: &HashSet<String>) -> f64 {
59        let mut extra = 0.0;
60        for (tag, &cost) in &self.tag_costs {
61            if edge_tags.contains(tag) {
62                extra += cost;
63            }
64        }
65        extra
66    }
67}
68/// Internal priority-queue node for province-level A\* and Dijkstra.
69struct AStarNode {
70    /// Province id held by this node.
71    province_id: u32,
72    /// f-score (g + h for A\*, g for Dijkstra).
73    f_score: f64,
74}
75/// Equality by f-score.
76impl PartialEq for AStarNode {
77    fn eq(&self, other: &Self) -> bool {
78        self.f_score == other.f_score
79    }
80}
81/// Marker trait required by `Ord`; delegates equality to `PartialEq`.
82impl Eq for AStarNode {}
83
84/// Delegates to `Ord`.
85impl PartialOrd for AStarNode {
86    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
87        Some(self.cmp(other))
88    }
89}
90/// Reverse ordering so `BinaryHeap` is a min-heap on f-score.
91impl Ord for AStarNode {
92    fn cmp(&self, other: &Self) -> Ordering {
93        other
94            .f_score
95            .partial_cmp(&self.f_score)
96            .unwrap_or(Ordering::Equal)
97    }
98}
99/// Run A\* across province adjacency; return the cheapest path and its cost, or `None` when unreachable.
100pub fn find_province_path(
101    neighbors: &HashMap<u32, Vec<u32>>,
102    centroids: &HashMap<u32, (f32, f32)>,
103    edge_tags: &HashMap<(u32, u32), HashSet<String>>,
104    from: u32,
105    to: u32,
106    cost_fn: &ProvinceCostFn,
107) -> Option<ProvincePath> {
108    if from == to {
109        return Some(ProvincePath {
110            provinces: vec![from],
111            total_cost: 0.0,
112        });
113    }
114    let goal_centroid = centroids.get(&to)?;
115    centroids.get(&from)?;
116    let mut open = BinaryHeap::new();
117    let mut g_score: HashMap<u32, f64> = HashMap::new();
118    let mut came_from: HashMap<u32, u32> = HashMap::new();
119    let mut closed: HashSet<u32> = HashSet::new();
120    g_score.insert(from, 0.0);
121    let h = centroid_distance(centroids.get(&from)?, goal_centroid);
122    open.push(AStarNode {
123        province_id: from,
124        f_score: h,
125    });
126    while let Some(current) = open.pop() {
127        let current_id = current.province_id;
128        if current_id == to {
129            let mut path = vec![to];
130            let mut node = to;
131            while let Some(&prev) = came_from.get(&node) {
132                path.push(prev);
133                node = prev;
134            }
135            path.reverse();
136            return Some(ProvincePath {
137                provinces: path,
138                total_cost: g_score[&to],
139            });
140        }
141        if !closed.insert(current_id) {
142            continue;
143        }
144        let current_g = g_score[&current_id];
145        let empty_neighbors = Vec::new();
146        let neighbor_ids = neighbors.get(&current_id).unwrap_or(&empty_neighbors);
147        for &neighbor_id in neighbor_ids {
148            if closed.contains(&neighbor_id) {
149                continue;
150            }
151            let step_cost = match cost_fn.cost_for(neighbor_id) {
152                Some(c) => c,
153                None => continue,
154            };
155            let edge_key = if current_id <= neighbor_id {
156                (current_id, neighbor_id)
157            } else {
158                (neighbor_id, current_id)
159            };
160            let edge_extra = edge_tags
161                .get(&edge_key)
162                .map(|tags| cost_fn.edge_cost(tags))
163                .unwrap_or(0.0);
164            let tentative_g = current_g + step_cost + edge_extra;
165            if tentative_g < *g_score.get(&neighbor_id).unwrap_or(&f64::INFINITY) {
166                g_score.insert(neighbor_id, tentative_g);
167                came_from.insert(neighbor_id, current_id);
168                let nc = match centroids.get(&neighbor_id) {
169                    Some(c) => c,
170                    None => continue,
171                };
172                let h = centroid_distance(nc, goal_centroid);
173                open.push(AStarNode {
174                    province_id: neighbor_id,
175                    f_score: tentative_g + h,
176                });
177            }
178        }
179    }
180    None
181}
182/// Return all provinces reachable from `start` within `max_cost` as a map of `province_id → cost`.
183pub fn province_reachable(
184    neighbors: &HashMap<u32, Vec<u32>>,
185    edge_tags: &HashMap<(u32, u32), HashSet<String>>,
186    start: u32,
187    max_cost: f64,
188    cost_fn: &ProvinceCostFn,
189) -> HashMap<u32, f64> {
190    let mut dist: HashMap<u32, f64> = HashMap::new();
191    let mut heap = BinaryHeap::new();
192    let mut visited: HashSet<u32> = HashSet::new();
193    dist.insert(start, 0.0);
194    heap.push(AStarNode {
195        province_id: start,
196        f_score: 0.0,
197    });
198    while let Some(current) = heap.pop() {
199        let current_id = current.province_id;
200        let current_dist = *dist.get(&current_id).unwrap_or(&f64::INFINITY);
201        if current_dist > max_cost || !visited.insert(current_id) {
202            continue;
203        }
204        let empty_neighbors = Vec::new();
205        let neighbor_ids = neighbors.get(&current_id).unwrap_or(&empty_neighbors);
206        for &neighbor_id in neighbor_ids {
207            if visited.contains(&neighbor_id) {
208                continue;
209            }
210            let step_cost = match cost_fn.cost_for(neighbor_id) {
211                Some(c) => c,
212                None => continue,
213            };
214            let edge_key = if current_id <= neighbor_id {
215                (current_id, neighbor_id)
216            } else {
217                (neighbor_id, current_id)
218            };
219            let edge_extra = edge_tags
220                .get(&edge_key)
221                .map(|tags| cost_fn.edge_cost(tags))
222                .unwrap_or(0.0);
223            let new_dist = current_dist + step_cost + edge_extra;
224            if new_dist <= max_cost && new_dist < *dist.get(&neighbor_id).unwrap_or(&f64::INFINITY)
225            {
226                dist.insert(neighbor_id, new_dist);
227                heap.push(AStarNode {
228                    province_id: neighbor_id,
229                    f_score: new_dist,
230                });
231            }
232        }
233    }
234    dist.into_iter()
235        .filter(|(id, _)| visited.contains(id))
236        .collect()
237}
238/// Return Euclidean distance between two centroid points.
239fn centroid_distance(a: &(f32, f32), b: &(f32, f32)) -> f64 {
240    let dx = (a.0 - b.0) as f64;
241    let dy = (a.1 - b.1) as f64;
242    (dx * dx + dy * dy).sqrt()
243}
