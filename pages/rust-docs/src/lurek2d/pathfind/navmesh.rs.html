1//! Polygon-based navigation mesh for 2D pathfinding.
2//!
3//! - A\* search over polygon adjacency graph with centroid heuristic.
4//! - Ray-cast point-in-polygon containment test.
5//! - Centroid waypoint extraction from polygon corridors.
6//! - Directed and bidirectional polygon connectivity.
7
8use std::cmp::Ordering;
9use std::collections::{BinaryHeap, HashMap};
10/// Polygon-based 2D navigation mesh supporting A\* pathfinding.
11#[derive(Debug, Clone, Default)]
12pub struct NavMesh {
13    /// Vertex lists for each polygon region.
14    polygons: Vec<Vec<(f32, f32)>>,
15    /// Adjacency list: `neighbors[i]` holds polygon indices reachable from polygon `i`.
16    neighbors: Vec<Vec<usize>>,
17}
18/// Construction and pathfinding methods for `NavMesh`.
19impl NavMesh {
20    /// Create an empty mesh. This function is part of the public API.
21    pub fn new() -> Self {
22        Self::default()
23    }
24    /// Add a polygon region with at least 3 vertices; return its index or `None` if fewer than 3 vertices.
25    pub fn add_polygon(&mut self, vertices: Vec<(f32, f32)>) -> Option<usize> {
26        if vertices.len() < 3 {
27            return None;
28        }
29        let id = self.polygons.len();
30        self.polygons.push(vertices);
31        self.neighbors.push(Vec::new());
32        Some(id)
33    }
34    /// Add a directed edge from polygon `a` to `b`; if `bidirectional`, also add the reverse edge.
35    pub fn connect(&mut self, a: usize, b: usize, bidirectional: bool) -> bool {
36        if a >= self.polygons.len() || b >= self.polygons.len() || a == b {
37            return false;
38        }
39        if !self.neighbors[a].contains(&b) {
40            self.neighbors[a].push(b);
41        }
42        if bidirectional && !self.neighbors[b].contains(&a) {
43            self.neighbors[b].push(a);
44        }
45        true
46    }
47    /// Return the total number of registered polygons.
48    pub fn polygon_count(&self) -> usize {
49        self.polygons.len()
50    }
51    /// Find a world-space path from `start` to `goal`; return centroid waypoints or `None` if either point is outside all polygons.
52    pub fn find_path(&self, start: (f32, f32), goal: (f32, f32)) -> Option<Vec<(f32, f32)>> {
53        let start_poly = self.find_polygon_for_point(start)?;
54        let goal_poly = self.find_polygon_for_point(goal)?;
55        if start_poly == goal_poly {
56            return Some(vec![start, goal]);
57        }
58        let corridor = self.astar_polygons(start_poly, goal_poly)?;
59        let mut out = Vec::with_capacity(corridor.len() + 2);
60        out.push(start);
61        for poly in corridor
62            .iter()
63            .skip(1)
64            .take(corridor.len().saturating_sub(2))
65        {
66            out.push(self.centroid(*poly));
67        }
68        out.push(goal);
69        Some(out)
70    }
71    /// Return the index of the first polygon containing `point`, using ray-cast point-in-polygon test.
72    fn find_polygon_for_point(&self, point: (f32, f32)) -> Option<usize> {
73        self.polygons
74            .iter()
75            .position(|poly| point_in_polygon(point, poly))
76    }
77    /// Compute the centroid (average vertex) of polygon `poly_id`.
78    fn centroid(&self, poly_id: usize) -> (f32, f32) {
79        let poly = &self.polygons[poly_id];
80        let mut sx = 0.0;
81        let mut sy = 0.0;
82        for (x, y) in poly {
83            sx += *x;
84            sy += *y;
85        }
86        let inv = 1.0 / poly.len() as f32;
87        (sx * inv, sy * inv)
88    }
89    /// Run A\* over the polygon graph from `start` to `goal`, returning an ordered list of polygon indices.
90    fn astar_polygons(&self, start: usize, goal: usize) -> Option<Vec<usize>> {
91        let mut open = BinaryHeap::new();
92        let mut g_score: HashMap<usize, f32> = HashMap::new();
93        let mut parent: HashMap<usize, usize> = HashMap::new();
94        g_score.insert(start, 0.0);
95        open.push(Node {
96            id: start,
97            f: self.heuristic(start, goal),
98        });
99        while let Some(Node { id, .. }) = open.pop() {
100            if id == goal {
101                return Some(reconstruct(parent, goal));
102            }
103            let current_g = *g_score.get(&id).unwrap_or(&f32::INFINITY);
104            for &next in &self.neighbors[id] {
105                let tentative = current_g + self.distance(id, next);
106                if tentative < *g_score.get(&next).unwrap_or(&f32::INFINITY) {
107                    g_score.insert(next, tentative);
108                    parent.insert(next, id);
109                    open.push(Node {
110                        id: next,
111                        f: tentative + self.heuristic(next, goal),
112                    });
113                }
114            }
115        }
116        None
117    }
118    /// Return the Euclidean distance between the centroids of polygons `a` and `b`.
119    fn distance(&self, a: usize, b: usize) -> f32 {
120        let (ax, ay) = self.centroid(a);
121        let (bx, by) = self.centroid(b);
122        ((bx - ax).powi(2) + (by - ay).powi(2)).sqrt()
123    }
124    /// Heuristic distance estimate for A\* (centroid-to-centroid).
125    fn heuristic(&self, a: usize, b: usize) -> f32 {
126        self.distance(a, b)
127    }
128}
129/// Even-odd ray-cast point-in-polygon test.
130fn point_in_polygon(point: (f32, f32), polygon: &[(f32, f32)]) -> bool {
131    let (px, py) = point;
132    let mut inside = false;
133    let n = polygon.len();
134    let mut j = n - 1;
135    for i in 0..n {
136        let (xi, yi) = polygon[i];
137        let (xj, yj) = polygon[j];
138        let intersect = ((yi > py) != (yj > py))
139            && (px < (xj - xi) * (py - yi) / ((yj - yi).abs().max(f32::EPSILON)) + xi);
140        if intersect {
141            inside = !inside;
142        }
143        j = i;
144    }
145    inside
146}
147/// Walk `parent` chain from `goal` to start and return the path in forward order.
148fn reconstruct(parent: HashMap<usize, usize>, mut current: usize) -> Vec<usize> {
149    let mut out = vec![current];
150    while let Some(prev) = parent.get(&current) {
151        current = *prev;
152        out.push(current);
153    }
154    out.reverse();
155    out
156}
157/// Internal A\* heap node for polygon-level search.
158#[derive(Clone)]
159struct Node {
160    /// Polygon index.
161    id: usize,
162    /// f-score = g + h.
163    f: f32,
164}
165/// Equality by f-score.
166impl PartialEq for Node {
167    /// Compare two nodes by f-score equality.
168    fn eq(&self, other: &Self) -> bool {
169        self.f == other.f
170    }
171}
172/// Marker trait asserting total equality for `Node`.
173impl Eq for Node {}
174
175/// Delegates to `Ord`.
176impl PartialOrd for Node {
177    /// Delegate partial comparison to the total `Ord` implementation.
178    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
179        Some(self.cmp(other))
180    }
181}
182/// Reverse ordering so `BinaryHeap` is a min-heap.
183impl Ord for Node {
184    /// Compare in reverse order to produce a min-heap from `BinaryHeap`.
185    fn cmp(&self, other: &Self) -> Ordering {
186        other.f.partial_cmp(&self.f).unwrap_or(Ordering::Equal)
187    }
188}
