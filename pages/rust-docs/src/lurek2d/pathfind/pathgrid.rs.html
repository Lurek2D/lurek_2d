1//! Grid-based A* pathfinding with 8-directional movement and variable cell costs.
2//!
3//! - Bresenham line-of-sight checks for post-search path smoothing (string-pull).
4//! - World-space coordinate conversion: cell indices map to centres via configurable cell size.
5//! - Diagonal corner-cutting prevention to avoid clipping through blocked corners.
6//! - Octile distance heuristic for consistent and admissible cost estimation.
7
8use std::cmp::Ordering;
9use std::collections::BinaryHeap;
10/// Single cell in a pathfinding grid, storing traversal permission and movement cost.
11#[derive(Debug, Clone)]
12pub struct Cell {
13    /// Whether agents may enter this cell.
14    pub walkable: bool,
15    /// Movement cost multiplier; `1.0` is standard.
16    pub cost: f32,
17}
18/// Default walkable cell with cost `1.0`.
19impl Default for Cell {
20    fn default() -> Self {
21        Self {
22            walkable: true,
23            cost: 1.0,
24        }
25    }
26}
27/// Internal A\* heap node for `PathGrid`.
28#[derive(Debug, Clone)]
29struct AStarNode {
30    /// Flat cell index.
31    idx: usize,
32    /// f-score = g + h.
33    f_cost: f32,
34}
35/// Equality by f-score.
36impl PartialEq for AStarNode {
37    fn eq(&self, other: &Self) -> bool {
38        self.f_cost == other.f_cost
39    }
40}
41/// Marker trait required by `Ord`; equality is by f-score.
42impl Eq for AStarNode {}
43
44/// Delegates to `Ord`.
45impl PartialOrd for AStarNode {
46    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
47        Some(self.cmp(other))
48    }
49}
50/// Reverse ordering so `BinaryHeap` is a min-heap.
51impl Ord for AStarNode {
52    fn cmp(&self, other: &Self) -> Ordering {
53        other
54            .f_cost
55            .partial_cmp(&self.f_cost)
56            .unwrap_or(Ordering::Equal)
57    }
58}
59/// World-space pathfinding grid: cells have world-space size `cell_size`; paths are in world coordinates.
60pub struct PathGrid {
61    /// Grid width in cells.
62    pub width: usize,
63    /// Grid height in cells.
64    pub height: usize,
65    /// World-space dimension of each cell.
66    pub cell_size: f32,
67    /// Per-cell data, row-major.
68    cells: Vec<Cell>,
69}
70/// Construction and pathfinding methods for `PathGrid`.
71impl PathGrid {
72    /// Create a fully walkable `width × height` grid with given world-space `cell_size`.
73    pub fn new(width: usize, height: usize, cell_size: f32) -> Self {
74        Self {
75            width,
76            height,
77            cell_size,
78            cells: vec![Cell::default(); width * height],
79        }
80    }
81    /// Return the flat index for cell `(x, y)`.
82    fn idx(&self, x: usize, y: usize) -> usize {
83        y * self.width + x
84    }
85    /// Return true when `(x, y)` is within the grid dimensions.
86    pub fn in_bounds(&self, x: usize, y: usize) -> bool {
87        x < self.width && y < self.height
88    }
89    /// Set walkability of cell `(x, y)`.
90    pub fn set_walkable(&mut self, x: usize, y: usize, walkable: bool) {
91        if self.in_bounds(x, y) {
92            let idx = self.idx(x, y);
93            self.cells[idx].walkable = walkable;
94        }
95    }
96    /// Return true when `(x, y)` is in-bounds and walkable.
97    pub fn is_walkable(&self, x: usize, y: usize) -> bool {
98        if self.in_bounds(x, y) {
99            self.cells[self.idx(x, y)].walkable
100        } else {
101            false
102        }
103    }
104    /// Set movement cost of cell `(x, y)`.
105    pub fn set_cost(&mut self, x: usize, y: usize, cost: f32) {
106        if self.in_bounds(x, y) {
107            let idx = self.idx(x, y);
108            self.cells[idx].cost = cost;
109        }
110    }
111    /// Return movement cost of cell `(x, y)`, or `f32::INFINITY` when out-of-bounds.
112    pub fn get_cost(&self, x: usize, y: usize) -> f32 {
113        if self.in_bounds(x, y) {
114            self.cells[self.idx(x, y)].cost
115        } else {
116            f32::INFINITY
117        }
118    }
119    /// Run 8-directional A\* from cell `(sx, sy)` to `(gx, gy)`; return world-space waypoints or `None`.
120    pub fn find_path(&self, sx: usize, sy: usize, gx: usize, gy: usize) -> Option<Vec<(f32, f32)>> {
121        if !self.in_bounds(sx, sy) || !self.in_bounds(gx, gy) {
122            return None;
123        }
124        if !self.is_walkable(sx, sy) || !self.is_walkable(gx, gy) {
125            return None;
126        }
127        let total = self.width * self.height;
128        let mut g_cost = vec![f32::INFINITY; total];
129        let mut came_from = vec![usize::MAX; total];
130        let mut closed = vec![false; total];
131        let start_idx = self.idx(sx, sy);
132        let goal_idx = self.idx(gx, gy);
133        g_cost[start_idx] = 0.0;
134        let mut open = BinaryHeap::new();
135        open.push(AStarNode {
136            idx: start_idx,
137            f_cost: self.heuristic(sx, sy, gx, gy),
138        });
139        while let Some(current) = open.pop() {
140            if current.idx == goal_idx {
141                return Some(self.reconstruct_path(&came_from, goal_idx));
142            }
143            if closed[current.idx] {
144                continue;
145            }
146            closed[current.idx] = true;
147            let cx = current.idx % self.width;
148            let cy = current.idx / self.width;
149            for &(dx, dy) in &[
150                (-1i32, -1),
151                (-1, 0),
152                (-1, 1),
153                (0, -1),
154                (0, 1),
155                (1, -1),
156                (1, 0),
157                (1, 1),
158            ] {
159                let nx = cx as i32 + dx;
160                let ny = cy as i32 + dy;
161                if nx < 0 || ny < 0 || nx >= self.width as i32 || ny >= self.height as i32 {
162                    continue;
163                }
164                let nux = nx as usize;
165                let nuy = ny as usize;
166                if !self.is_walkable(nux, nuy) {
167                    continue;
168                }
169                if dx != 0 && dy != 0 && (!self.is_walkable(cx, nuy) || !self.is_walkable(nux, cy))
170                {
171                    continue;
172                }
173                let nidx = self.idx(nux, nuy);
174                if closed[nidx] {
175                    continue;
176                }
177                let move_cost = if dx != 0 && dy != 0 { 1.414 } else { 1.0 };
178                let tentative = g_cost[current.idx] + move_cost * self.cells[nidx].cost;
179                if tentative < g_cost[nidx] {
180                    g_cost[nidx] = tentative;
181                    came_from[nidx] = current.idx;
182                    open.push(AStarNode {
183                        idx: nidx,
184                        f_cost: tentative + self.heuristic(nux, nuy, gx, gy),
185                    });
186                }
187            }
188        }
189        None
190    }
191    /// Run A\* then apply string-pull smoothing to reduce waypoint count.
192    pub fn find_path_smoothed(
193        &self,
194        sx: usize,
195        sy: usize,
196        gx: usize,
197        gy: usize,
198    ) -> Option<Vec<(f32, f32)>> {
199        let path = self.find_path(sx, sy, gx, gy)?;
200        if path.len() <= 2 {
201            return Some(path);
202        }
203        let mut smoothed = vec![path[0]];
204        let mut anchor = 0;
205        let mut current = 1;
206        while current < path.len() - 1 {
207            let next = current + 1;
208            if !self.has_line_of_sight(path[anchor], path[next]) {
209                smoothed.push(path[current]);
210                anchor = current;
211            }
212            current += 1;
213        }
214        smoothed.push(
215            *path
216                .last()
217                .expect("path has >= 3 elements after length guard"),
218        );
219        Some(smoothed)
220    }
221    /// Octile distance heuristic.
222    fn heuristic(&self, x1: usize, y1: usize, x2: usize, y2: usize) -> f32 {
223        let dx = (x1 as f32 - x2 as f32).abs();
224        let dy = (y1 as f32 - y2 as f32).abs();
225        let min = dx.min(dy);
226        let max = dx.max(dy);
227        min * 1.414 + (max - min)
228    }
229    /// Walk `came_from` from `goal_idx` to start and return the world-space path in forward order.
230    fn reconstruct_path(&self, came_from: &[usize], goal_idx: usize) -> Vec<(f32, f32)> {
231        let mut path = Vec::new();
232        let mut current = goal_idx;
233        while current != usize::MAX {
234            let x = current % self.width;
235            let y = current / self.width;
236            path.push(self.cell_center(x, y));
237            current = came_from[current];
238        }
239        path.reverse();
240        path
241    }
242    /// Return the world-space centre of cell `(x, y)`.
243    pub fn cell_center(&self, x: usize, y: usize) -> (f32, f32) {
244        (
245            (x as f32 + 0.5) * self.cell_size,
246            (y as f32 + 0.5) * self.cell_size,
247        )
248    }
249    /// Return true when the Bresenham line from `from` to `to` passes only through walkable cells.
250    fn has_line_of_sight(&self, from: (f32, f32), to: (f32, f32)) -> bool {
251        let fx = (from.0 / self.cell_size) as i32;
252        let fy = (from.1 / self.cell_size) as i32;
253        let tx = (to.0 / self.cell_size) as i32;
254        let ty = (to.1 / self.cell_size) as i32;
255        let mut x = fx;
256        let mut y = fy;
257        let dx = (tx - fx).abs();
258        let dy = (ty - fy).abs();
259        let sx = if fx < tx { 1 } else { -1 };
260        let sy = if fy < ty { 1 } else { -1 };
261        let mut err = dx - dy;
262        loop {
263            if x >= 0 && y >= 0 && (x as usize) < self.width && (y as usize) < self.height {
264                if !self.cells[self.idx(x as usize, y as usize)].walkable {
265                    return false;
266                }
267            } else {
268                return false;
269            }
270            if x == tx && y == ty {
271                break;
272            }
273            let e2 = 2 * err;
274            if e2 > -dy {
275                err -= dy;
276                x += sx;
277            }
278            if e2 < dx {
279                err += dx;
280                y += sy;
281            }
282        }
283        true
284    }
