2//!
3//! - A* pathfinding with Euclidean heuristic for shortest-path queries between regions.
4//! - Bounded Dijkstra reachability returning all regions within a cumulative travel cost.
5//! - Minimum spanning tree computation via Kruskal's algorithm.

9use std::cmp::Ordering;

12/// A named point-of-interest region in the world graph.
13#[derive(Debug, Clone)]
14pub struct WorldRegion {
15    /// Unique region identifier assigned at insertion.
16    pub id: u32,
17    /// Human-readable name used for display or lookup.
18    pub name: String,
19    /// X world-space coordinate.
20    pub x: f32,
21    /// Y world-space coordinate.
22    pub y: f32,
23    /// Arbitrary classification tags attached to this region.
24    pub tags: Vec<String>,
25}
26
27/// A directed or bidirectional connection between two regions.
28#[derive(Debug, Clone)]
29pub struct WorldEdge {
30    /// Source region ID.
31    pub from: u32,
32    /// Destination region ID.
33    pub to: u32,
34    /// Traversal cost; clamped to >= 0 during pathfinding.
35    pub cost: f32,
36    /// When true, the edge can be traversed in both directions.
37    pub bidirectional: bool,
38}

40/// Directed world graph with regions, edges, and graph algorithm methods.
41#[derive(Debug, Clone, Default)]
42pub struct WorldGraph {
43    /// All regions in insertion order.
44    pub regions: Vec<WorldRegion>,
45    /// All edges in insertion order.
46    pub edges: Vec<WorldEdge>,
47    /// Auto-increment counter for new region IDs.
48    next_id: u32,
49}
50
51impl WorldGraph {
52    /// Create an empty world graph. This function is part of the public API.
53    pub fn new() -> Self {
54        Self::default()
55    }
56
57    /// Add a named region at `(x, y)` and return its assigned ID.
58    pub fn add_region(&mut self, name: &str, x: f32, y: f32) -> u32 {
59        let id = self.next_id;
60        self.next_id += 1;
61        self.regions.push(WorldRegion {
62            id,
63            name: name.to_string(),
64            x,
65            y,
66            tags: Vec::new(),
67        });
68        id
69    }
70
71    /// Add an edge from `from` to `to` with the given `cost`; bidirectional edges traverse both ways.
72    pub fn add_edge(&mut self, from: u32, to: u32, cost: f32, bidirectional: bool) {
73        self.edges.push(WorldEdge {
74            from,
75            to,
76            cost,
77            bidirectional,
78        });
79    }
80
81    /// Find the shortest path from `from` to `to` using A* with Euclidean heuristic; returns `None` when no path exists.
82    pub fn find_path(&self, from: u32, to: u32) -> Option<Vec<u32>> {
83        if from == to {
84            return Some(vec![from]);
85        }
86        let pos: HashMap<u32, (f32, f32)> =
87            self.regions.iter().map(|r| (r.id, (r.x, r.y))).collect();
88        let goal_pos = pos.get(&to).copied().unwrap_or((0.0, 0.0));
89        let heuristic = |a: u32| -> f32 {
90            let (ax, ay) = pos.get(&a).copied().unwrap_or((0.0, 0.0));
91            ((ax - goal_pos.0).powi(2) + (ay - goal_pos.1).powi(2)).sqrt()
92        };
93        let mut g_cost: HashMap<u32, f32> = HashMap::new();
94        let mut came_from: HashMap<u32, u32> = HashMap::new();
95        let mut open: BinaryHeap<WNode> = BinaryHeap::new();
96        g_cost.insert(from, 0.0);
97        open.push(WNode {
98            id: from,
99            f: heuristic(from),
100        });
101        while let Some(WNode { id, .. }) = open.pop() {
102            if id == to {
103                return Some(self.reconstruct(&came_from, to));
104            }
105            let cur_g = *g_cost.get(&id).unwrap_or(&f32::MAX);
106            for edge in self.edges.iter() {
107                let neighbor = if edge.from == id {
108                    edge.to
109                } else if edge.bidirectional && edge.to == id {
110                    edge.from
111                } else {
112                    continue;
113                };
114                let new_g = cur_g + edge.cost.max(0.0);
115                if new_g < *g_cost.get(&neighbor).unwrap_or(&f32::MAX) {
116                    g_cost.insert(neighbor, new_g);
117                    came_from.insert(neighbor, id);
118                    open.push(WNode {
119                        id: neighbor,
120                        f: new_g + heuristic(neighbor),
121                    });
122                }
123            }
124        }
125        None
126    }
127
128    /// Return all region IDs reachable from `start` within cumulative edge cost `max_cost` using bounded Dijkstra.
129    pub fn reachable_from(&self, start: u32, max_cost: f32) -> Vec<u32> {
130        let mut dist: HashMap<u32, f32> = HashMap::new();
131        let mut heap: BinaryHeap<WNode> = BinaryHeap::new();
132        dist.insert(start, 0.0);
133        heap.push(WNode { id: start, f: 0.0 });
134        let mut result = Vec::new();
135        while let Some(WNode { id, f: cost }) = heap.pop() {
136            if cost > *dist.get(&id).unwrap_or(&f32::MAX) {
137                continue;
138            }
139            result.push(id);
140            for edge in self.edges.iter() {
141                let nb = if edge.from == id {
142                    edge.to
143                } else if edge.bidirectional && edge.to == id {
144                    edge.from
145                } else {
146                    continue;
147                };
148                let new_cost = cost + edge.cost.max(0.0);
149                if new_cost <= max_cost && new_cost < *dist.get(&nb).unwrap_or(&f32::MAX) {
150                    dist.insert(nb, new_cost);
151                    heap.push(WNode {
152                        id: nb,
153                        f: new_cost,
154                    });
155                }
156            }
157        }
158        result
159    }

161    /// Compute a minimum spanning tree using Kruskal's algorithm; returns `(from, to, cost)` triples.
162    pub fn mst(&self) -> Vec<(u32, u32, f32)> {
163        let mut sorted_edges: Vec<&WorldEdge> = self.edges.iter().collect();
164        sorted_edges.sort_by(|a, b| a.cost.partial_cmp(&b.cost).unwrap_or(Ordering::Equal));
165        let ids: Vec<u32> = self.regions.iter().map(|r| r.id).collect();
166        let mut parent: HashMap<u32, u32> = ids.iter().map(|&id| (id, id)).collect();
167        fn find(parent: &mut HashMap<u32, u32>, x: u32) -> u32 {
168            if parent[&x] != x {
169                let p = find(parent, parent[&x]);
170                parent.insert(x, p);
171            }
172            parent[&x]
173        }
174        let mut result = Vec::new();
175        for edge in sorted_edges {
176            let rx = find(&mut parent, edge.from);
177            let ry = find(&mut parent, edge.to);
178            if rx != ry {
179                parent.insert(rx, ry);
180                result.push((edge.from, edge.to, edge.cost));
181            }
182        }
183        result
184    }
185
186    /// Return references to all regions in insertion order.
187    pub fn to_regions_list(&self) -> Vec<&WorldRegion> {
188        self.regions.iter().collect()
189    }
190
191    /// Walk `came_from` back from `cur` to the start and return the forward path.
192    fn reconstruct(&self, came_from: &HashMap<u32, u32>, mut cur: u32) -> Vec<u32> {
193        let mut path = vec![cur];
194        while let Some(&prev) = came_from.get(&cur) {
195            path.push(prev);
196            cur = prev;
197        }
198        path.reverse();
199        path
200    }
201}

203/// Generate a random world graph with `region_count` regions inside a `width × height` bounding box, connecting each to its 3 nearest neighbours.
204pub fn generate_world_graph(width: f32, height: f32, region_count: u32, seed: u64) -> WorldGraph {
205    let mut rng = Lcg::new(seed);
206    let mut graph = WorldGraph::new();
207    let mut positions: Vec<(u32, f32, f32)> = Vec::with_capacity(region_count as usize);
208    for i in 0..region_count {
209        let x = rng.next_f32() * width;
210        let y = rng.next_f32() * height;
211        let id = graph.add_region(&format!("Region{i}"), x, y);
212        positions.push((id, x, y));
213    }
214    let k = 3usize;
215    let ids: Vec<u32> = positions.iter().map(|p| p.0).collect();
216    for &(id, x, y) in &positions {
217        let mut dists: Vec<(u32, f32)> = ids
218            .iter()
219            .filter(|&&other| other != id)
220            .map(|&other| {
221                let (_, ox, oy) = positions[other as usize];
222                let d = ((x - ox).powi(2) + (y - oy).powi(2)).sqrt();
223                (other, d)
224            })
225            .collect();
226        dists.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap_or(Ordering::Equal));
227        for (other, cost) in dists.into_iter().take(k) {
228            graph.add_edge(id, other, cost, true);
229        }
230    }
231    graph
232}

234/// Min-heap entry used by A* and bounded Dijkstra; `f` is negated to make `BinaryHeap` act as a min-heap.
235#[derive(Clone)]
236struct WNode {
237    /// Region ID at this heap entry.
238    id: u32,
239    /// Priority value; lower is better — `Ord` implementation reverses comparison.
240    f: f32,
241}
242
243/// Delegate equality to `f` field comparison.
244impl PartialEq for WNode {
245    fn eq(&self, o: &Self) -> bool {
246        self.f == o.f
247    }
248}
249
250/// Satisfy `Eq` bound required by `BinaryHeap`.
251impl Eq for WNode {}
252
253/// Provide partial ordering over `f` for `Ord`.
254impl PartialOrd for WNode {
255    fn partial_cmp(&self, o: &Self) -> Option<Ordering> {
256        Some(self.cmp(o))
257    }
258}
259
260/// Reverse `f` ordering so `BinaryHeap` operates as a min-heap.
261impl Ord for WNode {
262    fn cmp(&self, o: &Self) -> Ordering {
263        o.f.partial_cmp(&self.f).unwrap_or(Ordering::Equal)
264    }
