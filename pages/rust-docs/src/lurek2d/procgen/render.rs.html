1//! Tileable Perlin noise grid generation and cell access.
2//!
3//! - Conversion to grayscale RGBA byte buffers and `ImageData`.
4//! - Batch render-command generation for grid visualization.
5
6use crate::image::ImageData;
7use crate::procgen::noise::perlin_noise_periodic;
8use crate::procgen::scalar_map_to_rgba_bytes;
9use crate::render::renderer::{DrawMode, RenderCommand};
10
11/// Flat noise grid with tiling-Perlin cell values in 0.0–1.0.
12pub struct NoiseGrid {
13    /// Grid width in cells.
14    pub width: u32,
15    /// Grid height in cells.
16    pub height: u32,
17    /// Row-major cell values; index = `y * width + x`.
18    pub cells: Vec<f32>,
19}
20
21impl NoiseGrid {
22    /// Build a tileable Perlin noise grid at the given `scale`; scale is clamped to >= 1e-6.
23    pub fn from_perlin(width: u32, height: u32, scale: f64) -> Self {
24        let scale = scale.max(1e-6);
25        let px = width as f64 * scale;
26        let py = height as f64 * scale;
27        let size = (width * height) as usize;
28        let mut cells = Vec::with_capacity(size);
29        for y in 0..height {
30            for x in 0..width {
31                let nx = x as f64 * scale;
32                let ny = y as f64 * scale;
33                let v = (perlin_noise_periodic(nx, ny, px, py) * 0.5 + 0.5).clamp(0.0, 1.0) as f32;
34                cells.push(v);
35            }
36        }
37        Self {
38            width,
39            height,
40            cells,
41        }
42    }
43    /// Convert the cell grid to a flat grayscale RGBA byte buffer at 4 bytes per cell.
44    pub fn to_rgba_bytes(&self) -> Vec<u8> {
45        scalar_map_to_rgba_bytes(&self.cells)
46    }
47    /// Generate `SetColor` + `Rectangle` render commands for each cell at `cell_size` pixels; returns an empty vec for empty grids.
48    pub fn generate_render_commands(&self, cell_size: f32) -> Vec<RenderCommand> {
49        if self.cells.is_empty() {
50            return Vec::new();
51        }
52        let mut cmds = Vec::with_capacity(self.cells.len() * 2);
53        for (i, &v) in self.cells.iter().enumerate() {
54            let col = i as u32;
55            let row = i as u32 / self.width;
56            let x = (col % self.width) as f32 * cell_size;
57            let y = row as f32 * cell_size;
58            cmds.push(RenderCommand::SetColor(v, v, v, 1.0));
59            cmds.push(RenderCommand::Rectangle {
60                mode: DrawMode::Fill,
61                x,
62                y,
63                w: cell_size,
64                h: cell_size,
65            });
66        }
67        cmds
68    }
69    /// Render the grid into a new `ImageData` as grayscale RGBA pixels.
70    pub fn draw_to_image(&self) -> ImageData {
71        let mut img = ImageData::new(self.width, self.height);
72        for y in 0..self.height {
73            for x in 0..self.width {
74                let v = self.cells[(y * self.width + x) as usize];
75                let g = (v * 255.0) as u8;
76                img.set_pixel(x, y, g, g, g, 255);
77            }
78        }
79        img
80    }
81}
