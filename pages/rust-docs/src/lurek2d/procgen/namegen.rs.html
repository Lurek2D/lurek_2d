1//! Markov-chain name generator trained on arbitrary word corpora.
2//!
3//! - Configurable n-gram order controls fidelity-vs-variety tradeoff.
4//! - Deterministic output via internal LCG seeding for reproducible generation.
5//! - Batch generation with length constraints and bounded retry logic.
6
7use crate::procgen::lcg::Lcg;
8use std::collections::HashMap;
9
10/// Markov-chain name generator trained on a corpus of example strings.
11pub struct NameGen {
12    /// N-gram order; higher values produce more faithful but less varied names.
13    order: usize,
14    /// Map from n-gram key string to list of observed following characters.
15    chain: HashMap<String, Vec<char>>,
16    /// Internal LCG for reproducible sampling.
17    rng: Lcg,
18}
19
20impl NameGen {
21    /// Build a chain from `training` words at the given `order` and deterministic `seed`.
22    pub fn new(training: &[&str], order: usize, seed: u64) -> Self {
23        let order = order.max(1);
24        let mut chain: HashMap<String, Vec<char>> = HashMap::new();
25        for word in training {
26            let padded: String = std::iter::repeat_n('\x00', order)
27                .chain(word.chars())
28                .chain(std::iter::once('\x01'))
29                .collect();
30            let chars: Vec<char> = padded.chars().collect();
31            for i in 0..chars.len().saturating_sub(order) {
32                let key: String = chars[i..i + order].iter().collect();
33                chain.entry(key).or_default().push(chars[i + order]);
34            }
35        }
36        Self {
37            order,
38            chain,
39            rng: Lcg::new(seed),
40        }
41    }
42
43    /// Sample up to 64 candidate names and return the first one with `min_len..=max_len` characters; returns an empty string when all attempts fail.
44    pub fn generate(&mut self, min_len: usize, max_len: usize) -> String {
45        for _ in 0..64 {
46            if let Some(name) = self.try_generate(max_len) {
47                if name.len() >= min_len {
48                    return name;
49                }
50            }
51        }
52        String::new()
53    }
54
55    /// Generate `n` names each satisfying `min_len`/`max_len`; names that fail all attempts are empty strings.
56    pub fn generate_n(&mut self, n: usize, min_len: usize, max_len: usize) -> Vec<String> {
57        (0..n).map(|_| self.generate(min_len, max_len)).collect()
58    }
59
60    /// Attempt one chain walk up to `max_len` characters; returns `None` when the chain has no successor.
61    fn try_generate(&mut self, max_len: usize) -> Option<String> {
62        let mut context: Vec<char> = std::iter::repeat_n('\x00', self.order).collect();
63        let mut name = String::new();
64        for _ in 0..max_len + self.order + 4 {
65            let key: String = context.iter().collect();
66            let options = self.chain.get(&key)?;
67            if options.is_empty() {
68                return None;
69            }
70            let idx = (self.rng.next() as usize) % options.len();
71            let next = options[idx];
72            if next == '\x01' {
73                break;
74            }
75            if next != '\x00' {
76                name.push(next);
77            }
78            if name.len() >= max_len {
79                break;
80            }
81            context.remove(0);
82            context.push(next);
83        }
84        if name.is_empty() {
85            None
86        } else {
87            Some(capitalise(name))
88        }
89    }
90}
91
92/// Capitalise the first character of `s`; returns `s` unchanged when empty.
93fn capitalise(s: String) -> String {
94    let mut chars = s.chars();
95    match chars.next() {
96        None => s,
97        Some(c) => c.to_uppercase().to_string() + chars.as_str(),
98    }
99}
