1//! Poisson-disk sampling: generate evenly-spaced random 2D point distributions.
2//!
3//! - Uses Bridson's algorithm with grid acceleration for O(n) rejection.
4//! - Deterministic via seeded LCG; produces `(x, y)` pair vectors.
5
6use super::lcg::Lcg;
7
8/// Generate a Poisson disk sample set in a `width × height` rectangle with minimum distance `min_dist`.
9///
10/// Uses `max_attempts` candidate trials per active point and `seed` for reproducibility.
11/// Returns all accepted points as `(x, y)` pairs.
12pub fn poisson_disk(
13    width: f32,
14    height: f32,
15    min_dist: f32,
16    max_attempts: u32,
17    seed: u64,
18) -> Vec<(f32, f32)> {
19    use std::f32::consts::PI;
20    let mut rng = Lcg::new(seed);
21    let cell_size = min_dist / std::f32::consts::SQRT_2;
22    let grid_w = (width / cell_size).ceil() as usize + 1;
23    let grid_h = (height / cell_size).ceil() as usize + 1;
24    let mut grid: Vec<Option<usize>> = vec![None; grid_w * grid_h];
25    let mut points: Vec<(f32, f32)> = Vec::new();
26    let mut active: Vec<usize> = Vec::new();
27    let first = (rng.next_f32() * width, rng.next_f32() * height);
28    points.push(first);
29    active.push(0);
30    let gx = (first.0 / cell_size) as usize;
31    let gy = (first.1 / cell_size) as usize;
32    if gx < grid_w && gy < grid_h {
33        grid[gy * grid_w + gx] = Some(0);
34    }
35    while !active.is_empty() {
36        let idx = (rng.next() as usize) % active.len();
37        let point = points[active[idx]];
38        let mut found = false;
39        for _ in 0..max_attempts {
40            let angle = rng.next_f32() * 2.0 * PI;
41            let dist = min_dist + rng.next_f32() * min_dist;
42            let nx = point.0 + angle.cos() * dist;
43            let ny = point.1 + angle.sin() * dist;
44            if nx < 0.0 || ny < 0.0 || nx >= width || ny >= height {
45                continue;
46            }
47            let gx = (nx / cell_size) as usize;
48            let gy = (ny / cell_size) as usize;
49            let mut too_close = false;
50            let search_radius = 2usize;
51            for dy in 0..=(search_radius * 2) {
52                let cy = (gy + dy).wrapping_sub(search_radius);
53                if cy >= grid_h {
54                    continue;
55                }
56                for dx_off in 0..=(search_radius * 2) {
57                    let cx = (gx + dx_off).wrapping_sub(search_radius);
58                    if cx >= grid_w {
59                        continue;
60                    }
61                    if let Some(pi) = grid[cy * grid_w + cx] {
62                        let (qx, qy) = points[pi];
63                        let ddx = nx - qx;
64                        let ddy = ny - qy;
65                        if ddx * ddx + ddy * ddy < min_dist * min_dist {
66                            too_close = true;
67                            break;
68                        }
69                    }
70                }
71                if too_close {
72                    break;
73                }
74            }
75            if !too_close {
76                let new_idx = points.len();
77                points.push((nx, ny));
78                active.push(new_idx);
79                if gx < grid_w && gy < grid_h {
80                    grid[gy * grid_w + gx] = Some(new_idx);
81                }
82                found = true;
83                break;
84            }
85        }
86        if !found {
87            active.swap_remove(idx);
88        }
89    }
90    points
91}
