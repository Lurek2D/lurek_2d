1//! Random room placement with overlap rejection and configurable size ranges.
2//!
3//! - L-shaped corridor carving between consecutive room centres.
4//! - Flat row-major tile grid output (wall / floor / corridor byte values).
5//! - Prefab stamp system that centre-pastes named mask patterns into placed rooms.
6//! - Round-robin prefab assignment across all placed rooms.
7
8use crate::procgen::lcg::Lcg;

10/// Axis-aligned room rectangle placed in a rooms dungeon grid.
11#[derive(Debug, Clone)]
12pub struct Room {
13    /// Left edge in tile coordinates.
14    pub x: u32,
15    /// Top edge in tile coordinates.
16    pub y: u32,
17    /// Room width in tiles.
18    pub w: u32,
19    /// Room height in tiles.
20    pub h: u32,
21}
22
23/// Geometric helpers for room overlap testing and centre calculation.
24impl Room {
25    /// Return true if this room overlaps `other` with a 1-tile buffer on each side.
26    fn overlaps(&self, other: &Room) -> bool {
27        self.x < other.x + other.w + 1
28            && self.x + self.w + 1 > other.x
29            && self.y < other.y + other.h + 1
30            && self.y + self.h + 1 > other.y
31    }
32
33    /// Return the centre tile coordinates of this room.
34    fn center(&self) -> (u32, u32) {
35        (self.x + self.w / 2, self.y + self.h / 2)
36    }
37}

39/// Configuration for a random-room dungeon generation run.
40#[derive(Debug, Clone)]
41pub struct RoomsOpts {
42    /// Total dungeon width in tiles.
43    pub width: u32,
44    /// Total dungeon height in tiles.
45    pub height: u32,
46    /// Maximum number of rooms to place; actual count may be lower when space is scarce.
47    pub max_rooms: u32,
48    /// Minimum room dimension in tiles (both width and height).
49    pub min_room_size: u32,
50    /// Maximum room dimension in tiles (both width and height).
51    pub max_room_size: u32,
52    /// Seed for the internal `Lcg` RNG.
53    pub seed: u64,
54}
55
56/// Provide defaults for a 64×64 dungeon: up to 20 rooms of 4–12 tiles.
57impl Default for RoomsOpts {
58    fn default() -> Self {
59        Self {
60            width: 64,
61            height: 64,
62            max_rooms: 20,
63            min_room_size: 4,

70/// Completed rooms dungeon: placed rooms, corridor segments, and the flat tile grid.
71#[derive(Debug, Clone)]
72pub struct RoomsDungeon {
73    /// All successfully placed rooms.
74    pub rooms: Vec<Room>,
75    /// Corridor centre-line segments as `(x1, y1, x2, y2)` tile pairs.
76    pub corridors: Vec<(u32, u32, u32, u32)>,
77    /// Flat row-major tile grid: `0` = wall, `1` = floor, `2` = corridor.
78    pub grid: Vec<u8>,
79}

81/// Template shape used to overwrite a room's interior with a named stamp pattern.
82#[derive(Debug, Clone)]
83pub struct RoomPrefabStamp {
84    /// Prefab identifier matched to placements in the output.
85    pub name: String,
86    /// Non-zero mask bytes indicate which cells are stamped with `stamp_value`.
87    pub mask: Vec<u8>,
88    /// Mask width in tiles; must be <= room width to be placed.
89    pub width: u32,
90    /// Mask height in tiles; must be <= room height to be placed.
91    pub height: u32,
92}

94/// A `RoomPrefabStamp` placed at a concrete tile position within a room.
95#[derive(Debug, Clone)]
96pub struct PlacedRoomPrefab {
97    /// Originating prefab name.
98    pub name: String,
99    /// Top-left tile X position in the dungeon grid.
100    pub x: u32,
101    /// Top-left tile Y position in the dungeon grid.
102    pub y: u32,
103    /// Placed width in tiles (copied from stamp).
104    pub width: u32,
105    /// Placed height in tiles (copied from stamp).
106    pub height: u32,
107}

109/// Generate a rooms dungeon from `opts`; returns rooms, L-shaped corridors, and the tile grid.
110pub fn rooms_dungeon(opts: &RoomsOpts) -> RoomsDungeon {
111    let mut rng = Lcg::new(opts.seed);
112    let mut grid = vec![0u8; (opts.width * opts.height) as usize];
113    let mut rooms: Vec<Room> = Vec::new();
114    let size_range = opts.max_room_size - opts.min_room_size + 1;
115    for _ in 0..opts.max_rooms * 4 {
116        if rooms.len() as u32 >= opts.max_rooms {
117            break;
118        }
119        let rw = opts.min_room_size + (rng.next() as u32) % size_range.max(1);
120        let rh = opts.min_room_size + (rng.next() as u32) % size_range.max(1);
121        if opts.width <= rw + 2 || opts.height <= rh + 2 {
122            continue;
123        }
124        let rx = 1 + (rng.next() as u32) % (opts.width - rw - 2).max(1);
125        let ry = 1 + (rng.next() as u32) % (opts.height - rh - 2).max(1);
126        let candidate = Room {
127            x: rx,
128            y: ry,
129            w: rw,
130            h: rh,
131        };
132        if rooms.iter().any(|r| r.overlaps(&candidate)) {
133            continue;
134        }
135        for dy in 0..rh {
136            for dx in 0..rw {
137                let idx = ((ry + dy) * opts.width + rx + dx) as usize;
138                grid[idx] = 1;
139            }
140        }
141        rooms.push(candidate);
142    }
143    let mut corridors = Vec::new();
144    for i in 1..rooms.len() {
145        let (x1, y1) = rooms[i - 1].center();
146        let (x2, y2) = rooms[i].center();
147        corridors.push((x1, y1, x2, y2));
148        let (sx, ex) = if x1 < x2 { (x1, x2) } else { (x2, x1) };
149        for cx in sx..=ex {
150            let idx = (y1 * opts.width + cx) as usize;
151            if idx < grid.len() && grid[idx] == 0 {
152                grid[idx] = 2;
153            }
154        }
155        let (sy, ey) = if y1 < y2 { (y1, y2) } else { (y2, y1) };
156        for cy in sy..=ey {
157            let idx = (cy * opts.width + x2) as usize;
158            if idx < grid.len() && grid[idx] == 0 {
159                grid[idx] = 2;
160            }
161        }
162    }
163    RoomsDungeon {
164        rooms,
165        corridors,
166        grid,
167    }
168}
169/// Generate a rooms dungeon, centre-stamp `prefabs` (round-robin) in each room, and return `(dungeon, placements)`.
170pub fn rooms_dungeon_with_prefabs(
171    opts: &RoomsOpts,
172    prefabs: &[RoomPrefabStamp],
173    stamp_value: u8,
174) -> (RoomsDungeon, Vec<PlacedRoomPrefab>) {
175    let mut dungeon = rooms_dungeon(opts);
176    let placements = apply_prefab_stamps(
177        &mut dungeon.grid,
178        opts.width,
179        opts.height,
180        &dungeon.rooms,
181        prefabs,
182        stamp_value,
183    );
184    (dungeon, placements)
185}
186/// Centre-stamp `prefabs` into `rooms` using non-zero mask bytes written as `stamp_value`; returns all placements.
187fn apply_prefab_stamps(
188    grid: &mut [u8],
189    grid_w: u32,
190    grid_h: u32,
191    rooms: &[Room],
192    prefabs: &[RoomPrefabStamp],
193    stamp_value: u8,
194) -> Vec<PlacedRoomPrefab> {
195    if prefabs.is_empty() {
196        return Vec::new();
197    }
198    let mut out = Vec::new();
199    for (i, room) in rooms.iter().enumerate() {
200        let prefab = &prefabs[i % prefabs.len()];
201        if prefab.width == 0 || prefab.height == 0 {
202            continue;
203        }
204        if prefab.width > room.w || prefab.height > room.h {
205            continue;
206        }
207        let ox = room.x + (room.w - prefab.width) / 2;
208        let oy = room.y + (room.h - prefab.height) / 2;
209        for py in 0..prefab.height {
210            for px in 0..prefab.width {
211                let pi = (py * prefab.width + px) as usize;
212                if prefab.mask.get(pi).copied().unwrap_or(0) == 0 {
213                    continue;
214                }
215                let gx = ox + px;
216                let gy = oy + py;
217                if gx < grid_w && gy < grid_h {
218                    let gi = (gy * grid_w + gx) as usize;
219                    if let Some(cell) = grid.get_mut(gi) {
220                        *cell = stamp_value;
221                    }
222                }
223            }
224        }
225        out.push(PlacedRoomPrefab {
226            name: prefab.name.clone(),
227            x: ox,
228            y: oy,
229            width: prefab.width,
230            height: prefab.height,
231        });
232    }
233    out
