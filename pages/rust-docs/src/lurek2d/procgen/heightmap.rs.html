1//! Procedural heightmap generation from FBM Perlin noise with configurable octaves, lacunarity, and persistence.
2//!
3//! - Simple hydraulic erosion pass that redistributes height differences across 4-connected neighbours.
4//! - Construction from raw noise maps or cellular automata grids with automatic normalisation to 0.0–1.0.
5//! - Clamped coordinate access and flat RGBA byte export for GPU texture upload.
6//! - Deterministic output controlled by a seed value passed through to the noise generator.
7
8use crate::procgen::noise::{FractalType, MapGenOptions, NoiseGenerator, NoiseKind};
9use crate::procgen::scalar_map_to_rgba_bytes;
10
11/// Parameters for procedural heightmap generation.
12#[derive(Debug, Clone)]
13pub struct HeightmapOpts {
14    /// Grid width in cells.
15    pub width: u32,
16    /// Grid height in cells.
17    pub height: u32,
18    /// Noise frequency scale applied to both axes; smaller = smoother features.
19    pub scale: f64,
20    /// Number of FBM octaves summed together.
21    pub octaves: u32,
22    /// Frequency multiplier per octave (typically 2.0).
23    pub lacunarity: f64,
24    /// Amplitude multiplier per octave; controls how much each octave contributes.
25    pub persistence: f64,
26    /// Seed passed to `NoiseGenerator::new`.
27    pub seed: u64,
28    /// Number of simple hydraulic erosion passes run after noise generation; 0 = none.
29    pub erosion_passes: u32,
30}
31
32/// Default for a 64×64 map with 5 octaves, no erosion.
33impl Default for HeightmapOpts {
34    fn default() -> Self {
35        Self {
36            width: 64,
37            height: 64,
38            scale: 1.0 / 32.0,
39            octaves: 5,
40            lacunarity: 2.0,
41            persistence: 0.5,
42            seed: 0,
43            erosion_passes: 0,
44        }
45    }
46}

48/// Normalised heightmap grid with float cells in 0.0–1.0 after construction.
49#[derive(Debug, Clone)]
50pub struct Heightmap {
51    /// Grid width in cells.
52    pub width: u32,
53    /// Grid height in cells.
54    pub height: u32,
55    /// Flat row-major cell values; index = `y * width + x`.
56    pub cells: Vec<f32>,
57}
58
59/// Core construction, sampling, erosion, and export methods.
60impl Heightmap {
61    /// Generate a heightmap from `opts` using FBM Perlin noise, normalised and optionally eroded.
62    pub fn generate(opts: &HeightmapOpts) -> Self {
63        let gen = NoiseGenerator::new(opts.seed);
64        let map_opts = MapGenOptions {
65            scale_x: opts.scale,
66            scale_y: opts.scale,
67            octaves: opts.octaves,
68            lacunarity: opts.lacunarity,
69            persistence: opts.persistence,
70            kind: NoiseKind::Perlin,
71            fractal: FractalType::Fbm,
72            offset_x: 0.0,
73            offset_y: 0.0,
74        };
75        let raw = gen.generate_map(opts.width, opts.height, &map_opts);
76        let cells: Vec<f32> = raw.iter().map(|&v| v as f32).collect();
77        let mut hm = Self {
78            width: opts.width,
79            height: opts.height,
80            cells,
81        };
82        hm.normalize();
83        if opts.erosion_passes > 0 {
84            hm.erode(opts.erosion_passes);
85            hm.normalize();
86        }
87        hm
88    }
89
90    /// Build a heightmap from a pre-computed `f64` noise slice; normalises the result.
91    pub fn from_noise_map(width: u32, height: u32, values: &[f64]) -> Self {
92        let len = (width * height) as usize;
93        let mut cells = Vec::with_capacity(len);
94        for i in 0..len {
95            cells.push(values.get(i).copied().unwrap_or(0.0) as f32);
96        }
97        let mut hm = Self {
98            width,
99            height,
100            cells,
101        };
102        hm.normalize();
103        hm
104    }
105
106    /// Build a heightmap from a cellular automata `u8` grid: cells != `floor_value` map to 1.0, others to 0.0.
107    pub fn from_cellular(width: u32, height: u32, cells: &[u8], floor_value: u8) -> Self {
108        let len = (width * height) as usize;
109        let mut out = Vec::with_capacity(len);
110        for i in 0..len {
111            let v = cells.get(i).copied().unwrap_or(floor_value);
112            out.push(if v == floor_value { 0.0 } else { 1.0 });
113        }
114        Self {
115            width,
116            height,
117            cells: out,
118        }
119    }
120
121    /// Return the cell value at `(x, y)`, clamping out-of-bounds coordinates to the grid edge.
122    pub fn get(&self, x: u32, y: u32) -> f32 {
123        let x = x.min(self.width.saturating_sub(1));
124        let y = y.min(self.height.saturating_sub(1));
125        self.cells[(y * self.width + x) as usize]
126    }
127
128    /// Overwrite the cell at `(x, y)` with `v`; assumes coordinates are in-bounds.
129    fn set(&mut self, x: u32, y: u32, v: f32) {
130        let idx = (y * self.width + x) as usize;
131        self.cells[idx] = v;
132    }
133
134    /// Remap all cells to 0.0–1.0 by dividing by the current min/max range; no-op if range < 1e-7.
135    pub fn normalize(&mut self) {
136        let min = self.cells.iter().cloned().fold(f32::MAX, f32::min);
137        let max = self.cells.iter().cloned().fold(f32::MIN, f32::max);
138        let range = max - min;
139        if range < 1e-7 {
140            return;
141        }
142        for v in &mut self.cells {
143            *v = (*v - min) / range;
144        }
145    }
146
147    /// Apply `passes` rounds of simple hydraulic erosion: each cell deposits 10 % of its height difference into its lowest 4-connected neighbour.
148    pub fn erode(&mut self, passes: u32) {
149        for _ in 0..passes {
150            let w = self.width;
151            let h = self.height;
152            for y in 0..h {
153                for x in 0..w {
154                    let center = self.get(x, y);
155                    let dirs: [(i32, i32); 4] = [(1, 0), (-1, 0), (0, 1), (0, -1)];
156                    let mut lowest_val = center;
157                    let mut lowest_dir: Option<(u32, u32)> = None;
158                    for (dx, dy) in dirs {
159                        let nx = x as i32 + dx;
160                        let ny = y as i32 + dy;
161                        if nx >= 0 && ny >= 0 && nx < w as i32 && ny < h as i32 {
162                            let nv = self.get(nx as u32, ny as u32);
163                            if nv < lowest_val {
164                                lowest_val = nv;
165                                lowest_dir = Some((nx as u32, ny as u32));
166                            }
167                        }
168                    }
169                    if let Some((lx, ly)) = lowest_dir {
170                        let diff = (center - lowest_val) * 0.1;
171                        let new_center = center - diff;
172                        let new_low = lowest_val + diff;
173                        self.set(x, y, new_center);
174                        self.set(lx, ly, new_low);
175                    }
176                }
177            }
178        }
179    }
180
181    /// Convert the cell grid to a flat grayscale RGBA byte buffer at 4 bytes per cell.
182    pub fn to_rgba_bytes(&self) -> Vec<u8> {
183        scalar_map_to_rgba_bytes(&self.cells)
184    }
