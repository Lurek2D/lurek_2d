1//! Four-connected flood fill on a flat `u8` grid with threshold-based matching.
2//!
3//! - Return a binary mask of reachable cells from a seed coordinate.
4//! - Support both above-threshold and below-threshold fill modes.
5
6/// Flood-fill from `(sx, sy)` over `data` and return a flat mask where 1 = reached cell.
7///
8/// Cells match the fill region when `above=true` and value >= `threshold`,
9/// or when `above=false` and value <= `threshold`. Returns an all-zero mask
10/// when `data.len() != width * height` or the seed cell does not match.
11pub fn flood_fill(
12    data: &[u8],
13    width: u32,
14    height: u32,
15    sx: u32,
16    sy: u32,
17    threshold: u8,
18    above: bool,
19) -> Vec<u8> {
20    let size = (width * height) as usize;
21    if data.len() != size {
22        return vec![0; size];
23    }
24    let mut result = vec![0u8; size];
25    let start_idx = (sy * width + sx) as usize;
26    if start_idx >= size {
27        return result;
28    }
29    let matches = |v: u8| -> bool {
30        if above {
31            v >= threshold
32        } else {
33            v <= threshold
34        }
35    };
36    if !matches(data[start_idx]) {
37        return result;
38    }
39    let mut queue = std::collections::VecDeque::new();
40    queue.push_back((sx, sy));
41    result[start_idx] = 1;
42    while let Some((x, y)) = queue.pop_front() {
43        for &(dx, dy) in &[(-1i32, 0), (1, 0), (0, -1i32), (0, 1)] {
44            let nx = x as i32 + dx;
45            let ny = y as i32 + dy;
46            if nx < 0 || ny < 0 || nx >= width as i32 || ny >= height as i32 {
47                continue;
48            }
49            let nx = nx as u32;
50            let ny = ny as u32;
51            let ni = (ny * width + nx) as usize;
52            if result[ni] == 0 && matches(data[ni]) {
53                result[ni] = 1;
54                queue.push_back((nx, ny));
55            }
56        }
57    }
58    result
59}
