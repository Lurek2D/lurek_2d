1//! Cellular automata cave generator producing flat grid maps from configurable birth/survive rules.
2//!
3//! - Supports reproducible output via seeded LCG randomisation.
4//! - Treats out-of-bounds neighbours as solid, forming natural cave walls at map edges.
5
6use super::lcg::Lcg;
7
8/// Configuration for one cellular automata cave generation run.
9#[derive(Debug, Clone)]
10pub struct CellularOpts {
11    /// Initial fill probability per cell; range 0.0–1.0 (0.45 = 45 % solid).
12    pub fill: f32,
13    /// Number of birth/survive update passes applied to the initial grid.
14    pub iterations: u32,
15    /// Moore-neighbour count at or above which a dead cell becomes alive (birth rule).
16    pub birth: u32,
17    /// Moore-neighbour count at or above which a live cell survives (survive rule).
18    pub survive: u32,
19    /// Seed passed to the internal `Lcg` for reproducible initialisation.
20    pub seed: u64,
21}
22
23/// Provide cave-friendly defaults: fill=0.45, iterations=5, birth=6, survive=4.
24impl Default for CellularOpts {
25    fn default() -> Self {
26        Self {
27            fill: 0.45,
28            iterations: 5,
29            birth: 6,
30            survive: 4,
31            seed: 12345,
32        }
33    }
34}
35
36/// Run cellular automata on a `width × height` grid and return a flat `0`/`1` slice (1 = solid).
37pub fn cellular_automata(width: u32, height: u32, opts: &CellularOpts) -> Vec<u8> {
38    let size = (width * height) as usize;
39    let mut grid = vec![0u8; size];
40    let mut rng = Lcg::new(opts.seed);
41    for cell in grid.iter_mut() {
42        *cell = if rng.next_f32() < opts.fill { 1 } else { 0 };
43    }
44    let mut next = vec![0u8; size];
45    for _ in 0..opts.iterations {
46        for y in 0..height {
47            for x in 0..width {
48                let idx = (y * width + x) as usize;
49                let mut neighbors = 0u32;
50                for dy in -1i32..=1 {
51                    for dx in -1i32..=1 {
52                        if dx == 0 && dy == 0 {
53                            continue;
54                        }
55                        let nx = x as i32 + dx;
56                        let ny = y as i32 + dy;
57                        if nx < 0 || ny < 0 || nx >= width as i32 || ny >= height as i32 {
58                            neighbors += 1;
59                        } else {
60                            neighbors += grid[(ny as u32 * width + nx as u32) as usize] as u32;
61                        }
62                    }
63                }
64                next[idx] = if grid[idx] == 1 {
65                    if neighbors >= opts.survive {
66                        1
67                    } else {
68                        0
69                    }
70                } else if neighbors >= opts.birth {
71                    1
72                } else {
73                    0
74                };
75            }
76        }
77        std::mem::swap(&mut grid, &mut next);
78    }
79    grid
80}
