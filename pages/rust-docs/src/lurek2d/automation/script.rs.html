1//! Automation script container: named, time-sorted step sequences for deterministic replay.
2//!
3//! - Expands repeat markers into cloned steps at computed time offsets.
4//! - Parses TOML input with meta description and typed step fields.
5//! - Enforces a configurable step limit (default MAX_STEPS = 100,000).
6//! - Sorts steps by time after expansion for correct playback ordering.
7
8use super::{Action, Step};
9/// Maximum number of steps retained in a single automation script.
10pub(crate) const MAX_STEPS: usize = 100_000;
11#[derive(Debug, Clone)]
12/// A time-sorted sequence of `Step` values that `Simulator` plays back deterministically.
13pub struct Script {
14    /// Human-readable identifier used as the registry key in `Simulator`.
15    pub name: String,
16    /// Optional human-readable summary loaded from `[meta] description` in TOML.
17    pub description: Option<String>,
18    /// Sorted, capped list of steps ready for playback; at most `step_limit` entries.
19    pub steps: Vec<Step>,
20    /// Maximum number of steps retained; defaults to `MAX_STEPS`, adjustable via `set_step_limit`.
21    step_limit: usize,
22}
23impl Script {
24    /// Expand repeat steps: clone each step with `repeat` N times at `repeat_interval` offsets.
25    fn expand_repeats(steps: Vec<Step>) -> Vec<Step> {
26        let mut out = Vec::new();
27        for step in steps {
28            let repeat_count = step.repeat.unwrap_or(0);
29            let interval = step.repeat_interval.unwrap_or(0.0);
30            let mut base = step.clone();
31            base.repeat = None;
32            base.repeat_interval = None;
33            out.push(base);
34            for i in 1..=repeat_count {
35                let mut clone = step.clone();
36                clone.time = step.time + (i as f32 * interval);
37                clone.repeat = None;
38                clone.repeat_interval = None;
39                out.push(clone);
40            }
41        }
42        out
43    }
44    /// Create a `Script` from a name and raw steps: expands repeats, sorts by time, caps to `MAX_STEPS`.
45    pub fn new(name: impl Into<String>, steps: Vec<Step>) -> Self {
46        let mut steps = Self::expand_repeats(steps);
47        steps.sort_by(|a, b| {
48            a.time
49                .partial_cmp(&b.time)
50                .unwrap_or(std::cmp::Ordering::Equal)
51        });
52        steps.truncate(MAX_STEPS);
53        Self {
54            name: name.into(),
55            description: None,
56            steps,
57            step_limit: MAX_STEPS,
58        }
59    }
60    /// Create a `Script` identical to `new` but also sets the human-readable `description` field.
61    pub fn with_description(
62        name: impl Into<String>,
63        description: impl Into<String>,
64        steps: Vec<Step>,
65    ) -> Self {
66        let mut script = Self::new(name, steps);
67        script.description = Some(description.into());
68        script
69    }
70    /// Return the total number of steps in this script.
71    pub fn step_count(&self) -> usize {
72        self.steps.len()
73    }
74    /// Clamp and apply a new step limit, truncating the step list if it exceeds `limit`; range 1..=MAX_STEPS.
75    pub fn set_step_limit(&mut self, limit: usize) {
76        self.step_limit = limit.clamp(1, MAX_STEPS);
77        self.steps.truncate(self.step_limit);
78    }
79    /// Return the current step limit for this script.
80    pub fn get_step_limit(&self) -> usize {
81        self.step_limit
82    }
83    /// Parse a TOML string and construct a `Script`; return an error string on invalid TOML or unknown action.
84    pub fn from_toml(name: impl Into<String>, toml_str: &str) -> Result<Self, String> {
85        let doc: toml::Value =
86            toml::from_str(toml_str).map_err(|e| format!("invalid TOML: {e}"))?;
87        let description = doc
88            .get("meta")
89            .and_then(|m| m.get("description"))
90            .and_then(|d| d.as_str())
91            .map(|s| s.to_string());
92        let steps_values = doc
93            .get("steps")
94            .and_then(|s| s.as_array())
95            .cloned()
96            .unwrap_or_default();
97        let mut steps = Vec::with_capacity(steps_values.len());
98        for sv in &steps_values {
99            let action_str = sv
100                .get("action")
101                .and_then(|a| a.as_str())
102                .ok_or_else(|| "each step needs 'action'".to_string())?;
103            let action = Action::parse_action(action_str)
104                .ok_or_else(|| format!("unknown action '{action_str}'"))?;
105            let time = sv.get("time").and_then(|t| t.as_float()).unwrap_or(0.0) as f32;
106            let mut step = Step::new(time, action);
107            step.key = sv.get("key").and_then(|v| v.as_str()).map(str::to_string);
108            step.scancode = sv
109                .get("scancode")
110                .and_then(|v| v.as_str())
111                .map(str::to_string);
112            step.x = sv.get("x").and_then(|v| v.as_float());
113            step.y = sv.get("y").and_then(|v| v.as_float());
114            step.dx = sv.get("dx").and_then(|v| v.as_float());
115            step.dy = sv.get("dy").and_then(|v| v.as_float());
116            step.button = sv
117                .get("button")
118                .and_then(|v| v.as_integer())
119                .map(|n| n as u32);
120            step.text = sv.get("text").and_then(|v| v.as_str()).map(str::to_string);
121            step.is_repeat = sv
122                .get("isRepeat")
123                .and_then(|v| v.as_bool())
124                .unwrap_or(false);
125            step.clicks = sv
126                .get("clicks")
127                .and_then(|v| v.as_integer())
128                .map(|n| n as u32);
129            step.repeat = sv
130                .get("repeat")
131                .and_then(|v| v.as_integer())
132                .map(|n| n as u32);
133            step.repeat_interval = sv
134                .get("repeatInterval")
135                .and_then(|v| v.as_float())
136                .map(|v| v as f32);
137            step.macro_name = sv.get("macro").and_then(|v| v.as_str()).map(str::to_string);
138            step.when = sv.get("when").and_then(|v| v.as_str()).map(str::to_string);
139            step.assert = sv
140                .get("assert")
141                .and_then(|v| v.as_str())
142                .map(str::to_string);
143            step.baseline = sv
144                .get("baseline")
145                .and_then(|v| v.as_str())
146                .map(str::to_string);
147            step.actual = sv
148                .get("actual")
149                .and_then(|v| v.as_str())
150                .map(str::to_string);
151            step.max_diff = sv
152                .get("maxDiff")
153                .and_then(|v| v.as_integer())
154                .map(|n| n as u32);
155            steps.push(step);
156        }
157        Ok(match description {
158            Some(desc) => Self::with_description(name, desc, steps),
159            None => Self::new(name, steps),
160        })
161    }
