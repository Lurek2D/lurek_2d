1//! Axis-aligned rectangle defined by top-left corner and size (y-down convention).
2//!
3//! - Containment, intersection, union, and bounding-box construction from point sets.
4//! - Center-based and corner-based constructors for layout and collision use cases.
5
6use super::vec2::Vec2;
7
8/// Axis-aligned rectangle with position and size; used for bounds, layout regions, and UV rects.
9#[derive(Debug, Clone, Copy, PartialEq)]
10pub struct Rect {
11    /// Left edge x coordinate.
12    pub x: f32,
13    /// Top edge y coordinate (y-down convention).
14    pub y: f32,
15    /// Width in the x direction; must be non-negative for correct containment checks.
16    pub width: f32,
17    /// Height in the y direction; must be non-negative for correct containment checks.
18    pub height: f32,
19}
20
21impl Rect {
22    /// Construct a Rect from top-left position and size.
23    pub fn new(x: f32, y: f32, width: f32, height: f32) -> Self {
24        Rect {
25            x,
26            y,
27            width,
28            height,
29        }
30    }
31
32    /// Return the center point of this rectangle.
33    pub fn center(&self) -> Vec2 {
34        Vec2::new(self.x + self.width / 2.0, self.y + self.height / 2.0)
35    }
36
37    /// Return the area (width × height).
38    pub fn area(&self) -> f32 {
39        self.width * self.height
40    }
41
42    /// Return true when `(point_x, point_y)` lies inside or on the boundary of this rect.
43    pub fn contains(&self, point_x: f32, point_y: f32) -> bool {
44        point_x >= self.x
45            && point_x <= self.x + self.width
46            && point_y >= self.y
47            && point_y <= self.y + self.height
48    }
49
50    /// Return true when this rect overlaps `other` (touching edges count as overlap).
51    pub fn intersects(&self, other: &Rect) -> bool {
52        self.x < other.x + other.width
53            && self.x + self.width > other.x
54            && self.y < other.y + other.height
55            && self.y + self.height > other.y
56    }
57
58    /// Return the overlapping region of `self` and `other`; returns zero-size rect when disjoint.
59    pub fn intersect(&self, other: &Rect) -> Rect {
60        let left = self.x.max(other.x);
61        let top = self.y.max(other.y);
62        let right = (self.x + self.width).min(other.x + other.width);
63        let bottom = (self.y + self.height).min(other.y + other.height);
64        if right > left && bottom > top {
65            Rect::new(left, top, right - left, bottom - top)
66        } else {
67            Rect::new(0.0, 0.0, 0.0, 0.0)
68        }
69    }
70
71    /// Return the smallest rect that contains both `self` and `other`.
72    pub fn union(&self, other: &Rect) -> Rect {
73        let left = self.x.min(other.x);
74        let top = self.y.min(other.y);
75        let right = (self.x + self.width).max(other.x + other.width);
76        let bottom = (self.y + self.height).max(other.y + other.height);
77        Rect::new(left, top, right - left, bottom - top)
78    }
79
80    /// Construct a rect centered at `(cx, cy)` with given `w` and `h`.
81    pub fn from_center(cx: f32, cy: f32, w: f32, h: f32) -> Rect {
82        Rect::new(cx - w / 2.0, cy - h / 2.0, w, h)
83    }
84
85    /// Return the tight bounding rect around a slice of `(x, y)` points; returns zero rect for empty slice.
86    pub fn from_points(points: &[(f32, f32)]) -> Rect {
87        if points.is_empty() {
88            return Rect::new(0.0, 0.0, 0.0, 0.0);
89        }
90        let mut min_x = f32::MAX;
91        let mut min_y = f32::MAX;
92        let mut max_x = f32::MIN;
93        let mut max_y = f32::MIN;
94        for &(px, py) in points {
95            if px < min_x {
96                min_x = px;
97            }
98            if py < min_y {
99                min_y = py;
100            }
101            if px > max_x {
102                max_x = px;
103            }
104            if py > max_y {
105                max_y = py;
106            }
107        }
108        Rect::new(min_x, min_y, max_x - min_x, max_y - min_y)
109    }
110}
