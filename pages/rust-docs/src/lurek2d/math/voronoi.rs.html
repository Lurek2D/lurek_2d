1//! Voronoi diagram generation from 2D point sets via Bowyer-Watson Delaunay triangulation.
2//!
3//! - Circumcenter and circumcircle predicates for incremental insertion.
4//! - Boundary-edge extraction and super-triangle cleanup.
5//! - CCW vertex sorting and deduplication to produce closed polygonal cells.
6//! - Input deduplication to handle coincident sites gracefully.
7
8use std::collections::HashMap;
9
10/// One Voronoi region around a seed site with its polygon vertices in CCW order.
11#[derive(Debug, Clone)]
12pub struct VoronoiCell {
13    /// Seed point this cell surrounds.
14    pub site: (f32, f32),
15    /// Circumcenter vertices of adjacent Delaunay triangles, sorted CCW around the site.
16    pub vertices: Vec<(f32, f32)>,
17}
18
19/// Return the circumcenter of triangle (ax,ay)-(bx,by)-(cx,cy), or `None` when collinear.
20fn circumcenter(ax: f32, ay: f32, bx: f32, by: f32, cx: f32, cy: f32) -> Option<(f32, f32)> {
21    let d = 2.0 * ((bx - ax) * (cy - ay) - (by - ay) * (cx - ax));
22    if d.abs() < 1e-10 {
23        return None;
24    }
25    let ax2 = ax * ax + ay * ay;
26    let bx2 = bx * bx + by * by;
27    let cx2 = cx * cx + cy * cy;
28    let ux = ((bx2 - ax2) * (cy - ay) - (cx2 - ax2) * (by - ay)) / d;
29    let uy = ((bx - ax) * (cx2 - ax2) - (cx - ax) * (bx2 - ax2)) / d;
30    Some((ux, uy))
31}

33/// Return true when point `(px, py)` lies strictly inside the circumcircle of triangle (ax,ay)-(bx,by)-(cx,cy).
34#[allow(clippy::too_many_arguments)]
35fn in_circumcircle(ax: f32, ay: f32, bx: f32, by: f32, cx: f32, cy: f32, px: f32, py: f32) -> bool {
36    let d = 2.0 * ((bx - ax) * (cy - ay) - (by - ay) * (cx - ax));
37    if d.abs() < 1e-10 {
38        return false;
39    }
40    let ax2 = ax * ax + ay * ay;
41    let bx2 = bx * bx + by * by;
42    let cx2 = cx * cx + cy * cy;
43    let ux = ((bx2 - ax2) * (cy - ay) - (cx2 - ax2) * (by - ay)) / d;
44    let uy = ((bx - ax) * (cx2 - ax2) - (cx - ax) * (bx2 - ax2)) / d;
45    let r2 = (ax - ux) * (ax - ux) + (ay - uy) * (ay - uy);
46    let dp2 = (px - ux) * (px - ux) + (py - uy) * (py - uy);
47    dp2 < r2
48}

50/// Index triple representing one Delaunay triangle by vertex indices into `pts`.
51#[derive(Clone, Copy, PartialEq, Eq, Debug)]
52struct Tri(usize, usize, usize);
53
54/// Methods for querying edges and super-triangle membership of a Delaunay triangle.
55impl Tri {
56    /// Return the three directed edges of this triangle.
57    fn edges(self) -> [(usize, usize); 3] {
58        [(self.0, self.1), (self.1, self.2), (self.2, self.0)]
59    }
60    /// Return true when any vertex index belongs to the super-triangle (indices >= `n_real`).
61    fn contains_super(self, n_real: usize) -> bool {
62        self.0 >= n_real || self.1 >= n_real || self.2 >= n_real
63    }
64}
65
66/// Return the canonical edge key with the smaller index first.
67fn edge_key(a: usize, b: usize) -> (usize, usize) {
68    if a < b {
69        (a, b)
70    } else {
71        (b, a)
72    }
73}

75/// Run Bowyer-Watson incremental Delaunay triangulation on `pts`; strips super-triangle faces.
76fn bowyer_watson(pts: &[(f32, f32)], n_real: usize) -> Vec<Tri> {
77    let n = pts.len();
78    let mut tris: Vec<Tri> = vec![Tri(n - 3, n - 2, n - 1)];
79    for i in 0..n_real {
80        let (px, py) = pts[i];
81        let bad: Vec<Tri> = tris
82            .iter()
83            .filter(|&&t| {
84                let (ax, ay) = pts[t.0];
85                let (bx, by) = pts[t.1];
86                let (cx, cy) = pts[t.2];
87                in_circumcircle(ax, ay, bx, by, cx, cy, px, py)
88            })
89            .copied()
90            .collect();
91        if bad.is_empty() {
92            continue;
93        }
94        let mut edge_count: HashMap<(usize, usize), usize> = HashMap::new();
95        for &t in &bad {
96            for (a, b) in t.edges() {
97                *edge_count.entry(edge_key(a, b)).or_insert(0) += 1;
98            }
99        }
100        let boundary: Vec<(usize, usize)> = edge_count
101            .into_iter()
102            .filter(|(_, c)| *c == 1)
103            .map(|(e, _)| e)
104            .collect();
105        tris.retain(|t| !bad.contains(t));
106        for (a, b) in boundary {
107            tris.push(Tri(i, a, b));
108        }
109    }
110    tris.into_iter()
111        .filter(|t| !t.contains_super(n_real))
112        .collect()
113}

115/// Compute Voronoi cells for `points`; deduplicates input and returns one cell per unique site.
116pub fn voronoi_from_points(points: &[(f32, f32)]) -> Vec<VoronoiCell> {
117    if points.is_empty() {
118        return Vec::new();
119    }
120    let mut pts: Vec<(f32, f32)> = Vec::with_capacity(points.len());
121    'outer: for &p in points {
122        for &q in &pts {
123            let dx = p.0 - q.0;
124            let dy = p.1 - q.1;
125            if dx * dx + dy * dy < 1e-10 {
126                continue 'outer;
127            }
128        }
129        pts.push(p);
130    }
131    let n_real = pts.len();
132    if n_real < 3 {
133        return pts
134            .iter()
135            .map(|&site| VoronoiCell {
136                site,
137                vertices: Vec::new(),
138            })
139            .collect();
140    }
141    let (mut min_x, mut min_y, mut max_x, mut max_y) = pts.iter().fold(
142        (f32::MAX, f32::MAX, f32::MIN, f32::MIN),
143        |(lx, ly, hx, hy), &(x, y)| (lx.min(x), ly.min(y), hx.max(x), hy.max(y)),
144    );
145    let dx = (max_x - min_x).max(1.0);
146    let dy = (max_y - min_y).max(1.0);
147    let delta = dx.max(dy) * 10.0;
148    min_x -= delta;
149    min_y -= delta;
150    max_x += delta;
151    max_y += delta;
152    pts.push((min_x - dy, min_y - dx));
153    pts.push(((min_x + max_x) * 0.5, max_y + dx));
154    pts.push((max_x + dy, min_y - dx));
155    let triangles = bowyer_watson(&pts, n_real);
156    let mut cells: Vec<VoronoiCell> = (0..n_real)
157        .map(|i| VoronoiCell {
158            site: pts[i],
159            vertices: Vec::new(),
160        })
161        .collect();
162    for &t in &triangles {
163        let (ax, ay) = pts[t.0];
164        let (bx, by) = pts[t.1];
165        let (cx, cy) = pts[t.2];
166        if let Some(cc) = circumcenter(ax, ay, bx, by, cx, cy) {
167            for &vi in &[t.0, t.1, t.2] {
168                if vi < n_real {
169                    cells[vi].vertices.push(cc);
170                }
171            }
172        }
173    }
174    for cell in &mut cells {
175        let (sx, sy) = cell.site;
176        cell.vertices.sort_by(|&(ax, ay), &(bx, by)| {
177            let a_ang = (ay - sy).atan2(ax - sx);
178            let b_ang = (by - sy).atan2(bx - sx);
179            a_ang
180                .partial_cmp(&b_ang)
181                .unwrap_or(std::cmp::Ordering::Equal)
182        });
183        cell.vertices.dedup_by(|a, b| {
184            let ddx = a.0 - b.0;
185            let ddy = a.1 - b.1;
186            ddx * ddx + ddy * ddy < 1e-8
187        });
188    }
189    cells
190}
