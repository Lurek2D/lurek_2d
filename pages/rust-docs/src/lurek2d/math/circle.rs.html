1//! Circle primitive defined by center + radius, clamped non-negative on construction.
2//!
3//! - Point-containment, circle-circle intersection, and AABB queries.
4//! - Area and perimeter helpers using `std::f32::consts::PI`.
5
6use super::vec2::Vec2;
7
8/// Axis-aligned circle defined by center position and non-negative radius.
9#[derive(Debug, Clone, Copy, PartialEq)]
10pub struct Circle {
11    /// X coordinate of the center.
12    pub x: f32,
13    /// Y coordinate of the center.
14    pub y: f32,
15    /// Radius; clamped to >= 0 on construction.
16    pub radius: f32,
17}
18
19/// Core geometry methods for `Circle`.
20impl Circle {
21    /// Construct a Circle; radius is clamped to >= 0.
22    pub fn new(x: f32, y: f32, radius: f32) -> Self {
23        Circle {
24            x,
25            y,
26            radius: radius.max(0.0),
27        }
28    }
29
30    /// Return the center as a Vec2. This function is part of the public API.
31    pub fn center(&self) -> Vec2 {
32        Vec2::new(self.x, self.y)
33    }
34
35    /// Return π × r². This function is part of the public API.
36    pub fn area(&self) -> f32 {
37        std::f32::consts::PI * self.radius * self.radius
38    }
39
40    /// Return 2 × π × r. This function is part of the public API.
41    pub fn perimeter(&self) -> f32 {
42        2.0 * std::f32::consts::PI * self.radius
43    }
44
45    /// Return true when `(px, py)` lies inside or on the boundary of this circle.
46    pub fn contains(&self, px: f32, py: f32) -> bool {
47        let dx = px - self.x;
48        let dy = py - self.y;
49        dx * dx + dy * dy <= self.radius * self.radius
50    }
51
52    /// Return true when this circle and `other` overlap (touching counts as intersection).
53    pub fn intersects(&self, other: &Circle) -> bool {
54        let dx = other.x - self.x;
55        let dy = other.y - self.y;
56        let sum_r = self.radius + other.radius;
57        dx * dx + dy * dy < sum_r * sum_r
58    }
59
60    /// Return the axis-aligned bounding box as `(min_x, min_y, max_x, max_y)`.
61    pub fn aabb(&self) -> (f32, f32, f32, f32) {
62        (
63            self.x - self.radius,
64            self.y - self.radius,
65            self.x + self.radius,
66            self.y + self.radius,
67        )
68    }
69}
