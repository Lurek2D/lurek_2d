1//! Uniform-grid spatial hashing for broad-phase 2D collision and proximity queries.
2//!
3//! - AABB insert/remove/update with automatic cell-bucket management.
4//! - Rectangle, circle, and segment query shapes with deduplication.
5//! - Parametric slab-based segment-vs-AABB intersection test.
6//! - O(1) cell lookup per query tile; scales with world density, not total item count.
7
8use std::collections::{HashMap, HashSet};
9
10/// A spatial item registered in the hash grid, describing its AABB.
11#[derive(Debug, Clone)]
12pub struct SpatialItem {
13    /// Caller-assigned identifier matching the key in the hash.
14    pub id: String,
15    /// Left edge of the item's AABB.
16    pub x: f32,
17    /// Top edge of the item's AABB.
18    pub y: f32,
19    /// Width of the item's AABB.
20    pub w: f32,
21    /// Height of the item's AABB.
22    pub h: f32,
23}
24
25/// Uniform-grid hash mapping `(cell_x, cell_y)` integer keys to sets of item ids.
26pub struct SpatialHash {
27    /// World-space size of each grid cell; smaller values improve query precision at higher memory cost.
28    cell_size: f32,
29    /// All currently registered items indexed by their string id.
30    items: HashMap<String, SpatialItem>,
31    /// Cell buckets mapping grid coordinates to sets of overlapping item ids.
32    buckets: HashMap<(i32, i32), HashSet<String>>,
33}
34/// Core operations for inserting, removing, querying, and maintaining spatial items.
35impl SpatialHash {
36    /// Construct an empty hash with the given uniform `cell_size`.
37    pub fn new(cell_size: f32) -> Self {
38        Self {
39            cell_size,
40            items: HashMap::new(),
41            buckets: HashMap::new(),
42        }
43    }
44    /// Return the world-space cell size.
45    pub fn cell_size(&self) -> f32 {
46        self.cell_size
47    }
48    /// Return the number of items currently registered.
49    pub fn item_count(&self) -> usize {
50        self.items.len()
51    }
52    /// Map world-space coordinate `v` to a signed grid cell index.
53    #[inline]
54    fn cell(&self, v: f32) -> i32 {
55        (v / self.cell_size).floor() as i32
56    }
57    /// Return the inclusive cell-grid range `(min_cx, min_cy, max_cx, max_cy)` covering the given AABB.
58    fn cell_range(&self, x: f32, y: f32, w: f32, h: f32) -> (i32, i32, i32, i32) {
59        (
60            self.cell(x),
61            self.cell(y),
62            self.cell(x + w),
63            self.cell(y + h),
64        )
65    }
66    /// Register or replace item `id` with AABB `(x, y, w, h)`, inserting it into all overlapping cells.
67    pub fn insert(&mut self, id: String, x: f32, y: f32, w: f32, h: f32) {
68        self.remove_internal(&id);
69        let (cx0, cy0, cx1, cy1) = self.cell_range(x, y, w, h);
70        for cy in cy0..=cy1 {
71            for cx in cx0..=cx1 {
72                self.buckets.entry((cx, cy)).or_default().insert(id.clone());
73            }
74        }
75        self.items
76            .insert(id.clone(), SpatialItem { id, x, y, w, h });
77    }
78    /// Remove item `id` from all grid cells and delete it.
79    pub fn remove(&mut self, id: &str) {
80        self.remove_internal(id);
81        self.items.remove(id);
82    }
83    /// Erase item `id` from all its current cells without removing the item record.
84    fn remove_internal(&mut self, id: &str) {
85        if let Some(item) = self.items.get(id) {
86            let (cx0, cy0, cx1, cy1) = self.cell_range(item.x, item.y, item.w, item.h);
87            for cy in cy0..=cy1 {
88                for cx in cx0..=cx1 {
89                    if let Some(bucket) = self.buckets.get_mut(&(cx, cy)) {
90                        bucket.remove(id);
91                    }
92                }
93            }
94        }
95    }
96    /// Replace item `id`'s AABB; equivalent to `insert` (re-registers in new cells).
97    pub fn update(&mut self, id: String, x: f32, y: f32, w: f32, h: f32) {
98        self.insert(id, x, y, w, h);
99    }
100    /// Remove all items and clear all cell buckets.
101    pub fn clear(&mut self) {
102        self.items.clear();
103        self.buckets.clear();
104    }
105    /// Return ids of all items whose AABB overlaps the query rectangle, deduplicated.
106    pub fn query_rect(&self, x: f32, y: f32, w: f32, h: f32) -> Vec<String> {
107        let (cx0, cy0, cx1, cy1) = self.cell_range(x, y, w, h);
108        let mut seen: HashSet<&str> = HashSet::new();
109        let mut result = Vec::new();
110        for cy in cy0..=cy1 {
111            for cx in cx0..=cx1 {
112                if let Some(bucket) = self.buckets.get(&(cx, cy)) {
113                    for id in bucket {
114                        let id_ref = id.as_str();
115                        if seen.contains(id_ref) {
116                            continue;
117                        }
118                        if let Some(item) = self.items.get(id) {
119                            if item.x < x + w
120                                && item.x + item.w > x
121                                && item.y < y + h
122                                && item.y + item.h > y
123                            {
124                                seen.insert(id_ref);
125                                result.push(id.clone());
126                            }
127                        }
128                    }
129                }
130            }
131        }
132        result
133    }
134    /// Return ids of all items whose AABB overlaps the circle, verified by nearest-point distance.
135    pub fn query_circle(&self, cx: f32, cy: f32, radius: f32) -> Vec<String> {
136        let rx = cx - radius;
137        let ry = cy - radius;
138        let rw = radius * 2.0;
139        let rh = radius * 2.0;
140        let (cx0, cy0, cx1, cy1) = self.cell_range(rx, ry, rw, rh);
141        let mut seen: HashSet<&str> = HashSet::new();
142        let mut result = Vec::new();
143        let r2 = radius * radius;
144        for cyi in cy0..=cy1 {
145            for cxi in cx0..=cx1 {
146                if let Some(bucket) = self.buckets.get(&(cxi, cyi)) {
147                    for id in bucket {
148                        let id_ref = id.as_str();
149                        if seen.contains(id_ref) {
150                            continue;
151                        }
152                        if let Some(item) = self.items.get(id) {
153                            let nearest_x = cx.max(item.x).min(item.x + item.w);
154                            let nearest_y = cy.max(item.y).min(item.y + item.h);
155                            let dx = cx - nearest_x;
156                            let dy = cy - nearest_y;
157                            if dx * dx + dy * dy <= r2 {
158                                seen.insert(id_ref);
159                                result.push(id.clone());
160                            }
161                        }
162                    }
163                }
164            }
165        }
166        result
167    }
168    /// Return ids of all items whose AABB the segment (x1,y1)-(x2,y2) passes through, using a slab test.
169    pub fn query_segment(&self, x1: f32, y1: f32, x2: f32, y2: f32) -> Vec<String> {
170        let min_x = x1.min(x2);
171        let min_y = y1.min(y2);
172        let max_x = x1.max(x2);
173        let max_y = y1.max(y2);
174        let (cx0, cy0, cx1, cy1) = self.cell_range(min_x, min_y, max_x - min_x, max_y - min_y);
175        let mut seen = HashSet::new();
176        let mut result = Vec::new();
177        for cy in cy0..=cy1 {
178            for cx in cx0..=cx1 {
179                if let Some(bucket) = self.buckets.get(&(cx, cy)) {
180                    for id in bucket {
181                        if seen.contains(id) {
182                            continue;
183                        }
184                        if let Some(item) = self.items.get(id) {
185                            if Self::segment_aabb(x1, y1, x2, y2, item.x, item.y, item.w, item.h) {
186                                seen.insert(id.clone());
187                                result.push(id.clone());
188                            }
189                        }
190                    }
191                }
192            }
193        }
194        result
195    }
196    /// Return true when segment (x1,y1)-(x2,y2) intersects AABB at (ax, ay, aw, ah) via parametric slab.
197    #[allow(clippy::too_many_arguments)]
198    fn segment_aabb(
199        x1: f32,
200        y1: f32,
201        x2: f32,
202        y2: f32,
203        ax: f32,
204        ay: f32,
205        aw: f32,
206        ah: f32,
207    ) -> bool {
208        let dx = x2 - x1;
209        let dy = y2 - y1;
210        let mut tmin = 0.0f32;
211        let mut tmax = 1.0f32;
212        if dx.abs() < 1e-12 {
213            if x1 < ax || x1 > ax + aw {
214                return false;
215            }
216        } else {
217            let inv = 1.0 / dx;
218            let mut t1 = (ax - x1) * inv;
219            let mut t2 = (ax + aw - x1) * inv;
220            if t1 > t2 {
221                std::mem::swap(&mut t1, &mut t2);
222            }
223            tmin = tmin.max(t1);
224            tmax = tmax.min(t2);
225            if tmin > tmax {
226                return false;
227            }
228        }
229        if dy.abs() < 1e-12 {
230            if y1 < ay || y1 > ay + ah {
231                return false;
232            }
233        } else {
234            let inv = 1.0 / dy;
235            let mut t1 = (ay - y1) * inv;
236            let mut t2 = (ay + ah - y1) * inv;
237            if t1 > t2 {
238                std::mem::swap(&mut t1, &mut t2);
239            }
240            tmin = tmin.max(t1);
241            tmax = tmax.min(t2);
242            if tmin > tmax {
243                return false;
244            }
245        }
246        true
247    }
