1//! Arbitrary-degree Bézier curve with dynamic control-point list.
2//!
3//! - Evaluation via Bernstein basis, clamped to `[0,1]`.
4//! - Sampling helpers for full curves, sub-segments, and arc-length walks.
5//! - First-derivative computation and tangent-angle extraction.
6//! - Geometric transforms: translate, rotate, scale relative to an origin.
7//! - Control-point CRUD with minimum-count safety.
8
9use crate::math::vec2::Vec2;
10
11/// Arbitrary-degree Bézier curve backed by a dynamic control-point list.
12pub struct BezierCurve {
13    /// Ordered control points defining the curve; at least 2 are required.
14    control_points: Vec<Vec2>,
15}
16/// Core methods for construction, evaluation, sampling, transforms, and control-point access.
17impl BezierCurve {
18    /// Construct a Bézier curve from `points`; panics when fewer than 2 are supplied.
19    pub fn new(points: Vec<Vec2>) -> Self {
20        assert!(
21            points.len() >= 2,
22            "BezierCurve needs at least 2 control points"
23        );
24        Self {
25            control_points: points,
26        }
27    }
28    /// Evaluate the curve at parameter `t` (clamped to `[0,1]`) using Bernstein basis.
29    pub fn evaluate(&self, t: f32) -> Vec2 {
30        let n = self.control_points.len();
31        if n == 2 {
32            return self.control_points[0].lerp(self.control_points[1], t);
33        }
34        let tc = t.clamp(0.0, 1.0);
35        let omt = 1.0 - tc;
36        let mut point = Vec2::ZERO;
37        let mut coeff = 1.0f32;
38        let degree = (n - 1) as f32;
39        for (i, cp) in self.control_points.iter().enumerate() {
40            let i_f = i as f32;
41            let basis = coeff * omt.powf(degree - i_f) * tc.powf(i_f);
42            point.x += cp.x * basis;
43            point.y += cp.y * basis;
44            if i + 1 < n {
45                coeff = coeff * (degree - i_f) / (i_f + 1.0);
46            }
47        }
48        point
49    }
50    /// Sample the full curve at `segments+1` evenly spaced parameter values.
51    pub fn render(&self, segments: usize) -> Vec<Vec2> {
52        let segments = segments.max(1);
53        let mut result = Vec::with_capacity(segments + 1);
54        for i in 0..=segments {
55            let t = i as f32 / segments as f32;
56            result.push(self.evaluate(t));
57        }
58        result
59    }
60    /// Sample the curve between `t_start` and `t_end` at `segments+1` evenly spaced values.
61    pub fn render_segment(&self, t_start: f32, t_end: f32, segments: usize) -> Vec<Vec2> {
62        let segments = segments.max(1);
63        let mut result = Vec::with_capacity(segments + 1);
64        for i in 0..=segments {
65            let t = t_start + (t_end - t_start) * (i as f32 / segments as f32);
66            result.push(self.evaluate(t));
67        }
68        result
69    }
70    /// Return the first derivative curve as a new `BezierCurve` with degree reduced by one.
71    pub fn get_derivative(&self) -> BezierCurve {
72        let n = self.control_points.len();
73        if n < 2 {
74            return BezierCurve {
75                control_points: vec![Vec2::ZERO, Vec2::ZERO],
76            };
77        }
78        let degree = (n - 1) as f32;
79        let mut derivative_points = Vec::with_capacity(n - 1);
80        for i in 0..(n - 1) {
81            derivative_points.push(Vec2::new(
82                degree * (self.control_points[i + 1].x - self.control_points[i].x),
83                degree * (self.control_points[i + 1].y - self.control_points[i].y),
84            ));
85        }
86        if derivative_points.len() < 2 {
87            derivative_points.push(derivative_points[0]);
88        }
89        BezierCurve {
90            control_points: derivative_points,
91        }
92    }
93    /// Return the control point at `index`, or `None` when out of range.
94    pub fn get_control_point(&self, index: usize) -> Option<Vec2> {
95        self.control_points.get(index).copied()
96    }
97    /// Set control point at `index`; returns `false` when out of range.
98    pub fn set_control_point(&mut self, index: usize, point: Vec2) -> bool {
99        if index < self.control_points.len() {
100            self.control_points[index] = point;
101            true
102        } else {
103            false
104        }
105    }
106    /// Insert `point` at `index`, or append when `index` is `None` or out of range.
107    pub fn insert_control_point(&mut self, point: Vec2, index: Option<usize>) {
108        match index {
109            Some(i) if i <= self.control_points.len() => {
110                self.control_points.insert(i, point);
111            }
112            _ => self.control_points.push(point),
113        }
114    }
115    /// Remove the control point at `index`; returns `false` when fewer than 3 points remain or index is out of range.
116    pub fn remove_control_point(&mut self, index: usize) -> bool {
117        if self.control_points.len() <= 2 || index >= self.control_points.len() {
118            return false;
119        }
120        self.control_points.remove(index);
121        true
122    }
123    /// Return the number of control points.
124    pub fn get_control_point_count(&self) -> usize {
125        self.control_points.len()
126    }
127    /// Translate all control points by `(dx, dy)`.
128    pub fn translate(&mut self, dx: f32, dy: f32) {
129        for p in &mut self.control_points {
130            p.x += dx;
131            p.y += dy;
132        }
133    }
134    /// Rotate all control points by `angle` radians around origin `(ox, oy)`.
135    pub fn rotate(&mut self, angle: f32, ox: f32, oy: f32) {
136        let cos_a = angle.cos();
137        let sin_a = angle.sin();
138        for p in &mut self.control_points {
139            let dx = p.x - ox;
140            let dy = p.y - oy;
141            p.x = ox + dx * cos_a - dy * sin_a;
142            p.y = oy + dx * sin_a + dy * cos_a;
143        }
144    }
145    /// Scale all control points by factor `s` relative to origin `(ox, oy)`.
146    pub fn scale(&mut self, s: f32, ox: f32, oy: f32) {
147        for p in &mut self.control_points {
148            p.x = ox + (p.x - ox) * s;
149            p.y = oy + (p.y - oy) * s;
150        }
151    }
152    /// Return the approximate arc length via 100-sample numeric integration.
153    pub fn length(&self) -> f32 {
154        const SAMPLES: usize = 100;
155        let mut total = 0.0f32;
156        let mut prev = self.evaluate(0.0);
157        for i in 1..=SAMPLES {
158            let t = i as f32 / SAMPLES as f32;
159            let curr = self.evaluate(t);
160            let dx = curr.x - prev.x;
161            let dy = curr.y - prev.y;
162            total += (dx * dx + dy * dy).sqrt();
163            prev = curr;
164        }
165        total
166    }
167    /// Return the evaluated point as `(x, y)` at parameter `t`.
168    pub fn get_interpolated_position(&self, t: f32) -> (f32, f32) {
169        let p = self.evaluate(t);
170        (p.x, p.y)
171    }
172    /// Return the point at arc-length `distance` from t=0 via `samples`-step linear walk.
173    pub fn evaluate_at_distance(&self, distance: f32, samples: usize) -> Vec2 {
174        let samples = samples.max(8);
175        if distance <= 0.0 {
176            return self.evaluate(0.0);
177        }
178        let mut prev = self.evaluate(0.0);
179        let mut walked = 0.0f32;
180        for i in 1..=samples {
181            let t = i as f32 / samples as f32;
182            let curr = self.evaluate(t);
183            let seg_len = prev.distance(curr);
184            let next_walked = walked + seg_len;
185            if distance <= next_walked {
186                if seg_len <= f32::EPSILON {
187                    return curr;
188                }
189                let local = (distance - walked) / seg_len;
190                return prev.lerp(curr, local);
191            }
192            walked = next_walked;
193            prev = curr;
194        }
195        self.evaluate(1.0)
196    }
197    /// Return the tangent angle in radians at parameter `t` using the derivative curve.
198    pub fn get_interpolated_angle(&self, t: f32) -> f32 {
199        let deriv = self.get_derivative();
200        let tangent = deriv.evaluate(t);
201        tangent.y.atan2(tangent.x)
202    }
203}
204/// Clone the curve by duplicating its control-point list.
205impl Clone for BezierCurve {
206    /// Return a deep copy of this curve with cloned control points.
207    fn clone(&self) -> Self {
208        Self {
209            control_points: self.control_points.clone(),
210        }
211    }
