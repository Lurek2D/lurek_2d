10/// Triangulate a simple (non-self-intersecting) polygon using ear clipping; returns error on failure.
11pub fn triangulate(polygon: &[Vec2]) -> Result<Vec<[Vec2; 3]>, String> {
12    let n = polygon.len();
13    if n < 3 {
14        return Err("Polygon needs at least 3 vertices".to_string());
15    }
16    if n == 3 {
17        return Ok(vec![[polygon[0], polygon[1], polygon[2]]]);
18    }
19    let area = signed_area(polygon);
20    let ccw: Vec<Vec2> = if area < 0.0 {
21        polygon.iter().rev().copied().collect()
22    } else {
23        polygon.to_vec()
24    };
25    let mut indices: Vec<usize> = (0..n).collect();
26    let mut triangles = Vec::new();
27    while indices.len() > 3 {
28        let mut ear_found = false;
29        let len = indices.len();
30        for i in 0..len {
31            let prev = (i + len - 1) % len;
32            let next = (i + 1) % len;
33            let pi = indices[prev];
34            let ci = indices[i];
35            let ni = indices[next];
36            if is_ear(&ccw, &indices, pi, ci, ni) {
37                triangles.push([ccw[pi], ccw[ci], ccw[ni]]);
38                indices.remove(i);
39                ear_found = true;
40                break;
41            }
42        }
43        if !ear_found {
44            return Err("Failed to triangulate: polygon may be self-intersecting".to_string());
45        }
46    }
47    if indices.len() == 3 {
48        triangles.push([ccw[indices[0]], ccw[indices[1]], ccw[indices[2]]]);
49    }
50    Ok(triangles)
51}

53/// Return true when all cross products of consecutive edge pairs have the same sign.
54pub fn is_convex(polygon: &[Vec2]) -> bool {
55    let n = polygon.len();
56    if n < 3 {
57        return false;
58    }
59    let mut sign = 0i32;
60    for i in 0..n {
61        let a = polygon[i];
62        let b = polygon[(i + 1) % n];
63        let c = polygon[(i + 2) % n];
64        let cross = (b.x - a.x) * (c.y - b.y) - (b.y - a.y) * (c.x - b.x);
65        let s = if cross > 0.0 {
66            1
67        } else if cross < 0.0 {
68            -1
69        } else {
70            0
71        };
72        if s != 0 {
73            if sign != 0 && s != sign {
74                return false;
75            }
76            sign = s;
77        }
78    }
79    true
80}

82/// Return the signed area of a Vec2 polygon (positive = CCW, negative = CW).
83fn signed_area(polygon: &[Vec2]) -> f32 {
84    let n = polygon.len();
85    let mut area = 0.0;
86    for i in 0..n {
87        let j = (i + 1) % n;
88        area += polygon[i].x * polygon[j].y;
89        area -= polygon[j].x * polygon[i].y;
90    }
91    area / 2.0
92}

94/// Return true when vertex `curr` forms a valid ear (convex, no other vertex inside the triangle).
95fn is_ear(polygon: &[Vec2], indices: &[usize], prev: usize, curr: usize, next: usize) -> bool {
96    let a = polygon[prev];
97    let b = polygon[curr];
98    let c = polygon[next];
99    let cross = (b.x - a.x) * (c.y - b.y) - (b.y - a.y) * (c.x - b.x);
100    if cross <= 0.0 {
101        return false;
102    }
103    for &idx in indices {
104        if idx == prev || idx == curr || idx == next {
105            continue;
106        }
107        if point_in_triangle(polygon[idx], a, b, c) {
108            return false;
109        }
110    }
111    true
112}

114/// Return true when point `p` lies inside or on triangle (a, b, c).
115fn point_in_triangle(p: Vec2, a: Vec2, b: Vec2, c: Vec2) -> bool {
116    let d1 = cross_sign(p, a, b);
117    let d2 = cross_sign(p, b, c);
118    let d3 = cross_sign(p, c, a);
119    let has_neg = (d1 < 0.0) || (d2 < 0.0) || (d3 < 0.0);
120    let has_pos = (d1 > 0.0) || (d2 > 0.0) || (d3 > 0.0);
121    !(has_neg && has_pos)
122}

124/// Return the signed cross product magnitude used for triangle winding tests.
125fn cross_sign(p1: Vec2, p2: Vec2, p3: Vec2) -> f32 {
126    (p1.x - p3.x) * (p2.y - p3.y) - (p2.x - p3.x) * (p1.y - p3.y)
127}

129/// Clip a polygon against a half-plane defined by normal `(nx, ny)` and offset `d` (Sutherland-Hodgman step).
130pub fn polygon_clip(polygon: &[(f32, f32)], nx: f32, ny: f32, d: f32) -> Vec<(f32, f32)> {
131    if polygon.is_empty() {
132        return Vec::new();
133    }
134    let inside = |(x, y): (f32, f32)| nx * x + ny * y >= d;
135    let intersect = |(ax, ay): (f32, f32), (bx, by): (f32, f32)| {
136        let da = nx * ax + ny * ay;
137        let db = nx * bx + ny * by;
138        let denom = db - da;
139        if denom.abs() < f32::EPSILON {
140            return (ax, ay);
141        }
142        let t = (d - da) / denom;
143        (ax + t * (bx - ax), ay + t * (by - ay))
144    };
145    let n = polygon.len();
146    let mut out = Vec::with_capacity(n);
147    for i in 0..n {
148        let curr = polygon[i];
149        let next = polygon[(i + 1) % n];
150        let curr_inside = inside(curr);
151        let next_inside = inside(next);
152        if curr_inside {
153            out.push(curr);
154            if !next_inside {
155                out.push(intersect(curr, next));
156            }
157        } else if next_inside {
158            out.push(intersect(curr, next));
159        }
160    }
161    out
162}

164/// Return the intersection of `subject` and convex `clip` polygon via Sutherland-Hodgman; returns empty on no overlap.
165pub fn polygon_intersection(subject: &[(f32, f32)], clip: &[(f32, f32)]) -> Vec<(f32, f32)> {
166    if subject.is_empty() || clip.is_empty() {
167        return Vec::new();
168    }
169    let clip_ccw = ensure_ccw(clip);
170    let mut output = subject.to_vec();
171    let n = clip_ccw.len();
172    for i in 0..n {
173        if output.is_empty() {
174            break;
175        }
176        let (ax, ay) = clip_ccw[i];
177        let (bx, by) = clip_ccw[(i + 1) % n];
178        let nx = -(by - ay);
179        let ny = bx - ax;
180        let d = nx * ax + ny * ay;
181        output = polygon_clip(&output, nx, ny, d);
182    }
183    output
184}

186/// Return the convex hull of the union of `a` and `b` as an approximation.
187pub fn polygon_union(a: &[(f32, f32)], b: &[(f32, f32)]) -> Vec<(f32, f32)> {
188    let mut pts: Vec<(f32, f32)> = a.iter().chain(b.iter()).copied().collect();
189    convex_hull(&mut pts)
190}

192/// Return the approximate set difference `a − b` as a convex hull; returns `a` when `b` is empty.
193pub fn polygon_difference(a: &[(f32, f32)], b: &[(f32, f32)]) -> Vec<(f32, f32)> {
194    if a.is_empty() {
195        return Vec::new();
196    }
197    if b.is_empty() {
198        return a.to_vec();
199    }
200    let b_ccw = ensure_ccw(b);
201    let n = b_ccw.len();
202    let mut fragments: Vec<Vec<(f32, f32)>> = Vec::new();
203    for i in 0..n {
204        let (ax, ay) = b_ccw[i];
205        let (bx, by) = b_ccw[(i + 1) % n];
206        let nx = -(by - ay);
207        let ny = bx - ax;
208        let d = nx * ax + ny * ay;
209        let frag = polygon_clip(a, -nx, -ny, -d);
210        if !frag.is_empty() {
211            fragments.push(frag);
212        }
213    }
214    if fragments.is_empty() {
215        Vec::new()
216    } else {
217        let mut all_pts: Vec<(f32, f32)> = fragments.into_iter().flatten().collect();
218        convex_hull(&mut all_pts)
219    }
220}

222/// Return `polygon` in CCW winding order, reversing if necessary.
223fn ensure_ccw(polygon: &[(f32, f32)]) -> Vec<(f32, f32)> {
224    let area = signed_area_tuples(polygon);
225    if area >= 0.0 {
226        polygon.to_vec()
227    } else {
228        let mut rev = polygon.to_vec();
229        rev.reverse();
230        rev
231    }
232}

234/// Return the signed area of a tuple polygon (positive = CCW).
235fn signed_area_tuples(polygon: &[(f32, f32)]) -> f32 {
236    let n = polygon.len();
237    let mut area = 0.0f32;
238    for i in 0..n {
239        let (ax, ay) = polygon[i];
240        let (bx, by) = polygon[(i + 1) % n];
241        area += ax * by - bx * ay;
242    }
243    area / 2.0
244}

246/// Compute the convex hull of `pts` using the Andrew monotone chain algorithm; deduplicates input.
247fn convex_hull(pts: &mut Vec<(f32, f32)>) -> Vec<(f32, f32)> {
248    if pts.len() < 3 {
249        return pts.clone();
250    }
251    pts.sort_by(|a, b| {
252        a.0.partial_cmp(&b.0)
253            .unwrap_or(std::cmp::Ordering::Equal)
254            .then(a.1.partial_cmp(&b.1).unwrap_or(std::cmp::Ordering::Equal))
255    });
256    pts.dedup_by(|a, b| (a.0 - b.0).abs() < f32::EPSILON && (a.1 - b.1).abs() < f32::EPSILON);
257    if pts.len() < 3 {
258        return pts.clone();
259    }
260    let cross = |(ox, oy): (f32, f32), (ax, ay): (f32, f32), (bx, by): (f32, f32)| {
261        (ax - ox) * (by - oy) - (ay - oy) * (bx - ox)
262    };
263    let mut lower: Vec<(f32, f32)> = Vec::new();
264    for &p in pts.iter() {
265        while lower.len() >= 2 && cross(lower[lower.len() - 2], lower[lower.len() - 1], p) <= 0.0 {
266            lower.pop();
267        }
268        lower.push(p);
269    }
270    let mut upper: Vec<(f32, f32)> = Vec::new();
271    for &p in pts.iter().rev() {
272        while upper.len() >= 2 && cross(upper[upper.len() - 2], upper[upper.len() - 1], p) <= 0.0 {
273            upper.pop();
274        }
275        upper.push(p);
276    }
277    lower.pop();
278    upper.pop();
279    lower.extend(upper);
280    lower
