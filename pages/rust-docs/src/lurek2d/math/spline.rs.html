3//! - Hermite cubic segment defined by endpoints and tangents.

6/// Multi-point Catmull-Rom spline with dynamic control-point list.
7pub struct CatmullRomSpline {
8    /// Ordered (x, y) control points; at least 2 are required for sampling.
9    control_points: Vec<(f32, f32)>,
10}
11/// Methods for building, sampling, and modifying the Catmull-Rom spline.
12impl CatmullRomSpline {
13    /// Construct a spline from a Vec of `(x, y)` control points.
14    pub fn new(points: Vec<(f32, f32)>) -> Self {
15        Self {
16            control_points: points,
17        }
18    }
19    /// Sample the full spline at normalized parameter `t` in `[0,1]`, mapping to the appropriate segment.
20    pub fn sample(&self, t: f32) -> (f32, f32) {
21        let n = self.control_points.len();
22        if n < 2 {
23            return self.control_points.first().copied().unwrap_or((0.0, 0.0));
24        }
25        let segs = (n - 1).max(1) as f32;
26        let scaled = (t * segs).clamp(0.0, segs);
27        let seg = (scaled.floor() as usize).min(n - 2);
28        let local_t = scaled - seg as f32;
29        self.sample_segment(seg, local_t)
30    }
31    /// Sample segment `seg` at local parameter `t` in `[0,1]` using 4-point Catmull-Rom weights.
32    pub fn sample_segment(&self, seg: usize, t: f32) -> (f32, f32) {
33        let n = self.control_points.len();
34        if n == 0 {
35            return (0.0, 0.0);
36        }
37        let p = |i: i64| -> (f32, f32) {
38            let idx = i.clamp(0, (n as i64) - 1) as usize;
39            self.control_points[idx]
40        };
41        let seg_i = seg.min(n.saturating_sub(2)) as i64;
42        let (p0, p1, p2, p3) = (p(seg_i - 1), p(seg_i), p(seg_i + 1), p(seg_i + 2));
43        let t2 = t * t;
44        let t3 = t2 * t;
45        let h0 = -0.5 * t3 + t2 - 0.5 * t;
46        let h1 = 1.5 * t3 - 2.5 * t2 + 1.0;
47        let h2 = -1.5 * t3 + 2.0 * t2 + 0.5 * t;
48        let h3 = 0.5 * t3 - 0.5 * t2;
49        (
50            h0 * p0.0 + h1 * p1.0 + h2 * p2.0 + h3 * p3.0,
51            h0 * p0.1 + h1 * p1.1 + h2 * p2.1 + h3 * p3.1,
52        )
53    }
54    /// Return the number of control points.
55    pub fn len(&self) -> usize {
56        self.control_points.len()
57    }
58    /// Return true when the spline has no control points.
59    pub fn is_empty(&self) -> bool {
60        self.control_points.is_empty()
61    }
62    /// Append a control point to the end of the spline.
63    pub fn add_point(&mut self, point: (f32, f32)) {
64        self.control_points.push(point);
65    }
66    /// Remove and return the control point at `index`, or `None` when out of range.
67    pub fn remove_point(&mut self, index: usize) -> Option<(f32, f32)> {
68        if index < self.control_points.len() {
69            Some(self.control_points.remove(index))
70        } else {
71            None
72        }
73    }
74}

76/// Single Hermite cubic spline segment defined by two endpoints and their tangents.
77pub struct HermiteSpline {
78    /// Start point.
79    p0: (f32, f32),
80    /// End point.
81    p1: (f32, f32),
82    /// Start tangent.
83    m0: (f32, f32),
84    /// End tangent.
85    m1: (f32, f32),
86}
87/// Methods for constructing and sampling the Hermite segment.
88impl HermiteSpline {
89    /// Construct a Hermite segment from endpoints `p0`, `p1` and tangents `m0`, `m1`.
90    pub fn new(p0: (f32, f32), p1: (f32, f32), m0: (f32, f32), m1: (f32, f32)) -> Self {
91        Self { p0, p1, m0, m1 }
92    }
93    /// Sample the segment at `t` in `[0,1]` using Hermite basis polynomials.
94    pub fn sample(&self, t: f32) -> (f32, f32) {
95        let t2 = t * t;
96        let t3 = t2 * t;
97        let h00 = 2.0 * t3 - 3.0 * t2 + 1.0;
98        let h10 = t3 - 2.0 * t2 + t;
99        let h01 = -2.0 * t3 + 3.0 * t2;
100        let h11 = t3 - t2;
101        (
102            h00 * self.p0.0 + h10 * self.m0.0 + h01 * self.p1.0 + h11 * self.m1.0,
103            h00 * self.p0.1 + h10 * self.m0.1 + h01 * self.p1.1 + h11 * self.m1.1,
104        )
105    }
