1//! Player and faction ownership of shared visibility and discovery state.
2//!
3//! - `OwnerMap` tracks which faction owns each region for fog-of-war sharing.
4//! - Allied factions share visibility when `share_vision` is enabled per-alliance.
5//! - `OwnerMap::visible_to(faction_id, region_id)` is the hot-path query.
6//! - Ownership changes trigger re-evaluation of all visibility states for affected factions.
7
8use std::collections::HashMap;
9
10/// Manages which players share visibility (alliances/teams).
11#[derive(Debug, Clone)]
12pub struct PlayerOwnership {
13    /// Total number of players.
14    player_count: u32,
15    /// Group assignments: player_id → group_id.
16    groups: HashMap<u32, u32>,
17    /// Next available group ID.
18    next_group: u32,
19}
20
21impl PlayerOwnership {
22    /// Create a new `PlayerOwnership` tracking the given number of players, all ungrouped.
23    pub fn new(player_count: u32) -> Self {
24        Self {
25            player_count,
26            groups: HashMap::new(),
27            next_group: 0,
28        }
29    }
30
31    /// Assign multiple players to a shared visibility group.
32    pub fn set_group(&mut self, players: &[u32]) -> u32 {
33        let gid = self.next_group;
34        self.next_group += 1;
35        for &p in players {
36            if p < self.player_count {
37                self.groups.insert(p, gid);
38            }
39        }
40        gid
41    }
42
43    /// Remove a player from their group (becomes independent).
44    pub fn remove_from_group(&mut self, player_id: u32) {
45        self.groups.remove(&player_id);
46    }
47
48    /// Get all players that share visibility with a given player.
49    pub fn allies_of(&self, player_id: u32) -> Vec<u32> {
50        if let Some(&gid) = self.groups.get(&player_id) {
51            self.groups
52                .iter()
53                .filter(|(_, &g)| g == gid)
54                .map(|(&p, _)| p)
55                .collect()
56        } else {
57            vec![player_id]
58        }
59    }
60
61    /// Check if two players share visibility.
62    pub fn shares_visibility(&self, a: u32, b: u32) -> bool {
63        if a == b {
64            return true;
65        }
66        match (self.groups.get(&a), self.groups.get(&b)) {
67            (Some(ga), Some(gb)) => ga == gb,
68            _ => false,
69        }
70    }
71
72    /// Return the total number of players managed by this ownership tracker.
73    pub fn player_count(&self) -> u32 {
74        self.player_count
75    }
76}
