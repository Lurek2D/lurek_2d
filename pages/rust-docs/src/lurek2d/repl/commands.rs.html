1//! Declares `ReplCommand` enum for the five built-in colon commands: `:help`, `:quit`, `:clear`, `:reset`, and `:load <path>`.
2//!
3//! - `display_text` returns a short human-readable confirmation string for each command variant.
4//! - Command data only; dispatch logic and Lua eval live in `session.rs`.
5
6#[derive(Debug, Clone, PartialEq, Eq)]
7/// Special colon command recognised by `ReplSession`.
8pub enum ReplCommand {
9    /// Display command help.
10    Help,
11    /// Request REPL shutdown.
12    Quit,
13    /// Clear the current REPL display or history.
14    Clear,
15    /// Request a fresh Lua VM from the embedding CLI.
16    Reset,
17    /// Load and execute a Lua source file.
18    Load { path: String },
19}
20
21impl ReplCommand {
22    /// Return a short human-readable command result.
23    pub fn display_text(&self) -> String {
24        match self {
25            Self::Help => ":help, :quit, :clear, :reset, :load <file>".to_string(),
26            Self::Quit => "(quit)".to_string(),
27            Self::Clear => "(cleared)".to_string(),
28            Self::Reset => "(reset)".to_string(),
29            Self::Load { path } => format!("(ok) loaded {}", path),
30        }
31    }
32}
