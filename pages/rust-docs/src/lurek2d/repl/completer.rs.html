129/// Return sorted completions matching the supplied prefix.
130pub fn complete_prefix(prefix: &str, lua: Option<&Lua>) -> Vec<String> {
131    let mut completions: Vec<String> = STATIC_COMPLETIONS
132        .iter()
133        .filter(|candidate| candidate.starts_with(prefix))
134        .map(|candidate| (*candidate).to_string())
135        .collect();
136    if let Some(lua) = lua {
137        collect_lua_completions(lua, prefix, &mut completions);
138    }
139    completions.sort();
140    completions.dedup();
141    completions
142}
143
144/// Walk a Lua table at `path` and collect string keys that match `prefix`.
145fn collect_lua_completions(lua: &Lua, prefix: &str, completions: &mut Vec<String>) {
146    let Some((table_path, member_prefix)) = prefix.rsplit_once('.') else {
147        return;
148    };
149    let Ok(table) = resolve_table(lua, table_path) else {
150        return;
151    };
152    for pair in table.pairs::<mlua::Value, mlua::Value>() {
153        let Ok((key, _)) = pair else {
154            continue;
155        };
156        let key_text = match key {
157            mlua::Value::String(text) => text.to_str().unwrap_or_default().to_string(),
158            mlua::Value::Integer(number) => number.to_string(),
159            _ => continue,
160        };
161        if key_text.starts_with(member_prefix) {
162            completions.push(format!("{}.{}", table_path, key_text));
163        }
164    }
165}

167/// Navigate a dot-separated path through nested Lua globals and return the final table.
168fn resolve_table<'lua>(lua: &'lua Lua, path: &str) -> LuaResult<LuaTable<'lua>> {
169    let mut segments = path.split('.');
170    let Some(first) = segments.next() else {
171        return Err(mlua::Error::RuntimeError(
172            "empty completion path".to_string(),
173        ));
174    };
175    let mut table: LuaTable = lua.globals().get(first)?;
176    for segment in segments {
177        table = table.get(segment)?;
178    }
179    Ok(table)
180}
