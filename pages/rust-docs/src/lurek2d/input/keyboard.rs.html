1//! Per-frame keyboard state machine tracking logical keys, physical scan-codes, and frame deltas.
2//!
3//! - Modifier bitmask flags (Shift, Ctrl, Alt, Meta) updated each event.
4//! - OS key-repeat and text-input (IME) buffer toggling.
5//! - Bidirectional mapping between logical key names and physical scan-code names.
6//! - Winit-to-Lurek translation for both logical `Key` and physical `KeyCode` enums.
7//! - Frame lifecycle: `begin_frame` clears deltas, events accumulate, queries read snapshot.
8//! - Scan-code layer allows layout-independent bindings for WASD-style controls.
9//! - Text-input buffer collects composed characters for chat and text fields.
10
11use std::collections::HashSet;
12
13/// Bitmask flag for the Shift modifier.
14pub const MOD_SHIFT: u8 = 0b0001;
15/// Bitmask flag for the Ctrl modifier.
16pub const MOD_CTRL: u8 = 0b0010;
17/// Bitmask flag for the Alt modifier.
18pub const MOD_ALT: u8 = 0b0100;
19/// Bitmask flag for the Meta/Super modifier.
20pub const MOD_META: u8 = 0b1000;
21
22/// Per-frame keyboard state: held keys, per-frame deltas, modifier flags, and text input buffer.
23pub struct KeyboardState {
24    /// Logical key names currently held down.
25    keys_down: HashSet<String>,
26    /// Logical key names that transitioned to pressed this frame.
27    keys_pressed: Vec<String>,
28    /// Logical key names that transitioned to released this frame.
29    keys_released: Vec<String>,
30    /// Physical scan-code names currently held down.
31    scancodes_down: HashSet<String>,
32    /// Physical scan-code names that transitioned to pressed this frame.
33    scancodes_pressed: Vec<String>,
34    /// Physical scan-code names that transitioned to released this frame.
35    scancodes_released: Vec<String>,
36    /// Packed modifier bitmask using `MOD_*` constants.
37    modifiers: u8,
38    /// When true, OS key-repeat events are forwarded as additional press events.
39    key_repeat_enabled: bool,
40    /// When true, the text-input buffer is filled by OS IME/character events.
41    text_input_enabled: bool,
42    /// Characters delivered by the OS text-input system this frame.
43    text_input_buffer: Vec<String>,
44}
45
46/// Provide a zeroed default keyboard state.
47impl Default for KeyboardState {
48    fn default() -> Self {
49        Self::new()
50    }
51}
52
53/// Core keyboard query and mutation methods.
54impl KeyboardState {
55    /// Create a keyboard state with all keys up and all buffers empty.
56    pub fn new() -> Self {
57        KeyboardState {
58            keys_down: HashSet::new(),
59            keys_pressed: Vec::new(),
60            keys_released: Vec::new(),
61            scancodes_down: HashSet::new(),
62            scancodes_pressed: Vec::new(),
63            scancodes_released: Vec::new(),
64            key_repeat_enabled: false,
65            text_input_enabled: false,
66            text_input_buffer: Vec::new(),
67            modifiers: 0,
68        }
69    }
70
71    /// Clear per-frame delta lists and the text buffer; call at the start of each game frame.
72    pub fn begin_frame(&mut self) {
73        self.keys_pressed.clear();
74        self.keys_released.clear();
75        self.scancodes_pressed.clear();
76        self.scancodes_released.clear();
77        self.text_input_buffer.clear();
78    }
79
80    /// Record a scan-code press and add it to the pressed delta list when not already down.
81    pub(crate) fn press_scancode(&mut self, scancode: String) {
82        if self.scancodes_down.insert(scancode.clone()) {
83            self.scancodes_pressed.push(scancode);
84        }
85    }
86
87    /// Record a scan-code release and add it to the released delta list when it was down.
88    pub(crate) fn release_scancode(&mut self, scancode: String) {
89        if self.scancodes_down.remove(&scancode) {
90            self.scancodes_released.push(scancode);
91        }
92    }
93
94    /// Return true when `scancode` is currently held down.
95    pub fn is_scancode_down(&self, scancode: &str) -> bool {
96        self.scancodes_down.contains(scancode)
97    }
98
99    /// Enable or disable OS key-repeat forwarding.
100    pub(crate) fn set_key_repeat(&mut self, enabled: bool) {
101        self.key_repeat_enabled = enabled;
102    }
103
104    /// Return true when key-repeat is enabled.
105    pub fn has_key_repeat(&self) -> bool {
106        self.key_repeat_enabled
107    }
108
109    /// Enable or disable the text-input character buffer.
110    pub(crate) fn set_text_input(&mut self, enabled: bool) {
111        self.text_input_enabled = enabled;
112    }
113
114    /// Return true when text-input mode is active.
115    pub fn has_text_input(&self) -> bool {
116        self.text_input_enabled
117    }
118
119    /// Append a character string to the text-input buffer for this frame.
120    pub(crate) fn push_text_input(&mut self, text: String) {
121        self.text_input_buffer.push(text);
122    }
123
124    /// Return all text-input characters delivered this frame.
125    pub fn get_text_input(&self) -> &[String] {
126        &self.text_input_buffer
127    }
128
129    /// Record a logical key press; adds to the pressed delta list only on first down.
130    pub fn set_key_down(&mut self, key: &str) {
131        if self.keys_down.insert(key.to_string()) {
132            self.keys_pressed.push(key.to_string());
133        }
134    }
135
136    /// Record a logical key release; adds to the released delta list only when it was down.
137    pub fn set_key_up(&mut self, key: &str) {
138        if self.keys_down.remove(key) {
139            self.keys_released.push(key.to_string());
140        }
141    }
142
143    /// Return true when `key` is currently held down.
144    pub fn is_down(&self, key: &str) -> bool {
145        self.keys_down.contains(key)
146    }
147
148    /// Return true when any key in `keys` is currently held down.
149    pub fn is_any_down(&self, keys: &[String]) -> bool {
150        keys.iter().any(|k| self.keys_down.contains(k.as_str()))
151    }
152
153    /// Return all logical keys that transitioned to pressed this frame.
154    pub fn get_pressed(&self) -> &[String] {
155        &self.keys_pressed
156    }
157
158    /// Return all logical keys that transitioned to released this frame.
159    pub fn get_released(&self) -> &[String] {
160        &self.keys_released
161    }
162
163    /// Clear all held-key sets and delta lists; does not reset modifier flags.
164    pub fn clear(&mut self) {
165        self.keys_down.clear();
166        self.keys_pressed.clear();
167        self.keys_released.clear();
168        self.scancodes_down.clear();
169        self.scancodes_pressed.clear();
170        self.scancodes_released.clear();
171        self.text_input_buffer.clear();
172    }
173
174    /// Return true when `modifier` name (`"shift"`, `"ctrl"`, `"alt"`, `"meta"`) flag is set.
175    pub fn is_modifier_active(&self, modifier: &str) -> bool {
176        let mask = match modifier {
177            "shift" => MOD_SHIFT,
178            "ctrl" => MOD_CTRL,
179            "alt" => MOD_ALT,
180            "meta" | "super" => MOD_META,
181            _ => return false,
182        };
183        self.modifiers & mask != 0
184    }
185
186    /// Update the packed modifier bitmask from four individual boolean flags.
187    pub fn set_modifiers(&mut self, shift: bool, ctrl: bool, alt: bool, meta: bool) {
188        self.modifiers = 0;
189        if shift {
190            self.modifiers |= MOD_SHIFT;
191        }
192        if ctrl {
193            self.modifiers |= MOD_CTRL;
194        }
195        if alt {
196            self.modifiers |= MOD_ALT;
197        }
198        if meta {
199            self.modifiers |= MOD_META;
200        }
201    }
202}

204/// Map a logical key name to its canonical scan-code name, or `None` for unmapped keys.
205pub(crate) fn get_scancode_from_key(key: &str) -> Option<String> {
206    let normalized = key.to_ascii_lowercase();
207    if normalized.len() == 1
208        && normalized
209            .bytes()
210            .all(|byte| byte.is_ascii_lowercase() || byte.is_ascii_digit())
211    {
212        return Some(normalized);
213    }
214    match normalized.as_str() {
215        "space" | "escape" | "return" | "tab" | "backspace" | "delete" | "insert" | "home"
216        | "end" | "pageup" | "pagedown" | "left" | "right" | "up" | "down" | "capslock"
217        | "numlock" => Some(normalized),
218        "shift" => Some("lshift".to_string()),
219        "ctrl" => Some("lctrl".to_string()),
220        "alt" => Some("lalt".to_string()),
221        "f1" | "f2" | "f3" | "f4" | "f5" | "f6" | "f7" | "f8" | "f9" | "f10" | "f11" | "f12" => {
222            Some(normalized)
223        }
224        _ => None,
225    }
226}

228/// Map a scan-code name to its canonical logical key name, or `None` for unmapped codes.
229pub(crate) fn get_key_from_scancode(scancode: &str) -> Option<String> {
230    let normalized = scancode.to_ascii_lowercase();
231    if normalized.len() == 1
232        && normalized
233            .bytes()
234            .all(|byte| byte.is_ascii_lowercase() || byte.is_ascii_digit())
235    {
236        return Some(normalized);
237    }
238    match normalized.as_str() {
239        "space" | "escape" | "return" | "tab" | "backspace" | "delete" | "insert" | "home"
240        | "end" | "pageup" | "pagedown" | "left" | "right" | "up" | "down" | "capslock"
241        | "numlock" | "scrolllock" | "f1" | "f2" | "f3" | "f4" | "f5" | "f6" | "f7" | "f8"
242        | "f9" | "f10" | "f11" | "f12" | "kp+" | "kp-" | "kp*" | "kp/" | "kp0" | "kp1" | "kp2"
243        | "kp3" | "kp4" | "kp5" | "kp6" | "kp7" | "kp8" | "kp9" => Some(normalized),
244        "lshift" | "rshift" => Some("shift".to_string()),
245        "lctrl" | "rctrl" => Some("ctrl".to_string()),
246        "lalt" | "ralt" => Some("alt".to_string()),
247        _ => None,
248    }
249}
