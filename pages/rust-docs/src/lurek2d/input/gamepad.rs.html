1//! Per-slot gamepad state tracking: buttons, axes, connection lifecycle, and per-frame delta sets.
2//!
3//! - Vibration request queuing for delivery to the OS force-feedback driver.
4//! - SDL2-style GUID-based mapping store with file and string parsing.
5//! - Gilrs button/axis to SDL2 string conversion helpers.
6//! - Virtual D-pad synthesis from analog stick values with configurable deadzone.
7//! - Hat (D-pad) direction queries returning 8-way compass strings.

15/// Pending vibration command for one gamepad, queued for delivery to the OS driver.
16#[derive(Debug, Clone, Copy, PartialEq)]
17pub struct GamepadVibrationRequest {
18    /// Gamepad slot index, matches `GamepadState::id`.
19    pub id: usize,
20    /// Low-frequency (rumble) motor intensity in [0.0, 1.0].
21    pub low_freq: f32,
22    /// High-frequency (buzzer) motor intensity in [0.0, 1.0].
23    pub high_freq: f32,
24    /// Vibration duration in milliseconds.
25    pub duration_ms: u32,
26}

28/// Per-frame state for one physical gamepad slot, including buttons, axes, and connection flags.
29pub struct GamepadState {
30    /// Slot index assigned by gilrs.
31    pub id: u32,
32    /// Human-readable controller name from the OS or gilrs database.
33    pub name: String,
34    /// True when the physical device is connected.
35    pub connected: bool,
36    /// True when the OS driver reports force-feedback capability.
37    pub vibration_supported: bool,
38    /// SDL2-style GUID string used to look up custom mappings.
39    guid: String,
40    /// Current held state for each button code.
41    buttons: HashMap<u32, bool>,
42    /// Button codes that transitioned to pressed this frame.
43    buttons_pressed: HashSet<u32>,
44    /// Button codes that transitioned to released this frame.
45    buttons_released: HashSet<u32>,
46    /// Current axis values keyed by axis code.
47    axes: HashMap<u32, f32>,
48    /// True only during the frame when the device first connected.
49    connected_this_frame: bool,
50    /// True only during the frame when the device disconnected.
51    disconnected_this_frame: bool,
52}
53
54impl GamepadState {
55    /// Create a disconnected slot for `id`; all buttons/axes start at default.
56    pub fn new(id: u32) -> Self {
57        log_msg!(debug, GD01, "id={}", id);
58        GamepadState {
59            id,
60            name: String::from("Unknown Controller"),
61            connected: false,
62            vibration_supported: false,
63            guid: String::new(),
64            buttons: HashMap::new(),
65            buttons_pressed: HashSet::new(),
66            buttons_released: HashSet::new(),
67            axes: HashMap::new(),
68            connected_this_frame: false,
69            disconnected_this_frame: false,
70        }
71    }
72
73    /// Clear per-frame delta sets; call once at the start of each game frame.
74    pub fn begin_frame(&mut self) {
75        self.buttons_pressed.clear();
76        self.buttons_released.clear();
77        self.connected_this_frame = false;
78        self.disconnected_this_frame = false;
79    }
80
81    /// Record a button state change and update pressed/released delta sets.
82    pub fn update_button(&mut self, button: u32, pressed: bool) {
83        log_msg!(trace, GD02, "button={} pressed={}", button, pressed);
84        let was_pressed = self.is_button_pressed(button);
85        if !was_pressed && pressed {
86            self.buttons_pressed.insert(button);
87        }
88        if was_pressed && !pressed {
89            self.buttons_released.insert(button);
90        }
91        self.buttons.insert(button, pressed);
92    }
93
94    /// Return true when `button` transitioned to pressed this frame.
95    pub fn was_button_pressed(&self, button: u32) -> bool {
96        self.buttons_pressed.contains(&button)
97    }
98
99    /// Return true when `button` transitioned to released this frame.
100    pub fn was_button_released(&self, button: u32) -> bool {
101        self.buttons_released.contains(&button)
102    }
103
104    /// Record a new axis value; replaces the previous value for `axis`.
105    pub fn update_axis(&mut self, axis: u32, value: f32) {
106        log_msg!(trace, GD03, "axis={} value={:.3}", axis, value);
107        self.axes.insert(axis, value);
108    }
109
110    /// Return true when `button` is currently held down.
111    pub fn is_button_pressed(&self, button: u32) -> bool {
112        *self.buttons.get(&button).unwrap_or(&false)
113    }
114
115    /// Return the current value for `axis`, or 0.0 when the axis has never been seen.
116    pub fn get_axis_value(&self, axis: u32) -> f32 {
117        *self.axes.get(&axis).unwrap_or(&0.0)
118    }
119
120    /// Return the OS-reported controller name.
121    pub fn get_name(&self) -> &str {
122        &self.name
123    }
124
125    /// Return true when the device is currently connected.
126    pub fn is_connected(&self) -> bool {
127        self.connected
128    }
129
130    /// Update connection state and set the per-frame connection-change flags.
131    pub fn set_connected(&mut self, connected: bool) {
132        if connected && !self.connected {
133            self.connected_this_frame = true;
134        }
135        if !connected && self.connected {
136            self.disconnected_this_frame = true;
137        }
138        self.connected = connected;
139    }
140
141    /// Return true only during the frame the device first connected.
142    pub fn was_connected_this_frame(&self) -> bool {
143        self.connected_this_frame
144    }
145
146    /// Return true only during the frame the device disconnected.
147    pub fn was_disconnected_this_frame(&self) -> bool {
148        self.disconnected_this_frame
149    }
150
151    /// Set whether the OS driver supports force feedback for this device.
152    pub fn set_vibration_supported(&mut self, supported: bool) {
153        self.vibration_supported = supported;
154    }
155
156    /// Return true when force feedback is supported.
157    pub fn is_vibration_supported(&self) -> bool {
158        self.vibration_supported
159    }
160
161    /// Return the number of distinct button codes seen on this device.
162    pub fn get_button_count(&self) -> usize {
163        self.buttons.len()
164    }
165
166    /// Return the number of distinct axis codes seen on this device.
167    pub fn get_axis_count(&self) -> usize {
168        self.axes.len()
169    }
170
171    /// Set the SDL2-style GUID string; crate-internal, called from the runtime event loop.
172    pub(crate) fn set_guid(&mut self, guid: impl Into<String>) {
173        self.guid = guid.into();
174    }
175
176    /// Return the SDL2-style GUID string for mapping lookup.
177    pub fn get_guid(&self) -> &str {
178        &self.guid
179    }
180
181    /// Return a D-pad direction string for `hat`; reads buttons 10–13 for hat 0.
182    pub fn get_hat(&self, hat: u32) -> &'static str {
183        if hat != 0 {
184            return "c";
185        }
186        let up = self.is_button_pressed(10);
187        let down = self.is_button_pressed(11);
188        let left = self.is_button_pressed(12);
189        let right = self.is_button_pressed(13);
190        match (up, down, left, right) {
191            (true, false, false, false) => "u",
192            (true, false, false, true) => "ru",
193            (false, false, false, true) => "r",
194            (false, true, false, true) => "rd",
195            (false, true, false, false) => "d",
196            (false, true, true, false) => "ld",
197            (false, false, true, false) => "l",
198            (true, false, true, false) => "lu",
199            _ => "c",
200        }
201    }
202}

204/// Map a gilrs `Button` variant to its SDL2-style string name used in mappings.
205pub fn gilrs_button_to_string(button: gilrs::Button) -> &'static str {
206    match button {
207        gilrs::Button::South => "a",
208        gilrs::Button::East => "b",
209        gilrs::Button::West => "x",
210        gilrs::Button::North => "y",
211        gilrs::Button::LeftTrigger => "leftshoulder",
212        gilrs::Button::RightTrigger => "rightshoulder",
213        gilrs::Button::LeftTrigger2 => "leftstick",
214        gilrs::Button::RightTrigger2 => "rightstick",
215        gilrs::Button::Select => "back",
216        gilrs::Button::Start => "start",
217        gilrs::Button::Mode => "guide",
218        gilrs::Button::DPadUp => "dpup",
219        gilrs::Button::DPadDown => "dpdown",
220        gilrs::Button::DPadLeft => "dpleft",
221        gilrs::Button::DPadRight => "dpright",
222        _ => "unknown",
223    }
224}

239/// Convert a 2D stick position to four directional booleans and an 8-way direction string.
240///
241/// Returns `(up, down, left, right, direction)` where `direction` is one of
242/// `"u"`, `"d"`, `"l"`, `"r"`, `"lu"`, `"ru"`, `"ld"`, `"rd"`, or `"c"` (centered).
243pub fn virtual_dpad(x: f32, y: f32, deadzone: f32) -> (bool, bool, bool, bool, &'static str) {
244    let dz = deadzone.clamp(0.0, 1.0);
245    let left = x <= -dz;
246    let right = x >= dz;
247    let up = y <= -dz;
248    let down = y >= dz;
249    let direction = match (up, down, left, right) {
250        (true, false, false, false) => "u",
251        (true, false, true, false) => "lu",
252        (true, false, false, true) => "ru",
253        (false, true, false, false) => "d",
254        (false, true, true, false) => "ld",
255        (false, true, false, true) => "rd",
256        (false, false, true, false) => "l",
257        (false, false, false, true) => "r",
258        _ => "c",
259    };
260    (up, down, left, right, direction)
261}

263/// In-memory store of SDL2-style gamepad mappings keyed by device GUID.
264pub struct GamepadMappings {
265    /// GUID → raw SDL2 mapping string.
266    map: HashMap<String, String>,
267}
268
269/// Provide a default empty mapping table.
270impl Default for GamepadMappings {
271    fn default() -> Self {
272        Self::new()
273    }
274}
275
276impl GamepadMappings {
277    /// Create an empty mapping store.
278    pub fn new() -> Self {
279        Self {
280            map: HashMap::new(),
281        }
282    }
283
284    /// Insert or overwrite the mapping string for `guid`.
285    pub fn set_mapping(&mut self, guid: &str, mapping: &str) {
286        self.map.insert(guid.to_string(), mapping.to_string());
287    }
288
289    /// Parse SDL2 gamecontrollerdb lines from `source`; return the number of entries loaded.
290    pub fn load_from_string(&mut self, source: &str) -> usize {
291        let mut count = 0usize;
292        for raw_line in source.lines() {
293            let trimmed = raw_line.trim();
294            if trimmed.is_empty() || trimmed.starts_with('#') {
295                continue;
296            }
297            if let Some(guid) = trimmed.split(',').next() {
298                self.map.insert(guid.to_string(), trimmed.to_string());
299                count += 1;
300            }
301        }
302        count
303    }
304
305    /// Return the raw mapping string for `guid`, or `None` when not present.
306    pub fn get_mapping_string(&self, guid: &str) -> Option<&str> {
307        self.map.get(guid).map(|s| s.as_str())
308    }
309
310    /// Load mappings from a file at `path`; return entry count or `EngineError` on I/O failure.
311    pub fn load_from_file(&mut self, path: &str) -> Result<usize, EngineError> {
312        let file = std::fs::File::open(path)
313            .map_err(|e| EngineError::FileSystemError(format!("Cannot open {}: {}", path, e)))?;
314        let reader = std::io::BufReader::new(file);
315        let mut content = String::new();
316        for line in reader.lines() {
317            let line = line.map_err(|e| {
318                EngineError::FileSystemError(format!("Read error in {}: {}", path, e))
319            })?;
320            content.push_str(&line);
321            content.push('\n');
322        }
323        Ok(self.load_from_string(&content))
324    }
325
326    /// Write all stored mappings to a file at `path`; return `EngineError` on I/O failure.
327    pub fn save_to_file(&self, path: &str) -> Result<(), EngineError> {
328        let mut file = std::fs::File::create(path)
329            .map_err(|e| EngineError::FileSystemError(format!("Cannot create {}: {}", path, e)))?;
330        for mapping in self.map.values() {
331            writeln!(file, "{}", mapping)
332                .map_err(|e| EngineError::FileSystemError(format!("Write error: {}", e)))?;
333        }
334        Ok(())
335    }
