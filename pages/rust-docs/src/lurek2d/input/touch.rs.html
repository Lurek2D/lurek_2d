1//! Multi-touch contact tracking with per-frame pressed/released deltas.
2//!
3//! - Position and pressure state for each active touch id.
4//! - Frame-boundary lifecycle: begin_frame clears deltas, start/move/end mutate state.
5
6use std::collections::{HashMap, HashSet};
7
8/// A single active touch contact point with its position and pressure.
9#[derive(Debug, Clone, Copy)]
10pub struct TouchPoint {
11    /// OS-assigned touch identifier; unique per active contact.
12    pub id: u64,
13    /// Horizontal position in window pixels.
14    pub x: f64,
15    /// Vertical position in window pixels.
16    pub y: f64,
17    /// Contact pressure in [0.0, 1.0]; 0.0 when unsupported by the hardware.
18    pub pressure: f64,
19}
20
21/// Per-frame state for all active touch contacts.
22#[derive(Debug, Default)]
23pub struct TouchState {
24    /// Currently active touch contacts keyed by their OS-assigned id.
25    touches: HashMap<u64, TouchPoint>,
26    /// Touch ids that first contacted the screen this frame.
27    touches_pressed: HashSet<u64>,
28    /// Touch ids that lifted from the screen this frame.
29    touches_released: HashSet<u64>,
30}
31
32/// Provide `Default` via the derive; `new` delegates to it.
33impl TouchState {
34    /// Create an empty touch state with no active contacts.
35    pub fn new() -> Self {
36        Self::default()
37    }
38
39    /// Clear per-frame delta sets; call at the start of each game frame.
40    pub fn begin_frame(&mut self) {
41        self.touches_pressed.clear();
42        self.touches_released.clear();
43    }
44
45    /// Record a new touch contact at `(x, y)` with the given `pressure`; adds `id` to pressed delta.
46    pub fn touch_start(&mut self, id: u64, x: f64, y: f64, pressure: f64) {
47        self.touches_pressed.insert(id);
48        self.touches.insert(id, TouchPoint { id, x, y, pressure });
49    }
50
51    /// Update position and pressure for an existing contact `id`; no-op when `id` is unknown.
52    pub fn touch_move(&mut self, id: u64, x: f64, y: f64, pressure: f64) {
53        if let Some(touch) = self.touches.get_mut(&id) {
54            touch.x = x;
55            touch.y = y;
56            touch.pressure = pressure;
57        }
58    }
59
60    /// Remove a contact `id` and add it to the released delta set.
61    pub fn touch_end(&mut self, id: u64) {
62        self.touches_released.insert(id);
63        self.touches.remove(&id);
64    }
65
66    /// Return true when touch `id` first contacted the screen this frame.
67    pub fn was_pressed(&self, id: u64) -> bool {
68        self.touches_pressed.contains(&id)
69    }
70
71    /// Return true when touch `id` lifted from the screen this frame.
72    pub fn was_released(&self, id: u64) -> bool {
73        self.touches_released.contains(&id)
74    }
75
76    /// Return all currently active touch contacts as a `Vec`.
77    pub fn get_touches(&self) -> Vec<TouchPoint> {
78        self.touches.values().copied().collect()
79    }
80
81    /// Return the `TouchPoint` for `id`, or `None` when that contact is not active.
82    pub fn get_touch(&self, id: u64) -> Option<TouchPoint> {
83        self.touches.get(&id).copied()
84    }
85
86    /// Return the count of currently active touch contacts.
87    pub fn get_touch_count(&self) -> usize {
88        self.touches.len()
89    }
90}
