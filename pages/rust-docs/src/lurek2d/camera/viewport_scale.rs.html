1//! Viewport scale state object used by the engine resize flow.
2//!
3//! - Stores computed scale, offset, and scaled dimensions after each resize.
4//! - Provides bidirectional game/screen coordinate conversion helpers.
5
6use crate::camera::viewport::ScaleMode;
7
8/// Stores runtime viewport scaling values for game and window surfaces.
9pub struct ViewportScale {
10    /// Stores logical game width used as scaling source.
11    pub game_width: f32,
12    /// Stores logical game height used as scaling source.
13    pub game_height: f32,
14    /// Stores active mode controlling transform behavior.
15    pub mode: ScaleMode,
16    /// Stores computed horizontal scale factor.
17    pub scale_x: f32,
18    /// Stores computed vertical scale factor.
19    pub scale_y: f32,
20    /// Stores computed horizontal offset in screen space.
21    pub offset_x: f32,
22    /// Stores computed vertical offset in screen space.
23    pub offset_y: f32,
24    /// Stores scaled game width in screen units.
25    pub scaled_width: f32,
26    /// Stores scaled game height in screen units.
27    pub scaled_height: f32,
28}
29impl ViewportScale {
30    /// Create viewport scale state and return it with identity transforms.
31    pub fn new(game_width: f32, game_height: f32, mode: ScaleMode) -> Self {
32        Self {
33            game_width,
34            game_height,
35            mode,
36            scale_x: 1.0,
37            scale_y: 1.0,
38            offset_x: 0.0,
39            offset_y: 0.0,
40            scaled_width: game_width,
41            scaled_height: game_height,
42        }
43    }
44    /// Recompute transforms from window dimensions and return after updating fields.
45    pub fn resize(&mut self, window_width: f32, window_height: f32) {
46        let (scale_x, scale_y, offset_x, offset_y) = self.mode.compute_transforms(
47            self.game_width,
48            self.game_height,
49            window_width,
50            window_height,
51        );
52        self.scale_x = scale_x;
53        self.scale_y = scale_y;
54        self.offset_x = offset_x;
55        self.offset_y = offset_y;
56        self.scaled_width = self.game_width * self.scale_x;
57        self.scaled_height = self.game_height * self.scale_y;
58    }
59    /// Read logical game dimensions and return (width, height).
60    pub fn get_game_dimensions(&self) -> (f32, f32) {
61        (self.game_width, self.game_height)
62    }
63    /// Read scaled dimensions and return (scaled_width, scaled_height).
64    pub fn get_scaled_dimensions(&self) -> (f32, f32) {
65        (self.scaled_width, self.scaled_height)
66    }
67    /// Read viewport offset and return (offset_x, offset_y).
68    pub fn get_offset(&self) -> (f32, f32) {
69        (self.offset_x, self.offset_y)
70    }
71    /// Read scale factors and return (scale_x, scale_y).
72    pub fn get_scale(&self) -> (f32, f32) {
73        (self.scale_x, self.scale_y)
74    }
75    /// Read active scale mode and return immutable reference to it.
76    pub fn get_mode(&self) -> &ScaleMode {
77        &self.mode
78    }
79    /// Convert screen coordinates into game-space coordinates and return mapped pair.
80    pub fn to_game_coords(&self, screen_x: f32, screen_y: f32) -> (f32, f32) {
81        (
82            (screen_x - self.offset_x) / self.scale_x,
83            (screen_y - self.offset_y) / self.scale_y,
84        )
85    }
86    /// Convert game coordinates into screen-space coordinates and return mapped pair.
87    pub fn to_screen_coords(&self, game_x: f32, game_y: f32) -> (f32, f32) {
88        (
89            game_x * self.scale_x + self.offset_x,
90            game_y * self.scale_y + self.offset_y,
91        )
92    }
93}
