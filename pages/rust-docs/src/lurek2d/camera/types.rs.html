1//! Core camera state containers: Camera (minimal) and Camera2D (full runtime).
2//!
3//! - Camera2D drives follow-target tracking with dead-zone, smoothing, and look-ahead.
4//! - Integrates shake, zoom pulse, sway, and breathing effects into effective transforms.
5//! - Viewport, bounds, and coordinate conversion for world/screen mapping.
6//! - Zoom and rotation damping with configurable constraint ranges.
7//! - Easing selection for follow interpolation: linear, smooth-step, ease-out-cubic.
8//! - View matrix generation composing position, rotation, zoom, and all active effects.
9//! - Presets for common follow behaviors: tight, cinematic, balanced, aggressive.
10

14#[derive(Clone, Copy, Debug, Eq, PartialEq)]
15/// Selects easing behavior used by camera tween and follow interpolation.
16pub enum CameraEasing {
17    /// Uses linear interpolation with constant progression speed.
18    Linear,
19    /// Uses smooth-step interpolation for eased start and end.
20    SmoothStep,
21    /// Uses cubic ease-out interpolation for fast start and soft stop.
22    EaseOutCubic,
23}
24impl CameraEasing {
25    /// Evaluate easing curve and return clamped interpolation factor.
26    pub fn apply(self, t: f32) -> f32 {
27        let clamped = t.clamp(0.0, 1.0);
28        match self {
29            Self::Linear => crate::math::easing::linear(clamped),
30            Self::SmoothStep => crate::math::easing::smooth_step(clamped),
31            Self::EaseOutCubic => crate::math::easing::ease_out_cubic(clamped),
32        }
33    }
34}

36/// Stores minimal camera transform state used by render command helpers.
37pub struct Camera {
38    /// Stores camera world position in scene units.
39    pub position: Vec2,
40    /// Stores uniform zoom multiplier applied to world coordinates.
41    pub zoom: f32,
42    /// Stores camera rotation angle in radians.
43    pub rotation: f32,
44}
45impl Camera {
46    /// Create camera state and return it with provided transform values.
47    pub fn new(position: Vec2, zoom: f32, rotation: f32) -> Self {
48        Camera {
49            position,
50            zoom,
51            rotation,
52        }
53    }
54    /// Build camera view matrix and return world-to-view transform.
55    pub fn view_matrix(&self) -> Mat3 {
56        let translation = Mat3::from_translation(Vec2::new(-self.position.x, -self.position.y));
57        let rotation = Mat3::from_rotation(self.rotation);
58        let scale = Mat3::from_scale(Vec2::splat(self.zoom));
59        scale * rotation * translation
60    }
61    /// Set camera position and return after replacing previous value.
62    pub fn set_position(&mut self, position: Vec2) {
63        self.position = position;
64    }
65    /// Set camera zoom and return after replacing previous value.
66    pub fn set_zoom(&mut self, zoom: f32) {
67        self.zoom = zoom;
68    }
69    /// Set camera rotation and return after replacing previous value.
70    pub fn set_rotation(&mut self, rotation: f32) {
71        self.rotation = rotation;
72    }
73}
74/// Provides default camera transform values for new contexts.
75impl Default for Camera {
76    /// Create default camera and return identity-like transform values.
77    fn default() -> Self {
78        Camera {
79            position: Vec2::ZERO,
80            zoom: 1.0,
81            rotation: 0.0,
82        }
83    }

356    pub fn to_world_coords(&self, screen_x: f32, screen_y: f32) -> (f32, f32) {
357        let z = if self.zoom.abs() > f32::EPSILON {
358            self.zoom
359        } else {
360            1.0

362        let wx = (screen_x - self.viewport.width * 0.5) / z + self.position.x + self.shake_offset.x;

434    /// Advance camera simulation and return after follow, bounds, shake, and damping updates.
435    pub fn update(&mut self, dt: f32) {
436        if let Some(target) = self.target {
437            let mut desired = target;
438            if let Some((hw, hh)) = self.dead_zone {
439                let dx = target.x - self.position.x;
440                let dy = target.y - self.position.y;
441                if dx.abs() <= hw && dy.abs() <= hh {
442                    desired = self.position;
443                }
444            }
445            if self.look_ahead > 0.0 {
446                if let Some(prev) = self.prev_target {
447                    let vx = target.x - prev.x;
448                    let vy = target.y - prev.y;
449                    desired.x += vx * self.look_ahead;
450                    desired.y += vy * self.look_ahead;
451                }
452            }
453            if self.follow_smooth > 0.0 {
454                let t = self.follow_easing.apply((self.follow_smooth * dt).min(1.0));
455                self.position.x += (desired.x - self.position.x) * t;
456                self.position.y += (desired.y - self.position.y) * t;
457            } else {
458                self.position = desired;
459            }
460            self.prev_target = Some(target);
461        }
462        if let Some(bounds) = self.bounds {
463            let z = if self.zoom_target.abs() > f32::EPSILON {
464                self.zoom_target
465            } else {
466                1.0
467            };
468            let half_w = self.viewport.width * 0.5 / z;
469            let half_h = self.viewport.height * 0.5 / z;
470            let min_x = bounds.x + half_w;
471            let max_x = bounds.x + bounds.width - half_w;
472            let min_y = bounds.y + half_h;
473            let max_y = bounds.y + bounds.height - half_h;
474            if min_x <= max_x {
475                self.position.x = self.position.x.clamp(min_x, max_x);
476            } else {
477                self.position.x = bounds.x + bounds.width * 0.5;
478            }
479            if min_y <= max_y {
480                self.position.y = self.position.y.clamp(min_y, max_y);
481            } else {
482                self.position.y = bounds.y + bounds.height * 0.5;
483            }
484        }
485        if self.shake_timer > 0.0 {
486            self.shake_timer -= dt;
487            if self.shake_timer <= 0.0 {
488                self.shake_timer = 0.0;
489                self.shake_offset = Vec2::ZERO;
490            } else {
491                let ratio = self.shake_timer / self.shake_duration;
492                let t = self.shake_timer;
493                let ox = (t * 53.0).sin() * self.shake_intensity * ratio;
494                let oy = (t * 97.0).sin() * self.shake_intensity * ratio;
495                self.shake_offset = Vec2::new(ox, oy);
496            }
497        }
498        if let Some(min_z) = self.zoom_min {
499            self.zoom_target = self.zoom_target.max(min_z);
500        }
501        if let Some(max_z) = self.zoom_max {
502            self.zoom_target = self.zoom_target.min(max_z);
503        }
504        if self.zoom_damping <= f32::EPSILON {
505            self.zoom = self.zoom_target;
506        } else {
507            let alpha = (dt.max(0.0) / (dt.max(0.0) + self.zoom_damping.max(1e-4))).clamp(0.0, 1.0);
508            self.zoom += (self.zoom_target - self.zoom) * alpha;
509        }
510        if let Some(min_r) = self.rotation_min {
511            self.rotation_target = self.rotation_target.max(min_r);
512        }
513        if let Some(max_r) = self.rotation_max {
514            self.rotation_target = self.rotation_target.min(max_r);
515        }
516        if self.rotation_damping <= f32::EPSILON {
517            self.rotation = self.rotation_target;
518        } else {
519            let alpha =
520                (dt.max(0.0) / (dt.max(0.0) + self.rotation_damping.max(1e-4))).clamp(0.0, 1.0);
521            self.rotation += (self.rotation_target - self.rotation) * alpha;
522        }
523        self.zoom_pulse.update(dt);
524        self.sway.update(dt);
525        self.breathing.update(dt);
526    }
527    /// Read zoom including active effects and return effective zoom value.
528    pub fn effective_zoom(&self) -> f32 {
529        self.zoom + self.zoom_pulse.current_delta() + self.breathing.current_delta()
530    }
531    /// Read active effect offset and return sway contribution tuple.
532    pub fn effect_offset(&self) -> (f32, f32) {
533        self.sway.current_offset()
534    }
535    /// Read render offset and return sum of sway and shake offsets.
536    pub fn render_offset(&self) -> (f32, f32) {
537        let (sx, sy) = self.sway.current_offset();
538        (sx + self.shake_offset.x, sy + self.shake_offset.y)
539    }
540    /// Build view matrix from camera state and return world-to-view transform.
541    pub fn view_matrix(&self) -> Mat3 {
542        let (ox, oy) = self.render_offset();
543        let pos = Vec2::new(self.position.x + ox, self.position.y + oy);
544        let translation = Mat3::from_translation(Vec2::new(-pos.x, -pos.y));
545        let rotation = Mat3::from_rotation(self.rotation);
546        let scale = Mat3::from_scale(Vec2::splat(self.effective_zoom()));
547        scale * rotation * translation
548    }
