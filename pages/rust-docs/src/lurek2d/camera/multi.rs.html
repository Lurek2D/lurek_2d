1//! Multi-camera rig that stores and manages named Camera2D instances.
2//!
3//! - Provides preset viewport layouts: split-screen, minimap, and picture-in-picture.
4//! - Supports bulk update and deterministic iteration for multi-view rendering passes.
5
6use crate::camera::types::Camera2D;
7use std::collections::HashMap;
8
9#[derive(Default)]
10/// Stores named camera instances used by multi-view rendering flows.
11pub struct CameraRig2D {
12    /// Maps stable camera names to mutable camera states.
13    cameras: HashMap<String, Camera2D>,
14}
15impl CameraRig2D {
16    /// Create an empty rig and return it with no registered cameras.
17    pub fn new() -> Self {
18        Self {
19            cameras: HashMap::new(),
20        }
21    }
22    /// Check camera presence by name and return true when it exists.
23    pub fn has_camera(&self, name: &str) -> bool {
24        self.cameras.contains_key(name)
25    }
26    /// Remove camera by name and return true when an entry was removed.
27    pub fn remove_camera(&mut self, name: &str) -> bool {
28        self.cameras.remove(name).is_some()
29    }
30    /// Return mutable camera by name, creating one with the provided viewport when missing.
31    pub fn ensure_camera(&mut self, name: &str, viewport_w: f32, viewport_h: f32) -> &mut Camera2D {
32        self.cameras
33            .entry(name.to_string())
34            .or_insert_with(|| Camera2D::new(viewport_w, viewport_h))
35    }
36    /// Return mutable camera reference for a registered name, or none when absent.
37    pub fn camera_mut(&mut self, name: &str) -> Option<&mut Camera2D> {
38        self.cameras.get_mut(name)
39    }
40    /// Return immutable camera reference for a registered name, or none when absent.
41    pub fn camera(&self, name: &str) -> Option<&Camera2D> {
42        self.cameras.get(name)
43    }
44    /// Apply left-right split layout and return after updating both camera viewports.
45    pub fn apply_split_screen_layout(&mut self, window_w: f32, window_h: f32) {
46        let half_w = window_w * 0.5;
47        self.ensure_camera("left", half_w, window_h)
48            .set_viewport(0.0, 0.0, half_w, window_h);
49        self.ensure_camera("right", half_w, window_h)
50            .set_viewport(half_w, 0.0, half_w, window_h);
51    }
52    /// Apply main-plus-minimap layout and return after updating camera viewports.
53    pub fn apply_minimap_layout(&mut self, window_w: f32, window_h: f32, minimap_ratio: f32) {
54        let ratio = minimap_ratio.clamp(0.1, 0.5);
55        let mm_w = window_w * ratio;
56        let mm_h = window_h * ratio;
57        self.ensure_camera("main", window_w, window_h)
58            .set_viewport(0.0, 0.0, window_w, window_h);
59        self.ensure_camera("minimap", mm_w, mm_h)
60            .set_viewport(window_w - mm_w, 0.0, mm_w, mm_h);
61    }
62    /// Apply picture-in-picture layout and return after clamping overlay viewport size.
63    pub fn apply_picture_in_picture_layout(
64        &mut self,
65        window_w: f32,
66        window_h: f32,
67        pip_w: f32,
68        pip_h: f32,
69    ) {
70        let clamped_w = pip_w.clamp(32.0, window_w.max(32.0));
71        let clamped_h = pip_h.clamp(32.0, window_h.max(32.0));
72        self.ensure_camera("main", window_w, window_h)
73            .set_viewport(0.0, 0.0, window_w, window_h);
74        self.ensure_camera("pip", clamped_w, clamped_h)
75            .set_viewport(
76                window_w - clamped_w,
77                window_h - clamped_h,
78                clamped_w,
79                clamped_h,
80            );
81    }
82    /// Update every camera with delta time and return when all states are advanced.
83    pub fn update_all(&mut self, dt: f32) {
84        for camera in self.cameras.values_mut() {
85            camera.update(dt);
86        }
87    }
88    /// Read viewport for named camera and return none when camera is missing.
89    pub fn viewport_of(&self, name: &str) -> Option<(f32, f32, f32, f32)> {
90        self.camera(name).map(Camera2D::get_viewport)
91    }
92    /// Return sorted camera names for deterministic iteration in callers.
93    pub fn camera_names(&self) -> Vec<String> {
94        let mut names: Vec<String> = self.cameras.keys().cloned().collect();
95        names.sort();
96        names
97    }
98}
