1//! Render command generation from camera transform state.
2//!
3//! - Builds PushTransform/Translate/Rotate/Scale/PopTransform sequences for Camera and Camera2D.
4//! - Separates begin/end phases so callers can sandwich scene commands between transforms.
5
6use crate::camera::types::{Camera, Camera2D};
7use crate::render::renderer::RenderCommand;
8impl Camera {
9    /// Append camera begin-transform commands and return after extending output.
10    pub fn append_begin_render_commands(&self, out: &mut Vec<RenderCommand>) {
11        out.push(RenderCommand::PushTransform);
12        out.push(RenderCommand::Translate {
13            x: -self.position.x,
14            y: -self.position.y,
15        });
16        if self.rotation.abs() > f32::EPSILON {
17            out.push(RenderCommand::Rotate {
18                angle: self.rotation,
19            });
20        }
21        if (self.zoom - 1.0).abs() > f32::EPSILON {
22            out.push(RenderCommand::Scale {
23                sx: self.zoom,
24                sy: self.zoom,
25            });
26        }
27    }
28    /// Build begin-transform command list and return it for immediate submission.
29    pub fn begin_render_commands(&self) -> Vec<RenderCommand> {
30        let mut cmds = Vec::with_capacity(4);
31        self.append_begin_render_commands(&mut cmds);
32        cmds
33    }
34    /// Return render command that restores transform stack after camera pass.
35    pub fn end_render_command() -> RenderCommand {
36        RenderCommand::PopTransform
37    }
38    /// Wrap scene commands with camera begin/end transforms and return combined list.
39    pub fn generate_render_commands(
40        &self,
41        scene_commands: Vec<RenderCommand>,
42    ) -> Vec<RenderCommand> {
43        let mut cmds = self.begin_render_commands();
44        cmds.extend(scene_commands);
45        cmds.push(RenderCommand::PopTransform);
46        cmds
47    }
48}
49impl Camera2D {
50    /// Append 2D camera begin-transform commands and return after extending output.
51    pub fn append_begin_render_commands(&self, out: &mut Vec<RenderCommand>) {
52        out.push(RenderCommand::PushTransform);
53        let (offset_x, offset_y) = self.render_offset();
54        out.push(RenderCommand::Translate {
55            x: -self.position.x - offset_x,
56            y: -self.position.y - offset_y,
57        });
58        if self.rotation.abs() > f32::EPSILON {
59            out.push(RenderCommand::Rotate {
60                angle: self.rotation,
61            });
62        }
63        let eff_zoom = self.effective_zoom();
64        if (eff_zoom - 1.0).abs() > f32::EPSILON {
65            out.push(RenderCommand::Scale {
66                sx: eff_zoom,
67                sy: eff_zoom,
68            });
69        }
70    }
71    /// Build 2D begin-transform command list and return it for submission.
72    pub fn begin_render_commands(&self) -> Vec<RenderCommand> {
73        let mut cmds = Vec::with_capacity(4);
74        self.append_begin_render_commands(&mut cmds);
75        cmds
76    }
77    /// Return render command that restores transform stack after 2D camera pass.
78    pub fn end_render_command() -> RenderCommand {
79        RenderCommand::PopTransform
80    }
81    /// Wrap scene commands with 2D camera transforms and return combined list.
82    pub fn generate_render_commands(
83        &self,
84        scene_commands: Vec<RenderCommand>,
85    ) -> Vec<RenderCommand> {
86        let mut cmds = self.begin_render_commands();
87        cmds.extend(scene_commands);
88        cmds.push(RenderCommand::PopTransform);
89        cmds
90    }
