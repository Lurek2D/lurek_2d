1//! Custom image cursor built from RGBA pixel data with configurable hotspot offset.
2//!
3//! - `CustomCursor` stores width, height, hotspot `(x, y)`, and a flat RGBA `Vec<u8>`.
4//! - Pixel data is validated at construction; mismatched dimensions return an error.
5//! - Used directly or as frames inside `AnimatedCursor`.
6//! - Exposed to Lua scripts via `lurek.cursor.set_custom()`.
7/// A custom cursor from RGBA pixel data.
8#[derive(Debug, Clone)]
9pub struct CustomCursor {
10    /// Width in pixels.
11    pub width: u32,
12    /// Height in pixels.
13    pub height: u32,
14    /// Horizontal hotspot offset from the cursor image origin.
15    pub hotspot_x: u32,
16    /// Vertical hotspot offset from the cursor image origin.
17    pub hotspot_y: u32,
18    pixels: Vec<u8>,
19}
20
21impl CustomCursor {
22    /// Create a blank (all-zero) custom cursor with the given dimensions and hotspot.
23    pub fn new(width: u32, height: u32, hotspot_x: u32, hotspot_y: u32) -> Self {
24        Self {
25            width,
26            height,
27            hotspot_x: hotspot_x.min(width.saturating_sub(1)),
28            hotspot_y: hotspot_y.min(height.saturating_sub(1)),
29            pixels: vec![0u8; (width * height * 4) as usize],
30        }
31    }
32
33    /// Create a cursor from raw RGBA byte data; returns `None` if `data.len()` does not match dimensions.
34    pub fn from_rgba(width: u32, height: u32, hotspot_x: u32, hotspot_y: u32, data: Vec<u8>) -> Option<Self> {
35        let expected = (width * height * 4) as usize;
36        if data.len() != expected {
37            return None;
38        }
39        Some(Self {
40            width,
41            height,
42            hotspot_x: hotspot_x.min(width.saturating_sub(1)),
43            hotspot_y: hotspot_y.min(height.saturating_sub(1)),
44            pixels: data,
45        })
46    }
47
48    /// Set the RGBA value of the pixel at `(x, y)`; out-of-bounds writes are silently ignored.
49    pub fn set_pixel(&mut self, x: u32, y: u32, r: u8, g: u8, b: u8, a: u8) {
50        if x < self.width && y < self.height {
51            let idx = ((y * self.width + x) * 4) as usize;
52            self.pixels[idx] = r;
53            self.pixels[idx + 1] = g;
54            self.pixels[idx + 2] = b;
55            self.pixels[idx + 3] = a;
56        }
57    }
58
59    /// Return the RGBA value at `(x, y)`, or `None` if the coordinates are out of bounds.
60    pub fn get_pixel(&self, x: u32, y: u32) -> Option<(u8, u8, u8, u8)> {
61        if x < self.width && y < self.height {
62            let idx = ((y * self.width + x) * 4) as usize;
63            Some((self.pixels[idx], self.pixels[idx + 1], self.pixels[idx + 2], self.pixels[idx + 3]))
64        } else {
65            None
66        }
67    }
68
69    /// Return the full flat RGBA pixel buffer (width × height × 4 bytes).
70    pub fn pixels(&self) -> &[u8] {
71        &self.pixels
72    }
73
74    /// Return the cursor image dimensions as `(width, height)` in pixels.
75    pub fn size(&self) -> (u32, u32) {
76        (self.width, self.height)
77    }
78}
