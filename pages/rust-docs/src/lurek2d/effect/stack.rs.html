1//! Ordered post-processing effect stack with per-entry enable flags.
2//!
3//! - Index-based effect references aligned with a parallel enabled vector.
4//! - Stack manipulation: add, remove, insert, reorder, deduplicate.
5//! - Query helpers for enabled subset, dimensions, and positional lookup.
6//! - Debug visualization renderers for stack state, catalogs, parameters, and type bars.
7
8use crate::log_msg;
9use crate::runtime::log_messages::{FX01, FX02};
10#[derive(Debug, Clone)]
11/// Stores an ordered list of effect indices together with enable flags and target size.
12pub struct PostFxStack {
13    /// Effect indices in application order.
14    pub effects: Vec<usize>,
15    /// Per-entry enable flags parallel to `effects`.
16    pub enabled: Vec<bool>,
17    /// Target render width for stack application.
18    pub width: u32,
19    /// Target render height for stack application.
20    pub height: u32,
21    /// Indicates whether the renderer is currently capturing into this stack.
22    pub capturing: bool,
23}
24impl PostFxStack {
25    /// Creates an empty post-effect stack for the given render size.
26    pub fn new(width: u32, height: u32) -> Self {
27        log_msg!(debug, FX01);
28        Self {
29            effects: Vec::new(),
30            enabled: Vec::new(),
31            width,
32            height,
33            capturing: false,
34        }
35    }
36    /// Appends an enabled effect index to the end of the stack.
37    pub fn add(&mut self, effect_idx: usize) {
38        log_msg!(debug, FX02);
39        self.effects.push(effect_idx);
40        self.enabled.push(true);
41    }
42    /// Removes the first stack entry that references the given effect index.
43    pub fn remove(&mut self, effect_idx: usize) -> bool {
44        if let Some(pos) = self.effects.iter().position(|&e| e == effect_idx) {
45            self.effects.remove(pos);
46            self.enabled.remove(pos);
47            true
48        } else {
49            false
50        }
51    }
52    /// Inserts an enabled effect index at a one-based stack position.
53    pub fn insert(&mut self, position: usize, effect_idx: usize) {
54        let idx = (position.saturating_sub(1)).min(self.effects.len());
55        self.effects.insert(idx, effect_idx);
56        self.enabled.insert(idx, true);
57    }
58    /// Sets the enable flag for the first stack entry that references the effect index.
59    pub fn set_enabled(&mut self, effect_idx: usize, is_enabled: bool) {
60        if let Some(pos) = self.effects.iter().position(|&e| e == effect_idx) {
61            self.enabled[pos] = is_enabled;
62        }
63    }
64    /// Returns the enable flag for the first stack entry that references the effect index.
65    pub fn is_enabled(&self, effect_idx: usize) -> bool {
66        self.effects
67            .iter()
68            .position(|&e| e == effect_idx)
69            .map(|pos| self.enabled[pos])
70            .unwrap_or(false)
71    }
72    /// Returns the number of stack entries.
73    pub fn get_effect_count(&self) -> usize {
74        self.effects.len()
75    }
76    /// Returns the effect index at a one-based stack position.
77    pub fn get_effect(&self, index: usize) -> Option<usize> {
78        if index >= 1 && index <= self.effects.len() {
79            Some(self.effects[index - 1])
80        } else {
81            None
82        }
83    }
84    /// Returns the effect indices whose stack entries are currently enabled.
85    pub fn enabled_effects(&self) -> Vec<usize> {
86        self.effects
87            .iter()
88            .zip(self.enabled.iter())
89            .filter(|(_, &en)| en)
90            .map(|(&idx, _)| idx)
91            .collect()
92    }
93    /// Updates the target render dimensions stored on the stack.
94    pub fn resize(&mut self, width: u32, height: u32) {
95        self.width = width;
96        self.height = height;
97    }
98    /// Returns the target render width.
99    pub fn get_width(&self) -> u32 {
100        self.width
101    }
102    /// Returns the target render height.
103    pub fn get_height(&self) -> u32 {
104        self.height
105    }
106    /// Returns the target render dimensions as `(width, height)`.
107    pub fn get_dimensions(&self) -> (u32, u32) {
108        (self.width, self.height)
109    }
110    /// Returns the number of stack entries.
111    pub fn len(&self) -> usize {
112        self.effects.len()
113    }
114    /// Returns whether the stack has no entries.
115    pub fn is_empty(&self) -> bool {
116        self.effects.is_empty()
117    }
118    /// Removes every stack entry and enable flag.
119    pub fn clear(&mut self) {
120        self.effects.clear();
121        self.enabled.clear();
122    }
123    /// Removes duplicate effect indices while preserving first occurrence order.
124    pub fn dedup_indices(&mut self) -> usize {
125        let mut seen = std::collections::HashSet::new();
126        let before = self.effects.len();
127        let mut new_effects = Vec::with_capacity(before);
128        let mut new_enabled = Vec::with_capacity(before);
129        for (idx, enabled) in self.effects.iter().zip(self.enabled.iter()) {
130            if seen.insert(*idx) {
131                new_effects.push(*idx);
132                new_enabled.push(*enabled);
133            }
134        }
135        let removed = before - new_effects.len();
136        self.effects = new_effects;
137        self.enabled = new_enabled;
138        removed
139    }
140    /// Renders a debug overview of stack entries and their enabled state.
141    pub fn draw_info_to_image(&self, width: u32, height: u32) -> crate::image::ImageData {
142        let mut img = crate::image::ImageData::new(width, height);
143        img.fill(20, 20, 30, 255);
144        let count = self.effects.len();
145        if count == 0 {
146            img.draw_label("EMPTY STACK", 10, 10, 180, 180, 190);
147            return img;
148        }
149        let box_gap = 10u32;
150        let total_gap = box_gap * (count as u32 + 1);
151        let box_w = if count > 0 {
152            (width.saturating_sub(total_gap)) / count as u32
153        } else {
154            60
155        };
156        let box_h = height.saturating_sub(60).min(100);
157        let box_y = (height - box_h) / 2;
158        for (i, &_effect_idx) in self.effects.iter().enumerate() {
159            let bx = box_gap + i as u32 * (box_w + box_gap);
160            let enabled = self.enabled.get(i).copied().unwrap_or(false);
161            let (r, g, b) = if enabled {
162                (80u8, 200u8, 80u8)
163            } else {
164                (200u8, 60u8, 60u8)
165            };
166            img.draw_rect(
167                bx as i32,
168                box_y as i32,
169                box_w,
170                box_h,
171                r / 3,
172                g / 3,
173                b / 3,
174                200,
175            );
176            img.draw_rect(bx as i32, box_y as i32, box_w, 2, r, g, b, 255);
177            img.draw_rect(
178                bx as i32,
179                (box_y + box_h - 2) as i32,
180                box_w,
181                2,
182                r,
183                g,
184                b,
185                255,
186            );
187            if !enabled {
188                img.draw_line(
189                    bx as i32 + 10,
190                    box_y as i32 + 10,
191                    (bx + box_w) as i32 - 10,
192                    (box_y + box_h) as i32 - 10,
193                    200,
194                    60,
195                    60,
196                    255,
197                );
198                img.draw_line(
199                    (bx + box_w) as i32 - 10,
200                    box_y as i32 + 10,
201                    bx as i32 + 10,
202                    (box_y + box_h) as i32 - 10,
203                    200,
204                    60,
205                    60,
206                    255,
207                );
208            }
209            let label = format!("FX{}", i);
210            img.draw_label(&label, bx as i32 + 4, box_y as i32 + 4, r, g, b);
211        }
212        img
213    }
214    /// Renders a labeled debug panel for stack management operations.
215    pub fn draw_stack_management_to_image(
216        &self,
217        width: u32,
218        height: u32,
219        labels: &[&str],
220    ) -> crate::image::ImageData {
221        let mut img = crate::image::ImageData::new(width, height);
222        img.fill(20, 18, 28, 255);
223        img.draw_label("STACK OPS", (width / 2 - 30) as i32, 4, 200, 180, 255);
224        for i in 0..self.effects.len() {
225            let y = 24 + i as i32 * 22;
226            let enabled = self.enabled.get(i).copied().unwrap_or(false);
227            let (cr, cg, cb) = if enabled {
228                (80u8, 200u8, 80u8)
229            } else {
230                (200u8, 80u8, 80u8)
231            };
232            img.draw_rect(
233                10,
234                y,
235                (width - 20).min(300),
236                18,
237                cr / 5,
238                cg / 5,
239                cb / 5,
240                255,
241            );
242            let label = labels.get(i).copied().unwrap_or("FX");
243            let text = format!("{} - {}", label, if enabled { "ON" } else { "OFF" });
244            img.draw_label(&text, 14, y + 4, cr, cg, cb);
245        }
246        img
247    }
248    /// Renders a tiled debug catalog for effect labels and representative colors.
249    pub fn draw_effect_catalog_to_image(
250        entries: &[(&str, (u8, u8, u8))],
251        width: u32,
252        height: u32,
253    ) -> crate::image::ImageData {
254        let mut img = crate::image::ImageData::new(width, height);
255        img.fill(20, 18, 28, 255);
256        img.draw_label("POSTFX CATALOG", (width / 2 - 40) as i32, 4, 220, 180, 255);
257        let cols = 4u32;
258        let cell_w = width / cols;
259        let rows = (entries.len() as u32).div_ceil(cols);
260        let cell_h = if rows > 0 {
261            (height - 20) / rows
262        } else {
263            height
264        };
265        for (i, &(label, (cr, cg, cb))) in entries.iter().enumerate() {
266            let col = (i as u32) % cols;
267            let row = (i as u32) / cols;
268            let px = col * cell_w;
269            let py = 20 + row * cell_h;
270            img.draw_rect(
271                (px + 2) as i32,
272                (py + 2) as i32,
273                cell_w - 4,
274                cell_h - 4,
275                cr / 5,
276                cg / 5,
277                cb / 5,
278                200,
279            );
280            img.draw_label(label, (px + 4) as i32, (py + 4) as i32, cr, cg, cb);
281        }
282        img
283    }
284    /// Renders a labeled debug panel showing effect parameter names and values.
285    pub fn draw_effect_parameters_to_image(
286        entries: &[(&str, &[(&str, f32)])],
287        width: u32,
288        height: u32,
289    ) -> crate::image::ImageData {
290        let mut img = crate::image::ImageData::new(width, height);
291        img.fill(20, 18, 28, 255);
292        let title_x = width.saturating_sub(76) / 2;
293        img.draw_label("POSTFX PARAMETERS", title_x as i32, 4, 200, 180, 255);
294        let mut y = 24i32;
295        for &(label, params) in entries {
296            if y + 52 > height as i32 {
297                break;
298            }
299            img.draw_rect(10, y, width - 20, 50, 30, 28, 42, 255);
300            img.draw_label(label, 14, y + 4, 220, 180, 100);
301            let mut px = 14i32;
302            for &(name, val) in params {
303                let text = format!("{}:{:.1}", name.to_uppercase(), val);
304                img.draw_label(&text, px, y + 18, 100, 200, 100);
305                px += (text.len() as i32 + 1) * 4;
306            }
307            y += 58;
308        }
309        img

311    /// Renders one colored debug row per effect type together with its parameter count.
312    pub fn draw_effect_type_bars_to_image(
313        entries: &[(&str, (u8, u8, u8), usize)],
314        width: u32,
315        height: u32,
316    ) -> crate::image::ImageData {
317        let mut img = crate::image::ImageData::new(width, height);
318        img.fill(25, 25, 35, 255);
319        img.draw_label(
320            "POSTFX EFFECT TYPES",
321            (width / 2).saturating_sub(60) as i32,
322            4,
323            200,
324            180,
325            255,
326        );
327        let row_h = if entries.is_empty() {
328            height
329        } else {
330            (height - 20) / entries.len() as u32
331        };
332        for (i, &(label, (cr, cg, cb), param_count)) in entries.iter().enumerate() {
333            let y_base = (20 + i as u32 * row_h) as i32;
334            let row_h_i = row_h.saturating_sub(4);
335            img.draw_rect(20, y_base, width - 40, row_h_i, cr / 3, cg / 3, cb / 3, 200);
336            img.draw_rect(20, y_base, width - 40, 2, cr, cg, cb, 255);
337            img.draw_label(label, 28, y_base + 4, cr, cg, cb);
338            for p in 0..param_count {
339                let dot_x = 40 + p as i32 * 20;
340                let dot_y = y_base + row_h as i32 / 2;
341                img.draw_circle(dot_x, dot_y, 5, cr, cg, cb, 255);
342            }
343        }
344        img
345    }
346    /// Renders a debug catalog for a list of effect types using synthetic colors.
347    pub fn draw_effect_types_to_image(
348        types: &[super::PostFxEffectType],
349        width: u32,
350        height: u32,
351    ) -> crate::image::ImageData {
352        let palette: &[(u8, u8, u8)] = &[
353            (180, 80, 80),
354            (80, 180, 80),
355            (80, 80, 180),
356            (180, 180, 80),
357            (180, 80, 180),
358            (80, 180, 180),
359            (200, 130, 60),
360            (130, 60, 200),
361        ];
362        let entries: Vec<(&str, (u8, u8, u8), usize)> = types
363            .iter()
364            .enumerate()
365            .map(|(i, t)| {
366                let effect = super::PostFxEffect::new(*t);
367                let param_count = effect.get_parameter_names().len();
368                let label = t.debug_label();
369                let color = palette[i % palette.len()];
370                (label, color, param_count)
371            })
372            .collect();
373        Self::draw_effect_type_bars_to_image(&entries, width, height)
374    }
