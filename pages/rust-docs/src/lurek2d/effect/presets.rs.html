1//! Built-in post-processing effect presets (retro TV, horror, dream, neon, sepia).
2//!
3//! - Preset construction with viewport-sized stack initialization.
4//! - Static name lookup for canonical preset identifiers.
5
6use super::effect::PostFxEffect;
7use super::effect_type::PostFxEffectType;
8use super::stack::PostFxStack;
9
10/// Bundle a preset name, its effect list, and a prepared stack ordering.
11#[derive(Debug, Clone)]
12pub struct EffectPreset {
13    /// Stable preset identifier exposed to callers.
14    pub name: &'static str,
15    /// Effect instances that make up this preset.
16    pub effects: Vec<PostFxEffect>,
17    /// Stack ordering that enables every effect in the preset.
18    pub stack: PostFxStack,
19}
20/// Returns the canonical names of all built-in post-effect presets.
21pub fn preset_names() -> Vec<&'static str> {
22    vec!["retro_tv", "horror", "dream", "neon", "sepia_age"]
23}
24/// Builds a named preset and initializes a stack sized for the target viewport.
25pub fn build_preset(name: &str, width: u32, height: u32) -> Option<EffectPreset> {
26    let effects: Vec<PostFxEffect> = match name.to_ascii_lowercase().as_str() {
27        "retro_tv" => build_retro_tv(),
28        "horror" => build_horror(),
29        "dream" => build_dream(),
30        "neon" => build_neon(),
31        "sepia_age" => build_sepia_age(),
32        _ => return None,
33    };
34    let count = effects.len();
35    let mut stack = PostFxStack::new(width, height);
36    for i in 0..count {
37        stack.add(i);
38    }
39    Some(EffectPreset {
40        name: preset_static_name(name),
41        effects,
42        stack,
43    })
44}
45/// Builds the retro TV preset effect chain.
46fn build_retro_tv() -> Vec<PostFxEffect> {
47    let mut bloom = PostFxEffect::new(PostFxEffectType::Bloom);
48    bloom.set_param("threshold", 0.6);
49    bloom.set_param("intensity", 0.4);
50    let mut crt = PostFxEffect::new(PostFxEffectType::Crt);
51    crt.set_param("curvature", 0.3);
52    let mut scan = PostFxEffect::new(PostFxEffectType::Scanlines);
53    scan.set_param("intensity", 0.35);
54    let mut chrom = PostFxEffect::new(PostFxEffectType::Chromatic);
55    chrom.set_param("offset", 0.003);
56    vec![bloom, crt, scan, chrom]
57}
58/// Builds the horror preset effect chain.
59fn build_horror() -> Vec<PostFxEffect> {
60    let mut vignette = PostFxEffect::new(PostFxEffectType::Vignette);
61    vignette.set_param("radius", 0.55);
62    vignette.set_param("softness", 0.4);
63    let mut noise = PostFxEffect::new(PostFxEffectType::Noise);
64    noise.set_param("strength", 0.12);
65    let mut gray = PostFxEffect::new(PostFxEffectType::Grayscale);
66    gray.set_param("strength", 0.6);
67    let mut chrom = PostFxEffect::new(PostFxEffectType::Chromatic);
68    chrom.set_param("offset", 0.002);
69    vec![vignette, noise, gray, chrom]
70}
71/// Builds the dream preset effect chain.
72fn build_dream() -> Vec<PostFxEffect> {
73    let mut bloom = PostFxEffect::new(PostFxEffectType::Bloom);
74    bloom.set_param("threshold", 0.3);
75    bloom.set_param("intensity", 0.9);
76    let mut blur = PostFxEffect::new(PostFxEffectType::Blur);
77    blur.set_param("radius", 2.0);
78    let mut hue = PostFxEffect::new(PostFxEffectType::HueShift);
79    hue.set_param("hue", 20.0);
80    vec![bloom, blur, hue]
81}
82/// Builds the neon preset effect chain.
83fn build_neon() -> Vec<PostFxEffect> {
84    let mut edge = PostFxEffect::new(PostFxEffectType::EdgeDetect);
85    edge.set_param("threshold", 0.1);
86    let mut hue = PostFxEffect::new(PostFxEffectType::HueShift);
87    hue.set_param("hue", 160.0);
88    let mut bloom = PostFxEffect::new(PostFxEffectType::Bloom);
89    bloom.set_param("threshold", 0.4);
90    bloom.set_param("intensity", 0.7);
91    vec![edge, hue, bloom]
92}
93/// Builds the aged sepia preset effect chain.
94fn build_sepia_age() -> Vec<PostFxEffect> {
95    let mut sepia = PostFxEffect::new(PostFxEffectType::Sepia);
96    sepia.set_param("strength", 0.85);
97    let mut noise = PostFxEffect::new(PostFxEffectType::Noise);
98    noise.set_param("strength", 0.08);
99    let mut vignette = PostFxEffect::new(PostFxEffectType::Vignette);
100    vignette.set_param("radius", 0.65);
101    vignette.set_param("softness", 0.35);
102    vec![sepia, noise, vignette]
103}
104/// Maps a user-supplied preset name to the stored static preset identifier.
105fn preset_static_name(name: &str) -> &'static str {
106    match name.to_ascii_lowercase().as_str() {
107        "horror" => "horror",
108        "dream" => "dream",
109        "neon" => "neon",
110        "sepia_age" => "sepia_age",
111        _ => "retro_tv",
112    }
113}
