1//! Collect bounded rolling history of frame-delta samples
2//!
3//! - Compute aggregate metrics: FPS, average, min, max, and percentiles
4//! - Produce immutable snapshots summarizing recent frame performance
5
6use std::collections::VecDeque;
7#[derive(Debug)]
8/// Store a bounded history of frame delta samples for aggregate stats.
9pub struct FrameStats {
10    /// Hold recent frame delta values in seconds.
11    pub history: VecDeque<f64>,
12    /// Define maximum number of samples retained in history.
13    pub capacity: usize,
14}
15impl FrameStats {
16    /// Create frame stats with bounded capacity and return the instance.
17    pub fn new(capacity: usize) -> Self {
18        Self {
19            history: VecDeque::new(),
20            capacity: capacity.max(10),
21        }
22    }
23    /// Append one frame delta sample and drop oldest samples past capacity.
24    pub fn record(&mut self, dt: f64) {
25        self.history.push_back(dt);
26        if self.history.len() > self.capacity {
27            let _ = self.history.pop_front();
28        }
29    }
30    /// Set the sample capacity and trim stored history to the new bound.
31    pub fn set_capacity(&mut self, cap: usize) {
32        self.capacity = cap.clamp(10, 10_000);
33        while self.history.len() > self.capacity {
34            let _ = self.history.pop_front();
35        }
36    }
37    /// Compute aggregate frame metrics and return zeros when history is empty.
38    pub fn snapshot(&self) -> FrameSnapshot {
39        if self.history.is_empty() {
40            return FrameSnapshot::zero();
41        }
42        let mut sorted: Vec<f64> = self.history.iter().copied().collect();
43        sorted.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
44        let n = sorted.len();
45        let sum: f64 = sorted.iter().sum();
46        let avg = sum / n as f64;
47        let pct = |p: f64| {
48            let idx = ((p / 100.0) * (n as f64 - 1.0)).round() as usize;
49            sorted[idx.min(n - 1)]
50        };
51        FrameSnapshot {
52            fps: if avg > 0.0 { 1.0 / avg } else { 0.0 },
53            dt: *sorted.last().unwrap_or(&0.0),
54            avg,
55            min: sorted[0],
56            max: sorted[n - 1],
57            p50: pct(50.0),
58            p95: pct(95.0),
59            p99: pct(99.0),
60            samples: n,
61        }
62    }
63}
64/// Provide default frame stats with a 300-sample rolling window.
65impl Default for FrameStats {
66    fn default() -> Self {
67        Self::new(300)
68    }
69}
70#[derive(Debug, Clone)]
71/// Hold computed frame-time and FPS summary metrics from sampled history.
72pub struct FrameSnapshot {
73    /// Store estimated frames per second derived from average delta.
74    pub fps: f64,
75    /// Store most recent frame delta value in seconds.
76    pub dt: f64,
77    /// Store average frame delta across retained samples.
78    pub avg: f64,
79    /// Store minimum frame delta observed in retained samples.
80    pub min: f64,
81    /// Store maximum frame delta observed in retained samples.
82    pub max: f64,
83    /// Store median-like 50th percentile frame delta.
84    pub p50: f64,
85    /// Store 95th percentile frame delta for tail-latency tracking.
86    pub p95: f64,
87    /// Store 99th percentile frame delta for worst-frame tracking.
88    pub p99: f64,
89    /// Store number of samples included in this snapshot.
90    pub samples: usize,
91}
92impl FrameSnapshot {
93    /// Create a zeroed snapshot and return it for empty-history cases.
94    fn zero() -> Self {
95        Self {
96            fps: 0.0,
97            dt: 0.0,
98            avg: 0.0,
99            min: 0.0,
100            max: 0.0,
101            p50: 0.0,
102            p95: 0.0,
103            p99: 0.0,
104            samples: 0,
105        }
106    }
107}
