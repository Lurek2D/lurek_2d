1//! Compatibility wrapper around the release-safe REPL core
2//!
3//! - Preserves the devtools `ReplConsole` API and bounded history behavior
4//! - Returns expression results, success markers, command text, or formatted error text
5
6use crate::repl::ReplSession;
7#[derive(Debug, Clone)]
8/// Store REPL command history and capacity limits for local dev sessions.
9pub struct ReplConsole {
10    /// Shared release-safe REPL implementation.
11    inner: ReplSession,
12}
13/// Provide default REPL history capacity for convenience construction.
14impl Default for ReplConsole {
15    fn default() -> Self {
16        Self::new(200)
17    }
18}
19impl ReplConsole {
20    /// Create a REPL console with bounded history and return the instance.
21    pub fn new(max_history: usize) -> Self {
22        Self {
23            inner: ReplSession::new(max_history),
24        }
25    }
26    /// Evaluate input and return expression result, ok marker, or error text.
27    pub fn eval(&mut self, input: &str, lua: &mlua::Lua) -> String {
28        let result = self.inner.eval_line(input, lua).display_text();
29        log::debug!("devtools: repl eval → {result}");
30        result
31    }
32    /// Return an immutable slice of stored history entries.
33    pub fn history(&self) -> &[String] {
34        self.inner.history()
35    }
36    /// Clear command history and return unit.
37    pub fn clear(&mut self) {
38        self.inner.clear();
39        log::debug!("devtools: repl history cleared");
40    }
41    /// Return the number of stored history entries.
42    pub fn len(&self) -> usize {
43        self.inner.len()
44    }
45    /// Return true when history contains no entries.
46    pub fn is_empty(&self) -> bool {
47        self.inner.is_empty()
48    }
49}
