1//! Worker VM lifecycle: spawn an OS thread with an isolated Lua VM, track Pending/Running/Completed/Error states.
2//!
3//! - Restricted API surface: inject only `lurek.thread.getChannel`, `lurek.fs.read`, and `arg` into worker VMs.
4//! - Channel-based communication: workers receive a shared channel registry for typed cross-VM messaging.
5//! - Blocking and timeout joins: wait indefinitely or poll with a deadline for worker completion.
6//! - Path-traversal guard: `fs.read` in worker VMs rejects `..` segments to prevent sandbox escape.
7
8use crate::log_msg;
9use crate::runtime::log_messages::{TH01_WORKER_INIT, TH02_WORKER_START, TH04_WORKER_ERROR};
10use crate::thread::channel::{channel_value_to_lua, Channel, ChannelValue, LuaChannel};
11use std::collections::HashMap;
12use std::sync::{Arc, Mutex};
13use std::thread;
14use std::time::{Duration, Instant};
15
16/// Lifecycle state of a `LuaThread`.
17#[derive(Debug, Clone, PartialEq)]
18pub enum ThreadState {
19    /// Thread has been created but `start` has not been called yet.
20    Pending,
21    /// Thread is actively executing Lua code.
22    Running,
23    /// Lua code returned without error.
24    Completed,
25    /// Lua code terminated with the given error string.
26    Error(String),
27}
28
29/// A single Lua VM running on its own OS thread with a restricted `lurek.*` API.
30pub struct LuaThread {
31    /// Lua source code to execute in the worker VM.
32    code: String,
33    /// Current lifecycle state shared with the worker OS thread.
34    state: Arc<Mutex<ThreadState>>,
35    /// OS thread join handle; `None` before `start` or after `wait`.
36    handle: Option<thread::JoinHandle<()>>,
37    /// Named channel registry injected into the worker VM via `lurek.thread.getChannel`.
38    channels: Arc<Mutex<HashMap<String, Arc<Channel>>>>,
39}
40/// Lifecycle management for a worker Lua VM running on a dedicated OS thread.
41impl LuaThread {
42    /// Create a `LuaThread` ready to run `code` with access to `channels`; does not start the OS thread.
43    pub fn new(code: String, channels: Arc<Mutex<HashMap<String, Arc<Channel>>>>) -> Self {
44        log_msg!(debug, TH01_WORKER_INIT);
45        Self {
46            code,
47            state: Arc::new(Mutex::new(ThreadState::Pending)),
48            handle: None,
49            channels,
50        }
51    }
52    /// Spawn the OS thread, inject `args` into the `arg` global, and begin executing `code`; returns an error when already running.
53    pub fn start(&mut self, args: Vec<ChannelValue>) -> Result<(), String> {
54        if *self.state.lock().unwrap() == ThreadState::Running {
55            log_msg!(error, TH04_WORKER_ERROR, "already running");
56            return Err("Thread is already running".into());
57        }
58        let code = self.code.clone();
59        let state = self.state.clone();
60        let channels = self.channels.clone();
61        *state.lock().unwrap() = ThreadState::Running;
62        log_msg!(info, TH02_WORKER_START);
63        let handle = thread::spawn(move || {
64            let lua = mlua::Lua::new();
65            if let Err(e) = register_thread_safe_modules(&lua, &channels, &args) {
66                *state.lock().unwrap() = ThreadState::Error(e.to_string());
67                return;
68            }
69            match lua.load(&code).exec() {
70                Ok(()) => {
71                    *state.lock().unwrap() = ThreadState::Completed;
72                }
73                Err(e) => {
74                    *state.lock().unwrap() = ThreadState::Error(e.to_string());
75                }
76            }
77        });
78        self.handle = Some(handle);
79        Ok(())
80    }
81    /// Block the calling thread until the worker finishes; no-op when no handle is present.
82    pub fn wait(&mut self) {
83        if let Some(handle) = self.handle.take() {
84            let _ = handle.join();
85        }
86    }
87
88    /// Poll every millisecond until the worker finishes or `timeout_secs` elapses; return `true` when finished in time.
89    pub fn wait_timeout(&mut self, timeout_secs: f64) -> bool {
90        if self.handle.is_none() {
91            return true;
92        }
93        let timeout = Duration::from_secs_f64(timeout_secs.max(0.0));
94        let deadline = Instant::now() + timeout;
95        loop {
96            let finished = self
97                .handle
98                .as_ref()
99                .map(|h| h.is_finished())
100                .unwrap_or(true);
101            if finished {
102                if let Some(handle) = self.handle.take() {
103                    let _ = handle.join();
104                }
105                return true;
106            }
107            if Instant::now() >= deadline {
108                return false;
109            }
110            thread::sleep(Duration::from_millis(1));
111        }
112    }
113    /// Return `true` while the worker OS thread state is `Running`.
114    pub fn is_running(&self) -> bool {
115        *self.state.lock().unwrap() == ThreadState::Running
116    }
117
118    /// Return the error message when the worker terminated with `ThreadState::Error`, or `None`.
119    pub fn get_error(&self) -> Option<String> {
120        match &*self.state.lock().unwrap() {
121            ThreadState::Error(e) => Some(e.clone()),
122            _ => None,
123        }
124    }
125}
126/// Return the `lurek.*` capabilities available inside worker VMs.
127pub fn worker_capabilities() -> &'static [&'static str] {
128    &[
129        "lurek.thread.getChannel",
130        "lurek.fs.read",
131        "arg",
132        "package.path",
133    ]
134}
135/// Register the minimal `lurek.thread`, `lurek.fs`, and `arg` API into `lua` for worker use.
136fn register_thread_safe_modules(
137    lua: &mlua::Lua,
138    channels: &Arc<Mutex<HashMap<String, Arc<Channel>>>>,
139    args: &[ChannelValue],
140) -> mlua::Result<()> {
141    let lurek = lua.create_table()?;
142    let thread_table = lua.create_table()?;
143    let channels_clone = channels.clone();
144    thread_table.set(
145        "getChannel",
146        lua.create_function(move |_, name: String| {
147            let channels = channels_clone.lock().unwrap();
148            match channels.get(&name) {
149                Some(ch) => Ok(LuaChannel { inner: ch.clone() }),
150                None => Err(mlua::Error::RuntimeError(format!(
151                    "Channel '{}' not found",
152                    name
153                ))),
154            }
155        })?,
156    )?;
157    lurek.set("thread", thread_table)?;
158    let fs_table = lua.create_table()?;
159    fs_table.set(
160        "read",
161        lua.create_function(|lua_ctx, path: String| {
162            if path.contains("..") {
163                return Err(mlua::Error::RuntimeError(
164                    "lurek.filesystem.read: path traversal not allowed".to_string(),
165                ));
166            }
167            match std::fs::read_to_string(&path) {
168                Ok(content) => Ok(mlua::Value::String(lua_ctx.create_string(&content)?)),
169                Err(e) => Err(mlua::Error::RuntimeError(format!("fs.read: {}", e))),
170            }
171        })?,
172    )?;
173    lurek.set("fs", fs_table)?;
174    lua.globals().set("lurek", lurek)?;
175    if let Ok(package) = lua.globals().get::<_, mlua::Table>("package") {
176        let current: String = package.get("path").unwrap_or_default();
177        let new_path = if current.is_empty() {
178            "./?.lua;./?/init.lua".to_string()
179        } else {
180            format!("{}:./?.lua:./?.lua/init.lua", current)
181        };
182        let _ = package.set("path", new_path);
183    }
184    let arg_table = lua.create_table()?;
185    for (i, val) in args.iter().enumerate() {
186        let lua_val = channel_value_to_lua(lua, val.clone())?;
187        arg_table.set(i + 1, lua_val)?;
188    }
189    lua.globals().set("arg", arg_table)?;
190    Ok(())
191}
