1//! Dialog gate conditions: guards that control branch and topic visibility.
2//!
3//! - `Condition` is an enum with variants for flag checks, stat comparisons, and Lua callbacks.
4//! - Evaluated lazily at the point where the dialog engine requests the next node.
5//! - Lua callback conditions receive the current `DialogState` as a table argument.
6//! - Composed with `And` / `Or` / `Not` wrappers for complex gating logic.
7
8use std::collections::HashMap;
9
10/// Context provided to gate evaluation.
11#[derive(Debug, Clone, Default)]
12pub struct GateContext {
13    /// Current FSM state name, if any.
14    pub fsm_state: Option<String>,
15    /// Current behavior tree status, if any.
16    pub bt_status: Option<String>,
17    /// Named utility scores.
18    pub utility_scores: HashMap<String, f64>,
19}
20
21/// A condition that guards whether a dialog branch is available.
22#[derive(Debug, Clone)]
23pub enum DialogueCondition {
24    /// Branch requires this FSM state.
25    FsmState(String),
26    /// Branch requires this BT status.
27    BtStatus(String),
28    /// Branch requires utility score above threshold.
29    UtilityAbove { key: String, threshold: f64 },
30    /// Branch requires utility score below threshold.
31    UtilityBelow { key: String, threshold: f64 },
32    /// All conditions must pass.
33    All(Vec<DialogueCondition>),
34    /// At least one condition must pass.
35    Any(Vec<DialogueCondition>),
36    /// Always passes (no guard).
37    Always,
38}
39
40impl DialogueCondition {
41    /// Evaluate this condition against the provided context.
42    pub fn evaluate(&self, ctx: &GateContext) -> bool {
43        match self {
44            Self::FsmState(s) => ctx.fsm_state.as_deref() == Some(s.as_str()),
45            Self::BtStatus(s) => ctx.bt_status.as_deref() == Some(s.as_str()),
46            Self::UtilityAbove { key, threshold } => {
47                ctx.utility_scores.get(key).copied().unwrap_or(0.0) > *threshold
48            }
49            Self::UtilityBelow { key, threshold } => {
50                ctx.utility_scores.get(key).copied().unwrap_or(0.0) < *threshold
51            }
52            Self::All(conditions) => conditions.iter().all(|c| c.evaluate(ctx)),
53            Self::Any(conditions) => conditions.iter().any(|c| c.evaluate(ctx)),
54            Self::Always => true,
55        }
56    }
57}
