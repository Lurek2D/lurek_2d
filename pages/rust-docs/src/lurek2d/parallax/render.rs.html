1//! Convert parallax layer state into flat `RenderCommand` lists for the renderer.
2//!
3//! - Tile position batches into draw-image sequences with color and blend pre-applied.
4//! - Bridge between the parallax camera math and the GPU submission pipeline.
5
6use crate::parallax::layer::{ParallaxDrawBatch, ParallaxLayer};
7use crate::render::renderer::RenderCommand;
8
9/// Render-command generation for a single parallax layer.
10impl ParallaxLayer {
11    /// Generate a `Vec<RenderCommand>` for this layer at the given camera position and screen size.
12    pub fn generate_render_commands(
13        &self,
14        cam_x: f32,
15        cam_y: f32,
16        screen_w: f32,
17        screen_h: f32,
18    ) -> Vec<RenderCommand> {
19        let batch = match self.build_draw_calls(cam_x, cam_y, screen_w, screen_h) {
20            Some(b) => b,
21            None => return Vec::new(),
22        };
23        batch_to_render_commands(&batch)
24    }
25}
26/// Convert a `ParallaxDrawBatch` into a flat list of `RenderCommand` values ready for submission.
27pub fn batch_to_render_commands(batch: &ParallaxDrawBatch) -> Vec<RenderCommand> {
28    let [r, g, b, a] = batch.color;
29    let mut cmds = Vec::with_capacity(2 + batch.tiles.len());
30    cmds.push(RenderCommand::SetColor(r, g, b, a));
31    cmds.push(RenderCommand::SetBlendMode(batch.blend_mode));
32    for &(x, y) in &batch.tiles {
33        cmds.push(RenderCommand::DrawImageEx {
34            texture_key: batch.texture_key,
35            x,
36            y,
37            rotation: 0.0,
38            sx: batch.sx,
39            sy: batch.sy,
40            ox: 0.0,
41            oy: 0.0,
42            effect: batch.effect.clone(),
43        });
44    }
45    cmds
46}
