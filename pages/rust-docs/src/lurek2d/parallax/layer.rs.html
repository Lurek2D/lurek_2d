1//! Single parallax layer definition with scroll factor, autoscroll, tiling, opacity, and tint.
2//!
3//! - Draw-batch struct that collects tile positions and render state for submission.
4//! - Camera-relative pixel offset computation with optional scroll clamping.
5//! - Tile repetition logic delegated to `tile_iter` for viewport coverage.
6//! - Motion-stretch blur effect injection based on autoscroll velocity.
7//! - Shader effect chain management (set, clear, count) per layer.
8
9use crate::render::BlendMode;
10use crate::render::ShaderPassDescriptor;
11use crate::runtime::resource_keys::TextureKey;
12use std::collections::HashMap;
13/// Minimum tile edge size in pixels; prevents degenerate zero-area tile geometry.
14const MIN_TILE_SIZE: f32 = 16.0;
15/// Accumulated draw data for one parallax layer ready to submit to the renderer.
16pub struct ParallaxDrawBatch {
17    /// Texture to draw for every tile position.
18    pub texture_key: TextureKey,
19    /// World-space `(x, y)` positions for each tile instance to draw.
20    pub tiles: Vec<(f32, f32)>,
21    /// Horizontal scale factor applied to the texture at draw time.
22    pub sx: f32,
23    /// Vertical scale factor applied to the texture at draw time.
24    pub sy: f32,
25    /// RGBA tint colour with pre-multiplied alpha.
26    pub color: [f32; 4],
27    /// Blend mode to use for every tile in this batch.
28    pub blend_mode: BlendMode,
29    /// Optional post-process shader chain applied after the batch is drawn.
30    pub effect: Option<Vec<ShaderPassDescriptor>>,
31}
32/// Single scrolling background layer with optional tiling, tint, autoscroll, and shader effects.
33pub struct ParallaxLayer {
34    /// Handle to the layer's source texture.
35    pub texture_key: TextureKey,
36    /// Source texture width in pixels before scaling.
37    pub texture_width: f32,
38    /// Source texture height in pixels before scaling.
39    pub texture_height: f32,
40    /// Scroll multiplier `[x, y]` relative to camera movement (0.0 = fixed, 1.0 = matches camera).
41    pub scroll_factor: [f32; 2],
42    /// Constant world-space offset `[x, y]` added after scroll calculation.
43    pub offset: [f32; 2],
44    /// Constant autoscroll velocity `[vx, vy]` in pixels per second.
45    pub autoscroll: [f32; 2],
46    /// Accumulated autoscroll distance since last reset; driven by `update`.
47    pub autoscroll_accum: [f32; 2],
48    /// Whether to tile the texture horizontally.
49    pub repeat_x: bool,
50    /// Whether to tile the texture vertically.
51    pub repeat_y: bool,
52    /// Minimum scroll clamp `[x, y]`; `None` means no minimum.
53    pub clamp_min: Option<[f32; 2]>,
54    /// Maximum scroll clamp `[x, y]`; `None` means no maximum.
55    pub clamp_max: Option<[f32; 2]>,
56    /// Integer Z sort key; lower values draw behind higher.
57    pub z: i32,
58    /// Layer opacity in `[0.0, 1.0]`; multiplied into tint alpha at draw time.
59    pub opacity: f32,
60    /// RGBA tint multiplier; `[1,1,1,1]` = no tint.
61    pub tint: [f32; 4],
62    /// Blend mode applied when drawing this layer.
63    pub blend_mode: BlendMode,
64    /// When `false` the layer is skipped in `build_draw_calls`.
65    pub visible: bool,
66    /// Per-axis scale `[sx, sy]` applied to the texture size.
67    pub scale: [f32; 2],
68    /// When `true` forces both `repeat_x` and `repeat_y`.
69    pub tiling: bool,
70    /// Override tile width; `None` uses `texture_width * scale[0]`.
71    pub tile_w: Option<f32>,
72    /// Override tile height; `None` uses `texture_height * scale[1]`.
73    pub tile_h: Option<f32>,
74    /// Depth value passed to the renderer for sorting within a Z layer.
75    pub depth: f32,
76    /// Optional shader pass chain applied to this layer.
77    pub effect_chain: Option<Vec<ShaderPassDescriptor>>,
78    /// Whether motion-stretch blur is active for this layer.
79    pub motion_stretch_enabled: bool,
80    /// Strength of the motion-stretch effect; higher values stretch more per pixel/sec.
81    pub motion_stretch_strength: f32,
82    /// Maximum scale multiplier applied by motion stretch (clamped to `>= 1.0`).
83    pub motion_stretch_max_scale: f32,
84}
85impl ParallaxLayer {
86    /// Create a new layer for `texture_key` with the given texture dimensions and sensible defaults.
87    pub fn new(texture_key: TextureKey, texture_width: f32, texture_height: f32) -> Self {
88        ParallaxLayer {
89            texture_key,
90            texture_width,
91            texture_height,
92            scroll_factor: [1.0, 0.0],
93            offset: [0.0, 0.0],
94            autoscroll: [0.0, 0.0],
95            autoscroll_accum: [0.0, 0.0],
96            repeat_x: true,
97            repeat_y: false,
98            clamp_min: None,
99            clamp_max: None,
100            z: 0,
101            opacity: 1.0,
102            tint: [1.0, 1.0, 1.0, 1.0],
103            blend_mode: BlendMode::Alpha,
104            visible: true,
105            scale: [1.0, 1.0],
106            tiling: false,
107            tile_w: None,
108            tile_h: None,
109            depth: 0.0,
110            effect_chain: None,
111            motion_stretch_enabled: false,
112            motion_stretch_strength: 0.001,
113            motion_stretch_max_scale: 2.0,
114        }
115    }
116    /// Advance the autoscroll accumulator by `dt` seconds and wrap it to one tile width/height.
117    pub fn update(&mut self, dt: f32) {
118        self.autoscroll_accum[0] += self.autoscroll[0] * dt;
119        self.autoscroll_accum[1] += self.autoscroll[1] * dt;
120        let (tw, th) = self.resolved_tile_dimensions();
121        if tw > 0.0 {
122            self.autoscroll_accum[0] = self.autoscroll_accum[0].rem_euclid(tw);
123        }
124        if th > 0.0 {
125            self.autoscroll_accum[1] = self.autoscroll_accum[1].rem_euclid(th);
126        }
127    }
128    /// Return the effective `(tile_w, tile_h)` in pixels after applying scale and `MIN_TILE_SIZE`.
129    fn resolved_tile_dimensions(&self) -> (f32, f32) {
130        let base_w = self.texture_width * self.scale[0];
131        let base_h = self.texture_height * self.scale[1];
132        let tw = self.tile_w.unwrap_or(base_w).max(MIN_TILE_SIZE);
133        let th = self.tile_h.unwrap_or(base_h).max(MIN_TILE_SIZE);
134        (tw, th)
135    }
136    /// Return the pixel offset `(px, py)` for the layer given camera position `(cam_x, cam_y)`.
137    fn compute_pixel_offset(&self, cam_x: f32, cam_y: f32) -> (f32, f32) {
138        let mut px = cam_x * self.scroll_factor[0] + self.offset[0] + self.autoscroll_accum[0];
139        let mut py = cam_y * self.scroll_factor[1] + self.offset[1] + self.autoscroll_accum[1];
140        if let (Some(mn), Some(mx)) = (self.clamp_min, self.clamp_max) {
141            px = px.clamp(mn[0], mx[0]);
142            py = py.clamp(mn[1], mx[1]);
143        }
144        (px, py)
145    }
146    /// Build a `ParallaxDrawBatch` for the current camera position; returns `None` when invisible or zero-size.
147    pub fn build_draw_calls(
148        &self,
149        cam_x: f32,
150        cam_y: f32,
151        screen_w: f32,
152        screen_h: f32,
153    ) -> Option<ParallaxDrawBatch> {
154        if !self.visible || self.opacity <= 0.0 {
155            return None;
156        }
157        let (tex_w, tex_h) = self.resolved_tile_dimensions();
158        let repeat_x = self.tiling || self.repeat_x;
159        let repeat_y = self.tiling || self.repeat_y;
160        if tex_w <= 0.0 || tex_h <= 0.0 {
161            return None;
162        }
163        let (px, py) = self.compute_pixel_offset(cam_x, cam_y);
164        let start_x = if repeat_x { -px.rem_euclid(tex_w) } else { -px };
165        let start_y = if repeat_y { -py.rem_euclid(tex_h) } else { -py };
166        let tiles = crate::parallax::tile_iter::collect_tiled_positions(
167            start_x,
168            start_y,
169            tex_w,
170            tex_h,
171            repeat_x,
172            repeat_y,
173            [screen_w, screen_h],
174        );
175        let [tr, tg, tb, ta] = self.tint;
176        let color = [tr, tg, tb, ta * self.opacity];
177        let mut sx = if self.texture_width > 0.0 {
178            tex_w / self.texture_width
179        } else {
180            self.scale[0]
181        };
182        let mut sy = if self.texture_height > 0.0 {
183            tex_h / self.texture_height
184        } else {
185            self.scale[1]
186        };
187        let mut effect = self.effect_chain.clone();
188        if self.motion_stretch_enabled {
189            let speed_x = self.autoscroll[0].abs();
190            let speed_y = self.autoscroll[1].abs();
191            let max_extra = (self.motion_stretch_max_scale - 1.0).max(0.0);
192            let sx_extra = (speed_x * self.motion_stretch_strength).min(max_extra);
193            let sy_extra = (speed_y * self.motion_stretch_strength).min(max_extra);
194            sx *= 1.0 + sx_extra;
195            sy *= 1.0 + sy_extra;
196            let speed = (speed_x * speed_x + speed_y * speed_y).sqrt();
197            if speed > 1.0 {
198                let mut params = HashMap::new();
199                params.insert("strength".to_string(), (speed * 0.001).clamp(0.0, 1.0));
200                params.insert(
201                    "direction_x".to_string(),
202                    if speed > 0.0 {
203                        self.autoscroll[0] / speed
204                    } else {
205                        0.0
206                    },
207                );
208                params.insert(
209                    "direction_y".to_string(),
210                    if speed > 0.0 {
211                        self.autoscroll[1] / speed
212                    } else {
213                        0.0
214                    },
215                );
216                let mut chain = effect.unwrap_or_default();
217                chain.push(ShaderPassDescriptor {
218                    effect_name: "motion_blur".to_string(),
219                    params,
220                    enabled: true,
221                });
222                effect = Some(chain);
223            }
224        }
225        Some(ParallaxDrawBatch {
226            texture_key: self.texture_key,
227            tiles,
228            sx,
229            sy,
230            color,
231            blend_mode: self.blend_mode,
232            effect,
233        })
234    }
235    /// Reset the autoscroll accumulator to `[0.0, 0.0]`.
236    pub fn reset_autoscroll(&mut self) {
237        self.autoscroll_accum = [0.0, 0.0];
238    }
239    /// Set whether tiling mode (forces repeat on both axes) is active.
240    pub fn set_tiling(&mut self, enabled: bool) {
241        self.tiling = enabled;
242    }
243    /// Return `true` when tiling mode is active.
244    pub fn get_tiling(&self) -> bool {
245        self.tiling
246    }
247    /// Override tile size to `(w, h)` pixels; `0.0` or negative resets to texture-derived size.
248    pub fn set_tile_size(&mut self, w: f32, h: f32) {
249        self.tile_w = if w > 0.0 {
250            Some(w.max(MIN_TILE_SIZE))
251        } else {
252            None
253        };
254        self.tile_h = if h > 0.0 {
255            Some(h.max(MIN_TILE_SIZE))
256        } else {
257            None
258        };
259    }
260    /// Set the depth sort value for this layer.
261    pub fn set_depth(&mut self, z: f32) {
262        self.depth = z;
263    }
264    /// Return the current depth sort value.
265    pub fn get_depth(&self) -> f32 {
266        self.depth
267    }
268    /// Replace the shader effect chain; an empty `chain` clears the existing chain.
269    pub fn set_effect_chain(&mut self, chain: Vec<ShaderPassDescriptor>) {
270        self.effect_chain = if chain.is_empty() { None } else { Some(chain) };
271    }
272    /// Remove all shader effects from this layer.
273    pub fn clear_effect_chain(&mut self) {
274        self.effect_chain = None;
275    }
276    /// Return the number of shader passes currently in the effect chain.
277    pub fn effect_count(&self) -> usize {
278        self.effect_chain.as_ref().map_or(0, Vec::len)
279    }
280    /// Configure motion-stretch blur; `strength` controls pixels-per-sec sensitivity, `max_scale` caps the scale boost.
281    pub fn set_motion_stretch(&mut self, enabled: bool, strength: f32, max_scale: f32) {
282        self.motion_stretch_enabled = enabled;
283        self.motion_stretch_strength = strength.max(0.0);
284        self.motion_stretch_max_scale = max_scale.max(1.0);
285    }
