1//! Bar chart renderer: vertical or horizontal grouped/stacked bars.
2//!
3//! - Rasterises a `BarChartSpec` into an RGBA pixel buffer.
4//! - Supports grouped and stacked layouts; bar width and gap are configurable.
5//! - Uses `render_utils::draw_rect_filled` for individual bar segments.
6//! - Owned by `lurek.charts.bar`; output is uploaded as a per-frame texture.
7
8use crate::charts::config::{ChartConfig, ChartSeries};
9use crate::charts::render_utils::{
10    auto_range, draw_line, draw_rect_filled, fill_buffer, world_to_screen,
11};
12
13/// A bar chart that renders series as grouped vertical bars.
14#[derive(Debug, Clone)]
15pub struct BarChart {
16    /// Chart appearance configuration.
17    pub config: ChartConfig,
18    /// Data series to plot.
19    series: Vec<ChartSeries>,
20    /// Width of each bar in pixels.
21    bar_width: f32,
22    /// Gap between bar groups in pixels.
23    gap: f32,
24}
25
26impl BarChart {
27    /// Create a new bar chart with the given configuration.
28    pub fn new(config: ChartConfig) -> Self {
29        Self {
30            config,
31            series: Vec::new(),
32            bar_width: 20.0,
33            gap: 4.0,
34        }
35    }
36
37    /// Add a data series (low-level: accepts a ChartSeries directly).
38    pub fn push_series(&mut self, series: ChartSeries) {
39        self.series.push(series);
40    }
41
42    /// Register a named series slot with a color.
43    pub fn add_series(&mut self, name: &str, color: crate::color::Color) {
44        self.push_series(ChartSeries {
45            name: name.to_string(),
46            color: [color.r, color.g, color.b, color.a],
47            data: Vec::new(),
48        });
49    }
50
51    /// Add a category with one value per registered series.
52    pub fn add_category(&mut self, _label: &str, vals: &[f32]) {
53        let cat_idx = self.series.iter().map(|s| s.data.len()).max().unwrap_or(0) as f32;
54        for (i, series) in self.series.iter_mut().enumerate() {
55            let v = vals.get(i).copied().unwrap_or(0.0);
56            series.data.push((cat_idx, v));
57        }
58    }
59
60    /// Add categories from dataframe rows, returning the number of categories added.
61    pub fn add_categories_from_dataframe(
62        &mut self,
63        df: &crate::dataframe::DataFrame,
64        label_col: &str,
65        value_cols: &[String],
66        opts: crate::charts::config::ChartDataFrameOptions,
67    ) -> Result<usize, String> {
68        use crate::dataframe::frame::ColRef;
69        let label_data = df.get_column(ColRef::Name(label_col.to_string()))?;
70        let n = label_data.len().min(opts.max_rows.unwrap_or(usize::MAX));
71        for row in 0..n {
72            let mut vals: Vec<f32> = Vec::new();
73            for vc in value_cols.iter() {
74                let col = df.get_column(ColRef::Name(vc.clone()))?;
75                vals.push(col.get(row).and_then(|c| c.as_number()).unwrap_or(0.0) as f32);
76            }
77            let label = label_data.get(row).map(|c| format!("{c}")).unwrap_or_default();
78            self.add_category(&label, &vals);
79        }
80        Ok(n)
81    }
82
83    /// Render this chart into an ImageData buffer.
84    pub fn draw_to_image(&self, img: &mut crate::image::ImageData) {
85        self.render(img.as_mut_bytes());
86    }
87
88    /// Remove all series from the chart.
89    pub fn clear(&mut self) {
90        self.series.clear();
91    }
92
93    /// Set the individual bar width in pixels.
94    pub fn set_bar_width(&mut self, width: f32) {
95        self.bar_width = width.max(1.0);
96    }
97
98    /// Render the chart into an RGBA8 pixel buffer.
99    ///
100    /// The buffer must be exactly `width * height * 4` bytes.
101    pub fn render(&self, buffer: &mut [u8]) {
102        let w = self.config.width;
103        let h = self.config.height;
104
105        fill_buffer(buffer, self.config.bg_color);
106
107        let margin = &self.config.margin;
108        let plot_x = margin.left;
109        let plot_y = margin.top;
110        let plot_w = w as f32 - margin.left - margin.right;
111        let plot_h = h as f32 - margin.top - margin.bottom;
112
113        if plot_w <= 0.0 || plot_h <= 0.0 {
114            return;
115        }
116
117        let (_, _, min_y, max_y) = auto_range(&self.series);
118        // Ensure y-axis starts at zero for bar charts when all values are positive.
119        let min_y = if min_y > 0.0 { 0.0 } else { min_y };
120
121        // Draw grid lines.
122        if self.config.show_grid {
123            let grid_lines = 5u32;
124            for i in 0..=grid_lines {
125                let frac = i as f32 / grid_lines as f32;
126                let gy = plot_y + plot_h * (1.0 - frac);
127                draw_line(
128                    buffer,
129                    w,
130                    h,
131                    plot_x,
132                    gy,
133                    plot_x + plot_w,
134                    gy,
135                    self.config.grid_color,
136                );
137            }
138        }
139
140        // Draw axes.
141        let baseline_y = plot_y + plot_h - world_to_screen(0.0_f32.max(min_y), min_y, max_y, plot_h);
142        draw_line(
143            buffer,
144            w,
145            h,
146            plot_x,
147            plot_y + plot_h,
148            plot_x + plot_w,
149            plot_y + plot_h,
150            self.config.axis_color,
151        );
152        draw_line(
153            buffer,
154            w,
155            h,
156            plot_x,
157            plot_y,
158            plot_x,
159            plot_y + plot_h,
160            self.config.axis_color,
161        );
162
163        // Determine total number of data points across all series.
164        let max_points = self.series.iter().map(|s| s.data.len()).max().unwrap_or(0);
165        if max_points == 0 {
166            return;
167        }
168
169        let num_series = self.series.len() as f32;
170        let group_width = num_series * self.bar_width + (num_series - 1.0).max(0.0) * self.gap;
171        let step = plot_w / max_points as f32;
172
173        for (si, s) in self.series.iter().enumerate() {
174            for (di, &(_x, y)) in s.data.iter().enumerate() {
175                let group_center = plot_x + step * (di as f32 + 0.5);
176                let bar_x = group_center - group_width * 0.5
177                    + si as f32 * (self.bar_width + self.gap);
178
179                let bar_top =
180                    plot_y + plot_h - world_to_screen(y, min_y, max_y, plot_h);
181                let bar_bottom = baseline_y;
182
183                let (draw_y, draw_h) = if bar_top < bar_bottom {
184                    (bar_top as u32, (bar_bottom - bar_top) as u32)
185                } else {
186                    (bar_bottom as u32, (bar_top - bar_bottom) as u32)
187                };
188
189                draw_rect_filled(
190                    buffer,
191                    w,
192                    bar_x as u32,
193                    draw_y,
194                    self.bar_width as u32,
195                    draw_h,
196                    s.color,
197                );
198            }
199        }
200    }
