1//! Line chart renderer: connected data-point series over time or categories.
2//!
3//! - Rasterises a `LineChartSpec` into an RGBA pixel buffer.
4//! - Supports multiple named series with per-series colour and line width.
5//! - Pixel coordinates are mapped via `render_utils::world_to_screen`.
6//! - Owned by `lurek.charts.line`; result is uploaded as a texture each frame.
7
8use crate::charts::config::{ChartConfig, ChartSeries};
9use crate::charts::render_utils::{
10    auto_range, draw_circle_filled, draw_line, fill_buffer, world_to_screen,
11};
12
13/// A line chart that renders one or more series as connected polylines.
14#[derive(Debug, Clone)]
15pub struct LineChart {
16    /// Chart appearance configuration.
17    pub config: ChartConfig,
18    /// Data series to plot.
19    series: Vec<ChartSeries>,
20    /// Optional Y-axis maximum override.
21    pub y_max: f32,
22    /// Optional X-axis maximum override.
23    pub x_max: f32,
24}
25
26impl LineChart {
27    /// Create a new line chart with the given configuration.
28    pub fn new(config: ChartConfig) -> Self {
29        Self {
30            config,
31            series: Vec::new(),
32            y_max: 0.0,
33            x_max: 0.0,
34        }
35    }
36
37    /// Add a data series (low-level: accepts a ChartSeries directly).
38    pub fn push_series(&mut self, series: ChartSeries) {
39        self.series.push(series);
40    }
41
42    /// Add a named series from point data and a color.
43    pub fn add_series(&mut self, name: &str, pts: &[(f32, f32)], color: crate::color::Color) {
44        self.push_series(ChartSeries {
45            name: name.to_string(),
46            color: [color.r, color.g, color.b, color.a],
47            data: pts.to_vec(),
48        });
49    }
50
51    /// Add a named series from dataframe columns, returning the number of points added.
52    pub fn add_series_from_dataframe(
53        &mut self,
54        name: &str,
55        df: &crate::dataframe::DataFrame,
56        x_col: &str,
57        y_col: &str,
58        color: crate::color::Color,
59        opts: crate::charts::config::ChartDataFrameOptions,
60    ) -> Result<usize, String> {
61        use crate::dataframe::frame::ColRef;
62        let xs = df.get_column(ColRef::Name(x_col.to_string()))?;
63        let ys = df.get_column(ColRef::Name(y_col.to_string()))?;
64        let limit = opts.max_rows.unwrap_or(usize::MAX);
65        let mut pts: Vec<(f32, f32)> = Vec::new();
66        for (xc, yc) in xs.iter().zip(ys.iter()).take(limit) {
67            if let (Some(x), Some(y)) = (xc.as_number(), yc.as_number()) {
68                pts.push((x as f32, y as f32));
69            }
70        }
71        let added = pts.len();
72        self.push_series(ChartSeries {
73            name: name.to_string(),
74            color: [color.r, color.g, color.b, color.a],
75            data: pts,
76        });
77        Ok(added)
78    }
79
80    /// Render this chart into an ImageData buffer.
81    pub fn draw_to_image(&self, img: &mut crate::image::ImageData) {
82        self.render(img.as_mut_bytes());
83    }
84
85    /// Remove all series from the chart.
86    pub fn clear(&mut self) {
87        self.series.clear();
88    }
89
90    /// Render the chart into an RGBA8 pixel buffer.
91    ///
92    /// The buffer must be exactly `width * height * 4` bytes.
93    pub fn render(&self, buffer: &mut [u8]) {
94        let w = self.config.width;
95        let h = self.config.height;
96
97        fill_buffer(buffer, self.config.bg_color);
98
99        let margin = &self.config.margin;
100        let plot_x = margin.left;
101        let plot_y = margin.top;
102        let plot_w = w as f32 - margin.left - margin.right;
103        let plot_h = h as f32 - margin.top - margin.bottom;
104
105        if plot_w <= 0.0 || plot_h <= 0.0 {
106            return;
107        }
108
109        let (min_x, max_x, min_y, max_y) = auto_range(&self.series);
110
111        // Draw grid lines.
112        if self.config.show_grid {
113            let grid_lines = 5u32;
114            for i in 0..=grid_lines {
115                let frac = i as f32 / grid_lines as f32;
116                // Horizontal grid line.
117                let gy = plot_y + plot_h * (1.0 - frac);
118                draw_line(
119                    buffer,
120                    w,
121                    h,
122                    plot_x,
123                    gy,
124                    plot_x + plot_w,
125                    gy,
126                    self.config.grid_color,
127                );
128                // Vertical grid line.
129                let gx = plot_x + plot_w * frac;
130                draw_line(
131                    buffer,
132                    w,
133                    h,
134                    gx,
135                    plot_y,
136                    gx,
137                    plot_y + plot_h,
138                    self.config.grid_color,
139                );
140            }
141        }
142
143        // Draw axes.
144        draw_line(
145            buffer,
146            w,
147            h,
148            plot_x,
149            plot_y + plot_h,
150            plot_x + plot_w,
151            plot_y + plot_h,
152            self.config.axis_color,
153        );
154        draw_line(
155            buffer,
156            w,
157            h,
158            plot_x,
159            plot_y,
160            plot_x,
161            plot_y + plot_h,
162            self.config.axis_color,
163        );
164
165        // Draw each series.
166        for s in &self.series {
167            if s.data.is_empty() {
168                continue;
169            }
170
171            let to_screen = |x: f32, y: f32| -> (f32, f32) {
172                let sx = plot_x + world_to_screen(x, min_x, max_x, plot_w);
173                let sy = plot_y + plot_h - world_to_screen(y, min_y, max_y, plot_h);
174                (sx, sy)
175            };
176
177            // Draw connecting lines.
178            for pair in s.data.windows(2) {
179                let (sx0, sy0) = to_screen(pair[0].0, pair[0].1);
180                let (sx1, sy1) = to_screen(pair[1].0, pair[1].1);
181                draw_line(buffer, w, h, sx0, sy0, sx1, sy1, s.color);
182            }
183
184            // Draw dots at data points.
185            for &(x, y) in &s.data {
186                let (sx, sy) = to_screen(x, y);
187                draw_circle_filled(buffer, w, h, sx, sy, 3.0, s.color);
188            }
189        }
190    }
191}
