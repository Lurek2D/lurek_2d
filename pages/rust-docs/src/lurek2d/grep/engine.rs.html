1//! High-level search engine: wires configuration, file filter, and pattern matcher.
2//!
3//! - `GrepEngine::run(root, pattern)` returns a `GrepResult` across all matching files.
4//! - Delegates file discovery to `FileFilter` and matching to `Matcher`.
5//! - Work is split across a Rayon thread pool sized from `GrepConfig::thread_count`.
6//! - Used by `lurek.grep.*` Lua API; the Lua binding owns the config lifecycle.
7use super::config::GrepConfig;
8use super::filter::FileFilter;
9use super::matcher::Matcher;
10use super::parallel::ParallelSearch;
11use super::pattern::PatternKind;
12use super::result::SearchResult;
13use std::path::{Path, PathBuf};
14
15/// Main search engine combining configuration, file filter, and pattern matcher.
16pub struct GrepEngine {
17    config: GrepConfig,
18    parallel: ParallelSearch,
19}
20
21impl GrepEngine {
22    /// Create a new `GrepEngine` using the provided configuration.
23    pub fn new(config: GrepConfig) -> Self {
24        let parallel = ParallelSearch::new(config.thread_count, config.max_file_size);
25        Self { config, parallel }
26    }
27
28    /// Search a directory with a literal pattern.
29    pub fn search_literal(&self, root: &Path, pattern: &str, filter: &FileFilter) -> SearchResult {
30        let matcher = Matcher::new(
31            PatternKind::literal(pattern),
32            self.config.case_sensitive,
33            self.config.whole_word,
34        );
35        self.parallel.search(root, &matcher, filter)
36    }
37
38    /// Search a directory with a regex pattern.
39    pub fn search_regex(&self, root: &Path, pattern: &str, filter: &FileFilter) -> SearchResult {
40        let matcher = Matcher::new(
41            PatternKind::regex(pattern),
42            self.config.case_sensitive,
43            self.config.whole_word,
44        );
45        self.parallel.search(root, &matcher, filter)
46    }
47
48    /// Search a directory with multiple literal patterns.
49    pub fn search_multi(&self, root: &Path, patterns: Vec<String>, filter: &FileFilter) -> SearchResult {
50        let matcher = Matcher::new(
51            PatternKind::multi_literal(patterns),
52            self.config.case_sensitive,
53            self.config.whole_word,
54        );
55        self.parallel.search(root, &matcher, filter)
56    }
57
58    /// Search a specific list of files for matches.
59    pub fn search_files(&self, files: &[PathBuf], pattern: &str) -> SearchResult {
60        let matcher = Matcher::new(
61            PatternKind::literal(pattern),
62            self.config.case_sensitive,
63            self.config.whole_word,
64        );
65        self.parallel.search_files(files, &matcher)
66    }
67
68    /// Count matches without collecting line details.
69    pub fn count(&self, root: &Path, pattern: &str, filter: &FileFilter) -> usize {
70        let result = self.search_literal(root, pattern, filter);
71        result.total_matches
72    }
73
74    /// Return a shared reference to this engine's configuration.
75    pub fn config(&self) -> &GrepConfig {
76        &self.config
77    }
78}
79
80impl Default for GrepEngine {
81    fn default() -> Self {
82        Self::new(GrepConfig::default())
83    }
84}
