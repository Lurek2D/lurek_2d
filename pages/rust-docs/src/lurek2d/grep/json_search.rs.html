1//! JSON path search: query structured key-value paths within JSON files.
2//!
3//! - `search_json_path` scans a directory for JSON files and extracts values at a path.
4//! - `search_json_file` operates on a single file; returns `Option<serde_json::Value>`.
5//! - Path syntax uses `/`-separated keys; arrays are addressed by numeric index.
6//! - Exposed to Lua via `lurek.grep.json_path(dir, path)` in `grep_api.rs`.
7use std::path::Path;
8
9/// Result of a JSON key-path search.
10#[derive(Debug, Clone)]
11pub struct JsonMatch {
12    /// File system path.
13    pub path: String,
14    /// Matched value at the JSON path.
15    pub value: String,
16    /// JSON key path expression to match.
17    pub json_path: String,
18}
19
20/// Search JSON files for values at specific paths.
21pub fn search_json_path(content: &str, query_path: &str, value_pattern: Option<&str>) -> Vec<JsonMatch> {
22    let mut results = Vec::new();
23
24    // Simple JSON key-value search (line-based heuristic for TOML/JSON without serde)
25    for line in content.lines() {
26        let trimmed = line.trim();
27        if trimmed.contains(query_path) {
28            let value = extract_json_value(trimmed);
29            let matches_value = match value_pattern {
30                Some(pat) => value.contains(pat),
31                None => true,
32            };
33            if matches_value {
34                results.push(JsonMatch {
35                    path: query_path.to_string(),
36                    value,
37                    json_path: query_path.to_string(),
38                });
39            }
40        }
41    }
42
43    results
44}
45
46/// Search a JSON file for a key path.
47pub fn search_json_file(path: &Path, query_path: &str, value_pattern: Option<&str>) -> Vec<JsonMatch> {
48    match std::fs::read_to_string(path) {
49        Ok(content) => search_json_path(&content, query_path, value_pattern),
50        Err(_) => Vec::new(),
51    }
52}
53
54fn extract_json_value(line: &str) -> String {
55    // Extract value after : or =
56    if let Some(pos) = line.find(':') {
57        line[pos + 1..].trim().trim_matches(|c| c == '"' || c == ',' || c == '}').to_string()
58    } else if let Some(pos) = line.find('=') {
59        line[pos + 1..].trim().trim_matches('"').to_string()
60    } else {
61        line.to_string()
62    }
63}
