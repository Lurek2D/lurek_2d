1//! Structured log file search with level, time-range, and text pattern filters.
2//!
3//! - `parse_log_lines` parses lines of the form `[LEVEL TIMESTAMP] MESSAGE`.
4//! - `search_logs` filters `Vec<LogEntry>` by level, time bounds, and text pattern.
5//! - `LogSearchOpts` drives the filter; all fields are optional (zero = no filter).
6//! - Used by `lurek.grep.logs` to let game scripts query the engine's runtime log.
7/// A single parsed log entry with line number, timestamp, level, and message.
8#[derive(Debug, Clone)]
9pub struct LogEntry {
10    /// Line number (1-based) in the source file.
11    pub line_number: usize,
12    /// Log entry timestamp string.
13    pub timestamp: Option<String>,
14    /// Log entry severity level.
15    pub level: Option<String>,
16    /// Log message text.
17    pub message: String,
18}
19
20/// Options for filtering log search by level, time range, and text pattern.
21#[derive(Debug, Clone)]
22pub struct LogSearchOpts {
23    /// Minimum severity level to include in results.
24    pub level_filter: Option<String>,
25    /// Include only log entries after this timestamp.
26    pub time_after: Option<String>,
27    /// Include only log entries before this timestamp.
28    pub time_before: Option<String>,
29    /// Search pattern string.
30    pub pattern: Option<String>,
31    /// Number of context lines around each match.
32    pub context_lines: usize,
33}
34
35impl LogSearchOpts {
36    /// Create `LogSearchOpts` with no filters and zero context lines.
37    pub fn new() -> Self {
38        Self {
39            level_filter: None,
40            time_after: None,
41            time_before: None,
42            pattern: None,
43            context_lines: 0,
44        }
45    }
46}
47
48impl Default for LogSearchOpts {
49    fn default() -> Self {
50        Self::new()
51    }
52}

54/// Parse log lines into structured entries.
55pub fn parse_log_lines(lines: &[String]) -> Vec<LogEntry> {
56    let mut entries = Vec::new();
57
58    for (idx, line) in lines.iter().enumerate() {
59        let trimmed = line.trim();
60        if trimmed.is_empty() {
61            continue;
62        }
63
64        let (timestamp, level, message) = parse_log_line(trimmed);
65        entries.push(LogEntry {
66            line_number: idx + 1,
67            timestamp,
68            level,
69            message,
70        });
71    }
72
73    entries
74}

76/// Search log entries with filters.
77pub fn search_logs<'a>(entries: &'a [LogEntry], opts: &LogSearchOpts) -> Vec<&'a LogEntry> {
78    entries.iter().filter(|entry| {
79        // Level filter
80        if let Some(ref level) = opts.level_filter {
81            match &entry.level {
82                Some(el) => {
83                    if !el.eq_ignore_ascii_case(level) {
84                        return false;
85                    }
86                }
87                None => return false,
88            }
89        }
90
91        // Pattern filter
92        if let Some(ref pat) = opts.pattern {
93            if !entry.message.to_lowercase().contains(&pat.to_lowercase()) {
94                return false;
95            }
96        }
97
98        // Time filters (string comparison - assumes ISO-like format)
99        if let Some(ref after) = opts.time_after {
100            if let Some(ref ts) = entry.timestamp {
101                if ts.as_str() < after.as_str() {
102                    return false;
103                }
104            }
105        }
106        if let Some(ref before) = opts.time_before {
107            if let Some(ref ts) = entry.timestamp {
108                if ts.as_str() > before.as_str() {
109                    return false;
110                }
111            }
112        }
113
114        true
115    }).collect()
116}

118/// Parse a single log line into (timestamp, level, message).
119fn parse_log_line(line: &str) -> (Option<String>, Option<String>, String) {
120    // Try common formats: [TIMESTAMP] [LEVEL] message
121    // or: TIMESTAMP LEVEL message
122    // or: [LEVEL] message
123
124    let trimmed = line.trim();
125
126    // Format: [2024-01-01 12:00:00] [INFO] message
127    if trimmed.starts_with('[') {
128        let mut parts = Vec::new();
129        let mut rest = trimmed;
130        while rest.starts_with('[') {
131            if let Some(end) = rest.find(']') {
132                parts.push(&rest[1..end]);
133                rest = rest[end + 1..].trim_start();
134            } else {
135                break;
136            }
137        }
138
139        return match parts.len() {
140            0 => (None, None, trimmed.to_string()),
141            1 => {
142                let p = parts[0];
143                if is_log_level(p) {
144                    (None, Some(p.to_uppercase()), rest.to_string())
145                } else {
146                    (Some(p.to_string()), None, rest.to_string())
147                }
148            }
149            _ => {
150                let ts = parts[0].to_string();
151                let lvl = parts[1].to_uppercase();
152                (Some(ts), Some(lvl), rest.to_string())
153            }
154        };
155    }
156
157    // Format: 2024-01-01T12:00:00 INFO message
158    if let Some(space_pos) = trimmed.find(' ') {
159        let first = &trimmed[..space_pos];
160        if first.len() >= 10 && (first.contains('-') || first.contains('T')) {
161            let rest = &trimmed[space_pos + 1..];
162            if let Some(space2) = rest.find(' ') {
163                let level_candidate = &rest[..space2];
164                if is_log_level(level_candidate) {
165                    return (
166                        Some(first.to_string()),
167                        Some(level_candidate.to_uppercase()),
168                        rest[space2 + 1..].to_string(),
169                    );
170                }
171            }
172            return (Some(first.to_string()), None, rest.to_string());
173        }
174    }
175
176    (None, None, trimmed.to_string())
177}
178
179fn is_log_level(s: &str) -> bool {
180    matches!(
181        s.to_uppercase().as_str(),
182        "TRACE" | "DEBUG" | "INFO" | "WARN" | "WARNING" | "ERROR" | "FATAL" | "CRITICAL"
183    )
