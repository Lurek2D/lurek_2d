1//! Template string interpolation with `{name}` placeholder syntax.
2//!
3//! - Supports escaping via `{{` / `}}` and preserves unknown placeholders.
4//! - Provides both map-based and pair-based variable substitution.
5
6use std::collections::HashMap;
7/// Replace `{name}` placeholders from a string map and keep unknown placeholders intact.
8pub fn interpolate(template: &str, vars: &HashMap<String, String>) -> String {
9    let mut out = String::with_capacity(template.len() + 16);
10    let mut chars = template.chars().peekable();
11    while let Some(ch) = chars.next() {
12        if ch == '{' {
13            if chars.peek() == Some(&'{') {
14                chars.next();
15                out.push('{');
16            } else {
17                let mut key = String::new();
18                let mut closed = false;
19                for inner in chars.by_ref() {
20                    if inner == '}' {
21                        closed = true;
22                        break;
23                    }
24                    key.push(inner);
25                }
26                if closed {
27                    if let Some(val) = vars.get(&key) {
28                        out.push_str(val);
29                    } else {
30                        out.push('{');
31                        out.push_str(&key);
32                        out.push('}');
33                    }
34                } else {
35                    out.push('{');
36                    out.push_str(&key);
37                }
38            }
39        } else if ch == '}' && chars.peek() == Some(&'}') {
40            chars.next();
41            out.push('}');
42        } else {
43            out.push(ch);
44        }
45    }
46    out
47}
48/// Build a variable map from key-value pairs and interpolate the template.
49pub fn interpolate_pairs(template: &str, pairs: &[(String, String)]) -> String {
50    let vars: HashMap<String, String> = pairs.iter().cloned().collect();
51    interpolate(template, &vars)
52}
