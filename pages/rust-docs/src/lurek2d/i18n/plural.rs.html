1//! Plural-form enum covering CLDR categories (zero, one, two, few, many, other).
2//!
3//! - Language-specific selection: English (singular/other) and Slavic (mod-10/mod-100 rules).
4//! - Key round-trip: convert between enum and string keys for translation maps.
5//! - Pluralize helpers that pick the best form from a HashMap and fall back gracefully.
6
7/// Plural categories used for translation lookup.
8#[derive(Debug, Clone, PartialEq, Eq, Hash)]
9pub enum PluralForm {
10    /// Quantity is exactly zero.
11    Zero,
12    /// Singular form for one item.
13    One,
14    /// Dual form for languages that distinguish two items.
15    Two,
16    /// Few form for Slavic-style plural rules.
17    Few,
18    /// Many form for large or exceptional quantities.
19    Many,
20    /// Fallback plural form.
21    Other,
22}
23/// Provide string keys and language-specific selection logic for `PluralForm`.
24impl PluralForm {
25    /// Return the lookup key for this form.
26    pub fn key(&self) -> &'static str {
27        match self {
28            Self::Zero => "zero",
29            Self::One => "one",
30            Self::Two => "two",
31            Self::Few => "few",
32            Self::Many => "many",
33            Self::Other => "other",
34        }
35    }
36    /// Select the English plural form for a numeric value.
37    pub fn english(n: f64) -> Self {
38        if n == 1.0 {
39            Self::One
40        } else {
41            Self::Other
42        }
43    }
44    /// Select the Slavic plural form for a numeric value.
45    pub fn slavic(n: u64) -> Self {
46        let abs = n % 100;
47        if (11..=19).contains(&abs) {
48            return Self::Many;
49        }
50        match n % 10 {
51            1 => Self::One,
52            2..=4 => Self::Few,
53            _ => Self::Many,
54        }
55    }
56    /// Parse a plural-form key into a `PluralForm` value.
57    pub fn from_key(s: &str) -> Option<Self> {
58        match s {
59            "zero" => Some(Self::Zero),
60            "one" => Some(Self::One),
61            "two" => Some(Self::Two),
62            "few" => Some(Self::Few),
63            "many" => Some(Self::Many),
64            "other" => Some(Self::Other),
65            _ => None,
66        }
67    }
68}
69/// Select a pluralized string using English plural rules and fall back to `other` or any value.
70pub fn pluralize(n: f64, forms: &std::collections::HashMap<String, String>) -> String {
71    let form = PluralForm::english(n);
72    if let Some(v) = forms.get(form.key()) {
73        return v.clone();
74    }
75    if let Some(v) = forms.get("other") {
76        return v.clone();
77    }
78    forms.values().next().cloned().unwrap_or_default()
79}
80/// Select a pluralized string using Slavic rules and fall back to English plural selection.
81pub fn pluralize_slavic(n: u64, forms: &std::collections::HashMap<String, String>) -> String {
82    let form = PluralForm::slavic(n);
83    if let Some(v) = forms.get(form.key()) {
84        return v.clone();
85    }
86    pluralize(n as f64, forms)
87}
