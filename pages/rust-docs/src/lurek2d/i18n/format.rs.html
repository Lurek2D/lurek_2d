1//! Locale-aware number formatting with configurable decimal and thousands separators.
2//!
3//! - Date formatting from Unix timestamps with ISO, long, and short patterns.

7/// Return decimal and thousands separators for a locale prefix.
8pub fn locale_separators(locale: &str) -> (char, char) {
9    const COMMA_DECIMAL: &[&str] = &[
10        "de", "fr", "es", "it", "pt", "nl", "pl", "ru", "tr", "sv", "da", "fi", "nb", "cs", "hu",
11        "ro", "hr", "sk", "uk", "bg",
12    ];
13    let prefix = locale.split(['-', '_']).next().unwrap_or(locale);
14    if COMMA_DECIMAL.contains(&prefix) {
15        (',', '.')
16    } else {
17        ('.', ',')
18    }
19}
20/// Format a number with explicit separators and fixed decimal places.
21pub fn format_number(n: f64, decimals: usize, decimal_sep: char, thousands_sep: char) -> String {
22    let factor = 10_f64.powi(decimals as i32);
23    let rounded = (n * factor).round() / factor;
24    let integer_part = rounded.abs().trunc() as i64;
25    let frac_scaled = ((rounded.abs() - integer_part as f64) * factor).round() as u64;
26    let int_str = integer_part.to_string();
27    let mut int_grouped = String::new();
28    for (i, ch) in int_str.chars().rev().enumerate() {
29        if i > 0 && i % 3 == 0 {
30            int_grouped.push(thousands_sep);
31        }
32        int_grouped.push(ch);
33    }
34    let int_grouped: String = int_grouped.chars().rev().collect();
35    let sign = if n < 0.0 && !(integer_part == 0 && frac_scaled == 0) {
36        "-"
37    } else {
38        ""
39    };
40    if decimals == 0 {
41        format!("{}{}", sign, int_grouped)
42    } else {
43        format!(
44            "{}{}{}{:0>width$}",
45            sign,
46            int_grouped,
47            decimal_sep,
48            frac_scaled,
49            width = decimals
50        )
51    }
52}
53/// Format a Unix timestamp using locale-aware date patterns.
54pub fn format_date(timestamp: i64, fmt: &str, locale: &str) -> String {
55    let days_total = timestamp.div_euclid(86_400);
56    let (year, month, day) = days_to_ymd(days_total);
57    let prefix = locale.split(['-', '_']).next().unwrap_or(locale);
58    let (month_names_long, month_names_short) = month_name_tables();
59    match fmt {
60        "iso" => format!("{:04}-{:02}-{:02}", year, month, day),
61        "long" => {
62            let mname = month_names_long[(month - 1) as usize];
63            match prefix {
64                "ja" | "ko" | "zh" => format!("{}年{}月{}日", year, month, day),
65                "de" | "fr" | "es" | "it" | "pt" | "nl" | "pl" | "ru" | "sv" | "da" | "fi"
66                | "nb" | "cs" | "hu" | "ro" | "hr" | "sk" | "uk" | "bg" => {
67                    format!("{} {} {}", day, mname, year)
68                }
69                _ => format!("{} {}, {}", mname, day, year),
70            }
71        }
72        _ => {
73            let mname_s = month_names_short[(month - 1) as usize];
74            match prefix {
75                "ja" | "ko" | "zh" => format!("{}/{}/{}", year, month, day),
76                "de" | "fr" | "es" | "it" | "pt" | "nl" | "pl" | "ru" | "sv" | "da" | "fi"
77                | "nb" | "cs" | "hu" | "ro" | "hr" | "sk" | "uk" | "bg" => {
78                    format!("{} {} {}", day, mname_s, year)
79                }
80                _ => format!("{} {}, {}", mname_s, day, year),
81            }
82        }
83    }
84}
85/// Convert a day count since the Unix epoch to year, month, and day.
86pub fn days_to_ymd(days: i64) -> (i32, u32, u32) {
87    let z = days + 719_468;
88    let era = (if z >= 0 { z } else { z - 146_096 }).div_euclid(146_097);
89    let doe = (z - era * 146_097) as u32;
90    let yoe = (doe - doe / 1_460 + doe / 36_524 - doe / 146_096) / 365;
91    let y = yoe as i64 + era * 400;
92    let doy = doe - (365 * yoe + yoe / 4 - yoe / 100);
93    let mp = (5 * doy + 2) / 153;
94    let d = doy - (153 * mp + 2) / 5 + 1;
95    let m = if mp < 10 { mp + 3 } else { mp - 9 };
96    let y = if m <= 2 { y + 1 } else { y };
97    (y as i32, m, d)
98}
