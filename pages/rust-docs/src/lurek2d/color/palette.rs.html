1//! Named colour palettes: retro console, web-safe, and designer presets.
2//!
3//! - `retro` sub-module provides PICO-8, Game Boy, CGA, and ZX Spectrum palettes.
4//! - Each palette is a static `&[&str]` of hex strings; no heap allocation.
5//! - Exposed to Lua via `lurek.color.palette.*`.
6//! - Palettes are additive — new sets can be registered via the Lua API.
7
8use super::color_core::Color;
9
10/// A named collection of colors.
11pub struct Palette {
12    /// Human-readable palette name.
13    pub name: &'static str,
14    /// Slice of colors in this palette.
15    pub colors: &'static [Color],
16}
17
