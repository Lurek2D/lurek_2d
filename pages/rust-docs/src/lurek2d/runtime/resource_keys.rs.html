1//! Typed slotmap keys for every engine resource pool (textures, fonts, sounds, particles, etc.).
2//!
3//! - Each key is a lightweight handle safe to store in Lua userdata and pass across frames.
4//! - Generated via `slotmap::new_key_type!` for O(1) lookup with generational validity checks.
5
6use slotmap::new_key_type;
7new_key_type! {
8    /// Key for texture storage entries.
9    pub struct TextureKey;
10    /// Key for font storage entries.
11    pub struct FontKey;
12    /// Key for render-canvas storage entries.
13    pub struct CanvasKey;
14    /// Key for sound storage entries.
15    pub struct SoundKey;
16    /// Key for particle system storage entries.
17    pub struct ParticleKey;
18    /// Key for sprite-batch storage entries.
19    pub struct SpriteBatchKey;
20    /// Key for shader storage entries.
21    pub struct ShaderKey;
22    /// Key for mesh storage entries.
23    pub struct MeshKey;
24    /// Key for compound-shape storage entries.
25    pub struct ShapeKey;
26    /// Key for audio bus storage entries.
27    pub struct BusKey;
28    /// Key for MIDI player storage entries.
29    pub struct MidiPlayerKey;
30    /// Key for queued-audio playback entries.
31    pub struct QueueableKey;
32    /// Key for light storage entries.
33    pub struct LightKey;
34    /// Key for occluder storage entries.
35    pub struct OccluderKey;
36}
