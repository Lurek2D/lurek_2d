1//! Implements the no-window headless runtime path for script automation and CI use.
2//!
3//! - `HeadlessOptions` carries game directory, eval snippets, and an optional frame-count override.
4//! - `run_headless` maps engine errors to process exit codes; `run_headless_checked` preserves structured errors for test callers.
5//! - Init sequence installs a stdout-routed `print` global and prepends game-directory roots to `package.path`.
6//! - Frame loop drives `process_physics`, `fixedUpdate`, `process`, and `process_late` in order; count and dt come from config or CLI flag.
7//! - Callback timeout is enforced via Lua instruction-count hooks when a limit is configured in `PerformanceConfig`.
8

18use std::time::{Duration, Instant};

20#[derive(Debug, Clone)]
21/// Inputs required to run the headless runtime once.
22pub struct HeadlessOptions {
23    /// Game directory used for `main.lua`, GameFS, and package path roots.
24    pub game_dir: PathBuf,
25    /// Whether the user supplied a game directory on the CLI.
26    pub explicit_game_dir: bool,
27    /// Lua snippets supplied through `--eval`.
28    pub eval: Vec<String>,
29    /// Optional frame-count override supplied through `--frames`.
30    pub frames_override: Option<u32>,
31}
32
33/// Run the headless runtime and map engine errors to process exit status.
34pub fn run_headless(config: Config, options: HeadlessOptions) -> ExitCode {
35    match run_headless_checked(config, options) {
36        Ok(()) => ExitCode::SUCCESS,
37        Err(error) => {
38            eprintln!("{}", error);
39            ExitCode::FAILURE
40        }
41    }
42}

44/// Run the headless runtime and preserve structured engine errors for tests.
45pub fn run_headless_checked(config: Config, options: HeadlessOptions) -> EngineResult<()> {
46    crate::runtime::messages::init();
47    let state = create_headless_state(&config, &options.game_dir);
48    let lua = create_headless_vm(state.clone(), &config.modules)
49        .map_err(|error| EngineError::LuaError(format!("headless VM init: {}", error)))?;
50    install_stdout_print(&lua)?;
51    install_game_package_path(&lua, &options.game_dir)?;
52    load_main_if_present(&lua, &options)?;
53    for code in &options.eval {
54        lua.load(code)
55            .set_name("--eval")
56            .exec()
57            .map_err(|error| EngineError::LuaError(format!("--eval: {}", error)))?;
58    }
59    let timeout_ms = config.performance.lua_callback_timeout_ms;
60    call_lurek_callback(&lua, "init", (), timeout_ms)?;
61    call_lurek_callback(&lua, "ready", (), timeout_ms)?;
62
63    let frames = options
64        .frames_override
65        .or(config.headless.frames)
66        .unwrap_or(0);
67    let dt = config.headless.dt.max(0.0);
68    for _ in 0..frames {
69        {
70            let mut shared = state.borrow_mut();
71            shared.delta_time = dt;
72            shared.total_time += dt;
73            shared.frame_counter = shared.frame_counter.wrapping_add(1);
74        }
75        call_lurek_callback(&lua, "process_physics", dt, timeout_ms)?;
76        call_lurek_callback(&lua, "fixedUpdate", dt, timeout_ms)?;
77        call_lurek_callback(&lua, "process", dt, timeout_ms)?;
78        call_lurek_callback(&lua, "process_late", dt, timeout_ms)?;
79    }
80    Ok(())
81}

83/// Build the initial `SharedState` for a headless run and configure physics tick rates.
84fn create_headless_state(config: &Config, game_dir: &Path) -> Rc<RefCell<SharedState>> {
85    let mut shared = SharedState::new(
86        config.window.width,
87        config.window.height,
88        &config.window.title,
89        game_dir.to_path_buf(),
90    );
91    if let Some(identity) = &config.identity {
92        shared.filesystem_identity = identity.clone();
93    }
94    shared.physics_run.fixed_dt = 1.0 / config.performance.physics_tick_rate.max(1) as f64;
95    shared.physics_run.fixed_update_dt = match config.performance.fixed_update_tick_rate {
96        Some(rate) if rate > 0 => 1.0 / rate as f64,
97        _ => 0.0,
98    };
99    shared.frame_budget_warn_ms = config.performance.frame_budget_warn_ms;
100    shared.lua_callback_timeout_ms = config.performance.lua_callback_timeout_ms;
101    shared.runtime_mode = crate::runtime::RuntimeMode::Headless;
102    Rc::new(RefCell::new(shared))
103}

105/// Replace the default Lua `print` global with a version that writes to stdout.
106fn install_stdout_print(lua: &Lua) -> EngineResult<()> {
107    let print = lua
108        .create_function(|_, values: mlua::Variadic<mlua::Value>| {
109            let parts: Vec<String> = values.iter().map(value_to_string).collect();
110            println!("{}", parts.join("\t"));
111            Ok(())
112        })
113        .map_err(|error| EngineError::LuaError(format!("headless print: {}", error)))?;
114    lua.globals()
115        .set("print", print)
116        .map_err(|error| EngineError::LuaError(format!("headless print: {}", error)))
117}

119/// Prepend the game directory to `package.path` so `require` resolves game scripts.
120fn install_game_package_path(lua: &Lua, game_dir: &Path) -> EngineResult<()> {
121    let package: LuaTable = lua
122        .globals()
123        .get("package")
124        .map_err(|error| EngineError::LuaError(format!("package.path: {}", error)))?;
125    let old_path: String = package
126        .get("path")
127        .map_err(|error| EngineError::LuaError(format!("package.path: {}", error)))?;
128    let root = game_dir.to_string_lossy().replace('\\', "/");
129    let new_path = format!(
130        "{};{}/?.lua;{}/?/init.lua;{}/content/?.lua;{}/content/?/init.lua",
131        old_path, root, root, root, root
132    );
133    package
134        .set("path", new_path)
135        .map_err(|error| EngineError::LuaError(format!("package.path: {}", error)))
136}

138/// Load and execute `main.lua` when an explicit game directory was supplied.
139fn load_main_if_present(lua: &Lua, options: &HeadlessOptions) -> EngineResult<()> {
140    if !options.explicit_game_dir {
141        return Ok(());
142    }
143    let main_lua = options.game_dir.join("main.lua");
144    if !main_lua.exists() {
145        return Ok(());
146    }
147    let code = std::fs::read_to_string(&main_lua).map_err(|error| {
148        EngineError::FileSystemError(format!("failed to read main.lua: {}", error))
149    })?;
150    lua.load(&code)
151        .set_name("main.lua")
152        .exec()
153        .map_err(|error| EngineError::LuaError(format!("main.lua: {}", error)))
154}

156/// Invoke a named `lurek.*` callback if it exists; skip silently when absent.
157fn call_lurek_callback<'lua, Args>(
158    lua: &'lua Lua,
159    name: &str,
160    args: Args,
161    timeout_ms: Option<f32>,
162) -> EngineResult<()>
163where
164    Args: IntoLuaMulti<'lua>,
165{
166    let globals = lua.globals();
167    let Ok(lurek) = globals.get::<_, LuaTable>("lurek") else {
168        return Ok(());
169    };
170    let Ok(function) = lurek.get::<_, LuaFunction>(name) else {
171        return Ok(());
172    };
173    let result = if let Some(ms) = timeout_ms.filter(|value| *value > 0.0) {
174        call_with_timeout(lua, name, function, args, ms)
175    } else {
176        function.call::<_, ()>(args)
177    };
178    result.map_err(|error| EngineError::LuaError(format!("lurek.{}: {}", name, error)))
179}

181/// Call a Lua function and abort it with an error if execution exceeds the timeout.
182fn call_with_timeout<'lua, Args>(
183    lua: &'lua Lua,
184    name: &str,
185    function: LuaFunction<'lua>,
186    args: Args,
187    timeout_ms: f32,
188) -> LuaResult<()>
189where
190    Args: IntoLuaMulti<'lua>,
191{
192    let timeout = Duration::from_secs_f64((timeout_ms as f64 / 1000.0).max(0.000_001));
193    let deadline = Instant::now() + timeout;
194    let callback_name = name.to_string();
195    lua.set_hook(
196        HookTriggers {
197            on_calls: false,
198            on_returns: false,
199            every_line: false,
200            every_nth_instruction: Some(20_000),
201        },
202        move |_, _| {
203            if Instant::now() >= deadline {
204                return Err(mlua::Error::RuntimeError(format!(
205                    "lurek.{}() exceeded callback timeout ({:.2} ms)",
206                    callback_name, timeout_ms
207                )));
208            }
209            Ok(())
210        },
211    );
212    let result = function.call::<_, ()>(args);
213    lua.remove_hook();
214    result
215}
