1//! Stable, structured log message identifiers for all engine subsystems.
2//!
3//! - Each constant provides a short code (e.g. "L001") used as prefix in log output.
4//! - Identifiers grouped by domain: L=lifecycle, A=audio, G=graphics, P=physics, FS=filesystem.
5//! - Additional prefixes: AN=animation, EN=ECS, TM=tilemap, SV=save, SC=scene, TH=thread, PF=pathfind.
6//! - Extended prefixes: MD=mods, NW=network, PL=pipeline, AT=automation, CP=compute, SR=serial, GU=GUI.
7//! - Runtime log level control via set_log_level/get_log_level with atomic override.
8//! - log_msg! macro for consistent formatted log output with message lookup.
9//! - Codes are stable across versions for log parsing, alerting, and external tool integration.
10
11use std::sync::atomic::{AtomicU8, Ordering};
12static LOG_LEVEL_OVERRIDE: AtomicU8 = AtomicU8::new(0);
13/// Set the global runtime log level from a string name (off, error, warn, info, debug, trace).
14pub fn set_log_level(level: &str) {
15    let filter = match level.to_lowercase().as_str() {
16        "off" | "none" => log::LevelFilter::Off,
17        "error" => log::LevelFilter::Error,
18        "warn" | "warning" => log::LevelFilter::Warn,
19        "info" => log::LevelFilter::Info,
20        "debug" => log::LevelFilter::Debug,
21        "trace" => log::LevelFilter::Trace,
22        _ => {
23            log::warn!(
24                "[{}] Unknown log level '{}', ignoring",
25                L022_UNKNOWN_LOG_LEVEL,
26                level
27            );
28            return;
29        }
30    };
31    log::set_max_level(filter);
32    LOG_LEVEL_OVERRIDE.store(filter as u8, Ordering::Relaxed);
33}
34/// Return the current global log level as a static string.
35pub fn get_log_level() -> &'static str {
36    match log::max_level() {
37        log::LevelFilter::Off => "off",
38        log::LevelFilter::Error => "error",
39        log::LevelFilter::Warn => "warn",
40        log::LevelFilter::Info => "info",
41        log::LevelFilter::Debug => "debug",
42        log::LevelFilter::Trace => "trace",
43    }
44}

123/// Emit a structured log message using a stable identifier and optional format arguments.
124#[macro_export]
125macro_rules! log_msg {
126    (error, $id:expr) => {
127        ::log::error!("[{}] {}", $id, $crate::runtime::messages::get_message($id))
128    };
129    (warn, $id:expr) => {
130        ::log::warn!("[{}] {}", $id, $crate::runtime::messages::get_message($id))
131    };
132    (info, $id:expr) => {
133        ::log::info!("[{}] {}", $id, $crate::runtime::messages::get_message($id))
134    };
135    (debug, $id:expr) => {
136        ::log::debug!("[{}] {}", $id, $crate::runtime::messages::get_message($id))
137    };
138    (trace, $id:expr) => {
139        ::log::trace!("[{}] {}", $id, $crate::runtime::messages::get_message($id))
140    };
141    (error, $id:expr, $($arg:tt)+) => {
142        ::log::error!("[{}] {}: {}", $id, $crate::runtime::messages::get_message($id), format_args!($($arg)+))
143    };
144    (warn, $id:expr, $($arg:tt)+) => {
145        ::log::warn!("[{}] {}: {}", $id, $crate::runtime::messages::get_message($id), format_args!($($arg)+))
146    };
147    (info, $id:expr, $($arg:tt)+) => {
148        ::log::info!("[{}] {}: {}", $id, $crate::runtime::messages::get_message($id), format_args!($($arg)+))
149    };
150    (debug, $id:expr, $($arg:tt)+) => {
151        ::log::debug!("[{}] {}: {}", $id, $crate::runtime::messages::get_message($id), format_args!($($arg)+))
152    };
153    (trace, $id:expr, $($arg:tt)+) => {
154        ::log::trace!("[{}] {}: {}", $id, $crate::runtime::messages::get_message($id), format_args!($($arg)+))
155    };
156}
