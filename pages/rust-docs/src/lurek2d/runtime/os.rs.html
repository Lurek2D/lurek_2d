1//! Operating system detection utilities for platform-specific code paths.
2//!
3//! - `get_os_name()` returns a lowercase string: `"windows"`, `"linux"`, or `"macos"`.
4//! - Used at startup to set OS-specific defaults (e.g. font paths, config directories).
5//! - Exposed to Lua via `lurek.runtime.os()` for platform-conditional game scripts.
6//! - Built on `cfg!` macros; no runtime OS probing, so the result is always correct.
7
8use crate::log_msg;
9use crate::runtime::log_messages::{LA03_OPEN_URL_REJECTED};
10
11/// Returns the current operating system name.
12pub fn get_os_name() -> &'static str {
13    if cfg!(target_os = "windows") {
14        "Windows"
15    } else if cfg!(target_os = "linux") {
16        "Linux"
17    } else if cfg!(target_os = "macos") {
18        "macOS"
19    } else if cfg!(target_os = "android") {
20        "Android"
21    } else if cfg!(target_os = "ios") {
22        "iOS"
23    } else {
24        "Unknown"
25    }
26}

28/// Returns the number of logical processors available on the host system.
29pub fn get_processor_count() -> usize {
30    std::thread::available_parallelism()
31        .map(|n| n.get())
32        .unwrap_or(1)
33}

35/// Returns total physical memory in megabytes reported by the operating system.
36pub fn get_memory_size() -> u64 {
37    use sysinfo::System;
38    let sys = System::new_with_specifics(
39        sysinfo::RefreshKind::new().with_memory(sysinfo::MemoryRefreshKind::everything()),
40    );
41    sys.total_memory() / (1024 * 1024)
42}

44/// Opens a URL in the default system browser. Only `http://`, `https://`, and `mailto:` schemes are allowed.
45pub fn open_url(url: &str) -> bool {
46    let url_lower = url.to_lowercase();
47    if !url_lower.starts_with("http://")
48        && !url_lower.starts_with("https://")
49        && !url_lower.starts_with("mailto:")
50    {
51        log_msg!(warn, LA03_OPEN_URL_REJECTED);
52        return false;
53    }
54    #[cfg(target_os = "windows")]
55    {
56        std::process::Command::new("cmd")
57            .args(["/C", "start", "", url])
58            .spawn()
59            .is_ok()
60    }
61    #[cfg(target_os = "macos")]
62    {
63        std::process::Command::new("open").arg(url).spawn().is_ok()
64    }
65    #[cfg(target_os = "linux")]
66    {
67        std::process::Command::new("xdg-open")
68            .arg(url)
69            .spawn()
70            .is_ok()
71    }
72    #[cfg(not(any(target_os = "windows", target_os = "macos", target_os = "linux")))]
73    {
74        let _ = url;
75        false
76    }
77}

79/// Returns the user's preferred locale list from the operating system, falling back to `"en_US"` if unavailable.
80pub fn get_preferred_locales() -> Vec<String> {
81    let locales: Vec<String> = sys_locale::get_locales().map(|l| l.to_string()).collect();
82    if locales.is_empty() {
83        if let Ok(lang) = std::env::var("LANG") {
84            vec![lang.split('.').next().unwrap_or("en_US").to_string()]
85        } else {
86            vec!["en_US".to_string()]
87        }
88    } else {
89        locales
90    }
91}

93/// Describes the current power supply state of the host device.
94#[derive(Debug, Clone, Copy, PartialEq)]
95pub enum PowerState {
96    Unknown,
97    Battery,
98    NoBattery,
99    Charging,
100    Charged,
101}
102
103impl PowerState {
104    /// Returns the Lua-visible power state string.
105    pub fn as_str(&self) -> &'static str {
106        match self {
107            PowerState::Unknown => "unknown",
108            PowerState::Battery => "battery",
109            PowerState::NoBattery => "nobattery",
110            PowerState::Charging => "charging",
111            PowerState::Charged => "charged",
112        }
113    }
114}
