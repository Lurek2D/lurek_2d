1//! ZIP-backed virtual filesystem mount with path-indexed entry lookup.
2//!
3//! - Reads individual files from a ZIP archive on demand without full extraction.
4//! - Normalizes virtual paths and rejects directory-traversal attempts.
5
6use std::collections::HashMap;
7use std::io::Read;
8use std::path::{Path, PathBuf};
9/// ZIP-backed virtual mount with a cached virtual-path index.
10pub struct ZipMount {
11    /// Archive file path on disk.
12    pub archive_path: PathBuf,
13    /// Virtual path prefix applied to indexed entries.
14    pub prefix: String,
15    /// Normalized virtual path to archive entry name mapping.
16    index: HashMap<String, String>,
17}
18/// Methods for building, querying, and reading files from a ZIP mount.
19impl ZipMount {
20    /// Build a ZIP mount index or return an error on archive open or parse failure.
21    pub fn new<P: AsRef<Path>>(archive_path: P, prefix: &str) -> Result<Self, String> {
22        let path = archive_path.as_ref().to_path_buf();
23        let file = std::fs::File::open(&path)
24            .map_err(|e| format!("ZipMount: cannot open '{}': {}", path.display(), e))?;
25        let archive = zip::ZipArchive::new(file)
26            .map_err(|e| format!("ZipMount: invalid ZIP '{}': {}", path.display(), e))?;
27        let clean_prefix = prefix.trim_matches('/').to_string();
28        let mut index = HashMap::new();
29        for i in 0..archive.len() {
30            let file2 = std::fs::File::open(&path)
31                .map_err(|e| format!("ZipMount: re-open failed: {}", e))?;
32            let mut arc2 = zip::ZipArchive::new(file2)
33                .map_err(|e| format!("ZipMount: re-open parse failed: {}", e))?;
34            let entry = arc2
35                .by_index(i)
36                .map_err(|e| format!("ZipMount: index {}: {}", i, e))?;
37            let entry_name = entry.name().to_string();
38            if entry.is_file() {
39                let virtual_path = if clean_prefix.is_empty() {
40                    entry_name.clone()
41                } else {
42                    format!("{}/{}", clean_prefix, entry_name)
43                };
44                index.insert(normalise(&virtual_path), entry_name);
45            }
46        }
47        Ok(Self {
48            archive_path: path,
49            prefix: clean_prefix,
50            index,
51        })
52    }
53    /// Read a virtual file from the archive or return an error when missing or invalid.
54    pub fn read_file(&self, virtual_path: &str) -> Result<Vec<u8>, String> {
55        let norm = normalise(virtual_path);
56        if is_traversal(&norm) {
57            return Err(format!(
58                "ZipMount: path traversal rejected: '{}'",
59                virtual_path
60            ));
61        }
62        let entry_name = self
63            .index
64            .get(&norm)
65            .ok_or_else(|| format!("ZipMount: file not found: '{}'", virtual_path))?
66            .clone();
67        let file = std::fs::File::open(&self.archive_path)
68            .map_err(|e| format!("ZipMount: cannot open archive: {}", e))?;
69        let mut archive =
70            zip::ZipArchive::new(file).map_err(|e| format!("ZipMount: parse error: {}", e))?;
71        let mut entry = archive
72            .by_name(&entry_name)
73            .map_err(|e| format!("ZipMount: entry '{}' not found: {}", entry_name, e))?;
74        let mut buf = Vec::with_capacity(entry.size() as usize);
75        entry
76            .read_to_end(&mut buf)
77            .map_err(|e| format!("ZipMount: read error: {}", e))?;
78        Ok(buf)
79    }
80    /// Return true when the virtual path is indexed in the archive.
81    pub fn contains(&self, virtual_path: &str) -> bool {
82        let norm = normalise(virtual_path);
83        self.index.contains_key(&norm)
84    }
85    /// Return all indexed virtual paths in sorted order.
86    pub fn list_files(&self) -> Vec<String> {
87        let mut paths: Vec<String> = self.index.keys().cloned().collect();
88        paths.sort();
89        paths
90    }
91}
92/// Normalize a path to forward slashes and remove empty segments.
93pub fn normalise(path: &str) -> String {
94    let cleaned = path.replace('\\', "/");
95    let parts: Vec<&str> = cleaned.split('/').filter(|s| !s.is_empty()).collect();
96    parts.join("/")
97}
98/// Return true when a path contains traversal or absolute-path markers.
99pub fn is_traversal(path: &str) -> bool {
100    path.contains("..") || path.starts_with('/') || (path.len() >= 2 && path.as_bytes()[1] == b':')
101}
