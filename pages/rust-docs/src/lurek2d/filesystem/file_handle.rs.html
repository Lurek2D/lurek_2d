1//! Buffered file handle abstraction for GameFS read, write, and append streams.
2//!
3//! - Mode-based state machine: Read, Write, Append, or Closed.
4//! - Resolves logical game paths through GameFS before opening OS files.
5//! - Provides line-oriented and byte-oriented read APIs with EOF detection.
6//! - Seek, tell, flush, and auto-close on drop for safe resource cleanup.
7
8use crate::filesystem::GameFS;
9use crate::runtime::error::{EngineError, EngineResult};
10use std::io::{BufRead, BufReader, BufWriter, Read, Seek, SeekFrom, Write};
11use std::path::PathBuf;
12/// File access mode used by buffered handles.
13#[derive(Debug, Clone, Copy, PartialEq)]
14pub enum FileMode {
15    /// Open for reading from the mounted filesystem.
16    Read,
17    /// Open for truncating writes into the save filesystem.
18    Write,
19    /// Open for appending writes into the save filesystem.
20    Append,
21    /// Handle is closed and no stream is attached.
22    Closed,
23}
24impl FileMode {
25    /// Parse a mode string into a file mode or error on unsupported input.
26    pub fn parse_mode(s: &str) -> EngineResult<Self> {
27        match s {
28            "r" => Ok(FileMode::Read),
29            "w" => Ok(FileMode::Write),
30            "a" => Ok(FileMode::Append),
31            _ => Err(EngineError::FileSystemError(format!(
32                "Invalid file mode '{}': expected 'r', 'w', or 'a'",
33                s
34            ))),
35        }
36    }
37    /// Return the single-letter mode string used by Lua and save data.
38    pub fn as_str(&self) -> &'static str {
39        match self {
40            FileMode::Read => "r",
41            FileMode::Write => "w",
42            FileMode::Append => "a",
43            FileMode::Closed => "c",
44        }
45    }
46}
47/// Buffered handle around a resolved game file path used by GameFS readers and writers.
48pub struct FileHandle {
49    /// Current access state for the underlying stream.
50    mode: FileMode,
51    /// Resolved host path used for the backing OS file.
52    #[allow(dead_code)]
53    path: PathBuf,
54    /// Logical path exposed to callers and error messages.
55    logical_path: String,
56    /// Buffered reader attached while the handle is open for reads.
57    reader: Option<BufReader<std::fs::File>>,
58    /// Buffered writer attached while the handle is open for writes.
59    writer: Option<BufWriter<std::fs::File>>,
60    /// File size in bytes when the handle was opened, or zero for new writes.
61    size: u64,
62}
63impl FileHandle {
64    /// Open a GameFS file handle or error on path resolution, access, or OS failure.
65    pub fn open(vfs: &GameFS, path: &str, mode: FileMode) -> EngineResult<Self> {
66        match mode {
67            FileMode::Read => {
68                let resolved = vfs.resolve_read_path(path)?;
69                let file = std::fs::File::open(&resolved).map_err(|e| {
70                    EngineError::FileSystemError(format!(
71                        "Cannot open '{}' for reading: {}",
72                        path, e
73                    ))
74                })?;
75                let size = file.metadata().map(|m| m.len()).unwrap_or(0);
76                Ok(Self {
77                    mode,
78                    path: resolved,
79                    logical_path: path.to_string(),
80                    reader: Some(BufReader::new(file)),
81                    writer: None,
82                    size,
83                })
84            }
85            FileMode::Write => {
86                let resolved = vfs.resolve_save_path(path)?;
87                if let Some(parent) = resolved.parent() {
88                    std::fs::create_dir_all(parent).map_err(|e| {
89                        EngineError::FileSystemError(format!("Failed to create directories: {}", e))
90                    })?;
91                }
92                let file = std::fs::File::create(&resolved).map_err(|e| {
93                    EngineError::FileSystemError(format!(
94                        "Cannot open '{}' for writing: {}",
95                        path, e
96                    ))
97                })?;
98                Ok(Self {
99                    mode,
100                    path: resolved,
101                    logical_path: path.to_string(),
102                    reader: None,
103                    writer: Some(BufWriter::new(file)),
104                    size: 0,
105                })
106            }
107            FileMode::Append => {
108                let resolved = vfs.resolve_save_path(path)?;
109                if let Some(parent) = resolved.parent() {
110                    std::fs::create_dir_all(parent).map_err(|e| {
111                        EngineError::FileSystemError(format!("Failed to create directories: {}", e))
112                    })?;
113                }
114                let file = std::fs::OpenOptions::new()
115                    .create(true)
116                    .append(true)
117                    .open(&resolved)
118                    .map_err(|e| {
119                        EngineError::FileSystemError(format!(
120                            "Cannot open '{}' for appending: {}",
121                            path, e
122                        ))
123                    })?;
124                let size = file.metadata().map(|m| m.len()).unwrap_or(0);
125                Ok(Self {
126                    mode,
127                    path: resolved,
128                    logical_path: path.to_string(),
129                    reader: None,
130                    writer: Some(BufWriter::new(file)),
131                    size,
132                })
133            }
134            FileMode::Closed => Err(EngineError::FileSystemError(
135                "Cannot open a file in Closed mode".to_string(),
136            )),
137        }
138    }
139    /// Read bytes from the open reader or error if the handle is not readable.
140    pub fn read(&mut self, count: Option<usize>) -> EngineResult<Vec<u8>> {
141        let reader = self
142            .reader
143            .as_mut()
144            .ok_or_else(|| EngineError::FileSystemError("File not open for reading".to_string()))?;
145        match count {
146            Some(n) => {
147                let mut buf = vec![0u8; n];
148                let bytes_read = reader
149                    .read(&mut buf)
150                    .map_err(|e| EngineError::FileSystemError(format!("Read error: {}", e)))?;
151                buf.truncate(bytes_read);
152                Ok(buf)
153            }
154            None => {
155                let mut buf = Vec::new();
156                reader
157                    .read_to_end(&mut buf)
158                    .map_err(|e| EngineError::FileSystemError(format!("Read error: {}", e)))?;
159                Ok(buf)
160            }
161        }
162    }
163    /// Read the next line without its trailing newline or return None at EOF.
164    pub fn read_line(&mut self) -> EngineResult<Option<String>> {
165        let reader = self
166            .reader
167            .as_mut()
168            .ok_or_else(|| EngineError::FileSystemError("File not open for reading".to_string()))?;
169        let mut line = String::new();
170        let bytes = reader
171            .read_line(&mut line)
172            .map_err(|e| EngineError::FileSystemError(format!("Read error: {}", e)))?;
173        if bytes == 0 {
174            Ok(None)
175        } else {
176            if line.ends_with('\n') {
177                line.pop();
178            }
179            if line.ends_with('\r') {
180                line.pop();
181            }
182            Ok(Some(line))
183        }
184    }
185    /// Write bytes to the open writer or error if the handle is not writable.
186    pub fn write(&mut self, data: &[u8]) -> EngineResult<usize> {
187        let writer = self
188            .writer
189            .as_mut()
190            .ok_or_else(|| EngineError::FileSystemError("File not open for writing".to_string()))?;
191        let written = writer
192            .write(data)
193            .map_err(|e| EngineError::FileSystemError(format!("Write error: {}", e)))?;
194        Ok(written)
195    }
196    /// Seek the active stream to an absolute byte offset or error when closed.
197    pub fn seek(&mut self, pos: u64) -> EngineResult<u64> {
198        if let Some(reader) = self.reader.as_mut() {
199            reader
200                .seek(SeekFrom::Start(pos))
201                .map_err(|e| EngineError::FileSystemError(format!("Seek error: {}", e)))
202        } else if let Some(writer) = self.writer.as_mut() {
203            writer
204                .seek(SeekFrom::Start(pos))
205                .map_err(|e| EngineError::FileSystemError(format!("Seek error: {}", e)))
206        } else {
207            Err(EngineError::FileSystemError("File is closed".to_string()))
208        }
209    }
210    /// Read the current absolute byte offset or error when closed.
211    pub fn tell(&mut self) -> EngineResult<u64> {
212        if let Some(reader) = self.reader.as_mut() {
213            reader
214                .stream_position()
215                .map_err(|e| EngineError::FileSystemError(format!("Tell error: {}", e)))
216        } else if let Some(writer) = self.writer.as_mut() {
217            writer
218                .stream_position()
219                .map_err(|e| EngineError::FileSystemError(format!("Tell error: {}", e)))
220        } else {
221            Err(EngineError::FileSystemError("File is closed".to_string()))
222        }
223    }
224    /// Return the captured file size in bytes.
225    pub fn get_size(&self) -> u64 {
226        self.size
227    }
228    /// Return the current file mode.
229    pub fn get_mode(&self) -> FileMode {
230        self.mode
231    }
232    /// Return the logical path used to open the handle.
233    pub fn get_path(&self) -> &str {
234        &self.logical_path
235    }
236    /// Flush buffered writes and return an error on write failure.
237    pub fn flush(&mut self) -> EngineResult<()> {
238        if let Some(writer) = self.writer.as_mut() {
239            writer
240                .flush()
241                .map_err(|e| EngineError::FileSystemError(format!("Flush error: {}", e)))?;
242        }
243        Ok(())
244    }
245    /// Close the handle, flush pending writes, and release both buffered streams.
246    pub fn close(&mut self) -> EngineResult<()> {
247        if self.mode == FileMode::Closed {
248            return Ok(());
249        }
250        self.flush()?;
251        self.reader = None;
252        self.writer = None;
253        self.mode = FileMode::Closed;
254        Ok(())
255    }
256    /// Check whether the reader has no buffered bytes left or error if unreadable.
257    pub fn is_eof(&mut self) -> EngineResult<bool> {
258        let reader = self
259            .reader
260            .as_mut()
261            .ok_or_else(|| EngineError::FileSystemError("File not open for reading".to_string()))?;
262        let buf = reader
263            .fill_buf()
264            .map_err(|e| EngineError::FileSystemError(format!("EOF check error: {}", e)))?;
265        Ok(buf.is_empty())
266    }
267}
268/// Closes the file handle on scope exit.
269impl Drop for FileHandle {
270    /// Flush and release all streams when the handle leaves scope.
271    fn drop(&mut self) {
272        let _ = self.close();
273    }
