1//! Pair raw file bytes with the logical path they were loaded from.
2//!
3//! - Provide length, emptiness, and UTF-8 decode helpers on the cached payload.
4//! - Serve as the common return type for GameFS load operations.
5
6/// Cached file bytes paired with the logical path they came from.
7pub struct FileData {
8    /// Logical path used to load or label the payload.
9    pub path: String,
10    /// Raw file contents in load order.
11    pub bytes: Vec<u8>,
12}
13
14/// Core constructors and accessors for `FileData`.
15impl FileData {
16    /// Create a file payload from a path and raw bytes.
17    pub fn new(path: String, bytes: Vec<u8>) -> Self {
18        Self { path, bytes }
19    }
20    /// Return the payload length in bytes.
21    pub fn len(&self) -> usize {
22        self.bytes.len()
23    }
24    /// Return true when the payload has no bytes.
25    pub fn is_empty(&self) -> bool {
26        self.bytes.is_empty()
27    }
28    /// Decode the payload as UTF-8 or return the decode error.
29    pub fn as_str(&self) -> Result<&str, std::str::Utf8Error> {
30        std::str::from_utf8(&self.bytes)
31    }
32}
