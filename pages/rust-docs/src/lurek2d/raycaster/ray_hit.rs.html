1//! DDA ray-cast result record holding wall distance, hit coordinates, and texture sampling data.
2//!
3//! - Carries both fish-eye-corrected and raw distances for flexible column rendering.
4//! - Provides side, alpha, and cell value for shading and transparency decisions.
5
6/// Result record for one DDA ray; produced by `Raycaster2D::cast_ray`.
7#[derive(Debug, Clone)]
8pub struct RayHit {
9    /// Fish-eye-corrected perpendicular distance from the camera plane to the wall.
10    pub distance: f32,
11    /// Euclidean ray travel distance before fish-eye correction.
12    pub raw_distance: f32,
13    /// Tile value of the wall cell that was hit.
14    pub cell_value: u32,
15    /// Alpha of the wall type; 1.0 = fully opaque, < 1.0 = transparent/glass.
16    pub alpha: f32,
17    /// Hit side: 0 = X-aligned face, 1 = Y-aligned face.
18    pub side: u8,
19    /// Horizontal texture coordinate at the hit point, 0.0..1.0.
20    pub tex_u: f32,
21    /// World X coordinate of the exact hit point.
22    pub hit_x: f32,
23    /// World Y coordinate of the exact hit point.
24    pub hit_y: f32,
25    /// True when the ray struck a solid or semi-transparent wall before `max_dist`.
26    pub hit: bool,
27}
