1//! Convert a built raycaster scene into GPU-ready render commands.
2//!
3//! - Emit textured quads for ceilings, floors, walls, and sprites in correct painter order.
4//! - Fall back to solid-color rectangles when a surface has no texture assigned.
5
6use crate::raycaster::scene::RaycasterScene;
7use crate::render::renderer::{DrawMode, RenderCommand};
8use crate::render::BlendMode;
9
10/// Render-command generation for a fully built raycaster scene.
11impl RaycasterScene {
12    /// Build a `Vec<RenderCommand>` for the full scene: ceilings, floors, walls, then sprites back-to-front.
13    pub fn generate_render_commands(&self) -> Vec<RenderCommand> {
14        let mut cmds = Vec::with_capacity(self.quad_count() + 2);
15        cmds.push(RenderCommand::SetBlendMode(BlendMode::Alpha));
16        for ceil in &self.ceilings {
17            match ceil.texture_key {
18                Some(tex) => {
19                    cmds.push(RenderCommand::DrawTexturedQuad {
20                        corners: ceil.corners,
21                        uvs: ceil.uvs,
22                        corner_w: ceil.corner_w,
23                        texture_key: tex,
24                        color: ceil.light,
25                    });
26                }
27                None => {
28                    let [r, g, b, a] = ceil.light;
29                    cmds.push(RenderCommand::SetColor(r, g, b, a));
30                    cmds.push(RenderCommand::Rectangle {
31                        mode: DrawMode::Fill,
32                        x: ceil.corners[0].x,
33                        y: ceil.corners[0].y,
34                        w: ceil.corners[1].x - ceil.corners[0].x,
35                        h: ceil.corners[3].y - ceil.corners[0].y,
36                    });
37                }
38            }
39        }
40        for floor in &self.floors {
41            match floor.texture_key {
42                Some(tex) => {
43                    cmds.push(RenderCommand::DrawTexturedQuad {
44                        corners: floor.corners,
45                        uvs: floor.uvs,
46                        corner_w: floor.corner_w,
47                        texture_key: tex,
48                        color: floor.light,
49                    });
50                }
51                None => {
52                    let [r, g, b, a] = floor.light;
53                    cmds.push(RenderCommand::SetColor(r, g, b, a));
54                    cmds.push(RenderCommand::Rectangle {
55                        mode: DrawMode::Fill,
56                        x: floor.corners[0].x,
57                        y: floor.corners[0].y,
58                        w: floor.corners[1].x - floor.corners[0].x,
59                        h: floor.corners[3].y - floor.corners[0].y,
60                    });
61                }
62            }
63        }
64        for wall in &self.walls {
65            match wall.texture_key {
66                Some(tex) => {
67                    cmds.push(RenderCommand::DrawTexturedQuad {
68                        corners: wall.corners,
69                        uvs: wall.uvs,
70                        corner_w: wall.corner_w,
71                        texture_key: tex,
72                        color: wall.light,
73                    });
74                }
75                None => {
76                    let [r, g, b, a] = wall.light;
77                    cmds.push(RenderCommand::SetColor(r, g, b, a));
78                    cmds.push(RenderCommand::Rectangle {
79                        mode: DrawMode::Fill,
80                        x: wall.corners[0].x,
81                        y: wall.corners[0].y,
82                        w: wall.corners[1].x - wall.corners[0].x,
83                        h: wall.corners[3].y - wall.corners[0].y,
84                    });
85                }
86            }
87        }
88        for sprite in &self.sprites {
89            cmds.push(RenderCommand::DrawTexturedQuad {
90                corners: sprite.corners,
91                uvs: sprite.uvs,
92                corner_w: [1.0, 1.0, 1.0, 1.0],
93                texture_key: sprite.texture_key,
94                color: sprite.light,
95            });
96        }
97        cmds
98    }
99}
