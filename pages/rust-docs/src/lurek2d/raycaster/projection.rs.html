1//! Wall-column projection from ray distance to screen-pixel height and vertical bounds.
2//!
3//! - Distance-based shading for depth fog attenuation.
4//! - Returns `(wall_height, draw_start, draw_end)` clamped to valid screen-pixel bounds.
5
6/// Project a wall column at `distance` using `fov` and `screen_height`;.
7/// return `(wall_height, draw_start_y, draw_end_y)` in screen pixels.
8pub fn project_column(distance: f32, fov: f32, screen_height: f32) -> (f32, f32, f32) {
9    if distance <= 0.0 {
10        return (screen_height, 0.0, screen_height);
11    }
12    let wall_height = screen_height / (distance * (fov / 2.0).tan());
13    let draw_start = (screen_height - wall_height) / 2.0;
14    let draw_end = draw_start + wall_height;
15    (
16        wall_height,
17        draw_start.max(0.0),
18        draw_end.min(screen_height),
19    )
20}
21/// Return a brightness multiplier 0.0..1.0 based on `distance` relative to `max_distance`.
22pub fn distance_shade(distance: f32, max_distance: f32) -> f32 {
23    if max_distance <= 0.0 {
24        return 0.0;
25    }
26    (1.0 - distance / max_distance).clamp(0.0, 1.0)
27}
