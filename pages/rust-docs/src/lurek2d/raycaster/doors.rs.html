1//! Animated sliding doors placed on raycaster grid tiles.
2//!
3//! - Per-door state machine: Closed → Opening → Open → Closing → Closed.
4//! - DoorManager registry drives batch updates and spatial lookups.
5
6/// Slide axis of a door: horizontal (slides along X) or vertical (slides along Y).
7#[derive(Debug, Clone, Copy, PartialEq)]
8pub enum DoorDirection {
9    /// Door slides along the X axis when opening.
10    Horizontal,
11    /// Door slides along the Y axis when opening.
12    Vertical,
13}
14/// Current animation phase of a door.
15#[derive(Debug, Clone, Copy, PartialEq)]
16pub enum DoorState {
17    /// Fully closed; blocks passage.
18    Closed,
19    /// Animating toward fully open.
20    Opening,
21    /// Fully open; passage allowed.
22    Open,
23    /// Animating toward fully closed.
24    Closing,
25}
26/// A single animated door placed at a grid tile.
27#[derive(Debug, Clone)]
28pub struct Door {
29    /// Grid X coordinate of the door tile.
30    pub x: u32,
31    /// Grid Y coordinate of the door tile.
32    pub y: u32,
33    /// How far the door has slid open, 0.0 (closed) to 1.0 (fully open).
34    pub open_amount: f32,
35    /// Animation speed in open_amount units per second.
36    pub speed: f32,
37    /// Slide direction of this door.
38    pub direction: DoorDirection,
39    /// Current animation state.
40    pub state: DoorState,
41}
42/// Registry and update driver for all doors in the current map.
43pub struct DoorManager {
44    /// All registered doors, indexed by the integer handle returned from `add_door`.
45    doors: Vec<Door>,
46}
47impl DoorManager {
48    /// Create an empty `DoorManager`.
49    pub fn new() -> Self {
50        Self { doors: Vec::new() }
51    }
52    /// Register a new closed door at `(x, y)` with the given `direction` and `speed`; return its index handle.
53    pub fn add_door(&mut self, x: u32, y: u32, direction: DoorDirection, speed: f32) -> usize {
54        let index = self.doors.len();
55        self.doors.push(Door {
56            x,
57            y,
58            open_amount: 0.0,
59            speed,
60            direction,
61            state: DoorState::Closed,
62        });
63        index
64    }
65    /// Start opening door `index` if it is Closed or Closing; no-op otherwise.
66    pub fn open_door(&mut self, index: usize) {
67        if let Some(door) = self.doors.get_mut(index) {
68            if door.state == DoorState::Closed || door.state == DoorState::Closing {
69                door.state = DoorState::Opening;
70            }
71        }
72    }
73    /// Start closing door `index` if it is Open or Opening; no-op otherwise.
74    pub fn close_door(&mut self, index: usize) {
75        if let Some(door) = self.doors.get_mut(index) {
76            if door.state == DoorState::Open || door.state == DoorState::Opening {
77                door.state = DoorState::Closing;
78            }
79        }
80    }
81    /// Advance all door animations by `dt` seconds.
82    pub fn update(&mut self, dt: f32) {
83        for door in &mut self.doors {
84            match door.state {
85                DoorState::Opening => {
86                    door.open_amount += door.speed * dt;
87                    if door.open_amount >= 1.0 {
88                        door.open_amount = 1.0;
89                        door.state = DoorState::Open;
90                    }
91                }
92                DoorState::Closing => {
93                    door.open_amount -= door.speed * dt;
94                    if door.open_amount <= 0.0 {
95                        door.open_amount = 0.0;
96                        door.state = DoorState::Closed;
97                    }
98                }
99                _ => {}
100            }
101        }
102    }
103    /// Return the first door at grid tile `(x, y)`, or `None` if none is registered there.
104    pub fn get_door_at(&self, x: u32, y: u32) -> Option<&Door> {
105        self.doors.iter().find(|d| d.x == x && d.y == y)
106    }
107    /// Return a slice of all registered doors.
108    pub fn doors(&self) -> &[Door] {
109        &self.doors
110    }
111}
112/// Implement `Default` for `DoorManager` delegating to `new()`.
113impl Default for DoorManager {
114    fn default() -> Self {
115        Self::new()
116    }
117}
