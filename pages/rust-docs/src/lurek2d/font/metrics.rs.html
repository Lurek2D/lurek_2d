1//! Font metrics and multi-line text measurement utilities.
2//!
3//! - `measure_text` splits on `\n` and returns a `TextMetrics` with width/height.
4//! - `measure_line` operates on a single line and accounts for kerning pairs.
5//! - Line height includes ascender, descender, and the configurable line-gap.
6//! - Results are in logical pixels; caller must apply DPI scale if needed.
7
8use crate::font::bitmap_font::BitmapFont;
9
10/// Per-glyph metric data including UV coordinates and pixel dimensions.
11#[derive(Debug, Clone, Copy)]
12pub struct GlyphMetrics {
13    /// Horizontal advance width in pixels.
14    pub advance_width: f32,
15    /// Horizontal bearing (offset from cursor to glyph left edge).
16    pub bearing_x: f32,
17    /// Vertical bearing (offset from baseline to glyph top edge).
18    pub bearing_y: f32,
19    /// Glyph bitmap width in pixels.
20    pub width: u32,
21    /// Glyph bitmap height in pixels.
22    pub height: u32,
23    /// U coordinate of the glyph in the atlas (normalised 0..1).
24    pub uv_x: f32,
25    /// V coordinate of the glyph in the atlas (normalised 0..1).
26    pub uv_y: f32,
27    /// Width of the glyph region in the atlas (normalised 0..1).
28    pub uv_w: f32,
29    /// Height of the glyph region in the atlas (normalised 0..1).
30    pub uv_h: f32,
31}
32
33/// Metrics for a single line of measured text.
34#[derive(Debug, Clone)]
35pub struct LineMetrics {
36    /// Total width of the line in pixels.
37    pub width: f32,
38    /// Number of glyphs in the line.
39    pub glyph_count: u32,
40    /// Start byte index in the source string.
41    pub start_index: usize,
42    /// End byte index (exclusive) in the source string.
43    pub end_index: usize,
44}
45
46/// Result of measuring a block of text.
47#[derive(Debug, Clone)]
48pub struct TextMetrics {
49    /// Maximum width across all lines.
50    pub width: f32,
51    /// Total height of all lines.
52    pub height: f32,
53    /// Number of lines.
54    pub line_count: u32,
55    /// Per-line metric data.
56    pub lines: Vec<LineMetrics>,
57}
58
59/// Measures a full block of text, splitting on newline characters.
60///
61/// Returns aggregate dimensions and per-line metrics.
62pub fn measure_text(font: &BitmapFont, text: &str, scale: f32) -> TextMetrics {
63    let mut lines = Vec::new();
64    let mut max_width: f32 = 0.0;
65    let mut offset = 0usize;
66
67    for line_str in text.split('\n') {
68        let start = offset;
69        let end = offset + line_str.len();
70        let (w, _h) = measure_line(font, line_str, scale);
71        let glyph_count = line_str.chars().count() as u32;
72
73        max_width = max_width.max(w);
74        lines.push(LineMetrics {
75            width: w,
76            glyph_count,
77            start_index: start,
78            end_index: end,
79        });
80
81        // +1 for the newline character itself (except at the very end)
82        offset = end + 1;
83    }
84
85    let line_count = lines.len() as u32;
86    let height = line_count as f32 * font.line_height() * scale;
87
88    TextMetrics {
89        width: max_width,
90        height,
91        line_count,
92        lines,
93    }
94}

96/// Measures a single line of text, returning (width, height).
97///
98/// Does not handle newline characters — use [`measure_text`] for multi-line.
99pub fn measure_line(font: &BitmapFont, text: &str, scale: f32) -> (f32, f32) {
100    let mut width = 0.0f32;
101    for ch in text.chars() {
102        width += char_advance(font, ch, scale);
103    }
104    let height = font.line_height() * scale;
105    (width, height)
106}
107
108/// Returns the horizontal advance for a single character at the given scale.
109///
110/// Falls back to the space character advance if the glyph is missing.
111pub fn char_advance(font: &BitmapFont, ch: char, scale: f32) -> f32 {
112    let codepoint = ch as u32;
113    if let Some(glyph) = font.glyph_info(codepoint) {
114        glyph.advance_width * scale
115    } else {
116        // Fallback: use space advance or zero
117        font.glyph_info(0x20)
118            .map(|g| g.advance_width * scale)
119            .unwrap_or(0.0)
120    }
121}
