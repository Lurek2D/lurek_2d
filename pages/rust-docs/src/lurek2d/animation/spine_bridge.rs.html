1//! Bridges a Spine skeleton to an animation state machine via name mapping.
2//!
3//! - Plays mapped Spine clips automatically when the FSM transitions to a new state.
4//! - Owns the skeleton instance and advances its animation and world transforms each frame.
5//! - Exposes read and write access to the skeleton and the last applied FSM state.
6
7use super::state_machine::AnimStateMachine;
8use crate::spine::skeleton::Skeleton;
9use std::collections::HashMap;
10/// Maps FSM state names to Spine animation clips.
11#[derive(Debug, Clone)]
12pub struct SpineAnimBridge {
13    /// Owned skeleton instance.
14    skeleton: Skeleton,
15    /// Mapping from FSM state to skeleton clip.
16    state_map: HashMap<String, String>,
17    /// Last applied FSM state.
18    last_state: String,
19    /// Looping override per mapped state.
20    looping_states: HashMap<String, bool>,
21}
22impl SpineAnimBridge {
23    /// Create a bridge for a skeleton.
24    pub fn new(skeleton: Skeleton) -> Self {
25        Self {
26            skeleton,
27            state_map: HashMap::new(),
28            last_state: String::new(),
29            looping_states: HashMap::new(),
30        }
31    }
32    /// Map a FSM state to a skeleton clip.
33    pub fn map(&mut self, fsm_state: &str, skeleton_clip: &str) {
34        self.state_map
35            .insert(fsm_state.to_string(), skeleton_clip.to_string());
36    }
37    /// Map a FSM state to a skeleton clip and set its looping override.
38    pub fn map_looping(&mut self, fsm_state: &str, skeleton_clip: &str, looping: bool) {
39        self.state_map
40            .insert(fsm_state.to_string(), skeleton_clip.to_string());
41        self.looping_states.insert(fsm_state.to_string(), looping);
42    }
43    /// Advance the FSM and play the mapped Spine animation when the state changes.
44    pub fn update(&mut self, dt: f32, fsm: &mut AnimStateMachine) {
45        let state_before = fsm.get_state().to_string();
46        fsm.update(dt);
47        let state_after = fsm.get_state().to_string();
48        if state_after != self.last_state {
49            self.last_state = state_after.clone();
50            if let Some(clip) = self.state_map.get(&state_after) {
51                let looping = self
52                    .looping_states
53                    .get(&state_after)
54                    .copied()
55                    .unwrap_or(true);
56                let _ = self.skeleton.play_animation(clip, looping);
57            }
58        }
59        let _ = state_before;
60        self.skeleton.update_animation(dt);
61        self.skeleton.update_world_transforms();
62    }
63    /// Return the current skeleton.
64    pub fn skeleton(&self) -> &Skeleton {
65        &self.skeleton
66    }
67    /// Return the skeleton mutably.
68    pub fn skeleton_mut(&mut self) -> &mut Skeleton {
69        &mut self.skeleton
70    }
71    /// Return the last applied FSM state.
72    pub fn last_applied_state(&self) -> &str {
73        &self.last_state
74    }
75    /// Return the mapped clip for a FSM state.
76    pub fn get_mapped_clip(&self, fsm_state: &str) -> Option<&str> {
77        self.state_map.get(fsm_state).map(String::as_str)
78    }
79}
