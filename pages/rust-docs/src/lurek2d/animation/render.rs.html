1//! Converts the current animation frame quad into a textured draw command.
2//!
3//! - Stores atlas reference, position, rotation, and scale in `AnimRenderParams`.
4//! - Provides a standalone `quad_to_draw_command` helper reusable outside the controller.
5
6use crate::animation::controller::Animation;
7use crate::math::Rect;
8use crate::render::renderer::RenderCommand;
9use crate::runtime::resource_keys::TextureKey;
10
11/// Rendering inputs shared by animation frame draw helpers.
12#[derive(Debug, Clone)]
13pub struct AnimRenderParams {
14    /// Texture atlas that contains the frame quad.
15    pub texture_key: TextureKey,
16    /// Full atlas width in pixels.
17    pub tex_w: f32,
18    /// Full atlas height in pixels.
19    pub tex_h: f32,
20    /// World-space or screen-space X position for the draw call.
21    pub x: f32,
22    /// World-space or screen-space Y position for the draw call.
23    pub y: f32,
24    /// Rotation in radians.
25    pub rotation: f32,
26    /// Horizontal scale.
27    pub sx: f32,
28    /// Vertical scale.
29    pub sy: f32,
30    /// Origin X used as the rotation and scale pivot.
31    pub ox: f32,
32    /// Origin Y used as the rotation and scale pivot.
33    pub oy: f32,
34}
35
36impl Animation {
37    /// Build a draw command for the current frame when the animation has an active quad.
38    pub fn generate_render_command(&self, params: &AnimRenderParams) -> Option<RenderCommand> {
39        let quad = self.current_quad()?;
40        Some(quad_to_draw_command(&quad, params))
41    }
42}
43
44/// Convert a texture quad plus render parameters into a renderer draw command.
45pub fn quad_to_draw_command(quad: &Rect, params: &AnimRenderParams) -> RenderCommand {
46    RenderCommand::DrawQuad {
47        texture_key: params.texture_key,
48        quad_x: quad.x,
49        quad_y: quad.y,
50        quad_w: quad.width,
51        quad_h: quad.height,
52        tex_w: params.tex_w,
53        tex_h: params.tex_h,
54        x: params.x,
55        y: params.y,
56        rotation: params.rotation,
57        sx: params.sx,
58        sy: params.sy,
59        ox: params.ox,
60        oy: params.oy,
61        effect: None,
62    }
63}
