1//! `MidiState` storage for loaded SoundFont binary data and its source path.
2//!
3//! - RIFF+sfbk header validation on `set_soundfont` to reject malformed SF2 files.
4//! - Query and clear helpers for SoundFont availability and data access.
5
6#[derive(Debug, Clone, Default)]
7/// Stores the loaded SoundFont binary data and its source path for MIDI synthesis.
8pub struct MidiState {
9    /// SF2-format SoundFont binary loaded from disk; validated on `set_soundfont`.
10    soundfont_data: Option<Vec<u8>>,
11    /// Source path of the loaded SoundFont file, if provided at load time.
12    soundfont_path: Option<String>,
13}
14impl MidiState {
15    /// Create a new `MidiState` with no SoundFont loaded.
16    pub fn new() -> Self {
17        Self {
18            soundfont_data: None,
19            soundfont_path: None,
20        }
21    }
22    /// Load `data` as an SF2 SoundFont, validating the RIFF+sfbk header; error if invalid.
23    pub fn set_soundfont(&mut self, data: Vec<u8>, path: Option<String>) -> Result<(), String> {
24        if data.len() < 12 {
25            return Err("SoundFont data too small (expected RIFF header)".to_string());
26        }
27        if &data[0..4] != b"RIFF" {
28            return Err("Invalid SoundFont: missing RIFF header".to_string());
29        }
30        if &data[8..12] != b"sfbk" {
31            return Err("Invalid SoundFont: not an SF2 file (missing sfbk chunk)".to_string());
32        }
33        self.soundfont_data = Some(data);
34        self.soundfont_path = path;
35        Ok(())
36    }
37    /// Return `true` when a SoundFont is loaded and ready for synthesis.
38    pub fn has_soundfont(&self) -> bool {
39        self.soundfont_data.is_some()
40    }
41    /// Unload the current SoundFont and clear its path.
42    pub fn clear_soundfont(&mut self) {
43        self.soundfont_data = None;
44        self.soundfont_path = None;
45    }
46    /// Return the source path of the loaded SoundFont, or `None` if none was loaded or path was not provided.
47    pub fn soundfont_path(&self) -> Option<&str> {
48        self.soundfont_path.as_deref()
49    }
50    /// Return a byte slice of the loaded SoundFont binary, or `None` if not loaded.
51    pub fn soundfont_data(&self) -> Option<&[u8]> {
52        self.soundfont_data.as_deref()
53    }
54}
