1//! Debug visualisation of physics bodies as render commands or rasterised images.
2//!
3//! - Colour-codes bodies by type: dynamic, static, kinematic, sensor.
4//! - Draws velocity arrows for dynamic bodies and shape outlines for all bodies.
5
6use crate::image::ImageData;
7use crate::physics::body::{BodyShape, BodyType};
8use crate::physics::world::World;
9use crate::render::renderer::{DrawMode, RenderCommand};
10
11/// Debug render and image-rasterisation helpers added to `World`.
12impl World {
13    /// Build a list of `RenderCommand`s drawing outlines and velocity arrows for all bodies.
14    pub fn generate_render_commands(&self) -> Vec<RenderCommand> {
15        let ids = self.get_body_ids();
16        if ids.is_empty() {
17            return Vec::new();
18        }
19        let mut cmds = Vec::with_capacity(ids.len() * 3);
20        for id in ids {
21            let body = match self.get_body(id) {
22                Some(b) => b,
23                None => continue,
24            };
25            let (r, g, b) = match body.body_type {
26                BodyType::Dynamic => (0.1f32, 0.9, 0.2),
27                BodyType::Static => (0.3, 0.6, 1.0),
28                BodyType::Kinematic => (0.2, 0.9, 1.0),
29                BodyType::Sensor => (1.0, 1.0, 0.0),
30            };
31            cmds.push(RenderCommand::SetColor(r, g, b, 0.85));
32            let px = body.position.x;
33            let py = body.position.y;
34            match body.shape {
35                BodyShape::Rect { width, height } => {
36                    cmds.push(RenderCommand::Rectangle {
37                        mode: DrawMode::Line,
38                        x: px - width * 0.5,
39                        y: py - height * 0.5,
40                        w: width,
41                        h: height,
42                    });
43                }
44                BodyShape::Circle { radius } => {
45                    cmds.push(RenderCommand::Circle {
46                        mode: DrawMode::Line,
47                        x: px,
48                        y: py,
49                        r: radius,
50                    });
51                }
52            }
53            if matches!(body.body_type, BodyType::Dynamic) {
54                let vx = body.velocity.x;
55                let vy = body.velocity.y;
56                if vx.abs() + vy.abs() > 0.01 {
57                    cmds.push(RenderCommand::SetColor(1.0, 0.3, 0.1, 0.9));
58                    cmds.push(RenderCommand::Line {
59                        x1: px,
60                        y1: py,
61                        x2: px + vx * 0.1,
62                        y2: py + vy * 0.1,
63                    });
64                }
65            }
66        }
67        cmds
68    }
69    /// Rasterise all bodies onto a `width`×`height` `ImageData` centered at the origin.
70    pub fn draw_to_image(&self, width: u32, height: u32) -> ImageData {
71        let mut img = ImageData::new(width, height);
72        img.fill(20, 20, 30, 255);
73        let scale = width as f32 / 200.0;
74        let cx = width as i32 / 2;
75        let cy = height as i32 / 2;
76        for id in self.get_body_ids() {
77            let body = match self.get_body(id) {
78                Some(b) => b,
79                None => continue,
80            };
81            let (r, g, b) = match body.body_type {
82                BodyType::Dynamic => (80u8, 220, 80),
83                BodyType::Static => (80, 150, 220),
84                BodyType::Kinematic => (80, 220, 220),
85                BodyType::Sensor => (220, 220, 80),
86            };
87            let bx = cx + (body.position.x * scale) as i32;
88            let by = cy + (body.position.y * scale) as i32;
89            match body.shape {
90                BodyShape::Rect {
91                    width: bw,
92                    height: bh,
93                } => {
94                    let hw = ((bw * scale * 0.5) as u32).max(1);
95                    let hh = ((bh * scale * 0.5) as u32).max(1);
96                    img.draw_rect(bx - hw as i32, by - hh as i32, hw * 2, hh * 2, r, g, b, 200);
97                }
98                BodyShape::Circle { radius } => {
99                    let rad = ((radius * scale) as u32).max(1);
100                    img.draw_circle(bx, by, rad, r, g, b, 200);
101                }
102            }
103        }
104        img
105    }
