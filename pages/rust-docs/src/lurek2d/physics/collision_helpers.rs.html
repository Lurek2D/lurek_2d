1//! Pure-geometry collision tests for AABB, circle, and point queries.
2//!
3//! - No allocations or side effects; suitable for hot-path per-frame checks.
4//! - All coordinates assume x-right, y-down screen space.
5
6/// Return true when two AABBs overlap (axes: x-right, y-down).
7#[allow(clippy::too_many_arguments)]
8pub fn test_aabb(ax: f32, ay: f32, aw: f32, ah: f32, bx: f32, by: f32, bw: f32, bh: f32) -> bool {
9    ax < bx + bw && ax + aw > bx && ay < by + bh && ay + ah > by
10}
11/// Return true when two circles overlap given center positions and radii.
12pub fn test_circles(ax: f32, ay: f32, ar: f32, bx: f32, by: f32, br: f32) -> bool {
13    let dx = ax - bx;
14    let dy = ay - by;
15    let dist_sq = dx * dx + dy * dy;
16    let r_sum = ar + br;
17    dist_sq < r_sum * r_sum
18}
19/// Return true when point `(px,py)` lies inside the AABB.
20pub fn test_point_aabb(px: f32, py: f32, ax: f32, ay: f32, aw: f32, ah: f32) -> bool {
21    px >= ax && px < ax + aw && py >= ay && py < ay + ah
22}
23/// Return true when circle `(cx,cy,cr)` overlaps the AABB.
24pub fn test_circle_aabb(cx: f32, cy: f32, cr: f32, ax: f32, ay: f32, aw: f32, ah: f32) -> bool {
25    let closest_x = cx.clamp(ax, ax + aw);
26    let closest_y = cy.clamp(ay, ay + ah);
27    let dx = cx - closest_x;
28    let dy = cy - closest_y;
29    dx * dx + dy * dy < cr * cr
30}
