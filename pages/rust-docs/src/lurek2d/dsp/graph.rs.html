1//! DSP processing graph: nodes connected by typed audio-rate and control-rate edges.
2//!
3//! - `DspGraph` owns a topologically sorted list of `DspNode` processing units.
4//! - Edges carry either audio frames (f32 interleaved) or scalar control signals.
5//! - Evaluated once per audio buffer in the rodio callback on the audio thread.
6//! - Graph mutation (add/remove node, patch edge) is performed from the game thread
7//!   via a lock-free command queue consumed at the start of each audio callback.
8

13/// Stable DSP graph node handle.
14pub type NodeId = u32;
15
16/// DSP node processing kind.
17#[derive(Debug, Clone, Copy, PartialEq, Eq)]
18pub enum DspNodeType {
19	/// Low-pass filter node.
20	Lowpass,
21	/// High-pass filter node.
22	Highpass,
23	/// Band-pass filter node.
24	Bandpass,
25	/// Gain node.
26	Gain,
27	/// Pass-through node for accepted effect types that need graph identity only.
28	Passthrough,
29}
30
31impl DspNodeType {
32	/// Parse a Lua-facing node kind.
33	pub fn parse(kind: &str) -> Self {
34		match kind {
35			"lowpass" => Self::Lowpass,
36			"highpass" => Self::Highpass,
37			"bandpass" => Self::Bandpass,
38			"gain" => Self::Gain,
39			_ => Self::Passthrough,
40		}
41	}
42
43	/// Return the Lua-facing node kind.
44	pub fn as_str(self) -> &'static str {
45		match self {
46			Self::Lowpass => "lowpass",
47			Self::Highpass => "highpass",
48			Self::Bandpass => "bandpass",
49			Self::Gain => "gain",
50			Self::Passthrough => "passthrough",
51		}
52	}
53}
54
55/// One DSP graph node with simple named parameters.
56#[derive(Debug, Clone)]
57pub struct DspNode {
58	node_type: DspNodeType,
59	p1: f32,
60	p2: f32,
61	p3: f32,
62}
63
64impl DspNode {
65	/// Create a node from a Lua-facing kind.
66	pub fn new(kind: &str) -> Self {
67		let node_type = DspNodeType::parse(kind);
68		let (p1, p2, p3) = match node_type {
69			DspNodeType::Lowpass => (1000.0, 0.0, 0.0),
70			DspNodeType::Highpass => (200.0, 0.0, 0.0),
71			DspNodeType::Bandpass => (200.0, 2000.0, 0.0),
72			DspNodeType::Gain => (1.0, 0.0, 0.0),
73			DspNodeType::Passthrough => (0.0, 0.0, 0.0),
74		};
75		Self {
76			node_type,
77			p1,
78			p2,
79			p3,
80		}
81	}
82
83	/// Set a named node parameter.
84	pub fn set_param(&mut self, name: &str, value: f32) -> Result<(), String> {
85		match name {
86			"cutoff" | "frequency" | "gain" | "value" | "low" => {
87				self.p1 = value;
88				Ok(())
89			}
90			"high" | "q" | "mix" => {
91				self.p2 = value;
92				Ok(())
93			}
94			"p3" => {
95				self.p3 = value;
96				Ok(())
97			}
98			other => Err(format!("invalid DSP node parameter: {}", other)),
99		}
100	}
101
102	/// Get a named node parameter.
103	pub fn get_param(&self, name: &str) -> Option<f32> {
104		match name {
105			"cutoff" | "frequency" | "gain" | "value" | "low" => Some(self.p1),
106			"high" | "q" | "mix" => Some(self.p2),
107			"p3" => Some(self.p3),
108			_ => None,
109		}
110	}
111
112	/// Return this node's kind.
113	pub fn node_type(&self) -> DspNodeType {
114		self.node_type
115	}
116
117	fn process(&self, sound_data: &mut SoundData) {
118		match self.node_type {
119			DspNodeType::Lowpass => sound_data.apply_lowpass(self.p1),
120			DspNodeType::Highpass => sound_data.apply_highpass(self.p1),
121			DspNodeType::Bandpass => sound_data.apply_bandpass(self.p1, self.p2),
122			DspNodeType::Gain => sound_data.apply_gain(self.p1),
123			DspNodeType::Passthrough => {}
124		}
125	}
126}

128/// Minimal deterministic DSP graph used by Lua for offline SoundData processing.
129#[derive(Debug, Clone)]
130pub struct DspGraph {
131	next_id: NodeId,
132	nodes: Vec<(NodeId, DspNode)>,
133	edges: Vec<(NodeId, NodeId)>,
134}
135
136impl DspGraph {
137	/// Create an empty DSP graph.
138	pub fn new() -> Self {
139		Self {
140			next_id: 1,
141			nodes: Vec::new(),
142			edges: Vec::new(),
143		}
144	}
145
146	/// Add a node and return its stable ID.
147	pub fn add_node(&mut self, node: DspNode) -> NodeId {
148		let id = self.next_id;
149		self.next_id = self.next_id.saturating_add(1);
150		self.nodes.push((id, node));
151		id
152	}
153
154	/// Connect two existing nodes.
155	pub fn connect(&mut self, from: NodeId, to: NodeId) -> bool {
156		if !self.has_node(from) || !self.has_node(to) {
157			return false;
158		}
159		if !self.edges.contains(&(from, to)) {
160			self.edges.push((from, to));
161		}
162		true
163	}
164
165	/// Disconnect two nodes.
166	pub fn disconnect(&mut self, from: NodeId, to: NodeId) -> bool {
167		let before = self.edges.len();
168		self.edges.retain(|edge| *edge != (from, to));
169		self.edges.len() != before
170	}
171
172	/// Process a sound buffer through all nodes in insertion order.
173	pub fn process(&self, sound_data: &SoundData) -> SoundData {
174		let mut output = sound_data.clone();
175		for (_, node) in &self.nodes {
176			node.process(&mut output);
177		}
178		output
179	}
180
181	/// Clear all graph nodes and connections.
182	pub fn clear(&mut self) {
183		self.nodes.clear();
184		self.edges.clear();
185		self.next_id = 1;
186	}
187
188	fn has_node(&self, id: NodeId) -> bool {
189		self.nodes.iter().any(|(node_id, _)| *node_id == id)
190	}
191}
192
193impl Default for DspGraph {
194	fn default() -> Self {
195		Self::new()
196	}
