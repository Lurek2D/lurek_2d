11/// Create parent directories for `path` when needed.
12fn ensure_parent_dir(path: &str) -> Result<(), String> {
13    if let Some(parent) = std::path::Path::new(path).parent() {
14        if !parent.as_os_str().is_empty() && !parent.exists() {
15            std::fs::create_dir_all(parent).map_err(|e| {
16                format!(
17                    "cannot create output directory '{}': {}",
18                    parent.display(),
19                    e
20                )
21            })?;
22        }
23    }
24    Ok(())
25}
26/// Render waveform overview of `input_wav` to `output_png` with given image size.
27pub fn waveform_to_png(
28    input_wav: &str,
29    output_png: &str,
30    width: u32,
31    height: u32,
32) -> Result<(), String> {
33    let samples = read_mono_f32(input_wav)?;
34    let bg: Rgba<u8> = Rgba([12, 14, 28, 255]);
35    let fg: Rgba<u8> = Rgba([0, 200, 220, 255]);
36    let mut img: ImageBuffer<Rgba<u8>, Vec<u8>> = ImageBuffer::from_fn(width, height, |_, _| bg);
37    let n = samples.len().max(1);
38    let samples_per_col = n as f32 / width as f32;
39    let half_h = height as f32 * 0.5;
40    for x in 0..width {
41        let start = ((x as f32 * samples_per_col) as usize).min(n - 1);
42        let end = (((x + 1) as f32 * samples_per_col) as usize).min(n);
43        let slice = &samples[start..end];
44        let (mn, mx) = if slice.is_empty() {
45            (0.0_f32, 0.0_f32)
46        } else {
47            let mn = slice.iter().cloned().fold(f32::INFINITY, f32::min);
48            let mx = slice.iter().cloned().fold(f32::NEG_INFINITY, f32::max);
49            (mn, mx)
50        };
51        let y_top = ((half_h - mx.clamp(-1.0, 1.0) * half_h) as u32).min(height - 1);
52        let y_bot = ((half_h - mn.clamp(-1.0, 1.0) * half_h) as u32).min(height - 1);
53        for y in y_top..=y_bot {
54            img.put_pixel(x, y, fg);
55        }
56    }
57    ensure_parent_dir(output_png)?;
58    img.save(output_png)
59        .map_err(|e| format!("cannot save PNG '{}': {}", output_png, e))
60}

62/// Render a spectrogram heatmap of `input_wav` to `output_png` using a Hann-windowed DFT.
63pub fn spectrogram_to_png(
64    input_wav: &str,
65    output_png: &str,
66    width: u32,
67    height: u32,
68) -> Result<(), String> {
69    let samples = read_mono_f32(input_wav)?;
70    const WIN: usize = 512;
71    let half_win = WIN / 2;
72    let total_windows = (samples.len() as f32 / WIN as f32).ceil() as usize;
73    let hop = if total_windows > 1 {
74        (samples.len() - WIN) / (total_windows - 1).max(1)
75    } else {
76        samples.len().max(1)
77    };
78    let n_frames = if total_windows < 2 { 1 } else { total_windows };
79    let mut spectra: Vec<Vec<f32>> = Vec::with_capacity(n_frames);
80    for i in 0..n_frames {
81        let start = (i * hop).min(samples.len().saturating_sub(WIN));
82        let slice = &samples[start..(start + WIN).min(samples.len())];
83        let mut window = vec![0.0_f32; WIN];
84        window[..slice.len()].copy_from_slice(slice);
85        for (k, s) in window.iter_mut().enumerate() {
86            let hann =
87                0.5 * (1.0 - (2.0 * std::f32::consts::PI * k as f32 / (WIN - 1) as f32).cos());
88            *s *= hann;
89        }
90        let mut mags = vec![0.0_f32; half_win];
91        for k in 0..half_win {
92            let mut re = 0.0_f32;
93            let mut im = 0.0_f32;
94            let angle = -2.0 * std::f32::consts::PI * k as f32 / WIN as f32;
95            for (n, &xn) in window.iter().enumerate() {
96                re += xn * (angle * n as f32).cos();
97                im += xn * (angle * n as f32).sin();
98            }
99            mags[k] = (re * re + im * im).sqrt();
100        }
101        spectra.push(mags);
102    }
103    let global_max = spectra
104        .iter()
105        .flat_map(|m| m.iter())
106        .cloned()
107        .fold(0.0_f32, f32::max)
108        .max(1e-6);
109    let bg: Rgba<u8> = Rgba([0, 0, 0, 255]);
110    let mut img: ImageBuffer<Rgba<u8>, Vec<u8>> = ImageBuffer::from_fn(width, height, |_, _| bg);
111    for px in 0..width {
112        let frame_idx = ((px as f32 / width as f32) * n_frames as f32) as usize;
113        let frame_idx = frame_idx.min(n_frames - 1);
114        let mags = &spectra[frame_idx];
115        for py in 0..height {
116            let bin = (((height - 1 - py) as f32 / height as f32) * half_win as f32) as usize;
117            let bin = bin.min(half_win - 1);
118            let mag_norm = (mags[bin] / global_max).clamp(0.0, 1.0);
119            let colour = heat_colour(mag_norm);
120            img.put_pixel(px, py, colour);
121        }
122    }
123    ensure_parent_dir(output_png)?;
124    img.save(output_png)
125        .map_err(|e| format!("cannot save PNG '{}': {}", output_png, e))
126}
127/// Decode an audio file and return mono f32 samples (stereo/multichannel is averaged).
128fn read_mono_f32(path: &str) -> Result<Vec<f32>, String> {
129    let file = File::open(path).map_err(|e| format!("file not found: {}: {}", path, e))?;
130    let reader = BufReader::new(file);
131    let decoder =
132        Decoder::new(reader).map_err(|e| format!("failed to decode '{}': {}", path, e))?;
133    let channels = decoder.channels() as usize;
134    let raw: Vec<f32> = decoder.convert_samples::<f32>().collect();
135    if channels <= 1 {
136        return Ok(raw);
137    }
138    let inv = 1.0 / channels as f32;
139    let mono: Vec<f32> = raw
140        .chunks(channels)
141        .map(|c| c.iter().sum::<f32>() * inv)
142        .collect();
143    Ok(mono)
144}
145/// Map normalized magnitude `t` in [0,1] to an RGB heatmap colour.
146fn heat_colour(t: f32) -> Rgba<u8> {
147    let (r, g, b) = if t < 0.25 {
148        let u = t / 0.25;
149        (0.0_f32, 0.0, u)
150    } else if t < 0.5 {
151        let u = (t - 0.25) / 0.25;
152        (0.0, u, 1.0)
153    } else if t < 0.75 {
154        let u = (t - 0.5) / 0.25;
155        (u, 1.0, 1.0 - u)
156    } else {
157        let u = (t - 0.75) / 0.25;
158        (1.0, 1.0, u)
159    };
160    Rgba([(r * 255.0) as u8, (g * 255.0) as u8, (b * 255.0) as u8, 255])
161}
