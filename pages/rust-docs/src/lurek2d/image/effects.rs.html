1//! Pixel-level color adjustments: brightness, contrast, saturation, gamma, tint, grayscale, sepia, invert, threshold, and posterize applied via parallel pixel mapping.
2//!
3//! - Alpha channel masking and deterministic per-pixel noise injection with repeatable seed.
4//! - Geometric transforms: horizontal and vertical flip, 90-degree clockwise rotation, and rectangular crop with bounds validation.
5//! - Resize operations using nearest-neighbor sampling, bilinear interpolation, and Lanczos3 windowed-sinc filtering.
6//! - Separable box blur with configurable radius and 3x3 unsharp-mask sharpening kernel.
7//! - General-purpose NxN kernel convolution with clamped-edge boundary handling and validation of odd kernel dimensions.
8//! - Compositing via alpha-blended blit with fast-path for fully opaque sources and nine-slice stretch drawing.
9//! - Bytewise image difference scoring across same-sized and differently-sized images for test comparison.
10//! - `ResizeFilter` enum for selecting resampling kernels via string parsing at the Lua boundary.

12use super::image_data::ImageData;
13/// Resize kernels supported by the image resampler.
14#[derive(Clone, Copy, Debug, PartialEq, Eq)]
15pub enum ResizeFilter {
16    /// Bilinear interpolation.
17    Bilinear,
18    /// Lanczos3 windowed-sinc interpolation.
19    Lanczos3,
20}
21impl ResizeFilter {
22    /// Parse a resize filter name and return the selected filter when recognized.
23    pub fn parse(value: &str) -> Option<Self> {
24        match value.to_ascii_lowercase().as_str() {
25            "bilinear" | "linear" => Some(Self::Bilinear),
26            "lanczos3" => Some(Self::Lanczos3),
27            _ => None,
28        }
29    }
30}
31/// Return the normalized sinc weight for the input value.
32fn sinc(x: f32) -> f32 {
33    if x.abs() < f32::EPSILON {
34        1.0
35    } else {
36        let px = std::f32::consts::PI * x;
37        px.sin() / px
38    }
39}
40/// Return the Lanczos window weight for a distance and window size.
41fn lanczos_weight(x: f32, a: f32) -> f32 {
42    let ax = x.abs();
43    if ax >= a {
44        0.0
45    } else {
46        sinc(x) * sinc(x / a)
47    }
48}
49impl ImageData {
50    /// Scale RGB channels by a factor in place.
51    pub fn brightness(&mut self, factor: f32) {
52        self.map_pixel_par(|_, _, r, g, b, a| {
53            let r = (r as f32 * factor).clamp(0.0, 255.0) as u8;
54            let g = (g as f32 * factor).clamp(0.0, 255.0) as u8;
55            let b = (b as f32 * factor).clamp(0.0, 255.0) as u8;
56            (r, g, b, a)
57        });
58    }
59    /// Adjust contrast around the midpoint in place.
60    pub fn contrast(&mut self, factor: f32) {
61        self.map_pixel_par(|_, _, r, g, b, a| {
62            let apply = |ch: u8| ((ch as f32 - 128.0) * factor + 128.0).clamp(0.0, 255.0) as u8;
63            (apply(r), apply(g), apply(b), a)
64        });
65    }
66    /// Blend RGB channels toward luminance by the given factor.
67    pub fn saturation(&mut self, factor: f32) {
68        self.map_pixel_par(|_, _, r, g, b, a| {
69            let luma = 0.2126 * r as f32 + 0.7152 * g as f32 + 0.0722 * b as f32;
70            let lerp = |ch: f32| (luma + (ch - luma) * factor).clamp(0.0, 255.0) as u8;
71            (lerp(r as f32), lerp(g as f32), lerp(b as f32), a)
72        });
73    }
74    /// Apply gamma correction to RGB channels in place.
75    pub fn gamma(&mut self, gamma: f32) {
76        self.map_pixel_par(|_, _, r, g, b, a| {
77            let apply =
78                |ch: u8| ((ch as f32 / 255.0).powf(1.0 / gamma) * 255.0).clamp(0.0, 255.0) as u8;
79            (apply(r), apply(g), apply(b), a)
80        });
81    }
82    /// Blend RGB channels toward a tint color by the given factor.
83    pub fn tint(&mut self, tr: u8, tg: u8, tb: u8, factor: f32) {
84        self.map_pixel_par(move |_, _, r, g, b, a| {
85            let lerp = |from: u8, to: u8| {
86                (from as f32 + (to as f32 - from as f32) * factor).clamp(0.0, 255.0) as u8
87            };
88            (lerp(r, tr), lerp(g, tg), lerp(b, tb), a)
89        });
90    }
91    /// Convert the image to grayscale in place.
92    pub fn grayscale(&mut self) {
93        self.map_pixel_par(|_, _, r, g, b, a| {
94            let luma = (0.2126 * r as f32 + 0.7152 * g as f32 + 0.0722 * b as f32).round() as u8;
95            (luma, luma, luma, a)
96        });
97    }
98    /// Apply a sepia tone in place. This function is part of the public API.
99    pub fn sepia(&mut self) {
100        self.map_pixel_par(|_, _, r, g, b, a| {
101            let rf = r as f32;
102            let gf = g as f32;
103            let bf = b as f32;
104            let nr = (0.393 * rf + 0.769 * gf + 0.189 * bf).clamp(0.0, 255.0) as u8;
105            let ng = (0.349 * rf + 0.686 * gf + 0.168 * bf).clamp(0.0, 255.0) as u8;
106            let nb = (0.272 * rf + 0.534 * gf + 0.131 * bf).clamp(0.0, 255.0) as u8;
107            (nr, ng, nb, a)
108        });
109    }
110    /// Invert the RGB channels in place.
111    pub fn invert(&mut self) {
112        self.map_pixel_par(|_, _, r, g, b, a| (255 - r, 255 - g, 255 - b, a));
113    }
114    /// Convert the image to a hard thresholded grayscale image.
115    pub fn threshold(&mut self, value: u8) {
116        self.map_pixel_par(move |_, _, r, g, b, a| {
117            let luma = (0.2126 * r as f32 + 0.7152 * g as f32 + 0.0722 * b as f32).round() as u8;
118            let v = if luma >= value { 255 } else { 0 };
119            (v, v, v, a)

122    /// Reduce color depth to the requested number of levels.
123    pub fn posterize(&mut self, levels: u8) {
124        let levels = levels.max(2);
125        let l = levels as f32 - 1.0;
126        self.map_pixel_par(move |_, _, r, g, b, a| {
127            let apply = |ch: u8| ((ch as f32 / 255.0 * l).round() / l * 255.0).round() as u8;
128            (apply(r), apply(g), apply(b), a)
129        });
130    }
131    /// Fill the whole image with a solid color.
132    pub fn fill(&mut self, r: u8, g: u8, b: u8, a: u8) {
133        self.map_pixel_par(move |_, _, _, _, _, _| (r, g, b, a));
134    }
135    /// Add repeatable random noise to RGB channels.
136    pub fn noise(&mut self, amount: u8) {
137        if amount == 0 {
138            return;
139        }
140        let w = self.width;
141        let h = self.height;
142        let mut seed = (w as u64)
143            .wrapping_mul(6_364_136_223_846_793_005)
144            .wrapping_add(1_442_695_040_888_963_407);
145        let range = amount as i32 * 2 + 1;
146        let pixels = self.pixels.as_mut_slice();
147        for _y in 0..h {
148            for _x in 0..w {
149                let idx = ((_y * w + _x) * 4) as usize;
150                for ch in 0..3usize {
151                    seed = seed
152                        .wrapping_mul(6_364_136_223_846_793_005)
153                        .wrapping_add(1_442_695_040_888_963_407);
154                    let offset = ((seed >> 33) as i32 % range) - amount as i32;
155                    pixels[idx + ch] = (pixels[idx + ch] as i32 + offset).clamp(0, 255) as u8;
156                }
157            }
158        }
159    }

167    /// Flip the image horizontally in place.
168    pub fn flip_horizontal(&mut self) {
169        let w = self.width as usize;
170        let h = self.height as usize;
171        for y in 0..h {
172            let row_start = y * w * 4;
173            let row = &mut self.pixels[row_start..row_start + w * 4];
174            for x in 0..w / 2 {
175                let left = x * 4;
176                let right = (w - 1 - x) * 4;
177                for i in 0..4 {
178                    row.swap(left + i, right + i);
179                }
180            }
181        }
182    }
183    /// Flip the image vertically in place.
184    pub fn flip_vertical(&mut self) {
185        let w = self.width as usize;
186        let h = self.height as usize;
187        for y in 0..h / 2 {
188            let top = y * w * 4;
189            let bottom = (h - 1 - y) * w * 4;
190            for i in 0..w * 4 {
191                self.pixels.swap(top + i, bottom + i);
192            }
193        }
194    }
195    /// Return a new image rotated 90 degrees clockwise.
196    pub fn rotate_90_cw(&self) -> ImageData {
197        let old_w = self.width;
198        let old_h = self.height;
199        let new_w = old_h;
200        let new_h = old_w;
201        let mut out = ImageData::new(new_w, new_h);
202        for y in 0..old_h {
203            for x in 0..old_w {
204                if let Some((r, g, b, a)) = self.get_pixel(x, y) {
205                    let nx = old_h - 1 - y;
206                    let ny = x;
207                    out.set_pixel(nx, ny, r, g, b, a);
208                }
209            }
210        }
211        out
212    }
213    /// Return a cropped copy of the image or `None` when the region is invalid.
214    pub fn crop(&self, x: u32, y: u32, w: u32, h: u32) -> Option<ImageData> {
215        if w == 0 || h == 0 || x + w > self.width || y + h > self.height {
216            return None;
217        }
218        let mut out = ImageData::new(w, h);
219        for row in 0..h {
220            let src_start = ((y + row) * self.width + x) as usize * 4;
221            let dst_start = (row * w) as usize * 4;
222            out.pixels[dst_start..dst_start + w as usize * 4]
223                .copy_from_slice(&self.pixels[src_start..src_start + w as usize * 4]);
224        }
225        Some(out)
226    }
227    /// Resize the image with nearest-neighbor sampling.
228    pub fn resize_nearest(&self, new_w: u32, new_h: u32) -> ImageData {
229        let mut out = ImageData::new(new_w, new_h);
230        if new_w == 0 || new_h == 0 || self.width == 0 || self.height == 0 {
231            return out;
232        }
233        for ny in 0..new_h {
234            for nx in 0..new_w {
235                let src_x = (nx * self.width / new_w).min(self.width - 1);
236                let src_y = (ny * self.height / new_h).min(self.height - 1);
237                if let Some((r, g, b, a)) = self.get_pixel(src_x, src_y) {
238                    out.set_pixel(nx, ny, r, g, b, a);
239                }
240            }
241        }
242        out
243    }
244    /// Blur the image with a separable box kernel and return a new image.
245    #[allow(clippy::needless_range_loop)]
246    pub fn blur(&self, radius: u32) -> ImageData {
247        if radius == 0 {
248            return self.clone();
249        }
250        let w = self.width as usize;
251        let h = self.height as usize;
252        let r = radius as usize;
253        let mut tmp_pixels = vec![0u8; w * h * 4];
254        for y in 0..h {
255            for x in 0..w {
256                let x_min = x.saturating_sub(r);
257                let x_max = (x + r).min(w - 1);
258                let count = (x_max - x_min + 1) as u32;
259                let mut sums = [0u32; 4];
260                for sx in x_min..=x_max {
261                    let idx = (y * w + sx) * 4;
262                    for c in 0..4 {
263                        sums[c] += self.pixels[idx + c] as u32;
264                    }
265                }
266                let dst = (y * w + x) * 4;
267                for c in 0..4 {
268                    tmp_pixels[dst + c] = (sums[c] / count) as u8;
269                }
270            }
271        }
272        let mut out = ImageData::new(self.width, self.height);
273        for y in 0..h {
274            for x in 0..w {
275                let y_min = y.saturating_sub(r);
276                let y_max = (y + r).min(h - 1);
277                let count = (y_max - y_min + 1) as u32;
278                let mut sums = [0u32; 4];
279                for sy in y_min..=y_max {
280                    let idx = (sy * w + x) * 4;
281                    for c in 0..4 {
282                        sums[c] += tmp_pixels[idx + c] as u32;
283                    }
284                }
285                let dst = (y * w + x) * 4;
286                for c in 0..4 {
287                    out.pixels[dst + c] = (sums[c] / count) as u8;
288                }
289            }
290        }
291        out
292    }
293    /// Sharpen the image with a 3x3 kernel and return a new image.
294    pub fn sharpen(&self) -> ImageData {
295        let w = self.width as i32;
296        let h = self.height as i32;
297        let mut out = ImageData::new(self.width, self.height);
298        let clamp_coord = |v: i32, max: i32| v.clamp(0, max - 1) as u32;
299        let get = |px: i32, py: i32, c: usize| {
300            let x = clamp_coord(px, w);
301            let y = clamp_coord(py, h);
302            self.pixels[((y * self.width + x) * 4) as usize + c]
303        };
304        for y in 0..h {
305            for x in 0..w {
306                let src_a = get(x, y, 3);
307                for c in 0..3usize {
308                    let center = get(x, y, c) as i32;
309                    let top = get(x, y - 1, c) as i32;
310                    let bottom = get(x, y + 1, c) as i32;
311                    let left = get(x - 1, y, c) as i32;
312                    let right = get(x + 1, y, c) as i32;
313                    let v = (5 * center - top - bottom - left - right).clamp(0, 255) as u8;
314                    out.pixels[((y * w + x) * 4) as usize + c] = v;
315                }
316                out.pixels[((y * w + x) * 4) as usize + 3] = src_a;
317            }
318        }
319        out
320    }

325    /// Resize the image with the requested filter and return `None` for zero-sized output.
326    pub fn resize_with_filter(
327        &self,
328        new_w: u32,
329        new_h: u32,
330        filter: ResizeFilter,
331    ) -> Option<ImageData> {
332        if new_w == 0 || new_h == 0 {
333            return None;
334        }
335        match filter {
336            ResizeFilter::Bilinear => self.resize_bilinear(new_w, new_h),
337            ResizeFilter::Lanczos3 => Some(self.resize_lanczos3(new_w, new_h)),
338        }
339    }
340    /// Resize the image with bilinear sampling.
341    fn resize_bilinear(&self, new_w: u32, new_h: u32) -> Option<ImageData> {
342        let src_w = self.width as f32;
343        let src_h = self.height as f32;
344        let mut out = ImageData::new(new_w, new_h);
345        for dy in 0..new_h {
346            for dx in 0..new_w {
347                let sx = (dx as f32 + 0.5) * src_w / new_w as f32 - 0.5;
348                let sy = (dy as f32 + 0.5) * src_h / new_h as f32 - 0.5;
349                let x0 = sx.floor() as i32;
350                let y0 = sy.floor() as i32;
351                let x1 = x0 + 1;
352                let y1 = y0 + 1;
353                let tx = sx - sx.floor();
354                let ty = sy - sy.floor();
355                let clamp_x = |x: i32| x.clamp(0, self.width as i32 - 1) as u32;
356                let clamp_y = |y: i32| y.clamp(0, self.height as i32 - 1) as u32;
357                let get = |px: u32, py: u32, c: usize| -> f32 {
358                    self.pixels[((py * self.width + px) * 4) as usize + c] as f32
359                };
360                let cx0 = clamp_x(x0);
361                let cx1 = clamp_x(x1);
362                let cy0 = clamp_y(y0);
363                let cy1 = clamp_y(y1);
364                let dst_idx = ((dy * new_w + dx) * 4) as usize;
365                for c in 0..4usize {
366                    let top = get(cx0, cy0, c) * (1.0 - tx) + get(cx1, cy0, c) * tx;
367                    let bot = get(cx0, cy1, c) * (1.0 - tx) + get(cx1, cy1, c) * tx;
368                    out.pixels[dst_idx + c] = (top * (1.0 - ty) + bot * ty).round() as u8;
369                }
370            }
371        }
372        Some(out)
373    }
374    /// Resize the image with Lanczos3 sampling.
375    fn resize_lanczos3(&self, new_w: u32, new_h: u32) -> ImageData {
376        let a = 3.0f32;
377        let src_w = self.width as i32;
378        let src_h = self.height as i32;
379        let scale_x = self.width as f32 / new_w as f32;
380        let scale_y = self.height as f32 / new_h as f32;
381        let mut out = ImageData::new(new_w, new_h);
382        for dy in 0..new_h {
383            let sy = (dy as f32 + 0.5) * scale_y - 0.5;
384            let y0 = sy.floor() as i32;
385            for dx in 0..new_w {
386                let sx = (dx as f32 + 0.5) * scale_x - 0.5;
387                let x0 = sx.floor() as i32;
388                let mut accum = [0.0f32; 4];
389                let mut total_w = 0.0f32;
390                for ky in (y0 - 2)..=(y0 + 3) {
391                    let cy = ky.clamp(0, src_h - 1) as u32;
392                    let wy = lanczos_weight(sy - ky as f32, a);
393                    if wy.abs() < 1.0e-6 {
394                        continue;
395                    }
396                    for kx in (x0 - 2)..=(x0 + 3) {
397                        let cx = kx.clamp(0, src_w - 1) as u32;
398                        let wx = lanczos_weight(sx - kx as f32, a);
399                        let w = wx * wy;
400                        if w.abs() < 1.0e-6 {
401                            continue;
402                        }
403                        let si = ((cy * self.width + cx) * 4) as usize;
404                        for (c, channel) in accum.iter_mut().enumerate() {
405                            *channel += self.pixels[si + c] as f32 * w;
406                        }
407                        total_w += w;
408                    }
409                }
410                let di = ((dy * new_w + dx) * 4) as usize;
411                if total_w.abs() > 1.0e-6 {
412                    for (c, channel) in accum.iter().enumerate() {
413                        out.pixels[di + c] = (*channel / total_w).round().clamp(0.0, 255.0) as u8;
414                    }
415                } else {
416                    let sxn = sx.round().clamp(0.0, self.width as f32 - 1.0) as u32;
417                    let syn = sy.round().clamp(0.0, self.height as f32 - 1.0) as u32;
418                    let si = ((syn * self.width + sxn) * 4) as usize;
419                    out.pixels[di..di + 4].copy_from_slice(&self.pixels[si..si + 4]);
420                }
421            }
422        }
423        out
424    }
425    /// Blend another image into this image using alpha or overwrite when fully opaque.
426    pub fn blit(&mut self, src: &ImageData, dst_x: i32, dst_y: i32) {
427        let dw = self.width as i32;
428        let dh = self.height as i32;
429        let sw = src.width as i32;
430        let sh = src.height as i32;
431        let fully_opaque = src.pixels.chunks_exact(4).all(|px| px[3] == 255);
432        if fully_opaque {
433            for sy in 0..sh {
434                let dy = dst_y + sy;
435                if dy < 0 || dy >= dh {
436                    continue;
437                }
438                let mut sx0 = 0;
439                let mut dx0 = dst_x;
440                if dx0 < 0 {
441                    sx0 = -dx0;
442                    dx0 = 0;
443                }
444                let row_len = (sw - sx0).min(dw - dx0);
445                if row_len <= 0 {
446                    continue;
447                }
448                let si = ((sy * sw + sx0) * 4) as usize;
449                let di = ((dy * dw + dx0) * 4) as usize;
450                let bytes = (row_len as usize) * 4;
451                self.pixels[di..di + bytes].copy_from_slice(&src.pixels[si..si + bytes]);
452            }
453            return;
454        }
455        for sy in 0..sh {
456            let dy = dst_y + sy;
457            if dy < 0 || dy >= dh {
458                continue;
459            }
460            for sx in 0..sw {
461                let dx = dst_x + sx;
462                if dx < 0 || dx >= dw {
463                    continue;
464                }
465                let si = ((sy * sw + sx) * 4) as usize;
466                let di = ((dy * dw + dx) * 4) as usize;
467                let sa = src.pixels[si + 3] as f32 / 255.0;
468                if sa <= 0.0 {
469                    continue;
470                }
471                let da = self.pixels[di + 3] as f32 / 255.0;
472                let out_a = sa + da * (1.0 - sa);
473                if out_a <= 0.0 {
474                    continue;
475                }
476                for c in 0..3usize {
477                    let s = src.pixels[si + c] as f32 / 255.0;
478                    let d = self.pixels[di + c] as f32 / 255.0;
479                    let out_c = (s * sa + d * da * (1.0 - sa)) / out_a;
480                    self.pixels[di + c] = (out_c * 255.0).round().clamp(0.0, 255.0) as u8;
481                }
482                self.pixels[di + 3] = (out_a * 255.0).round().clamp(0.0, 255.0) as u8;
483            }
484        }
485    }
486    /// Draw a nine-slice region from a source image into this image.
487    #[allow(clippy::too_many_arguments)]
488    pub fn draw_nine_slice(
489        &mut self,
490        src: &ImageData,
491        src_x: u32,
492        src_y: u32,
493        src_w: u32,
494        src_h: u32,
495        dst_x: i32,
496        dst_y: i32,
497        dst_w: u32,
498        dst_h: u32,
499        inset_left: u32,
500        inset_right: u32,
501        inset_top: u32,
502        inset_bottom: u32,
503    ) -> Result<(), String> {
504        if dst_w == 0 || dst_h == 0 {
505            return Err("draw_nine_slice: destination width/height must be > 0".into());
506        }
507        if src_w == 0 || src_h == 0 {
508            return Err("draw_nine_slice: source width/height must be > 0".into());
509        }
510        if src_x + src_w > src.width || src_y + src_h > src.height {
511            return Err("draw_nine_slice: source rect out of bounds".into());
512        }
513        if inset_left + inset_right > src_w || inset_top + inset_bottom > src_h {
514            return Err("draw_nine_slice: insets exceed source region size".into());
515        }
516        let src_center_w = src_w.saturating_sub(inset_left + inset_right);
517        let src_center_h = src_h.saturating_sub(inset_top + inset_bottom);
518        let dst_center_w = dst_w.saturating_sub(inset_left + inset_right);
519        let dst_center_h = dst_h.saturating_sub(inset_top + inset_bottom);
520        let draw_patch = |dst: &mut ImageData,
521                          sx: u32,
522                          sy: u32,
523                          sw: u32,
524                          sh: u32,
525                          dx: i32,
526                          dy: i32,
527                          dw: u32,
528                          dh: u32|
529         -> Result<(), String> {
530            if sw == 0 || sh == 0 || dw == 0 || dh == 0 {
531                return Ok(());
532            }
533            let patch = src
534                .get_region(sx, sy, sw, sh)
535                .ok_or_else(|| "draw_nine_slice: failed to extract source patch".to_string())?;
536            let patch = if sw == dw && sh == dh {
537                patch
538            } else {
539                patch.resize_nearest(dw, dh)
540            };
541            dst.blit(&patch, dx, dy);
542            Ok(())
543        };
544        let s_left = inset_left;
545        let s_right = inset_right;
546        let s_top = inset_top;
547        let s_bottom = inset_bottom;
548        let s_mid_x = src_x + s_left;
549        let s_mid_y = src_y + s_top;
550        let s_right_x = src_x + src_w - s_right;
551        let s_bottom_y = src_y + src_h - s_bottom;
552        let d_mid_x = dst_x + s_left as i32;
553        let d_mid_y = dst_y + s_top as i32;
554        let d_right_x = dst_x + (s_left + dst_center_w) as i32;
555        let d_bottom_y = dst_y + (s_top + dst_center_h) as i32;
556        draw_patch(
557            self, src_x, src_y, s_left, s_top, dst_x, dst_y, s_left, s_top,
558        )?;
559        draw_patch(
560            self,
561            s_mid_x,
562            src_y,
563            src_center_w,
564            s_top,
565            d_mid_x,
566            dst_y,
567            dst_center_w,
568            s_top,
569        )?;
570        draw_patch(
571            self, s_right_x, src_y, s_right, s_top, d_right_x, dst_y, s_right, s_top,
572        )?;
573        draw_patch(
574            self,
575            src_x,
576            s_mid_y,
577            s_left,
578            src_center_h,
579            dst_x,
580            d_mid_y,
581            s_left,
582            dst_center_h,
583        )?;
584        draw_patch(
585            self,
586            s_mid_x,
587            s_mid_y,
588            src_center_w,
589            src_center_h,
590            d_mid_x,
591            d_mid_y,
592            dst_center_w,
593            dst_center_h,
594        )?;
595        draw_patch(
596            self,
597            s_right_x,
598            s_mid_y,
599            s_right,
600            src_center_h,
601            d_right_x,
602            d_mid_y,
603            s_right,
604            dst_center_h,
605        )?;
606        draw_patch(
607            self, src_x, s_bottom_y, s_left, s_bottom, dst_x, d_bottom_y, s_left, s_bottom,
608        )?;
609        draw_patch(
610            self,
611            s_mid_x,
612            s_bottom_y,
613            src_center_w,
614            s_bottom,
615            d_mid_x,
616            d_bottom_y,
617            dst_center_w,
618            s_bottom,
619        )?;
620        draw_patch(
621            self, s_right_x, s_bottom_y, s_right, s_bottom, d_right_x, d_bottom_y, s_right,
622            s_bottom,
623        )?;
624        Ok(())
625    }
626    /// Return a copied rectangular region or `None` when out of bounds.
627    pub fn get_region(&self, x: u32, y: u32, w: u32, h: u32) -> Option<ImageData> {
628        if w == 0 || h == 0 || x + w > self.width || y + h > self.height {
629            return None;
630        }
631        let mut out = ImageData::new(w, h);
632        for row in 0..h {
633            let src_off = ((y + row) * self.width + x) as usize * 4;
634            let dst_off = (row * w) as usize * 4;
635            let len = w as usize * 4;
636            out.pixels[dst_off..dst_off + len]
637                .copy_from_slice(&self.pixels[src_off..src_off + len]);
638        }
639        Some(out)
640    }
641    /// Compute a bytewise difference score between two images.
642    pub fn diff(&self, other: &ImageData) -> u32 {
643        let same_dims = self.width == other.width && self.height == other.height;
644        if same_dims {
645            self.pixels
646                .iter()
647                .zip(other.pixels.iter())
648                .map(|(&a, &b)| (a as i32 - b as i32).unsigned_abs())
649                .sum()
650        } else {
651            let shared_w = self.width.min(other.width);
652            let shared_h = self.height.min(other.height);
653            let mut total = 0u32;
654            for row in 0..shared_h {
655                for col in 0..shared_w {
656                    let ai = ((row * self.width + col) * 4) as usize;
657                    let bi = ((row * other.width + col) * 4) as usize;
658                    for c in 0..4usize {
659                        total += (self.pixels[ai + c] as i32 - other.pixels[bi + c] as i32)
660                            .unsigned_abs();
661                    }
662                }
663            }
664            let extra_self = (self.width * self.height - shared_w * shared_h) * 4 * 255;
665            let extra_other = (other.width * other.height - shared_w * shared_h) * 4 * 255;
666            total + extra_self + extra_other
667        }
668    }
669    /// Convolve the image with a square kernel and return an error on invalid input.
670    #[allow(clippy::needless_range_loop)]
671    pub fn convolve(&self, kernel: &[f64], ksize: usize) -> Result<ImageData, String> {
672        if ksize == 0 {
673            return Err("ksize must be >= 1".into());
674        }
675        if ksize.is_multiple_of(2) {
676            return Err(format!("ksize must be odd, got {}", ksize));
677        }
678        if kernel.len() != ksize * ksize {
679            return Err(format!(
680                "kernel length {} does not match ksize*ksize ({})",
681                kernel.len(),
682                ksize * ksize
683            ));
684        }
685        let w = self.width as i32;
686        let h = self.height as i32;
687        let half = (ksize / 2) as i32;
688        let mut out = ImageData::new(self.width, self.height);
689        let clamp_x = |x: i32| x.clamp(0, w - 1) as u32;
690        let clamp_y = |y: i32| y.clamp(0, h - 1) as u32;
691        let get = |px: u32, py: u32, c: usize| -> f64 {
692            self.pixels[((py * self.width + px) * 4) as usize + c] as f64
693        };
694        for py in 0..h {
695            for px in 0..w {
696                let src_a = get(clamp_x(px), clamp_y(py), 3) as u8;
697                let mut acc = [0.0f64; 3];
698                for ky in 0..ksize {
699                    for kx in 0..ksize {
700                        let sx = px + kx as i32 - half;
701                        let sy = py + ky as i32 - half;
702                        let w_val = kernel[ky * ksize + kx];
703                        for c in 0..3usize {
704                            acc[c] += get(clamp_x(sx), clamp_y(sy), c) * w_val;
705                        }
706                    }
707                }
708                let idx = ((py * w) as usize + px as usize) * 4;
709                for c in 0..3usize {
710                    out.pixels[idx + c] = acc[c].clamp(0.0, 255.0).round() as u8;
711                }
712                out.pixels[idx + 3] = src_a;
713            }
714        }
715        Ok(out)
716    }
