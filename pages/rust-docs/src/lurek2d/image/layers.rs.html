1//! Named image layers with opacity, visibility, and RGBA pixel data.
2//!
3//! - Layered image stack that composites layers front-to-back with alpha blending.
4//! - Layer manipulation: add, remove, reorder, swap, rename, set opacity/visibility.
5//! - Final merge produces a single `ImageData` using standard Porter-Duff over compositing.
6
7use super::image_data::ImageData;
8/// A single named image layer with opacity, visibility, and pixel data.
9#[derive(Debug, Clone)]
10pub struct ImageLayer {
11    /// Layer display name.
12    pub name: String,
13    /// Layer opacity in the 0.0-1.0 range.
14    pub opacity: f32,
15    /// Whether the layer participates in compositing.
16    pub visible: bool,
17    /// Layer pixel data.
18    pub data: ImageData,
19}
20impl ImageLayer {
21    /// Create a visible opaque layer with a blank canvas of the given size.
22    pub fn new(name: impl Into<String>, width: u32, height: u32) -> Self {
23        Self {
24            name: name.into(),
25            opacity: 1.0,
26            visible: true,
27            data: ImageData::new(width, height),
28        }
29    }
30}
31/// A same-sized stack of image layers that can be merged into one image.
32#[derive(Debug, Clone)]
33pub struct LayeredImage {
34    /// Canvas width in pixels.
35    pub(super) width: u32,
36    /// Canvas height in pixels.
37    pub(super) height: u32,
38    /// Ordered layer list from back to front.
39    pub(super) layers: Vec<ImageLayer>,
40}
41impl LayeredImage {
42    /// Create an empty layered image with the given canvas size.
43    pub fn new(width: u32, height: u32) -> Self {
44        Self {
45            width,
46            height,
47            layers: Vec::new(),
48        }
49    }
50    /// Return the canvas width in pixels.
51    pub fn width(&self) -> u32 {
52        self.width
53    }
54    /// Return the canvas height in pixels.
55    pub fn height(&self) -> u32 {
56        self.height
57    }
58    /// Return the number of layers in the stack.
59    pub fn layer_count(&self) -> usize {
60        self.layers.len()
61    }
62    /// Append a new blank layer and return its index.
63    pub fn add_layer(&mut self, name: impl Into<String>) -> usize {
64        self.layers
65            .push(ImageLayer::new(name, self.width, self.height));
66        self.layers.len() - 1
67    }
68    /// Remove a layer by index and return it when present.
69    pub fn remove_layer(&mut self, index: usize) -> Option<ImageLayer> {
70        if index < self.layers.len() {
71            Some(self.layers.remove(index))
72        } else {
73            None
74        }
75    }
76    /// Return a layer by index. This function is part of the public API.
77    pub fn get_layer(&self, index: usize) -> Option<&ImageLayer> {
78        self.layers.get(index)
79    }
80    /// Return a mutable layer by index.
81    pub fn get_layer_mut(&mut self, index: usize) -> Option<&mut ImageLayer> {
82        self.layers.get_mut(index)
83    }
84    /// Set a layer opacity and clamp it to the valid range.
85    pub fn set_opacity(&mut self, index: usize, opacity: f32) -> bool {
86        if let Some(layer) = self.layers.get_mut(index) {
87            layer.opacity = opacity.clamp(0.0, 1.0);
88            true
89        } else {
90            false
91        }
92    }
93    /// Set a layer visibility flag and return whether the layer existed.
94    pub fn set_visible(&mut self, index: usize, visible: bool) -> bool {
95        if let Some(layer) = self.layers.get_mut(index) {
96            layer.visible = visible;
97            true
98        } else {
99            false
100        }
101    }
102    /// Rename a layer and return whether the layer existed.
103    pub fn set_name(&mut self, index: usize, name: impl Into<String>) -> bool {
104        if let Some(layer) = self.layers.get_mut(index) {
105            layer.name = name.into();
106            true
107        } else {
108            false
109        }
110    }
111    /// Replace a layer image with a copied source image or a pasted canvas copy.
112    pub fn set_layer_image(&mut self, index: usize, source: &ImageData) -> bool {
113        if let Some(layer) = self.layers.get_mut(index) {
114            if source.width() == self.width && source.height() == self.height {
115                layer.data = source.clone();
116            } else {
117                let mut canvas = ImageData::new(self.width, self.height);
118                canvas.paste(source, 0, 0);
119                layer.data = canvas;
120            }
121            true
122        } else {
123            false
124        }
125    }
126    /// Swap two layers and return false when either index is invalid.
127    pub fn swap_layers(&mut self, a: usize, b: usize) -> bool {
128        let len = self.layers.len();
129        if a >= len || b >= len || a == b {
130            return false;
131        }
132        self.layers.swap(a, b);
133        true
134    }
135    /// Move a layer to another position and return false when either index is invalid.
136    pub fn move_layer(&mut self, from_index: usize, to_index: usize) -> bool {
137        let len = self.layers.len();
138        if from_index >= len || to_index >= len {
139            return false;
140        }
141        let layer = self.layers.remove(from_index);
142        self.layers.insert(to_index, layer);
143        true
144    }
145    /// Merge visible layers front-to-back into a new image.
146    pub fn merge(&self) -> ImageData {
147        let mut result = ImageData::new(self.width, self.height);
148        let pixels_len = (self.width * self.height * 4) as usize;
149        let dst = result.pixels.as_mut_slice();
150        for layer in &self.layers {
151            if !layer.visible {
152                continue;
153            }
154            let src = layer.data.pixels.as_slice();
155            let src_len = src.len().min(pixels_len);
156            let px_count = src_len / 4;
157            for i in 0..px_count {
158                let si = i * 4;
159                let sr = src[si] as f32;
160                let sg = src[si + 1] as f32;
161                let sb = src[si + 2] as f32;
162                let sa = src[si + 3] as f32 / 255.0 * layer.opacity;
163                if sa <= 0.0 {
164                    continue;
165                }
166                let dr = dst[si] as f32;
167                let dg = dst[si + 1] as f32;
168                let db = dst[si + 2] as f32;
169                let da = dst[si + 3] as f32 / 255.0;
170                let out_a = sa + da * (1.0 - sa);
171                if out_a <= 0.0 {
172                    dst[si] = 0;
173                    dst[si + 1] = 0;
174                    dst[si + 2] = 0;
175                    dst[si + 3] = 0;
176                } else {
177                    dst[si] = ((sr * sa + dr * da * (1.0 - sa)) / out_a).round() as u8;
178                    dst[si + 1] = ((sg * sa + dg * da * (1.0 - sa)) / out_a).round() as u8;
179                    dst[si + 2] = ((sb * sa + db * da * (1.0 - sa)) / out_a).round() as u8;
180                    dst[si + 3] = (out_a * 255.0).round().min(255.0) as u8;
181                }
182            }
183        }
184        result
185    }
186}
