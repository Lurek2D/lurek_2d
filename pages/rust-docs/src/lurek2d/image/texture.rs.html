1//! CPU-side texture loading, decoding, and storage into the SlotMap pool.
2//!
3//! - Premultiplied-alpha conversion for correct blending on the GPU.
4//! - Color-space tagging (sRGB vs linear) carried alongside pixel data.
5//! - Construction from file paths or raw RGBA byte buffers.
6//! - Dimension validation for caller-supplied pixel buffers.
7
8use crate::log_msg;
9use crate::render::renderer::TextureData;
10use crate::runtime::error::{EngineError, EngineResult};
11use crate::runtime::log_messages::TX01_TEX_DECODED;
12use crate::runtime::resource_keys::TextureKey;
13use slotmap::SlotMap;
14use std::path::Path;
15/// Texture color space stored alongside decoded pixels.
16#[derive(Clone, Copy, Debug, PartialEq, Eq)]
17pub enum TextureColorSpace {
18    /// Standard sRGB texture data.
19    Srgb,
20    /// Linear texture data.
21    Linear,
22}
23/// CPU-side texture handle and its uploaded dimensions.
24pub struct Texture {
25    /// Slot-map key for the backing texture data.
26    pub key: TextureKey,
27    /// Texture width in pixels.
28    pub width: u32,
29    /// Texture height in pixels.
30    pub height: u32,
31}
32/// Premultiply RGB channels by alpha for each RGBA pixel in place.
33pub fn premultiply_alpha_rgba8_in_place(pixels: &mut [u8]) {
34    for chunk in pixels.chunks_exact_mut(4) {
35        let a = chunk[3] as f32 / 255.0;
36        chunk[0] = (chunk[0] as f32 * a) as u8;
37        chunk[1] = (chunk[1] as f32 * a) as u8;
38        chunk[2] = (chunk[2] as f32 * a) as u8;
39    }
40}
41impl Texture {
42    /// Parse a texture color-space label and return the matching enum value.
43    pub fn parse_color_space(value: &str) -> Option<TextureColorSpace> {
44        match value.to_ascii_lowercase().as_str() {
45            "srgb" => Some(TextureColorSpace::Srgb),
46            "linear" => Some(TextureColorSpace::Linear),
47            _ => None,
48        }
49    }
50    /// Load a texture with sRGB color space by default.
51    pub fn load<P: AsRef<Path>>(
52        path: P,
53        textures: &mut SlotMap<TextureKey, TextureData>,
54    ) -> EngineResult<Self> {
55        Self::load_with_color_space(path, textures, TextureColorSpace::Srgb)
56    }
57    /// Load a texture from disk, premultiply alpha, and store it in the texture pool.
58    pub fn load_with_color_space<P: AsRef<Path>>(
59        path: P,
60        textures: &mut SlotMap<TextureKey, TextureData>,
61        color_space: TextureColorSpace,
62    ) -> EngineResult<Self> {
63        let img = ::image::open(&path).map_err(|e| {
64            let path_str = path.as_ref().display().to_string();
65            if path_str.contains("No such file") || matches!(e, ::image::ImageError::IoError(_)) {
66                EngineError::ResourceNotFound(format!("{}: {}", path_str, e))
67            } else {
68                EngineError::RenderError(format!("Failed to decode image '{}': {}", path_str, e))
69            }
70        })?;
71        let rgba = img.to_rgba8();
72        let (width, height) = rgba.dimensions();
73        let mut pixels = rgba.into_raw();
74        premultiply_alpha_rgba8_in_place(&mut pixels);
75        let key = textures.insert(TextureData {
76            pixels,
77            width,
78            height,
79            color_space,
80        });
81        log_msg!(debug, TX01_TEX_DECODED, "{}x{}", width, height);
82        Ok(Texture { key, width, height })
83    }
84    /// Create a texture from RGBA bytes using sRGB color space.
85    pub fn from_rgba(
86        width: u32,
87        height: u32,
88        pixels: Vec<u8>,
89        textures: &mut SlotMap<TextureKey, TextureData>,
90    ) -> EngineResult<Self> {
91        Self::from_rgba_with_color_space(width, height, pixels, textures, TextureColorSpace::Srgb)
92    }
93    /// Create a texture from RGBA bytes and store it in the texture pool.
94    pub fn from_rgba_with_color_space(
95        width: u32,
96        height: u32,
97        mut pixels: Vec<u8>,
98        textures: &mut SlotMap<TextureKey, TextureData>,
99        color_space: TextureColorSpace,
100    ) -> EngineResult<Self> {
101        let expected = (width * height * 4) as usize;
102        if pixels.len() != expected {
103            return Err(EngineError::RenderError(format!(
104                "Expected {} bytes for {}x{} RGBA, got {}",
105                expected,
106                width,
107                height,
108                pixels.len()
109            )));
110        }
111        premultiply_alpha_rgba8_in_place(&mut pixels);
112        let key = textures.insert(TextureData {
113            pixels,
114            width,
115            height,
116            color_space,
117        });
118        Ok(Texture { key, width, height })
119    }
120}
