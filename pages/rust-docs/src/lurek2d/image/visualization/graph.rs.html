9#[allow(clippy::too_many_arguments)]
10/// Render a node-edge graph with labels and transfer-weight edges into an image.
11pub fn draw_graph_operations_to_image(
12    positions: &[(f32, f32)],
13    labels: &[&str],
14    colors: &[(u8, u8, u8)],
15    edges: &[(usize, usize)],
16    removed_edges: &[(usize, usize)],
17    stats_text: &str,
18    title: &str,
19    width: u32,
20    height: u32,
21) -> ImageData {
22    let mut img = ImageData::new(width, height);
23    img.fill(25, 25, 35, 255);
24    for &(a, b) in edges {
25        if a < positions.len() && b < positions.len() {
26            let (ax, ay) = positions[a];
27            let (bx, by) = positions[b];
28            img.draw_line(
29                ax as i32, ay as i32, bx as i32, by as i32, 80, 120, 180, 200,
30            );
31        }
32    }
33    for &(a, b) in removed_edges {
34        if a < positions.len() && b < positions.len() {
35            let (ax, ay) = positions[a];
36            let (bx, by) = positions[b];
37            img.draw_line(ax as i32, ay as i32, bx as i32, by as i32, 80, 40, 40, 100);
38        }
39    }
40    for (i, &(px, py)) in positions.iter().enumerate() {
41        let (r, g, b) = if i < colors.len() {
42            colors[i]
43        } else {
44            (180, 180, 180)
45        };
46        img.draw_circle(px as i32, py as i32, 14, r, g, b, 255);
47        if i < labels.len() {
48            img.draw_label(labels[i], (px - 30.0) as i32, (py + 18.0) as i32, r, g, b);
49        }
50    }
51    img.draw_label(stats_text, 10, (height - 20) as i32, 200, 200, 200);
52    img.draw_label(title, (width / 2 - 40) as i32, 5, 100, 255, 100);
53    img
54}
55#[allow(clippy::too_many_arguments)]
56/// Render an item-flow graph with nodes, flows, accumulations, and path highlights into an image.
57pub fn draw_graph_item_flow_to_image(
58    node_pos: &[(f32, f32)],
59    node_names: &[&str],
60    node_colors: &[(u8, u8, u8)],
61    items: &[(i32, i32, u8, u8, u8, &str)],
62    stats_text: &str,
63    title: &str,
64    width: u32,
65    height: u32,
66) -> ImageData {
67    let mut img = ImageData::new(width, height);
68    img.fill(25, 25, 35, 255);
69    for i in 0..node_pos.len().saturating_sub(1) {
70        let (ax, ay) = node_pos[i];
71        let (bx, by) = node_pos[i + 1];
72        let start_x = ax as i32 + 30;
73        let end_x = bx as i32 - 30;
74        img.draw_line(start_x, ay as i32, end_x, by as i32, 100, 150, 200, 200);
75        img.draw_line(
76            end_x - 10,
77            by as i32 - 5,
78            end_x,
79            by as i32,
80            100,
81            150,
82            200,
83            200,
84        );
85        img.draw_line(
86            end_x - 10,
87            by as i32 + 5,
88            end_x,
89            by as i32,
90            100,
91            150,
92            200,
93            200,
94        );
95    }
96    for (i, &(px, py)) in node_pos.iter().enumerate() {
97        let (r, g, b) = if i < node_colors.len() {
98            node_colors[i]
99        } else {
100            (180, 180, 180)
101        };
102        img.draw_circle(px as i32, py as i32, 20, r, g, b, 255);
103        if i < node_names.len() {
104            img.draw_label(
105                node_names[i],
106                (px - 22.0) as i32,
107                (py + 25.0) as i32,
108                r,
109                g,
110                b,
111            );
112        }
113    }
114    for &(ix, iy, ir, ig, ib, label) in items {
115        img.draw_circle(ix, iy, 6, ir, ig, ib, 255);
116        img.draw_label(label, ix - 10, iy - 15, ir, ig, ib);
117    }
118    img.draw_label(title, (width / 2 - 40) as i32, 5, 100, 255, 100);
119    img.draw_label(stats_text, 10, (height - 20) as i32, 200, 200, 200);
120    img
121}
