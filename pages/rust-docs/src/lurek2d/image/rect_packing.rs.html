1//! Shelf-first rectangle packing for texture atlas layout.
2//!
3//! - Configurable atlas dimensions and uniform pixel padding between rects.
4//! - Tracks occupancy ratio and returns placement coordinates in insertion order.
5
6/// Placement result for a single packed rectangle.
7#[derive(Debug, Clone, PartialEq, Eq)]
8pub struct PackedRect {
9    /// Optional label assigned by the caller for mapping back to source assets.
10    pub id: Option<String>,
11    /// Left edge of the placed rectangle within the atlas, in pixels.
12    pub x: u32,
13    /// Top edge of the placed rectangle within the atlas, in pixels.
14    pub y: u32,
15    /// Width of the content area, not including padding.
16    pub w: u32,
17    /// Height of the content area, not including padding.
18    pub h: u32,
19}
20
21/// Internal shelf row tracking its Y position, height, and fill cursor.
22#[derive(Debug, Clone)]
23struct Shelf {
24    /// Y origin of this shelf in the atlas.
25    y: u32,
26    /// Row height reserved for this shelf (tallest rect + padding placed so far).
27    height: u32,
28    /// X position of the next free slot in this shelf.
29    cursor_x: u32,
30}
31
32/// Shelf-first texture-atlas packer with configurable atlas dimensions and uniform padding.
33#[derive(Debug, Clone)]
34pub struct RectPacker {
35    /// Atlas width in pixels.
36    width: u32,
37    /// Atlas height in pixels.
38    height: u32,
39    /// Uniform pixel gap added to every rect on the right and bottom.
40    padding: u32,
41    /// Ordered list of shelves from top to bottom.
42    shelves: Vec<Shelf>,
43    /// All successfully placed rects in insertion order.
44    packed: Vec<PackedRect>,
45}
46impl RectPacker {
47    /// Construct a packer with the given atlas `width × height` and uniform `padding`.
48    pub fn new(width: u32, height: u32, padding: u32) -> Self {
49        Self {
50            width,
51            height,
52            padding,
53            shelves: Vec::new(),
54            packed: Vec::new(),
55        }
56    }
57    /// Place a `w × h` rectangle with optional `id` label; returns placement or `None` when no space remains.
58    pub fn pack(&mut self, w: u32, h: u32, id: Option<String>) -> Option<PackedRect> {
59        if w == 0 || h == 0 {
60            return None;
61        }
62        let pw = w.saturating_add(self.padding);
63        let ph = h.saturating_add(self.padding);
64        if pw > self.width || ph > self.height {
65            return None;
66        }
67        for shelf in &mut self.shelves {
68            if ph <= shelf.height && shelf.cursor_x.saturating_add(pw) <= self.width {
69                let placed = PackedRect {
70                    id,
71                    x: shelf.cursor_x,
72                    y: shelf.y,
73                    w,
74                    h,
75                };
76                shelf.cursor_x = shelf.cursor_x.saturating_add(pw);
77                self.packed.push(placed.clone());
78                return Some(placed);
79            }
80        }
81        let next_y = self
82            .shelves
83            .last()
84            .map(|s| s.y.saturating_add(s.height))
85            .unwrap_or(0);
86        if next_y.saturating_add(ph) > self.height {
87            return None;
88        }
89        let mut shelf = Shelf {
90            y: next_y,
91            height: ph,
92            cursor_x: 0,
93        };
94        let placed = PackedRect {
95            id,
96            x: 0,
97            y: next_y,
98            w,
99            h,
100        };
101        shelf.cursor_x = pw;
102        self.shelves.push(shelf);
103        self.packed.push(placed.clone());
104        Some(placed)
105    }
106    /// Remove all placed rects and reset shelves, keeping atlas dimensions unchanged.
107    pub fn clear(&mut self) {
108        self.shelves.clear();
109        self.packed.clear();
110    }
111    /// Return a slice of all successfully placed rects in insertion order.
112    pub fn packed_rects(&self) -> &[PackedRect] {
113        &self.packed
114    }
115    /// Return the fraction `[0,1]` of the atlas area covered by placed rects, excluding padding.
116    pub fn occupancy(&self) -> f32 {
117        if self.width == 0 || self.height == 0 {
118            return 0.0;
119        }
120        let used: u64 = self
121            .packed
122            .iter()
123            .map(|r| (r.w as u64) * (r.h as u64))
124            .sum();
125        let total = (self.width as u64) * (self.height as u64);
126        (used as f32 / total as f32).clamp(0.0, 1.0)
127    }
128    /// Return the atlas dimensions as `(width, height)` in pixels.
129    pub fn size(&self) -> (u32, u32) {
130        (self.width, self.height)
131    }
132    /// Return the uniform padding value used during packing.
133    pub fn padding(&self) -> u32 {
134        self.padding
135    }
136}
