1//! Container widgets for the retained-mode GUI: Panel, Layout, ScrollPanel, NinePatch, GUIWindow, SplitPanel, DockPanel.
2//!
3//! - Layout engine supports vertical, horizontal, and grid stacking with spacing, alignment, and justification.
4//! - ScrollPanel provides viewport clipping with clamped 2D scroll offsets and configurable speed.
5//! - NinePatch implements resolution-independent 9-slice border rendering from pixel insets.
6//! - GUIWindow adds floating window semantics: title bar, close button, drag, and resize.
7//! - SplitPanel and DockPanel offer two-pane splitting and edge-based docking respectively.
8
9use crate::ui::widget::{WidgetBase, WidgetType};
10/// Plain box container that groups children with an optional title and scroll flag.

12pub struct Panel {

39/// Direction along which a `Layout` positions its children.
40#[derive(Debug, Clone, Copy, PartialEq, Eq)]
41pub enum LayoutDirection {
42    /// Stack children top-to-bottom.
43    Vertical,
44    /// Stack children left-to-right.
45    Horizontal,
46    /// Arrange children in a column-wrapped grid.
47    Grid,
48}
49impl LayoutDirection {
50    /// Parse a lowercase string to a variant; return `None` for unrecognised values.
51    pub fn parse_str(s: &str) -> Option<Self> {
52        match s {
53            "vertical" => Some(Self::Vertical),
54            "horizontal" => Some(Self::Horizontal),
55            "grid" => Some(Self::Grid),
56            _ => None,
57        }
58    }
59    /// Return the canonical lowercase string representation of this variant.
60    pub fn as_str(&self) -> &'static str {
61        match self {
62            Self::Vertical => "vertical",
63            Self::Horizontal => "horizontal",
64            Self::Grid => "grid",
65        }
66    }
67}
68/// Container that positions children according to a `LayoutDirection`, spacing, and optional grid columns.
69#[derive(Debug, Clone)]
70pub struct Layout {
71    /// Shared layout, style, and state fields.
72    pub base: WidgetBase,
73    /// Indices of child widgets positioned by this layout.
74    pub children: Vec<usize>,
75    /// Direction along which children are stacked or gridded.
76    pub direction: LayoutDirection,
77    /// Extra pixel gap inserted between consecutive children.
78    pub spacing: f32,
79    /// Number of columns when `direction` is `Grid`; ignored otherwise.
80    pub columns: usize,
81    /// Whether children wrap to the next row/column when they exceed the layout extent.
82    pub wrap: bool,
83    /// Cross-axis alignment token, e.g. `"start"`, `"center"`, `"end"`, or `"stretch"`.
84    pub align: String,
85    /// Main-axis justification token, e.g. `"start"`, `"space-between"`.
86    pub justify: String,
87}
88impl Layout {
89    /// Create a layout with the given direction; sets spacing=0, columns=1, wrap=false.
90    pub fn new(direction: LayoutDirection) -> Self {
91        Self {
92            base: WidgetBase::new(WidgetType::Layout),
93            children: Vec::new(),
94            direction,
95            spacing: 0.0,
96            columns: 1,
97            wrap: false,
98            align: "stretch".to_string(),
99            justify: "start".to_string(),
100        }
101    }
102    /// Apply this layout's direction and spacing to position the given widget bases in-place.
103    pub fn perform_layout(&self, bases: &mut [WidgetBase]) {
104        let pad = &self.base.padding;
105        let mut cx = self.base.x + pad[3];
106        let mut cy = self.base.y + pad[0];
107        for &child_idx in &self.children {
108            if child_idx >= bases.len() {
109                continue;
110            }
111            let child = &mut bases[child_idx];
112            if !child.visible {
113                continue;
114            }
115            child.x = cx + child.margin[3];
116            child.y = cy + child.margin[0];
117            match self.direction {
118                LayoutDirection::Vertical => {
119                    cy += child.height + child.margin[0] + child.margin[2] + self.spacing;
120                }
121                LayoutDirection::Horizontal => {
122                    cx += child.width + child.margin[1] + child.margin[3] + self.spacing;
123                }
124                LayoutDirection::Grid => {
125                    let col = self
126                        .children
127                        .iter()
128                        .position(|&i| i == child_idx)
129                        .unwrap_or(0);
130                    let col_in_row = col % self.columns;
131                    if col_in_row == self.columns - 1 {
132                        cx = self.base.x + pad[3];
133                        cy += child.height + child.margin[0] + child.margin[2] + self.spacing;
134                    } else {
135                        cx += child.width + child.margin[1] + child.margin[3] + self.spacing;
136                    }
137                }
138            }
139        }
140    }
141}
142/// Provide a default vertical `Layout` via `Self::new(LayoutDirection::Vertical)`.
143impl Default for Layout {
144    fn default() -> Self {
145        Self::new(LayoutDirection::Vertical)
146    }
147}
148/// Panel with independently scrollable content area larger than its visible bounds.
149#[derive(Debug, Clone)]
150pub struct ScrollPanel {
151    /// Shared layout, style, and state fields.
152    pub base: WidgetBase,
153    /// Indices of child widgets clipped and offset by scroll position.
154    pub children: Vec<usize>,
155    /// Full scrollable content width in pixels.
156    pub content_width: f32,
157    /// Full scrollable content height in pixels.
158    pub content_height: f32,
159    /// Current horizontal scroll offset in pixels.
160    pub scroll_x: f32,
161    /// Current vertical scroll offset in pixels.
162    pub scroll_y: f32,
163    /// Pixels scrolled per wheel tick.
164    pub scroll_speed: f32,
165}
166impl ScrollPanel {
167    /// Create a scroll panel with default content size 100×100 and scroll_speed 20.
168    pub fn new() -> Self {
169        Self {
170            base: WidgetBase::new(WidgetType::ScrollPanel),
171            children: Vec::new(),
172            content_width: 100.0,
173            content_height: 100.0,
174            scroll_x: 0.0,
175            scroll_y: 0.0,
176            scroll_speed: 20.0,
177        }
178    }
179    /// Return `(max_scroll_x, max_scroll_y)` derived from content and widget dimensions.
180    pub fn max_scroll(&self) -> (f32, f32) {
181        let mx = (self.content_width - self.base.width).max(0.0);
182        let my = (self.content_height - self.base.height).max(0.0);
183        (mx, my)
184    }
185    /// Clamp `scroll_x` and `scroll_y` to the valid `[0, max_scroll]` range.
186    pub fn clamp_scroll(&mut self) {
187        let (mx, my) = self.max_scroll();
188        self.scroll_x = self.scroll_x.clamp(0.0, mx);
189        self.scroll_y = self.scroll_y.clamp(0.0, my);
190    }
191}
192/// Provide a default `ScrollPanel` via `Self::new()`.
193impl Default for ScrollPanel {
194    fn default() -> Self {
195        Self::new()
196    }
197}
198/// Eight-tuple of `(src_x, src_y, src_w, src_h, dst_x, dst_y, dst_w, dst_h)` describing one 9-patch tile region.
199pub type NineSlice = (f32, f32, f32, f32, f32, f32, f32, f32);
200/// 9-patch scalable border widget; divides a source image into 9 regions for resolution-independent borders.
201#[derive(Debug, Clone)]
202pub struct NinePatch {
203    /// Shared layout, style, and state fields.
204    pub base: WidgetBase,
205    /// Pixel inset from the left edge of the source image to the inner column.
206    pub inset_left: u32,
207    /// Pixel inset from the top edge of the source image to the inner row.
208    pub inset_top: u32,
209    /// Pixel inset from the right edge of the source image to the inner column.
210    pub inset_right: u32,
211    /// Pixel inset from the bottom edge of the source image to the inner row.
212    pub inset_bottom: u32,
213    /// Full pixel width of the source image.
214    pub image_width: u32,
215    /// Full pixel height of the source image.
216    pub image_height: u32,
217}
218impl NinePatch {
219    /// Create a nine-patch widget with zero insets and zero source image size.
220    pub fn new() -> Self {
221        Self {
222            base: WidgetBase::new(WidgetType::NinePatch),
223            inset_left: 0,
224            inset_top: 0,
225            inset_right: 0,
226            inset_bottom: 0,
227            image_width: 0,
228            image_height: 0,
229        }
230    }
231    /// Return the 9 `NineSlice` tuples describing source and destination rects for each patch region.
232    pub fn get_slices(&self) -> Vec<NineSlice> {
233        let il = self.inset_left as f32;
234        let it = self.inset_top as f32;
235        let ir = self.inset_right as f32;
236        let ib = self.inset_bottom as f32;
237        let iw = self.image_width as f32;
238        let ih = self.image_height as f32;
239        let dw = self.base.width;
240        let dh = self.base.height;
241        let dx = self.base.x;
242        let dy = self.base.y;
243        let sm = iw - il - ir;
244        let smh = ih - it - ib;
245        let dm = dw - il - ir;
246        let dmh = dh - it - ib;
247        vec![
248            (0.0, 0.0, il, it, dx, dy, il, it),
249            (il, 0.0, sm, it, dx + il, dy, dm, it),
250            (iw - ir, 0.0, ir, it, dx + dw - ir, dy, ir, it),
251            (0.0, it, il, smh, dx, dy + it, il, dmh),
252            (il, it, sm, smh, dx + il, dy + it, dm, dmh),
253            (iw - ir, it, ir, smh, dx + dw - ir, dy + it, ir, dmh),
254            (0.0, ih - ib, il, ib, dx, dy + dh - ib, il, ib),
255            (il, ih - ib, sm, ib, dx + il, dy + dh - ib, dm, ib),
256            (iw - ir, ih - ib, ir, ib, dx + dw - ir, dy + dh - ib, ir, ib),
257        ]
258    }
259}
260/// Provide a default `NinePatch` via `Self::new()`.
261impl Default for NinePatch {
262    fn default() -> Self {
263        Self::new()
264    }
265}
