1//! Utility-AI scoring model storing actions, response curves, considerations, and evaluation results.
2//!
3//! - Response-curve mapping rules and action-side data binding Lua scorers with momentum weighting.
4//! - Evaluation flow calling registered scorers, tracking per-action scores, and selecting best action.
5
6use mlua::prelude::*;
7use mlua::RegistryKey;
8/// Response-curve variant used to transform raw consideration inputs.
9#[derive(Debug, Clone, PartialEq)]
10pub enum ResponseCurve {
11    /// Output = p1 * input + p2.
12    Linear,
13    /// Output = p1 * input² + p2 * input + p3.
14    Quadratic,
15    /// Sigmoid: 1 / (1 + exp(−p1 * (input − p2))).
16    Logistic,
17    /// Log-odds transform, clamped to avoid infinities.
18    Logit,
19    /// Returns p2 when input ≥ p1, otherwise p3.
20    Step,
21    /// Delegates to a Lua callback identified by `callback_id`.
22    Custom {
23        /// Registry index of the Lua curve callback.
24        callback_id: u32,
25    },
26}
27impl ResponseCurve {
28    /// Parse a string tag into a `ResponseCurve`; unknown strings map to `Linear`.
29    pub fn parse_str(s: &str) -> Self {
30        match s {
31            "quadratic" => Self::Quadratic,
32            "logistic" => Self::Logistic,
33            "logit" => Self::Logit,
34            "step" => Self::Step,
35            _ => Self::Linear,
36        }
37    }
38    /// Evaluate the curve at `input` using shape parameters p1, p2, p3.
39    pub fn apply(&self, input: f64, p1: f64, p2: f64, p3: f64) -> f64 {
40        match self {
41            Self::Linear => p1 * input + p2,
42            Self::Quadratic => p1 * input * input + p2 * input + p3,
43            Self::Logistic => 1.0 / (1.0 + (-p1 * (input - p2)).exp()),
44            Self::Logit => {
45                let clamped = input.clamp(0.001, 0.999);
46                (clamped / (1.0 - clamped)).ln() * p1 + p2
47            }
48            Self::Step => {
49                if input >= p1 {
50                    p2
51                } else {
52                    p3
53                }
54            }
55            Self::Custom { .. } => input,
56        }
57    }
58}
59/// One consideration that transforms a raw Lua score into a weighted utility value.
60pub struct Consideration {
61    /// Unique name for this consideration, used for debug display.
62    pub name: String,
63    /// Lua callback that returns a raw input score in `[0, 1]`.
64    pub callback: RegistryKey,
65    /// Response curve that maps the raw input to a utility value.
66    pub curve: ResponseCurve,
67    /// First shape parameter for the response curve.
68    pub p1: f64,
69    /// Second shape parameter for the response curve.
70    pub p2: f64,
71    /// Third shape parameter for the response curve.
72    pub p3: f64,
73    /// Multiplicative weight applied to the curve output before product scoring.
74    pub weight: f64,
75}
76/// One candidate action scored by the utility-AI system.
77pub struct UAAction {
78    /// Unique name identifying this action.
79    pub name: String,
80    /// Lua scorer callback returning the overall action utility.
81    pub scorer: RegistryKey,
82    /// Ordered list of considerations whose outputs are multiplied together.
83    pub considerations: Vec<Consideration>,
84    /// Bonus multiplier added to the score when this action was selected last tick.
85    pub momentum_bonus: f64,
86}
87/// Utility-AI runtime that stores actions and the latest evaluation results.
88pub struct UtilityAI {
89    /// All registered actions evaluated each tick.
90    pub actions: Vec<UAAction>,
91    /// Index of the action selected on the last `evaluate` call.
92    pub last_action: Option<usize>,
93    /// Per-action weighted scores from the last `evaluate` call.
94    pub last_scores: Vec<f64>,
95}
96impl UtilityAI {
97    /// Create a `UtilityAI` with no actions.
98    pub fn new() -> Self {
99        Self {
100            actions: Vec::new(),
101            last_action: None,
102            last_scores: Vec::new(),
103        }
104    }
105    /// Register a new action with an empty consideration list.
106    pub fn add_action(&mut self, name: String, scorer: RegistryKey, momentum_bonus: f64) {
107        self.actions.push(UAAction {
108            name,
109            scorer,
110            considerations: Vec::new(),
111            momentum_bonus,
112        });
113    }
114    #[allow(clippy::too_many_arguments)]
115    /// Append a consideration to the named action; no-op if the action is not found.
116    pub fn add_consideration(
117        &mut self,
118        action_name: &str,
119        name: String,
120        callback: RegistryKey,
121        curve: &str,
122        p1: f64,
123        p2: f64,
124        p3: f64,
125        weight: f64,
126    ) {
127        if let Some(a) = self.actions.iter_mut().find(|a| a.name == action_name) {
128            a.considerations.push(Consideration {
129                name,
130                callback,
131                curve: ResponseCurve::parse_str(curve),
132                p1,
133                p2,
134                p3,
135                weight,
136            });
137        }
138    }
139    /// Return the name of the action selected on the last `evaluate` call, or `None`.
140    pub fn last_action_name(&self) -> Option<&str> {
141        self.last_action
142            .and_then(|i| self.actions.get(i))
143            .map(|a| a.name.as_str())
144    }
145    /// Call all action scorers, apply momentum, and return the best action name.
146    pub fn evaluate(&mut self, lua: &Lua) -> LuaResult<Option<String>> {
147        if self.actions.is_empty() {
148            return Ok(None);
149        }
150        let mut best_idx = 0;
151        let mut best_score = f64::NEG_INFINITY;
152        let mut scores = Vec::with_capacity(self.actions.len());
153        for (i, action) in self.actions.iter().enumerate() {
154            let func: LuaFunction = lua.registry_value(&action.scorer)?;
155            let score: f64 = func.call(())?;
156            let weighted = score * action.momentum_bonus;
157            scores.push(weighted);
158            if weighted > best_score {
159                best_score = weighted;
160                best_idx = i;
161            }
162        }
163        self.last_scores = scores;
164        self.last_action = Some(best_idx);
165        Ok(Some(self.actions[best_idx].name.clone()))
166    }
167}
168/// `Default` delegates to `UtilityAI::new`.
169impl Default for UtilityAI {
170    /// `Default` delegates to `UtilityAI::new`.
171    fn default() -> Self {
172        Self::new()
173    }
174}
