3//! - Stimulus world for visual, auditory, and custom signals with insertion, decay, and removal.

7use std::collections::HashMap;
8/// Stimulus classification used by the sensor world.
9#[derive(Debug, Clone, PartialEq, Eq, Hash)]
10pub enum StimulusType {
11    /// Visual stimulus.
12    Visual,
13    /// Audible stimulus.
14    Auditory,
15    /// Custom stimulus label.
16    Custom(String),
17}
18impl StimulusType {
19    #[allow(clippy::should_implement_trait)]
20    /// Parse a stimulus type name; unknown strings become `Custom`.
21    pub fn from_str(s: &str) -> Self {
22        match s {
23            "visual" => Self::Visual,
24            "auditory" => Self::Auditory,
25            other => Self::Custom(other.to_string()),
26        }
27    }
28    /// Return a display string for the stimulus type.
29    pub fn as_str(&self) -> String {
30        match self {
31            Self::Visual => "visual".to_string(),
32            Self::Auditory => "auditory".to_string(),
33            Self::Custom(s) => s.clone(),
34        }
35    }
36}
37/// Source stimulus stored in the world.
38pub struct Stimulus {
39    /// Stable world id.
40    pub id: u64,
41    /// Stimulus kind.
42    pub stimulus_type: StimulusType,
43    /// World position.
44    pub position: (f32, f32),
45    /// Intensity in `[0, 1]`.
46    pub intensity: f32,
47    /// Effective range.
48    pub radius: f32,
49    /// Intensity decay per second.
50    pub decay_rate: f32,
51    /// Optional source name.
52    pub source_name: Option<String>,
53    /// Optional gameplay tag.
54    pub tag: Option<String>,
55}

57pub struct DetectedStimulus {

64    /// Source intensity.
65    pub intensity: f32,

69    pub source_name: Option<String>,

73/// Container for all stimuli available to sensors.
74pub struct StimulusWorld {
75    /// Stored stimuli.
76    stimuli: Vec<Stimulus>,
77    /// Next allocated stimulus id.
78    next_id: u64,
79}
80impl StimulusWorld {
81    /// Create an empty stimulus world.
82    pub fn new() -> Self {
83        Self {
84            stimuli: Vec::new(),
85            next_id: 0,
86        }
87    }
88    /// Insert a stimulus and return its assigned id.
89    pub fn add(&mut self, stimulus: Stimulus) -> u64 {
90        let id = self.next_id;
91        self.next_id += 1;
92        let mut s = stimulus;
93        s.id = id;
94        self.stimuli.push(s);
95        id
96    }
97    /// Add a visual stimulus. This function is part of the public API.
98    pub fn add_visual(
99        &mut self,
100        x: f32,
101        y: f32,
102        intensity: f32,
103        radius: f32,
104        tag: Option<String>,
105    ) -> u64 {
106        self.add(Stimulus {
107            id: 0,
108            stimulus_type: StimulusType::Visual,
109            position: (x, y),
110            intensity,
111            radius,
112            decay_rate: 0.0,
113            source_name: None,
114            tag,
115        })
116    }
117    /// Add an auditory stimulus. This function is part of the public API.
118    pub fn add_auditory(
119        &mut self,
120        x: f32,
121        y: f32,
122        intensity: f32,
123        radius: f32,
124        decay_rate: f32,
125        tag: Option<String>,
126    ) -> u64 {
127        self.add(Stimulus {
128            id: 0,
129            stimulus_type: StimulusType::Auditory,
130            position: (x, y),
131            intensity,
132            radius,
133            decay_rate,
134            source_name: None,
135            tag,
136        })
137    }
138    #[allow(clippy::too_many_arguments)]
139    /// Add a custom stimulus type. This function is part of the public API.
140    pub fn add_custom(
141        &mut self,
142        sense_type: &str,
143        x: f32,
144        y: f32,
145        intensity: f32,
146        radius: f32,
147        decay_rate: f32,
148        tag: Option<String>,
149    ) -> u64 {
150        self.add(Stimulus {
151            id: 0,
152            stimulus_type: StimulusType::Custom(sense_type.to_string()),
153            position: (x, y),
154            intensity,
155            radius,
156            decay_rate,
157            source_name: None,
158            tag,
159        })
160    }
161    /// Remove a stimulus by id and return `true` when one was removed.
162    pub fn remove(&mut self, id: u64) -> bool {
163        let before = self.stimuli.len();
164        self.stimuli.retain(|s| s.id != id);
165        self.stimuli.len() < before
166    }
167    /// Decay all stimuli and drop exhausted entries.
168    pub fn update(&mut self, dt: f32) {
169        for s in &mut self.stimuli {
170            if s.decay_rate > 0.0 {
171                s.intensity -= s.decay_rate * dt;
172            }
173        }
174        self.stimuli.retain(|s| s.intensity > 0.0);
175    }
176    /// Return the active stimuli slice.
177    pub fn stimuli(&self) -> &[Stimulus] {
178        &self.stimuli
179    }
180    /// Return the number of active stimuli.
181    pub fn count(&self) -> usize {
182        self.stimuli.len()
183    }
184    /// Remove all stimuli. This function is part of the public API.
185    pub fn clear(&mut self) {
186        self.stimuli.clear();
187    }
188}
189/// `Default` delegates to `StimulusWorld::new`.
190impl Default for StimulusWorld {
191    /// Build an empty stimulus world.
192    fn default() -> Self {
193        Self::new()
194    }
195}
196/// Perception configuration and transient awareness state.
197pub struct Sensor {
198    /// Sight radius.
199    pub sight_range: f32,
200    /// Sight cone width in degrees.
201    pub sight_angle: f32,
202    /// Hearing radius.
203    pub hearing_range: f32,
204    /// Facing angle in radians.
205    pub facing: f32,
206    /// Current awareness in `[0, 1]`.
207    pub awareness: f32,
208    /// Awareness rise rate.
209    pub awareness_rise: f32,
210    /// Awareness decay rate.
211    pub awareness_decay: f32,
212    /// Threshold used for alert state.
213    pub alert_threshold: f32,
214    /// Custom detection ranges per label.
215    pub custom_ranges: HashMap<String, f32>,
216}
217impl Sensor {
218    /// Create a sensor with default sight, hearing, and awareness settings.
219    pub fn new() -> Self {
220        Self {
221            sight_range: 200.0,
222            sight_angle: 120.0,
223            hearing_range: 100.0,
224            facing: 0.0,
225            awareness: 0.0,
226            awareness_rise: 0.5,
227            awareness_decay: 0.3,
228            alert_threshold: 0.8,
229            custom_ranges: HashMap::new(),
230        }
231    }
232    /// Return `true` when the target lies within sight range and the vision cone.
233    pub fn can_see(&self, sensor_pos: (f32, f32), target_pos: (f32, f32)) -> bool {
234        let dx = target_pos.0 - sensor_pos.0;
235        let dy = target_pos.1 - sensor_pos.1;
236        let dist_sq = dx * dx + dy * dy;
237        if dist_sq > self.sight_range * self.sight_range {
238            return false;
239        }
240        if self.sight_angle >= 360.0 {
241            return true;
242        }
243        let angle_to = dy.atan2(dx);
244        let half_cone = (self.sight_angle * 0.5).to_radians();
245        let diff = angle_diff(angle_to, self.facing);
246        diff.abs() <= half_cone
247    }
248    /// Return `true` when the auditory stimulus is within the effective hearing range.
249    pub fn can_hear(&self, sensor_pos: (f32, f32), stimulus: &Stimulus) -> bool {
250        if stimulus.stimulus_type != StimulusType::Auditory {
251            return false;
252        }
253        let dx = stimulus.position.0 - sensor_pos.0;
254        let dy = stimulus.position.1 - sensor_pos.1;
255        let dist = (dx * dx + dy * dy).sqrt();
256        dist <= self.hearing_range.min(stimulus.radius)
257    }
258    /// Return every stimulus currently detected from `sensor_pos`.
259    pub fn detect(&self, sensor_pos: (f32, f32), world: &StimulusWorld) -> Vec<DetectedStimulus> {
260        let mut out = Vec::new();
261        for stim in world.stimuli() {
262            let dx = stim.position.0 - sensor_pos.0;
263            let dy = stim.position.1 - sensor_pos.1;
264            let distance = (dx * dx + dy * dy).sqrt();
265            let detected = match &stim.stimulus_type {
266                StimulusType::Visual => self.can_see(sensor_pos, stim.position),
267                StimulusType::Auditory => self.can_hear(sensor_pos, stim),
268                StimulusType::Custom(label) => {
269                    if let Some(&range) = self.custom_ranges.get(label) {
270                        distance <= range.min(stim.radius)
271                    } else {
272                        false
273                    }
274                }
275            };
276            if detected {
277                out.push(DetectedStimulus {
278                    stimulus_id: stim.id,
279                    stimulus_type: stim.stimulus_type.clone(),
280                    position: stim.position,
281                    intensity: stim.intensity,
282                    distance,
283                    source_name: stim.source_name.clone(),
284                    tag: stim.tag.clone(),
285                });
286            }
287        }
288        out
289    }
290    /// Raise or decay awareness based on the current number of detections.
291    pub fn update_awareness(&mut self, detected_count: usize, dt: f32) {
292        if detected_count > 0 {
293            self.awareness = (self.awareness + self.awareness_rise * dt).min(1.0);
294        } else {
295            self.awareness = (self.awareness - self.awareness_decay * dt).max(0.0);
296        }
297    }
298    /// Return `true` when awareness reached the alert threshold.
299    pub fn is_alert(&self) -> bool {
300        self.awareness >= self.alert_threshold
301    }
302    /// Register a detection range override for one custom stimulus label.
303    pub fn add_custom_range(&mut self, type_label: &str, range: f32) {
304        self.custom_ranges.insert(type_label.to_string(), range);
305    }
306}
307/// `Default` delegates to `Sensor::new`.
308impl Default for Sensor {
309    /// Build a sensor with the default ranges and awareness settings.
310    fn default() -> Self {
311        Self::new()
312    }
313}
314/// Return the wrapped angular difference between `a` and `b` in radians.
315fn angle_diff(a: f32, b: f32) -> f32 {
316    let mut diff = a - b;
317    while diff > std::f32::consts::PI {
318        diff -= 2.0 * std::f32::consts::PI;
319    }
320    while diff <= -std::f32::consts::PI {
321        diff += 2.0 * std::f32::consts::PI;
322    }
323    diff
