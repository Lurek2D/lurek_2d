1//! ORCA local-avoidance data representing moving agents, solver constraints, and safe output velocities.
2//!
3//! - Per-agent motion inputs: current velocity, preferred velocity, collision radius, and max speed.
4//! - Solver pass building pairwise half-plane constraints and projecting collision-free velocities.
5
6/// One agent used by the ORCA solver.
7#[derive(Clone)]
8pub struct ORCAAgent {
9    /// Agent position.
10    pub position: (f32, f32),
11    /// Current velocity.
12    pub velocity: (f32, f32),
13    /// Preferred velocity before collision avoidance.
14    pub preferred_velocity: (f32, f32),
15    /// Velocity chosen by the solver.
16    pub safe_velocity: (f32, f32),
17    /// Collision radius.
18    pub radius: f32,
19    /// Maximum speed.
20    pub max_speed: f32,
21}
22impl ORCAAgent {
23    /// Create a new agent at `(x, y)`.
24    pub fn new(x: f32, y: f32, radius: f32, max_speed: f32) -> Self {
25        Self {
26            position: (x, y),
27            velocity: (0.0, 0.0),
28            preferred_velocity: (0.0, 0.0),
29            safe_velocity: (0.0, 0.0),
30            radius,
31            max_speed,
32        }
33    }
34}
35/// Internal half-plane constraint used by the linear solver.
36#[derive(Clone, Copy)]
37struct HalfPlane {
38    /// Point on the plane.
39    point: (f32, f32),
40    /// Outward normal.
41    normal: (f32, f32),
42}
43/// Solver that computes collision-free velocities for all registered agents.
44pub struct ORCASolver {
45    /// Lookahead time used by the constraints.
46    pub time_horizon: f32,
47    /// Registered agents.
48    pub agents: Vec<ORCAAgent>,
49}
50impl ORCASolver {
51    /// Create a solver with a minimum time horizon of 0.1 seconds.
52    pub fn new(time_horizon: f32) -> Self {
53        Self {
54            time_horizon: time_horizon.max(0.1),
55            agents: Vec::new(),
56        }
57    }
58    /// Add an agent and return its index.
59    pub fn add_agent(&mut self, agent: ORCAAgent) -> usize {
60        let idx = self.agents.len();
61        self.agents.push(agent);
62        idx
63    }
64    /// Remove and return the agent at `index`, or `None` when out of bounds.
65    pub fn remove_agent(&mut self, index: usize) -> Option<ORCAAgent> {
66        if index < self.agents.len() {
67            Some(self.agents.swap_remove(index))
68        } else {
69            None
70        }
71    }
72    /// Return the number of registered agents.
73    pub fn agent_count(&self) -> usize {
74        self.agents.len()
75    }
76    #[allow(clippy::needless_range_loop)]
77    /// Compute safe velocities for all agents.
78    pub fn compute(&mut self, _dt: f32) {
79        let n = self.agents.len();
80        let snapshot: Vec<(f32, f32, f32, f32, f32, f32)> = self
81            .agents
82            .iter()
83            .map(|a| {
84                (
85                    a.position.0,
86                    a.position.1,
87                    a.velocity.0,
88                    a.velocity.1,
89                    a.radius,
90                    a.max_speed,
91                )
92            })
93            .collect();
94        for i in 0..n {
95            let (px, py, vx, vy, ri, max_spd) = snapshot[i];
96            let pref = self.agents[i].preferred_velocity;
97            let mut planes: Vec<HalfPlane> = Vec::with_capacity(n - 1);
98            for j in 0..n {
99                if i == j {
100                    continue;
101                }
102                let (opx, opy, ovx, ovy, rj, _) = snapshot[j];
103                let rel_pos = (opx - px, opy - py);
104                let rel_vel = (vx - ovx, vy - ovy);
105                let combined_radius = ri + rj;
106                let dist_sq = rel_pos.0 * rel_pos.0 + rel_pos.1 * rel_pos.1;
107                let r_sq = combined_radius * combined_radius;
108                let hp = if dist_sq >= r_sq {
109                    let dist = dist_sq.sqrt();
110                    let tau = self.time_horizon.max(0.01);
111                    let w = (rel_vel.0 - rel_pos.0 / tau, rel_vel.1 - rel_pos.1 / tau);
112                    let w_len_sq = w.0 * w.0 + w.1 * w.1;
113                    let dot = w.0 * rel_pos.0 + w.1 * rel_pos.1;
114                    let (nx, ny) = if dot < 0.0 && dot * dot > r_sq * w_len_sq / (tau * tau) {
115                        let (lx, ly) = (rel_pos.0 / dist, rel_pos.1 / dist);
116                        (-lx, -ly)
117                    } else {
118                        let w_len = w_len_sq.sqrt().max(1e-6);
119                        (w.0 / w_len, w.1 / w_len)
120                    };
121                    let u = (
122                        nx * combined_radius / tau - rel_vel.0,
123                        ny * combined_radius / tau - rel_vel.1,
124                    );
125                    HalfPlane {
126                        point: (vx + u.0 * 0.5, vy + u.1 * 0.5),
127                        normal: (nx, ny),
128                    }
129                } else {
130                    let inv_dist = 1.0 / dist_sq.sqrt().max(1e-6);
131                    let nx = -rel_pos.0 * inv_dist;
132                    let ny = -rel_pos.1 * inv_dist;
133                    HalfPlane {
134                        point: (vx, vy),
135                        normal: (nx, ny),
136                    }
137                };
138                planes.push(hp);
139            }
140            let safe = Self::linear_program(pref, max_spd, &planes);
141            self.agents[i].safe_velocity = safe;
142        }
143    }
144    /// Project the preferred velocity against the half-plane set and clamp speed.
145    fn linear_program(preferred: (f32, f32), max_speed: f32, planes: &[HalfPlane]) -> (f32, f32) {
146        let mut vx = preferred.0;
147        let mut vy = preferred.1;
148        let spd_sq = vx * vx + vy * vy;
149        if spd_sq > max_speed * max_speed {
150            let s = max_speed / spd_sq.sqrt();
151            vx *= s;
152            vy *= s;
153        }
154        for plane in planes {
155            let dot = (vx - plane.point.0) * plane.normal.0 + (vy - plane.point.1) * plane.normal.1;
156            if dot < 0.0 {
157                let proj_len =
158                    dot / (plane.normal.0 * plane.normal.0 + plane.normal.1 * plane.normal.1);
159                vx -= proj_len * plane.normal.0;
160                vy -= proj_len * plane.normal.1;
161                let s2 = vx * vx + vy * vy;
162                if s2 > max_speed * max_speed {
163                    let s = max_speed / s2.sqrt();
164                    vx *= s;
165                    vy *= s;
166                }
167            }
168        }
169        (vx, vy)
170    }
