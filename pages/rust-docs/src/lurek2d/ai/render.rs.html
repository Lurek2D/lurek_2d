10impl StateMachine {
11    /// Build line and box commands for the FSM debug view.
12    pub fn generate_render_commands(&self) -> Vec<RenderCommand> {
13        let mut state_names: Vec<&str> = self.states.keys().map(|s| s.as_str()).collect();
14        state_names.sort_unstable();
15        if state_names.is_empty() {
16            return Vec::new();
17        }
18        let box_w = 80.0f32;
19        let box_h = 30.0f32;
20        let gap = 10.0f32;
21        let y = 20.0f32;
22        let mut cmds = Vec::with_capacity(state_names.len() * 4 + 2);
23        for (i, name) in state_names.iter().enumerate() {
24            let x = (box_w + gap) * i as f32 + 10.0;
25            let is_active = self
26                .current_state
27                .as_deref()
28                .map(|c| c == *name)
29                .unwrap_or(false);
30            if is_active {
31                cmds.push(RenderCommand::SetColor(1.0, 1.0, 1.0, 0.9));
32            } else {
33                cmds.push(RenderCommand::SetColor(0.3, 0.5, 0.8, 0.6));
34            }
35            cmds.push(RenderCommand::Rectangle {
36                mode: DrawMode::Line,
37                x,
38                y,
39                w: box_w,
40                h: box_h,
41            });
42        }
43        cmds.push(RenderCommand::SetColor(0.6, 0.6, 0.6, 0.5));
44        for t in &self.transitions {
45            let from_idx = state_names.iter().position(|&s| s == t.from);
46            let to_idx = state_names.iter().position(|&s| s == t.to);
47            if let (Some(fi), Some(ti)) = (from_idx, to_idx) {
48                if fi != ti {
49                    let fx = (box_w + gap) * fi as f32 + 10.0 + box_w;
50                    let fy = y + box_h * 0.5;
51                    let tx = (box_w + gap) * ti as f32 + 10.0;
52                    let ty = y + box_h * 0.5;
53                    cmds.push(RenderCommand::Line {
54                        x1: fx,
55                        y1: fy,
56                        x2: tx,
57                        y2: ty,
58                    });
59                }
60            }
61        }
62        cmds
63    }
64    /// Draw the FSM debug view into an `ImageData` buffer.
65    pub fn draw_to_image(&self, width: u32, height: u32) -> ImageData {
66        let mut img = ImageData::new(width, height);
67        img.fill(15, 15, 25, 255);
68        let mut state_names: Vec<&str> = self.states.keys().map(|s| s.as_str()).collect();
69        state_names.sort_unstable();
70        if state_names.is_empty() {
71            return img;
72        }
73        let row_h = (height / state_names.len() as u32).max(1);
74        for (i, name) in state_names.iter().enumerate() {
75            let is_active = self
76                .current_state
77                .as_deref()
78                .map(|c| c == *name)
79                .unwrap_or(false);
80            let (r, g, b) = if is_active {
81                (220u8, 220, 220)
82            } else {
83                (50, 80, 140)
84            };
85            let strip_y = (i as u32 * row_h) as i32;
86            img.draw_rect(
87                2,
88                strip_y + 2,
89                width - 4,
90                row_h.saturating_sub(4),
91                r,
92                g,
93                b,
94                200,
95            );
96        }
97        img
98    }
99}
100/// Count all BT nodes in a subtree.
101fn bt_node_count(node: &BTNode) -> usize {
102    match node {
103        BTNode::Selector { children, .. }
104        | BTNode::Sequence { children, .. }
105        | BTNode::Parallel { children, .. } => {
106            1 + children.iter().map(bt_node_count).sum::<usize>()
107        }
108        BTNode::Inverter { child }
109        | BTNode::Repeater { child, .. }
110        | BTNode::Succeeder { child }
111        | BTNode::Guard { child, .. } => 1 + bt_node_count(child),
112        BTNode::Action { .. } | BTNode::Condition { .. } => 1,
113    }
114}
115/// Return a short label for a BT node.
116fn bt_node_label(node: &BTNode) -> &'static str {
117    match node {
118        BTNode::Selector { .. } => "SEL",
119        BTNode::Sequence { .. } => "SEQ",
120        BTNode::Parallel { .. } => "PAR",
121        BTNode::Inverter { .. } => "INV",
122        BTNode::Repeater { .. } => "REP",
123        BTNode::Succeeder { .. } => "SUC",
124        BTNode::Guard { .. } => "GRD",
125        BTNode::Action { .. } => "ACT",
126        BTNode::Condition { .. } => "CND",
127    }
128}
129/// Return the maximum depth of a BT subtree.
130fn bt_depth(node: &BTNode) -> usize {
131    match node {
132        BTNode::Selector { children, .. }
133        | BTNode::Sequence { children, .. }
134        | BTNode::Parallel { children, .. } => 1 + children.iter().map(bt_depth).max().unwrap_or(0),
135        BTNode::Inverter { child }
136        | BTNode::Repeater { child, .. }
137        | BTNode::Succeeder { child }
138        | BTNode::Guard { child, .. } => 1 + bt_depth(child),
139        BTNode::Action { .. } | BTNode::Condition { .. } => 1,
140    }
141}
142/// Emit BT debug render commands for one subtree.
143fn emit_bt_commands(
144    node: &BTNode,
145    depth: usize,
146    slot: usize,
147    cmds: &mut Vec<RenderCommand>,
148    node_w: f32,
149    node_h: f32,
150    gap: f32,
151) {
152    let _ = bt_node_label(node);
153    let x = gap + depth as f32 * (node_w + gap);
154    let y = gap + slot as f32 * (node_h + gap);
155    cmds.push(RenderCommand::SetColor(0.3, 0.6, 0.9, 0.8));
156    cmds.push(RenderCommand::Rectangle {
157        mode: DrawMode::Line,
158        x,
159        y,
160        w: node_w,
161        h: node_h,
162    });
163    let children: Vec<&BTNode> = match node {
164        BTNode::Selector { children, .. }
165        | BTNode::Sequence { children, .. }
166        | BTNode::Parallel { children, .. } => children.iter().collect(),
167        BTNode::Inverter { child }
168        | BTNode::Repeater { child, .. }
169        | BTNode::Succeeder { child }
170        | BTNode::Guard { child, .. } => vec![child.as_ref()],
171        _ => vec![],
172    };
173    let mut child_slot = slot;
174    for child in children {
175        let cx = gap + (depth + 1) as f32 * (node_w + gap);
176        let cy = gap + child_slot as f32 * (node_h + gap);
177        cmds.push(RenderCommand::SetColor(0.5, 0.5, 0.5, 0.5));
178        cmds.push(RenderCommand::Line {
179            x1: x + node_w,
180            y1: y + node_h * 0.5,
181            x2: cx,
182            y2: cy + node_h * 0.5,
183        });
184        let sub_size = bt_node_count(child);
185        emit_bt_commands(child, depth + 1, child_slot, cmds, node_w, node_h, gap);
186        child_slot += sub_size;
187    }
188}
189impl BehaviorTree {
190    /// Build render commands for the BT debug view.
191    pub fn generate_render_commands(&self) -> Vec<RenderCommand> {
192        let root = match &self.root {
193            Some(r) => r,
194            None => return Vec::new(),
195        };
196        let node_count = bt_node_count(root);
197        let mut cmds = Vec::with_capacity(node_count * 4);
198        let (node_w, node_h, gap) = (50.0f32, 20.0f32, 8.0f32);
199        let (sr, sg, sb) = match self.last_status {
200            BTStatus::Success => (0.2, 0.9, 0.2),
201            BTStatus::Failure => (0.9, 0.2, 0.2),
202            BTStatus::Running => (0.9, 0.7, 0.1),
203        };
204        cmds.push(RenderCommand::SetColor(sr, sg, sb, 0.6));
205        cmds.push(RenderCommand::Circle {
206            mode: DrawMode::Fill,
207            x: 8.0,
208            y: 8.0,
209            r: 5.0,
210        });
211        emit_bt_commands(root, 0, 0, &mut cmds, node_w, node_h, gap);
212        cmds
213    }
214    /// Draw the BT debug view into an `ImageData` buffer.
215    pub fn draw_to_image(&self, width: u32, height: u32) -> ImageData {
216        let mut img = ImageData::new(width, height);
217        img.fill(15, 15, 25, 255);
218        let root = match &self.root {
219            Some(r) => r,
220            None => return img,
221        };
222        let depth = bt_depth(root);
223        if depth == 0 {
224            return img;
225        }
226        let node_count = bt_node_count(root);
227        let node_w = (width / (depth as u32 + 1)).max(4);
228        let node_h = (height / (node_count as u32 + 1)).max(4);
229        draw_bt_image(&mut img, root, 0, 0, node_w, node_h);
230        img
231    }
232}
233/// Draw one BT subtree into an image buffer.
234fn draw_bt_image(
235    img: &mut ImageData,
236    node: &BTNode,
237    depth: u32,
238    slot: u32,
239    node_w: u32,
240    node_h: u32,
241) {
242    let gap = 2u32;
243    let x = (depth * (node_w + gap)) as i32;
244    let y = (slot * (node_h + gap)) as i32;
245    let (r, g, b) = match node {
246        BTNode::Selector { .. } | BTNode::Sequence { .. } | BTNode::Parallel { .. } => {
247            (80u8, 120, 200)
248        }
249        BTNode::Inverter { .. }
250        | BTNode::Repeater { .. }
251        | BTNode::Succeeder { .. }
252        | BTNode::Guard { .. } => (200, 140, 60),
253        BTNode::Action { .. } | BTNode::Condition { .. } => (80, 200, 80),
254    };
255    img.draw_rect(
256        x,
257        y,
258        node_w.saturating_sub(gap),
259        node_h.saturating_sub(gap),
260        r,
261        g,
262        b,
263        220,
264    );
265    let children: Vec<&BTNode> = match node {
266        BTNode::Selector { children, .. }
267        | BTNode::Sequence { children, .. }
268        | BTNode::Parallel { children, .. } => children.iter().collect(),
269        BTNode::Inverter { child }
270        | BTNode::Repeater { child, .. }
271        | BTNode::Succeeder { child }
272        | BTNode::Guard { child, .. } => vec![child.as_ref()],
273        _ => vec![],
274    };
275    let mut child_slot = slot;
276    for child in children {
277        draw_bt_image(img, child, depth + 1, child_slot, node_w, node_h);
278        child_slot += bt_node_count(child) as u32;
279    }
