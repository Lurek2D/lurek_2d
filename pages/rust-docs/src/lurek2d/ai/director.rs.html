1//! Pacing director translating accumulated tension into pressure phases and runtime multipliers.
2//!
3//! - Tunable thresholds and timers moving between buildup, peak, sustain, and relief phases.
4//! - Derived outputs for spawn pressure, loot pressure, ambient intensity, and state inspection.
5//! - Slower tension decay during peak and sustain phases to hold pressure before relief.
6//! - Per-phase spawn, loot, and ambient multipliers scaling downstream gameplay intensity.
7
8/// Director pacing phase. This item is part of the public API.
9#[derive(Debug, Clone, Copy, PartialEq, Eq)]
10pub enum DirectorPhase {
11    /// Tension is rising toward a peak.
12    BuildUp,
13    /// Peak pressure phase.
14    Peak,
15    /// High pressure maintained after a peak.
16    Sustain,
17    /// Low pressure recovery phase.
18    Relief,
19}
20impl DirectorPhase {
21    /// Return the canonical string tag for this pacing phase.
22    pub fn as_str(self) -> &'static str {
23        match self {
24            Self::BuildUp => "build_up",
25            Self::Peak => "peak",
26            Self::Sustain => "sustain",
27            Self::Relief => "relief",
28        }
29    }
30}
31/// Tunable thresholds for the AI director.
32pub struct DirectorConfig {
33    /// Tension decay per second.
34    pub tension_decay_rate: f32,
35    /// Tension threshold that triggers a peak phase.
36    pub peak_threshold: f32,
37    /// Tension threshold below which recovery can begin.
38    pub relief_threshold: f32,
39    /// Minimum sustain time after a peak.
40    pub sustain_duration: f32,
41    /// Maximum tension added by one event.
42    pub max_tension_per_event: f32,
43    /// Spawn multiplier used during high pressure.
44    pub peak_spawn_factor: f32,
45    /// Loot multiplier used during relief.
46    pub relief_loot_factor: f32,
47}
48/// `Default` provides the tuned pacing config used by `AIDirector::new`.
49impl Default for DirectorConfig {
50    /// Build the default pacing configuration.
51    fn default() -> Self {
52        Self {
53            tension_decay_rate: 0.05,
54            peak_threshold: 0.8,
55            relief_threshold: 0.3,
56            sustain_duration: 15.0,
57            max_tension_per_event: 0.25,
58            peak_spawn_factor: 2.0,
59            relief_loot_factor: 2.5,
60        }
61    }
62}
63/// Runtime director state used by pacing systems.
64pub struct AIDirector {
65    /// Active tuning values.
66    pub config: DirectorConfig,
67    /// Current tension in `[0, 1]`.
68    tension: f32,
69    /// Current pacing phase.
70    phase: DirectorPhase,
71    /// Time spent in sustain phase.
72    sustain_timer: f32,
73    /// Time since creation or last reset.
74    elapsed: f32,
75    /// Total number of submitted events.
76    total_events: u32,
77}
78impl AIDirector {
79    /// Create a director with default config.
80    pub fn new() -> Self {
81        Self {
82            config: DirectorConfig::default(),
83            tension: 0.0,
84            phase: DirectorPhase::Relief,
85            sustain_timer: 0.0,
86            elapsed: 0.0,
87            total_events: 0,
88        }
89    }
90    /// Create a director with a custom config.
91    pub fn with_config(config: DirectorConfig) -> Self {
92        Self {
93            config,
94            tension: 0.0,
95            phase: DirectorPhase::Relief,
96            sustain_timer: 0.0,
97            elapsed: 0.0,
98            total_events: 0,
99        }
100    }
101    /// Return the current tension. This function is part of the public API.
102    pub fn tension(&self) -> f32 {
103        self.tension
104    }
105    /// Return the current phase. This function is part of the public API.
106    pub fn phase(&self) -> DirectorPhase {
107        self.phase
108    }
109    /// Return the current phase as a string tag.
110    pub fn phase_str(&self) -> &'static str {
111        self.phase.as_str()
112    }
113    /// Return elapsed seconds. This function is part of the public API.
114    pub fn elapsed(&self) -> f32 {
115        self.elapsed
116    }
117    /// Return total events received.
118    pub fn total_events(&self) -> u32 {
119        self.total_events
120    }
121    /// Add one event and clamp the resulting tension to `[0, 1]`.
122    pub fn push_event(&mut self, intensity: f32) {
123        let clamped = intensity.clamp(0.0, self.config.max_tension_per_event);
124        self.tension = (self.tension + clamped).clamp(0.0, 1.0);
125        self.total_events += 1;
126    }
127    /// Advance the director and update phase transitions.
128    pub fn update(&mut self, dt: f32) {
129        self.elapsed += dt;
130        if self.phase != DirectorPhase::Peak && self.phase != DirectorPhase::Sustain {
131            self.tension = (self.tension - self.config.tension_decay_rate * dt).max(0.0);
132        } else {
133            self.tension = (self.tension - self.config.tension_decay_rate * 0.3 * dt).max(0.0);
134        }
135        match self.phase {
136            DirectorPhase::Relief | DirectorPhase::BuildUp => {
137                if self.tension >= self.config.peak_threshold {
138                    self.phase = DirectorPhase::Peak;
139                    self.sustain_timer = 0.0;
140                } else if self.tension > self.config.relief_threshold {
141                    self.phase = DirectorPhase::BuildUp;
142                }
143            }
144            DirectorPhase::Peak => {
145                if self.tension < self.config.peak_threshold {
146                    self.phase = DirectorPhase::Sustain;
147                    self.sustain_timer = 0.0;
148                }
149            }
150            DirectorPhase::Sustain => {
151                self.sustain_timer += dt;
152                if self.sustain_timer >= self.config.sustain_duration
153                    && self.tension <= self.config.relief_threshold
154                {
155                    self.phase = DirectorPhase::Relief;
156                }
157            }
158        }
159    }
160    /// Return the current spawn rate multiplier.
161    pub fn spawn_rate_factor(&self) -> f32 {
162        match self.phase {
163            DirectorPhase::BuildUp => {
164                1.0 + self.tension * (self.config.peak_spawn_factor - 1.0) * 0.5
165            }
166            DirectorPhase::Peak | DirectorPhase::Sustain => self.config.peak_spawn_factor,
167            DirectorPhase::Relief => 0.25,
168        }
169    }
170    /// Return the current loot multiplier.
171    pub fn loot_factor(&self) -> f32 {
172        match self.phase {
173            DirectorPhase::Relief => self.config.relief_loot_factor,
174            DirectorPhase::BuildUp => 1.0,
175            DirectorPhase::Peak | DirectorPhase::Sustain => 0.5,
176        }
177    }
178    /// Return the current ambient intensity scalar.
179    pub fn ambient_intensity(&self) -> f32 {
180        match self.phase {
181            DirectorPhase::Peak => (self.tension * 0.5 + 0.5).clamp(0.0, 1.0),
182            DirectorPhase::Sustain => 0.6f32.max(self.tension),
183            DirectorPhase::BuildUp => self.tension,
184            DirectorPhase::Relief => (self.tension * 0.5).max(0.1),
185        }
186    }
187    /// Set tension directly and clamp it to `[0, 1]`.
188    pub fn set_tension(&mut self, value: f32) {
189        self.tension = value.clamp(0.0, 1.0);
190    }
191    /// Reset tension, phase, and timers to their initial state.
192    pub fn reset(&mut self) {
193        self.tension = 0.0;
194        self.phase = DirectorPhase::Relief;
195        self.sustain_timer = 0.0;
196    }
197}
198/// `Default` delegates to `AIDirector::new`.
199impl Default for AIDirector {
200    /// Build a director with the default configuration.
201    fn default() -> Self {
202        Self::new()
203    }
