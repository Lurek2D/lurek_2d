1//! Runtime state machine executor for AI agents.
2//!
3//! For structural FSM definition and graph validation, see [`crate::patterns::state_machine`].
4//! This module adds Lua callback execution, priority-based transitions, and per-frame time tracking.
5//!
6//! - Finite-state-machine storing named states, callback hooks, transition rules, and active state.
7//! - Callback bundles for state entry, update, and exit with priority-sorted transition records.
8//! - Mutable machine state tracking current state, initial state, elapsed time, and registration helpers.
9//! - Descending-priority transition sorting so the tick evaluator tests highest-priority guards first.
10//! - Elapsed time tracking in the current state for time-based guards and Lua update callbacks.
11
12use crate::log_msg;
13use crate::runtime::log_messages::{FN01, FN02};
14use mlua::RegistryKey;
15use std::collections::HashMap;
16/// Lua callback set attached to one FSM state.
17pub struct StateCallbacks {
18    /// Registry key of the Lua callback invoked when entering this state.
19    pub on_enter: Option<RegistryKey>,
20    /// Registry key of the Lua callback invoked each tick while in this state.
21    pub on_update: Option<RegistryKey>,
22    /// Registry key of the Lua callback invoked when leaving this state.
23    pub on_exit: Option<RegistryKey>,
24}
25/// One transition rule between FSM states.
26pub struct Transition {
27    /// Source state name; `"*"` matches any state.
28    pub from: String,
29    /// Destination state name activated when the guard passes.
30    pub to: String,
31    /// Optional Lua guard predicate; `None` means the transition is unconditional.
32    pub guard: Option<RegistryKey>,
33    /// Evaluation order; higher priority transitions are tested first.
34    pub priority: i32,
35}
36#[allow(dead_code)]
37/// Finite-state-machine storage for states, transitions, and runtime selection.
38pub struct StateMachine {
39    /// Map from state name to its enter/update/exit callbacks.
40    pub(crate) states: HashMap<String, StateCallbacks>,
41    /// All registered transitions, sorted descending by priority.
42    pub(crate) transitions: Vec<Transition>,
43    /// Name of the currently active state; `None` before first tick.
44    pub(crate) current_state: Option<String>,
45    /// State name to activate on the first tick if none is active.
46    pub(crate) initial_state: Option<String>,
47    /// Elapsed time in seconds since the current state was entered.
48    pub(crate) time_in_state: f32,
49}
50impl StateMachine {
51    /// Create an empty state machine with no states or transitions.
52    pub fn new() -> Self {
53        log_msg!(debug, FN01);
54        Self {
55            states: HashMap::new(),
56            transitions: Vec::new(),
57            current_state: None,
58            initial_state: None,
59            time_in_state: 0.0,
60        }
61    }
62    /// Register a transition and re-sort the transition list by descending priority.
63    pub fn add_transition(&mut self, transition: Transition) {
64        log_msg!(debug, FN02);
65        self.transitions.push(transition);
66        self.transitions.sort_by(|a, b| b.priority.cmp(&a.priority));
67    }
68    /// Return the name of the currently active state, or `None` before the first tick.
69    pub fn current_state(&self) -> Option<&str> {
70        self.current_state.as_deref()
71    }
72    /// Return elapsed seconds since the current state was entered.
73    pub fn time_in_state(&self) -> f32 {
74        self.time_in_state
75    }
76    /// Register a state by name with optional enter, update, and exit registry keys.
77    pub fn add_state_raw(
78        &mut self,
79        name: String,
80        on_enter: Option<RegistryKey>,
81        on_update: Option<RegistryKey>,
82        on_exit: Option<RegistryKey>,
83    ) {
84        self.states.insert(
85            name,
86            StateCallbacks {
87                on_enter,
88                on_update,
89                on_exit,
90            },
91        );
92    }
93    /// Build a `Transition` from raw parts and add it via `add_transition`.
94    pub fn add_transition_raw(
95        &mut self,
96        from: String,
97        to: String,
98        priority: i32,
99        guard: Option<RegistryKey>,
100    ) {
101        self.add_transition(Transition {
102            from,
103            to,
104            guard,
105            priority,
106        });
107    }
108    /// Set the state name that will be activated on the first tick.
109    pub fn set_initial_state(&mut self, name: String) {
110        self.initial_state = Some(name);
111    }
112}
113/// `Default` delegates to `StateMachine::new`.
114impl Default for StateMachine {
115    /// `Default` delegates to `StateMachine::new`.
116    fn default() -> Self {
117        Self::new()
118    }
119}
