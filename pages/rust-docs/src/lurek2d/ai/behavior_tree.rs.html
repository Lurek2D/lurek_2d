1//! Runtime behavior tree executor for AI agents with Lua callbacks.
2//!
3//! For structural BT building and inspection, see [`crate::patterns::behavior_tree`].
4//! This module adds Lua RegistryKey-driven actions, conditions, guards, and per-tick running state.
5//!
6//! - Behavior-tree node hierarchy with local runtime progress and last tick result.
7//! - Control-flow variants: selector, sequence, parallel, decorator, guard, and Lua leaves.
8//! - Subtree reset, node counting, status translation, and compact debug snapshots.
9//! - Running state per composite node enabling cross-tick resume from last active child.
10//! - Parallel policy configuration with independent success and failure thresholds.
11//! - Root node container used as the single-instance tree by the per-agent AI runtime.
12
13use mlua::RegistryKey;
14/// Execution result produced by a behavior-tree node or whole tree.
15#[derive(Debug, Clone, PartialEq)]
16pub enum BTStatus {
17    /// Node completed its task successfully.
18    Success,
19    /// Node could not complete its task.
20    Failure,
21    /// Node is still in progress and must be ticked again next frame.
22    Running,
23}
24impl BTStatus {
25    /// Parse a string tag into `BTStatus`; unknown strings default to `Running`.
26    pub fn parse_str(s: &str) -> Self {
27        match s {
28            "success" => Self::Success,
29            "failure" => Self::Failure,
30            _ => Self::Running,
31        }
32    }
33    /// Return the canonical lowercase string tag for this status.
34    pub fn as_str(&self) -> &'static str {
35        match self {
36            Self::Success => "success",
37            Self::Failure => "failure",
38            Self::Running => "running",
39        }
40    }
41}
42/// Success and failure rule used when a parallel node combines child results.
43#[derive(Debug, Clone, PartialEq)]
44pub enum ParallelPolicy {
45    /// Parallel succeeds as soon as any one child succeeds.
46    RequireOne,
47    /// Parallel succeeds only when all children succeed.
48    RequireAll,
49}
50impl ParallelPolicy {
51    /// Parse a string tag; unknown strings default to `RequireOne`.
52    pub fn parse_str(s: &str) -> Self {
53        match s {
54            "requireAll" => Self::RequireAll,
55            _ => Self::RequireOne,
56        }
57    }
58    /// Return the canonical string tag for this policy.
59    pub fn as_str(&self) -> &'static str {
60        match self {
61            Self::RequireOne => "requireOne",
62            Self::RequireAll => "requireAll",
63        }
64    }
65}
66/// Behavior-tree node with child links and any per-node progress it owns.
67pub enum BTNode {
68    /// Tries children in order; succeeds on the first child success, fails when all fail.
69    Selector {
70        /// Ordered list of child nodes.
71        children: Vec<BTNode>,
72        /// Index of the child currently in `Running` state; reset to 0 on restart.
73        running_idx: usize,
74    },
75    /// Runs children in order; fails on the first child failure, succeeds when all pass.
76    Sequence {
77        /// Ordered list of child nodes.
78        children: Vec<BTNode>,
79        /// Index of the child currently in `Running` state; reset to 0 on restart.
80        running_idx: usize,
81    },
82    /// Ticks all children each frame; result controlled by success and failure policies.
83    Parallel {
84        /// All child nodes ticked each update.
85        children: Vec<BTNode>,
86        /// Determines when the parallel node reports success.
87        success_policy: ParallelPolicy,
88        /// Determines when the parallel node reports failure.
89        failure_policy: ParallelPolicy,
90    },
91    /// Flips child result: `Success` ↔ `Failure`; `Running` passes through unchanged.
92    Inverter {
93        /// The single child whose result is inverted.
94        child: Box<BTNode>,
95    },
96    /// Runs its child `count` times before reporting `Success`.
97    Repeater {
98        /// The child to repeat.
99        child: Box<BTNode>,
100        /// Total repetition target; 0 means repeat indefinitely.
101        count: u32,
102        /// Number of repetitions completed so far.
103        done: u32,
104    },
105    /// Always returns `Success` regardless of the child's result.
106    Succeeder {
107        /// The child whose result is overridden to `Success`.
108        child: Box<BTNode>,
109    },
110    /// Evaluates a Lua predicate; runs child only when the predicate returns truthy.
111    Guard {
112        /// Registry key of the Lua predicate callback.
113        predicate: RegistryKey,
114        /// The child executed when the predicate passes.
115        child: Box<BTNode>,
116    },
117    /// Leaf that calls a Lua callback and converts the return value to `BTStatus`.
118    Action {
119        /// Registry key of the Lua action callback.
120        callback: RegistryKey,
121    },
122    /// Leaf that evaluates a Lua predicate: `true` → `Success`, `false` → `Failure`.
123    Condition {
124        /// Registry key of the Lua condition callback.
125        callback: RegistryKey,
126    },
127}
128impl BTNode {
129    /// Reset all running indices and repetition counters in this subtree recursively.
130    pub fn reset(&mut self) {
131        match self {
132            BTNode::Selector {
133                children,
134                running_idx,
135            } => {
136                *running_idx = 0;
137                for child in children.iter_mut() {
138                    child.reset();
139                }
140            }
141            BTNode::Sequence {
142                children,
143                running_idx,
144            } => {
145                *running_idx = 0;
146                for child in children.iter_mut() {
147                    child.reset();
148                }
149            }
150            BTNode::Parallel { children, .. } => {
151                for child in children.iter_mut() {
152                    child.reset();
153                }
154            }
155            BTNode::Inverter { child } => child.reset(),
156            BTNode::Repeater { child, done, .. } => {
157                *done = 0;
158                child.reset();
159            }
160            BTNode::Succeeder { child } => child.reset(),
161            BTNode::Guard { child, .. } => child.reset(),
162            BTNode::Action { .. } | BTNode::Condition { .. } => {}
163        }
164    }
165    /// Return the number of direct children; leaf nodes return 0.
166    pub fn child_count(&self) -> usize {
167        match self {
168            BTNode::Selector { children, .. }
169            | BTNode::Sequence { children, .. }
170            | BTNode::Parallel { children, .. } => children.len(),
171            BTNode::Inverter { .. }
172            | BTNode::Repeater { .. }
173            | BTNode::Succeeder { .. }
174            | BTNode::Guard { .. } => 1,
175            BTNode::Action { .. } | BTNode::Condition { .. } => 0,
176        }
177    }
178}
179/// Root node and last completed status for one behavior-tree instance.
180pub struct BehaviorTree {
181    /// Top-level node; `None` if no tree has been built yet.
182    pub root: Option<BTNode>,
183    /// Result returned by the last completed tick.
184    pub last_status: BTStatus,
185}
186impl BehaviorTree {
187    /// Create an empty tree with `last_status` initialised to `Success`.
188    pub fn new() -> Self {
189        Self {
190            root: None,
191            last_status: BTStatus::Success,
192        }
193    }
194}
195/// `Default` delegates to `BehaviorTree::new`.
196impl Default for BehaviorTree {
197    /// `Default` delegates to `BehaviorTree::new`.
198    fn default() -> Self {
199        Self::new()
200    }
201}
202/// Count all nodes in a subtree recursively, including the root node.
203fn count_bt_nodes(node: &BTNode) -> usize {
204    1 + match node {
205        BTNode::Selector { children, .. }
206        | BTNode::Sequence { children, .. }
207        | BTNode::Parallel { children, .. } => children.iter().map(count_bt_nodes).sum(),
208        BTNode::Inverter { child } | BTNode::Succeeder { child } | BTNode::Guard { child, .. } => {
209            count_bt_nodes(child)
210        }
211        BTNode::Repeater { child, .. } => count_bt_nodes(child),
212        BTNode::Action { .. } | BTNode::Condition { .. } => 0,
213    }
214}
215/// Debug summary containing node count and the last resolved tree status.
216pub struct BtDebugState {
217    /// Total node count in the associated tree.
218    pub node_count: usize,
219    /// String form of the last tick status.
220    pub last_status: String,
221}
222impl BehaviorTree {
223    /// Build a `BtDebugState` snapshot from the current tree shape and status.
224    pub fn debug_state(&self) -> BtDebugState {
225        let node_count = match &self.root {
226            Some(root) => count_bt_nodes(root),
227            None => 0,
228        };
229        BtDebugState {
230            node_count,
231            last_status: self.last_status.as_str().to_string(),
232        }
233    }
