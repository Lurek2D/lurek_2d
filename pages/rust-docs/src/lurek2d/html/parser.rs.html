1//! Parse raw HTML strings into a live element tree with parent-child relationships.
2//!
3//! - Split tag headers, extract and normalize attribute key-value pairs.
4//! - Decode and encode the small HTML entity set (amp, lt, gt, quot, #39).
5//! - Collapse whitespace in text nodes before attaching to elements.
6//! - Handle self-closing tags, void tags, closing tags, and comments.
7
8use crate::html::element::{HtmlElement, HtmlElementId};
9use std::collections::BTreeMap;
10/// Parse HTML into a live element tree and return the top-level child ids.
11pub(crate) fn parse_into(
12    html: &str,
13    elements: &mut Vec<HtmlElement>,
14    parent: HtmlElementId,
15    warnings: &mut Vec<String>,
16) -> Vec<HtmlElementId> {
17    let mut roots = Vec::new();
18    let mut stack = vec![parent];
19    let mut cursor = 0;
20    while cursor < html.len() {
21        let Some(relative_start) = html[cursor..].find('<') else {
22            push_text(elements, *stack.last().unwrap_or(&parent), &html[cursor..]);
23            break;
24        };
25        let start = cursor + relative_start;
26        push_text(
27            elements,
28            *stack.last().unwrap_or(&parent),
29            &html[cursor..start],
30        );
31        let Some(relative_end) = html[start..].find('>') else {
32            warnings.push("unterminated HTML tag".to_string());
33            break;
34        };
35        let end = start + relative_end;
36        let raw_tag = html[start + 1..end].trim();
37        cursor = end + 1;
38        if raw_tag.is_empty() || raw_tag.starts_with('!') || raw_tag.starts_with('?') {
39            continue;
40        }
41        if raw_tag.starts_with("!--") {
42            continue;
43        }
44        if raw_tag.starts_with('/') {
45            if stack.len() > 1 {
46                stack.pop();
47            }
48            continue;
49        }
50        let self_closing = raw_tag.ends_with('/');
51        let tag_body = raw_tag.trim_end_matches('/').trim();
52        let (tag_name, attr_source) = split_tag(tag_body);
53        if tag_name.is_empty() {
54            continue;
55        }
56        let tag_name = tag_name.to_ascii_lowercase();
57        let attrs = parse_attributes(attr_source);
58        let current_parent = *stack.last().unwrap_or(&parent);
59        if tag_name == "body" && current_parent == parent && elements[parent].tag_name() == "body" {
60            apply_attributes(&mut elements[parent], attrs);
61            if !self_closing {
62                stack.push(parent);
63            }
64            continue;
65        }
66        let id = elements.len();
67        let mut element = HtmlElement::new(id, tag_name, Some(current_parent));
68        apply_attributes(&mut element, attrs);
69        elements.push(element);
70        elements[current_parent].children.push(id);
71        if current_parent == parent {
72            roots.push(id);
73        }
74        if !self_closing && !elements[id].is_void_tag() {
75            stack.push(id);
76        }
77    }
78    roots
79}
80/// Split a tag header into the tag name and the remaining attribute text.
81fn split_tag(source: &str) -> (&str, &str) {
82    let mut parts = source.splitn(2, char::is_whitespace);
83    let tag = parts.next().unwrap_or_default();
84    let attrs = parts.next().unwrap_or_default();
85    (tag, attrs)
86}
87/// Parse a tag attribute string into a normalized attribute map.
88fn parse_attributes(source: &str) -> BTreeMap<String, String> {
89    let mut attrs = BTreeMap::new();
90    let bytes = source.as_bytes();
91    let mut cursor = 0;
92    while cursor < bytes.len() {
93        while cursor < bytes.len() && bytes[cursor].is_ascii_whitespace() {
94            cursor += 1;
95        }
96        let name_start = cursor;
97        while cursor < bytes.len() && !bytes[cursor].is_ascii_whitespace() && bytes[cursor] != b'='
98        {
99            cursor += 1;
100        }
101        if name_start == cursor {
102            break;
103        }
104        let name = source[name_start..cursor].to_ascii_lowercase();
105        while cursor < bytes.len() && bytes[cursor].is_ascii_whitespace() {
106            cursor += 1;
107        }
108        let value = if cursor < bytes.len() && bytes[cursor] == b'=' {
109            cursor += 1;
110            while cursor < bytes.len() && bytes[cursor].is_ascii_whitespace() {
111                cursor += 1;
112            }
113            if cursor < bytes.len() && (bytes[cursor] == b'\'' || bytes[cursor] == b'"') {
114                let quote = bytes[cursor];
115                cursor += 1;
116                let value_start = cursor;
117                while cursor < bytes.len() && bytes[cursor] != quote {
118                    cursor += 1;
119                }
120                let value = decode_entities(&source[value_start..cursor]);
121                if cursor < bytes.len() {
122                    cursor += 1;
123                }
124                value
125            } else {
126                let value_start = cursor;
127                while cursor < bytes.len() && !bytes[cursor].is_ascii_whitespace() {
128                    cursor += 1;
129                }
130                decode_entities(&source[value_start..cursor])
131            }
132        } else {
133            "true".to_string()
134        };
135        attrs.insert(name, value);
136    }
137    attrs
138}
139/// Apply parsed attributes to an element by reusing the element setter.
140fn apply_attributes(element: &mut HtmlElement, attrs: BTreeMap<String, String>) {
141    for (name, value) in attrs {
142        element.set_attribute(&name, Some(value));
143    }
144}
145/// Append collapsed text content to an element after decoding entities.
146fn push_text(elements: &mut [HtmlElement], element_id: HtmlElementId, text: &str) {
147    let text = decode_entities(text);
148    let collapsed = text.split_whitespace().collect::<Vec<_>>().join(" ");
149    if collapsed.is_empty() {
150        return;
151    }
152    let element = &mut elements[element_id];
153    if !element.text.is_empty() {
154        element.text.push(' ');
155    }
156    element.text.push_str(&collapsed);
157}
158/// Escape text for HTML text nodes.
159pub(crate) fn escape_text(text: &str) -> String {
160    text.replace('&', "&amp;")
161        .replace('<', "&lt;")
162        .replace('>', "&gt;")
163}
164/// Escape text for HTML attribute values.
165pub(crate) fn escape_attribute(text: &str) -> String {
166    escape_text(text).replace('"', "&quot;")
167}
168/// Decode the small entity set supported by the HTML parser.
169fn decode_entities(text: &str) -> String {
170    text.replace("&lt;", "<")
171        .replace("&gt;", ">")
172        .replace("&quot;", "\"")
173        .replace("&#39;", "'")
174        .replace("&amp;", "&")
175}
