1//! Chunk-based large-map renderer for tilemaps that exceed single-pass draw limits.
2//!
3//! - Splits the full tile grid into fixed-size square chunks with dirty-flag tracking.
4//! - Camera and viewport state drive visibility culling at chunk granularity.
5//! - Supports per-tile mutation with automatic chunk invalidation.
6//! - Optional LOD down-sampling controlled by configurable zoom thresholds.
7//! - Tileset column count stored for atlas UV computation by the draw backend.
8
9use std::collections::HashMap;
10
11/// A single rendered chunk: a region of tile IDs with a dirty flag for incremental updates.
12#[derive(Debug)]
13pub struct MapChunk {
14    /// Chunk column coordinate in chunk space.
15    pub cx: i32,
16    /// Chunk row coordinate in chunk space.
17    pub cy: i32,
18    /// True when the tiles were modified since last render; set by `invalidate_chunk`.
19    pub dirty: bool,
20    /// Flat tile-ID slice for this chunk, row-major within the chunk bounds.
21    pub tile_ids: Vec<u32>,
22}
23
24/// Stateful renderer context for maps too large to render in a single draw pass.
25pub struct LargeMapRenderer {
26    /// Pixel width of a single tile.
27    pub tile_width: u32,
28    /// Pixel height of a single tile.
29    pub tile_height: u32,
30    /// Map width in tiles.
31    pub map_width: u32,
32    /// Map height in tiles.
33    pub map_height: u32,
34    /// Flat row-major tile-ID array for the full map.
35    tile_data: Vec<u32>,
36    /// Number of columns in the source tileset texture.
37    pub tileset_columns: u32,
38    /// Side length in tiles of each square chunk.
39    pub chunk_size: u32,
40    /// Loaded chunk cache keyed by chunk coordinates.
41    chunks: HashMap<(i32, i32), MapChunk>,
42    /// World X position of the camera centre.
43    pub camera_x: f32,
44    /// World Y position of the camera centre.
45    pub camera_y: f32,
46    /// Camera zoom factor; values less than 1 zoom out.
47    pub camera_zoom: f32,
48    /// Viewport width in pixels at zoom=1.
49    pub viewport_w: f32,
50    /// Viewport height in pixels at zoom=1.
51    pub viewport_h: f32,
52    /// Whether LOD down-sampling is active.
53    pub lod_enabled: bool,
54    /// Zoom thresholds at which successive LOD levels activate; ascending order.
55    pub lod_thresholds: Vec<f32>,
56}
57impl LargeMapRenderer {
58    /// Create a renderer with `tile_w`×`tile_h` tile dimensions and default chunk size of 16.
59    pub fn new(tile_w: u32, tile_h: u32) -> Self {
60        Self {
61            tile_width: tile_w,
62            tile_height: tile_h,
63            map_width: 0,
64            map_height: 0,
65            tile_data: Vec::new(),
66            tileset_columns: 1,
67            chunk_size: 16,
68            chunks: HashMap::new(),
69            camera_x: 0.0,
70            camera_y: 0.0,
71            camera_zoom: 1.0,
72            viewport_w: 0.0,
73            viewport_h: 0.0,
74            lod_enabled: false,
75            lod_thresholds: Vec::new(),
76        }
77    }
78    /// Replace the map tile data and dimensions, then rebuild all chunks.
79    pub fn set_map_data(&mut self, data: Vec<u32>, width: u32, height: u32) {
80        self.map_width = width;
81        self.map_height = height;
82        self.tile_data = data;
83        let expected = (width * height) as usize;
84        self.tile_data.resize(expected, 0);
85        self.rebuild_chunks();
86    }
87    /// Write `tile_id` at map position `(x, y)` and mark its chunk dirty; no-op for out-of-bounds.
88    pub fn set_tile(&mut self, x: u32, y: u32, tile_id: u32) {
89        if x >= self.map_width || y >= self.map_height {
90            return;
91        }
92        let idx = (y * self.map_width + x) as usize;
93        if idx < self.tile_data.len() {
94            self.tile_data[idx] = tile_id;
95            let cx = (x / self.chunk_size) as i32;
96            let cy = (y / self.chunk_size) as i32;
97            self.invalidate_chunk(cx, cy);
98        }
99    }
100    /// Return the tile ID at `(x, y)`, or `None` for out-of-bounds.
101    pub fn get_tile(&self, x: u32, y: u32) -> Option<u32> {
102        if x >= self.map_width || y >= self.map_height {
103            return None;
104        }
105        let idx = (y * self.map_width + x) as usize;
106        self.tile_data.get(idx).copied()
107    }
108    /// Return the map dimensions as `(width, height)` in tiles.
109    pub fn get_map_size(&self) -> (u32, u32) {
110        (self.map_width, self.map_height)
111    }
112
113    /// Set the chunk side length; clamps to 1 and rebuilds all chunks.
114    pub fn set_chunk_size(&mut self, size: u32) {
115        self.chunk_size = size.max(1);
116        self.rebuild_chunks();
117    }
118
119    /// Return the current chunk side length.
120    pub fn get_chunk_size(&self) -> u32 {
121        self.chunk_size
122    }
123
124    /// Mark the chunk at `(cx, cy)` dirty; no-op when not cached.
125    pub fn invalidate_chunk(&mut self, cx: i32, cy: i32) {
126        if let Some(chunk) = self.chunks.get_mut(&(cx, cy)) {
127            chunk.dirty = true;
128        }
129    }
130    /// Mark every cached chunk dirty.
131    pub fn invalidate_all(&mut self) {
132        for chunk in self.chunks.values_mut() {
133            chunk.dirty = true;
134        }
135    }
136    /// Return the count of cached chunks that intersect the current camera viewport.
137    pub fn get_visible_chunks(&self) -> usize {
138        if self.chunk_size == 0 {
139            return 0;
140        }
141        let (min_cx, max_cx, min_cy, max_cy) = self.visible_chunk_range();
142        let mut count = 0;
143        for cy in min_cy..=max_cy {
144            for cx in min_cx..=max_cx {
145                if self.chunks.contains_key(&(cx, cy)) {
146                    count += 1;
147                }
148            }
149        }
150        count
151    }
152    /// Return the total number of cached chunks regardless of visibility.
153    pub fn get_total_chunks(&self) -> usize {
154        self.chunks.len()
155    }
156
157    /// Return the full chunk cache map.
158    pub fn chunks(&self) -> &HashMap<(i32, i32), MapChunk> {
159        &self.chunks
160    }
161
162    /// Set the camera centre `(x, y)` and `zoom` factor.
163    pub fn set_camera(&mut self, x: f32, y: f32, zoom: f32) {
164        self.camera_x = x;
165        self.camera_y = y;
166        self.camera_zoom = zoom;
167    }
168
169    /// Set the viewport dimensions in pixels.
170    pub fn set_viewport(&mut self, w: f32, h: f32) {
171        self.viewport_w = w;
172        self.viewport_h = h;
173    }
174    /// Enable or disable LOD down-sampling.
175    pub fn set_lod_enabled(&mut self, enabled: bool) {
176        self.lod_enabled = enabled;
177    }
178
179    /// Return `true` when LOD is currently enabled.
180    pub fn is_lod_enabled(&self) -> bool {
181        self.lod_enabled
182    }
183
184    /// Replace the LOD zoom-threshold list.
185    pub fn set_lod_thresholds(&mut self, levels: Vec<f32>) {
186        self.lod_thresholds = levels;
187    }
188
189    /// Set the tileset column count; clamps to 1.
190    pub fn set_tileset_columns(&mut self, cols: u32) {
191        self.tileset_columns = cols.max(1);
192    }
193
194    /// Return the current tileset column count.
195    pub fn get_tileset_columns(&self) -> u32 {
196        self.tileset_columns
197    }
198    /// Rebuild all chunks from the current tile data.
199    fn rebuild_chunks(&mut self) {
200        self.chunks.clear();
201        if self.chunk_size == 0 || self.map_width == 0 || self.map_height == 0 {
202            return;
203        }
204        let cols_chunks = self.map_width.div_ceil(self.chunk_size) as i32;
205        let rows_chunks = self.map_height.div_ceil(self.chunk_size) as i32;
206        for cy in 0..rows_chunks {
207            for cx in 0..cols_chunks {
208                let mut tile_ids = Vec::new();
209                let start_x = cx as u32 * self.chunk_size;
210                let start_y = cy as u32 * self.chunk_size;
211                let end_x = (start_x + self.chunk_size).min(self.map_width);
212                let end_y = (start_y + self.chunk_size).min(self.map_height);
213                for ty in start_y..end_y {
214                    for tx in start_x..end_x {
215                        let idx = (ty * self.map_width + tx) as usize;
216                        tile_ids.push(self.tile_data.get(idx).copied().unwrap_or(0));
217                    }
218                }
219                self.chunks.insert(
220                    (cx, cy),
221                    MapChunk {
222                        cx,
223                        cy,
224                        dirty: false,
225                        tile_ids,
226                    },
227                );
228            }
229        }
230    }
231    /// Compute the inclusive chunk coordinate range visible through the current camera viewport.
232    fn visible_chunk_range(&self) -> (i32, i32, i32, i32) {
233        if self.viewport_w <= 0.0 || self.viewport_h <= 0.0 {
234            let chunks_x = ((self.map_width as i32 * self.tile_width as i32).max(1)
235                + self.chunk_size as i32
236                - 1)
237                / self.chunk_size as i32;
238            let chunks_y = ((self.map_height as i32 * self.tile_height as i32).max(1)
239                + self.chunk_size as i32
240                - 1)
241                / self.chunk_size as i32;
242            return (-1, chunks_x, -1, chunks_y);
243        }
244        let zoom = if self.camera_zoom.abs() > f32::EPSILON {
245            self.camera_zoom
246        } else {
247            1.0
248        };
249        let half_w = self.viewport_w * 0.5 / zoom;
250        let half_h = self.viewport_h * 0.5 / zoom;
251        let world_left = self.camera_x - half_w;
252        let world_right = self.camera_x + half_w;
253        let world_top = self.camera_y - half_h;
254        let world_bottom = self.camera_y + half_h;
255        let chunk_w = self.chunk_size as f32 * self.tile_width as f32;
256        let chunk_h = self.chunk_size as f32 * self.tile_height as f32;
257        let min_cx = if chunk_w > 0.0 {
258            (world_left / chunk_w).floor() as i32
259        } else {
260            0
261        };
262        let max_cx = if chunk_w > 0.0 {
263            (world_right / chunk_w).floor() as i32
264        } else {
265            0
266        };
267        let min_cy = if chunk_h > 0.0 {
268            (world_top / chunk_h).floor() as i32
269        } else {
270            0
271        };
272        let max_cy = if chunk_h > 0.0 {
273            (world_bottom / chunk_h).floor() as i32
274        } else {
275            0
276        };
277        (min_cx, max_cx, min_cy, max_cy)
278    }
279}
280/// Default `LargeMapRenderer` uses 32×32 pixel tiles.
281impl Default for LargeMapRenderer {
282    fn default() -> Self {
283        Self::new(32, 32)
284    }
