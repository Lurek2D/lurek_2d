1//! Tileset geometry: tile dimensions, spacing, margin, column count, and GID range ownership.
2//!
3//! - Source-rect lookup: compute pixel `Rect` for any local tile ID within the sprite-sheet.
4//! - Collision metadata: per-tile solid flag storage and query.
5//! - Animation sequences: frame-based tile animations keyed by local ID.
6//! - Autotile rules: 4-bit and 8-bit bitmask-to-tile mappings for terrain transitions.
7
8use crate::log_msg;
9use crate::math::Rect;
10use crate::runtime::log_messages::{TS01, TS02, TS03};
11use std::collections::HashMap;
12
13/// A single frame in a tile sprite-sheet animation.
14#[derive(Debug, Clone)]
15pub struct TileAnimFrame {
16    /// Local tile ID this frame displays.
17    pub tile_id: u32,
18    /// How long this frame is shown in milliseconds.
19    pub duration_ms: f32,
20}
21
22/// A tileset slice of a sprite-sheet texture with collision, animation, and autotile data.
23#[derive(Debug, Clone)]
24pub struct TileSet {
25    /// First global GID owned by this tileset; used to map GIDs to local IDs.
26    first_gid: u32,
27    /// Total number of tiles in the tileset.
28    tile_count: u32,
29    /// Number of tile columns in the source image.
30    columns: u32,
31    /// Width of each tile in pixels.
32    tile_width: u32,
33    /// Height of each tile in pixels.
34    tile_height: u32,
35    /// Pixel gap between tiles in the source image.
36    spacing: u32,
37    /// Pixel margin around the edge of the source image.
38    margin: u32,
39    /// Solid flag per local tile ID; indexed by local ID, absent entries are `false`.
40    solids: Vec<bool>,
41    /// Frame animation sequences keyed by local tile ID.
42    animations: HashMap<u32, Vec<TileAnimFrame>>,
43    /// 4-bit autotile rules: `(type_name, bitmask) -> local_tile_id`.
44    auto_rules_4: HashMap<(String, u8), u32>,
45    /// 8-bit autotile rules: `(type_name, bitmask) -> local_tile_id`.
46    auto_rules_8: HashMap<(String, u16), u32>,
47}
48
49/// Construction, geometry queries, collision, animation, and autotile rule management.
50impl TileSet {
51    /// Create a `TileSet` with the given layout parameters and empty solid, animation, and autotile tables.
52    pub fn new(
53        first_gid: u32,
54        tile_count: u32,
55        columns: u32,
56        tile_width: u32,
57        tile_height: u32,
58        spacing: u32,
59        margin: u32,
60    ) -> Self {
61        log_msg!(debug, TS01, "first_gid={} tiles={}", first_gid, tile_count);
62        Self {
63            first_gid,
64            tile_count,
65            columns,
66            tile_width,
67            tile_height,
68            spacing,
69            margin,
70            solids: Vec::new(),
71            animations: HashMap::new(),
72            auto_rules_4: HashMap::new(),
73            auto_rules_8: HashMap::new(),
74        }
75    }
76    /// Return the first global GID owned by this tileset.
77    pub fn get_first_gid(&self) -> u32 {
78        self.first_gid
79    }
80    /// Return the total tile count. This function is part of the public API.
81    pub fn get_tile_count(&self) -> u32 {
82        self.tile_count
83    }
84    /// Return the number of tile columns in the source image.
85    pub fn get_columns(&self) -> u32 {
86        self.columns
87    }
88    /// Return tile width in pixels. This function is part of the public API.
89    pub fn get_tile_width(&self) -> u32 {
90        self.tile_width
91    }
92    /// Return tile height in pixels.
93    pub fn get_tile_height(&self) -> u32 {
94        self.tile_height
95    }
96    /// Return tile dimensions as `(width, height)` in pixels.
97    pub fn get_tile_dimensions(&self) -> (u32, u32) {
98        (self.tile_width, self.tile_height)
99    }
100    /// Return pixel spacing between tiles in the source image.
101    pub fn get_spacing(&self) -> u32 {
102        self.spacing
103    }
104    /// Return pixel margin around the source image edge.
105    pub fn get_margin(&self) -> u32 {
106        self.margin
107    }
108    /// Return the source-image `Rect` (in pixels) for `local_tile_id`.
109    pub fn get_quad(&self, local_tile_id: u32) -> Rect {
110        let col = local_tile_id % self.columns;
111        let row = local_tile_id / self.columns;
112        let x = self.margin + col * (self.tile_width + self.spacing);
113        let y = self.margin + row * (self.tile_height + self.spacing);
114        Rect::new(
115            x as f32,
116            y as f32,
117            self.tile_width as f32,
118            self.tile_height as f32,
119        )
120    }
121    /// Register or replace the animation frame sequence for `local_tile_id`.
122    pub fn set_animation(&mut self, local_tile_id: u32, frames: Vec<TileAnimFrame>) {
123        log_msg!(
124            debug,
125            TS02,
126            "tile={} frames={}",
127            local_tile_id,
128            frames.len()
129        );
130        self.animations.insert(local_tile_id, frames);
131    }
132    /// Return the animation frames for `local_tile_id`, or `None` when not animated.
133    pub fn get_animation(&self, local_tile_id: u32) -> Option<&Vec<TileAnimFrame>> {
134        self.animations.get(&local_tile_id)
135    }
136    /// Set or clear the solid flag for `local_tile_id`; grows the solids vec as needed.
137    pub fn set_solid(&mut self, local_tile_id: u32, solid: bool) {
138        log_msg!(trace, TS03, "tile={} solid={}", local_tile_id, solid);
139        let idx = local_tile_id as usize;
140        if idx >= self.solids.len() {
141            self.solids.resize(idx + 1, false);
142        }
143        self.solids[idx] = solid;
144    }
145    /// Return `true` when `local_tile_id` is marked solid; returns `false` for IDs beyond the solids vec.
146    pub fn is_solid(&self, local_tile_id: u32) -> bool {
147        let idx = local_tile_id as usize;
148        if idx < self.solids.len() {
149            self.solids[idx]
150        } else {
151            false
152        }
153    }
154    /// Register a 4-bit autotile rule mapping `(type_name, bitmask)` to `local_tile_id`.
155    pub fn set_auto_tile_rule(&mut self, type_name: &str, bitmask: u8, local_tile_id: u32) {
156        self.auto_rules_4
157            .insert((type_name.to_string(), bitmask), local_tile_id);
158    }
159    /// Look up the 4-bit autotile local ID for `(type_name, bitmask)`, or `None` when no rule matches.
160    pub fn get_auto_tile_id(&self, type_name: &str, bitmask: u8) -> Option<u32> {
161        self.auto_rules_4
162            .get(&(type_name.to_string(), bitmask))
163            .copied()
164    }
165    /// Register an 8-bit autotile rule mapping `(type_name, bitmask)` to `local_tile_id`.
166    pub fn set_auto_tile_rule_8(&mut self, type_name: &str, bitmask: u16, local_tile_id: u32) {
167        self.auto_rules_8
168            .insert((type_name.to_string(), bitmask), local_tile_id);
169    }
170    /// Look up the 8-bit autotile local ID for `(type_name, bitmask)`, or `None` when no rule matches.
171    pub fn get_auto_tile_id_8(&self, type_name: &str, bitmask: u16) -> Option<u32> {
172        self.auto_rules_8
173            .get(&(type_name.to_string(), bitmask))
174            .copied()
175    }
176}
