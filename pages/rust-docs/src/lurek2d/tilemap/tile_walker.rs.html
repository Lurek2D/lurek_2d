1//! Cardinal facing direction with angle, delta, and rotation helpers.
2//!
3//! - Discrete grid walker with forward, backward, and strafe movement.
4//! - Previous-state snapshot for smooth frame interpolation of position and heading.
5//! - Relative-facing query to classify adjacent tiles as front, back, left, or right.
6//! - Passability checks decoupled from actual collision data.

8use std::f32::consts::PI;
9
10/// Cardinal facing direction for a grid-aligned entity.
11#[derive(Debug, Clone, Copy, PartialEq)]
12pub enum Facing {
13    /// Facing toward decreasing Y (up on screen in top-down maps).
14    North = 0,
15    /// Facing toward increasing X (right on screen).
16    East = 1,
17    /// Facing toward increasing Y (down on screen).
18    South = 2,
19    /// Facing toward decreasing X (left on screen).
20    West = 3,
21}
22/// Methods for parsing, converting, and rotating cardinal directions.
23impl Facing {
24    /// Parse a direction string (`"north"`, `"n"`, etc.); returns `None` on unknown input.
25    pub fn parse(s: &str) -> Option<Self> {
26        match s.to_lowercase().as_str() {
27            "north" | "n" => Some(Facing::North),
28            "east" | "e" => Some(Facing::East),
29            "south" | "s" => Some(Facing::South),
30            "west" | "w" => Some(Facing::West),
31            _ => None,
32        }
33    }
34    /// Return the lowercase canonical name for this direction.
35    pub fn to_str(self) -> &'static str {
36        match self {
37            Facing::North => "north",
38            Facing::East => "east",
39            Facing::South => "south",
40            Facing::West => "west",
41        }
42    }
43    /// Return the heading angle in radians: East=0, South=π/2, West=π, North=3π/2.
44    pub fn angle(self) -> f32 {
45        match self {
46            Facing::North => 3.0 * PI / 2.0,
47            Facing::East => 0.0,
48            Facing::South => PI / 2.0,
49            Facing::West => PI,
50        }
51    }
52    /// Return the X grid delta for one step in this direction: East=+1, West=-1, N/S=0.
53    pub fn dx(self) -> i32 {
54        match self {
55            Facing::East => 1,
56            Facing::West => -1,
57            _ => 0,
58        }
59    }
60    /// Return the Y grid delta for one step in this direction: South=+1, North=-1, E/W=0.
61    pub fn dy(self) -> i32 {
62        match self {
63            Facing::North => -1,
64            Facing::South => 1,
65            _ => 0,
66        }
67    }
68    /// Return the direction 90° counter-clockwise from `self`.
69    fn turn_left(self) -> Facing {
70        match self {
71            Facing::North => Facing::West,
72            Facing::West => Facing::South,
73            Facing::South => Facing::East,
74            Facing::East => Facing::North,
75        }
76    }
77    /// Return the direction 90° clockwise from `self`.
78    fn turn_right(self) -> Facing {
79        match self {
80            Facing::North => Facing::East,
81            Facing::East => Facing::South,
82            Facing::South => Facing::West,
83            Facing::West => Facing::North,
84        }
85    }
86    /// Return the direction 180° from `self`.
87    fn opposite(self) -> Facing {
88        match self {
89            Facing::North => Facing::South,
90            Facing::South => Facing::North,
91            Facing::East => Facing::West,
92            Facing::West => Facing::East,
93        }
94    }
95}
96/// A discrete grid walker that records its previous state for smooth frame interpolation.
97pub struct TileWalker {
98    /// Current grid X position.
99    x: i32,
100    /// Current grid Y position.
101    y: i32,
102    /// Current facing direction.
103    facing: Facing,
104    /// Grid X position at the last `begin_move` snapshot.
105    prev_x: i32,
106    /// Grid Y position at the last `begin_move` snapshot.
107    prev_y: i32,
108    /// Facing at the last `begin_move` snapshot.
109    prev_facing: Facing,
110}
111/// Grid movement, rotation, interpolation, and relative-facing queries.
112impl TileWalker {
113    /// Create a `TileWalker` at `(x, y)` facing `facing`; previous state initialised to the same values.
114    pub fn new(x: i32, y: i32, facing: Facing) -> Self {
115        Self {
116            x,
117            y,
118            facing,
119            prev_x: x,
120            prev_y: y,
121            prev_facing: facing,
122        }
123    }
124    /// Return current grid X. This function is part of the public API.
125    pub fn x(&self) -> i32 {
126        self.x
127    }
128    /// Return current grid Y. This function is part of the public API.
129    pub fn y(&self) -> i32 {
130        self.y
131    }
132    /// Return current facing direction.
133    pub fn facing(&self) -> Facing {
134        self.facing
135    }
136    /// Teleport to `(x, y)` without updating the previous-position snapshot.
137    pub fn set_position(&mut self, x: i32, y: i32) {
138        self.x = x;
139        self.y = y;
140    }
141    /// Set facing without updating the previous-facing snapshot.
142    pub fn set_facing(&mut self, facing: Facing) {
143        self.facing = facing;
144    }
145    /// Internal passability check; always returns `true` — callers override with collision guards before calling movement methods.
146    fn can_move_to(&self, _tx: i32, _ty: i32) -> bool {
147        true
148    }
149    /// Return `true` when the tile one step forward is passable.
150    pub fn can_move_forward(&self) -> bool {
151        self.can_move_to(self.x + self.facing.dx(), self.y + self.facing.dy())
152    }
153    /// Return `true` when the tile one step backward is passable.
154    pub fn can_move_backward(&self) -> bool {
155        let back = self.facing.opposite();
156        self.can_move_to(self.x + back.dx(), self.y + back.dy())
157    }
158    /// Return `true` when the tile one step to the left is passable.
159    pub fn can_strafe_left(&self) -> bool {
160        let left = self.facing.turn_left();
161        self.can_move_to(self.x + left.dx(), self.y + left.dy())
162    }
163    /// Return `true` when the tile one step to the right is passable.
164    pub fn can_strafe_right(&self) -> bool {
165        let right = self.facing.turn_right();
166        self.can_move_to(self.x + right.dx(), self.y + right.dy())
167    }
168    /// Move one step forward; returns `true` when movement succeeded.
169    pub fn move_forward(&mut self) -> bool {
170        let tx = self.x + self.facing.dx();
171        let ty = self.y + self.facing.dy();
172        if self.can_move_to(tx, ty) {
173            self.x = tx;
174            self.y = ty;
175            true
176        } else {
177            false
178        }
179    }
180    /// Move one step backward; returns `true` when movement succeeded.
181    pub fn move_backward(&mut self) -> bool {
182        let back = self.facing.opposite();
183        let tx = self.x + back.dx();
184        let ty = self.y + back.dy();
185        if self.can_move_to(tx, ty) {
186            self.x = tx;
187            self.y = ty;
188            true
189        } else {
190            false
191        }
192    }
193    /// Strafe one step to the left; returns `true` when movement succeeded.
194    pub fn strafe_left(&mut self) -> bool {
195        let left = self.facing.turn_left();
196        let tx = self.x + left.dx();
197        let ty = self.y + left.dy();
198        if self.can_move_to(tx, ty) {
199            self.x = tx;
200            self.y = ty;
201            true
202        } else {
203            false
204        }
205    }
206    /// Strafe one step to the right; returns `true` when movement succeeded.
207    pub fn strafe_right(&mut self) -> bool {
208        let right = self.facing.turn_right();
209        let tx = self.x + right.dx();
210        let ty = self.y + right.dy();
211        if self.can_move_to(tx, ty) {
212            self.x = tx;
213            self.y = ty;
214            true
215        } else {
216            false
217        }
218    }
219    /// Rotate facing 90° counter-clockwise.
220    pub fn turn_left(&mut self) {
221        self.facing = self.facing.turn_left();
222    }
223    /// Rotate facing 90° clockwise.
224    pub fn turn_right(&mut self) {
225        self.facing = self.facing.turn_right();
226    }
227    /// Rotate facing 180°. This function is part of the public API.
228    pub fn turn_around(&mut self) {
229        self.facing = self.facing.opposite();
230    }
231    /// Snapshot current position and facing into `prev_*` for interpolation; call before each discrete move.
232    pub fn begin_move(&mut self) {
233        self.prev_x = self.x;
234        self.prev_y = self.y;
235        self.prev_facing = self.facing;
236    }
237    /// Linearly interpolate between `prev` and current position at blend factor `t` clamped to `[0, 1]`.
238    pub fn get_interpolated_position(&self, t: f32) -> (f32, f32) {
239        let t = t.clamp(0.0, 1.0);
240        let ix = self.prev_x as f32 + (self.x - self.prev_x) as f32 * t;
241        let iy = self.prev_y as f32 + (self.y - self.prev_y) as f32 * t;
242        (ix, iy)
243    }
244    /// Interpolate heading angle between previous and current facing at `t`, handling wrap-around correctly.
245    pub fn get_interpolated_angle(&self, t: f32) -> f32 {
246        let t = t.clamp(0.0, 1.0);
247        let a1 = self.prev_facing.angle();
248        let a2 = self.facing.angle();
249        let mut diff = a2 - a1;
250        if diff > PI {
251            diff -= 2.0 * PI;
252        } else if diff < -PI {
253            diff += 2.0 * PI;
254        }
255        a1 + diff * t
256    }
257    /// Return `"front"`, `"back"`, `"left"`, or `"right"` describing where `(tx, ty)` is relative to the walker.
258    pub fn get_relative_facing(&self, tx: i32, ty: i32) -> &'static str {
259        let dx = tx - self.x;
260        let dy = ty - self.y;
261        let fwd_x = self.facing.dx();
262        let fwd_y = self.facing.dy();
263        if dx == fwd_x && dy == fwd_y {
264            return "front";
265        }
266        let back = self.facing.opposite();
267        if dx == back.dx() && dy == back.dy() {
268            return "back";
269        }
270        let left = self.facing.turn_left();
271        if dx == left.dx() && dy == left.dy() {
272            return "left";
273        }
274        let right = self.facing.turn_right();
275        if dx == right.dx() && dy == right.dy() {
276            return "right";
277        }
278        let dot_fwd = dx * fwd_x + dy * fwd_y;
279        if dot_fwd > 0 {
280            "front"
281        } else if dot_fwd < 0 {
282            "back"
283        } else {
284            let cross = fwd_x * dy - fwd_y * dx;
285            if cross > 0 {
286                "right"
287            } else {
288                "left"
289            }
290        }
291    }
