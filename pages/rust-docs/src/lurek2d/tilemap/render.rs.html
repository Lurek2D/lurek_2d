1//! Camera-culled render-command generation for tile-map layers.
2//!
3//! - GID-to-color debug palette for fallback colored tile rendering.
4//! - Per-layer visibility and tint applied during command emission.
5
6use super::tilemap::TileMap;
7use crate::render::renderer::{DrawMode, RenderCommand};
8
9/// Map a GID to a debug-palette RGB triple for fallback colored tile rendering.
10fn gid_to_color(gid: u32) -> (f32, f32, f32) {
11    if gid >= 10 {
12        match gid {
13            10 => (200.0 / 255.0, 50.0 / 255.0, 50.0 / 255.0),
14            11 => (50.0 / 255.0, 50.0 / 255.0, 200.0 / 255.0),
15            12 => (200.0 / 255.0, 200.0 / 255.0, 50.0 / 255.0),
16            _ => (1.0, 1.0, 1.0),
17        }
18    } else {
19        match gid {
20            1 => (80.0 / 255.0, 160.0 / 255.0, 80.0 / 255.0),
21            2 => (60.0 / 255.0, 120.0 / 255.0, 60.0 / 255.0),
22            _ => (40.0 / 255.0, 40.0 / 255.0, 40.0 / 255.0),
23        }
24    }
25}
26/// Render-command generation for `TileMap`.
27impl TileMap {
28    #[allow(clippy::too_many_arguments)]
29    #[allow(clippy::manual_clamp)]
30    /// Generate camera-culled `RenderCommand` primitives for all visible layers; returns an empty vec when the map has no layers or zero tile size.
31    pub fn generate_render_commands(
32        &self,
33        offset_x: f32,
34        offset_y: f32,
35        cam_x: f32,
36        cam_y: f32,
37        cam_w: f32,
38        cam_h: f32,
39    ) -> Vec<RenderCommand> {
40        let mut cmds = Vec::new();
41        if self.get_layer_count() == 0 {
42            return cmds;
43        }
44        let tw = self.get_tile_width() as f32;
45        let th = self.get_tile_height() as f32;
46        if tw <= 0.0 || th <= 0.0 {
47            return cmds;
48        }
49        let tile_x0 = ((cam_x / tw).floor() as i64).max(0) as u32;
50        let tile_y0 = ((cam_y / th).floor() as i64).max(0) as u32;
51        let tile_x1_cam = ((cam_x + cam_w) / tw).ceil() as u32;
52        let tile_y1_cam = ((cam_y + cam_h) / th).ceil() as u32;
53        for layer_idx in 0..self.get_layer_count() {
54            if !self.get_layer_visible(layer_idx) {
55                continue;
56            }
57            let Some((lw, lh)) = self.get_layer_dimensions(layer_idx) else {
58                continue;
59            };
60            let [lt_r, lt_g, lt_b, lt_a] = self.get_layer_color(layer_idx);
61            let x_end = lw.min(tile_x1_cam);
62            let y_end = lh.min(tile_y1_cam);
63            for ty in tile_y0.min(lh)..y_end {
64                for tx in tile_x0.min(lw)..x_end {
65                    let gid = self.get_tile(layer_idx, tx, ty);
66                    if gid == 0 {
67                        continue;
68                    }
69                    let (gr, gg, gb) = gid_to_color(gid);
70                    let cr = gr * lt_r;
71                    let cg = gg * lt_g;
72                    let cb = gb * lt_b;
73                    let ca = lt_a;
74                    let world_x = tx as f32 * tw;
75                    let world_y = ty as f32 * th;
76                    let sx = offset_x + world_x;
77                    let sy = offset_y + world_y;
78                    cmds.push(RenderCommand::SetColor(cr, cg, cb, ca));
79                    if gid >= 10 {
80                        cmds.push(RenderCommand::Circle {
81                            mode: DrawMode::Fill,
82                            x: sx + tw * 0.5,
83                            y: sy + th * 0.5,
84                            r: (tw * 0.5).min(6.0).max(3.0),
85                        });
86                    } else {
87                        cmds.push(RenderCommand::Rectangle {
88                            mode: DrawMode::Fill,
89                            x: sx,
90                            y: sy,
91                            w: tw,
92                            h: th,
93                        });
94                    }
95                }
96            }
97        }
98        cmds
99    }
