1//! Import LDtk project JSON into the engine tilemap representation.
2//!
3//! - Parse levels, tile layers, and auto-layers with tileset geometry reconstruction.
4//! - Map LDtk pixel-based tile coordinates to grid-cell indices.
5
6use super::tilemap::TileMap;
7use super::tileset::TileSet;
8
9/// Parse `json_str` as an LDtk project, import the level named `level_name` (or the first level when `None`), and return a `TileMap`.
10pub fn load_ldtk(json_str: &str, level_name: Option<&str>) -> Result<TileMap, String> {
11    let root: serde_json::Value =
12        serde_json::from_str(json_str).map_err(|e| format!("LDtk JSON parse error: {}", e))?;
13    let levels = root
14        .get("levels")
15        .and_then(|v| v.as_array())
16        .ok_or("LDtk JSON missing 'levels' array")?;
17    let level = if let Some(name) = level_name {
18        levels
19            .iter()
20            .find(|l| l.get("identifier").and_then(|v| v.as_str()) == Some(name))
21            .ok_or_else(|| format!("LDtk level '{}' not found", name))?
22    } else {
23        levels.first().ok_or("LDtk project has no levels")?
24    };
25    let layer_instances = level
26        .get("layerInstances")
27        .and_then(|v| v.as_array())
28        .ok_or("LDtk level missing 'layerInstances'")?;
29    let tile_layer = layer_instances.iter().rev().find(|l| is_tile_layer(l));
30    let (grid_size, map_width, map_height) = if let Some(tl) = tile_layer {
31        (
32            tl.get("__gridSize").and_then(|v| v.as_u64()).unwrap_or(16) as u32,
33            tl.get("__cWid").and_then(|v| v.as_u64()).unwrap_or(0) as u32,
34            tl.get("__cHei").and_then(|v| v.as_u64()).unwrap_or(0) as u32,
35        )
36    } else {
37        return Err("LDtk level contains no tile or auto-layer layers".into());
38    };
39    let mut map = TileMap::new(grid_size, grid_size, 16);
40    for layer in layer_instances.iter().rev() {
41        if !is_tile_layer(layer) {
42            continue;
43        }
44        let layer_name = layer
45            .get("__identifier")
46            .and_then(|v| v.as_str())
47            .unwrap_or("layer");
48        let layer_idx = map.add_layer(layer_name, map_width, map_height);
49        let grid_tiles = layer
50            .get("gridTiles")
51            .or_else(|| layer.get("autoLayerTiles"))
52            .and_then(|v| v.as_array());
53        let max_tile_id = grid_tiles
54            .map(|tiles| {
55                tiles
56                    .iter()
57                    .filter_map(|t| t.get("t").and_then(|v| v.as_u64()))
58                    .max()
59                    .unwrap_or(0)
60            })
61            .unwrap_or(0) as u32;
62        let tileset_w_px = layer
63            .get("__tilesetRelPath")
64            .and_then(|_| {
65                layer.get("__tilesetDefUid").and_then(|_| {
66                    root.get("defs")
67                        .and_then(|d| d.get("tilesets"))
68                        .and_then(|ts| ts.as_array())
69                        .and_then(|arr| {
70                            let uid = layer.get("__tilesetDefUid")?.as_u64()?;
71                            arr.iter()
72                                .find(|ts| ts.get("uid").and_then(|v| v.as_u64()) == Some(uid))
73                        })
74                        .and_then(|ts| ts.get("pxWid").and_then(|v| v.as_u64()))
75                })
76            })
77            .unwrap_or((max_tile_id + 1) as u64 * grid_size as u64)
78            as u32;
79        let columns = if grid_size > 0 {
80            (tileset_w_px / grid_size).max(1)
81        } else {
82            1
83        };
84        let tile_count = max_tile_id + 1;
85        let tileset = TileSet::new(1, tile_count.max(1), columns, grid_size, grid_size, 0, 0);
86        map.add_tileset(tileset);
87        if let Some(tiles) = grid_tiles {
88            for tile in tiles {
89                let px = tile.get("px").and_then(|v| v.as_array());
90                let tile_id = tile.get("t").and_then(|v| v.as_u64()).unwrap_or(0) as u32;
91                if let Some(px) = px {
92                    let px_x = px.first().and_then(|v| v.as_u64()).unwrap_or(0) as u32;
93                    let px_y = px.get(1).and_then(|v| v.as_u64()).unwrap_or(0) as u32;
94                    let tx = if grid_size > 0 { px_x / grid_size } else { 0 };
95                    let ty = if grid_size > 0 { px_y / grid_size } else { 0 };
96                    map.set_tile(layer_idx, tx, ty, tile_id + 1);
97                }
98            }
99        }
100    }
101    Ok(map)
102}
103/// Return `true` when `layer` is a `"Tiles"` or `"AutoLayer"` layer type.
104fn is_tile_layer(layer: &serde_json::Value) -> bool {
105    matches!(
106        layer.get("__type").and_then(|v| v.as_str()),
107        Some("Tiles") | Some("AutoLayer")
108    )
