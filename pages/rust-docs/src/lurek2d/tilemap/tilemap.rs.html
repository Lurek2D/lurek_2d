1//! Multi-layer tile map with per-tile GID storage, tint overrides, and parallax scroll factors.
2//!
3//! - Tileset attachment and GID resolution across multiple tileset ranges.
4//! - 4-neighbour and 8-neighbour autotile bitmask computation and GID substitution.
5//! - Continuous AABB sweep-cast collision against solid tiles for platformer and top-down physics.
6//! - Per-GID animation timer advancement using tileset frame data.
7//! - World-to-tile and tile-to-world coordinate conversion respecting tile dimensions.
8//! - Viewport-aware culled render-command generation for debug colour-coded output.
9//! - Debug image rendering: full-map, per-layer side-by-side, and highlight-overlay modes.
10//! - Boolean walkability grid export for pathfinding integration.
11//! - GID-to-position reverse index cache for fast spatial queries by tile type.
12

20/// A single tile layer with per-tile GID and optional per-tile tint data.
21#[derive(Debug, Clone)]
22pub struct TileLayer {
23    /// Layer identifier shown in editors and log messages.
24    pub name: String,
25    /// Tile count along the X axis.
26    pub width: u32,
27    /// Tile count along the Y axis.
28    pub height: u32,
29    /// Whether this layer is included in render-command output.
30    pub visible: bool,
31    /// Multiplicative tint `[r, g, b, a]` applied to every tile in the layer.
32    pub tint: [f32; 4],
33    /// World-space draw offset applied to this layer.
34    pub offset: Vec2,
35    /// Per-axis parallax scroll factor; `(1, 1)` = full scroll, `(0, 0)` = fixed.
36    pub parallax: Vec2,
37    /// Flat GID array in row-major order; `0` means empty.
38    tiles: Vec<u32>,
39    /// Optional per-tile tint override; `None` falls back to `tint`.
40    tile_tints: Vec<Option<[f32; 4]>>,
41}
42impl TileLayer {
43    /// Create a fully-empty `TileLayer` of `width × height` tiles with all GIDs set to `0`.
44    fn new(name: &str, width: u32, height: u32) -> Self {
45        let cap = (width * height) as usize;
46        Self {
47            name: name.to_string(),
48            width,
49            height,
50            visible: true,
51            tint: [1.0, 1.0, 1.0, 1.0],
52            offset: Vec2::ZERO,
53            parallax: Vec2::new(1.0, 1.0),
54            tiles: vec![0u32; cap],
55            tile_tints: vec![None; cap],
56        }
57    }
58    /// Convert `(x, y)` into a flat index; returns `None` when out of bounds.
59    fn index(&self, x: u32, y: u32) -> Option<usize> {
60        if x < self.width && y < self.height {
61            Some((y * self.width + x) as usize)
62        } else {
63            None
64        }
65    }
66}

200    /// Write GID `gid` at `(x, y)` in layer `layer`; updates the type-index cache; no-op when out of bounds.

214            }
215        }

239    /// Fill all tiles in `layer` with `gid`; rebuilds the type-index cache for that layer.
240    pub fn fill(&mut self, layer: usize, gid: u32) {
241        if let Some(l) = self.layers.get_mut(layer) {
242            for tile in l.tiles.iter_mut() {
243                *tile = gid;
244            }
245            if let Some(layer_index) = self.tile_type_index_cache.get_mut(layer) {
246                layer_index.clear();
247                if gid != 0 {
248                    let mut positions = Vec::with_capacity((l.width * l.height) as usize);
249                    for y in 0..l.height {
250                        for x in 0..l.width {
251                            positions.push((x, y));
252                        }
253                    }
254                    layer_index.insert(gid, positions);
255                }
256            }
257        }
258    }

281    /// Advance all GID animation timers by `dt` seconds; updates frame indices for each animated tileset.
282    pub fn update(&mut self, dt: f32) {
283        let dt_ms = dt * 1000.0;
284        for ts in &self.tilesets {
285            let first_gid = ts.get_first_gid();
286            let count = ts.get_tile_count();
287            for local_id in 0..count {
288                if let Some(frames) = ts.get_animation(local_id) {
289                    if frames.is_empty() {
290                        continue;
291                    }
292                    let gid = first_gid + local_id;
293                    let (frame_idx, elapsed) = self.anim_timers.entry(gid).or_insert((0, 0.0));
294                    *elapsed += dt_ms;
295                    while *elapsed >= frames[*frame_idx].duration_ms {
296                        *elapsed -= frames[*frame_idx].duration_ms;
297                        *frame_idx = (*frame_idx + 1) % frames.len();
298                    }
299                }
300            }
301        }
302    }
303    /// Convert world position `(wx, wy)` to tile grid coordinates; clamps negative values to `0`.
304    pub fn world_to_tile(&self, wx: f32, wy: f32) -> (u32, u32) {
305        let tx = if wx < 0.0 {
306            0
307        } else {
308            (wx / self.tile_width as f32) as u32
309        };
310        let ty = if wy < 0.0 {
311            0
312        } else {
313            (wy / self.tile_height as f32) as u32
314        };
315        (tx, ty)
316    }

348    /// Resolve global GID `gid` to a `(tileset_index, local_id)` pair; returns `None` for GID `0` or unknown GIDs.
349    fn resolve_gid(&self, gid: u32) -> Option<(usize, u32)> {
350        if gid == 0 {
351            return None;
352        }
353        for (i, ts) in self.tilesets.iter().enumerate() {
354            let first = ts.get_first_gid();
355            if gid >= first && gid < first + ts.get_tile_count() {
356                return Some((i, gid - first));
357            }
358        }
359        None
360    }

370    /// Return `true` when any tile overlapped by `rect` in `layer` is solid.
371    pub fn rect_overlaps_solid(&self, layer: usize, rect: Rect) -> bool {
372        let (tx0, ty0) = self.world_to_tile(rect.x, rect.y);
373        let (tx1, ty1) =
374            self.world_to_tile(rect.x + rect.width - 0.001, rect.y + rect.height - 0.001);
375        for ty in ty0..=ty1 {
376            for tx in tx0..=tx1 {
377                if self.is_solid(layer, tx, ty) {
378                    return true;
379                }
380            }
381        }
382        false
383    }
384    /// Continuous AABB sweep through solid tiles in `layer` along `(dx, dy)`; returns the earliest hit or `None`.
385    pub fn sweep_rect(&self, layer: usize, rect: Rect, dx: f32, dy: f32) -> Option<SweepResult> {
386        if dx == 0.0 && dy == 0.0 {
387            return None;
388        }
389        let min_x = rect.x.min(rect.x + dx);
390        let min_y = rect.y.min(rect.y + dy);
391        let max_x = (rect.x + rect.width).max(rect.x + rect.width + dx);
392        let max_y = (rect.y + rect.height).max(rect.y + rect.height + dy);
393        let (tx0, ty0) = self.world_to_tile(min_x, min_y);
394        let (tx1, ty1) = self.world_to_tile(max_x, max_y);
395        let tw = self.tile_width as f32;
396        let th = self.tile_height as f32;
397        let mut best: Option<SweepResult> = None;
398        for ty in ty0..=ty1 {
399            for tx in tx0..=tx1 {
400                if !self.is_solid(layer, tx, ty) {
401                    continue;
402                }
403                let tile_rect = Rect::new(tx as f32 * tw, ty as f32 * th, tw, th);
404                if let Some(result) = sweep_aabb_vs_aabb(rect, dx, dy, tile_rect, tx, ty) {
405                    if best.is_none()
406                        || result.t < best.as_ref().expect("best is Some when not is_none").t
407                    {
408                        best = Some(result);
409                    }
410                }
411            }
412        }
413        best
414    }
415    /// Apply 4-neighbour autotile GID substitution to all non-empty tiles in `layer` matching `type_name`.
416    pub fn apply_autotile(&mut self, layer: usize, type_name: &str) {
417        let (width, height) = match self.layers.get(layer) {
418            Some(l) => (l.width, l.height),
419            None => return,
420        };
421        let mut replacements = Vec::new();
422        for y in 0..height {
423            for x in 0..width {
424                if self.get_tile(layer, x, y) == 0 {
425                    continue;
426                }
427                let mask = self.compute_bitmask_4(layer, x, y, width, height);
428                if let Some(new_gid) = self.lookup_autotile_4(type_name, mask) {
429                    replacements.push((x, y, new_gid));
430                }
431            }
432        }
433        for (x, y, gid) in replacements {
434            self.set_tile(layer, x, y, gid);
435        }
436    }
437    /// Apply 4-neighbour autotile substitution to the 3×3 neighbourhood around `(x, y)` only.
438    pub fn apply_autotile_at(&mut self, layer: usize, x: u32, y: u32, type_name: &str) {

440            Some(l) => (l.width, l.height),
441            None => return,
442        };
443        let mut replacements = Vec::new();

445        let y_start = y.saturating_sub(1);
446        let x_end = (x + 1).min(width - 1);
447        let y_end = (y + 1).min(height - 1);

458        }

461        }
462    }

511    /// Compute the 4-bit cardinal-neighbour bitmask for `(x, y)` in `layer`; bits: N=1, E=2, S=4, W=8.
512    fn compute_bitmask_4(&self, layer: usize, x: u32, y: u32, width: u32, height: u32) -> u8 {
513        let mut mask = 0u8;
514        if y > 0 && self.get_tile(layer, x, y - 1) != 0 {
515            mask |= 1;
516        }
517        if x + 1 < width && self.get_tile(layer, x + 1, y) != 0 {
518            mask |= 2;
519        }
520        if y + 1 < height && self.get_tile(layer, x, y + 1) != 0 {
521            mask |= 4;
522        }
523        if x > 0 && self.get_tile(layer, x - 1, y) != 0 {
524            mask |= 8;
525        }
526        mask
527    }
528    /// Compute the 8-bit full-neighbour bitmask for `(x, y)` in `layer`; cardinals=1..8, diagonals=16..128.
529    fn compute_bitmask_8(&self, layer: usize, x: u32, y: u32, width: u32, height: u32) -> u16 {
530        let n = y > 0 && self.get_tile(layer, x, y - 1) != 0;
531        let e = x + 1 < width && self.get_tile(layer, x + 1, y) != 0;
532        let s = y + 1 < height && self.get_tile(layer, x, y + 1) != 0;
533        let w = x > 0 && self.get_tile(layer, x - 1, y) != 0;
534        let mut mask: u16 = 0;
535        if n {
536            mask |= 1;
537        }
538        if e {
539            mask |= 2;
540        }
541        if s {
542            mask |= 4;
543        }
544        if w {
545            mask |= 8;
546        }
547        if n && e && y > 0 && x + 1 < width && self.get_tile(layer, x + 1, y - 1) != 0 {
548            mask |= 16;
549        }
550        if s && e && y + 1 < height && x + 1 < width && self.get_tile(layer, x + 1, y + 1) != 0 {
551            mask |= 32;
552        }
553        if s && w && y + 1 < height && x > 0 && self.get_tile(layer, x - 1, y + 1) != 0 {
554            mask |= 64;
555        }
556        if n && w && y > 0 && x > 0 && self.get_tile(layer, x - 1, y - 1) != 0 {
557            mask |= 128;
558        }
559        mask
560    }

579    /// Render all layers to an `ImageData` using the debug color palette; `tile_size` is the render pixel size per tile.
580    pub fn draw_to_image(&self, tile_size: u32) -> crate::image::ImageData {
581        if self.layers.is_empty() {
582            return crate::image::ImageData::new(1, 1);
583        }
584        let lw = self.layers[0].width;
585        let lh = self.layers[0].height;
586        let mut img = crate::image::ImageData::new(lw * tile_size, lh * tile_size);
587        img.fill(20, 20, 30, 255);
588        for (li, layer) in self.layers.iter().enumerate() {
589            let w = layer.width.min(lw);
590            let h = layer.height.min(lh);
591            for y in 0..h {
592                for x in 0..w {
593                    let idx = layer.index(x, y).unwrap_or(0);
594                    let gid = layer.tiles[idx];
595                    if gid == 0 && li > 0 {
596                        continue;
597                    }
598                    if gid >= 10 {
599                        let (r, g, b) = match gid {
600                            10 => (200u8, 50, 50),
601                            11 => (50, 50, 200),
602                            12 => (200, 200, 50),
603                            _ => (255, 255, 255),
604                        };
605                        let cx = (x * tile_size + tile_size / 2) as i32;
606                        let cy = (y * tile_size + tile_size / 2) as i32;
607                        img.draw_circle(cx, cy, 6, r, g, b, 255);
608                    } else {
609                        let (r, g, b) = match gid {
610                            1 => (80u8, 160, 80),
611                            2 => (60, 120, 60),
612                            _ => (40, 40, 40),
613                        };
614                        for py in 0..tile_size {
615                            for px in 0..tile_size {
616                                img.set_pixel(x * tile_size + px, y * tile_size + py, r, g, b, 255);
617                            }
618                        }
619                        if li == 0 {
620                            for px in 0..tile_size {
621                                img.set_pixel(x * tile_size + px, y * tile_size, 30, 30, 50, 255);
622                            }
623                            for py in 0..tile_size {
624                                img.set_pixel(x * tile_size, y * tile_size + py, 30, 30, 50, 255);
625                            }
626                        }
627                    }
628                }
629            }
630        }
631        img
632    }
633    /// Build a flat `RenderCommand` list for all visible layers using the debug color palette and `offset`.
634    pub fn build_render_commands(
635        &self,
636        offset_x: f32,
637        offset_y: f32,
638    ) -> Vec<crate::render::renderer::RenderCommand> {
639        use crate::render::renderer::{DrawMode, RenderCommand};
640        let mut cmds: Vec<RenderCommand> = Vec::new();
641        if self.layers.is_empty() {
642            return cmds;
643        }
644        let lw = self.layers[0].width;
645        let lh = self.layers[0].height;
646        let tw = self.tile_width as f32;
647        let th = self.tile_height as f32;
648        for (li, layer) in self.layers.iter().enumerate() {
649            if !layer.visible {
650                continue;
651            }
652            let w = layer.width.min(lw);
653            let h = layer.height.min(lh);
654            for y in 0..h {
655                for x in 0..w {
656                    let idx = layer.index(x, y).unwrap_or(0);
657                    let gid = layer.tiles[idx];
658                    if gid == 0 && li > 0 {
659                        continue;
660                    }
661                    let (r, g, b): (f32, f32, f32) = if gid >= 10 {
662                        match gid {
663                            10 => (200.0 / 255.0, 50.0 / 255.0, 50.0 / 255.0),
664                            11 => (50.0 / 255.0, 50.0 / 255.0, 200.0 / 255.0),
665                            12 => (200.0 / 255.0, 200.0 / 255.0, 50.0 / 255.0),
666                            _ => (1.0, 1.0, 1.0),
667                        }
668                    } else {
669                        match gid {
670                            1 => (80.0 / 255.0, 160.0 / 255.0, 80.0 / 255.0),
671                            2 => (60.0 / 255.0, 120.0 / 255.0, 60.0 / 255.0),
672                            _ => (40.0 / 255.0, 40.0 / 255.0, 40.0 / 255.0),
673                        }
674                    };
675                    let tint = layer.tile_tints[idx].unwrap_or(layer.tint);
676                    cmds.push(RenderCommand::SetColor(
677                        r * tint[0],
678                        g * tint[1],
679                        b * tint[2],
680                        tint[3],
681                    ));
682                    if gid >= 10 {
683                        let cx = offset_x + x as f32 * tw + tw * 0.5;
684                        let cy = offset_y + y as f32 * th + th * 0.5;
685                        cmds.push(RenderCommand::Circle {
686                            mode: DrawMode::Fill,
687                            x: cx,
688                            y: cy,
689                            r: 6.0,
690                        });
691                    } else {
692                        cmds.push(RenderCommand::Rectangle {
693                            mode: DrawMode::Fill,
694                            x: offset_x + x as f32 * tw,
695                            y: offset_y + y as f32 * th,
696                            w: tw,
697                            h: th,
698                        });
699                    }
700                }
701            }
702        }
703        cmds
704    }
705    /// Render world-space highlight points and a grid overlay into an `ImageData` of `img_width × img_height`.

751    /// Render all layers side-by-side with colour-coded GIDs into an `ImageData` of `width × height` pixels.
752    pub fn draw_layers_to_image(
753        &self,
754        tile_px: u32,
755        width: u32,
756        height: u32,
757    ) -> crate::image::ImageData {
758        let mut img = crate::image::ImageData::new(width, height);
759        img.fill(25, 25, 35, 255);
760        let margin = 10i32;
761        for layer_idx in 0..self.get_layer_count() {
762            let dims = self.get_layer_dimensions(layer_idx);
763            let (lw, lh) = dims.unwrap_or((0, 0));
764            for y in 0..lh {
765                for x in 0..lw {
766                    let gid = self.get_tile(layer_idx, x, y);
767                    if gid == 0 {
768                        continue;
769                    }
770                    let px = x as i32 * tile_px as i32 + margin;
771                    let py = y as i32 * tile_px as i32 + margin;
772                    let (r, g, b) = match (layer_idx, gid) {
773                        (0, 1) => (40u8, 80, 40),
774                        (_, 2) => (120, 80, 60),
775                        (_, 3) => (100, 70, 50),
776                        (_, 4) => (80, 60, 40),
777                        _ => (100, 100, 100),
778                    };
779                    if layer_idx == 0 {
780                        img.draw_rect(px, py, tile_px, tile_px, r, g, b, 255);
781                    } else {
782                        img.draw_rect(
783                            px + 1,
784                            py + 1,
785                            tile_px.saturating_sub(2),
786                            tile_px.saturating_sub(2),
787                            r,
788                            g,
789                            b,
790                            255,
791                        );
792                    }
793                }
794            }
795        }
796        let label = format!("{} LAYERS", self.get_layer_count());
797        img.draw_label(&label, margin, (height - 20) as i32, 200, 200, 200);
798        img.draw_label("TILEMAP LAYERS OK", 80, (height - 20) as i32, 100, 255, 100);
799        img
800    }
801    /// Convert a layer to a boolean walkability grid; `walkable_gids` are treated as passable, GID `0` is always passable.
802    pub fn to_nav_grid(&self, layer: usize, walkable_gids: &[u32]) -> Vec<Vec<bool>> {
803        let (width, height) = self.get_layer_dimensions(layer).unwrap_or((0, 0));
804        let mut grid: Vec<Vec<bool>> = Vec::with_capacity(height as usize);
805        for y in 0..height {
806            let mut row: Vec<bool> = Vec::with_capacity(width as usize);
807            for x in 0..width {
808                let gid = self.get_tile(layer, x, y);
809                let walkable = gid == 0 || walkable_gids.contains(&gid);
810                row.push(walkable);
811            }
812            grid.push(row);
813        }
814        grid
815    }

817/// Remove position `(x, y)` from the GID entry in `layer_index`; removes the key entirely when the list becomes empty.
818fn remove_pos_from_gid(layer_index: &mut HashMap<u32, Vec<(u32, u32)>>, gid: u32, x: u32, y: u32) {
819    if let Some(list) = layer_index.get_mut(&gid) {
820        if let Some(pos_idx) = list.iter().position(|&(px, py)| px == x && py == y) {
821            list.swap_remove(pos_idx);
822        }
823        if list.is_empty() {
824            layer_index.remove(&gid);
825        }
826    }
827}
828/// Swept AABB-vs-AABB narrow-phase test; returns a `SweepResult` when the expanded target is hit in `[0, 1)`, or `None`.
829fn sweep_aabb_vs_aabb(
830    mover: Rect,
831    dx: f32,
832    dy: f32,
833    target: Rect,
834    tile_x: u32,
835    tile_y: u32,
836) -> Option<SweepResult> {
837    let ex = Rect::new(
838        target.x - mover.width,
839        target.y - mover.height,
840        target.width + mover.width,
841        target.height + mover.height,
842    );
843    let origin_x = mover.x;
844    let origin_y = mover.y;
845    let (t_near_x, t_far_x) = if dx != 0.0 {
846        let inv = 1.0 / dx;
847        let t1 = (ex.x - origin_x) * inv;
848        let t2 = (ex.x + ex.width - origin_x) * inv;
849        if t1 < t2 {
850            (t1, t2)
851        } else {
852            (t2, t1)
853        }
854    } else {
855        if origin_x < ex.x || origin_x >= ex.x + ex.width {
856            return None;
857        }
858        (f32::NEG_INFINITY, f32::INFINITY)
859    };
860    let (t_near_y, t_far_y) = if dy != 0.0 {
861        let inv = 1.0 / dy;
862        let t1 = (ex.y - origin_y) * inv;
863        let t2 = (ex.y + ex.height - origin_y) * inv;
864        if t1 < t2 {
865            (t1, t2)
866        } else {
867            (t2, t1)
868        }
869    } else {
870        if origin_y < ex.y || origin_y >= ex.y + ex.height {
871            return None;
872        }
873        (f32::NEG_INFINITY, f32::INFINITY)
874    };
875    let t_near = t_near_x.max(t_near_y);
876    let t_far = t_far_x.min(t_far_y);
877    if t_near >= t_far || t_far <= 0.0 || t_near >= 1.0 {
878        return None;
879    }
880    let t = t_near.max(0.0);
881    let normal = if t_near_x > t_near_y {
882        Vec2::new(if dx > 0.0 { -1.0 } else { 1.0 }, 0.0)
883    } else {
884        Vec2::new(0.0, if dy > 0.0 { -1.0 } else { 1.0 })
885    };
886    let contact = Vec2::new(origin_x + dx * t, origin_y + dy * t);
887    Some(SweepResult {
888        contact_point: contact,
889        normal,
890        tile_x,
891        tile_y,
892        t,
893    })
