1//! Infinite sparse tile grid partitioned into fixed-size square chunks.
2//!
3//! - On-demand chunk allocation and explicit load/unload lifecycle.
4//! - Tile read/write by world coordinates with automatic chunk decomposition.
5//! - Rectangular fill, chunk enumeration, and view-frustum culling helpers.
6//! - World-space geometry queries for chunk bounds and overlap testing.
7
8use crate::log_msg;
9use crate::math::Rect;
10use crate::runtime::log_messages::{CK01, CK02, CK03};
11use std::collections::HashMap;
12
13/// Infinite tile grid partitioned into fixed-size square chunks loaded on demand.
14#[derive(Debug, Clone)]
15pub struct ChunkMap {
16    /// Side length in tiles of each square chunk.
17    chunk_size: u32,
18    /// Sparse map from chunk coordinates to flattened tile GID arrays.
19    chunks: HashMap<(i32, i32), Vec<u32>>,
20}
21impl ChunkMap {
22    /// GID value used for tiles in unloaded or unset chunks.
23    pub const DEFAULT_GID: u32 = 0;
24
25    /// Create a `ChunkMap` with the given `chunk_size`; panics when `chunk_size` is zero.
26    pub fn new(chunk_size: u32) -> Self {
27        assert!(chunk_size >= 1, "chunk_size must be >= 1");
28        log_msg!(debug, CK01, "chunk_size={}", chunk_size);
29        Self {
30            chunk_size,
31            chunks: HashMap::new(),
32        }
33    }
34    /// Return the side length in tiles of each chunk.
35    pub fn get_chunk_size(&self) -> u32 {
36        self.chunk_size
37    }
38
39    /// Return the GID at tile `(x, y)`; returns `DEFAULT_GID` when the chunk is not loaded.
40    pub fn get_tile(&self, x: i32, y: i32) -> u32 {
41        let (cx, cy, lx, ly) = self.decompose(x, y);
42        match self.chunks.get(&(cx, cy)) {
43            Some(chunk) => chunk[(ly * self.chunk_size + lx) as usize],
44            None => Self::DEFAULT_GID,
45        }
46    }
47    /// Write `gid` to tile `(x, y)`, allocating the chunk if needed.
48    pub fn set_tile(&mut self, x: i32, y: i32, gid: u32) {
49        let (cx, cy, lx, ly) = self.decompose(x, y);
50        let cs = self.chunk_size;
51        let chunk = self
52            .chunks
53            .entry((cx, cy))
54            .or_insert_with(|| vec![0u32; (cs * cs) as usize]);
55        chunk[(ly * cs + lx) as usize] = gid;
56    }
57    /// Reset tile `(x, y)` to GID 0, allocating the chunk if needed.
58    pub fn clear_tile(&mut self, x: i32, y: i32) {
59        self.set_tile(x, y, 0);
60    }
61
62    /// Fill all tiles in the rectangle `[x0,x1) × [y0,y1)` with `gid`.
63    pub fn fill_rect(&mut self, x0: i32, y0: i32, x1: i32, y1: i32, gid: u32) {
64        for y in y0..y1 {
65            for x in x0..x1 {
66                self.set_tile(x, y, gid);
67            }
68        }
69    }
70    /// Ensure the chunk at `(cx, cy)` is allocated; no-op when already loaded.
71    pub fn load_chunk(&mut self, cx: i32, cy: i32) {
72        log_msg!(debug, CK02, "({}, {})", cx, cy);
73        let cs = self.chunk_size;
74        self.chunks
75            .entry((cx, cy))
76            .or_insert_with(|| vec![0u32; (cs * cs) as usize]);
77    }
78    /// Discard the chunk at `(cx, cy)` and free its memory.
79    pub fn unload_chunk(&mut self, cx: i32, cy: i32) {
80        log_msg!(debug, CK03, "({}, {})", cx, cy);
81        self.chunks.remove(&(cx, cy));
82    }
83    /// Return the coordinates of all currently loaded chunks.
84    pub fn get_loaded_chunks(&self) -> Vec<(i32, i32)> {
85        self.chunks.keys().copied().collect()
86    }
87
88    /// Return the number of currently loaded chunks.
89    pub fn get_loaded_chunk_count(&self) -> usize {
90        self.chunks.len()
91    }
92
93    /// Return `true` when the chunk at `(cx, cy)` is loaded.
94    pub fn is_chunk_loaded(&self, cx: i32, cy: i32) -> bool {
95        self.chunks.contains_key(&(cx, cy))
96    }
97
98    /// Convert tile coordinates `(x, y)` to their parent chunk coordinates.
99    pub fn tile_to_chunk(&self, x: i32, y: i32) -> (i32, i32) {
100        let cs = self.chunk_size as i32;
101        (x.div_euclid(cs), y.div_euclid(cs))
102    }
103    /// Return the inclusive tile range `(min_x, min_y, max_x, max_y)` covered by chunk `(cx, cy)`.
104    pub fn chunk_tile_range(&self, cx: i32, cy: i32) -> (i32, i32, i32, i32) {
105        let cs = self.chunk_size as i32;
106        (cx * cs, cy * cs, cx * cs + cs, cy * cs + cs)
107    }
108    /// Return all chunk coordinates overlapping a view rectangle `(vx,vy,vw,vh)` with tile size `(tw,th)`.
109    pub fn get_chunks_in_view(
110        &self,
111        vx: f32,
112        vy: f32,
113        vw: f32,
114        vh: f32,
115        tw: f32,
116        th: f32,
117    ) -> Vec<(i32, i32)> {
118        let cs = self.chunk_size as f32;
119        let cpw = cs * tw;
120        let cph = cs * th;
121        let cx_min = (vx / cpw).floor() as i32;
122        let cy_min = (vy / cph).floor() as i32;
123        let cx_max = ((vx + vw) / cpw).ceil() as i32;
124        let cy_max = ((vy + vh) / cph).ceil() as i32;
125        let mut result = Vec::new();
126        for cy in cy_min..=cy_max {
127            for cx in cx_min..=cx_max {
128                result.push((cx, cy));
129            }
130        }
131        result
132    }
133    /// Return the world-space `Rect` occupied by chunk `(cx, cy)` given tile dimensions `(tw, th)`.
134    pub fn chunk_world_rect(&self, cx: i32, cy: i32, tw: f32, th: f32) -> Rect {
135        let cs = self.chunk_size as f32;
136        Rect::new(cx as f32 * cs * tw, cy as f32 * cs * th, cs * tw, cs * th)
137    }
138    /// Return the tile slice of chunk `(cx, cy)`, or `None` when the chunk is not loaded.
139    pub fn iter_chunk(&self, cx: i32, cy: i32) -> Option<&[u32]> {
140        self.chunks.get(&(cx, cy)).map(|v| v.as_slice())
141    }
142
143    /// Decompose world tile `(x, y)` into chunk coordinates `(cx, cy)` and local tile offsets `(lx, ly)`.
144    fn decompose(&self, x: i32, y: i32) -> (i32, i32, u32, u32) {
145        let cs = self.chunk_size as i32;
146        let cx = x.div_euclid(cs);
147        let cy = y.div_euclid(cs);
148        let lx = x.rem_euclid(cs) as u32;
149        let ly = y.rem_euclid(cs) as u32;
150        (cx, cy, lx, ly)
151    }
152}
