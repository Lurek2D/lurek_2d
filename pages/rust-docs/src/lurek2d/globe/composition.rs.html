1//! Compose multiple globe views into a single frame via split viewports.
2//!
3//! - Emit render commands for each named globe with per-entry screen center overrides.
4//! - Iterate the registry, clone camera state, and collect draw output into one batch.
5
6use crate::globe::draw::emit_globe_frame;
7use crate::globe::registry::GlobeRegistry;
8use crate::render::renderer::RenderCommand;
9use crate::runtime::resource_keys::FontKey;
10/// Screen center override for one split viewport.
11#[derive(Debug, Clone, Copy)]
12pub struct SplitViewport {
13    /// Screen center x coordinate for the viewport.
14    pub cx: f32,
15    /// Screen center y coordinate for the viewport.
16    pub cy: f32,
17}
18/// Emit render commands for several globes with per-entry viewport centers.
19pub fn emit_split_frame(
20    reg: &GlobeRegistry,
21    entries: &[(&str, SplitViewport)],
22    font: Option<FontKey>,
23) -> Vec<RenderCommand> {
24    let mut out = Vec::new();
25    for (name, vp) in entries {
26        if let Some(globe) = reg.get(name) {
27            let mut clone = globe.camera.clone();
28            clone.screen_cx = vp.cx;
29            clone.screen_cy = vp.cy;
30            let mut g = emit_globe_frame(
31                &globe.spec,
32                &clone,
33                &globe.graph,
34                &globe.fog,
35                &globe.markers,
36                &globe.labels,
37                &globe.layers,
38                &globe.heat_layers,
39                &globe.arcs,
40                globe.active_viewer.as_deref(),
41                font,
42                globe.sim_time_sec,
43            );
44            out.append(&mut g);
45        }
46    }
47    out
48}
