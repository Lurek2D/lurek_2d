1//! Id-keyed label storage for globe map annotations.
2//!
3//! - Insert, remove, move, and toggle visibility of positioned text labels.
4//! - LOD-aware iteration filters labels by minimum detail tier.
5
6use crate::globe::types::{Label, LabelStyle};
7use std::collections::HashMap;
8/// Label collection keyed by stable id.
9#[derive(Debug, Clone, Default)]
10pub struct LabelStore {
11    /// Stored labels by id.
12    labels: HashMap<u32, Label>,
13    /// Next label id to assign.
14    next_id: u32,
15}
16impl LabelStore {
17    /// Create an empty label store. This function is part of the public API.
18    pub fn new() -> Self {
19        Self::default()
20    }
21    /// Insert a label and return its assigned id.
22    pub fn add(
23        &mut self,
24        label_type: impl Into<String>,
25        lat_deg: f32,
26        lon_deg: f32,
27        text: impl Into<String>,
28        style: LabelStyle,
29        min_lod: u8,
30    ) -> u32 {
31        let id = self.next_id;
32        self.next_id += 1;
33        self.labels.insert(
34            id,
35            Label {
36                id,
37                label_type: label_type.into(),
38                lat_deg,
39                lon_deg,
40                text: text.into(),
41                visible: true,
42                style,
43                min_lod,
44            },
45        );
46        id
47    }
48    /// Remove a label by id and return it when found.
49    pub fn remove(&mut self, id: u32) -> Option<Label> {
50        self.labels.remove(&id)
51    }
52    /// Return a shared label reference when the id exists.
53    pub fn get(&self, id: u32) -> Option<&Label> {
54        self.labels.get(&id)
55    }
56    /// Return a mutable label reference when the id exists.
57    pub fn get_mut(&mut self, id: u32) -> Option<&mut Label> {
58        self.labels.get_mut(&id)
59    }
60    /// Set label visibility and return true when the id exists.
61    pub fn set_visible(&mut self, id: u32, visible: bool) -> bool {
62        if let Some(l) = self.labels.get_mut(&id) {
63            l.visible = visible;
64            true
65        } else {
66            false
67        }
68    }
69    /// Replace label text and return true when the id exists.
70    pub fn set_text(&mut self, id: u32, text: impl Into<String>) -> bool {
71        if let Some(l) = self.labels.get_mut(&id) {
72            l.text = text.into();
73            true
74        } else {
75            false
76        }
77    }
78    /// Move a label to a new latitude and longitude and return true when it exists.
79    pub fn move_to(&mut self, id: u32, lat_deg: f32, lon_deg: f32) -> bool {
80        if let Some(l) = self.labels.get_mut(&id) {
81            l.lat_deg = lat_deg;
82            l.lon_deg = lon_deg;
83            true
84        } else {
85            false
86        }
87    }
88    /// Iterate over all stored labels.
89    pub fn iter(&self) -> impl Iterator<Item = &Label> {
90        self.labels.values()
91    }
92    /// Iterate over visible labels whose minimum LOD fits the supplied tier.
93    pub fn iter_visible(&self, lod_tier: u8) -> impl Iterator<Item = &Label> {
94        self.labels
95            .values()
96            .filter(move |l| l.visible && l.min_lod <= lod_tier)
97    }
98    /// Return the number of stored labels.
99    pub fn len(&self) -> usize {
100        self.labels.len()
101    }
102    /// Return true when no labels are stored.
103    pub fn is_empty(&self) -> bool {
104        self.labels.is_empty()
105    }
106}
