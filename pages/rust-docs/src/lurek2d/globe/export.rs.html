1//! Export globe region geometry to standard mesh formats.
2//!
3//! - Generate flat OBJ output with one named object per region polygon.
4//! - Vertex data uses (lon, lat) mapping onto a 2D plane at z=0.
5
6use crate::globe::registry::Globe;
7use std::fmt::Write;
8/// Export region polygons as a flat OBJ string with one object per region.
9pub fn export_regions_to_obj(globe: &Globe) -> String {
10    let mut out = String::new();
11    let mut vertex_base: u32 = 1;
12    for region in globe.graph.iter() {
13        let _ = writeln!(&mut out, "o region_{}", region.id);
14        for (lat, lon) in &region.vertices {
15            let _ = writeln!(&mut out, "v {} {} 0.0", lon, lat);
16        }
17        if region.vertices.len() >= 3 {
18            let mut face = String::from("f");
19            for i in 0..region.vertices.len() {
20                let _ = write!(&mut face, " {}", vertex_base + i as u32);
21            }
22            let _ = writeln!(&mut out, "{}", face);
23        }
24        vertex_base += region.vertices.len() as u32;
25    }
26    out
27}
28
29/// Backward compatibility alias.
30#[inline]
31pub fn export_provinces_to_obj(globe: &Globe) -> String {
32    export_regions_to_obj(globe)
33}
