1//! Snapshot serialization of globe state for cross-thread transfer.
2//!
3//! - Channel pair for sending and receiving globe snapshots.
4//! - Build and apply helpers to capture or restore globe state.
5
6use crate::globe::registry::Globe;
7use crate::globe::types::RegionId;
8use std::collections::HashMap;
9use std::sync::mpsc::{channel, Receiver, Sender};
10/// Serializable globe state used for snapshot transfer.
11#[derive(Debug, Clone)]
12pub struct GlobeSyncSnapshot {
13    /// Globe name included in the snapshot.
14    pub name: String,
15    /// Camera latitude, longitude, and zoom.
16    pub camera: (f32, f32, f32),
17    /// Globe rotation in degrees.
18    pub rotation_deg: f32,
19    /// Time of day captured in the snapshot.
20    pub time_of_day: f32,
21    /// Region owner strings keyed by region id.
22    pub owner_attrs: HashMap<RegionId, String>,
23}
24/// Channel pair used to send and receive globe snapshots.
25#[derive(Debug)]
26pub struct GlobeSyncChannel {
27    /// Sender side of the snapshot channel.
28    pub tx: Sender<GlobeSyncSnapshot>,
29    /// Receiver side of the snapshot channel.
30    pub rx: Receiver<GlobeSyncSnapshot>,
31}
32impl GlobeSyncChannel {
33    /// Create a new snapshot channel pair.
34    pub fn new() -> Self {
35        let (tx, rx) = channel();
36        Self { tx, rx }
37    }
38}
39/// Create a default snapshot channel pair.
40impl Default for GlobeSyncChannel {
41    /// Delegate to `GlobeSyncChannel::new`.
42    fn default() -> Self {
43        Self::new()
44    }
45}
46/// Build a snapshot from the current globe state.
47pub fn build_snapshot(globe: &Globe) -> GlobeSyncSnapshot {
48    let mut owner_attrs = HashMap::new();
49    for p in globe.graph.iter() {
50        if let Some(owner) = p.attrs.get("owner") {
51            owner_attrs.insert(p.id, owner.clone());
52        }
53    }
54    GlobeSyncSnapshot {
55        name: globe.name.clone(),
56        camera: (
57            globe.camera.lat_deg,
58            globe.camera.lon_deg,
59            globe.camera.zoom,
60        ),
61        rotation_deg: globe.spec.rotation_deg,
62        time_of_day: globe.spec.time_of_day,
63        owner_attrs,
64    }
65}
66/// Apply a snapshot to a mutable globe instance.
67pub fn apply_snapshot(globe: &mut Globe, snap: &GlobeSyncSnapshot) {
68    globe.camera.lat_deg = snap.camera.0;
69    globe.camera.lon_deg = snap.camera.1;
70    globe.camera.zoom = snap.camera.2;
71    globe.camera.clamp();
72    globe.spec.rotation_deg = snap.rotation_deg;
73    globe.spec.time_of_day = snap.time_of_day;
74    for (id, owner) in &snap.owner_attrs {
75        if let Some(p) = globe.get_province_mut(*id) {
76            p.attrs.insert("owner".to_string(), owner.clone());
77        }
78    }
79}
