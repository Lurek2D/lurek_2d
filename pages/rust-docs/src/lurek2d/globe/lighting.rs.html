1//! Globe day/night lighting: sun direction from rotation and time-of-day.
2//!
3//! - Per-region diffuse intensity with ambient floor.
4//! - Batch intensity computation for region centroid sequences.
5//! - Terminator-band alpha for smooth day/night transition rendering.
6
7use crate::globe::types::GlobeSpec;
8use super::sphere::{lat_lon_to_unit, rot_y};
9use crate::math::Vec3;
10/// Compute the sun direction in world space from globe rotation and time of day.
11pub fn sun_direction(spec: &GlobeSpec) -> Vec3 {
12    let sun_lon_deg = 180.0 - spec.time_of_day * 360.0;
13    let sun_lat_deg = spec.axial_tilt_deg * (spec.time_of_day * std::f32::consts::TAU).sin();
14    let base = lat_lon_to_unit(sun_lat_deg, sun_lon_deg);
15    let spin = rot_y(spec.rotation_deg);
16    let sun_world = spin.mul_vec(base);
17    let len = sun_world.length().max(1e-12);
18    Vec3::new(sun_world.x / len, sun_world.y / len, sun_world.z / len)
19}
20/// Compute region light intensity from the centroid and sun direction.
21pub fn province_intensity(
22    centroid_lat_deg: f32,
23    centroid_lon_deg: f32,
24    sun_dir: &Vec3,
25    ambient: f32,
26) -> f32 {
27    let normal = lat_lon_to_unit(centroid_lat_deg, centroid_lon_deg);
28    let dot = normal.x * sun_dir.x + normal.y * sun_dir.y + normal.z * sun_dir.z;
29    dot.max(ambient).min(1.0)
30}
31/// Compute region intensities for a sequence of centroid positions.
32#[allow(clippy::extra_unused_lifetimes)]
33pub fn compute_intensities<'a>(
34    centroids: impl Iterator<Item = (f32, f32)>,
35    sun_dir: &Vec3,
36    ambient: f32,
37) -> Vec<f32> {
38    centroids
39        .map(|(lat, lon)| province_intensity(lat, lon, sun_dir, ambient))
40        .collect()
41}
42/// Convert sun alignment into an alpha value around the terminator band.
43pub fn terminator_alpha(
44    centroid_lat_deg: f32,
45    centroid_lon_deg: f32,
46    sun_dir: &Vec3,
47    transition_deg: f32,
48) -> f32 {
49    let normal = lat_lon_to_unit(centroid_lat_deg, centroid_lon_deg);
50    let dot = normal.x * sun_dir.x + normal.y * sun_dir.y + normal.z * sun_dir.z;
51    let half = (transition_deg / 2.0).to_radians().cos();
52    let t = (dot + half) / (2.0 * half);
53    t.clamp(0.0, 1.0)
54}
