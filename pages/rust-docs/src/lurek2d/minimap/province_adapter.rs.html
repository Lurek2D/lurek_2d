1//! Bridge between `ProvinceRegistry` terrain/visibility data and the minimap grid.
2//!
3//! - Copy terrain types, fog levels, and political palette colours into a `Minimap`.
4//! - Clips to the smaller of the two grids so mismatched sizes never panic.
5
6use crate::minimap::minimap::Minimap;
7use crate::minimap::types::FogLevel;
8use crate::province::registry::ProvinceRegistry;
9use crate::province::types::ProvinceId;
10
11/// Copy terrain types from `registry` cells into `minimap`, clipped to the smaller of the two grids.
12pub fn apply_terrain(minimap: &mut Minimap, registry: &ProvinceRegistry) {
13    let w = minimap.grid_width().min(registry.width());
14    let h = minimap.grid_height().min(registry.height());
15    for y in 0..h {
16        for x in 0..w {
17            let pid = registry.get_at(x, y);
18            if pid == 0 {
19                continue;
20            }
21            if let Some(snap) = registry.get_province(ProvinceId(pid)) {
22                minimap.set_terrain(x, y, snap.style.terrain_type);
23            }
24        }
25    }
26}
27
28/// Translate province `visibility_state` bytes into `FogLevel` values on `minimap`.
29pub fn apply_visibility(minimap: &mut Minimap, registry: &ProvinceRegistry) {
30    let w = minimap.grid_width().min(registry.width());
31    let h = minimap.grid_height().min(registry.height());
32    for y in 0..h {
33        for x in 0..w {
34            let pid = registry.get_at(x, y);
35            if pid == 0 {
36                continue;
37            }
38            if let Some(snap) = registry.get_province(ProvinceId(pid)) {
39                let level = if snap.style.visibility_state == 0 {
40                    FogLevel::Hidden
41                } else if snap.style.visibility_state < 128 {
42                    FogLevel::Explored
43                } else {
44                    FogLevel::Visible
45                };
46                minimap.set_fog_level(x, y, level);
47            }
48        }
49    }
50}

52/// Register each province's `political_color` as the terrain colour in `minimap`.
53pub fn apply_terrain_palette(minimap: &mut Minimap, registry: &ProvinceRegistry) {
54    for id in registry.province_ids() {
55        if let Some(snap) = registry.get_province(id) {
56            minimap.set_terrain_color(snap.style.terrain_type, snap.style.political_color);
57        }
58    }
59}
