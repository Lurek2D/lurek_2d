1//! Convert minimap state into an ordered list of `RenderCommand` values for the renderer.
2//!
3//! - Draw the background, terrain grid cells with zoom and center offset, and fog-of-war tinting.
4//! - Render overlay shapes (lines, rectangles), multi-segment paths, and the viewport indicator.
5//! - Draw animated pings with fade, map objects with optional icons, and markers with crosshairs.
6//! - All coordinates are projected from grid-space to screen-space via the minimap's transform.
7
8use super::minimap::Minimap;
9use super::types::{FogLevel, OverlayShape};
10use crate::render::renderer::{DrawMode, RenderCommand};
11
12/// Provide `RenderCommand` generation for `Minimap`.
13impl Minimap {
14    /// Build the full ordered `RenderCommand` list for this minimap at screen origin `(screen_x, screen_y)`.
15    pub fn generate_render_commands(&self, screen_x: f32, screen_y: f32) -> Vec<RenderCommand> {
16        let mut cmds = Vec::new();
17        let owner_colors = self.owner_colors_by_cell();
18        let dw = self.display_width() as f32;
19        let dh = self.display_height() as f32;
20        if dw <= 0.0 || dh <= 0.0 {
21            return cmds;
22        }
23        cmds.push(RenderCommand::SetColor(0.0, 0.0, 0.0, 0.85));
24        cmds.push(RenderCommand::Rectangle {
25            mode: DrawMode::Fill,
26            x: screen_x,
27            y: screen_y,
28            w: dw,
29            h: dh,
30        });
31        let gw = self.grid_width() as f32;
32        let gh = self.grid_height() as f32;
33        let zoom = self.zoom();
34        let cx = self.center_x();
35        let cy = self.center_y();
36        let cells_vis_x = gw / zoom;
37        let cells_vis_y = gh / zoom;
38        let cell_px_w = dw / cells_vis_x;
39        let cell_px_h = dh / cells_vis_y;
40        let start_gx = ((cx - cells_vis_x / 2.0).floor() as i64).max(0) as u32;
41        let start_gy = ((cy - cells_vis_y / 2.0).floor() as i64).max(0) as u32;
42        let end_gx = ((cx + cells_vis_x / 2.0).ceil() as u32).min(self.grid_width());
43        let end_gy = ((cy + cells_vis_y / 2.0).ceil() as u32).min(self.grid_height());
44        let fog_enabled = self.fog_enabled();
45        let [fcr, fcg, fcb, fca] = self.fog_color();
46        for gy in start_gy..end_gy {
47            for gx in start_gx..end_gx {
48                let (sx, sy) = self.grid_to_screen(gx as f32, gy as f32, screen_x, screen_y);
49                let terrain = self.get_terrain(gx, gy);
50                let [r, g, b, a] = self.resolve_cell_color(gx, gy, terrain, &owner_colors);
51                let (cr, cg, cb, ca) = if fog_enabled {
52                    match self.get_fog_level(gx, gy) {
53                        FogLevel::Hidden => (fcr, fcg, fcb, fca),
54                        FogLevel::Explored => (r, g, b, a * 0.4),
55                        FogLevel::Visible => (r, g, b, a),
56                    }
57                } else {
58                    (r, g, b, a)
59                };
60                cmds.push(RenderCommand::SetColor(cr, cg, cb, ca));
61                cmds.push(RenderCommand::Rectangle {
62                    mode: DrawMode::Fill,
63                    x: sx,
64                    y: sy,
65                    w: cell_px_w,
66                    h: cell_px_h,
67                });
68            }
69        }
70        for shape in self.overlay_shapes() {
71            match shape {
72                OverlayShape::Line {
73                    x1,
74                    y1,
75                    x2,
76                    y2,
77                    color,
78                } => {
79                    let (sx1, sy1) = self.grid_to_screen(*x1, *y1, screen_x, screen_y);
80                    let (sx2, sy2) = self.grid_to_screen(*x2, *y2, screen_x, screen_y);
81                    cmds.push(RenderCommand::SetColor(
82                        color[0] as f32 / 255.0,
83                        color[1] as f32 / 255.0,
84                        color[2] as f32 / 255.0,
85                        color[3] as f32 / 255.0,
86                    ));
87                    cmds.push(RenderCommand::Line {
88                        x1: sx1,
89                        y1: sy1,
90                        x2: sx2,
91                        y2: sy2,
92                    });
93                }
94                OverlayShape::Rect { x, y, w, h, color } => {
95                    let (sx, sy) = self.grid_to_screen(*x, *y, screen_x, screen_y);
96                    let (ex, ey) = self.grid_to_screen(*x + *w, *y + *h, screen_x, screen_y);
97                    cmds.push(RenderCommand::SetColor(
98                        color[0] as f32 / 255.0,
99                        color[1] as f32 / 255.0,
100                        color[2] as f32 / 255.0,
101                        color[3] as f32 / 255.0,
102                    ));
103                    cmds.push(RenderCommand::Rectangle {
104                        mode: DrawMode::Line,
105                        x: sx,
106                        y: sy,
107                        w: (ex - sx).abs(),
108                        h: (ey - sy).abs(),
109                    });
110                }
111            }
112        }
113        for path in self.paths() {
114            if path.points.len() < 2 {
115                continue;
116            }
117            cmds.push(RenderCommand::SetColor(
118                path.color[0] as f32 / 255.0,
119                path.color[1] as f32 / 255.0,
120                path.color[2] as f32 / 255.0,
121                path.color[3] as f32 / 255.0,
122            ));
123            for window in path.points.windows(2) {
124                let (sx1, sy1) = self.grid_to_screen(window[0].0, window[0].1, screen_x, screen_y);
125                let (sx2, sy2) = self.grid_to_screen(window[1].0, window[1].1, screen_x, screen_y);
126                cmds.push(RenderCommand::Line {
127                    x1: sx1,
128                    y1: sy1,
129                    x2: sx2,
130                    y2: sy2,
131                });
132            }
133        }
134        if self.viewport_visible() {
135            if let Some((vx, vy, vw, vh)) = self.viewport_rect() {
136                let [vr, vg, vb, va] = self.viewport_color();
137                let (sx, sy) = self.grid_to_screen(vx, vy, screen_x, screen_y);
138                let (ex, ey) = self.grid_to_screen(vx + vw, vy + vh, screen_x, screen_y);
139                cmds.push(RenderCommand::SetColor(vr, vg, vb, va));
140                cmds.push(RenderCommand::Rectangle {
141                    mode: DrawMode::Line,
142                    x: sx,
143                    y: sy,
144                    w: (ex - sx).abs(),
145                    h: (ey - sy).abs(),
146                });
147            }
148        }
149        let ping_radius = (cell_px_w * 1.5).max(4.0);
150        for ping in self.pings() {
151            let [pr, pg, pb, pa] = ping.color;
152            let fade = if ping.duration > 0.0 {
153                ping.remaining / ping.duration
154            } else {
155                1.0
156            };
157            let (sx, sy) = self.grid_to_screen(ping.x, ping.y, screen_x, screen_y);
158            cmds.push(RenderCommand::SetColor(pr, pg, pb, pa * fade));
159            cmds.push(RenderCommand::Circle {
160                mode: DrawMode::Line,
161                x: sx,
162                y: sy,
163                r: ping_radius,
164            });
165        }
166        for object in self.objects_iter() {
167            let Some(object_type) = self.object_type(object.type_index) else {
168                continue;
169            };
170            if !object_type.visible {
171                continue;
172            }
173            let (sx, sy) = self.grid_to_screen(object.x, object.y, screen_x, screen_y);
174            if let Some(icon) = self.object_type_icon(object.type_index) {
175                cmds.push(RenderCommand::DrawImageEx {
176                    texture_key: icon.texture_key,
177                    x: sx,
178                    y: sy,
179                    rotation: 0.0,
180                    sx: icon.display_width / icon.texture_width,
181                    sy: icon.display_height / icon.texture_height,
182                    ox: icon.texture_width * 0.5,
183                    oy: icon.texture_height * 0.5,
184                    effect: None,
185                });
186                continue;
187            }
188            cmds.push(RenderCommand::SetColor(
189                object_type.color[0],
190                object_type.color[1],
191                object_type.color[2],
192                object_type.color[3],
193            ));
194            cmds.push(RenderCommand::Circle {
195                mode: DrawMode::Fill,
196                x: sx,
197                y: sy,
198                r: (cell_px_w.min(cell_px_h) * 0.35).max(3.0),
199            });
200        }
201        for (marker_id, marker) in self.markers_with_ids() {
202            let (sx, sy) = self.grid_to_screen(marker.x, marker.y, screen_x, screen_y);
203            if let Some(icon) = self.marker_icon(*marker_id) {
204                cmds.push(RenderCommand::DrawImageEx {
205                    texture_key: icon.texture_key,
206                    x: sx,
207                    y: sy,
208                    rotation: 0.0,
209                    sx: icon.display_width / icon.texture_width,
210                    sy: icon.display_height / icon.texture_height,
211                    ox: icon.texture_width * 0.5,
212                    oy: icon.texture_height * 0.5,
213                    effect: None,
214                });
215                continue;
216            }
217            cmds.push(RenderCommand::SetColor(
218                marker.color[0],
219                marker.color[1],
220                marker.color[2],
221                marker.color[3],
222            ));
223            cmds.push(RenderCommand::Circle {
224                mode: DrawMode::Fill,
225                x: sx,
226                y: sy,
227                r: (cell_px_w.min(cell_px_h) * 0.3).max(3.0),
228            });
229            cmds.push(RenderCommand::Line {
230                x1: sx - 4.0,
231                y1: sy,
232                x2: sx + 4.0,
233                y2: sy,
234            });
235            cmds.push(RenderCommand::Line {
236                x1: sx,
237                y1: sy - 4.0,
238                x2: sx,
239                y2: sy + 4.0,
240            });
241        }
242        cmds
243    }
244}
