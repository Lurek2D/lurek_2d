1//! Shared data types for the minimap subsystem: enums, structs, and overlay shapes.
2//!
3//! - `ColorMode` selects between terrain-coloured and political-coloured cell rendering.
4//! - `FogLevel` encodes per-cell fog-of-war visibility as a three-state enum.
5//! - Object, ping, and marker structs hold live map overlays with position, colour, and animation.
6//! - `OverlayShape` and `OverlayPath` describe vector geometry drawn over the terrain grid.
7//! - `LayerData` stores raw cell bytes for named minimap layers.
8
9/// Whether minimap cells are coloured by terrain type or by political owner.
10#[derive(Debug, Clone, Copy, PartialEq)]
11pub enum ColorMode {
12    /// Colour each cell by its terrain type's registered colour.
13    Terrain,
14    /// Colour each cell by the owning player's registered colour.
15    Political,
16}
17
18impl ColorMode {
19    /// Parse `"terrain"` or `"political"` to a `ColorMode`; returns `None` on unknown strings.
20    pub fn parse_mode(s: &str) -> Option<Self> {
21        match s {
22            "terrain" => Some(ColorMode::Terrain),
23            "political" => Some(ColorMode::Political),
24            _ => None,
25        }
26    }
27
28    /// Return the canonical string name for this colour mode.
29    pub fn as_str(self) -> &'static str {
30        match self {
31            ColorMode::Terrain => "terrain",
32            ColorMode::Political => "political",
33        }
34    }
35}

37/// Fog-of-war visibility level for a single minimap cell.
38#[derive(Debug, Clone, Copy, PartialEq, Eq)]
39#[repr(u8)]
40pub enum FogLevel {
41    /// Cell has never been seen; rendered with full fog tint.
42    Hidden = 0,
43    /// Cell was seen in a past turn; rendered at reduced brightness.
44    Explored = 1,
45    /// Cell is currently visible; rendered at full brightness.
46    Visible = 2,
47}
48
49impl FogLevel {
50    /// Convert a raw `u8` byte to a `FogLevel`; values >= 2 map to `Visible`.
51    pub fn from_u8(val: u8) -> Self {
52        match val {
53            0 => FogLevel::Hidden,
54            1 => FogLevel::Explored,
55            _ => FogLevel::Visible,
56        }
57    }
58}
