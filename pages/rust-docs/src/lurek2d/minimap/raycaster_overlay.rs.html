1//! Raycaster-specific minimap overlay rendering.
2//!
3//! - Tile-based minimap window construction with per-tile lighting and line-of-sight checks.
4//! - Bresenham grid traversal for fast obstruction testing between player and map cells.
5//! - FOV ray fan that reveals all traversed cells within a max distance and step size.
6//! - Pixel-grid minimap extraction producing raw RGBA buffers with wall/floor coloring.
7//! - Player arrow rendering (filled circle plus direction line) composited onto the minimap.
8
9use crate::raycaster::dda::Raycaster2D;
10use crate::raycaster::lighting::{compute_lighting, PointLight};
11use std::collections::HashSet;
12/// A tile sample used to build a minimap view window centered on the player.
13#[derive(Debug, Clone)]
14pub struct MinimapTileSample {
15    /// Grid X coordinate of this tile.
16    pub x: u32,
17    /// Grid Y coordinate of this tile.
18    pub y: u32,
19    /// True when this tile is a solid wall.
20    pub blocked: bool,
21    /// True when this tile has direct line-of-sight from the player.
22    pub visible: bool,
23    /// Accumulated RGB light at this tile's center.
24    pub light: [f32; 3],
25    /// Luminance (average of `light` channels) in 0.0..1.0.
26    pub luma: f32,
27}
28/// Return true when the Bresenham grid path from `(x0,y0)` to `(x1,y1)` is unobstructed.
29fn tile_line_of_sight(raycaster: &Raycaster2D, x0: i32, y0: i32, x1: i32, y1: i32) -> bool {
30    let mut x = x0;
31    let mut y = y0;
32    let dx = (x1 - x0).abs();
33    let dy = (y1 - y0).abs();
34    let sx = if x0 < x1 { 1 } else { -1 };
35    let sy = if y0 < y1 { 1 } else { -1 };
36    let mut err = dx - dy;
37    while !(x == x1 && y == y1) {
38        let e2 = 2 * err;
39        if e2 > -dy {
40            err -= dy;
41            x += sx;
42        }
43        if e2 < dx {
44            err += dx;
45            y += sy;
46        }
47        if (x != x1 || y != y1) && (x < 0 || y < 0 || raycaster.get_cell(x as u32, y as u32) > 0) {
48            return false;
49        }
50    }
51    true
52}
53/// Compute accumulated RGB light for the center of tile `(x, y)` using `compute_lighting`.
54pub fn compute_tile_light(
55    raycaster: &Raycaster2D,
56    x: u32,
57    y: u32,
58    ambient: f32,
59    lights: &[PointLight],
60) -> [f32; 3] {
61    if x >= raycaster.width() || y >= raycaster.height() {
62        let a = ambient.clamp(0.0, 1.0);
63        return [a, a, a];
64    }
65    let wx = x as f32 + 0.5;
66    let wy = y as f32 + 0.5;
67    let wall_at = |cx: i32, cy: i32| -> bool {
68        cx < 0 || cy < 0 || raycaster.get_cell(cx as u32, cy as u32) > 0
69    };
70    compute_lighting(wx, wy, ambient, lights, &wall_at)
71}
72/// Return all `MinimapTileSample`s in a `radius`-tile square window centered on `(center_x, center_y)`.
73pub fn build_minimap_tile_window(
74    raycaster: &Raycaster2D,
75    center_x: f32,
76    center_y: f32,
77    radius: u32,
78    ambient: f32,
79    lights: &[PointLight],
80) -> Vec<MinimapTileSample> {
81    let cx = center_x.floor() as i32;
82    let cy = center_y.floor() as i32;
83    let r = radius as i32;
84    let mut out = Vec::new();
85    for gy in (cy - r)..=(cy + r) {
86        for gx in (cx - r)..=(cx + r) {
87            if gx < 0 || gy < 0 {
88                continue;
89            }
90            let ux = gx as u32;
91            let uy = gy as u32;
92            if ux >= raycaster.width() || uy >= raycaster.height() {
93                continue;
94            }
95            let blocked = raycaster.get_cell(ux, uy) > 0;
96            let visible = tile_line_of_sight(raycaster, cx, cy, gx, gy);
97            let light = compute_tile_light(raycaster, ux, uy, ambient, lights);
98            let luma = ((light[0] + light[1] + light[2]) / 3.0).clamp(0.0, 1.0);
99            out.push(MinimapTileSample {
100                x: ux,
101                y: uy,
102                blocked,
103                visible,
104                light,
105                luma,
106            });
107        }
108    }
109    out
110}
111/// Cast a FOV fan of `count` rays from `(ox, oy)` and collect all traversed tile coordinates.
112#[allow(clippy::too_many_arguments)]
113pub fn reveal_cells_from_rays(
114    raycaster: &Raycaster2D,
115    ox: f32,
116    oy: f32,
117    angle: f32,
118    fov: f32,
119    count: u32,
120    max_dist: f32,
121    step: f32,
122) -> Vec<(u32, u32)> {
123    let step = step.max(0.05);
124    let mut visited: HashSet<(u32, u32)> = HashSet::new();
125    let mut cells = Vec::new();
126    let add_cell =
127        |visited: &mut HashSet<(u32, u32)>, cells: &mut Vec<(u32, u32)>, x: f32, y: f32| {
128            if x.is_finite() && y.is_finite() && x >= 0.0 && y >= 0.0 {
129                let gx = x.floor() as u32;
130                let gy = y.floor() as u32;
131                if gx < raycaster.width() && gy < raycaster.height() && visited.insert((gx, gy)) {
132                    cells.push((gx, gy));
133                }
134            }
135        };
136    add_cell(&mut visited, &mut cells, ox, oy);
137    let hits = raycaster.cast_rays(ox, oy, angle, fov, count, max_dist);
138    for hit in hits {
139        let hx = if hit.hit {
140            hit.hit_x
141        } else {
142            ox + angle.cos() * max_dist
143        };
144        let hy = if hit.hit {
145            hit.hit_y
146        } else {
147            oy + angle.sin() * max_dist
148        };
149        let dx = hx - ox;
150        let dy = hy - oy;
151        let dist = (dx * dx + dy * dy).sqrt();
152        let steps = (dist / step).max(1.0).floor() as u32;
153        for i in 0..=steps {
154            let t = i as f32 / steps as f32;
155            add_cell(&mut visited, &mut cells, ox + dx * t, oy + dy * t);
156        }
157    }
158    cells
159}
160/// Render a `view_radius`-tile minimap pixel grid centered on the player; return `(pixels, width, height)`.
161#[allow(clippy::too_many_arguments)]
162pub fn extract_minimap(
163    raycaster: &Raycaster2D,
164    player_x: f32,
165    player_y: f32,
166    player_angle: f32,
167    view_radius: u32,
168    cell_size: u32,
169    wall_color: [u8; 4],
170    floor_color: [u8; 4],
171    player_color: [u8; 4],
172) -> (Vec<u8>, u32, u32) {
173    let diameter = view_radius * 2 + 1;
174    let pixel_w = diameter * cell_size;
175    let pixel_h = diameter * cell_size;
176    let mut pixels = vec![0u8; (pixel_w * pixel_h * 4) as usize];
177    let player_cell_x = player_x.floor() as i32;
178    let player_cell_y = player_y.floor() as i32;
179    for vy in 0..diameter {
180        for vx in 0..diameter {
181            let cell_x = player_cell_x - view_radius as i32 + vx as i32;
182            let cell_y = player_cell_y - view_radius as i32 + vy as i32;
183            let is_wall = if cell_x >= 0 && cell_y >= 0 {
184                raycaster.get_cell(cell_x as u32, cell_y as u32) > 0
185            } else {
186                false
187            };
188            let color = if is_wall { wall_color } else { floor_color };
189            for py in 0..cell_size {
190                for px in 0..cell_size {
191                    let img_x = vx * cell_size + px;
192                    let img_y = vy * cell_size + py;
193                    let idx = ((img_y * pixel_w + img_x) * 4) as usize;
194                    if idx + 3 < pixels.len() {
195                        pixels[idx] = color[0];
196                        pixels[idx + 1] = color[1];
197                        pixels[idx + 2] = color[2];
198                        pixels[idx + 3] = color[3];
199                    }
200                }
201            }
202        }
203    }
204    let center_px = view_radius * cell_size + cell_size / 2;
205    let center_py = view_radius * cell_size + cell_size / 2;
206    draw_player_arrow(
207        &mut pixels,
208        pixel_w,
209        center_px,
210        center_py,
211        player_angle,
212        cell_size.max(3),
213        player_color,
214    );
215    (pixels, pixel_w, pixel_h)
216}
217/// Draw a filled circle with a forward-direction line at `(center_x, center_y)` into a raw RGBA pixel slice.
218#[allow(clippy::too_many_arguments)]
219pub fn draw_player_arrow(
220    pixels: &mut [u8],
221    img_width: u32,
222    center_x: u32,
223    center_y: u32,
224    angle: f32,
225    size: u32,
226    color: [u8; 4],
227) {
228    let half = size as f32 / 2.0;
229    let radius = (half * 0.6).max(1.0);
230    let r2 = radius * radius;
231    for dy in -(radius as i32)..=(radius as i32) {
232        for dx in -(radius as i32)..=(radius as i32) {
233            if (dx * dx + dy * dy) as f32 <= r2 {
234                let px = center_x as i32 + dx;
235                let py = center_y as i32 + dy;
236                if px >= 0 && py >= 0 && (px as u32) < img_width {
237                    let idx = ((py as u32 * img_width + px as u32) * 4) as usize;
238                    if idx + 3 < pixels.len() {
239                        pixels[idx] = color[0];
240                        pixels[idx + 1] = color[1];
241                        pixels[idx + 2] = color[2];
242                        pixels[idx + 3] = color[3];
243                    }
244                }
245            }
246        }
247    }
248    let line_len = half;
249    let tip_x = center_x as f32 + angle.cos() * line_len;
250    let tip_y = center_y as f32 + angle.sin() * line_len;
251    let steps = (line_len * 2.0) as i32;
252    for i in 0..=steps {
253        let t = i as f32 / steps.max(1) as f32;
254        let lx = center_x as f32 + (tip_x - center_x as f32) * t;
255        let ly = center_y as f32 + (tip_y - center_y as f32) * t;
256        let px = lx as u32;
257        let py = ly as u32;
258        if px < img_width {
259            let idx = ((py * img_width + px) * 4) as usize;
260            if idx + 3 < pixels.len() {
261                pixels[idx] = color[0];
262                pixels[idx + 1] = color[1];
263                pixels[idx + 2] = color[2];
264                pixels[idx + 3] = color[3];
265            }
266        }
267    }
268}
