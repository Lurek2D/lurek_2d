1//! Priority-ordered demand matching against available supply nodes.
2//!
3//! - Pathfinding-based item routing from supplier to consumer.
4//! - Event emission on supply depletion and demand fulfillment.
5
6use super::core::Graph;
7use super::item::ItemPosition;
8use super::simulation::GraphEvent;
9impl Graph {
10    /// Process node demands and return the resulting graph events.
11    pub fn process_demand(&mut self) -> Vec<GraphEvent> {
12        let mut events = Vec::new();
13        let mut all_demands: Vec<(u64, String, i32, i32)> = Vec::new();
14        for node in self.nodes.values() {
15            for d in &node.demands {
16                all_demands.push((node.id.raw(), d.item_type.clone(), d.quantity, d.priority));
17            }
18        }
19        all_demands.sort_by(|a, b| b.3.cmp(&a.3));
20        for (demand_node_id, item_type, quantity, _priority) in all_demands {
21            let mut remaining = quantity;
22            let supply_nodes: Vec<u64> = self
23                .nodes
24                .values()
25                .filter(|n| {
26                    n.id.raw() != demand_node_id
27                        && n.supplies
28                            .iter()
29                            .any(|s| s.item_type == item_type && s.quantity != 0)
30                })
31                .map(|n| n.id.0)
32                .collect();
33            for supply_node_id in supply_nodes {
34                if remaining <= 0 {
35                    break;
36                }
37                let path = match self.find_path(supply_node_id, demand_node_id) {
38                    Some(p) => p,
39                    None => continue,
40                };
41                let available = self
42                    .nodes
43                    .get(&supply_node_id)
44                    .map(|n| n.get_available_supply(&item_type))
45                    .unwrap_or(0);
46                let to_send = if available < 0 {
47                    remaining
48                } else {
49                    remaining.min(available)
50                };
51                if to_send <= 0 {
52                    continue;
53                }
54                let mut sent = 0;
55                for _ in 0..to_send {
56                    let item_id = self.create_item(&item_type, -1.0);
57                    if let Some(item) = self.items.get_mut(&item_id) {
58                        item.position = ItemPosition::AtNode(supply_node_id);
59                    }
60                    if let Some(node) = self.nodes.get_mut(&supply_node_id) {
61                        node.items.push(item_id);
62                    }
63                    if let Some(&first_edge) = path.edges.first() {
64                        match self.send_item(item_id, first_edge) {
65                            Ok(true) => {
66                                sent += 1;
67                            }
68                            _ => {
69                                self.remove_item(item_id);
70                                break;
71                            }
72                        }
73                    } else {
74                        sent += 1;
75                    }
76                }
77                if sent > 0 {
78                    remaining -= sent;
79                    if let Some(node) = self.nodes.get_mut(&supply_node_id) {
80                        for s in &mut node.supplies {
81                            if s.item_type == item_type && s.quantity > 0 {
82                                s.quantity -= sent;
83                                if s.quantity <= 0 {
84                                    events.push(GraphEvent::SupplyDepleted {
85                                        node_id: supply_node_id,
86                                        item_type: item_type.clone(),
87                                    });
88                                }
89                                break;
90                            }
91                        }
92                    }
93                    events.push(GraphEvent::DemandFulfilled {
94                        demand_node: demand_node_id,
95                        supply_node: supply_node_id,
96                        item_type: item_type.clone(),
97                        count: sent as u32,
98                    });
99                }
100            }
101        }
102        events
103    }
104}
