1//! Graph container managing nodes, edges, and items with id-based lookup.
2//!
3//! - Adjacency indexes for fast outgoing and incoming edge queries.
4//! - Node CRUD with cascade removal of connected edges and displaced items.
5//! - Edge CRUD with transit capacity, cooldown, and type filtering.
6//! - Item lifecycle: creation, node placement (with overflow policy), transit, and removal.
7//! - Subgraph extraction preserving topology and item positions.
8//! - Aggregate stats computation across nodes and edges.
9//! - Direction-based edge queries (in, out, both).
10//! - Simple circular-layout image rendering for debug preview.
11//! - JSON-like serialize and deserialize for persistence.
12

21pub struct GraphStats {

25    pub edges: usize,

27    pub items: usize,

29    pub active_nodes: usize,
30    /// Number of active edges.
31    pub active_edges: usize,
32    /// Number of items currently in transit.
33    pub items_in_transit: usize,

36    /// Sum of node demand quantities.

43/// Main graph container with nodes, edges, items, and adjacency indexes.
44pub struct Graph {
45    /// Stored nodes by id.
46    pub nodes: HashMap<u64, Node>,
47    /// Stored edges by id.
48    pub edges: HashMap<u64, Edge>,
49    /// Stored items by id.
50    pub items: HashMap<u64, GraphItem>,
51    /// Outgoing edge ids keyed by source node id.
52    outgoing_index: HashMap<u64, Vec<u64>>,
53    /// Incoming edge ids keyed by destination node id.
54    incoming_index: HashMap<u64, Vec<u64>>,
55    /// Next node id to assign.
56    next_node_id: u64,
57    /// Next edge id to assign.
58    next_edge_id: u64,
59    /// Next item id to assign.
60    next_item_id: u64,
61}
62/// Create an empty graph with fresh id counters.
63impl Default for Graph {
64    fn default() -> Self {
65        Self::new()
66    }
67}
68/// Core operations for building, querying, and mutating the graph.
69impl Graph {
70    /// Create an empty graph with fresh id counters.
71    pub fn new() -> Self {
72        Self {
73            nodes: HashMap::new(),
74            edges: HashMap::new(),
75            items: HashMap::new(),
76            outgoing_index: HashMap::new(),
77            incoming_index: HashMap::new(),
78            next_node_id: 1,
79            next_edge_id: 1,
80            next_item_id: 1,
81        }
82    }
83    /// Add an edge id to the outgoing and incoming indexes.
84    fn index_edge(&mut self, edge_id: u64, from_node: u64, to_node: u64) {
85        self.outgoing_index
86            .entry(from_node)
87            .or_default()
88            .push(edge_id);
89        self.incoming_index
90            .entry(to_node)
91            .or_default()
92            .push(edge_id);
93    }
94    /// Remove an edge id from the outgoing and incoming indexes.
95    fn unindex_edge(&mut self, edge_id: u64, from_node: u64, to_node: u64) {
96        if let Some(ids) = self.outgoing_index.get_mut(&from_node) {
97            ids.retain(|&id| id != edge_id);
98        }
99        if let Some(ids) = self.incoming_index.get_mut(&to_node) {
100            ids.retain(|&id| id != edge_id);
101        }
102    }
103    /// Return the indexed outgoing edge ids for a node.
104    pub(crate) fn outgoing_edge_ids_slice(&self, node_id: u64) -> &[u64] {
105        self.outgoing_index
106            .get(&node_id)
107            .map(Vec::as_slice)
108            .unwrap_or(&[])
109    }
110    /// Return the indexed incoming edge ids for a node.
111    pub(crate) fn incoming_edge_ids_slice(&self, node_id: u64) -> &[u64] {
112        self.incoming_index
113            .get(&node_id)
114            .map(Vec::as_slice)
115            .unwrap_or(&[])
116    }
117    /// Add a node and return its assigned id.
118    pub fn add_node(&mut self, node_type: &str, capacity: i32) -> u64 {
119        let id = self.next_node_id;
120        self.next_node_id += 1;
121        log_msg!(debug, GC01, "id={} type={}", id, node_type);
122        self.nodes.insert(id, Node::new(id, node_type, capacity));
123        self.outgoing_index.entry(id).or_default();
124        self.incoming_index.entry(id).or_default();
125        id
126    }
127    /// Remove a node and all connected edges, returning true when it existed.
128    pub fn remove_node(&mut self, node_id: u64) -> bool {
129        if self.nodes.remove(&node_id).is_none() {
130            return false;
131        }
132        let edge_ids: Vec<u64> = self
133            .edges
134            .values()
135            .filter(|e| e.from_node == node_id || e.to_node == node_id)
136            .map(|e| e.id.raw())
137            .collect();
138        for eid in edge_ids {
139            self.remove_edge(eid);
140        }
141        for item in self.items.values_mut() {
142            if item.position == ItemPosition::AtNode(node_id) {
143                item.position = ItemPosition::Unplaced;
144            }
145        }
146        self.outgoing_index.remove(&node_id);
147        self.incoming_index.remove(&node_id);
148        log_msg!(debug, GC02, "{}", node_id);
149        true
150    }
151    /// Return true when the node id exists.
152    pub fn has_node(&self, node_id: u64) -> bool {
153        self.nodes.contains_key(&node_id)
154    }
155    /// Return all node ids in arbitrary order.
156    pub fn get_node_ids(&self) -> Vec<u64> {
157        self.nodes.keys().copied().collect()
158    }
159    /// Return the number of nodes. This function is part of the public API.
160    pub fn get_node_count(&self) -> usize {

162    }
163    /// Add an edge and return its assigned id or an error when either endpoint is missing.
164    pub fn add_edge(&mut self, from: u64, to: u64, edge_type: Option<&str>) -> Result<u64, String> {
165        if !self.nodes.contains_key(&from) {
166            return Err(format!("source node {from} does not exist"));
167        }
168        if !self.nodes.contains_key(&to) {
169            return Err(format!("destination node {to} does not exist"));
170        }
171        let id = self.next_edge_id;
172        self.next_edge_id += 1;
173        let e = Edge::new(id, from, to, edge_type.unwrap_or("default"));
174        self.edges.insert(id, e);
175        self.index_edge(id, from, to);
176        log_msg!(debug, GC03, "{} -> {} (id={})", from, to, id);
177        Ok(id)
178    }
179    /// Add an edge without validating that source/target nodes exist.
180    /// Use for hot paths where node existence is already guaranteed by construction.
181    pub fn add_edge_unchecked(&mut self, from: u64, to: u64, edge_type: Option<&str>) -> u64 {
182        let id = self.next_edge_id;
183        self.next_edge_id += 1;
184        let e = Edge::new(id, from, to, edge_type.unwrap_or("default"));
185        self.edges.insert(id, e);
186        self.index_edge(id, from, to);
187        id
188    }
189    /// Remove an edge and detach any items in transit, returning true when it existed.
190    pub fn remove_edge(&mut self, edge_id: u64) -> bool {
191        if let Some(edge) = self.edges.remove(&edge_id) {
192            self.unindex_edge(edge_id, edge.from_node, edge.to_node);
193            for &iid in &edge.items_in_transit {
194                if let Some(item) = self.items.get_mut(&iid) {
195                    item.position = ItemPosition::Unplaced;
196                }
197            }
198            log_msg!(debug, GC04, "{}", edge_id);
199            true
200        } else {
201            false
202        }
203    }

207    }

216    /// Return the first outgoing edge id that connects the supplied nodes.
217    pub fn get_edge_between(&self, from: u64, to: u64) -> Option<u64> {
218        self.outgoing_edge_ids_slice(from)
219            .iter()
220            .find_map(|edge_id| {
221                self.edges.get(edge_id).and_then(|edge| {
222                    if edge.to_node == to {
223                        Some(edge.id.raw())
224                    } else {
225                        None
226                    }
227                })
228            })
229    }
230    /// Build a new graph containing only the selected nodes and connected data.
231    pub fn subgraph(&self, node_ids: &[u64]) -> Self {
232        let requested: HashSet<u64> = node_ids
233            .iter()
234            .copied()
235            .filter(|id| self.nodes.contains_key(id))
236            .collect();
237        let mut sorted_nodes: Vec<u64> = requested.iter().copied().collect();
238        sorted_nodes.sort_unstable();
239        let mut sub = Graph::new();
240        let mut node_map: HashMap<u64, u64> = HashMap::new();
241        for old_id in sorted_nodes {
242            let old = &self.nodes[&old_id];
243            let new_id = sub.add_node(&old.node_type, old.capacity);
244            let new_node = sub.nodes.get_mut(&new_id).expect("new node must exist");
245            new_node.active = old.active;
246            new_node.overflow_policy = old.overflow_policy.clone();
247            new_node.flow_mode = old.flow_mode.clone();
248            new_node.push_rate = old.push_rate;
249            new_node.pull_rate = old.pull_rate;
250            new_node.push_filter = old.push_filter.clone();
251            new_node.pull_filter = old.pull_filter.clone();
252            new_node.process_time = old.process_time;
253            new_node.queue_enabled = old.queue_enabled;
254            new_node.queue_capacity = old.queue_capacity;
255            new_node.conversions = old.conversions.clone();
256            new_node.demands = old.demands.clone();
257            new_node.supplies = old.supplies.clone();
258            new_node.tags = old.tags.clone();
259            node_map.insert(old_id, new_id);
260        }
261        let mut edge_map: HashMap<u64, u64> = HashMap::new();
262        let mut edge_ids: Vec<u64> = self.edges.keys().copied().collect();
263        edge_ids.sort_unstable();
264        for old_edge_id in edge_ids {
265            let old_edge = &self.edges[&old_edge_id];
266            let Some(&new_from) = node_map.get(&old_edge.from_node) else {
267                continue;
268            };
269            let Some(&new_to) = node_map.get(&old_edge.to_node) else {
270                continue;
271            };
272            let new_edge_id = sub
273                .add_edge(new_from, new_to, Some(&old_edge.edge_type))
274                .expect("subgraph edge endpoints should be valid");
275            let new_edge = sub
276                .edges
277                .get_mut(&new_edge_id)
278                .expect("new edge must exist");
279            new_edge.capacity = old_edge.capacity;
280            new_edge.throughput = old_edge.throughput;
281            new_edge.travel_time = old_edge.travel_time;
282            new_edge.weight = old_edge.weight;
283            new_edge.speed_modifier = old_edge.speed_modifier;
284            new_edge.cooldown = old_edge.cooldown;
285            new_edge.cooldown_timer = old_edge.cooldown_timer;
286            new_edge.bidirectional = old_edge.bidirectional;
287            new_edge.active = old_edge.active;
288            new_edge.allowed_types = old_edge.allowed_types.clone();
289            edge_map.insert(old_edge_id, new_edge_id);
290        }
291        let mut item_map: HashMap<u64, u64> = HashMap::new();
292        let mut item_ids: Vec<u64> = self.items.keys().copied().collect();
293        item_ids.sort_unstable();
294        for old_item_id in item_ids {
295            let old_item = &self.items[&old_item_id];
296            let new_position = match old_item.position {
297                ItemPosition::AtNode(old_node_id) => {
298                    let Some(&new_node_id) = node_map.get(&old_node_id) else {
299                        continue;
300                    };
301                    ItemPosition::AtNode(new_node_id)
302                }
303                ItemPosition::InTransit {
304                    edge_id: old_edge_id,
305                    progress,
306                } => {
307                    let Some(&new_edge_id) = edge_map.get(&old_edge_id) else {
308                        continue;
309                    };
310                    ItemPosition::InTransit {
311                        edge_id: new_edge_id,
312                        progress,
313                    }
314                }
315                ItemPosition::Unplaced => ItemPosition::Unplaced,
316            };
317            let new_item_id = sub.create_item(&old_item.item_type, old_item.decay_time);
318            let new_item = sub
319                .items
320                .get_mut(&new_item_id)
321                .expect("new item must exist");
322            new_item.remaining_life = old_item.remaining_life;
323            new_item.alive = old_item.alive;
324            new_item.priority = old_item.priority;
325            new_item.position = new_position;
326            item_map.insert(old_item_id, new_item_id);
327        }
328        for (&old_node_id, &new_node_id) in &node_map {
329            if let Some(old_node) = self.nodes.get(&old_node_id) {
330                if let Some(new_node) = sub.nodes.get_mut(&new_node_id) {
331                    new_node.items = old_node
332                        .items
333                        .iter()
334                        .filter_map(|old_item_id| item_map.get(old_item_id).copied())
335                        .collect();
336                    new_node.queue = old_node
337                        .queue
338                        .iter()
339                        .filter_map(|old_item_id| item_map.get(old_item_id).copied())
340                        .collect();
341                }
342            }
343        }
344        for (&old_edge_id, &new_edge_id) in &edge_map {
345            if let Some(old_edge) = self.edges.get(&old_edge_id) {
346                if let Some(new_edge) = sub.edges.get_mut(&new_edge_id) {
347                    new_edge.items_in_transit = old_edge
348                        .items_in_transit
349                        .iter()
350                        .filter_map(|old_item_id| item_map.get(old_item_id).copied())
351                        .collect();
352                }
353            }
354        }
355        sub
356    }

365    /// Add an item to a node and return whether the placement succeeded.
366    pub fn add_item_to_node(&mut self, item_id: u64, node_id: u64) -> Result<bool, String> {
367        if !self.items.contains_key(&item_id) {
368            return Err(format!("item {item_id} does not exist"));
369        }
370        let node = self
371            .nodes
372            .get(&node_id)
373            .ok_or_else(|| format!("node {node_id} does not exist"))?;
374        if node.is_full() {
375            match node.overflow_policy {
376                OverflowPolicy::Reject => return Ok(false),
377                OverflowPolicy::Destroy => {
378                    if let Some(item) = self.items.get_mut(&item_id) {
379                        item.kill();
380                    }
381                    return Ok(false);
382                }
383                OverflowPolicy::Queue => {
384                    let node = self.nodes.get_mut(&node_id).unwrap();
385                    let queued = node.enqueue(item_id);
386                    if queued {
387                        if let Some(item) = self.items.get_mut(&item_id) {
388                            item.position = ItemPosition::AtNode(node_id);
389                        }
390                    }
391                    return Ok(queued);
392                }
393            }
394        }
395        let node = self.nodes.get_mut(&node_id).unwrap();
396        node.items.push(item_id);
397        if let Some(item) = self.items.get_mut(&item_id) {
398            item.position = ItemPosition::AtNode(node_id);
399        }
400        Ok(true)
401    }
402    /// Remove an item from the graph and all node or edge containers.
403    pub fn remove_item(&mut self, item_id: u64) -> bool {
404        if self.items.remove(&item_id).is_none() {
405            return false;
406        }
407        for node in self.nodes.values_mut() {
408            node.items.retain(|&id| id != item_id);
409            node.queue.retain(|&id| id != item_id);
410        }
411        for edge in self.edges.values_mut() {
412            edge.items_in_transit.retain(|&id| id != item_id);
413        }
414        true
415    }

428    /// Send an item onto an edge and return whether the transfer succeeded.
429    pub fn send_item(&mut self, item_id: u64, edge_id: u64) -> Result<bool, String> {
430        let item = self
431            .items
432            .get(&item_id)
433            .ok_or_else(|| format!("item {item_id} does not exist"))?;
434        let item_type = item.item_type.clone();
435        let edge = self
436            .edges
437            .get(&edge_id)
438            .ok_or_else(|| format!("edge {edge_id} does not exist"))?;
439        if !edge.active {
440            return Ok(false);
441        }
442        if edge.is_on_cooldown() {
443            return Ok(false);
444        }
445        if !edge.is_item_type_allowed(&item_type) {
446            return Ok(false);
447        }
448        if edge.is_transit_full() {
449            return Ok(false);
450        }
451        let from_node = edge.from_node;
452        if let Some(node) = self.nodes.get_mut(&from_node) {
453            node.items.retain(|&id| id != item_id);
454        }
455        let edge = self.edges.get_mut(&edge_id).unwrap();
456        edge.items_in_transit.push(item_id);
457        edge.cooldown_timer = edge.cooldown;
458        let item = self.items.get_mut(&item_id).unwrap();
459        item.position = ItemPosition::InTransit {
460            edge_id,
461            progress: 0.0,
462        };
463        Ok(true)
464    }
465    /// Return aggregate counts derived from the current graph state.
466    pub fn get_stats(&self) -> GraphStats {
467        let mut stats = GraphStats {
468            nodes: self.nodes.len(),
469            edges: self.edges.len(),
470            items: self.items.len(),
471            active_nodes: 0,
472            active_edges: 0,
473            items_in_transit: 0,
474            items_on_nodes: 0,
475            total_demand: 0,
476            total_supply: 0,
477            queued_items: 0,
478        };
479        for node in self.nodes.values() {
480            if node.active {
481                stats.active_nodes += 1;
482            }
483            stats.items_on_nodes += node.items.len();
484            stats.queued_items += node.queue.len();
485            for d in &node.demands {
486                stats.total_demand += d.quantity;
487            }
488            for s in &node.supplies {
489                stats.total_supply += s.quantity;
490            }
491        }
492        for edge in self.edges.values() {
493            if edge.active {
494                stats.active_edges += 1;
495            }
496            stats.items_in_transit += edge.items_in_transit.len();
497        }
498        stats
499    }

508    /// Return edge ids by requested direction or an error when the direction is invalid.
509    pub fn get_edges_by_direction(
510        &self,
511        node_id: u64,
512        direction: &str,
513    ) -> Result<Vec<u64>, String> {
514        if !self.nodes.contains_key(&node_id) {
515            return Err("node not found".into());
516        }
517        match direction {
518            "in" => Ok(self.get_incoming_edges(node_id)),
519            "out" => Ok(self.get_outgoing_edges(node_id)),
520            "both" => {
521                let mut combined = self.get_outgoing_edges(node_id);
522                combined.extend(self.get_incoming_edges(node_id));
523                combined.sort();
524                combined.dedup();
525                Ok(combined)
526            }
527            _ => Err(format!(
528                "invalid direction: '{}'. Use 'in', 'out', or 'both'",
529                direction
530            )),
531        }
532    }
533    /// Draw a simple circular graph preview into an image buffer.
534    pub fn draw_to_image(&self, width: u32, height: u32) -> crate::image::ImageData {
535        let mut img = crate::image::ImageData::new(width, height);
536        img.fill(25, 25, 35, 255);
537        let cx = width as f32 / 2.0;
538        let cy = height as f32 / 2.0;
539        let radius = (width.min(height) as f32) * 0.35;
540        let mut node_ids: Vec<u64> = self.nodes.keys().copied().collect();
541        node_ids.sort();
542        let n = node_ids.len();
543        if n == 0 {
544            return img;
545        }
546        let positions: Vec<(f32, f32)> = (0..n)
547            .map(|i| {
548                let angle =
549                    i as f32 * std::f32::consts::PI * 2.0 / n as f32 - std::f32::consts::FRAC_PI_2;
550                (cx + radius * angle.cos(), cy + radius * angle.sin())
551            })
552            .collect();
553        let id_to_idx: std::collections::HashMap<u64, usize> = node_ids
554            .iter()
555            .enumerate()
556            .map(|(i, &id)| (id, i))
557            .collect();
558        for edge in self.edges.values() {
559            if let (Some(&ai), Some(&bi)) =
560                (id_to_idx.get(&edge.from_node), id_to_idx.get(&edge.to_node))
561            {
562                let (ax, ay) = positions[ai];
563                let (bx, by) = positions[bi];
564                img.draw_line(
565                    ax as i32, ay as i32, bx as i32, by as i32, 80, 120, 160, 200,
566                );
567            }
568        }
569        for (i, &nid) in node_ids.iter().enumerate() {
570            let (px, py) = positions[i];
571            let (r, g, b) = if self.nodes[&nid].node_type == "city" {
572                (200u8, 80, 80)
573            } else {
574                (80, 200, 80)
575            };
576            img.draw_circle(px as i32, py as i32, 8, r, g, b, 255);
577        }
578        img
579    }
580    /// Serialize nodes and edges into a JSON-like value map.
581    pub fn serialize(&self) -> HashMap<String, serde_json::Value> {
582        use serde_json::{json, Value};

584            .nodes
585            .values()
586            .map(|n| json!({ "id": n.id.raw(), "node_type": n.node_type, "capacity": n.capacity }))
587            .collect();
588        nodes_arr.sort_by_key(|v| v["id"].as_u64().unwrap_or(0));
589        let mut edges_arr: Vec<Value> = self
590            .edges
591            .values()
592            .map(|e| {
593                json!({ "id": e.id.raw(), "from": e.from_node, "to": e.to_node,
594                    "weight": e.weight, "edge_type": e.edge_type,
595                    "bidirectional": e.bidirectional })
596            })
597            .collect();
598        edges_arr.sort_by_key(|v| v["id"].as_u64().unwrap_or(0));
599        let mut map = HashMap::new();
600        map.insert("nodes".to_string(), Value::Array(nodes_arr));
601        map.insert("edges".to_string(), Value::Array(edges_arr));
602        map
603    }
604    /// Deserialize a graph from the JSON-like value map or return a shape error.
605    pub fn deserialize(data: &HashMap<String, serde_json::Value>) -> Result<Self, String> {
606        let mut g = Self::new();
607        if let Some(nodes_val) = data.get("nodes") {
608            let arr = nodes_val.as_array().ok_or("nodes must be an array")?;
609            for n in arr {
610                let node_type = n["node_type"].as_str().unwrap_or("default");
611                let capacity = n["capacity"].as_i64().unwrap_or(-1) as i32;
612                g.add_node(node_type, capacity);
613            }
614        }
615        if let Some(edges_val) = data.get("edges") {
616            let arr = edges_val.as_array().ok_or("edges must be an array")?;
617            for e in arr {
618                let from = e["from"].as_u64().ok_or("edge missing 'from'")?;
619                let to = e["to"].as_u64().ok_or("edge missing 'to'")?;
620                let edge_type = e["edge_type"].as_str();
621                let weight = e["weight"].as_f64().unwrap_or(1.0);
622                let bidirectional = e["bidirectional"].as_bool().unwrap_or(false);
623                let eid = g
624                    .add_edge(from, to, edge_type)
625                    .map_err(|e| format!("add_edge error: {e}"))?;
626                if let Some(edge) = g.edges.get_mut(&eid) {
627                    edge.weight = weight;
628                    edge.bidirectional = bidirectional;
629                }
630            }
631        }
632        Ok(g)
633    }
