1//! Directed edge connecting two graph nodes with capacity, throughput, and cooldown constraints.
2//!
3//! - Type-based filtering restricts which items may transit an edge.
4//! - Supports bidirectional flag and per-edge speed/weight modifiers for pathfinding.
5
6use std::collections::HashSet;
7
8use crate::flownet::types::EdgeId;
9
10/// Directed connection between two graph nodes.
11pub struct Edge {
12    /// Stable edge identifier.
13    pub id: EdgeId,
14    /// Edge type name.
15    pub edge_type: String,
16    /// Source node id.
17    pub from_node: u64,
18    /// Destination node id.
19    pub to_node: u64,
20    /// Maximum items in transit, or negative for unlimited.
21    pub capacity: i32,
22    /// Maximum items that can move per update window.
23    pub throughput: f64,
24    /// Travel time in seconds.
25    pub travel_time: f64,
26    /// Pathfinding weight.
27    pub weight: f64,
28    /// Speed multiplier applied during transit.
29    pub speed_modifier: f64,
30    /// Cooldown duration between sends.
31    pub cooldown: f64,
32    /// Remaining cooldown time.
33    pub cooldown_timer: f64,
34    /// Flag that marks the edge as bidirectional.
35    pub bidirectional: bool,
36    /// Flag that enables the edge in simulation.
37    pub active: bool,
38    /// Allowed item types, or empty for no restriction.
39    pub allowed_types: HashSet<String>,
40    /// Item ids currently moving along the edge.
41    pub items_in_transit: Vec<u64>,
42}
43impl Edge {
44    /// Create an edge with default transit settings.
45    pub fn new(id: u64, from: u64, to: u64, edge_type: &str) -> Self {
46        Self {
47            id: EdgeId(id),
48            edge_type: edge_type.to_string(),
49            from_node: from,
50            to_node: to,
51            capacity: -1,
52            throughput: 1.0,
53            travel_time: 1.0,
54            weight: 1.0,
55            speed_modifier: 1.0,
56            cooldown: 0.0,
57            cooldown_timer: 0.0,
58            bidirectional: false,
59            active: true,
60            allowed_types: HashSet::new(),
61            items_in_transit: Vec::new(),
62        }
63    }
64    /// Return the edge type string. This function is part of the public API.
65    pub fn get_type(&self) -> &str {
66        &self.edge_type
67    }
68    /// Set the edge type string. This function is part of the public API.
69    pub fn set_type(&mut self, t: &str) {
70        self.edge_type = t.to_string();
71    }
72    /// Return true when the edge is still on cooldown.
73    pub fn is_on_cooldown(&self) -> bool {
74        self.cooldown_timer > 0.0
75    }
76    /// Return true when the item type is allowed by the edge filter.
77    pub fn is_item_type_allowed(&self, t: &str) -> bool {
78        self.allowed_types.is_empty() || self.allowed_types.contains(t)
79    }
80    /// Allow an item type on the edge.
81    pub fn add_allowed_type(&mut self, t: &str) {
82        self.allowed_types.insert(t.to_string());
83    }
84    /// Remove an allowed item type and return true when it existed.
85    pub fn remove_allowed_type(&mut self, t: &str) -> bool {
86        self.allowed_types.remove(t)
87    }
88    /// Remove all allowed item type filters.
89    pub fn clear_allowed_types(&mut self) {
90        self.allowed_types.clear();
91    }
92    /// Return true when the transit buffer is at or above capacity.
93    pub fn is_transit_full(&self) -> bool {
94        if self.capacity < 0 {
95            false
96        } else {
97            self.items_in_transit.len() >= self.capacity as usize
98        }
99    }
100}
