1//! Render the composited terminal cell grid as a list of `RenderCommand` draw calls.
2//!
3//! - Rasterise the composited grid into an `ImageData` thumbnail for previews and tests.
4//! - Both paths include terminal widgets and map foreground/background colours to output.
5
6use super::terminal_state::Terminal;
7use crate::image::ImageData;
8use crate::render::renderer::{DrawMode, RenderCommand};
9use crate::runtime::resource_keys::FontKey;
10
11/// Rendering helpers for the terminal cell grid.
12impl Terminal {
13    /// Build a `RenderCommand` list for the current cell grid using `font_key`, `char_w`/`char_h` cell dimensions, and `scale`.
14    pub fn generate_render_commands(
15        &self,
16        font_key: FontKey,
17        char_w: f32,
18        char_h: f32,
19        scale: f32,
20    ) -> Vec<RenderCommand> {
21        let (cols, rows) = self.get_dimensions();
22        let cells = self.render_cells();
23        let mut cmds = Vec::with_capacity(cols * rows * 2);
24        for row in 0..rows {
25            for col in 0..cols {
26                let cell = cells[row * cols + col];
27                let x = col as f32 * char_w;
28                let y = row as f32 * char_h;
29                let [br, bg_clr, bb, ba] = cell.bg;
30                if ba > 0.0 {
31                    cmds.push(RenderCommand::SetColor(br, bg_clr, bb, ba));
32                    cmds.push(RenderCommand::Rectangle {
33                        mode: DrawMode::Fill,
34                        x,
35                        y,
36                        w: char_w,
37                        h: char_h,
38                    });
39                }
40                let ch = char::from_u32(cell.ch).unwrap_or(' ');
41                if ch != ' ' {
42                    let [r, g, b, a] = cell.fg;
43                    cmds.push(RenderCommand::SetColor(r, g, b, a));
44                    cmds.push(RenderCommand::Print {
45                        font_key,
46                        text: ch.to_string(),
47                        x,
48                        y,
49                        scale,
50                    });
51                }
52            }
53        }
54        cmds
55    }
56
57    /// Rasterise the cell grid into a `width`×`height` `ImageData` thumbnail; non-space cells are drawn as solid colored rectangles.
58    pub fn draw_to_image(&self, width: u32, height: u32) -> ImageData {
59        let mut img = ImageData::new(width, height);
60        img.fill(18, 18, 28, 255);
61        let (cols, rows) = self.get_dimensions();
62        if cols == 0 || rows == 0 || width == 0 || height == 0 {
63            return img;
64        }
65        let cells = self.render_cells();
66        let cell_w = (width / cols as u32).max(1);
67        let cell_h = (height / rows as u32).max(1);
68        for row in 0..rows {
69            for col in 0..cols {
70                let cell = cells[row * cols + col];
71                let [br, bg, bb, ba] = cell.bg;
72                let px = (col as u32 * cell_w) as i32;
73                let py = (row as u32 * cell_h) as i32;
74                if ba > 0.0 {
75                    img.draw_rect(
76                        px,
77                        py,
78                        cell_w,
79                        cell_h,
80                        (br * 255.0).min(255.0) as u8,
81                        (bg * 255.0).min(255.0) as u8,
82                        (bb * 255.0).min(255.0) as u8,
83                        (ba * 255.0).min(255.0) as u8,
84                    );
85                }
86                let ch = char::from_u32(cell.ch).unwrap_or(' ');
87                if ch == ' ' {
88                    continue;
89                }
90                let [r, g, b, _a] = cell.fg;
91                let pr = (r * 255.0).min(255.0) as u8;
92                let pg = (g * 255.0).min(255.0) as u8;
93                let pb = (b * 255.0).min(255.0) as u8;
94                img.draw_rect(px, py, cell_w, cell_h, pr, pg, pb, 255);
95            }
96        }
97        img
98    }
