1//! Pattern-based text highlighting: match literal strings and assign foreground/background colors.
2//!
3//! - Span splitting: decompose input into colored runs for terminal cell rendering.
4//! - Leftmost-first rule priority with default fallback for unmatched regions.
5
6/// A plain-string pattern with associated foreground and optional background colors.
7pub struct HighlightRule {
8    /// Literal string to match in the input text.
9    pub pattern: String,
10    /// RGBA foreground color applied to matched text, components in 0.0–1.0.
11    pub fg: [f32; 4],
12    /// Optional RGBA background color; `None` leaves background unchanged.
13    pub bg: Option<[f32; 4]>,
14}
15
16/// A run of text with resolved foreground and optional background colors, ready for cell writing.
17pub struct ColoredSpan {
18    /// Decoded text content for this span.
19    pub text: String,
20    /// RGBA foreground color, components in 0.0–1.0.
21    pub fg: [f32; 4],
22    /// Optional RGBA background color; `None` means inherit default.
23    pub bg: Option<[f32; 4]>,
24}
25
26/// Split `text` into `ColoredSpan`s by applying `rules` in leftmost-first order; unmatched runs use `default_fg`.
27pub fn highlight_spans(
28    text: &str,
29    rules: &[HighlightRule],
30    default_fg: [f32; 4],
31) -> Vec<ColoredSpan> {
32    let mut result = Vec::new();
33    let mut remaining = text;
34    while !remaining.is_empty() {
35        let best = rules
36            .iter()
37            .filter_map(|r| remaining.find(r.pattern.as_str()).map(|pos| (pos, r)))
38            .min_by_key(|(pos, _)| *pos);
39        match best {
40            None => {
41                result.push(ColoredSpan {
42                    text: remaining.to_string(),
43                    fg: default_fg,
44                    bg: None,
45                });
46                break;
47            }
48            Some((pos, rule)) => {
49                if pos > 0 {
50                    let prefix = &remaining[..pos];
51                    result.push(ColoredSpan {
52                        text: prefix.to_string(),
53                        fg: default_fg,
54                        bg: None,
55                    });
56                }
57                let end = pos + rule.pattern.len();
58                let token = &remaining[pos..end];
59                result.push(ColoredSpan {
60                    text: token.to_string(),
61                    fg: rule.fg,
62                    bg: rule.bg,
63                });
64                remaining = &remaining[end..];
65            }
66        }
67    }
68    result
69}
