1//! Z-ordered draw-callback queue flushed once per frame by the render loop.
2//!
3//! - Entries hold a depth key and an opaque callback ID returned to Lua.
4//! - Sorted at flush time so draw callbacks execute in front-to-back order.
5
6/// A pending draw-callback slot queued in `DrawLayer`.
7pub struct LayerEntry {
8    /// Depth key used to sort entries front-to-back before flush.
9    pub z_order: f64,
10    /// Opaque callback ID assigned at `queue` time; passed back to the Lua runtime.
11    pub callback_id: usize,
12}
13/// Z-ordered pending-callback queue flushed once per frame by the render loop.
14pub struct DrawLayer {
15    /// Pending entries in insertion order; sorted at flush.
16    entries: Vec<LayerEntry>,
17    /// Monotonically incrementing counter for callback IDs.
18    next_id: usize,
19}
20impl DrawLayer {
21    /// Create an empty `DrawLayer` with ID counter starting at 0.
22    pub fn new() -> Self {
23        Self {
24            entries: Vec::new(),
25            next_id: 0,
26        }
27    }
28    /// Enqueue a callback at `z_order` depth and return its unique callback ID.
29    pub fn queue(&mut self, z_order: f64) -> usize {
30        let id = self.next_id;
31        self.next_id += 1;
32        self.entries.push(LayerEntry {
33            z_order,
34            callback_id: id,
35        });
36        id
37    }
38    /// Sort entries by `z_order`, drain, and return them; leaves the layer empty.
39    pub fn flush(&mut self) -> Vec<LayerEntry> {
40        self.entries.sort_by(|a, b| {
41            a.z_order
42                .partial_cmp(&b.z_order)
43                .unwrap_or(std::cmp::Ordering::Equal)
44        });
45        std::mem::take(&mut self.entries)
46    }
47    /// Discard all pending entries without firing callbacks.
48    pub fn clear(&mut self) {
49        self.entries.clear();
50    }
51    /// Return the number of pending entries.
52    pub fn get_count(&self) -> usize {
53        self.entries.len()
54    }
55}
56/// Delegate `Default` to `DrawLayer::new`.
57impl Default for DrawLayer {
58    fn default() -> Self {
59        Self::new()
60    }
61}
