3//! - Provides blend, stencil, and depth mode enums for compositing and test configuration.
4//! - Contains text alignment and draw-mode enums shared across shape, font, and path rendering.

6//! - Declares particle instance and render-shape types for the particle system pipeline.

19/// Depth comparison function for stencil and depth-buffer tests.
20#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
21pub enum CompareMode {
22    /// Pass only when values are equal.
23    Equal,
24    /// Pass only when values differ.
25    NotEqual,
26    /// Pass when incoming < stored.
27    Less,
28    /// Pass when incoming <= stored.
29    LessEqual,
30    /// Pass when incoming > stored.
31    Greater,
32    /// Pass when incoming >= stored.
33    GreaterEqual,
34    /// Always pass.
35    Always,
36    /// Always fail.
37    Never,
38}
39/// Stencil-buffer write operation applied when a stencil test passes.
40#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
41pub enum StencilAction {
42    /// Leave current value unchanged.
43    Keep,
44    /// Write zero.
45    Zero,
46    /// Write the reference value.
47    Replace,
48    /// Clamp-increment the current value.
49    Increment,
50    /// Clamp-decrement the current value.
51    Decrement,
52    /// Wrap-increment the current value.
53    IncrementWrap,
54    /// Wrap-decrement the current value.
55    DecrementWrap,
56    /// Bitwise-invert the current value.
57    Invert,
58}
59/// Combined stencil action, comparison, and reference value for one draw call.
60#[derive(Debug, Clone, Copy, PartialEq, Eq)]
61pub struct StencilMode {
62    /// How to update the stencil buffer on pass.
63    pub action: StencilAction,
64    /// Comparison used to test each fragment.
65    pub compare: CompareMode,
66    /// Reference value compared against the stencil buffer.
67    pub value: u8,
68}
69/// Provide a no-op `Default` for `StencilMode` (Keep / Always / 0).
70impl Default for StencilMode {
71    fn default() -> Self {
72        Self {
73            action: StencilAction::Keep,
74            compare: CompareMode::Always,
75            value: 0,
76        }
77    }
78}

86    Never,

88    Less,
89    /// Pass when incoming <= stored depth.
90    LessEqual,
91    /// Pass when incoming == stored depth.
92    Equal,
93    /// Pass when incoming != stored depth.
94    NotEqual,
95    /// Pass when incoming > stored depth.
96    Greater,
97    /// Pass when incoming >= stored depth.
98    GreaterEqual,
99}
