1//! OBJ model loader for 2D projection.
2//!
3//! Loads Wavefront .obj files and projects 3D geometry into 2D for use with
4//! the raycaster and globe rendering systems. This is NOT a 3D rendering
5//! pipeline — models are reduced to 2D projections (orthographic or perspective)
6//! for display in the 2D engine.
7//!
8//! ## Feature Gate
9//!
10//! This module is gated behind the `obj-loader` feature (enabled by default).
11//! Disable it to reduce binary size if your game doesn't use 3D model loading:
12//!
13//! ```toml
14//! [dependencies]
15//! lurek2d = { version = "...", default-features = false, features = [...] }
16//! ```
17//!
18//! ## Capabilities
19//!
20//! - Wavefront OBJ and MTL file loading via a built-in hand parser.
21//! - Triangulated face model with per-vertex position, UV, and normal indices.
22//! - Named materials carrying diffuse colour and optional texture path.
23//! - CPU software rasteriser producing `ImageData` thumbnails with back-face culling, Z-buffer, and key lighting.
24//! - Perspective projection of OBJ models into engine `Mesh` geometry for GPU rendering.
25//! - Instance projection with Y-axis rotation, uniform scale, and depth output for scene sorting.
26//! - Local `Vec3`/`Vec2` types for self-contained 3-D math without engine-wide dependencies.
27//! - `ObjCamera` helper packing position, lookat target, and FOV for projection calls.
28//! - `ObjLoader` stateless parser facade with both file-based and in-memory entry points.
29//! - MTL parsing extracting `newmtl`, `Kd`, and `map_Kd` into a flat material list.
30//! - OBJ face-vertex index resolver handling 1-based and negative (relative) indices.
31//! - Edge-function barycentric rasterisation for the CPU renderer path.
32

38/// Error returned from OBJ loading or parsing.
39#[derive(Debug)]
40pub enum ObjError {
41    /// File I/O error.
42    Io(std::io::Error),
43    /// Structural parse error with a human-readable message.
44    Parse(String),
45}
46/// Implement `Display` for `ObjError`.
47impl std::fmt::Display for ObjError {
48    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
49        match self {
50            ObjError::Io(e) => write!(f, "OBJ IO error: {}", e),
51            ObjError::Parse(s) => write!(f, "OBJ parse error: {}", s),
52        }
53    }
54}
55/// Convert `std::io::Error` into `ObjError::Io`.
56impl From<std::io::Error> for ObjError {
57    fn from(e: std::io::Error) -> Self {
58        ObjError::Io(e)
59    }
60}
61/// Simple 3-D vector used for positions, normals, and scratch arithmetic during projection.
62#[derive(Debug, Clone, Copy)]
63pub struct Vec3 {
64    /// X component.
65    pub x: f32,
66    /// Y component.
67    pub y: f32,
68    /// Z component.
69    pub z: f32,
70}
71/// Arithmetic helpers for the local 3-D vector type.
72impl Vec3 {
73    /// Construct from components. This function is part of the public API.
74    pub fn new(x: f32, y: f32, z: f32) -> Self {
75        Self { x, y, z }
76    }
77    /// Return the scalar dot product of `self` and `other`.
78    pub fn dot(self, other: Vec3) -> f32 {
79        self.x * other.x + self.y * other.y + self.z * other.z
80    }
81    /// Return the Euclidean length of this vector.
82    pub fn len(self) -> f32 {
83        (self.x * self.x + self.y * self.y + self.z * self.z).sqrt()
84    }
85    /// Return a unit vector; returns zero vector when length < 1e-9.
86    pub fn normalise(self) -> Vec3 {
87        let l = self.len();
88        if l < 1e-9 {
89            Vec3::new(0.0, 0.0, 0.0)
90        } else {
91            Vec3::new(self.x / l, self.y / l, self.z / l)
92        }
93    }
94    /// Return `self - o`. This function is part of the public API.
95    #[allow(clippy::should_implement_trait)]
96    pub fn sub(self, o: Vec3) -> Vec3 {
97        Vec3::new(self.x - o.x, self.y - o.y, self.z - o.z)
98    }
99    /// Return the cross product `self × o`.
100    pub fn cross(self, o: Vec3) -> Vec3 {
101        Vec3::new(
102            self.y * o.z - self.z * o.y,
103            self.z * o.x - self.x * o.z,
104            self.x * o.y - self.y * o.x,
105        )
106    }
107    /// Return `self + o`. This function is part of the public API.
108    #[allow(clippy::should_implement_trait)]
109    pub fn add(self, o: Vec3) -> Vec3 {
110        Vec3::new(self.x + o.x, self.y + o.y, self.z + o.z)
111    }
112    /// Return `self * s` (scalar multiply).
113    #[allow(clippy::should_implement_trait)]
114    pub fn mul(self, s: f32) -> Vec3 {
115        Vec3::new(self.x * s, self.y * s, self.z * s)
116    }
117}
118/// Rotate `v` around the world Y-axis by `angle` radians.
119fn rotate_y(v: Vec3, angle: f32) -> Vec3 {
120    let c = angle.cos();
121    let s = angle.sin();
122    Vec3::new(v.x * c + v.z * s, v.y, -v.x * s + v.z * c)
123}
124/// Compute a 2D edge function value used for barycentric rasterization.
125fn edge_fn(ax: f32, ay: f32, bx: f32, by: f32, px: f32, py: f32) -> f32 {
126    (px - ax) * (by - ay) - (py - ay) * (bx - ax)
127}

186    /// CPU-rasterize the model into an `ImageData` with a virtual camera, Y-rotation, and a key light.
187    pub fn render_to_image(&self, width: u32, height: u32, rotation_quarters: u8) -> ImageData {
188        let mut image = ImageData::new(width, height);
189        if width == 0 || height == 0 || self.positions.is_empty() {
190            return image;
191        }
192        let angle = (rotation_quarters % 4) as f32 * std::f32::consts::FRAC_PI_2;
193        let rotated: Vec<Vec3> = self
194            .positions
195            .iter()
196            .copied()
197            .map(|p| rotate_y(p, angle))
198            .collect();
199        let mut min = Vec3::new(f32::INFINITY, f32::INFINITY, f32::INFINITY);
200        let mut max = Vec3::new(f32::NEG_INFINITY, f32::NEG_INFINITY, f32::NEG_INFINITY);
201        for p in &rotated {
202            min.x = min.x.min(p.x);
203            min.y = min.y.min(p.y);
204            min.z = min.z.min(p.z);
205            max.x = max.x.max(p.x);
206            max.y = max.y.max(p.y);
207            max.z = max.z.max(p.z);
208        }
209        let center = Vec3::new(
210            (min.x + max.x) * 0.5,
211            (min.y + max.y) * 0.5,
212            (min.z + max.z) * 0.5,
213        );
214        let extent = Vec3::new(max.x - min.x, max.y - min.y, max.z - min.z);
215        let radius = ((extent.x * extent.x + extent.y * extent.y + extent.z * extent.z).sqrt()
216            * 0.5)
217            .max(0.5);
218        let cam_pos = center.add(Vec3::new(radius * 1.2, radius * 0.8, radius * 2.4));
219        let cam_target = center.add(Vec3::new(0.0, extent.y * 0.15, 0.0));
220        let forward = cam_target.sub(cam_pos).normalise();
221        let world_up = Vec3::new(0.0, 1.0, 0.0);
222        let right = forward.cross(world_up).normalise();
223        let up = right.cross(forward).normalise();
224        let fov_y = 50.0_f32.to_radians();
225        let tan_half_fov = (fov_y * 0.5).tan();
226        let aspect = width as f32 / height as f32;
227        let near_z = 0.05_f32;
228        let light_dir = Vec3::new(0.45, 0.90, 0.55).normalise();
229        let mut zbuf = vec![f32::INFINITY; (width * height) as usize];
230        for face in &self.faces {
231            let wp = [
232                rotated[face.verts[0].0],
233                rotated[face.verts[1].0],
234                rotated[face.verts[2].0],
235            ];
236            let e0 = wp[1].sub(wp[0]);
237            let e1 = wp[2].sub(wp[0]);
238            let face_normal = e0.cross(e1).normalise();
239            let to_cam = cam_pos.sub(wp[0]).normalise();
240            if face_normal.dot(to_cam) <= 0.0 {
241                continue;
242            }
243            let z0 = wp[0].sub(cam_pos).dot(forward);
244            let z1 = wp[1].sub(cam_pos).dot(forward);
245            let z2 = wp[2].sub(cam_pos).dot(forward);
246            if z0 <= near_z || z1 <= near_z || z2 <= near_z {
247                continue;
248            }
249            let mut sx = [0.0_f32; 3];
250            let mut sy = [0.0_f32; 3];
251            for i in 0..3 {
252                let rel = wp[i].sub(cam_pos);
253                let vz = rel.dot(forward);
254                let vx = rel.dot(right);
255                let vy = rel.dot(up);
256                let ndcx = vx / (vz * tan_half_fov * aspect);
257                let ndcy = vy / (vz * tan_half_fov);
258                sx[i] = (ndcx + 1.0) * 0.5 * width as f32;
259                sy[i] = (1.0 - (ndcy + 1.0) * 0.5) * height as f32;
260            }
261            let area = edge_fn(sx[0], sy[0], sx[1], sy[1], sx[2], sy[2]);
262            if area.abs() < 1e-4 {
263                continue;
264            }
265            let ndotl = face_normal.dot(light_dir).max(0.0);
266            let shade = 0.28 + 0.72 * ndotl;
267            let (r, g, b) = if let Some(mat_idx) = face.material {
268                if let Some(mat) = self.materials.get(mat_idx) {
269                    (
270                        (mat.diffuse_color[0] * shade * 255.0).clamp(0.0, 255.0) as u8,
271                        (mat.diffuse_color[1] * shade * 255.0).clamp(0.0, 255.0) as u8,
272                        (mat.diffuse_color[2] * shade * 255.0).clamp(0.0, 255.0) as u8,
273                    )
274                } else {
275                    let c = (shade * 255.0).clamp(0.0, 255.0) as u8;
276                    (c, c, c)
277                }
278            } else {
279                let c = (shade * 255.0).clamp(0.0, 255.0) as u8;
280                (c, c, c)
281            };
282            let min_x = sx
283                .iter()
284                .copied()
285                .fold(f32::INFINITY, f32::min)
286                .floor()
287                .max(0.0) as u32;
288            let max_x = sx
289                .iter()
290                .copied()
291                .fold(f32::NEG_INFINITY, f32::max)
292                .ceil()
293                .min(width as f32 - 1.0) as u32;
294            let min_y = sy
295                .iter()
296                .copied()
297                .fold(f32::INFINITY, f32::min)
298                .floor()
299                .max(0.0) as u32;
300            let max_y = sy
301                .iter()
302                .copied()
303                .fold(f32::NEG_INFINITY, f32::max)
304                .ceil()
305                .min(height as f32 - 1.0) as u32;
306            for py in min_y..=max_y {
307                for px in min_x..=max_x {
308                    let fx = px as f32 + 0.5;
309                    let fy = py as f32 + 0.5;
310                    let w0 = edge_fn(sx[1], sy[1], sx[2], sy[2], fx, fy) / area;
311                    let w1 = edge_fn(sx[2], sy[2], sx[0], sy[0], fx, fy) / area;
312                    let w2 = edge_fn(sx[0], sy[0], sx[1], sy[1], fx, fy) / area;
313                    if w0 < 0.0 || w1 < 0.0 || w2 < 0.0 {
314                        continue;
315                    }
316                    let depth = w0 * z0 + w1 * z1 + w2 * z2;
317                    let idx = (py * width + px) as usize;
318                    if depth >= zbuf[idx] {
319                        continue;
320                    }
321                    zbuf[idx] = depth;
322                    image.set_pixel(px, py, r, g, b, 255);
323                }
324            }
325        }
326        image
327    }
328    /// Project the model from `cam_pos`/`cam_target` into a `Mesh` sorted back-to-front.
329    pub fn project_to_mesh(
330        &self,
331        cam_pos: Vec3,
332        cam_target: Vec3,
333        fov_y: f32,
334        screen_w: f32,
335        screen_h: f32,
336        texture_key: Option<TextureKey>,
337    ) -> Mesh {
338        let forward = cam_target.sub(cam_pos).normalise();
339        let world_up = Vec3::new(0.0, 1.0, 0.0);
340        let right = forward.cross(world_up).normalise();
341        let up = right.cross(forward).normalise();
342        let aspect = screen_w / screen_h;
343        let tan_half_fov = (fov_y * 0.5).tan();
344        let light_dir = Vec3::new(0.5, 1.0, 0.7).normalise();
345        let mut projected_tris: Vec<(f32, [MeshVertex; 3])> = Vec::with_capacity(self.faces.len());
346        let near_z = 0.05_f32;
347        for face in &self.faces {
348            let wp: [Vec3; 3] = [
349                self.positions[face.verts[0].0],
350                self.positions[face.verts[1].0],
351                self.positions[face.verts[2].0],
352            ];
353            let e0 = wp[1].sub(wp[0]);
354            let e1 = wp[2].sub(wp[0]);
355            let face_normal = e0.cross(e1).normalise();
356            let z0 = wp[0].sub(cam_pos).dot(forward);
357            let z1 = wp[1].sub(cam_pos).dot(forward);
358            let z2 = wp[2].sub(cam_pos).dot(forward);
359            if z0 <= near_z || z1 <= near_z || z2 <= near_z {
360                continue;
361            }
362            let to_cam = cam_pos.sub(wp[0]).normalise();
363            if face_normal.dot(to_cam) <= 0.0 {
364                continue;
365            }
366            let ndotl = face_normal.dot(light_dir).max(0.0);
367            let shade = 0.35 + 0.65 * ndotl;
368            let (r, g, b) = if let Some(mat_idx) = face.material {
369                if let Some(mat) = self.materials.get(mat_idx) {
370                    (
371                        mat.diffuse_color[0] * shade,
372                        mat.diffuse_color[1] * shade,
373                        mat.diffuse_color[2] * shade,
374                    )
375                } else {
376                    (shade, shade, shade)
377                }
378            } else {
379                (shade, shade, shade)
380            };
381            let mut tri = [MeshVertex::default(); 3];
382            for (vi, &wp_v) in wp.iter().enumerate() {
383                let rel = wp_v.sub(cam_pos);
384                let vz = rel.dot(forward);
385                let clip_z = vz.max(0.001);
386                let vx = rel.dot(right);
387                let vy = rel.dot(up);
388                let ndcx = vx / (clip_z * tan_half_fov * aspect);
389                let ndcy = vy / (clip_z * tan_half_fov);
390                let sx = (ndcx + 1.0) * 0.5 * screen_w;
391                let sy = (1.0 - (ndcy + 1.0) * 0.5) * screen_h;
392                let (u, v) = if let Some(uv_idx) = face.verts[vi].1 {
393                    if let Some(uv) = self.uvs.get(uv_idx) {
394                        (uv.u, 1.0 - uv.v)
395                    } else {
396                        (0.0, 0.0)
397                    }
398                } else {
399                    (0.0, 0.0)
400                };
401                tri[vi] = MeshVertex {
402                    x: sx,
403                    y: sy,
404                    u,
405                    v,
406                    r: r.min(1.0),
407                    g: g.min(1.0),
408                    b: b.min(1.0),
409                    a: 1.0,
410                };
411            }
412            let tri_depth = z0.max(z1).max(z2);
413            projected_tris.push((tri_depth, tri));
414        }
415        projected_tris.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(std::cmp::Ordering::Equal));
416        let mut vertices: Vec<MeshVertex> = Vec::with_capacity(projected_tris.len() * 3);
417        for (_, tri) in projected_tris {
418            vertices.push(tri[0]);
419            vertices.push(tri[1]);
420            vertices.push(tri[2]);
421        }
422        let mut mesh = Mesh::from_vertices(vertices, MeshDrawMode::Triangles);
423        mesh.texture = texture_key;
424        mesh
425    }
426    /// Project a single world-space instance with Y-rotation and uniform scale; return `(Mesh, depth)`.
427    #[allow(clippy::too_many_arguments)]
428    pub fn project_instance_to_mesh(
429        &self,
430        cam_pos: Vec3,
431        cam_target: Vec3,
432        fov_y: f32,
433        screen_w: f32,
434        screen_h: f32,
435        world_x: f32,
436        world_y: f32,
437        rotation_quarters: u8,
438        scale: f32,
439    ) -> (Mesh, f32) {
440        let forward = cam_target.sub(cam_pos).normalise();
441        let world_up = Vec3::new(0.0, 1.0, 0.0);
442        let right = forward.cross(world_up).normalise();

448        let rot = (rotation_quarters % 4) as f32 * std::f32::consts::FRAC_PI_2;
449        let c = rot.cos();
450        let s = rot.sin();
451        let mut projected_tris: Vec<(f32, [MeshVertex; 3])> = Vec::with_capacity(self.faces.len());
452        let mut min_x = f32::INFINITY;
453        let mut max_x = f32::NEG_INFINITY;
454        let mut min_y = f32::INFINITY;
455        let mut min_z = f32::INFINITY;
456        let mut max_z = f32::NEG_INFINITY;
457        for p in &self.positions {
458            min_x = min_x.min(p.x);
459            max_x = max_x.max(p.x);
460            min_y = min_y.min(p.y);
461            min_z = min_z.min(p.z);
462            max_z = max_z.max(p.z);
463        }
464        let center_x = (min_x + max_x) * 0.5;
465        let center_z = (min_z + max_z) * 0.5;
466        let base_y = min_y;
467        let mut instance_depth = f32::INFINITY;
468        for face in &self.faces {

471                let px = (p.x - center_x) * scale;
472                let py = (p.y - base_y) * scale;
473                let pz = (p.z - center_z) * scale;
474                let rx = px * c + pz * s;
475                let rz = -px * s + pz * c;
476                Vec3::new(world_x + rx, py, world_y + rz)
477            });
478            let e0 = wp[1].sub(wp[0]);
479            let e1 = wp[2].sub(wp[0]);

482            let z1 = wp[1].sub(cam_pos).dot(forward);

486            }
487            let to_cam = cam_pos.sub(wp[0]).normalise();

489                continue;
490            }

497                        mat.diffuse_color[1] * shade,
498                        mat.diffuse_color[2] * shade,
499                    )
500                } else {
501                    (shade, shade, shade)
502                }
503            } else {
504                (shade, shade, shade)
505            };
506            let mut tri = [MeshVertex::default(); 3];
507            for (vi, &wp_v) in wp.iter().enumerate() {
508                let rel = wp_v.sub(cam_pos);
509                let vz = rel.dot(forward);

527                    x: sx,
528                    y: sy,
529                    u,
530                    v,
531                    r: r.min(1.0),
532                    g: g.min(1.0),
533                    b: b.min(1.0),
534                    a: 1.0,
535                };
536            }
537            let tri_depth = z0.max(z1).max(z2);
538            instance_depth = instance_depth.min((z0 + z1 + z2) / 3.0);
539            projected_tris.push((tri_depth, tri));
540        }
541        projected_tris.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(std::cmp::Ordering::Equal));
542        let mut vertices: Vec<MeshVertex> = Vec::with_capacity(projected_tris.len() * 3);
543        for (_, tri) in projected_tris {
544            vertices.push(tri[0]);
545            vertices.push(tri[1]);
546            vertices.push(tri[2]);
547        }
548        (
549            Mesh::from_vertices(vertices, MeshDrawMode::Triangles),
550            instance_depth,
551        )
552    }
553}
554/// Stateless parser wrapper for Wavefront OBJ files.
555pub struct ObjLoader;
556/// File-based and in-memory OBJ/MTL parsing entry points.
557impl ObjLoader {
558    /// Load and triangulate an OBJ file from `path`, using `tobj`; return error on I/O or parse failure.
559    pub fn load_file(path: impl AsRef<Path>) -> Result<ObjModel, ObjError> {
560        let path = path.as_ref();
561        let (models, materials_result) = tobj::load_obj(
562            path,
563            &tobj::LoadOptions {
564                triangulate: true,
565                single_index: true,
566                ignore_lines: true,
567                ignore_points: true,
568            },
569        )
570        .map_err(|e| ObjError::Parse(format!("tobj load failed: {e}")))?;
571        let materials_raw = materials_result.unwrap_or_default();
572        let mut materials = Vec::with_capacity(materials_raw.len());
573        for mat in materials_raw {
574            let kd = mat.diffuse.unwrap_or([1.0, 1.0, 1.0]);
575            materials.push(ObjMaterial {
576                name: mat.name,
577                diffuse_color: kd,
578                diffuse_map: mat.diffuse_texture,
579            });
580        }
581        let mut positions = Vec::new();
582        let mut uvs = Vec::new();
583        let mut normals = Vec::new();
584        let mut faces = Vec::new();
585        for model in models {
586            let mesh = model.mesh;
587            let base_pos = positions.len();
588            let vcount = mesh.positions.len() / 3;
589            for i in 0..vcount {
590                positions.push(Vec3::new(
591                    mesh.positions[i * 3],
592                    mesh.positions[i * 3 + 1],
593                    mesh.positions[i * 3 + 2],
594                ));
595                if mesh.texcoords.len() >= (i + 1) * 2 {
596                    uvs.push(Vec2 {
597                        u: mesh.texcoords[i * 2],
598                        v: mesh.texcoords[i * 2 + 1],
599                    });
600                } else {
601                    uvs.push(Vec2 { u: 0.0, v: 0.0 });
602                }
603                if mesh.normals.len() >= (i + 1) * 3 {
604                    normals.push(Vec3::new(
605                        mesh.normals[i * 3],
606                        mesh.normals[i * 3 + 1],
607                        mesh.normals[i * 3 + 2],
608                    ));
609                } else {
610                    normals.push(Vec3::new(0.0, 1.0, 0.0));
611                }
612            }
613            for tri in mesh.indices.chunks_exact(3) {
614                let a = base_pos + tri[0] as usize;
615                let b = base_pos + tri[1] as usize;
616                let c = base_pos + tri[2] as usize;
617                let material = mesh.material_id;
618                faces.push(ObjFace {
619                    verts: [
620                        (a, Some(a), Some(a)),
621                        (b, Some(b), Some(b)),
622                        (c, Some(c), Some(c)),
623                    ],
624                    material,
625                });
626            }
627        }
628        Ok(ObjModel {
629            positions,
630            uvs,
631            normals,
632            faces,
633            materials,
634        })
635    }
636    /// Parse OBJ + MTL text in memory, resolving `mtllib` paths relative to `base_dir`.
637    pub fn parse_obj(src: &str, base_dir: &Path) -> Result<ObjModel, ObjError> {
638        let mut positions: Vec<Vec3> = Vec::new();
639        let mut uvs: Vec<Vec2> = Vec::new();
640        let mut normals: Vec<Vec3> = Vec::new();
641        let mut faces: Vec<ObjFace> = Vec::new();
642        let mut materials: Vec<ObjMaterial> = Vec::new();
643        let mut mat_index: HashMap<String, usize> = HashMap::new();
644        let mut current_mat: Option<usize> = None;
645        for line in src.lines() {
646            let line = line.trim();
647            if line.is_empty() || line.starts_with('#') {
648                continue;
649            }
650            let mut tokens = line.splitn(2, char::is_whitespace);
651            let keyword = tokens.next().unwrap_or("");
652            let rest = tokens.next().unwrap_or("").trim();
653            match keyword {
654                "v" => {
655                    let p = Self::parse_vec3(rest)?;
656                    positions.push(p);
657                }
658                "vt" => {
659                    let parts = Self::split_floats(rest, 2)?;
660                    uvs.push(Vec2 {
661                        u: parts[0],
662                        v: parts[1],
663                    });
664                }
665                "vn" => {
666                    let p = Self::parse_vec3(rest)?;
667                    normals.push(p);
668                }
669                "mtllib" => {
670                    let mtl_path = base_dir.join(rest);
671                    if let Ok(mtl_src) = std::fs::read_to_string(&mtl_path) {
672                        let loaded = Self::parse_mtl(&mtl_src, base_dir);
673                        for m in loaded {
674                            if !mat_index.contains_key(&m.name) {
675                                mat_index.insert(m.name.clone(), materials.len());
676                                materials.push(m);
677                            }
678                        }
679                    }
680                }
681                "usemtl" => {
682                    current_mat = mat_index.get(rest).copied();
683                }
684                "f" => {
685                    let verts =
686                        Self::parse_face_verts(rest, positions.len(), uvs.len(), normals.len())?;
687                    for i in 1..(verts.len() - 1) {
688                        faces.push(ObjFace {
689                            verts: [verts[0], verts[i], verts[i + 1]],
690                            material: current_mat,
691                        });
692                    }
693                }
694                _ => {}
695            }
696        }
697        Ok(ObjModel {
698            positions,
699            uvs,
700            normals,
701            faces,
702            materials,
703        })
704    }
705    /// Parse MTL text; extract `newmtl`, `Kd`, and `map_Kd` entries into `ObjMaterial` list.
706    fn parse_mtl(src: &str, _base_dir: &Path) -> Vec<ObjMaterial> {
707        let mut out: Vec<ObjMaterial> = Vec::new();
708        let mut current: Option<ObjMaterial> = None;
709        for line in src.lines() {
710            let line = line.trim();
711            if line.is_empty() || line.starts_with('#') {
712                continue;
713            }
714            let mut tokens = line.splitn(2, char::is_whitespace);
715            let keyword = tokens.next().unwrap_or("");
716            let rest = tokens.next().unwrap_or("").trim();
717            match keyword {
718                "newmtl" => {
719                    if let Some(m) = current.take() {
720                        out.push(m);
721                    }
722                    current = Some(ObjMaterial {
723                        name: rest.to_owned(),
724                        diffuse_color: [1.0, 1.0, 1.0],
725                        diffuse_map: None,
726                    });
727                }
728                "Kd" => {
729                    if let Some(m) = current.as_mut() {
730                        if let Ok(parts) = Self::split_floats(rest, 3) {
731                            m.diffuse_color = [parts[0], parts[1], parts[2]];
732                        }
733                    }
734                }
735                "map_Kd" => {
736                    if let Some(m) = current.as_mut() {
737                        m.diffuse_map = Some(rest.to_owned());
738                    }
739                }
740                _ => {}
741            }
742        }
743        if let Some(m) = current {
744            out.push(m);
745        }
746        out
747    }
748    /// Parse `s` as three whitespace-separated floats into a `Vec3`; return error on bad input.
749    fn parse_vec3(s: &str) -> Result<Vec3, ObjError> {
750        let parts = Self::split_floats(s, 3)?;
751        Ok(Vec3::new(parts[0], parts[1], parts[2]))
752    }
753    /// Split `s` into at least `expected` floats; return error when fewer are found.
754    fn split_floats(s: &str, expected: usize) -> Result<Vec<f32>, ObjError> {
755        let v: Vec<f32> = s
756            .split_whitespace()
757            .take(expected + 4)
758            .map(|t| {
759                t.parse::<f32>()
760                    .map_err(|_| ObjError::Parse(format!("expected float, got '{}'", t)))
761            })
762            .collect::<Result<_, _>>()?;
763        if v.len() < expected {
764            return Err(ObjError::Parse(format!(
765                "expected {} floats, got {} in '{}'",
766                expected,
767                v.len(),
768                s
769            )));
770        }
771        Ok(v)
772    }
773    /// Parse a whitespace-delimited face vertex string into `(pos, uv, normal)` index triples.
774    #[allow(clippy::type_complexity)]
775    fn parse_face_verts(
776        s: &str,
777        pos_count: usize,
778        uv_count: usize,
779        norm_count: usize,
780    ) -> Result<Vec<(usize, Option<usize>, Option<usize>)>, ObjError> {
781        let mut out = Vec::new();
782        for token in s.split_whitespace() {
783            let mut parts = token.split('/');
784            let pi = Self::parse_index(parts.next(), pos_count)?;
785            let ti = Self::parse_index_opt(parts.next(), uv_count);
786            let ni = Self::parse_index_opt(parts.next(), norm_count);
787            out.push((pi, ti, ni));
788        }
789        if out.len() < 3 {
790            return Err(ObjError::Parse(format!(
791                "face has only {} vertices (need >= 3): '{}'",
792                out.len(),
793                s
794            )));
795        }
796        Ok(out)
797    }
798    /// Resolve an OBJ 1-based or negative index into a 0-based index; return error when `s` is empty.
799    fn parse_index(s: Option<&str>, count: usize) -> Result<usize, ObjError> {
800        match s {
801            None | Some("") => Err(ObjError::Parse("missing face index".into())),
802            Some(t) => {
803                let i: i32 = t
804                    .parse()
805                    .map_err(|_| ObjError::Parse(format!("bad index '{}'", t)))?;
806                let idx = if i < 0 {
807                    (count as i32 + i) as usize
808                } else {
809                    (i - 1) as usize
810                };
811                Ok(idx)
812            }
813        }
814    }
815    /// Like `parse_index` but return `None` for empty or missing tokens.
816    fn parse_index_opt(s: Option<&str>, count: usize) -> Option<usize> {
817        match s {
818            None | Some("") => None,
819            Some(t) => {
820                let i: i32 = t.parse().ok()?;
821                let idx = if i < 0 {
822                    (count as i32 + i) as usize
823                } else {
824                    (i - 1) as usize
825                };
826                Some(idx)
827            }
828        }
829    }
830}
