1//! ENet host wrapper owning a non-blocking UDP socket and all peer slots for one endpoint.
2//!
3//! - Host role classification (Server, Client, combined Host) for session routing.
4//! - Event-driven poll loop yielding Connect, Disconnect, and Receive events.
5//! - Connection lifecycle: initiate, graceful disconnect, forced disconnect, and reset.
6//! - Unicast and broadcast packet sending with reliable or unreliable delivery.
7//! - Peer diagnostics: round-trip time, connection state, address, and full statistics snapshot.
8//! - Bandwidth and channel limit configuration at runtime.
9//! - Convenience constructors for common server and client bind patterns.
10
11use super::constants::{DEFAULT_CHANNELS, DEFAULT_PEERS, MAX_PEERS};
12use super::error::NetworkError;
13use crate::log_msg;
14use crate::runtime::log_messages::{NW01_HOST_BIND, NW04_NET_ERROR};
15use rusty_enet::{self as enet, Host, HostSettings, Packet, PacketKind, PeerID};
16use std::net::{SocketAddr, UdpSocket};
17use std::time::Duration;
18/// Logical role of this host in a multiplayer session.
19#[derive(Debug, Clone, Copy, PartialEq, Eq)]
20pub enum HostRole {
21    /// Dedicated or listen server that accepts incoming peers.
22    Server,
23    /// Client that connects to a remote server with a single peer slot.
24    Client,
25    /// Combined host/client running both sides locally.
26    Host,
27}
28/// ENet host wrapping a UDP socket; owns all peer slots for one network endpoint.
29pub struct NetworkHost {
30    /// Underlying ENet host; `None` after `destroy()` is called.
31    inner: Option<Host<UdpSocket>>,
32    /// Bound socket address resolved at creation time.
33    local_addr: SocketAddr,
34    /// Role assigned at creation or via `set_role`.
35    role: HostRole,
36}
37/// Events emitted by `NetworkHost::service` on each poll cycle.
38pub enum NetworkEvent {
39    /// A remote peer completed the ENet handshake and is now connected.
40    Connect {
41        /// ENet peer identifier assigned to the new peer.
42        peer_id: PeerID,
43        /// Application-defined connection data sent by the peer.
44        data: u32,
45    },
46    /// A remote peer disconnected or timed out.
47    Disconnect {
48        /// ENet peer identifier of the disconnected peer.
49        peer_id: PeerID,
50        /// Application-defined disconnection reason code.
51        data: u32,
52    },
53    /// A packet arrived from a remote peer on a specific channel.
54    Receive {
55        /// ENet peer identifier of the sender.
56        peer_id: PeerID,
57        /// Channel index the packet was sent on.
58        channel_id: u8,
59        /// Raw packet payload bytes.
60        data: Vec<u8>,
61    },
62}
63/// Core networking operations: creation, polling, sending, peer management, and teardown.
64impl NetworkHost {
65    /// Create and bind a new ENet host; returns `PeerLimitExceeded` if `peer_count` exceeds `MAX_PEERS`.
66    pub fn new(
67        bind_addr: SocketAddr,
68        peer_count: Option<usize>,
69        channel_count: Option<usize>,
70        in_bandwidth: Option<u32>,
71        out_bandwidth: Option<u32>,
72    ) -> Result<Self, NetworkError> {
73        let peers = peer_count.unwrap_or(DEFAULT_PEERS);
74        if peers > MAX_PEERS {
75            log_msg!(
76                error,
77                NW04_NET_ERROR,
78                "peer limit: {} > {}",
79                peers,
80                MAX_PEERS
81            );
82            return Err(NetworkError::PeerLimitExceeded {
83                requested: peers,
84                max: MAX_PEERS,
85            });
86        }
87        let channels = channel_count.unwrap_or(DEFAULT_CHANNELS);
88        let socket = UdpSocket::bind(bind_addr)?;
89        socket.set_nonblocking(true)?;
90        let local_addr = socket.local_addr()?;
91        let settings = HostSettings {
92            peer_limit: peers,
93            channel_limit: channels,
94            incoming_bandwidth_limit: in_bandwidth,
95            outgoing_bandwidth_limit: out_bandwidth,
96            ..HostSettings::default()
97        };
98        let host = Host::new(socket, settings).map_err(|e| NetworkError::Enet(format!("{e}")))?;
99        log_msg!(info, NW01_HOST_BIND, "{}", local_addr);
100        Ok(Self {
101            inner: Some(host),
102            local_addr,
103            role: HostRole::Host,
104        })
105    }
106    /// Return a shared reference to the inner ENet host; returns `HostDestroyed` if destroyed.
107    fn host(&self) -> Result<&Host<UdpSocket>, NetworkError> {
108        self.inner.as_ref().ok_or(NetworkError::HostDestroyed)
109    }
110    /// Return a mutable reference to the inner ENet host; returns `HostDestroyed` if destroyed.
111    fn host_mut(&mut self) -> Result<&mut Host<UdpSocket>, NetworkError> {
112        self.inner.as_mut().ok_or(NetworkError::HostDestroyed)
113    }
114    /// Poll the ENet host for one pending event; returns `None` when the queue is empty.
115    pub fn service(&mut self) -> Result<Option<NetworkEvent>, NetworkError> {
116        let host = self.host_mut()?;
117        match host.service() {
118            Ok(Some(event)) => {
119                let net_event = match event.no_ref() {
120                    enet::EventNoRef::Connect { peer, data } => NetworkEvent::Connect {
121                        peer_id: peer,
122                        data,
123                    },
124                    enet::EventNoRef::Disconnect { peer, data } => NetworkEvent::Disconnect {
125                        peer_id: peer,
126                        data,
127                    },
128                    enet::EventNoRef::Receive {
129                        peer,
130                        channel_id,
131                        packet,
132                    } => NetworkEvent::Receive {
133                        peer_id: peer,
134                        channel_id,
135                        data: packet.data().to_vec(),
136                    },
137                };
138                Ok(Some(net_event))
139            }
140            Ok(None) => Ok(None),
141            Err(e) => Err(NetworkError::Enet(format!("{e}"))),
142        }
143    }
144    /// Initiate a connection to `address` and return the assigned `PeerID` on success.
145    pub fn connect(
146        &mut self,
147        address: SocketAddr,
148        channel_count: usize,
149        data: u32,
150    ) -> Result<PeerID, NetworkError> {
151        let host = self.host_mut()?;
152        let peer = host
153            .connect(address, channel_count, data)
154            .map_err(|e| NetworkError::Enet(format!("{e}")))?;
155        Ok(peer.id())
156    }
157    /// Send a pre-built `Packet` to the given peer on `channel_id`.
158    pub fn send(
159        &mut self,
160        peer_id: PeerID,
161        channel_id: u8,
162        packet: Packet,
163    ) -> Result<(), NetworkError> {
164        let host = self.host_mut()?;
165        let peer = host.peer_mut(peer_id);
166        let _ = peer.send(channel_id, &packet);
167        Ok(())
168    }
169    /// Send raw bytes to a peer; uses reliable ordered delivery when `reliable` is `true`.
170    pub fn send_bytes(
171        &mut self,
172        peer_id: PeerID,
173        channel_id: u8,
174        data: &[u8],
175        reliable: bool,
176    ) -> Result<(), NetworkError> {
177        let kind = if reliable {
178            PacketKind::Reliable
179        } else {
180            PacketKind::Unreliable { sequenced: true }
181        };
182        self.send(peer_id, channel_id, Packet::new(data, kind))
183    }
184    /// Broadcast a pre-built `Packet` to all connected peers on `channel_id`.
185    pub fn broadcast(&mut self, channel_id: u8, packet: &Packet) -> Result<(), NetworkError> {
186        let host = self.host_mut()?;
187        host.broadcast(channel_id, packet);
188        Ok(())
189    }
190    /// Broadcast raw bytes to all connected peers; uses reliable delivery when `reliable` is `true`.
191    pub fn broadcast_bytes(
192        &mut self,
193        channel_id: u8,
194        data: &[u8],
195        reliable: bool,
196    ) -> Result<(), NetworkError> {
197        let kind = if reliable {
198            PacketKind::Reliable
199        } else {
200            PacketKind::Unreliable { sequenced: true }
201        };
202        let packet = Packet::new(data, kind);
203        self.broadcast(channel_id, &packet)
204    }
205    /// Flush the host's outgoing packet queue to the socket immediately.
206    pub fn flush(&mut self) -> Result<(), NetworkError> {
207        let host = self.host_mut()?;
208        host.flush();
209        Ok(())
210    }
211    /// Begin a graceful ENet disconnect handshake with the peer.
212    pub fn disconnect(&mut self, peer_id: PeerID, data: u32) -> Result<(), NetworkError> {
213        let host = self.host_mut()?;
214        host.peer_mut(peer_id).disconnect(data);
215        Ok(())
216    }
217    /// Immediately forcibly disconnect the peer without a handshake.
218    pub fn disconnect_now(&mut self, peer_id: PeerID, data: u32) -> Result<(), NetworkError> {
219        let host = self.host_mut()?;
220        host.peer_mut(peer_id).disconnect_now(data);
221        Ok(())
222    }
223    /// Queue a graceful disconnect after all queued outgoing packets have been sent.
224    pub fn disconnect_later(&mut self, peer_id: PeerID, data: u32) -> Result<(), NetworkError> {
225        let host = self.host_mut()?;
226        host.peer_mut(peer_id).disconnect_later(data);
227        Ok(())
228    }
229    /// Reset a peer's state to `Disconnected` without sending any disconnect notification.
230    pub fn reset_peer(&mut self, peer_id: PeerID) -> Result<(), NetworkError> {
231        let host = self.host_mut()?;
232        host.peer_mut(peer_id).reset();
233        Ok(())
234    }
235    /// Send a ping packet to the peer to refresh the round-trip-time estimate.
236    pub fn ping(&mut self, peer_id: PeerID) -> Result<(), NetworkError> {
237        let host = self.host_mut()?;
238        host.peer_mut(peer_id).ping();
239        Ok(())
240    }
241    /// Return the current measured round-trip time for the peer; returns `InvalidPeer` if unknown.
242    pub fn round_trip_time(&self, peer_id: PeerID) -> Result<Duration, NetworkError> {
243        let host = self.host()?;
244        let peer = host
245            .get_peer(peer_id)
246            .ok_or(NetworkError::InvalidPeer(peer_id.0))?;
247        Ok(peer.round_trip_time())
248    }
249    /// Return the ENet connection state of the peer as a static string label.
250    pub fn peer_state(&self, peer_id: PeerID) -> Result<&'static str, NetworkError> {
251        let host = self.host()?;
252        let peer = host
253            .get_peer(peer_id)
254            .ok_or(NetworkError::InvalidPeer(peer_id.0))?;
255        let state_str = match peer.state() {
256            enet::PeerState::Disconnected => "disconnected",
257            enet::PeerState::Connecting => "connecting",
258            enet::PeerState::AcknowledgingConnect => "acknowledging_connect",
259            enet::PeerState::ConnectionPending => "connection_pending",
260            enet::PeerState::ConnectionSucceeded => "connection_succeeded",
261            enet::PeerState::Connected => "connected",
262            enet::PeerState::DisconnectLater => "disconnect_later",
263            enet::PeerState::Disconnecting => "disconnecting",
264            enet::PeerState::AcknowledgingDisconnect => "acknowledging_disconnect",
265            enet::PeerState::Zombie => "zombie",
266        };
267        Ok(state_str)
268    }
269    /// Return the remote socket address of the peer, or `None` if the address is not available.
270    pub fn peer_address(&self, peer_id: PeerID) -> Result<Option<SocketAddr>, NetworkError> {
271        let host = self.host()?;
272        let peer = host
273            .get_peer(peer_id)
274            .ok_or(NetworkError::InvalidPeer(peer_id.0))?;
275        Ok(peer.address())
276    }
277    /// Return the local socket address this host is bound to.
278    pub fn local_address(&self) -> SocketAddr {
279        self.local_addr
280    }
281    /// Return the maximum number of peer slots configured for this host.
282    pub fn peer_limit(&self) -> Result<usize, NetworkError> {
283        let host = self.host()?;
284        Ok(host.peer_limit())
285    }
286    /// Return the current per-peer channel count limit.
287    pub fn channel_limit(&self) -> Result<usize, NetworkError> {
288        let host = self.host()?;
289        Ok(host.channel_limit())
290    }
291    /// Set the per-peer channel count limit; returns `Enet` error if rejected.
292    pub fn set_channel_limit(&mut self, limit: usize) -> Result<(), NetworkError> {
293        let host = self.host_mut()?;
294        host.set_channel_limit(limit)
295            .map_err(|e| NetworkError::Enet(format!("{e}")))
296    }
297    /// Return the current `(incoming, outgoing)` bandwidth limits in bytes/sec; `None` means unlimited.
298    pub fn bandwidth_limit(&self) -> Result<(Option<u32>, Option<u32>), NetworkError> {
299        let host = self.host()?;
300        Ok(host.bandwidth_limit())
301    }
302    /// Set the incoming and outgoing bandwidth limits in bytes/sec; `None` removes the limit.
303    pub fn set_bandwidth_limit(
304        &mut self,
305        incoming: Option<u32>,
306        outgoing: Option<u32>,
307    ) -> Result<(), NetworkError> {
308        let host = self.host_mut()?;
309        host.set_bandwidth_limit(incoming, outgoing)
310            .map_err(|e| NetworkError::Enet(format!("{e}")))
311    }
312    /// Return the number of currently connected peers.
313    pub fn connected_peer_count(&mut self) -> Result<usize, NetworkError> {
314        let host = self.host_mut()?;
315        Ok(host.connected_peers().count())
316    }
317    /// Drop the inner ENet host, releasing the socket; subsequent calls to methods return `HostDestroyed`.
318    pub fn destroy(&mut self) {
319        self.inner = None;
320    }
321    /// Return `true` if `destroy()` has been called and the host is no longer usable.
322    pub fn is_destroyed(&self) -> bool {
323        self.inner.is_none()
324    }
325    /// Return a list of `PeerID` values for all currently connected peers.
326    pub fn connected_peer_ids(&mut self) -> Result<Vec<PeerID>, NetworkError> {
327        let host = self.host_mut()?;
328        Ok(host.connected_peers().map(|p| p.id()).collect())
329    }
330    /// Convenience constructor that binds a server on `0.0.0.0:<port>` with `Server` role.
331    pub fn create_server(
332        port: u16,
333        max_peers: Option<usize>,
334        channels: Option<usize>,
335    ) -> Result<Self, NetworkError> {
336        let addr_str = format!("0.0.0.0:{}", port);
337        let addr: SocketAddr = addr_str
338            .parse()
339            .map_err(|e: std::net::AddrParseError| NetworkError::InvalidAddress(e.to_string()))?;
340        let mut host = Self::new(
341            addr,
342            max_peers.or(Some(DEFAULT_PEERS)),
343            channels,
344            None,
345            None,
346        )?;
347        host.role = HostRole::Server;
348        Ok(host)
349    }
350    /// Convenience constructor that binds an ephemeral port, connects to `address`, and sets `Client` role.
351    pub fn create_client(
352        address: SocketAddr,
353        channels: Option<usize>,
354        data: Option<u32>,
355    ) -> Result<Self, NetworkError> {
356        let bind_addr: SocketAddr = "0.0.0.0:0".parse().unwrap();
357        let mut host = Self::new(bind_addr, Some(1), channels, None, None)?;
358        host.role = HostRole::Client;
359        let _ = host.connect(
360            address,
361            channels.unwrap_or(DEFAULT_CHANNELS),
362            data.unwrap_or(0),
363        )?;
364        Ok(host)
365    }
366    /// Return the current `HostRole` of this host.
367    pub fn role(&self) -> HostRole {
368        self.role
369    }
370    /// Override the host role; used when role is determined after creation.
371    pub fn set_role(&mut self, role: HostRole) {
372        self.role = role;
373    }
374    /// Collect and return a `PeerStats` snapshot for the given peer; returns `InvalidPeer` if unknown.
375    pub fn peer_stats(&self, peer_id: PeerID) -> Result<PeerStats, NetworkError> {
376        let host = self.host()?;
377        let peer = host
378            .get_peer(peer_id)
379            .ok_or(NetworkError::InvalidPeer(peer_id.0))?;
380        Ok(PeerStats {
381            round_trip_time: peer.round_trip_time().as_millis() as u32,
382            round_trip_time_variance: peer.round_trip_time_variance().as_millis() as u32,
383            packets_sent: peer.packets_sent(),
384            packets_lost: peer.packets_lost(),
385            packet_loss: peer.packet_loss(),
386            incoming_bandwidth: peer.incoming_bandwidth(),
387            outgoing_bandwidth: peer.outgoing_bandwidth(),
388            incoming_data_total: peer.incoming_data_total(),
389            outgoing_data_total: peer.outgoing_data_total(),
390        })
391    }
392}
393/// Point-in-time network statistics snapshot for a single ENet peer.
394pub struct PeerStats {
395    /// Smoothed round-trip time in milliseconds.
396    pub round_trip_time: u32,
397    /// Variance of the round-trip time measurement in milliseconds.
398    pub round_trip_time_variance: u32,
399    /// Total packets sent to this peer since connection.
400    pub packets_sent: u32,
401    /// Total packets lost in transit to this peer since connection.
402    pub packets_lost: u32,
403    /// Fixed-point packet loss ratio (0 = no loss, 65536 = 100% loss).
404    pub packet_loss: u32,
405    /// Negotiated incoming bandwidth limit in bytes/sec (0 = unlimited).
406    pub incoming_bandwidth: u32,
407    /// Negotiated outgoing bandwidth limit in bytes/sec (0 = unlimited).
408    pub outgoing_bandwidth: u32,
409    /// Total bytes received from this peer since connection.
410    pub incoming_data_total: u32,
411    /// Total bytes sent to this peer since connection.
412    pub outgoing_data_total: u32,
413}
