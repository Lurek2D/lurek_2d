1//! LAN lobby discovery via timed UDP broadcast listen on a fixed port.
2//!
3//! - Lobby advertisement encoding and parsing in a `key=value;...` wire format.
4//! - In-process room registry for create, join, leave, and list operations.
5//! - Broadcast helper that sends a single SO_BROADCAST datagram on all interfaces.
6//! - Deduplication of discovered lobbies by host+port during the scan window.
7
8use std::net::{Ipv4Addr, SocketAddr, UdpSocket};
9use std::sync::{Mutex, OnceLock};
10use std::time::{Duration, Instant};
11/// UDP port used for LAN lobby broadcast and discovery.
12pub const LOBBY_PORT: u16 = 47_777;
13/// Lobby advertisement received via LAN UDP broadcast.
14#[derive(Debug, Clone, PartialEq)]
15pub struct LobbyInfo {
16    /// Human-readable lobby name set by the host.
17    pub name: String,
18    /// IP address or hostname string of the hosting machine.
19    pub host: String,
20    /// ENet port the host is listening on.
21    pub port: u16,
22    /// Current number of connected players.
23    pub player_count: u32,
24    /// Maximum players allowed before the lobby is full.
25    pub max_players: u32,
26}
27impl LobbyInfo {
28    /// Encode this `LobbyInfo` as a `key=value;...` wire string for UDP broadcast.
29    pub fn to_wire(&self) -> String {
30        format!(
31            "name={};host={};port={};players={};max={}",
32            self.name, self.host, self.port, self.player_count, self.max_players
33        )
34    }
35    /// Parse a wire string back into a `LobbyInfo`; uses `sender` IP when `host` field is absent; returns `None` on malformed input.
36    pub fn from_wire(s: &str, sender: SocketAddr) -> Option<Self> {
37        let mut name = None;
38        let mut host = None;
39        let mut port = None;
40        let mut player_count = None;
41        let mut max_players = None;
42        for kv in s.split(';') {
43            let mut parts = kv.splitn(2, '=');
44            let key = parts.next()?.trim();
45            let val = parts.next()?.trim();
46            match key {
47                "name" => name = Some(val.to_string()),
48                "host" => host = Some(val.to_string()),
49                "port" => port = val.parse::<u16>().ok(),
50                "players" => player_count = val.parse::<u32>().ok(),
51                "max" => max_players = val.parse::<u32>().ok(),
52                _ => {}
53            }
54        }
55        Some(Self {
56            name: name?,
57            host: host.unwrap_or_else(|| sender.ip().to_string()),
58            port: port?,
59            player_count: player_count.unwrap_or(0),
60            max_players: max_players.unwrap_or(0),
61        })
62    }
63}
64/// Send a single UDP broadcast on the LAN announcing `info`; returns an error string on socket failure.
65pub fn broadcast_lobby(info: &LobbyInfo) -> Result<(), String> {
66    let socket = UdpSocket::bind("0.0.0.0:0").map_err(|e| format!("lobby broadcast bind: {e}"))?;
67    socket
68        .set_broadcast(true)
69        .map_err(|e| format!("lobby broadcast SO_BROADCAST: {e}"))?;
70    let payload = info.to_wire();
71    let addr = SocketAddr::new(Ipv4Addr::BROADCAST.into(), LOBBY_PORT);
72    socket
73        .send_to(payload.as_bytes(), addr)
74        .map_err(|e| format!("lobby broadcast send: {e}"))?;
75    Ok(())
76}
77/// Listen on `LOBBY_PORT` for up to `timeout_ms` (capped at 5000) ms and return all unique lobbies found.
78pub fn discover_lobbies(timeout_ms: u64) -> Vec<LobbyInfo> {
79    let deadline = Duration::from_millis(timeout_ms.min(5_000));
80    let Ok(socket) = UdpSocket::bind(SocketAddr::new(Ipv4Addr::UNSPECIFIED.into(), LOBBY_PORT))
81    else {
82        return Vec::new();
83    };
84    if socket
85        .set_read_timeout(Some(Duration::from_millis(50)))
86        .is_err()
87    {
88        return Vec::new();
89    }
90    let mut results: Vec<LobbyInfo> = Vec::new();
91    let start = Instant::now();
92    let mut buf = [0u8; 512];
93    loop {
94        if start.elapsed() >= deadline {
95            break;
96        }
97        match socket.recv_from(&mut buf) {
98            Ok((len, sender)) => {
99                if let Ok(s) = std::str::from_utf8(&buf[..len]) {
100                    if let Some(info) = LobbyInfo::from_wire(s, sender) {
101                        let key = (info.host.clone(), info.port);
102                        if !results.iter().any(|r| (r.host.clone(), r.port) == key) {
103                            results.push(info);
104                        }
105                    }
106                }
107            }
108            Err(ref e)
109                if e.kind() == std::io::ErrorKind::WouldBlock
110                    || e.kind() == std::io::ErrorKind::TimedOut => {}
111            Err(_) => break,
112        }
113    }
114    results
115}
116/// Room advertisement stored in the in-process registry; used when a relay server is not available.
117#[derive(Debug, Clone, PartialEq)]
118pub struct RoomInfo {
119    /// Unique room identifier assigned at creation (e.g. `"room-1"`).
120    pub id: String,
121    /// Human-readable room name.
122    pub name: String,
123    /// IP or hostname of the room host.
124    pub host: String,
125    /// Current number of players in the room.
126    pub player_count: u32,
127    /// Maximum allowed players; at least 1.
128    pub max_players: u32,
129}
130/// Process-global in-memory store for all announced rooms.
131#[derive(Debug, Default)]
132struct RoomRegistry {
133    /// Ordered list of all rooms created in this process.
134    rooms: Vec<RoomInfo>,
135}
136/// Return the process-global `RoomRegistry` mutex, initialising it on first call.
137fn rooms() -> &'static Mutex<RoomRegistry> {
138    static REGISTRY: OnceLock<Mutex<RoomRegistry>> = OnceLock::new();
139    REGISTRY.get_or_init(|| Mutex::new(RoomRegistry::default()))
140}
141/// Create a room entry in the global registry and return a clone of the new `RoomInfo`.
142pub fn create_room(name: &str, host: &str, max_players: u32) -> RoomInfo {
143    let mut reg = rooms().lock().expect("room registry poisoned");
144    let id = format!("room-{}", reg.rooms.len() + 1);
145    let room = RoomInfo {
146        id,
147        name: name.to_string(),
148        host: host.to_string(),
149        player_count: 1,
150        max_players: max_players.max(1),
151    };
152    reg.rooms.push(room.clone());
153    room
154}
155/// Return a snapshot of all rooms currently in the global registry.
156pub fn list_rooms() -> Vec<RoomInfo> {
157    rooms()
158        .lock()
159        .expect("room registry poisoned")
160        .rooms
161        .clone()
162}
163/// Increment the player count for the given room and return the updated `RoomInfo`; returns `None` when full or not found.
164pub fn join_room(id: &str) -> Option<RoomInfo> {
165    let mut reg = rooms().lock().expect("room registry poisoned");
166    let room = reg.rooms.iter_mut().find(|r| r.id == id)?;
167    if room.player_count >= room.max_players {
168        return None;
169    }
170    room.player_count += 1;
171    Some(room.clone())
172}
173/// Decrement the player count for the given room and return the updated `RoomInfo`; returns `None` when not found.
174pub fn leave_room(id: &str) -> Option<RoomInfo> {
175    let mut reg = rooms().lock().expect("room registry poisoned");
176    let room = reg.rooms.iter_mut().find(|r| r.id == id)?;
177    if room.player_count > 0 {
178        room.player_count -= 1;
179    }
180    Some(room.clone())
181}
