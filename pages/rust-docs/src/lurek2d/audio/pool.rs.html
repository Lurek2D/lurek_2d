1//! `SoundPool` round-robin polyphonic voice pool for one-shot playback of a single sound asset.
2//!
3//! - Preloaded `SoundKey` voices cycled via `next_voice` for low-latency triggering.
4//! - Per-pool volume multiplier and optional bus routing assignment.
5//! - Validity check ensuring at least one voice is available.
6
7use crate::runtime::resource_keys::SoundKey;
8/// Round-robin pool of preloaded source keys representing voices for one sound asset.
9///
10/// # Fields
11pub struct SoundPool {
12    /// Source keys used as available voices.
13    keys: Vec<SoundKey>,
14    /// Next voice index used by `next_voice`.
15    next: usize,
16    /// Original source file path associated with these voices.
17    file_path: String,
18    /// Per-pool gain multiplier applied when triggering a voice.
19    volume: f32,
20    /// Optional target bus name for routed playback.
21    bus_name: Option<String>,
22}
23impl SoundPool {
24    /// Create a pool with `keys` and source `file_path`, defaulting to volume=1.0 and no bus.
25    pub fn new(keys: Vec<SoundKey>, file_path: String) -> Self {
26        Self {
27            keys,
28            next: 0,
29            file_path,
30            volume: 1.0,
31            bus_name: None,
32        }
33    }
34    /// Return number of voices in this pool.
35    pub fn voice_count(&self) -> usize {
36        self.keys.len()
37    }
38    /// Return source file path associated with this pool.
39    pub fn file_path(&self) -> &str {
40        &self.file_path
41    }
42    /// Return current pool gain multiplier.
43    pub fn volume(&self) -> f32 {
44        self.volume
45    }
46    /// Set pool gain multiplier; values below 0.0 are clamped to 0.0.
47    pub fn set_volume(&mut self, vol: f32) {
48        self.volume = vol.max(0.0);
49    }
50    /// Return assigned bus name, or `None` if unassigned.
51    pub fn bus_name(&self) -> Option<&str> {
52        self.bus_name.as_deref()
53    }
54    /// Assign this pool to bus `name`.
55    pub fn set_bus(&mut self, name: &str) {
56        self.bus_name = Some(name.to_owned());
57    }
58    /// Remove any bus assignment from this pool.
59    pub fn clear_bus(&mut self) {
60        self.bus_name = None;
61    }
62    /// Return next voice key in round-robin order and advance the cursor.
63    pub fn next_voice(&mut self) -> SoundKey {
64        let key = self.keys[self.next % self.keys.len()];
65        self.next = (self.next + 1) % self.keys.len();
66        key
67    }
68    /// Return all voice keys managed by this pool.
69    pub fn all_keys(&self) -> &[SoundKey] {
70        &self.keys
71    }
72    /// Return `true` when the pool contains at least one voice key.
73    pub fn is_valid(&self) -> bool {
74        !self.keys.is_empty()
75    }
76}
