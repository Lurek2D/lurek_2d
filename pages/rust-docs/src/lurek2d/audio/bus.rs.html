1//! Named audio routing bus with per-bus volume, pitch, pause, and duck-target controls.
2//!
3//! - Shared DSP effect chain stored as `Arc<RwLock<Vec<Arc<EffectParams>>>>` for lock-free audio-thread reads.
4//! - Duck-target assignment enabling automatic cross-bus volume suppression.
5//! - Boundary clamping on volume, pitch, and duck volume values.
6
7use crate::log_msg;
8use crate::runtime::log_messages::{BU01, BU02, BU03};
9#[derive(Debug, Clone)]
10/// Named audio routing channel that applies volume, pitch, effects, and ducking to its sources.
11///
12/// # Fields
13pub struct Bus {
14    /// Human-readable identifier used as the registry key in `Mixer`.
15    name: String,
16    /// Volume multiplier applied to all sources on this bus; clamped to >= 0.0.
17    volume: f32,
18    /// Pitch multiplier applied to all sources on this bus; clamped to >= 0.0.
19    pitch: f32,
20    /// When true, all sources assigned to this bus are suspended from playback.
21    paused: bool,
22    /// Shared DSP effect chain applied by `DynamicEffectSource` on every source in this bus.
23    pub effects: std::sync::Arc<std::sync::RwLock<Vec<std::sync::Arc<crate::dsp::EffectParams>>>>,
24    /// Optional duck target: `(bus_name, duck_volume)` to suppress when this bus is active.
25    pub duck_target: Option<(String, f32)>,
26}
27impl Bus {
28    /// Create a new bus with the given name, volume=1.0, pitch=1.0, unpaused, and no effects.
29    pub fn new(name: impl Into<String>) -> Self {
30        let name = name.into();
31        log_msg!(debug, BU01, "{}", name);
32        Bus {
33            name,
34            volume: 1.0,
35            pitch: 1.0,
36            paused: false,
37            effects: std::sync::Arc::new(std::sync::RwLock::new(Vec::new())),
38            duck_target: None,
39        }
40    }
41    /// Return the name of this bus. This function is part of the public API.
42    pub fn name(&self) -> &str {
43        &self.name
44    }
45    /// Return the current volume multiplier for this bus.
46    pub fn volume(&self) -> f32 {
47        self.volume
48    }
49    /// Set the volume multiplier; values below 0.0 are clamped to 0.0.
50    pub fn set_volume(&mut self, volume: f32) {
51        self.volume = volume.max(0.0);
52    }
53    /// Return the current pitch multiplier for this bus.
54    pub fn pitch(&self) -> f32 {
55        self.pitch
56    }
57    /// Set the pitch multiplier; values below 0.0 are clamped to 0.0.
58    pub fn set_pitch(&mut self, pitch: f32) {
59        self.pitch = pitch.max(0.0);
60    }
61    /// Pause all sources on this bus; no-op if already paused.
62    pub fn pause(&mut self) {
63        log_msg!(debug, BU02, "{}", self.name);
64        self.paused = true;
65    }
66    /// Resume all sources on this bus; no-op if already playing.
67    pub fn resume(&mut self) {
68        log_msg!(debug, BU03, "{}", self.name);
69        self.paused = false;
70    }
71    /// Return `true` when this bus is paused.
72    pub fn is_paused(&self) -> bool {
73        self.paused
74    }
75    /// Set the duck target bus name and duck volume; volume is clamped to 0.0..=1.0.
76    pub fn set_duck_target(&mut self, target_bus_name: impl Into<String>, duck_volume: f32) {
77        log_msg!(debug, BU02, "{}", duck_volume);
78        self.duck_target = Some((target_bus_name.into(), duck_volume.clamp(0.0, 1.0)));
79    }
80    /// Remove configured duck target from this bus.
81    pub fn clear_duck_target(&mut self) {
82        log_msg!(debug, BU03, "duck target cleared");
83        self.duck_target = None;
84    }
85}
