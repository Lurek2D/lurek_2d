1//! Mod registry: register, unregister, and look up mods by id or capability.
2//!
3//! - Manifest parsing: load `mod.toml` files, validate fields, and compute SHA-256 signatures.
4//! - Load ordering: topological sort with dependency resolution, priority tie-breaking, and custom override.
5//! - Asset conflict detection: prevent two mods from declaring the same asset path.
6//! - Hot-reload queue: mark mods dirty, re-parse their manifests, and re-register atomically.
7//! - Folder scanning: discover mod directories on disk and batch-register valid entries.
8//! - Dependency validation: detect missing deps and circular dependency cycles.
9//! - Config schema: carry typed key/default triples from manifests for runtime config UI.
10
11use crate::log_msg;

13use sha2::{Digest, Sha256};

15use std::collections::{BinaryHeap, HashMap, HashSet};
16use std::path::Path;
17/// Metadata and runtime state for a single content mod.
18#[derive(Debug, Clone)]
19pub struct ModInfo {
20    /// Unique mod identifier; used as the canonical key everywhere.
21    pub id: String,
22    /// Human-readable display name.
23    pub name: String,
24    /// SemVer-style version string from the manifest.
25    pub version: String,
26    /// Author name from the manifest.
27    pub author: String,
28    /// Short description from the manifest.
29    pub description: String,
30    /// Load priority; lower values load earlier when there are no dependency constraints.
31    pub priority: i32,
32    /// Ids of other mods that must load before this one.
33    pub dependencies: Vec<String>,
34    /// Whether this mod is allowed to participate in the load order.
35    pub enabled: bool,
36    /// Whether the mod's Lua entry point has been executed this session.
37    pub loaded: bool,
38    /// Absolute path to the `mod.toml` manifest file, when loaded from disk.
39    pub path: Option<String>,
40    /// Optional minimum engine API version string declared in the manifest.
41    pub api_version: Option<String>,
42    /// Feature capability tags declared in the manifest.
43    pub capabilities: Vec<String>,
44    /// Config schema entries as `(key, type_hint, default_value)` triples.
45    pub config_schema: Vec<(String, String, String)>,
46    /// Asset paths declared in the manifest; checked for conflicts during scan.
47    pub asset_paths: Vec<String>,
48    /// Optional SHA-256 hex signature of the manifest fields; verified on load.
49    pub signature: Option<String>,
50}

74    }
75    /// Create a `ModInfo` from explicit parts, falling back to defaults when `None` is supplied.
76    pub fn from_parts(
77        id: String,
78        name: Option<String>,
79        version: Option<String>,
80        author: Option<String>,
81        description: Option<String>,
82        priority: Option<i32>,
83        dependencies: Vec<String>,
84    ) -> Self {
85        let mut info = Self::new(id);
86        if let Some(n) = name {
87            info.name = n;
88        }
89        if let Some(v) = version {
90            info.version = v;
91        }
92        if let Some(a) = author {
93            info.author = a;
94        }
95        if let Some(d) = description {
96            info.description = d;
97        }
98        if let Some(p) = priority {
99            info.priority = p;
100        }
101        info.dependencies = dependencies;
102        info
103    }

105    /// Checks whether the mod's declared API version is compatible with the host engine version.
106    pub fn check_api_version(&self, host_version: &str) -> Result<(), String> {
107        let required = match &self.api_version {
108            None => return Ok(()),
109            Some(v) => v,
110        };
111        let parse = |s: &str| -> Option<(u32, u32, u32)> {
112            let mut parts = s.splitn(3, '.');
113            let maj = parts.next()?.parse::<u32>().ok()?;
114            let min = parts.next()?.parse::<u32>().ok()?;
115            let pat = parts
116                .next()
117                .and_then(|p| p.parse::<u32>().ok())
118                .unwrap_or(0);
119            Some((maj, min, pat))
120        };
121        let (req_maj, req_min, _) = match parse(required) {
122            Some(v) => v,
123            None => return Err(format!("mod api_version '{}' is not a valid semver", required)),
124        };
125        let (host_maj, host_min, _) = match parse(host_version) {
126            Some(v) => v,
127            None => return Err(format!("host api_version '{}' is not a valid semver", host_version)),
128        };
129        if req_maj != host_maj {
130            return Err(format!("mod requires API {}.x but host provides {}.x", req_maj, host_maj));
131        }
132        if req_min > host_min {
133            return Err(format!("mod requires API {}.{}.x but host provides {}.{}.x", req_maj, req_min, host_maj, host_min));
134        }
135        Ok(())
136    }
137}
138/// Registry of all known mods, their custom load order, and pending hot-reload requests.
139#[derive(Debug, Clone, Default)]
140pub struct ModManager {
141    /// Flat list of all registered `ModInfo` entries.
142    mods: Vec<ModInfo>,
143    /// Optional explicit load order overriding priority and dependency sort.
144    custom_load_order: Option<Vec<String>>,
145    /// Ordered queue of mod ids pending a hot-reload.
146    reload_queue: Vec<String>,
147    /// Set mirror of `reload_queue` for O(1) duplicate prevention.
148    reload_queue_set: HashSet<String>,
149}
150/// Mod lifecycle operations: registration, ordering, scanning, reload, and validation.
151impl ModManager {
152    /// Create an empty mod manager; logs MD01.
153    pub fn new() -> Self {
154        log_msg!(debug, MD01_MGR_INIT);
155        Self {
156            mods: Vec::new(),
157            custom_load_order: None,
158            reload_queue: Vec::new(),
159            reload_queue_set: HashSet::new(),
160        }
161    }
162    /// Insert or replace `info` by id; removes it from the reload queue if present.
163    pub fn register_mod(&mut self, info: ModInfo) {
164        self.reload_queue.retain(|queued| queued != &info.id);
165        self.reload_queue_set.remove(info.id.as_str());
166        if let Some(pos) = self.mods.iter().position(|m| m.id == info.id) {
167            self.mods[pos] = info;
168        } else {
169            self.mods.push(info);
170        }
171    }
172    /// Remove the mod with `id`; returns true when a mod was actually removed.
173    pub fn unregister_mod(&mut self, id: &str) -> bool {
174        if let Some(pos) = self.mods.iter().position(|m| m.id == id) {
175            self.mods.remove(pos);
176            self.reload_queue.retain(|q| q != id);
177            self.reload_queue_set.remove(id);
178            true
179        } else {
180            false
181        }
182    }
183    /// Return a shared reference to the mod with `id`, or `None` when not found.
184    pub fn get_mod(&self, id: &str) -> Option<&ModInfo> {
185        self.mods.iter().find(|m| m.id == id)
186    }
187    /// Return a mutable reference to the mod with `id`, or `None` when not found.
188    pub fn get_mod_mut(&mut self, id: &str) -> Option<&mut ModInfo> {
189        self.mods.iter_mut().find(|m| m.id == id)
190    }
191    /// Return true when a mod with `id` is registered.
192    pub fn has_mod(&self, id: &str) -> bool {
193        self.mods.iter().any(|m| m.id == id)
194    }
195    /// Return the total number of registered mods.
196    pub fn mod_count(&self) -> usize {
197        self.mods.len()
198    }
199    /// Return the full slice of registered mods in insertion order.
200    pub fn all_mods(&self) -> &[ModInfo] {
201        &self.mods
202    }
203    /// Return all mods that declare `capability` in their capabilities list.
204    pub fn get_mods_by_capability(&self, capability: &str) -> Vec<&ModInfo> {
205        self.mods
206            .iter()
207            .filter(|mod_info| mod_info.capabilities.iter().any(|cap| cap == capability))
208            .collect()
209    }
210    /// Return mods in their effective load order: custom order, then topological by priority; logs MD04.
211    pub fn load_order(&self) -> Vec<&ModInfo> {
212        log_msg!(debug, MD04_ORDER_OK);
213        if let Some(order) = &self.custom_load_order {
214            let mut result: Vec<&ModInfo> = Vec::new();
215            let mut seen: HashSet<&str> = HashSet::new();
216            for id in order {
217                if let Some(m) = self.mods.iter().find(|m| &m.id == id) {
218                    result.push(m);
219                    seen.insert(m.id.as_str());
220                }
221            }
222            let mut remainder: Vec<&ModInfo> = self
223                .mods
224                .iter()
225                .filter(|m| !seen.contains(m.id.as_str()))
226                .collect();
227            remainder = Self::sort_by_priority(remainder);
228            result.extend(remainder);
229            result
230        } else {
231            match self.topological_order() {
232                Ok(sorted) => sorted,
233                Err(cycle_ids) => {
234                    ::log::warn!("mod dependency cycle detected: {:?}", cycle_ids);
235                    Self::sort_by_priority(self.mods.iter().collect())
236                }
237            }
238        }
239    }
240    /// Sort `mods` by ascending priority then lexicographic id.
241    fn sort_by_priority(mods: Vec<&ModInfo>) -> Vec<&ModInfo> {
242        let mut sorted = mods;
243        sorted.sort_by(|a, b| a.priority.cmp(&b.priority).then(a.id.cmp(&b.id)));
244        sorted
245    }
246    /// Produce a topological load order respecting dependencies; returns `Err` with cycle ids on failure.
247    fn topological_order(&self) -> Result<Vec<&ModInfo>, Vec<String>> {
248        let mut index_by_id: HashMap<&str, usize> = HashMap::new();
249        for (index, info) in self.mods.iter().enumerate() {
250            index_by_id.insert(info.id.as_str(), index);
251        }
252        let mut indegree = vec![0usize; self.mods.len()];
253        let mut outgoing = vec![Vec::<usize>::new(); self.mods.len()];
254        for (index, info) in self.mods.iter().enumerate() {
255            for dependency in &info.dependencies {
256                if let Some(&dependency_index) = index_by_id.get(dependency.as_str()) {
257                    outgoing[dependency_index].push(index);
258                    indegree[index] += 1;
259                }
260            }
261        }
262        let mut ready: BinaryHeap<Reverse<(i32, String, usize)>> = BinaryHeap::new();
263        for (index, info) in self.mods.iter().enumerate() {
264            if indegree[index] == 0 {
265                ready.push(Reverse((info.priority, info.id.clone(), index)));
266            }
267        }
268        let mut ordered = Vec::with_capacity(self.mods.len());
269        while let Some(Reverse((_priority, _id, index))) = ready.pop() {
270            ordered.push(&self.mods[index]);
271            for &dependent in &outgoing[index] {
272                if indegree[dependent] > 0 {
273                    indegree[dependent] -= 1;
274                    if indegree[dependent] == 0 {
275                        let info = &self.mods[dependent];
276                        ready.push(Reverse((info.priority, info.id.clone(), dependent)));
277                    }
278                }
279            }
280        }
281        if ordered.len() != self.mods.len() {
282            let cycle_ids = self
283                .mods
284                .iter()
285                .enumerate()
286                .filter(|(index, _)| indegree[*index] > 0)
287                .map(|(_, info)| info.id.clone())
288                .collect();
289            Err(cycle_ids)
290        } else {
291            Ok(ordered)
292        }
293    }
294    /// Compute a deterministic SHA-256 hex signature over the public manifest fields of `info`.
295    fn manifest_signature(info: &ModInfo) -> String {
296        let mut hasher = Sha256::new();
297        hasher.update(info.id.as_bytes());
298        hasher.update([0]);
299        hasher.update(info.name.as_bytes());
300        hasher.update([0]);
301        hasher.update(info.version.as_bytes());
302        hasher.update([0]);
303        hasher.update(info.author.as_bytes());
304        hasher.update([0]);
305        hasher.update(info.description.as_bytes());
306        hasher.update([0]);
307        hasher.update(info.priority.to_le_bytes());
308        hasher.update([0]);
309        for dependency in &info.dependencies {
310            hasher.update(dependency.as_bytes());
311            hasher.update([0]);
312        }
313        for capability in &info.capabilities {
314            hasher.update(capability.as_bytes());
315            hasher.update([0]);
316        }
317        for asset_path in &info.asset_paths {
318            hasher.update(asset_path.as_bytes());
319            hasher.update([0]);
320        }
321        for (key, type_hint, default) in &info.config_schema {
322            hasher.update(key.as_bytes());
323            hasher.update([0]);
324            hasher.update(type_hint.as_bytes());
325            hasher.update([0]);
326            hasher.update(default.as_bytes());
327            hasher.update([0]);
328        }
329        hex::encode(hasher.finalize())
330    }
331    /// Return `(asset_path, conflicting_mod_id)` pairs for any asset paths `info` shares with registered mods.
332    fn manifest_conflicts(&self, info: &ModInfo) -> Vec<(String, String)> {
333        let mut conflicts = Vec::new();
334        for asset_path in &info.asset_paths {
335            for existing in &self.mods {
336                if existing.id != info.id
337                    && existing.asset_paths.iter().any(|path| path == asset_path)
338                {
339                    conflicts.push((asset_path.clone(), existing.id.clone()));
340                    break;
341                }
342            }
343        }
344        conflicts
345    }
346    /// Parse a `mod.toml` string into a `ModInfo` plus a list of non-fatal warnings; return error on fatal failures.
347    fn parse_manifest(entry_path: &Path, content: &str) -> Result<(ModInfo, Vec<String>), String> {
348        let value: toml::Value = content
349            .parse()
350            .map_err(|err| format!("{}: invalid TOML: {}", entry_path.display(), err))?;
351        let table = value.as_table().ok_or_else(|| {
352            format!(
353                "{}: mod.toml must contain a TOML table",
354                entry_path.display()
355            )
356        })?;
357        let id = table
358            .get("id")
359            .and_then(|value| value.as_str())
360            .ok_or_else(|| {
361                format!(
362                    "{}: missing required string field 'id'",
363                    entry_path.display()
364                )
365            })?
366            .to_string();
367        let dependencies = table
368            .get("dependencies")
369            .and_then(|value| value.as_array())
370            .map(|items| {
371                items
372                    .iter()
373                    .filter_map(|item| item.as_str().map(str::to_string))
374                    .collect()
375            })
376            .unwrap_or_default();
377        let mut info = ModInfo::from_parts(
378            id,
379            table
380                .get("name")
381                .and_then(|value| value.as_str())
382                .map(str::to_string),
383            table
384                .get("version")
385                .and_then(|value| value.as_str())
386                .map(str::to_string),
387            table
388                .get("author")
389                .and_then(|value| value.as_str())
390                .map(str::to_string),
391            table
392                .get("description")
393                .and_then(|value| value.as_str())
394                .map(str::to_string),
395            table
396                .get("priority")
397                .and_then(|value| value.as_integer())
398                .map(|value| value as i32),
399            dependencies,
400        );
401        info.path = Some(entry_path.to_string_lossy().into_owned());
402        let mut warnings = Vec::new();
403        if let Some(items) = table.get("capabilities").and_then(|value| value.as_array()) {
404            info.capabilities = items
405                .iter()
406                .filter_map(|item| item.as_str().map(str::to_string))
407                .collect();
408        }
409        if let Some(items) = table.get("assets").and_then(|value| value.as_array()) {
410            info.asset_paths = items
411                .iter()
412                .filter_map(|item| item.as_str().map(str::to_string))
413                .collect();
414        }
415        if let Some(signature) = table.get("signature").and_then(|value| value.as_str()) {
416            info.signature = Some(signature.to_string());
417        }
418        if let Some(schema) = table
419            .get("config_schema")
420            .and_then(|value| value.as_array())
421        {
422            let mut parsed_schema = Vec::new();
423            for entry in schema {
424                let Some(entry_table) = entry.as_table() else {
425                    warnings.push(format!(
426                        "{}: ignored non-table config_schema entry",
427                        entry_path.display()
428                    ));
429                    continue;
430                };
431                let Some(key) = entry_table.get("key").and_then(|value| value.as_str()) else {
432                    warnings.push(format!(
433                        "{}: config_schema entry missing string key",
434                        entry_path.display()
435                    ));
436                    continue;
437                };
438                let type_hint = entry_table
439                    .get("type")
440                    .and_then(|value| value.as_str())
441                    .unwrap_or("any");
442                let default = entry_table
443                    .get("default")
444                    .map(|value| value.to_string())
445                    .unwrap_or_default();
446                parsed_schema.push((key.to_string(), type_hint.to_string(), default));
447            }
448            info.config_schema = parsed_schema;
449        }
450        if let Some(signature) = &info.signature {
451            let expected = Self::manifest_signature(&info);
452            if signature != &expected {
453                return Err(format!(
454                    "{}: signature mismatch for mod '{}'",
455                    entry_path.display(),
456                    info.id
457                ));
458            }
459        }
460        Ok((info, warnings))
461    }

474    /// Scan `path` for subdirectories with a `mod.toml`; registers valid mods and returns discovered `ModInfo` list.
475    pub fn scan_folder(&mut self, path: &str) -> Vec<ModInfo> {
476        let mut discovered = Vec::new();
477        let dir_iter = match std::fs::read_dir(path) {
478            Ok(iter) => iter,
479            Err(_) => return discovered,
480        };
481        for entry in dir_iter.flatten() {
482            let entry_path = entry.path();
483            if !entry_path.is_dir() {
484                continue;
485            }
486            let toml_path = entry_path.join("mod.toml");
487            if !toml_path.exists() {
488                continue;
489            }
490            let content = match std::fs::read_to_string(&toml_path) {
491                Ok(c) => c,
492                Err(err) => {
493                    ::log::warn!("{}: {}", toml_path.display(), err);
494                    continue;
495                }
496            };
497            let (info, warnings) = match Self::parse_manifest(&toml_path, &content) {
498                Ok(value) => value,
499                Err(err) => {
500                    ::log::warn!("{}", err);
501                    continue;
502                }
503            };
504            for warning in warnings {
505                ::log::warn!("{}", warning);
506            }
507            let conflicts = self.manifest_conflicts(&info);
508            if !conflicts.is_empty() {
509                ::log::warn!(
510                    "{}: asset path conflict(s) detected for mod '{}': {:?}",
511                    toml_path.display(),
512                    info.id,
513                    conflicts
514                );
515                continue;
516            }
517            self.register_mod(info.clone());
518            discovered.push(info);
519        }
520        discovered
521    }
522    /// Enqueue mod `id` for hot-reload; marks it unloaded; returns false when the id is unknown.
523    pub fn mark_for_reload(&mut self, id: &str) -> bool {
524        if !self.has_mod(id) {
525            return false;
526        }
527        if let Some(info) = self.get_mod_mut(id) {
528            info.loaded = false;
529        }
530        if self.reload_queue_set.insert(id.to_string()) {
531            self.reload_queue.push(id.to_string());
532        }
533        true
534    }
535    /// Return the pending reload queue in submission order.
536    pub fn get_reload_queue(&self) -> &[String] {
537        &self.reload_queue
538    }
539    /// Clear the reload queue without processing it.
540    pub fn clear_reload_queue(&mut self) {
541        self.reload_queue.clear();
542        self.reload_queue_set.clear();
543    }
544    /// Re-parse manifests for all queued mods and re-register them; return the ids that succeeded.
545    pub fn process_reload_queue(&mut self) -> Vec<String> {
546        let queued = std::mem::take(&mut self.reload_queue);
547        self.reload_queue_set.clear();
548        let mut reloaded = Vec::new();
549        for id in queued {
550            let Some(path) = self.get_mod(&id).and_then(|info| info.path.clone()) else {
551                ::log::warn!(
552                    "mod '{}' cannot be reloaded because it has no manifest path",
553                    id
554                );
555                continue;
556            };
557            let content = match std::fs::read_to_string(&path) {
558                Ok(content) => content,
559                Err(err) => {
560                    ::log::warn!("{}: {}", path, err);
561                    continue;
562                }
563            };
564            let (mut info, warnings) = match Self::parse_manifest(Path::new(&path), &content) {
565                Ok(value) => value,
566                Err(err) => {
567                    ::log::warn!("{}", err);
568                    continue;
569                }
570            };
571            for warning in warnings {
572                ::log::warn!("{}", warning);
573            }
574            info.loaded = true;
575            self.register_mod(info);
576            reloaded.push(id);
577        }
578        reloaded
579    }
580    /// Return the ids of dependencies declared by any mod that are not themselves registered.
581    pub fn validate_dependencies(&self) -> Vec<String> {
582        let ids: HashSet<&str> = self.mods.iter().map(|m| m.id.as_str()).collect();
583        let mut missing = Vec::new();
584        for m in &self.mods {
585            for dep in &m.dependencies {
586                if !ids.contains(dep.as_str()) && !missing.contains(dep) {
587                    missing.push(dep.clone());
588                }
589            }
590        }
591        missing
592    }
593    /// Return true when the registered mods contain a dependency cycle.
594    pub fn has_circular_dependencies(&self) -> bool {
595        self.topological_order().is_err()
596    }
