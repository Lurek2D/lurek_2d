1//! Owned mutable byte buffer with indexed read and write access
2//!
3//! - UTF-8 string encoding and lossy decoding from raw bytes
4//! - Immutable and mutable slice views for zero-copy downstream use
5
6#[derive(Debug, Clone)]
7/// Hold owned raw bytes with convenience conversion helpers.
8pub struct ByteData {
9    /// Store raw bytes for this buffer.
10    data: Vec<u8>,
11}
12impl ByteData {
13    /// Create zero-filled buffer and return new value.
14    pub fn new(size: usize) -> Self {
15        Self {
16            data: vec![0; size],
17        }
18    }
19    /// Wrap existing bytes and return new value.
20    pub fn from_bytes(bytes: Vec<u8>) -> Self {
21        Self { data: bytes }
22    }
23    /// Encode UTF-8 text bytes and return new value.
24    pub fn from_string(s: &str) -> Self {
25        Self {
26            data: s.as_bytes().to_vec(),
27        }
28    }
29    /// Return buffer length in bytes.
30    pub fn len(&self) -> usize {
31        self.data.len()
32    }
33    /// Return true when buffer has no bytes.
34    pub fn is_empty(&self) -> bool {
35        self.data.is_empty()
36    }
37    /// Read byte at offset and return optional value.
38    pub fn get_byte(&self, offset: usize) -> Option<u8> {
39        self.data.get(offset).copied()
40    }
41    /// Write byte at offset and return success flag.
42    pub fn set_byte(&mut self, offset: usize, value: u8) -> bool {
43        if offset < self.data.len() {
44            self.data[offset] = value;
45            true
46        } else {
47            false
48        }
49    }
50    /// Decode bytes as UTF-8 lossily and return string.
51    pub fn get_string(&self) -> String {
52        String::from_utf8_lossy(&self.data).to_string()
53    }
54    /// Return immutable byte slice view.
55    pub fn as_bytes(&self) -> &[u8] {
56        &self.data
57    }
58    /// Return mutable byte slice view.
59    pub fn as_bytes_mut(&mut self) -> &mut [u8] {
60        &mut self.data
61    }
62    /// Clone internal bytes and return copied buffer.
63    pub fn clone_data(&self) -> Self {
64        Self {
65            data: self.data.clone(),
66        }
67    }
68}
