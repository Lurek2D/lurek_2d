1//! Behavior tree data structure with Sequence, Selector, Parallel, Inverter, Repeat, and Leaf node kinds.
2//!
3//! - Builder API for allocating nodes, linking children, and setting the root.
4//! - Per-tick runtime state tracking running nodes and repeat counters.
5//! - Integer `NodeId` addressing; no heap indirection between parent and child.
6//! - Fully deterministic tick ordering: left-to-right child evaluation.
7
8/// Tick result returned by a behavior tree node.
9#[derive(Debug, Clone, Copy, PartialEq, Eq)]
10pub enum BtStatus {
11    /// Node completed successfully.
12    Success,
13    /// Node completed with failure.
14    Failure,
15    /// Node is still executing.
16    Running,
17}
18/// Integer identifier for a node within a `BehaviorTree`.
19pub type NodeId = u32;
20
21/// Structural type of a behavior tree node.
22#[derive(Debug, Clone)]
23pub enum NodeKind {
24    /// Run children left-to-right; succeed only if all succeed.
25    Sequence,
26    /// Run children left-to-right; succeed on first success.
27    Selector,
28    /// Run all children; succeed when at least `min_success` succeed.
29    Parallel { min_success: usize },
30    /// Invert the single child's result.
31    Inverter,
32    /// Repeat the single child up to `count` times.
33    Repeat { count: usize },
34    /// Terminal leaf identified by action `name`.
35    Leaf { name: String },
36}
37/// A single node in the behavior tree with its structural kind and child list.
38#[derive(Debug, Clone)]
39pub struct BtNode {
40    /// Unique identifier within the owning `BehaviorTree`.
41    pub id: NodeId,
42    /// Structural variant that determines tick semantics.
43    pub kind: NodeKind,
44    /// Ordered child node identifiers.
45    pub children: Vec<NodeId>,
46    /// Human-readable debug label.
47    pub label: String,
48}
49/// Builder for constructing a behavior tree by allocating nodes and linking children.
50#[derive(Debug, Clone, Default)]
51pub struct BehaviorTree {
52    /// Allocated nodes in insertion order.
53    nodes: Vec<BtNode>,
54    /// Next available identifier.
55    next_id: NodeId,
56    /// Optional root node identifier.
57    pub root: Option<NodeId>,
58}
59/// Construction and query methods for `BehaviorTree`.
60impl BehaviorTree {
61    /// Create an empty behavior tree.
62    pub fn new() -> Self {
63        Self::default()
64    }
65    /// Allocate a new node with `kind` and `label`; return its `NodeId`.
66    fn alloc(&mut self, kind: NodeKind, label: &str) -> NodeId {
67        let id = self.next_id;
68        self.next_id += 1;
69        self.nodes.push(BtNode {
70            id,
71            kind,
72            children: Vec::new(),
73            label: label.to_string(),
74        });
75        id
76    }
77    /// Add a Sequence node with `label`; return its `NodeId`.
78    pub fn add_sequence(&mut self, label: &str) -> NodeId {
79        self.alloc(NodeKind::Sequence, label)
80    }
81    /// Add a Selector node with `label`; return its `NodeId`.
82    pub fn add_selector(&mut self, label: &str) -> NodeId {
83        self.alloc(NodeKind::Selector, label)
84    }
85    /// Add a Parallel node requiring `min_success` successes; return its `NodeId`.
86    pub fn add_parallel(&mut self, min_success: usize, label: &str) -> NodeId {
87        self.alloc(NodeKind::Parallel { min_success }, label)
88    }
89    /// Add an Inverter node with `label`; return its `NodeId`.
90    pub fn add_inverter(&mut self, label: &str) -> NodeId {
91        self.alloc(NodeKind::Inverter, label)
92    }
93    /// Add a Repeat node that loops its child up to `count` times; return its `NodeId`.
94    pub fn add_repeat(&mut self, count: usize, label: &str) -> NodeId {
95        self.alloc(NodeKind::Repeat { count }, label)
96    }
97    /// Add a Leaf node with action `name` and debug `label`; return its `NodeId`.
98    pub fn add_leaf(&mut self, name: &str, label: &str) -> NodeId {
99        self.alloc(
100            NodeKind::Leaf {
101                name: name.to_string(),
102            },
103            label,
104        )
105    }
106    /// Attach `child_id` as the last child of `parent_id`; return false when either id is missing.
107    pub fn add_child(&mut self, parent_id: NodeId, child_id: NodeId) -> bool {
108        if !self.has_node(child_id) {
109            return false;
110        }
111        if let Some(p) = self.nodes.iter_mut().find(|n| n.id == parent_id) {
112            p.children.push(child_id);
113            true
114        } else {
115            false
116        }
117    }
118    /// Set the tree root to `id`; return false when `id` does not exist.
119    pub fn set_root(&mut self, id: NodeId) -> bool {
120        if self.has_node(id) {
121            self.root = Some(id);
122            true
123        } else {
124            false
125        }
126    }
127    /// Return true when a node with `id` exists in this tree.
128    pub fn has_node(&self, id: NodeId) -> bool {
129        self.nodes.iter().any(|n| n.id == id)
130    }
131    /// Return a reference to the node with `id`, or `None` if missing.
132    pub fn get_node(&self, id: NodeId) -> Option<&BtNode> {
133        self.nodes.iter().find(|n| n.id == id)
134    }
135    /// Return the total number of allocated nodes.
136    pub fn node_count(&self) -> usize {
137        self.nodes.len()
138    }
139    /// Return all node identifiers in insertion order.
140    pub fn node_ids(&self) -> Vec<NodeId> {
141        self.nodes.iter().map(|n| n.id).collect()
142    }
143    /// Remove all nodes and reset the root.
144    pub fn clear(&mut self) {
145        self.nodes.clear();
146        self.next_id = 0;
147        self.root = None;
148    }
149}
150/// Per-tick runtime state for an executing `BehaviorTree`.
151#[derive(Debug, Clone, Default)]
152pub struct BtRunState {
153    /// Nodes currently in `Running` state from the last tick.
154    pub running: std::collections::HashSet<NodeId>,
155    /// Per-node repeat counters for `Repeat` nodes.
156    pub repeat_counters: std::collections::HashMap<NodeId, usize>,
157}
158/// Methods for `BtRunState`.
159impl BtRunState {
160    /// Create an empty run state. This function is part of the public API.
161    pub fn new() -> Self {
162        Self::default()
163    }
164    /// Clear all running-node markers and repeat counters.
165    pub fn reset(&mut self) {
166        self.running.clear();
167        self.repeat_counters.clear();
168    }
169}
