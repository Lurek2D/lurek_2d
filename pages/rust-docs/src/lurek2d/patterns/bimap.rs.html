1//! Bidirectional map with O(1) lookup by key or by value.
2//!
3//! - Mirrored forward and reverse `HashMap` tables kept in sync on every mutation.
4//! - Insert, remove-by-key, remove-by-value, and containment checks in both directions.
5
6use std::collections::HashMap;
7use std::hash::Hash;
8/// Hash map with O(1) lookups in both directions using mirrored forward and reverse tables.
9#[derive(Debug, Clone)]
10pub struct BiMap<K, V> {
11    /// Forward map: key → value.
12    fwd: HashMap<K, V>,
13    /// Reverse map: value → key.
14    rev: HashMap<V, K>,
15}
16/// All methods for `BiMap`.
17impl<K: Clone + Hash + Eq, V: Clone + Hash + Eq> BiMap<K, V> {
18    /// Create an empty `BiMap`. This function is part of the public API.
19    pub fn new() -> Self {
20        Self {
21            fwd: HashMap::new(),
22            rev: HashMap::new(),
23        }
24    }
25    /// Insert the `key`/`value` pair, removing any existing pair that shares the same key or value.
26    pub fn insert(&mut self, key: K, value: V) {
27        if let Some(old_val) = self.fwd.get(&key) {
28            self.rev.remove(old_val);
29        }
30        if let Some(old_key) = self.rev.get(&value) {
31            self.fwd.remove(old_key);
32        }
33        self.fwd.insert(key.clone(), value.clone());
34        self.rev.insert(value, key);
35    }
36    /// Return the value associated with `key`, or `None`.
37    pub fn get_by_key(&self, key: &K) -> Option<&V> {
38        self.fwd.get(key)
39    }
40    /// Return the key associated with `value`, or `None`.
41    pub fn get_by_value(&self, value: &V) -> Option<&K> {
42        self.rev.get(value)
43    }
44    /// Return true when `key` is present.
45    pub fn contains_key(&self, key: &K) -> bool {
46        self.fwd.contains_key(key)
47    }
48    /// Return true when `value` is present.
49    pub fn contains_value(&self, value: &V) -> bool {
50        self.rev.contains_key(value)
51    }
52    /// Remove the pair identified by `key`; return `(key, value)` or `None` when absent.
53    pub fn remove_by_key(&mut self, key: &K) -> Option<(K, V)> {
54        if let Some(val) = self.fwd.remove(key) {
55            self.rev.remove(&val);
56            Some((key.clone(), val))
57        } else {
58            None
59        }
60    }
61    /// Remove the pair identified by `value`; return `(key, value)` or `None` when absent.
62    pub fn remove_by_value(&mut self, value: &V) -> Option<(K, V)> {
63        if let Some(key) = self.rev.remove(value) {
64            self.fwd.remove(&key);
65            Some((key, value.clone()))
66        } else {
67            None
68        }
69    }
70    /// Return the number of key-value pairs.
71    pub fn len(&self) -> usize {
72        self.fwd.len()
73    }
74    /// Return true when the map contains no pairs.
75    pub fn is_empty(&self) -> bool {
76        self.fwd.is_empty()
77    }
78    /// Remove all pairs from the map.
79    pub fn clear(&mut self) {
80        self.fwd.clear();
81        self.rev.clear();
82    }
83}
84/// Delegates to `Self::new()`.
85impl<K: Clone + Hash + Eq, V: Clone + Hash + Eq> Default for BiMap<K, V> {
86    /// Return a new empty `BiMap`.
87    fn default() -> Self {
88        Self::new()
89    }
90}
