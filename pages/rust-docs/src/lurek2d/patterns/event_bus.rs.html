1//! Named event bus that routes events to prioritized subscriptions.
2//!
3//! - Supports wildcard listeners, one-shot subscriptions, and per-event clearing.
4//! - Returns ordered listener ID lists for the Lua callback layer to dispatch.
5
6use std::collections::HashMap;
7
8/// Single subscription entry tracking event name, priority, and one-shot flag.
9#[derive(Debug, Clone)]
10pub struct Subscription {
11    /// Unique subscription identifier.
12    pub id: u64,
13    /// Event name this subscription listens to; `"*"` matches all events.
14    pub event: String,
15    /// Higher values are dispatched first.
16    pub priority: i64,
17    /// When true, the subscription is removed after the first dispatch.
18    pub once: bool,
19}
20/// Named event router that tracks subscriptions and returns ordered listener ID lists.
21#[derive(Debug)]
22pub struct EventBus {
23    /// Debug name for this bus instance.
24    pub name: String,
25    /// When false, `get_listeners` returns an empty list.
26    pub enabled: bool,
27    /// Next subscription identifier to assign.
28    next_id: u64,
29    /// Active subscriptions keyed by id.
30    subs: HashMap<u64, Subscription>,
31}
32/// All methods for `EventBus`.
33impl EventBus {
34    /// Create an enabled bus named `name`.
35    pub fn new(name: &str) -> Self {
36        Self {
37            name: name.to_string(),
38            enabled: true,
39            next_id: 1,
40            subs: HashMap::new(),
41        }
42    }
43    /// Register a listener for `event` with `priority` and `once` flag; return the subscription id.
44    pub fn subscribe(&mut self, event: &str, priority: i64, once: bool) -> u64 {
45        let id = self.next_id;
46        self.next_id += 1;
47        self.subs.insert(
48            id,
49            Subscription {
50                id,
51                event: event.to_string(),
52                priority,
53                once,
54            },
55        );
56        id
57    }
58    /// Remove the subscription with `id`; return true when it existed.
59    pub fn unsubscribe(&mut self, id: u64) -> bool {
60        self.subs.remove(&id).is_some()
61    }
62    /// Return listener IDs for `event` sorted by descending priority; empty when bus is disabled.
63    pub fn get_listeners(&self, event: &str) -> Vec<u64> {
64        if !self.enabled {
65            return Vec::new();
66        }
67        let mut listeners: Vec<&Subscription> = self
68            .subs
69            .values()
70            .filter(|s| s.event == event || s.event == "*")
71            .collect();
72        listeners.sort_by(|a, b| b.priority.cmp(&a.priority));
73        listeners.iter().map(|s| s.id).collect()
74    }
75    /// Remove all `once` subscriptions from `ids` and return the removed ids.
76    pub fn drain_once(&mut self, ids: &[u64]) -> Vec<u64> {
77        let once: Vec<u64> = ids
78            .iter()
79            .copied()
80            .filter(|id| self.subs.get(id).map(|s| s.once).unwrap_or(false))
81            .collect();
82        for id in &once {
83            self.subs.remove(id);
84        }
85        once
86    }
87    /// Remove all subscriptions for `event`; return the removed ids.
88    pub fn clear_event(&mut self, event: &str) -> Vec<u64> {
89        let ids: Vec<u64> = self
90            .subs
91            .values()
92            .filter(|s| s.event == event)
93            .map(|s| s.id)
94            .collect();
95        for id in &ids {
96            self.subs.remove(id);
97        }
98        ids
99    }
100    /// Remove all subscriptions; return all removed ids.
101    pub fn clear_all(&mut self) -> Vec<u64> {
102        let ids: Vec<u64> = self.subs.keys().copied().collect();
103        self.subs.clear();
104        ids
105    }
106    /// Return the number of active subscriptions for exactly `event`.
107    pub fn listener_count(&self, event: &str) -> usize {
108        self.subs.values().filter(|s| s.event == event).count()
109    }
110    /// Return all distinct event names with at least one subscriber, sorted alphabetically.
111    pub fn event_names(&self) -> Vec<String> {
112        let mut events: Vec<String> = self
113            .subs
114            .values()
115            .map(|s| s.event.clone())
116            .collect::<std::collections::HashSet<_>>()
117            .into_iter()
118            .collect();
119        events.sort();
120        events
121    }
122    /// Return the total number of active subscriptions across all events.
123    pub fn total_count(&self) -> usize {
124        self.subs.len()
125    }
126}
