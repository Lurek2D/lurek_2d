1//! Sorted priority queue with stable FIFO tie-breaking for equal priorities.
2//!
3//! - Push, pop, peek, and remove by id with O(n) insertion via partition point.
4//! - Each item carries an auto-assigned id, priority, label, and sequence number.
5
6/// A single entry in the queue with a stable tie-breaking sequence number.
7#[derive(Debug, Clone)]
8pub struct PriorityItem {
9    /// Unique item id.
10    pub id: u64,
11    /// Scheduling priority; higher is dequeued first.
12    pub priority: i64,
13    /// Debug label.
14    pub label: String,
15    /// Insertion sequence number for FIFO ordering among equal priorities.
16    pub seq: u64,
17}
18/// Sorted queue of `PriorityItem` entries.
19#[derive(Debug)]
20pub struct PriorityQueue {
21    /// Debug name.
22    pub name: String,
23    /// Next item id to assign.
24    next_id: u64,
25    /// Insertion counter for stable ordering.
26    next_seq: u64,
27    /// Items stored in descending priority order.
28    items: Vec<PriorityItem>,
29}
30/// All methods for `PriorityQueue`.
31impl PriorityQueue {
32    /// Create an empty priority queue named `name`.
33    pub fn new(name: &str) -> Self {
34        Self {
35            name: name.to_string(),
36            next_id: 1,
37            next_seq: 0,
38            items: Vec::new(),
39        }
40    }
41    /// Insert an item with `priority` and `label`; return its id.
42    pub fn push(&mut self, priority: i64, label: &str) -> u64 {
43        let id = self.next_id;
44        self.next_id += 1;
45        let seq = self.next_seq;
46        self.next_seq += 1;
47        let item = PriorityItem {
48            id,
49            priority,
50            label: label.to_string(),
51            seq,
52        };
53        let pos = self
54            .items
55            .partition_point(|x| x.priority > priority || (x.priority == priority && x.seq < seq));
56        self.items.insert(pos, item);
57        id
58    }
59    /// Return a reference to the highest-priority item without removing it.
60    pub fn peek(&self) -> Option<&PriorityItem> {
61        self.items.first()
62    }
63    /// Remove and return the highest-priority item's id and priority.
64    pub fn pop(&mut self) -> Option<(u64, i64)> {
65        if self.items.is_empty() {
66            return None;
67        }
68        let item = self.items.remove(0);
69        Some((item.id, item.priority))
70    }
71    /// Remove the item with `id`; return true when it was found.
72    pub fn remove(&mut self, id: u64) -> bool {
73        let before = self.items.len();
74        self.items.retain(|i| i.id != id);
75        self.items.len() < before
76    }
77    /// Return the number of items in the queue.
78    pub fn len(&self) -> usize {
79        self.items.len()
80    }
81    /// Return true when the queue is empty.
82    pub fn is_empty(&self) -> bool {
83        self.items.is_empty()
84    }
85    /// Return all items in priority order.
86    pub fn items(&self) -> &[PriorityItem] {
87        &self.items
88    }
89    /// Remove all items. This function is part of the public API.
90    pub fn clear(&mut self) {
91        self.items.clear();
92    }
93}
