1//! Named type registry with alias support for dynamic object construction.
2//!
3//! - Register, unregister, and resolve canonical type names at runtime.
4//! - Alias mapping allows multiple names to reference the same underlying type.
5
6use std::collections::HashSet;
7/// Registry of named constructable types with optional string aliases.
8#[derive(Debug, Default)]
9pub struct Factory {
10    /// Canonical type names that have been registered.
11    pub types: HashSet<String>,
12    /// Alias-to-canonical-name mapping.
13    pub aliases: std::collections::HashMap<String, String>,
14}
15/// Registration and lookup methods for `Factory`.
16impl Factory {
17    /// Create an empty factory. This function is part of the public API.
18    pub fn new() -> Self {
19        Self::default()
20    }
21    /// Register `name` as a known type.
22    pub fn register(&mut self, name: &str) {
23        self.types.insert(name.to_string());
24    }
25    /// Remove `name` and any aliases pointing to it; return true when the type existed.
26    pub fn unregister(&mut self, name: &str) -> bool {
27        self.aliases.retain(|_, v| v != name);
28        self.types.remove(name)
29    }
30    /// Return true when `name` is a registered type or a known alias.
31    pub fn has(&self, name: &str) -> bool {
32        self.types.contains(name) || self.aliases.contains_key(name)
33    }
34    /// Return the canonical name for `name`, following one level of alias if present.
35    pub fn resolve<'a>(&'a self, name: &'a str) -> &'a str {
36        self.aliases.get(name).map(String::as_str).unwrap_or(name)
37    }
38    /// Map `alias` to `canonical`; overwrites any previous mapping for that alias.
39    pub fn add_alias(&mut self, alias: &str, canonical: &str) {
40        self.aliases
41            .insert(alias.to_string(), canonical.to_string());
42    }
43    /// Return all registered canonical type names sorted alphabetically.
44    pub fn type_names(&self) -> Vec<&str> {
45        let mut names: Vec<&str> = self.types.iter().map(String::as_str).collect();
46        names.sort();
47        names
48    }
49    /// Remove all types and aliases.
50    pub fn clear(&mut self) {
51        self.types.clear();
52        self.aliases.clear();
53    }
54}
