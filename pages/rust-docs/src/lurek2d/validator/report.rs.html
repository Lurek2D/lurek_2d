1//! Structured violation report: aggregates, formats, and summarises validation results.
2//!
3//! - `Violation` carries file path, line number, `Severity`, rule name, and message.
4//! - `ValidationReport` holds `Vec<Violation>` and provides filter/sort helpers.
5//! - `Severity` enum: `Info`, `Warning`, `Error` — ordered by increasing severity.
6//! - `ValidationReport::display_summary()` prints a compact human-readable table.
7use std::path::PathBuf;
8
9/// Severity level of a validation violation.
10#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
11pub enum Severity {
12    Hint,
13    Warning,
14    Error,
15    Critical,
16}
17
18impl Severity {
19    /// Parse a `Severity` level from a string; unknown values map to `Warning`.
20    pub fn from_name(s: &str) -> Self {
21        match s.to_lowercase().as_str() {
22            "hint" | "info" => Self::Hint,
23            "warning" | "warn" => Self::Warning,
24            "error" | "err" => Self::Error,
25            "critical" | "fatal" => Self::Critical,
26            _ => Self::Warning,
27        }
28    }
29
30    /// Return the lowercase canonical string name of this severity level.
31    pub fn as_str(&self) -> &'static str {
32        match self {
33            Self::Hint => "hint",
34            Self::Warning => "warning",
35            Self::Error => "error",
36            Self::Critical => "critical",
37        }
38    }
39}

41/// A single validation violation.
42#[derive(Debug, Clone)]
43pub struct Violation {
44    /// Rule id.
45    pub rule_id: String,
46    /// Severity.
47    pub severity: Severity,
48    /// File.
49    pub file: PathBuf,
50    /// Line.
51    pub line: Option<usize>,
52    /// Column.
53    pub column: Option<usize>,
54    /// Log message text.
55    pub message: String,
56    /// Suggestion.
57    pub suggestion: Option<String>,
58}
59
60impl Violation {
61    /// Create a new `Violation` for the given rule, severity, file, and message.
62    pub fn new(rule_id: impl Into<String>, severity: Severity, file: PathBuf, message: impl Into<String>) -> Self {
63        Self {
64            rule_id: rule_id.into(),
65            severity,
66            file,
67            line: None,
68            column: None,
69            message: message.into(),
70            suggestion: None,
71        }
72    }
73
74    /// Attach a source line number to this violation.
75    pub fn with_line(mut self, line: usize) -> Self {
76        self.line = Some(line);
77        self
78    }
79
80    /// Attach a source column number to this violation.
81    pub fn with_column(mut self, col: usize) -> Self {
82        self.column = Some(col);
83        self
84    }
85
86    /// Attach a suggested fix message to this violation.
87    pub fn with_suggestion(mut self, suggestion: impl Into<String>) -> Self {
88        self.suggestion = Some(suggestion.into());
89        self
90    }
91}

93/// Complete validation report.
94#[derive(Debug, Clone)]
95pub struct ValidationReport {
96    /// Violations.
97    pub violations: Vec<Violation>,
98    /// Files checked.
99    pub files_checked: usize,
100    /// Frame display duration in milliseconds.
101    pub duration_ms: u64,
102}
103
104impl ValidationReport {
105    /// Create an empty report with no violations and zero counters.
106    pub fn empty() -> Self {
107        Self { violations: Vec::new(), files_checked: 0, duration_ms: 0 }
108    }
109
110    /// Return the count of violations at `Error` severity or higher.
111    pub fn error_count(&self) -> usize {
112        self.violations.iter().filter(|v| v.severity >= Severity::Error).count()
113    }
114
115    /// Return the count of violations at exactly `Warning` severity.
116    pub fn warning_count(&self) -> usize {
117        self.violations.iter().filter(|v| v.severity == Severity::Warning).count()
118    }
119
120    /// Return `true` if any violation is at `Error` or `Critical` severity.
121    pub fn has_errors(&self) -> bool {
122        self.violations.iter().any(|v| v.severity >= Severity::Error)
123    }
124
125    /// Return `true` if the report contains no violations at any severity.
126    pub fn is_clean(&self) -> bool {
127        self.violations.is_empty()
128    }
129
130    /// Return all violations that match the given severity level.
131    pub fn by_severity(&self, sev: Severity) -> Vec<&Violation> {
132        self.violations.iter().filter(|v| v.severity == sev).collect()
133    }
134
135    /// Return all violations that were raised against the given file path.
136    pub fn by_file(&self, file: &std::path::Path) -> Vec<&Violation> {
137        self.violations.iter().filter(|v| v.file == file).collect()
138    }
139}
