1//! Validation engine: orchestrates rule execution across file trees with parallel workers.
2//!
3//! - `ValidationEngine` loads config, builds the rule set, and calls `parallel::validate_parallel`.
4//! - Returns a `ValidationReport` aggregating all violations from all rules and files.
5//! - Custom Lua rules registered via `lurek.validator.add_rule` are injected here.
6//! - Used by `lurek.validator.run()` and the `python tools/validate/` quality gate.
7use super::asset_check::AssetExistenceRule;
8use super::api_check::ApiComplianceRule;
9use super::config::ValidatorConfig;
10use super::import_check::ImportResolutionRule;
11use super::parallel::{collect_lua_files, validate_parallel};
12use super::report::ValidationReport;
13use super::rule::ValidationRule;
14use super::rules_lua::LuaPatternRule;
15use super::rules_toml;
16use std::path::{Path, PathBuf};
17use std::sync::Arc;
18
19/// Main validation engine that manages rules and executes validation.
20pub struct ValidationEngine {
21    config: ValidatorConfig,
22    rules: Vec<Arc<dyn ValidationRule>>,
23    root: PathBuf,
24}
25
26impl ValidationEngine {
27    /// Create a new `ValidationEngine` rooted at `root` with the given configuration.
28    pub fn new(root: impl Into<PathBuf>, config: ValidatorConfig) -> Self {
29        Self {
30            config,
31            rules: Vec::new(),
32            root: root.into(),
33        }
34    }
35
36    /// Add the built-in asset existence rule.
37    pub fn add_asset_rule(&mut self, asset_root: impl Into<PathBuf>) {
38        self.rules.push(Arc::new(AssetExistenceRule::new(asset_root)));
39    }
40
41    /// Add the built-in import resolution rule.
42    pub fn add_import_rule(&mut self, lua_paths: Vec<PathBuf>) {
43        self.rules.push(Arc::new(ImportResolutionRule::new(lua_paths)));
44    }
45
46    /// Add the built-in API compliance rule.
47    pub fn add_api_rule(&mut self) {
48        self.rules.push(Arc::new(ApiComplianceRule::with_defaults()));
49    }
50
51    /// Add a custom Lua pattern validation rule.
52    pub fn add_pattern_rule(&mut self, rule: LuaPatternRule) {
53        self.rules.push(Arc::new(rule));
54    }
55
56    /// Load validation rules from a TOML file.
57    pub fn load_toml_rules(&mut self, path: &Path) {
58        let loaded = rules_toml::load_rules_from_file(path);
59        for rule in loaded {
60            self.rules.push(Arc::new(rule));
61        }
62    }
63
64    /// Run all rules against Lua files under the root.
65    pub fn run(&self) -> ValidationReport {
66        let files = collect_lua_files(&self.root);
67        validate_parallel(&files, &self.rules, self.config.thread_count)
68    }
69
70    /// Run against a specific set of files.
71    pub fn run_files(&self, files: &[PathBuf]) -> ValidationReport {
72        validate_parallel(files, &self.rules, self.config.thread_count)
73    }
74
75    /// Run all rules against a single file.
76    pub fn run_single(&self, path: &Path) -> ValidationReport {
77        let content = match std::fs::read_to_string(path) {
78            Ok(c) => c,
79            Err(_) => return ValidationReport::empty(),
80        };
81        let mut violations = Vec::new();
82        for rule in &self.rules {
83            violations.extend(rule.validate(path, &content));
84        }
85        ValidationReport {
86            violations,
87            files_checked: 1,
88            duration_ms: 0,
89        }
90    }
91
92    /// Return the total number of rules registered in this engine.
93    pub fn rule_count(&self) -> usize {
94        self.rules.len()
95    }
96
97    /// Return the root directory this engine validates files under.
98    pub fn root(&self) -> &Path {
99        &self.root
100    }
101}
