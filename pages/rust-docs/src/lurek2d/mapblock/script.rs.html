1//! Script steps that drive the procedural map block generation sequence.
2//!
3//! - `MapScript` is a `Vec<ScriptStep>` executed in order by the generator.
4//! - `StepType` variants: `Fill`, `PlaceGroup`, `PlaceBlock`, `ApplyLayer`, `Repeat`.
5//! - Steps can be loaded from TOML or constructed programmatically from Lua.
6//! - `Repeat { count, steps }` nests a sub-list with its own RNG advancement.
7
8/// Type of procedural map generation step executed by the build script.
9#[derive(Debug, Clone, Copy, PartialEq, Eq)]
10pub enum StepType {
11    /// Fill the entire map randomly from a group's blocks.
12    FillRandom,
13    /// Place a specific block at a fixed or next-available position.
14    PlaceBlock,
15    /// Place a random block from a group at a random valid position.
16    PlaceRandom,
17    /// Place blocks along a line.
18    PlaceLine,
19    /// Flood fill from a position with a given tile.
20    FloodFill,
21    /// Fill a rectangular area with a tile.
22    FillArea,
23    /// Draw a random path between two points.
24    DrawPath,
25    /// Fill a rectangle with a specific tile ID.
26    FillRect,
27    /// Place blocks only on map edges.
28    FillEdges,
29    /// Place blocks that satisfy neighbor constraints automatically.
30    AutoPlace,
31}
32
33/// A single step in a map generation script.
34#[derive(Debug, Clone)]
35pub struct ScriptStep {
36    /// Type of step to execute.
37    pub step_type: StepType,
38    /// Group name to use for this step.
39    pub group_name: String,
40    /// Specific block index within group; -1 means random.
41    pub block_index: i32,
42    /// X position for placement (if applicable).
43    pub x: i32,
44    /// Y position for placement (if applicable).
45    pub y: i32,
46    /// Width for area operations.
47    pub width: u32,
48    /// Height for area operations.
49    pub height: u32,
50    /// Number of times to perform this step.
51    pub count: u32,
52    /// Rotation in 90-degree increments (0, 1, 2, 3).
53    pub rotation: u32,
54    /// Mirror the block horizontally.
55    pub mirror: bool,
56    /// Randomly rotate when placing.
57    pub random_rotation: bool,
58    /// Randomly mirror when placing.
59    pub random_mirror: bool,
60    /// Match edge sides when placing (Carcassonne-style).
61    pub match_sides: bool,
62    /// Probability of this step executing (0.0..=1.0).
63    pub chance: f32,
64    /// How many times this step is repeated.
65    pub repeat_count: u32,
66    /// Tile ID for fill operations.
67    pub tile_id: u32,
68    /// Slot index to write to (for multi-slot tiles).
69    pub slot_index: usize,
70    /// Tileset ID for fill operations.
71    pub tileset_id: u32,
72    /// Layer index to write to.
73    pub layer: u32,
74    /// Level (floor/storey) to operate on.
75    pub level: u32,
76}

105/// An ordered sequence of steps that drives one generation pass.
106#[derive(Debug, Clone)]
107pub struct MapScript {
108    /// Human-readable identifier.
109    name: String,
110    /// Ordered list of generation steps.
111    steps: Vec<ScriptStep>,
112}
113
114impl MapScript {
115    /// Create an empty script with the given name.
116    pub fn new(name: &str) -> Self {
117        Self {
118            name: name.to_string(),
119            steps: Vec::new(),
120        }
121    }
122
123    /// Append a new step to this script.
124    pub fn add_step(&mut self, step: ScriptStep) {
125        self.steps.push(step);
126    }
127
128    /// Add a step from type and group name with defaults.
129    pub fn add_step_simple(&mut self, step_type: StepType, group_name: &str, count: u32) {
130        let step = ScriptStep {
131            step_type,
132            group_name: group_name.to_string(),
133            count,
134            ..Default::default()
135        };
136        self.steps.push(step);
137    }
138
139    /// Get an immutable script step by index.
140    pub fn get_step(&self, index: usize) -> Option<&ScriptStep> {
141        self.steps.get(index)
142    }
143
144    /// Get a mutable script step by index.
145    pub fn get_step_mut(&mut self, index: usize) -> Option<&mut ScriptStep> {
146        self.steps.get_mut(index)
147    }
148
149    /// Get the total number of steps in this script.
150    pub fn step_count(&self) -> usize {
151        self.steps.len()
152    }
153
154    /// Remove a script step at the given index.
155    pub fn remove_step(&mut self, index: usize) {
156        if index < self.steps.len() {
157            self.steps.remove(index);
158        }
159    }
160
161    /// Clear all steps from this script.
162    pub fn clear(&mut self) {
163        self.steps.clear();
164    }
165
166    /// Get this script's display name.
167    pub fn name(&self) -> &str {
168        &self.name
169    }
170
171    /// Set this script's display name.
172    pub fn set_name(&mut self, name: &str) {
173        self.name = name.to_string();
174    }
175
176    /// Get all steps in this script as a slice.
177    pub fn steps(&self) -> &[ScriptStep] {
178        &self.steps
179    }
180}
