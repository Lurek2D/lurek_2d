1//! Z-layer management for multi-storey and multi-level map construction.
2//!
3//! - `LayerStack` holds a `Vec<MapBlockGrid>`, one per Z level starting from 0.
4//! - Layers are independent grids; block placement in one layer does not affect another.
5//! - Layer 0 is the ground floor; negative indices are not supported.
6//! - The `MapBlockConfig::layer_count` field pre-allocates the stack at generator init.
7
8use super::maptile::MapTile;
9
10/// A single layer within a map block — a 2D grid of map tiles.
11#[derive(Debug, Clone)]
12pub struct BlockLayer {
13    /// Width in tiles.
14    width: u32,
15    /// Height in tiles.
16    height: u32,
17    /// Number of slots per tile (from config).
18    slot_count: usize,
19    /// Tile data stored row-major: `tiles[y * width + x]`.
20    tiles: Vec<MapTile>,
21}
22
23impl BlockLayer {
24    /// Create a new layer with given dimensions and slot count per tile.
25    pub fn new(width: u32, height: u32, slot_count: usize) -> Self {
26        let count = (width * height) as usize;
27        let tiles = vec![MapTile::new(slot_count); count];
28        Self {
29            width,
30            height,
31            slot_count,
32            tiles,
33        }
34    }
35
36    /// Get the tile width of this layer.
37    pub fn width(&self) -> u32 {
38        self.width
39    }
40
41    /// Get the tile height of this layer.
42    pub fn height(&self) -> u32 {
43        self.height
44    }
45
46    /// Get a tile reference at (x, y).
47    pub fn get_tile(&self, x: u32, y: u32) -> Option<&MapTile> {
48        if x < self.width && y < self.height {
49            Some(&self.tiles[(y * self.width + x) as usize])
50        } else {
51            None
52        }
53    }
54
55    /// Get a mutable tile reference at (x, y).
56    pub fn get_tile_mut(&mut self, x: u32, y: u32) -> Option<&mut MapTile> {
57        if x < self.width && y < self.height {
58            Some(&mut self.tiles[(y * self.width + x) as usize])
59        } else {
60            None
61        }
62    }
63
64    /// Set a specific slot on a tile at (x, y).
65    pub fn set_tile_slot(
66        &mut self,
67        x: u32,
68        y: u32,
69        slot_index: usize,
70        tileset_id: u32,
71        gid: u32,
72    ) {
73        if let Some(tile) = self.get_tile_mut(x, y) {
74            tile.set_slot(slot_index, tileset_id, gid);
75        }
76    }
77
78    /// Get the GID of a specific slot at (x, y).
79    pub fn get_tile_gid(&self, x: u32, y: u32, slot_index: usize) -> u32 {
80        self.get_tile(x, y)
81            .and_then(|t| t.get_slot(slot_index))
82            .map(|s| s.gid)
83            .unwrap_or(0)
84    }
85
86    /// Fill entire layer with a single GID in a specific slot.
87    pub fn fill(&mut self, slot_index: usize, tileset_id: u32, gid: u32) {
88        for tile in &mut self.tiles {
89            tile.set_slot(slot_index, tileset_id, gid);
90        }
91    }
92
93    /// Clear all tiles in this layer.
94    pub fn clear(&mut self) {
95        for tile in &mut self.tiles {
96            *tile = MapTile::new(self.slot_count);
97        }
98    }
99
100    /// Return the number of slots per tile.
101    pub fn slot_count(&self) -> usize {
102        self.slot_count
103    }
104}
