1//! Map block definition: tile slots, edge connection points, and per-block metadata.
2//!
3//! - `MapBlock` holds a grid of `MapTile` slots and a set of edge constraint descriptors.
4//! - `BlockMeta` carries the name, weight, group membership, and tileset reference.
5//! - Blocks are loaded from TOML files under `content/maps/blocks/`.
6//! - Exposed to Lua via `lurek.mapblock.define(spec)` for runtime registration.
7
8use super::config::MapBlockConfig;
9use super::constraints::EdgeConstraint;
10use super::layer::BlockLayer;
11use std::collections::HashMap;
12
13/// Cardinal direction for block edges.
14#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
15pub enum Edge {
16    /// North edge (top).
17    North,
18    /// East edge (right).
19    East,
20    /// South edge (bottom).
21    South,
22    /// West edge (left).
23    West,
24}
25
26/// A composable map block containing layers of tiles with edge constraints.
27#[derive(Debug, Clone)]
28pub struct MapBlock {
29    /// Block name/identifier.
30    pub name: String,
31    /// Width in tiles.
32    width: u32,
33    /// Height in tiles.
34    height: u32,
35    /// Layers (1..=10), index 0 = bottom.
36    layers: Vec<BlockLayer>,
37    /// Number of slots per tile from config.
38    slot_count: usize,
39    /// Edge type IDs: `(Edge, segment_index) -> edge_type_id`.
40    sides: HashMap<(Edge, u32), u32>,
41    /// Block weight for random selection (higher = more likely).
42    pub weight: f32,
43    /// If true, this block must be placed on the map edge.
44    pub edge_only: bool,
45    /// If true, this block cannot be placed on the map edge.
46    pub interior_only: bool,
47    /// Number of vertical levels this block spans (for multi-level maps).
48    pub level_span: u32,
49    /// Legacy segment size (for compatibility with existing mapgen).
50    pub segment_size: u32,
51}
52
53impl MapBlock {
54    /// Create a new map block with given dimensions and layer count.
55    pub fn new(width: u32, height: u32, layer_count: u32, config: &MapBlockConfig) -> Self {
56        let slot_count = config.slot_count();
57        let layer_count = layer_count.clamp(1, config.max_layers);
58        let layers = (0..layer_count)
59            .map(|_| BlockLayer::new(width, height, slot_count))
60            .collect();
61
62        Self {
63            name: String::new(),
64            width,
65            height,
66            layers,
67            slot_count,
68            sides: HashMap::new(),
69            weight: 1.0,
70            edge_only: false,
71            interior_only: false,
72            level_span: 1,
73            segment_size: width,
74        }
75    }
76
77    /// Create a block from legacy mapgen format (single layer, raw GID data).
78    pub fn from_legacy(
79        width: u32,
80        height: u32,
81        segment_size: u32,
82        tile_data: &[Vec<u32>],
83        sides: &HashMap<(Edge, u32), u32>,
84        name: &str,
85        weight: f32,
86    ) -> Self {
87        let slot_count = 1; // Legacy uses single slot (floor only)
88        let mut layers = Vec::new();
89
90        for layer_data in tile_data {
91            let mut layer = BlockLayer::new(width, height, slot_count);
92            for y in 0..height {
93                for x in 0..width {
94                    let idx = (y * width + x) as usize;
95                    if idx < layer_data.len() && layer_data[idx] != 0 {
96                        layer.set_tile_slot(x, y, 0, 1, layer_data[idx]);
97                    }
98                }
99            }
100            layers.push(layer);
101        }
102
103        if layers.is_empty() {
104            layers.push(BlockLayer::new(width, height, slot_count));
105        }
106
107        Self {
108            name: name.to_string(),
109            width,
110            height,
111            layers,
112            slot_count,
113            sides: sides.clone(),
114            weight,
115            edge_only: false,
116            interior_only: false,
117            level_span: 1,
118            segment_size,
119        }
120    }
121
122    /// Get the block width in tile units.
123    pub fn get_width(&self) -> u32 {
124        self.width
125    }
126
127    /// Get the block height in tile units.
128    pub fn get_height(&self) -> u32 {
129        self.height
130    }
131
132    /// Get the total number of tile layers.
133    pub fn get_layer_count(&self) -> u32 {
134        self.layers.len() as u32
135    }
136
137    /// Get an immutable tile layer by index.
138    pub fn get_layer(&self, index: usize) -> Option<&BlockLayer> {
139        self.layers.get(index)
140    }
141
142    /// Get a mutable tile layer by index.
143    pub fn get_layer_mut(&mut self, index: usize) -> Option<&mut BlockLayer> {
144        self.layers.get_mut(index)
145    }
146
147    /// Add a new empty layer. Returns false if max layers reached.
148    pub fn add_layer(&mut self, max_layers: u32) -> bool {
149        if self.layers.len() < max_layers as usize {
150            self.layers
151                .push(BlockLayer::new(self.width, self.height, self.slot_count));
152            true
153        } else {
154            false
155        }
156    }
157
158    /// Set a tile slot value at (layer, x, y, slot).
159    pub fn set_tile(
160        &mut self,
161        layer: usize,
162        x: u32,
163        y: u32,
164        slot_index: usize,
165        tileset_id: u32,
166        gid: u32,
167    ) {
168        if let Some(l) = self.layers.get_mut(layer) {
169            l.set_tile_slot(x, y, slot_index, tileset_id, gid);
170        }
171    }
172
173    /// Get a tile GID at (layer, x, y, slot).
174    pub fn get_tile(&self, layer: usize, x: u32, y: u32, slot_index: usize) -> u32 {
175        self.layers
176            .get(layer)
177            .map(|l| l.get_tile_gid(x, y, slot_index))
178            .unwrap_or(0)
179    }
180
181    /// Legacy get_tile for single-slot blocks (slot 0, layer 0).
182    pub fn get_tile_legacy(&self, _layer: usize, x: u32, y: u32) -> u32 {
183        self.get_tile(0, x, y, 0)
184    }
185
186    /// Set edge type for a given side and segment index.
187    pub fn set_edge(&mut self, edge: Edge, segment: u32, edge_type: u32) {
188        self.sides.insert((edge, segment), edge_type);
189    }
190
191    /// Get edge type for a given side and segment index.
192    pub fn get_edge(&self, edge: Edge, segment: u32) -> Option<u32> {
193        self.sides.get(&(edge, segment)).copied()
194    }
195
196    /// Get all edge constraints for this block.
197    pub fn get_edge_constraints(&self) -> Vec<EdgeConstraint> {
198        self.sides
199            .iter()
200            .map(|((edge, seg), edge_type)| EdgeConstraint {
201                edge: *edge,
202                segment: *seg,
203                edge_type: *edge_type,
204            })
205            .collect()
206    }
207
208    /// Set the display name of this block.
209    pub fn set_name(&mut self, name: &str) {
210        self.name = name.to_string();
211    }
212
213    /// Get the display name of this block.
214    pub fn get_name(&self) -> &str {
215        &self.name
216    }
217
218    /// Get number of segments along an edge (width / segment_size or height / segment_size).
219    pub fn segments_horizontal(&self) -> u32 {
220        if self.segment_size > 0 {
221            self.width / self.segment_size
222        } else {
223            1
224        }
225    }
226
227    /// Get number of segments along vertical edge.
228    pub fn segments_vertical(&self) -> u32 {
229        if self.segment_size > 0 {
230            self.height / self.segment_size
231        } else {
232            1
233        }
234    }
