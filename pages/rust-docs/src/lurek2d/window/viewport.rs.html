1//! Logical game viewport size queries and scale-mode staging.
2//!
3//! - Coordinate conversion between logical game space and physical screen pixels.
4//! - Viewport scale/offset snapshot via `ScaleInfo`.
5
6use crate::runtime::shared_state::WindowState;
7/// Return the logical game viewport width in pixels.
8pub fn get_width(ws: &WindowState) -> f32 {
9    ws.game_width
10}
11/// Return the logical game viewport height in pixels.
12pub fn get_height(ws: &WindowState) -> f32 {
13    ws.game_height
14}
15/// Return the current scale mode string (`"none"`, `"letterbox"`, `"stretch"`, or `"pixel"`).
16pub fn get_scale_mode(ws: &WindowState) -> &str {
17    &ws.scale_mode_str
18}
19/// Stage a scale mode change to `mode`; validated and applied by the event loop next frame.
20pub fn set_scale_mode(ws: &mut WindowState, mode: &str) {
21    ws.pending_scale_mode = Some(mode.to_owned());
22}
23/// Convert logical game `(x, y)` to physical screen pixels using current scale and letterbox offset.
24pub fn to_pixels(ws: &WindowState, x: f32, y: f32) -> (f32, f32) {
25    (
26        x * ws.viewport_scale_x + ws.viewport_offset_x,
27        y * ws.viewport_scale_y + ws.viewport_offset_y,
28    )
29}
30/// Convert physical screen `(x, y)` back to logical game coordinates; return `(0, 0)` when scale is near-zero.
31pub fn from_pixels(ws: &WindowState, x: f32, y: f32) -> (f32, f32) {
32    let sx = if ws.viewport_scale_x.abs() > f32::EPSILON {
33        (x - ws.viewport_offset_x) / ws.viewport_scale_x
34    } else {
35        0.0
36    };
37    let sy = if ws.viewport_scale_y.abs() > f32::EPSILON {
38        (y - ws.viewport_offset_y) / ws.viewport_scale_y
39    } else {
40        0.0
41    };
42    (sx, sy)
43}
44/// Snapshot of all viewport scale and offset values returned by `get_scale_info`.
45pub struct ScaleInfo {
46    /// Horizontal scale factor from game to screen pixels.
47    pub scale_x: f32,
48    /// Vertical scale factor from game to screen pixels.
49    pub scale_y: f32,
50    /// Horizontal letterbox offset in screen pixels.
51    pub offset_x: f32,
52    /// Vertical letterbox offset in screen pixels.
53    pub offset_y: f32,
54    /// Logical game viewport width.
55    pub game_width: f32,
56    /// Logical game viewport height.
57    pub game_height: f32,
58}
59/// Return a `ScaleInfo` snapshot of the current viewport transform.
60pub fn get_scale_info(ws: &WindowState) -> ScaleInfo {
61    ScaleInfo {
62        scale_x: ws.viewport_scale_x,
63        scale_y: ws.viewport_scale_y,
64        offset_x: ws.viewport_offset_x,
65        offset_y: ws.viewport_offset_y,
66        game_width: ws.game_width,
67        game_height: ws.game_height,
68    }
69}
70/// Stage a scale mode change; return `false` and log a warning when `mode` is not a recognised value.
71pub fn set_scale_mode_validated(ws: &mut WindowState, mode: &str) -> bool {
72    let m = mode.to_lowercase();
73    if matches!(m.as_str(), "none" | "letterbox" | "stretch" | "pixel") {
74        ws.pending_scale_mode = Some(m);
75        true
76    } else {
77        log::warn!("Unknown scale mode: {}", mode);
78        false
79    }
80}
