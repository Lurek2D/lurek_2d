1//! Formats fatal Lua and engine errors into a user-facing screen.
2//!
3//! - Splits message text and traceback, word-wraps long lines, and cleans Lua string markers.
4//! - Builds full-screen render commands showing error title, body, traceback, and hint footer.
5//! - Provides clipboard export text for quick copy of error details.
6
7use crate::render::renderer::{DrawMode, RenderCommand};
8use crate::runtime::error::EngineError;
9use crate::runtime::resource_keys::FontKey;
10const ERROR_BG: [f32; 4] = [0.11, 0.22, 0.53, 1.0];
11const ERROR_TITLE_COLOR: [f32; 4] = [0.9, 0.6, 0.6, 1.0];
12const ERROR_TEXT_COLOR: [f32; 4] = [1.0, 1.0, 1.0, 1.0];
13const ERROR_FOOTER_COLOR: [f32; 4] = [0.7, 0.7, 0.8, 1.0];
14const WRAP_WIDTH: usize = 80;
15const GLYPH_W: f32 = 8.0;
16const LINE_H: f32 = 14.0;
17/// Render-ready error payload shown when Lua or engine execution fails fatally.
18pub struct ErrorScreen {
19    /// Title line shown near the top of the screen.
20    title: String,
21    /// Wrapped primary error message lines.
22    message_lines: Vec<String>,
23    /// Optional cleaned traceback lines.
24    traceback_lines: Vec<String>,
25}
26impl ErrorScreen {
27    /// Build error screen model from plain text where first non-empty line becomes the title and remainder is body.
28    pub fn from_error(msg: &str) -> Self {
29        let lines: Vec<&str> = msg.lines().collect();
30
31        // Find first non-empty line for title
32        let title_idx = lines.iter().position(|line| !line.trim().is_empty());
33        let title = if let Some(idx) = title_idx {
34            lines[idx].trim().to_string()
35        } else {
36            "Error".to_string()
37        };
38
39        // Body is everything after the title line
40        let body = if let Some(idx) = title_idx {
41            if idx + 1 < lines.len() {
42                lines[idx + 1..].join("\n")
43            } else {
44                String::new()
45            }
46        } else {
47            String::new()
48        };
49
50        let message_lines = wrap_text(&body, WRAP_WIDTH);
51        Self {
52            title,
53            message_lines,
54            traceback_lines: Vec::new(),
55        }
56    }
57    /// Build error screen model from `mlua::Error`, splitting message and traceback sections.
58    pub fn from_lua_error(err: &mlua::Error) -> Self {
59        let full = format_lua_error(err);
60        let (msg_part, tb_part) = split_traceback(&full);
61        let cleaned_msg = replace_string_markers(&msg_part);
62        let (title, body) = if let Some(colon2) = find_second_colon(&cleaned_msg) {
63            let location = cleaned_msg[..colon2].trim();
64            let description = cleaned_msg[colon2 + 1..].trim();
65            (
66                format!("Lua Error  —  {}", location),
67                description.to_string(),
68            )
69        } else {
70            ("Lua Error".to_string(), cleaned_msg.clone())
71        };
72        let message_lines = wrap_text(&body, WRAP_WIDTH);
73        let traceback_lines = format_traceback(&tb_part);
74        Self {
75            title,
76            message_lines,
77            traceback_lines,
78        }
79    }
80    /// Build error screen model from `EngineError` display text.
81    pub fn from_engine_error(err: &EngineError) -> Self {
82        Self::from_error(&err.to_string())
83    }
84    /// Build draw commands that render full-screen error background, text body, traceback, and footer hints.
85    pub fn build_render_commands(
86        &self,
87        screen_w: u32,
88        screen_h: u32,
89        heading_font: Option<FontKey>,
90        body_font: Option<FontKey>,
91    ) -> Vec<RenderCommand> {
92        let mut cmds = Vec::new();
93        let margin_x = 40.0_f32;
94        let mut y = 40.0_f32;
95        cmds.push(RenderCommand::SetColor(
96            ERROR_BG[0],
97            ERROR_BG[1],
98            ERROR_BG[2],
99            ERROR_BG[3],
100        ));
101        cmds.push(RenderCommand::Rectangle {
102            mode: DrawMode::Fill,
103            x: 0.0,
104            y: 0.0,
105            w: screen_w as f32,
106            h: screen_h as f32,
107        });
108        cmds.push(RenderCommand::SetColor(
109            ERROR_TITLE_COLOR[0],
110            ERROR_TITLE_COLOR[1],
111            ERROR_TITLE_COLOR[2],
112            ERROR_TITLE_COLOR[3],
113        ));
114        if let Some(fk) = heading_font {
115            cmds.push(RenderCommand::Print {
116                font_key: fk,
117                text: "Error".to_string(),
118                x: margin_x,
119                y,
120                scale: 1.0,
121            });
122            y += 50.0;
123        } else {
124            y += 50.0;
125        }
126        cmds.push(RenderCommand::SetColor(
127            ERROR_TEXT_COLOR[0],
128            ERROR_TEXT_COLOR[1],
129            ERROR_TEXT_COLOR[2],
130            ERROR_TEXT_COLOR[3],
131        ));
132        let body_line_h = if body_font.is_some() {
133            22.0
134        } else {
135            LINE_H * 2.0
136        };
137        self.push_text(&mut cmds, &self.title, margin_x, y, body_font);
138        y += body_line_h + 4.0;
139        for line in &self.message_lines {
140            if !line.is_empty() {
141                self.push_text(&mut cmds, line, margin_x, y, body_font);
142            }
143            y += body_line_h;
144        }
145        if !self.traceback_lines.is_empty() {
146            y += body_line_h * 0.5;
147            cmds.push(RenderCommand::SetColor(0.8, 0.8, 0.9, 1.0));
148            self.push_text(&mut cmds, "Traceback:", margin_x, y, body_font);
149            y += body_line_h;
150            let indent = if body_font.is_some() {
151                24.0
152            } else {
153                GLYPH_W * 2.0 * 2.0
154            };
155            for line in &self.traceback_lines {
156                self.push_text(&mut cmds, line, margin_x + indent, y, body_font);
157                y += body_line_h;
158            }
159        }
160        let footer = "Press Escape to quit  |  R to restart  |  Ctrl+C to copy error";
161        let footer_y = screen_h as f32 - 50.0;
162        cmds.push(RenderCommand::SetColor(
163            ERROR_FOOTER_COLOR[0],
164            ERROR_FOOTER_COLOR[1],
165            ERROR_FOOTER_COLOR[2],
166            ERROR_FOOTER_COLOR[3],
167        ));
168        self.push_text(&mut cmds, footer, margin_x, footer_y, body_font);
169        cmds
170    }
171    /// Return formatted text representation used for clipboard export and logs.
172    pub fn as_text(&self) -> String {
173        let mut parts = vec![self.title.clone()];
174        parts.extend(self.message_lines.iter().cloned());
175        if !self.traceback_lines.is_empty() {
176            parts.push(String::new());
177            parts.push("Traceback:".to_string());
178            for line in &self.traceback_lines {
179                parts.push(format!("  {}", line));
180            }
181        }
182        parts.join("\n")
183    }
184    /// Push one text draw command when `font_key` is available.
185    fn push_text(
186        &self,
187        cmds: &mut Vec<RenderCommand>,
188        text: &str,
189        x: f32,
190        y: f32,
191        font_key: Option<FontKey>,
192    ) {
193        if let Some(fk) = font_key {
194            cmds.push(RenderCommand::Print {
195                font_key: fk,
196                text: text.to_string(),
197                x,
198                y,
199                scale: 1.0,
200            });
201        }
202    }
203}
204/// Wrap `text` into lines not exceeding `max_chars`, preserving existing line breaks.
205pub fn wrap_text(text: &str, max_chars: usize) -> Vec<String> {
206    let mut result = Vec::new();
207    for line in text.lines() {
208        if line.len() <= max_chars {
209            result.push(line.to_string());
210            continue;
211        }
212        let mut current = String::new();
213        for word in line.split_whitespace() {
214            if current.is_empty() {
215                if word.len() > max_chars {
216                    let mut remaining = word;
217                    while remaining.len() > max_chars {
218                        result.push(remaining[..max_chars].to_string());
219                        remaining = &remaining[max_chars..];
220                    }
221                    current = remaining.to_string();
222                } else {
223                    current = word.to_string();
224                }
225            } else if current.len() + 1 + word.len() > max_chars {
226                result.push(current);
227                current = word.to_string();
228            } else {
229                current.push(' ');
230                current.push_str(word);
231            }
232        }
233        if !current.is_empty() {
234            result.push(current);
235        }
236    }
237    if result.is_empty() && !text.is_empty() {
238        result.push(String::new());
239    }
240    result
241}
242/// Normalize traceback block by removing boilerplate lines and replacing string markers.
243pub fn format_traceback(traceback: &str) -> Vec<String> {
244    traceback
245        .lines()
246        .filter(|l| {
247            let trimmed = l.trim();
248            !trimmed.is_empty() && !trimmed.starts_with("stack traceback:")
249        })
250        .map(|l| {
251            let cleaned = l.trim();
252            let cleaned = replace_string_markers(cleaned);
253            format!("  {}", cleaned)
254        })
255        .collect()
256}
257/// Convert `mlua::Error` variants into user-facing text, preserving nested callback tracebacks.
258fn format_lua_error(err: &mlua::Error) -> String {
259    match err {
260        mlua::Error::RuntimeError(msg) => msg.clone(),
261        mlua::Error::SyntaxError {
262            message,
263            incomplete_input: _,
264        } => format!("Syntax error: {}", message),
265        mlua::Error::CallbackError { traceback, cause } => {
266            let inner = format_lua_error(cause);
267            format!("{}\nstack traceback:\n{}", inner, traceback)
268        }
269        other => format!("{}", other),
270    }
271}
272/// Split combined Lua error text into message part and traceback part.
273fn split_traceback(full: &str) -> (String, String) {
274    if let Some(idx) = full.find("stack traceback:") {
275        let msg = full[..idx].trim_end().to_string();
276        let tb = full[idx..].to_string();
277        (msg, tb)
278    } else {
279        (full.to_string(), String::new())
280    }
281}
282/// Return byte index of the second colon in `s`, if present.
283fn find_second_colon(s: &str) -> Option<usize> {
284    let first = s.find(':')?;
285    s[first + 1..].find(':').map(|i| first + 1 + i)
286}
287/// Replace Lua `[string "..."]` markers with bare script names for cleaner display.
288fn replace_string_markers(s: &str) -> String {
289    let mut result = s.to_string();
290    while let Some(start) = result.find("[string \"") {
291        let after = start + 9;
292        if let Some(end_quote) = result[after..].find("\"]") {
293            let name = result[after..after + end_quote].to_string();
294            let end = after + end_quote + 2;
295            result = format!("{}{}{}", &result[..start], name, &result[end..]);
296        } else {
297            break;
298        }
299    }
300    result
