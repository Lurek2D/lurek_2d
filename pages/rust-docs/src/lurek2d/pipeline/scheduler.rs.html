1//! Frame-driven delay scheduler that counts down per-step timers each update.
2//!
3//! - Report which pipeline steps become ready once their configured delay expires.
4//! - Track wall-clock elapsed time and running state for the owning pipeline.
5
6use crate::pipeline::dag::Pipeline;
7use crate::pipeline::step::StepStatus;
8use std::collections::HashMap;
9
10/// Drives step-delay countdowns and reports which steps become ready each frame.
11pub struct PipelineScheduler {
12    /// Remaining delay seconds per step name; decremented each `update` call.
13    delay_timers: HashMap<String, f32>,
14    /// Whether the pipeline is currently executing.
15    pub is_running: bool,
16    /// Total wall-clock seconds since `start` was called.
17    pub elapsed: f32,
18}
19
20impl PipelineScheduler {
21    /// Create a stopped scheduler with no active timers.
22    pub fn new() -> Self {
23        Self {
24            delay_timers: HashMap::new(),
25            is_running: false,
26            elapsed: 0.0,
27        }
28    }
29
30    /// Initialize delay timers from `pipeline` step definitions and begin execution.
31    pub fn start(&mut self, pipeline: &Pipeline) {
32        self.delay_timers.clear();
33        self.elapsed = 0.0;
34        self.is_running = true;
35        for step in pipeline.get_steps() {
36            self.delay_timers.insert(step.name.clone(), step.delay);
37        }
38    }
39
40    /// Advance timers by `dt` seconds and return names of steps whose delay has expired and are still `Waiting`.
41    pub fn update(&mut self, dt: f32, pipeline: &Pipeline) -> Vec<String> {
42        if !self.is_running {
43            return Vec::new();
44        }
45        self.elapsed += dt;
46        let mut ready: Vec<String> = Vec::new();
47        for step in pipeline.get_steps() {
48            if step.status != StepStatus::Waiting {
49                continue;
50            }
51            let timer = self.delay_timers.entry(step.name.clone()).or_insert(0.0);
52            *timer -= dt;
53            if *timer <= 0.0 {
54                ready.push(step.name.clone());
55            }
56        }
57        ready
58    }
59
60    /// Reset and arm the delay timer for `name` using its configured delay from `pipeline`.
61    pub fn mark_step_waiting(&mut self, name: &str, pipeline: &Pipeline) {
62        let delay = pipeline.get_step(name).map(|s| s.delay).unwrap_or(0.0);
63        self.delay_timers.insert(name.to_owned(), delay);
64    }
65
66    /// Clear all timers and stop execution; does not reset step statuses in the pipeline.
67    pub fn reset(&mut self) {
68        self.delay_timers.clear();
69        self.is_running = false;
70        self.elapsed = 0.0;
71    }
72}
73
74/// Delegate to `PipelineScheduler::new`.
75impl Default for PipelineScheduler {
76    fn default() -> Self {
77        Self::new()
78    }
79}
