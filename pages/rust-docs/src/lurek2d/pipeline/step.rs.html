1//! Pipeline step definition: named unit of work with dependency, timing, and retry config.
2//!
3//! - Execution lifecycle via `StepStatus` (pending → waiting → running → terminal).
4//! - Per-step error policy overriding pipeline-level failure behavior.
5
6use std::collections::HashMap;
7
8/// Execution lifecycle state of a single pipeline step.
9#[derive(Debug, Clone, PartialEq, Eq)]
10pub enum StepStatus {
11    /// Created but not yet enqueued for execution.
12    Pending,
13    /// Enqueued; waiting for its delay timer to expire.
14    Waiting,
15    /// Currently executing.
16    Running,
17    /// Finished successfully.
18    Completed,
19    /// Execution ended with an error.
20    Failed,
21    /// Not executed because a required dependency failed.
22    Skipped,
23    /// Cancelled when the pipeline aborted before this step ran.
24    Cancelled,
25}
26
27impl StepStatus {
28    /// Return the canonical lowercase token string for this status.
29    pub fn as_str(&self) -> &'static str {
30        match self {
31            Self::Pending => "pending",
32            Self::Waiting => "waiting",
33            Self::Running => "running",
34            Self::Completed => "completed",
35            Self::Failed => "failed",
36            Self::Skipped => "skipped",
37            Self::Cancelled => "cancelled",
38        }
39    }
40}
41
42/// Per-step failure response, overriding the pipeline-level `ErrorMode`.
43#[derive(Debug, Clone, PartialEq, Eq)]
44pub enum ErrorPolicy {
45    /// Abort the entire pipeline on failure.
46    Abort,
47    /// Mark this step failed and continue scheduling other steps.
48    Continue,
49    /// Re-queue this step up to `retry_count` times before failing.
50    Retry,
51}
52
53/// Single named unit of work in a `Pipeline`, carrying dependency, timing, and retry configuration.
54#[derive(Debug, Clone)]
55pub struct PipelineStep {
56    /// Unique name within the owning `Pipeline`.
57    pub name: String,
58    /// Names of steps that must complete before this one can run.
59    pub deps: Vec<String>,
60    /// Seconds to wait after deps satisfy before transitioning from `Waiting` to `Running`.
61    pub delay: f32,
62    /// When `true`, downstream steps are not blocked if this step fails.
63    pub optional: bool,
64    /// Maximum number of automatic retries on failure; 0 means no retries.
65    pub retry_count: u32,
66    /// Seconds to wait between retry attempts.
67    pub retry_delay: f32,
68    /// Per-step policy that overrides the pipeline `ErrorMode` on failure.
69    pub on_error: ErrorPolicy,
70    /// Optional grouping tag used for filtering or reporting.
71    pub tag: Option<String>,
72    /// Arbitrary key-value metadata stored for Lua consumption.
73    pub metadata: HashMap<String, String>,
74    /// Current execution lifecycle state.
75    pub status: StepStatus,
76    /// Number of execution attempts made so far, including retries.
77    pub attempt: u32,
78    /// Wall-clock seconds the most recent execution attempt took.
79    pub duration: f32,
80    /// Error message from the most recent failure, if any.
81    pub error_msg: Option<String>,
82}
83
84impl PipelineStep {
85    /// Create a step with default settings: no deps, no delay, no retries, `ErrorPolicy::Abort`.
86    pub fn new(name: impl Into<String>) -> Self {
87        Self {
88            name: name.into(),
89            deps: Vec::new(),
90            delay: 0.0,
91            optional: false,
92            retry_count: 0,
93            retry_delay: 0.0,
94            on_error: ErrorPolicy::Abort,
95            tag: None,
96            metadata: HashMap::new(),
97            status: StepStatus::Pending,
98            attempt: 0,
99            duration: 0.0,
100            error_msg: None,
101        }
102    }
103
104    /// Reset runtime state to `Pending`; clears attempt count, duration, and error message.
105    pub fn reset(&mut self) {
106        self.status = StepStatus::Pending;
107        self.attempt = 0;
108        self.duration = 0.0;
109        self.error_msg = None;
110    }
111}
