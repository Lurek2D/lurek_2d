1//! Lifecycle status enum tracking pipeline progression from pending through completion or failure.
2//!
3//! - Aggregated result struct collecting per-step outcomes, durations, and error messages.
4//! - Convenience queries for success checks and human-readable summaries.
5
6/// Lifecycle state of a pipeline run.
7#[derive(Debug, Clone, PartialEq, Eq)]
8pub enum PipelineStatus {
9    /// Not yet started.
10    Pending,
11    /// Actively executing steps.
12    Running,
13    /// All steps resolved without unrecovered failures.
14    Completed,
15    /// At least one non-optional step failed and error mode was Abort.
16    Failed,
17    /// Execution was cancelled before all steps ran.
18    Cancelled,
19}
20
21/// Aggregated outcome produced after a pipeline finishes or aborts.
22#[derive(Debug, Clone)]
23pub struct PipelineResult {
24    /// Overall lifecycle state of this run.
25    pub status: PipelineStatus,
26    /// Names of steps that reached `StepStatus::Completed`.
27    pub completed: Vec<String>,
28    /// Names of steps that reached `StepStatus::Failed`.
29    pub failed: Vec<String>,
30    /// Names of steps that were skipped due to failed non-optional dependencies.
31    pub skipped: Vec<String>,
32    /// Names of steps cancelled when the pipeline aborted early.
33    pub cancelled: Vec<String>,
34    /// Wall-clock seconds from pipeline start to finish.
35    pub total_duration: f32,
36    /// Per-step error messages: `(step_name, error_text)` pairs.
37    pub errors: Vec<(String, String)>,
38}
39
40impl PipelineResult {
41    /// Create a blank result in `Pending` state with no recorded steps.
42    pub fn new() -> Self {
43        Self {
44            status: PipelineStatus::Pending,
45            completed: Vec::new(),
46            failed: Vec::new(),
47            skipped: Vec::new(),
48            cancelled: Vec::new(),
49            total_duration: 0.0,
50            errors: Vec::new(),
51        }
52    }
53
54    /// Return `true` if no steps failed (skipped and cancelled do not count as failure).
55    pub fn is_success(&self) -> bool {
56        self.failed.is_empty()
57    }
58
59    /// Return a single-line human-readable summary of counts and duration.
60    pub fn summary(&self) -> String {
61        format!(
62            "status={:?} completed={} failed={} skipped={} cancelled={} duration={:.3}s",
63            self.status,
64            self.completed.len(),
65            self.failed.len(),
66            self.skipped.len(),
67            self.cancelled.len(),
68            self.total_duration,
69        )
70    }
71}
72
73/// Delegate to `PipelineResult::new`.
74impl Default for PipelineResult {
75    fn default() -> Self {
76        Self::new()
77    }
