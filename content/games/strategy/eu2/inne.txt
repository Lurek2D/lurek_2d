        // --- PROCERURALNY GENERATOR MAPY DEMO (Gdy nie ma pliku) ---
        // Generuje mapę imitującą plik map.png z prowincjami o unikalnych barwach rozdzielonych czarnym kolorem (0,0,0)
        function generateDemoMap() {
            const w = 240;
            const h = 110;
            const demoCanvas = document.createElement('canvas');
            demoCanvas.width = w;
            demoCanvas.height = h;
            const dCtx = demoCanvas.getContext('2d');

            // Wypełnienie tła - morze (kolor 0,0,0)
            dCtx.fillStyle = '#000000';
            dCtx.fillRect(0, 0, w, h);

            // Generowanie wysp/kontynentów jako skupisk prowincji
            const numProvinces = 60;
            const centers = [];
            
            // Losowe centra prowincji (z unikaniem krawędzi)
            for (let i = 0; i < numProvinces; i++) {
                centers.push({
                    x: Math.floor(15 + Math.random() * (w - 30)),
                    y: Math.floor(15 + Math.random() * (h - 30)),
                    r: 40 + Math.floor(Math.random() * 40),
                    g: 40 + Math.floor(Math.random() * 40),
                    b: 40 + Math.floor(Math.random() * 40),
                    islandId: Math.floor(Math.random() * 3) // grupowanie w 3 kontynenty
                });
            }

            const imgData = dCtx.createImageData(w, h);
            const data = imgData.data;

            // Algorytm najbliższego sąsiada (diagram Voronoi) dla prowincji
            for (let y = 0; y < h; y++) {
                for (let x = 0; x < w; x++) {
                    let minDist = Infinity;
                    let closestCenter = null;

                    // Ograniczamy rysowanie do organicznych kształtów kontynentów
                    let noise = Math.sin(x * 0.05) * Math.cos(y * 0.05) * 15;
                    let isLand = false;
                    
                    // Definiujemy 3 okręgi jako bazy kontynentów
                    const continents = [
                        {x: w * 0.25, y: h * 0.5, r: 40},
                        {x: w * 0.6, y: h * 0.4, r: 35},
                        {x: w * 0.8, y: h * 0.7, r: 25}
                    ];

                    for (let c of continents) {
                        let distToCont = Math.hypot(x - c.x, y - c.y);
                        if (distToCont < c.r + noise) {
                            isLand = true;
                            break;
                        }
                    }

                    if (!isLand) {
                        // Zostaje głębokie czarne morze
                        const idx = (y * w + x) * 4;
                        data[idx] = 0;
                        data[idx+1] = 0;
                        data[idx+2] = 0;
                        data[idx+3] = 255;
                        continue;
                    }

                    for (let c of centers) {
                        let dist = Math.hypot(x - c.x, y - c.y);
                        if (dist < minDist) {
                            minDist = dist;
                            closestCenter = c;
                        }
                    }

                    if (closestCenter) {
                        const idx = (y * w + x) * 4;
                        data[idx] = closestCenter.r;
                        data[idx+1] = closestCenter.g;
                        data[idx+2] = closestCenter.b;
                        data[idx+3] = 255;
                    }
                }
            }

            // Dodanie czarnych granic pomiędzy prowincjami (symulacja map.png)
            const borderData = new Uint8ClampedArray(data);
            for (let y = 1; y < h - 1; y++) {
                for (let x = 1; x < w - 1; x++) {
                    const idx = (y * w + x) * 4;
                    // Jeśli piksel jest lądem, ale jego sąsiad ma inny kolor, tworzymy granicę
                    const r = borderData[idx];
                    const g = borderData[idx+1];
                    const b = borderData[idx+2];
                    if (r === 0 && g === 0 && b === 0) continue;

                    let hasDifferentNeighbor = false;
                    const neighbors = [
                        ((y-1) * w + x) * 4,
                        ((y+1) * w + x) * 4,
                        (y * w + (x-1)) * 4,
                        (y * w + (x+1)) * 4
                    ];

                    for (let nIdx of neighbors) {
                        if (borderData[nIdx] !== r || borderData[nIdx+1] !== g || borderData[nIdx+2] !== b) {
                            hasDifferentNeighbor = true;
                            break;
                        }
                    }

                    if (hasDifferentNeighbor) {
                        data[idx] = 0;
                        data[idx+1] = 0;
                        data[idx+2] = 0;
                    }
                }
            }

        // --- ZAAWANSOWANA WEKTORYZACJA 8:1 (Zoptymalizowany podział na komponenty i uproszczenie) ---
        async function runVectorization() {
            showLoader(true, "Inicjalizowanie analizatora...");
            await sleep(100);

            const width = originalImgWidth;
            const height = originalImgHeight;
            const data = originalImgData.data;

            // Krok 1: Skanowanie pikseli i zbieranie unikalnych kolorów
            showLoader(true, "Skanowanie kolorów prowincji...", 10);
            await sleep(50);

            // Tworzymy tablicę odwiedzonych do detekcji połączonych komponentów
            const visited = new Uint8Array(width * height);
            provinces = {};
            provinceList = [];

            const dx = [1, 0, -1, 0];
            const dy = [0, 1, 0, -1];

            // Krok 2: Analiza zalewowa (Connected Components) dla każdej prowincji
            showLoader(true, "Identyfikacja spójnych prowincji...", 30);
            await sleep(50);

            let provinceCounter = 0;

            for (let y = 0; y < height; y++) {
                // Pozwalamy przeglądarce oddychać co kilka rzędów pikseli
                if (y % 15 === 0) {
                    const progress = Math.floor(30 + (y / height) * 40);
                    showLoader(true, `Analiza rzędu pikseli ${y}/${height}...`, progress);
                    await sleep(5);
                }

                for (let x = 0; x < width; x++) {
                    const idx = (y * width + x) * 4;
                    const offset = y * width + x;

                    if (visited[offset]) continue;

                    const r = data[idx];
                    const g = data[idx+1];
                    const b = data[idx+2];
                    const a = data[idx+3];

                    // Ignorujemy czarny (0,0,0) - traktujemy jako granice lub morze bezpaństwowe
                    if (a < 50 || (r === 0 && g === 0 && b === 0)) {
                        visited[offset] = 1;
                        continue;
                    }

                    const hexKey = rgbToHex(r, g, b);

                    // Wykonujemy szybki BFS, aby znaleźć cały połączony komponent o tym kolorze (np. wyspa lub główny kontynent)
                    const componentPixels = [];
                    const queue = [x, y];
                    visited[offset] = 1;

                    let head = 0;
                    while (head < queue.length) {
                        const cx = queue[head++];
                        const cy = queue[head++];
                        componentPixels.push({ x: cx, y: cy });

                        for (let d = 0; d < 4; d++) {
                            const nx = cx + dx[d];
                            const ny = cy + dy[d];

                            if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
                                const noffset = ny * width + nx;
                                if (!visited[noffset]) {
                                    const nidx = noffset * 4;
                                    if (data[nidx] === r && data[nidx+1] === g && data[nidx+2] === b) {
                                        visited[noffset] = 1;
                                        queue.push(nx, ny);
                                    }
                                }
                            }
                        }
                    }

                    // Generujemy wektory dla tego połączonego komponentu
                    const polygons = traceContoursForComponent(componentPixels);

                    if (polygons.length > 0) {
                        if (!provinces[hexKey]) {
                            provinceCounter++;
                            provinces[hexKey] = {
                                id: hexKey,
                                r: r, g: g, b: b,
                                name: `Prowincja ${provinceCounter}`,
                                nation: 'none',
                                dev: { tax: 3, prod: 3, manpower: 3 },
                                polygons: [],
                                boundingBox: { minX: Infinity, minY: Infinity, maxX: -Infinity, maxY: -Infinity }
                            };
                        }

                        // Dodajemy polygony tego komponentu do prowincji
                        for (let poly of polygons) {
                            provinces[hexKey].polygons.push(poly);
                            
                            // Aktualizujemy globalny bounding box prowincji do szybkiej eliminacji rysowania poza ekranem
                            for (let pt of poly) {
                                if (pt.x < provinces[hexKey].boundingBox.minX) provinces[hexKey].boundingBox.minX = pt.x;
                                if (pt.y < provinces[hexKey].boundingBox.minY) provinces[hexKey].boundingBox.minY = pt.y;
                                if (pt.x > provinces[hexKey].boundingBox.maxX) provinces[hexKey].boundingBox.maxX = pt.x;
                                if (pt.y > provinces[hexKey].boundingBox.maxY) provinces[hexKey].boundingBox.maxY = pt.y;
                            }
                        }
                    }
                }
            }

            showLoader(true, "Wyliczanie uśrednionych centrów prowincji...", 80);
            await sleep(50);

            // Tworzymy finalną listę prowincji i obliczamy ich środki geometryczne (do rysowania nazw)
            provinceList = Object.values(provinces);
            for (let prov of provinceList) {
                let sumX = 0, sumY = 0, totalPts = 0;
                for (let poly of prov.polygons) {
                    for (let pt of poly) {
                        sumX += pt.x;
                        sumY += pt.y;
                        totalPts++;
                    }
                }
                prov.centerX = totalPts > 0 ? sumX / totalPts : (prov.boundingBox.minX + prov.boundingBox.maxX) / 2;
                prov.centerY = totalPts > 0 ? sumY / totalPts : (prov.boundingBox.minY + prov.boundingBox.maxY) / 2;
            }

            showLoader(true, "Układanie danych i odświeżanie mapy...", 95);
            await sleep(50);

            updateNationsDropdown();
            updateNationsList();
            showLoader(false);
            draw();
        }

        // --- MATEMATYCZNE TRASOWANIE KRAWĘDZI DO POLIGONU ---
        // Buduje skierowany graf krawędzi komórek pikseli i łączy je w idealne zamknięte wielokąty wektorowe (8:1)
        function traceContoursForComponent(pixels) {
            const pixelSet = new Set();
            for (let p of pixels) {
                pixelSet.add(`${p.x}_${p.y}`);
            }

            // Mapa adiacencji krawędzi skierowanych: "x,y" -> ["nextX,nextY"]
            const adj = {};
            const addEdge = (x1, y1, x2, y2) => {
                const k1 = `${x1},${y1}`;
                const k2 = `${x2},${y2}`;
                if (!adj[k1]) adj[k1] = [];
                adj[k1].push(k2);
            };

            // Dla każdego piksela sprawdzamy 4 sąsiadów. Jeśli sąsiada brak w zestawie komponentu - tworzymy krawędź
            for (let p of pixels) {
                const x = p.x;
                const y = p.y;

                // Góra pusta -> krawędź w prawo
                if (!pixelSet.has(`${x}_${y-1}`)) {
                    addEdge(x, y, x + 1, y);
                }
                // Prawa pusta -> krawędź w dół
                if (!pixelSet.has(`${x+1}_${y}`)) {
                    addEdge(x + 1, y, x + 1, y + 1);
                }
                // Dół pusta -> krawędź w lewo
                if (!pixelSet.has(`${x}_${y+1}`)) {
                    addEdge(x + 1, y + 1, x, y + 1);
                }
                // Lewa pusta -> krawędź w górę
                if (!pixelSet.has(`${x-1}_${y}`)) {
                    addEdge(x, y + 1, x, y);
                }
            }

            // Łączymy segmenty krawędziowe w zamknięte pętle (poligony)
            const polygons = [];
            const visitedKeys = new Set();

            for (const startKey in adj) {
                if (!adj[startKey] || adj[startKey].length === 0) continue;

                while (adj[startKey].length > 0) {
                    const path = [];
                    let curr = startKey;
                    let success = false;

                    while (curr && adj[curr] && adj[curr].length > 0) {
                        const next = adj[curr].pop();
                        path.push(curr);
                        if (next === startKey) {
                            success = true;
                            break;
                        }
                        curr = next;
                    }

                    if (success && path.length >= 3) {
                        // Przekształcamy klucze tekstowe na współrzędne i skalujemy wektorowo 8:1
                        const points = path.map(k => {
                            const [px, py] = k.split(',').map(Number);
                            return { x: px * scaleFactor, y: py * scaleFactor };
                        });

                        // Uproszczenie kolinearnych krawędzi prostoliniowych (redukcja punktów o ~70% bez utraty jakości)
                        polygons.push(simplifyCollinear(points));
                    }
                }
            }

            return polygons;
        }

        // Usuwa zbędne punkty leżące na liniach prostych
        function simplifyCollinear(points) {
            if (points.length < 3) return points;
            const result = [points[0]];

            for (let i = 1; i < points.length - 1; i++) {
                const prev = points[i-1];
                const curr = points[i];
                const next = points[i+1];

                // Obliczanie powierzchni trójkąta (wyznacznik)
                const area = prev.x * (curr.y - next.y) + curr.x * (next.y - prev.y) + next.x * (prev.y - curr.y);
                // Jeśli pole jest bliskie 0, punkty są kolinearne
                if (Math.abs(area) > 0.001) {
                    result.push(curr);
                }
            }
            result.push(points[points.length - 1]);
            return result;
        }

        // --- RYSOWANIE MAPY NA CANVAS ---
        function draw() {
            if (canvas.width === 0 || canvas.height === 0) return;

            // Czyszczenie tła - kolor głębi oceanu
            ctx.fillStyle = '#0f172a';
            ctx.fillRect(0, 0, canvas.width, canvas.height);

            ctx.save();
            // Aplikacja przesunięcia i przybliżenia
            ctx.translate(offsetX, offsetY);
            ctx.scale(scale, scale);

            // Wyznaczanie granic ekranu w układzie współrzędnych mapy dla frustum culling (optymalizacja)
            const minViewX = -offsetX / scale;
            const minViewY = -offsetY / scale;
            const maxViewX = (canvas.width - offsetX) / scale;
            const maxViewY = (canvas.height - offsetY) / scale;

            // Rysowanie wszystkich prowincji
            for (let i = 0; i < provinceList.length; i++) {
                const prov = provinceList[i];

                // Optymalizacja: Pomiń rysowanie, jeśli prowincja leży całkowicie poza widocznym obszarem
                if (prov.boundingBox.maxX < minViewX || prov.boundingBox.minX > maxViewX ||
                    prov.boundingBox.maxY < minViewY || prov.boundingBox.minY > maxViewY) {
                    continue;
                }

                // Kolorowanie w zależności od trybu mapy
                if (currentMapMode === 'provinces') {
                    ctx.fillStyle = `rgb(${prov.r}, ${prov.g}, ${prov.b})`;
                } else if (currentMapMode === 'political') {
                    if (prov.nation !== 'none' && nations[prov.nation]) {
                        ctx.fillStyle = nations[prov.nation].color;
                    } else {
                        // Niezasiedlone prowincje w trybie politycznym rysujemy wyblakłe
                        ctx.fillStyle = `rgba(${prov.r}, ${prov.g}, ${prov.b}, 0.25)`;
                    }
                } else if (currentMapMode === 'physical') {
                    // Tryb stylowego "papierowego" terenu
                    ctx.fillStyle = prov.nation !== 'none' ? '#1e293b' : '#334155';
                }

                // Rysujemy polygony prowincji (w tym wyspy)
                for (let poly of prov.polygons) {
                    ctx.beginPath();
                    ctx.moveTo(poly[0].x, poly[0].y);
                    for (let j = 1; j < poly.length; j++) {
                        ctx.lineTo(poly[j].x, poly[j].y);
                    }
                    ctx.closePath();
                    ctx.fill();

                    // Rysowanie cienkich czarnych granic
                    ctx.strokeStyle = '#000000';
                    ctx.lineWidth = 0.5 / scale; // Stała grubość granic na ekranie przy zoomowaniu
                    ctx.stroke();
                }
            }

            // Podświetlenie najechanego (hover) prowincji
            if (hoveredProvince) {
                ctx.fillStyle = 'rgba(255, 255, 255, 0.25)';
                for (let poly of hoveredProvince.polygons) {
                    ctx.beginPath();
                    ctx.moveTo(poly[0].x, poly[0].y);
                    for (let j = 1; j < poly.length; j++) {
                        ctx.lineTo(poly[j].x, poly[j].y);
                    }
                    ctx.closePath();
                    ctx.fill();
                }
            }

            // Podświetlenie wybranego prowincji (Złote, grube obramowanie)
            if (selectedProvince) {
                ctx.strokeStyle = '#f59e0b';
                ctx.lineWidth = 2.5 / scale;
                ctx.shadowColor = '#f59e0b';
                ctx.shadowBlur = 10;
                for (let poly of selectedProvince.polygons) {
                    ctx.beginPath();
                    ctx.moveTo(poly[0].x, poly[0].y);
                    for (let j = 1; j < poly.length; j++) {
                        ctx.lineTo(poly[j].x, poly[j].y);
                    }
                    ctx.closePath();
                    ctx.stroke();
                }
                ctx.shadowBlur = 0; // Reset cienia
            }

            // Rysowanie nazw potęg/państw na środku ich prowincji
            if (scale > 0.8) {
                ctx.fillStyle = '#ffffff';
                ctx.shadowColor = '#000000';
                ctx.shadowBlur = 4;
                ctx.textAlign = 'center';
                ctx.textBaseline = 'middle';
                
                // Dostosowujemy czcionkę do zoomu
                const fontSize = Math.max(7, Math.min(11, 8 / scale));
                ctx.font = `bold ${fontSize}px sans-serif`;

                for (let prov of provinceList) {
                    if (prov.boundingBox.maxX < minViewX || prov.boundingBox.minX > maxViewX) continue;

                    // Pokazujemy skróconą nazwę prowincji lub właściciela
                    let text = prov.name;
                    if (currentMapMode === 'political' && prov.nation !== 'none') {
                        text = nations[prov.nation].name.substring(0, 8).toUpperCase();
                    }
                    ctx.fillText(text, prov.centerX, prov.centerY);
                }
                ctx.shadowBlur = 0;
            }

            ctx.restore();
        }

        // --- DETEKCJA KLIKNIĘĆ (Błyskawiczne O(1) dzięki buforowi koloru) ---
        function getProvinceAtCoords(screenX, screenY) {
            if (!originalImgData) return null;

            // Przeliczamy współrzędne ekranowe na współrzędne wirtualne mapy
            const mapX = (screenX - offsetX) / scale;
            const mapY = (screenY - offsetY) / scale;

            // Przeliczamy na piksele oryginalnego obrazu (dzielimy przez scaleFactor = 8)
            const origX = Math.floor(mapX / scaleFactor);
            const origY = Math.floor(mapY / scaleFactor);

            if (origX < 0 || origX >= originalImgWidth || origY < 0 || origY >= originalImgHeight) {
                return null;
            }

            const idx = (origY * originalImgWidth + origX) * 4;
            const r = originalImgData.data[idx];
            const g = originalImgData.data[idx+1];
            const b = originalImgData.data[idx+2];
            const a = originalImgData.data[idx+3];

            // Jeśli trafiliśmy w czarne obramowanie lub przeźroczystość - brak prowincji
            if (a < 50 || (r === 0 && g === 0 && b === 0)) return null;

            const hexKey = rgbToHex(r, g, b);
            return provinces[hexKey] || null;
        }

        // --- INTERAKCJA MYSZY I GESTÓW (ZOOM & PAN) ---
        function setupInteractions() {
            // Przesunięcia widoku i interakcja
            canvas.addEventListener('mousedown', (e) => {
                isDragging = true;
                dragStartX = e.clientX - offsetX;
                dragStartY = e.clientY - offsetY;
                lastMouseX = e.clientX;
                lastMouseY = e.clientY;
            });

            canvas.addEventListener('mousemove', (e) => {
                const rect = canvas.getBoundingClientRect();
                const mouseX = e.clientX - rect.left;
                const mouseY = e.clientY - rect.top;

                // Przesuwanie mapy
                if (isDragging) {
                    offsetX = e.clientX - dragStartX;
                    offsetY = e.clientY - dragStartY;
                    draw();
                } else {
                    // Detekcja hover prowincji
                    const prevHover = hoveredProvince;
                    hoveredProvince = getProvinceAtCoords(mouseX, mouseY);
                    
                    if (hoveredProvince) {
                        document.getElementById('status-hover').innerText = `${hoveredProvince.name} (${nations[hoveredProvince.nation].name})`;
                    } else {
                        document.getElementById('status-hover').innerText = '-';
                    }

                    if (prevHover !== hoveredProvince) {
                        draw();
                    }
                }
            });

            window.addEventListener('mouseup', () => {
                isDragging = false;
            });

            // Kliknięcie i malowanie prowincji
            canvas.addEventListener('click', (e) => {
                // Jeśli przesunęliśmy kursor o więcej niż 3 piksele podczas przeciągania, traktujemy to jako Pan, nie Klik
                const dragDist = Math.hypot(e.clientX - lastMouseX, e.clientY - lastMouseY);
                if (dragDist > 4) return;

                const rect = canvas.getBoundingClientRect();
                const mouseX = e.clientX - rect.left;
                const mouseY = e.clientY - rect.top;

                const clickedProv = getProvinceAtCoords(mouseX, mouseY);

                if (clickedProv) {
                    if (currentTool === 'select') {
                        // Standardowy wybór prowincji
                        selectedProvince = clickedProv;
                        showProvinceDetails(clickedProv);
                    } else if (currentTool === 'paint') {
                        // Narzędzie malowania państwa
                        paintProvince(clickedProv, activeNationId);
                    }
                    draw();
                }
            });

            // Obsługa Zoom kółkiem myszy
            canvas.addEventListener('wheel', (e) => {
                e.preventDefault();
                const rect = canvas.getBoundingClientRect();
                const mouseX = e.clientX - rect.left;
                const mouseY = e.clientY - rect.top;

                const zoomIntensity = 0.12;
                const wheel = e.deltaY < 0 ? 1 : -1;
                const zoomFactor = Math.exp(wheel * zoomIntensity);

                // Matematyka zoomu z zachowaniem punktu pod kursorem jako centrum
                const mapX = (mouseX - offsetX) / scale;
                const mapY = (mouseY - offsetY) / scale;

                scale *= zoomFactor;
                // Bezpieczne granice zoomu
                scale = Math.max(0.1, Math.min(scale, 40));

                offsetX = mouseX - mapX * scale;
                offsetY = mouseY - mapY * scale;

                document.getElementById('status-scale').innerText = `${Math.round(scale * 100)}%`;
                draw();
            }, { passive: false });
        }

        // Ręczne przyciski Zoom
        function zoomMap(factor) {
            const centerX = canvas.width / 2;
            const centerY = canvas.height / 2;

            const mapX = (centerX - offsetX) / scale;
            const mapY = (centerY - offsetY) / scale;

            scale *= factor;
            scale = Math.max(0.1, Math.min(scale, 40));

            offsetX = centerX - mapX * scale;
            offsetY = centerY - mapY * scale;

            document.getElementById('status-scale').innerText = `${Math.round(scale * 100)}%`;
            draw();
        }

        function resetView() {
            if (virtualMapWidth === 0) return;
            // Dopasowujemy mapę do dostępnego kontenera
            const scaleX = canvas.width / virtualMapWidth;
            const scaleY = canvas.height / virtualMapHeight;
            scale = Math.min(scaleX, scaleY) * 0.95; // 95% dla marginesu

            offsetX = (canvas.width - virtualMapWidth * scale) / 2;
            offsetY = (canvas.height - virtualMapHeight * scale) / 2;

            document.getElementById('status-scale').innerText = `${Math.round(scale * 100)}%`;
            draw();
        }

        // --- ZARZĄDZANIE PAŃSTWAMI ---
        function createNationManual(name, color) {
            const id = name.toLowerCase().replace(/[^a-z0-9]/g, '_');
            nations[id] = {
                id: id,
                name: name,
                color: color,
                provincesCount: 0
            };
            return id;
        }

        function createNewNation() {
            const nameInput = document.getElementById('new-nat-name');
            const colorInput = document.getElementById('new-nat-color');
            const name = nameInput.value.trim();
            const color = colorInput.value;

            if (!name) return;

            const id = createNationManual(name, color);
            nameInput.value = '';

            updateNationsDropdown();
            updateNationsList();
            setActivePaintNation(id);
        }

        function deleteNation(id) {
            if (id === 'none') return;
            // Wszystkie prowincje wracają do ziemi niczyjej
            for (let prov of provinceList) {
                if (prov.nation === id) {
                    prov.nation = 'none';
                }
            }
            delete nations[id];
            updateNationsDropdown();
            updateNationsList();
            if (selectedProvince && selectedProvince.nation === id) {
                selectedProvince.nation = 'none';
                showProvinceDetails(selectedProvince);
            }
            draw();
        }

        function updateNationsDropdown() {
            const select = document.getElementById('prov-nation-select');
            select.innerHTML = '';
            
            for (let nat of Object.values(nations)) {
                const opt = document.createElement('option');
                opt.value = nat.id;
                opt.innerText = nat.name;
                select.appendChild(opt);
            }
        }

        function updateNationsList() {
            // Przeliczanie prowincji dla każdego państwa
            for (let nat of Object.values(nations)) {
                nat.provincesCount = 0;
            }
            for (let prov of provinceList) {
                if (nations[prov.nation]) {
                    nations[prov.nation].provincesCount++;
                }
            }

            const listContainer = document.getElementById('nations-list');
            listContainer.innerHTML = '';

            for (let nat of Object.values(nations)) {
                if (nat.id === 'none') continue; // Ignorujemy ziemię nieznaną na liście interaktywnej

                const div = document.createElement('div');
                div.className = `flex items-center justify-between p-2.5 rounded border transition cursor-pointer ${activeNationId === nat.id ? 'bg-slate-800 border-amber-500' : 'bg-slate-950 border-slate-800 hover:border-slate-700'}`;
                div.onclick = () => setActivePaintNation(nat.id);

                div.innerHTML = `
                    <div class="flex items-center space-x-3 overflow-hidden">
                        <span class="w-4 h-4 rounded-full border border-black/40 flex-shrink-0" style="background-color: ${nat.color}"></span>
                        <div class="truncate">
                            <span class="font-semibold text-xs block text-slate-100">${nat.name}</span>
                            <span class="text-[10px] text-slate-400">Prowincje: ${nat.provincesCount}</span>

        // Generuje losową epicką nazwę w klimacie europejskim/fantasy
        function generateLoreName() {
            if (!selectedProvince) return;
            const prefixes = ["Nowy ", "Górny ", "Dolny ", "Wielki ", "Święty ", "Ost-", "West-", "Nord-", "Süd-"];
            const roots = ["Krak", "Warsz", "Gdań", "Wroc", "Pozn", "Berl", "Lond", "Paryż", "Wied", "Madryt", "Rzym", "Kij", "Mosk", "Sztok", "Prag"];
            const suffixes = ["ów", "ice", "burg", "grad", "ia", "owo", "in", "by", "heim", "stadt", "shire", "fort"];

            let name = "";
            if (Math.random() < 0.25) name += prefixes[Math.floor(Math.random() * prefixes.length)];
            name += roots[Math.floor(Math.random() * roots.length)];
            name += suffixes[Math.floor(Math.random() * suffixes.length)];

            document.getElementById('prov-name-input').value = name;
            updateSelectedProvinceName(name);
        }

        // --- EKSPORT I IMPORT DANYCH SCENARIUSZA (JSON) ---
        function exportScenario() {
            const dataToExport = {
                nations: nations,
                provincesData: provinceList.map(p => ({
                    id: p.id,
                    name: p.name,
                    nation: p.nation,
                    dev: p.dev
                }))
            };

            const dataStr = "data:text/json;charset=utf-8," + encodeURIComponent(JSON.stringify(dataToExport, null, 2));
            const downloadAnchor = document.createElement('a');
            downloadAnchor.setAttribute("href", dataStr);
            downloadAnchor.setAttribute("download", "scenariusz_mapy.json");
            document.body.appendChild(downloadAnchor);
            downloadAnchor.click();
            downloadAnchor.remove();
        }

        function triggerImport() {
            document.getElementById('import-input').click();
        }

        function importScenario(event) {
            const file = event.target.files[0];
            if (!file) return;

            const reader = new FileReader();
            reader.onload = function(e) {
                try {
                    const imported = JSON.parse(e.target.result);
                    
                    if (imported.nations) {
                        nations = imported.nations;
                    }
                    if (imported.provincesData) {
                        for (let pData of imported.provincesData) {
                            if (provinces[pData.id]) {
                                provinces[pData.id].name = pData.name;
                                provinces[pData.id].nation = pData.nation;
                                provinces[pData.id].dev = pData.dev;
                            }
                        }
                    }

                    updateNationsDropdown();
                    updateNationsList();
                    if (selectedProvince) showProvinceDetails(selectedProvince);
                    draw();
                    
                    // Pokazujemy komunikat sukcesu za pomocą customowej notyfikacji w status barze
                    document.getElementById('status-hover').innerText = "Scenariusz wczytany poprawnie!";
                } catch (err) {
                    document.getElementById('status-hover').innerText = "Błąd importu pliku JSON!";
                }
            };
            reader.readAsText(file);
        }

        function rgbToHex(r, g, b) {
            return "#" + ((1 << 24) + (r << 16) + (g << 8) + b).toString(16).slice(1).toUpperCase();
        }
